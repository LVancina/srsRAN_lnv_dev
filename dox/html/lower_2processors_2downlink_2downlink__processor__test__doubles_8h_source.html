61class downlink_processor_baseband_spy : public downlink_processor_baseband
62{
63public:
64 struct entry_t {
65 baseband_gateway_buffer_read_only buffer;
66 baseband_gateway_timestamp timestamp;
67 };
68
69 baseband_gateway_transmitter_metadata process(baseband_gateway_buffer_writer& buffer,
70 baseband_gateway_timestamp timestamp) override
71 {
72 for (unsigned i_channel = 0, i_channel_end = buffer.get_nof_channels(); i_channel != i_channel_end; ++i_channel) {
73 span<cf_t> samples = buffer.get_channel_buffer(i_channel);
74 std::generate(samples.begin(), samples.end(), [this]() { return cf_t(sample_dist(rgen), sample_dist(rgen)); });
75 }
76
77 entries.emplace_back();
78 entry_t& entry = entries.back();
79 entry.timestamp = timestamp;
80 entry.buffer = baseband_gateway_buffer_read_only(buffer);
81
82 baseband_gateway_transmitter_metadata md;
83 md.is_empty = false;
84 return md;
85 }
86
87 const std::vector<entry_t>& get_entries() const { return entries; }
88
89 void clear() { entries.clear(); }
90
91private:
92 std::mt19937 rgen;
93 std::uniform_real_distribution<float> sample_dist;
94 std::vector<entry_t> entries;
95};
