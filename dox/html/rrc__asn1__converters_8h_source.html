82inline asn1::rrc_nr::pdcp_cfg_s pdcp_config_to_rrc_nr_asn1(pdcp_config pdcp_cfg)
83{
84 asn1::rrc_nr::pdcp_cfg_s rrc_pdcp_cfg;
85
86 // t reordering -- Need S
87 if (pdcp_cfg.rx.t_reordering != pdcp_t_reordering::infinity) {
88 rrc_pdcp_cfg.t_reordering_present = true;
89 rrc_pdcp_cfg.t_reordering = t_reordering_to_asn1(pdcp_cfg.rx.t_reordering);
90 }
91
92 // ciphering disabled present -- Cond ConnectedTo5GC
93 if (!pdcp_cfg.ciphering_required) {
94 rrc_pdcp_cfg.ciphering_disabled_present = true;
95 // this is an extension field.
96 rrc_pdcp_cfg.ext = true;
97 }
98
99 // more than one rlc
100 rrc_pdcp_cfg.more_than_one_rlc_present = false; // not supported.
101
102 // no more configurable parameters for SRBs
103 if (pdcp_cfg.rb_type == pdcp_rb_type::srb) {
104 return rrc_pdcp_cfg;
105 }
106
107 // drb -- Cond DRB
108 rrc_pdcp_cfg.drb_present = true;
109
110 // hdr compress
111 rrc_pdcp_cfg.drb.hdr_compress.set_not_used(); // not supported.
112
113 // discard timer -- Cond Setup
114 if (pdcp_cfg.tx.discard_timer.has_value()) {
115 rrc_pdcp_cfg.drb.discard_timer_present = true;
116 rrc_pdcp_cfg.drb.discard_timer = discard_timer_to_asn1(pdcp_cfg.tx.discard_timer.value());
117 }
118
119 // pdcp sn size ul -- Cond Setup2
120 rrc_pdcp_cfg.drb.pdcp_sn_size_ul_present = true;
121 asn1::number_to_enum(rrc_pdcp_cfg.drb.pdcp_sn_size_ul, pdcp_sn_size_to_uint(pdcp_cfg.rx.sn_size));
122
123 // pdcp sn size dl -- Cond Setup2
124 rrc_pdcp_cfg.drb.pdcp_sn_size_dl_present = true;
125 asn1::number_to_enum(rrc_pdcp_cfg.drb.pdcp_sn_size_dl, pdcp_sn_size_to_uint(pdcp_cfg.tx.sn_size));
126
127 // integrity protection present
128 rrc_pdcp_cfg.drb.integrity_protection_present = pdcp_cfg.integrity_protection_required;
129
130 // status report required present
131 rrc_pdcp_cfg.drb.status_report_required_present = pdcp_cfg.tx.status_report_required;
132
133 // out of order delivery present
134 rrc_pdcp_cfg.drb.out_of_order_delivery_present = pdcp_cfg.rx.out_of_order_delivery;
135
136 return rrc_pdcp_cfg;
137}

168inline asn1::rrc_nr::sdap_cfg_s sdap_config_to_rrc_asn1(sdap_config_t sdap_cfg)
169{
170 asn1::rrc_nr::sdap_cfg_s asn1_sdap_cfg;
171
172 // pdu session
173 asn1_sdap_cfg.pdu_session = pdu_session_id_to_uint(sdap_cfg.pdu_session);
174
175 // sdap hdr dl
176 asn1_sdap_cfg.sdap_hdr_dl = sdap_hdr_dl_cfg_to_rrc_asn1(sdap_cfg.sdap_hdr_dl);
177
178 // sdap hdr ul
179 asn1_sdap_cfg.sdap_hdr_ul = sdap_hdr_ul_cfg_to_rrc_asn1(sdap_cfg.sdap_hdr_ul);
180
181 // default drb
182 asn1_sdap_cfg.default_drb = sdap_cfg.default_drb;
183
184 // mapped qos flow to add
185 for (const auto& mapped_qps_flow_to_add : sdap_cfg.mapped_qos_flows_to_add) {
186 asn1_sdap_cfg.mapped_qos_flows_to_add.push_back(qos_flow_id_to_uint(mapped_qps_flow_to_add));
187 }
188
189 // mapped qos flow to release
190 for (const auto& mapped_qps_flow_to_release : sdap_cfg.mapped_qos_flows_to_release) {
191 asn1_sdap_cfg.mapped_qos_flows_to_release.push_back(qos_flow_id_to_uint(mapped_qps_flow_to_release));
192 }
193
194 return asn1_sdap_cfg;
195}

197inline asn1::rrc_nr::ciphering_algorithm_e
198ciphering_algorithm_to_rrc_asn1(const security::ciphering_algorithm& ciphering_algo)
199{
200 asn1::rrc_nr::ciphering_algorithm_e asn1_ciphering_algo;
201
202 switch (ciphering_algo) {
203 case srsran::security::ciphering_algorithm::nea0:
204 asn1_ciphering_algo = asn1::rrc_nr::ciphering_algorithm_opts::options::nea0;
205 break;
206 case srsran::security::ciphering_algorithm::nea1:
207 asn1_ciphering_algo = asn1::rrc_nr::ciphering_algorithm_opts::options::nea1;
208 break;
209 case srsran::security::ciphering_algorithm::nea2:
210 asn1_ciphering_algo = asn1::rrc_nr::ciphering_algorithm_opts::options::nea2;
211 break;
212 case srsran::security::ciphering_algorithm::nea3:
213 asn1_ciphering_algo = asn1::rrc_nr::ciphering_algorithm_opts::options::nea3;
214 break;
215 default:
216 // error
217 report_fatal_error("Cannot convert ciphering algorithm {} to ASN.1 type", ciphering_algo);
218 }
219
220 return asn1_ciphering_algo;
221}
