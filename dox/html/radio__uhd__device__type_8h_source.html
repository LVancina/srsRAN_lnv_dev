29class radio_uhd_device_type
30{
31public:
32 enum class types {
34 UNKNOWN = 0,
36 B2xx,
38 E3xx,
40 N3x0,
42 N32x,
44 X3x0,
46 X410
47 };
48
49 radio_uhd_device_type() = default;
50 radio_uhd_device_type(types type_) : type(type_) {}
51 radio_uhd_device_type(std::string mboard_name)
52 {
53 for (char& c : mboard_name) {
54 c = std::toupper(c);
55 }
56
57 if (mboard_name.find("B2") != std::string::npos) {
58 type = types::B2xx;
59 } else if (mboard_name.find("X3") != std::string::npos) {
60 type = types::X3x0;
61 } else if (mboard_name.find("X4") != std::string::npos) {
62 type = types::X410;
63 } else if (mboard_name.find("N32") != std::string::npos) {
64 type = types::N32x;
65 } else if (mboard_name.find("N3") != std::string::npos) {
66 type = types::N3x0;
67 } else {
68 type = types::UNKNOWN;
69 }
70 }
71 std::string to_string()
72 {
73 switch (type) {
74 case types::B2xx:
75 return "uhd_b200";
76 case types::E3xx:
77 return "uhd_e3x0";
78 case types::N3x0:
79 return "uhd_n300";
80 case types::N32x:
81 return "uhd_n320";
82 case types::X3x0:
83 return "uhd_x300";
84 case types::X410:
85 return "uhd_x400";
86 case types::UNKNOWN:
87 default:
88 return "uhd_unknown";
89 }
90 }
91 bool operator==(const radio_uhd_device_type& other) const { return type == other.type; }
92 bool operator==(const types& other) const { return type == other; }
93 operator const types&() { return type; }
94
95private:
96 types type = types::UNKNOWN;
97};
98} // namespace srsran
