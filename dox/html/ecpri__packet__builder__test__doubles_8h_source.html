40public:
41 // See interface for documentation.
42 units::bytes get_header_size(message_type msg_type) const override { return units::bytes(8); }
43
44 // See interface for documentation.
45 void build_control_packet(span<uint8_t> buffer, const realtime_control_parameters& msg_params) override
46 {
47 build_control_packet_method_called = true;
48 ctrl_params.push_back(msg_params);
49 }
50
51 // See interface for documentation.
52 void build_data_packet(span<uint8_t> buffer, const iq_data_parameters& msg_params) override
53 {
54 build_data_packet_method_called = true;
55 data_params.push_back(msg_params);
56 }
57
59 bool has_build_control_packet_method_been_called() const { return build_control_packet_method_called; }
60
62 bool has_build_data_packet_method_been_called() const { return build_data_packet_method_called; }
63
65 span<const realtime_control_parameters> get_control_parameters() const { return ctrl_params; }
66
68 span<const iq_data_parameters> get_data_parameters() const { return data_params; }
69};
