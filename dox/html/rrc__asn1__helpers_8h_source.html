37inline void
38fill_asn1_rrc_setup_msg(asn1::rrc_nr::rrc_setup_s& rrc_setup, const byte_buffer& mcg, uint8_t rrc_transaction_id)
39{
40 using namespace asn1::rrc_nr;
41 rrc_setup_ies_s& setup_ies = rrc_setup.crit_exts.set_rrc_setup();
42 rrc_setup.rrc_transaction_id = rrc_transaction_id;
43
44 // Add SRB1
45 // Note: See TS 38.331 section 5.3.5.6.3 - SRB addition/modification
46 auto& radio_bearer_cfg = setup_ies.radio_bearer_cfg;
47 radio_bearer_cfg.srb_to_add_mod_list.push_back(srb_to_add_mod_s{});
48 radio_bearer_cfg.srb_to_add_mod_list.back().srb_id = 1; // SRB1
49
50 // Copy cell config from DU_to_CU_RRC_Container to master cell group
51 auto& master_cell_group = setup_ies.master_cell_group;
52 master_cell_group.resize(mcg.length());
53 std::copy(mcg.begin(), mcg.end(), master_cell_group.begin());
54}

63inline expected<uint8_t> get_transaction_id(const asn1::rrc_nr::ul_dcch_msg_s& msg)
64{
65 using namespace asn1::rrc_nr;
66 switch (msg.msg.c1().type().value) {
67 case ul_dcch_msg_type_c::c1_c_::types_opts::rrc_setup_complete:
68 return msg.msg.c1().rrc_setup_complete().rrc_transaction_id;
69 // TODO: Remaining cases.
70 default:
71 break;
72 }
73 return {default_error_t{}};
74}

80inline void fill_asn1_rrc_reconfiguration_msg(asn1::rrc_nr::rrc_recfg_s& asn1_rrc_reconf,
81 uint8_t rrc_transaction_id,
82 rrc_reconfiguration_procedure_request rrc_reconf)
83{
84 using namespace asn1::rrc_nr;
85
86 asn1_rrc_reconf.rrc_transaction_id = rrc_transaction_id;
87
88 rrc_recfg_ies_s& asn1_reconfig_ies = asn1_rrc_reconf.crit_exts.set_rrc_recfg();
89
90 // radio bearer cfg
91 if (rrc_reconf.radio_bearer_cfg.has_value()) {
92 asn1_reconfig_ies.radio_bearer_cfg_present = true;
93 auto& asn1_radio_bearer_cfg = asn1_reconfig_ies.radio_bearer_cfg;
94
95 auto& cu_cp_radio_bearer_cfg = rrc_reconf.radio_bearer_cfg.value();
96
97 // srb to add mod list
98 for (const auto& srb_to_add : cu_cp_radio_bearer_cfg.srb_to_add_mod_list) {
99 srsran_assert(srb_to_add.srb_id != srb_id_t::nulltype, "Invalid SRB ID");
100
101 asn1::rrc_nr::srb_to_add_mod_s asn1_srb_to_add;
102 asn1_srb_to_add.srb_id = srb_id_to_uint(srb_to_add.srb_id);
103
104 asn1_srb_to_add.reestablish_pdcp_present = srb_to_add.reestablish_pdcp_present;
105
106 asn1_srb_to_add.discard_on_pdcp_present = srb_to_add.discard_on_pdcp_present;
107
108 // PDCP config
109 if (srb_to_add.pdcp_cfg.has_value()) {
110 asn1_srb_to_add.pdcp_cfg_present = true;
111 asn1_srb_to_add.pdcp_cfg = pdcp_config_to_rrc_nr_asn1(srb_to_add.pdcp_cfg.value());
112 }
113
114 asn1_radio_bearer_cfg.srb_to_add_mod_list.push_back(asn1_srb_to_add);
115 }
116
117 // drb to add mod list
118 for (const auto& drb_to_add : cu_cp_radio_bearer_cfg.drb_to_add_mod_list) {
119 srsran_assert(drb_to_add.drb_id != drb_id_t::invalid, "Invalid DRB ID");
120
121 asn1::rrc_nr::drb_to_add_mod_s asn1_drb_to_add;
122 asn1_drb_to_add.drb_id = drb_id_to_uint(drb_to_add.drb_id);
123
124 asn1_drb_to_add.reestablish_pdcp_present = drb_to_add.reestablish_pdcp_present;
125
126 // PDCP config
127 if (drb_to_add.pdcp_cfg.has_value()) {
128 asn1_drb_to_add.pdcp_cfg_present = true;
129 asn1_drb_to_add.pdcp_cfg = pdcp_config_to_rrc_nr_asn1(drb_to_add.pdcp_cfg.value());
130 }
131
132 // Add CN association and SDAP config
133 if (drb_to_add.cn_assoc.has_value()) {
134 asn1_drb_to_add.cn_assoc_present = true;
135 if (drb_to_add.cn_assoc.value().sdap_cfg.has_value()) {
136 asn1_drb_to_add.cn_assoc.set_sdap_cfg();
137 asn1_drb_to_add.cn_assoc.sdap_cfg() = sdap_config_to_rrc_asn1(drb_to_add.cn_assoc.value().sdap_cfg.value());
138 } else {
139 asn1_drb_to_add.cn_assoc.set_eps_bearer_id();
140 asn1_drb_to_add.cn_assoc.eps_bearer_id() = drb_to_add.cn_assoc.value().eps_bearer_id.value();
141 }
142 }
143
144 asn1_radio_bearer_cfg.drb_to_add_mod_list.push_back(asn1_drb_to_add);
145 }
146
147 // drb to release list
148 for (const auto& drb_to_release : cu_cp_radio_bearer_cfg.drb_to_release_list) {
149 srsran_assert(drb_to_release != drb_id_t::invalid, "Invalid DRB ID");
150 asn1_radio_bearer_cfg.drb_to_release_list.push_back(drb_id_to_uint(drb_to_release));
151 }
152
153 // security cfg
154 if (cu_cp_radio_bearer_cfg.security_cfg.has_value()) {
155 asn1_radio_bearer_cfg.security_cfg_present = true;
156
157 const auto& security_cfg = cu_cp_radio_bearer_cfg.security_cfg.value();
158
159 // security algorithm cfg
160 if (security_cfg.security_algorithm_cfg.has_value()) {
161 asn1_radio_bearer_cfg.security_cfg.security_algorithm_cfg_present = true;
162
163 // ciphering algorithm
164 asn1_radio_bearer_cfg.security_cfg.security_algorithm_cfg.ciphering_algorithm =
165 ciphering_algorithm_to_rrc_asn1(security_cfg.security_algorithm_cfg.value().ciphering_algorithm);
166
167 // integrity prot algorithm
168 if (security_cfg.security_algorithm_cfg.value().integrity_prot_algorithm.has_value()) {
169 asn1_radio_bearer_cfg.security_cfg.security_algorithm_cfg.integrity_prot_algorithm_present = true;
170 asn1_radio_bearer_cfg.security_cfg.security_algorithm_cfg.integrity_prot_algorithm =
171 integrity_prot_algorithm_to_rrc_asn1(
172 security_cfg.security_algorithm_cfg.value().integrity_prot_algorithm.value());
173 }
174 }
175 // key to use
176 if (security_cfg.key_to_use.has_value()) {
177 asn1_radio_bearer_cfg.security_cfg.key_to_use_present = true;
178 asn1::string_to_enum(asn1_radio_bearer_cfg.security_cfg.key_to_use, security_cfg.key_to_use.value());
179 }
180 }
181
182 // srb3 to release present
183 asn1_radio_bearer_cfg.srb3_to_release_present = cu_cp_radio_bearer_cfg.srb3_to_release_present;
184 }
185
186 // secondary cell group
187 asn1_reconfig_ies.secondary_cell_group = rrc_reconf.secondary_cell_group.copy();
188
189 // meas config
190 if (rrc_reconf.meas_cfg.has_value()) {
191 asn1_reconfig_ies.meas_cfg_present = true;
192 asn1_reconfig_ies.meas_cfg = meas_config_to_rrc_asn1(rrc_reconf.meas_cfg.value());
193 }
194
195 // non crit ext
196 if (rrc_reconf.non_crit_ext.has_value()) {
197 asn1_reconfig_ies.non_crit_ext_present = true;
198 const auto& non_crit_ext = rrc_reconf.non_crit_ext.value();
199
200 // full cfg
201 asn1_reconfig_ies.non_crit_ext.full_cfg_present = non_crit_ext.full_cfg_present;
202
203 // master cell group config
204 if (!non_crit_ext.master_cell_group.empty()) {
205 asn1_reconfig_ies.non_crit_ext.master_cell_group = non_crit_ext.master_cell_group.copy();
206 }
207
208 // ded nas msg list
209 for (const auto& nas_pdu : non_crit_ext.ded_nas_msg_list) {
210 asn1_reconfig_ies.non_crit_ext.ded_nas_msg_list.push_back(nas_pdu.copy());
211 }
212
213 // master key upd
214 if (non_crit_ext.master_key_upd.has_value()) {
215 asn1_reconfig_ies.non_crit_ext.master_key_upd_present = true;
216 asn1_reconfig_ies.non_crit_ext.master_key_upd.key_set_change_ind =
217 non_crit_ext.master_key_upd.value().key_set_change_ind;
218 asn1_reconfig_ies.non_crit_ext.master_key_upd.next_hop_chaining_count =
219 non_crit_ext.master_key_upd.value().next_hop_chaining_count;
220 asn1_reconfig_ies.non_crit_ext.master_key_upd.nas_container =
221 non_crit_ext.master_key_upd.value().nas_container.copy();
222 }
223
224 // ded sib1 delivery
225 if (!non_crit_ext.ded_sib1_delivery.empty()) {
226 asn1_reconfig_ies.non_crit_ext.ded_sib1_delivery = non_crit_ext.ded_sib1_delivery.copy();
227 }
228
229 // ded sys info delivery
230 if (!non_crit_ext.ded_sys_info_delivery.empty()) {
231 asn1_reconfig_ies.non_crit_ext.ded_sys_info_delivery = non_crit_ext.ded_sys_info_delivery.copy();
232 }
233
234 // other cfg
235 if (non_crit_ext.other_cfg.has_value()) {
236 asn1_reconfig_ies.non_crit_ext.other_cfg_present = true;
237 if (non_crit_ext.other_cfg.value().delay_budget_report_cfg.has_value()) {
238 asn1_reconfig_ies.non_crit_ext.other_cfg.delay_budget_report_cfg_present = true;
239 const auto& delay_budget_report_cfg = non_crit_ext.other_cfg.value().delay_budget_report_cfg.value();
240
241 if (delay_budget_report_cfg.type == "setup") {
242 asn1_reconfig_ies.non_crit_ext.other_cfg.delay_budget_report_cfg.set_setup();
243 asn1::string_to_enum(asn1_reconfig_ies.non_crit_ext.other_cfg.delay_budget_report_cfg.setup()
244 .delay_budget_report_prohibit_timer,
245 delay_budget_report_cfg.delay_budget_report_prohibit_timer);
246 } else if (delay_budget_report_cfg.type == "release") {
247 asn1_reconfig_ies.non_crit_ext.other_cfg.delay_budget_report_cfg.set_release();
248 } else {
249 asn1_reconfig_ies.non_crit_ext.other_cfg.delay_budget_report_cfg.set(
250 asn1::rrc_nr::other_cfg_s::delay_budget_report_cfg_c_::types_opts::nulltype);
251 }
252 }
253 }
