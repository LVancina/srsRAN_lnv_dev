32{
33private:

36
37 /// Implements TS 38.211 section 7.3.3.1 Scrambling
38 /// \param [in] b Inputs bits to scramble
39 /// \param [out] b_hat Output bits after scrambling
40 /// \param [in] config PBCH modulator arguments
41 void scramble(span<const uint8_t> b, span<uint8_t> b_hat, const config_t& config);
42
43 /// Implements TS 38.211 section 7.3.3.2 Modulation
44 /// \param [in] b_hat Inputs bits to scramble
45 /// \param [out] d_pbch Output symbols
46 void modulate(span<const uint8_t> b_hat, span<cf_t> d_pbch);
47
48 /// Implements TS 38.211 section 7.3.3.3 Mapping to physical resources
49 /// \param [in] d_pbch Provides the symbols to map
50 /// \param [in, out] grid is the destination resource grid
51 /// \param [in] config PBCH modulator arguments
52 void map(span<const cf_t> d_pbch, resource_grid_writer& grid, const config_t& config);
53
54public:
55 pbch_modulator_impl(std::unique_ptr<modulation_mapper> modulator_, std::unique_ptr<pseudo_random_generator> prg_) :
56 modulator(std::move(modulator_)), scrambler(std::move(prg_))
57 {
58 srsran_assert(modulator, "Invalid modulator.");
59 srsran_assert(scrambler, "Invalid scrambler.");
60 }
61
62 void put(span<const uint8_t> bits, resource_grid_writer& grid, const config_t& config) override;
63};
