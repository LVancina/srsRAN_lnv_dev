40class uhd_exception_handler
41{
42public:
43 template <typename S, typename... Args>
44 void on_error(const S& format_str, Args&&... args)
45 {
46 fmt::memory_buffer str_buf;
47 fmt::format_to(str_buf, format_str, std::forward<Args>(args)...);
48 error_message = to_string(str_buf);
49 }
50
51 template <typename F>
52 bool safe_execution(F task)
53 {
54 // Clears the error message.
55 error_message.clear();
56
57 static_assert(std::is_convertible<F, std::function<void()>>::value, "The function signature must be void()");
58 // Try to execute task and catch exception.
59 try {
60 task();
61 } catch (const uhd::exception& e) {
62 error_message = e.what();
63 return false;
64 } catch (const boost::exception& e) {
65 error_message = boost::diagnostic_information(e);
66 return false;
67 } catch (const std::exception& e) {
68 error_message = e.what();
69 return false;
70 } catch (...) {
71 error_message = "Unrecognized exception caught.";
72 return false;
73 }
74
75 return is_successful();
76 }
77
78 bool is_successful() const { return error_message.empty(); }
79 const std::string& get_error_message() const { return error_message; }
80
81private:
82 std::string error_message;
83};
