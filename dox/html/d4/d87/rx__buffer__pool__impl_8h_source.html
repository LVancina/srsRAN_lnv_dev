38/// Implements a PUSCH rate matcher buffer pool.
39class rx_buffer_pool_impl : public rx_buffer_pool_controller, private rx_buffer_pool
40{
41private:
42 /// No expiration time value.
43 const slot_point null_expiration = slot_point();
44
45 /// Set to true when the buffer pool stopped.
46 std::atomic<bool> stopped = {false};
47 /// Codeblock buffer pool.
48 rx_buffer_codeblock_pool codeblock_pool;
49 /// Actual buffer pool.
50 std::vector<rx_buffer_impl> buffers;
51 /// \brief List of buffer identifiers.
52 ///
53 /// Maps buffer identifiers to buffers. Each position indicates the buffer identifier assign to each of the buffers.
54 /// Buffers with \c trx_buffer_identifier::invalid() identifier are not reserved.
55 std::vector<trx_buffer_identifier> identifiers;
56 /// Tracks expiration times.
57 std::vector<slot_point> expirations;
58
59 /// Indicates the lifetime of a buffer reservation as a number of slots.
60 unsigned expire_timeout_slots;
61 /// Logger.
62 srslog::basic_logger& logger;
63
64 // See interface for documentation.
65 unique_rx_buffer
66 reserve(const slot_point& slot, trx_buffer_identifier id, unsigned nof_codeblocks, bool new_data) override;
67
68 // See interface for documentation.
69 void run_slot(const slot_point& slot) override;
70
71public:
72 /// \brief Creates a generic receiver buffer pool.
73 /// \param[in] config Provides the pool required parameters.
74 rx_buffer_pool_impl(const rx_buffer_pool_config& config) :
75 codeblock_pool(config.nof_codeblocks, config.max_codeblock_size, config.external_soft_bits),
76 buffers(config.nof_buffers, rx_buffer_impl(codeblock_pool)),
77 identifiers(config.nof_buffers, trx_buffer_identifier::invalid()),
78 expirations(config.nof_buffers, null_expiration),
79 expire_timeout_slots(config.expire_timeout_slots),
80 logger(srslog::fetch_basic_logger("PHY", true))
81 {
82 }
83
84 // See rx_buffer_pool_controller interface for documentation.
85 ~rx_buffer_pool_impl() override
86 {
87 srsran_assert(std::none_of(buffers.begin(), buffers.end(), [](const auto& buffer) { return buffer.is_locked(); }),
88 "A buffer is still locked.");
89 }
90
91 // See rx_buffer_pool_controller interface for documentation.
92 rx_buffer_pool& get_pool() override;
93
94 // See rx_buffer_pool_controller interface for documentation.
95 void stop() override;
96};
