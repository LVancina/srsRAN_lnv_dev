35/// Type to store a dynamic amount of aligned contiguous elements
36template <typename T>
37class aligned_vec : public span<T>
38{
39 void dealloc() { detail::mem_free(this->data()); }
40
41public:
42 aligned_vec<T>& operator=(aligned_vec<T>&& other) = delete;
43 aligned_vec<T>& operator=(const aligned_vec<T>& other) = delete;
44 aligned_vec(const aligned_vec<T>& other) = delete;
45 aligned_vec(aligned_vec<T>&& other) noexcept = delete;
46
47 aligned_vec() : span<T>(nullptr, 0UL){};
48 explicit aligned_vec(std::size_t size) { resize(size); }
49
50 void resize(unsigned new_size)
51 {
52 if (new_size == this->size()) {
53 return;
54 }
55 dealloc();
56
57 T* ptr_ = (T*)detail::mem_alloc(sizeof(T) * new_size);
58 span<T> o = span<T>(ptr_, new_size);
59 *(span<T>*)this = o;
60 }
61
62 ~aligned_vec() { dealloc(); }
63};
