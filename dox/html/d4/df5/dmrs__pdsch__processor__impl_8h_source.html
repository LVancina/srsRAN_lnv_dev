32namespace srsran {
33
34/// Defines a DMRS for PDSCH generic implementation.
35class dmrs_pdsch_processor_impl : public dmrs_pdsch_processor
36{
37private:
38 /// Defines the maximum number of DMRS per RB.
39 static constexpr unsigned MAX_DRMS_PER_RB =
40 std::max(dmrs_type(dmrs_type::TYPE1).nof_dmrs_per_rb(), dmrs_type(dmrs_type::TYPE2).nof_dmrs_per_rb());
41
42 /// Define the maximum number of DMRS per symbol.
43 static constexpr unsigned MAX_DMRS_PER_SYMBOL = MAX_RB * MAX_DRMS_PER_RB;
44
45 /// \brief Maximum number of OFDM symbols containing DM-RS on a PDSCH allocation
46 ///
47 /// It corresponds to a single symbol DM-RS with three additional positions, or to a double symbol DM-RS with one
48 /// additional position, as per TS38.211 Section 7.4.1.1.2. and TS38.214 Section 5.6.1.2.
49 static constexpr unsigned MAX_DMRS_SYMBOLS = 4;
50
51 /// Defines generation parameters.
52 struct params_t {
53 std::array<float, 2> w_f;
54 std::array<float, 2> w_t;
55 };
56
57 /// Provides TS 38.211 Table 7.4.1.1.2-1: Parameters for PDSCH DM-RS configuration type 1.
58 static const std::array<params_t, 8> params_type1;
59
60 /// Provides TS 38.211 Table 7.4.1.1.2-2: Parameters for PDSCH DM-RS configuration type 2.
61 static const std::array<params_t, 12> params_type2;
62
63 /// Pseudo-random sequence generator instance.
64 std::unique_ptr<pseudo_random_generator> prg;
65
66 /// \brief Implements TS 38.211 section 7.4.1.1.1 Sequence generation.
67 ///
68 /// This method generates the sequence described in TS 38.211 section 7.4.1.1.1, considering the only values required
69 /// in TS 38.211 section 7.4.1.1.2.
70 ///
71 /// \param[out] sequence Provides the sequence destination.
72 /// \param[in] symbol Denotes the symbol index.
73 /// \param[in] config Provides the required parameters to calculate the sequences.
74 void sequence_generation(span<cf_t> sequence, unsigned symbol, const config_t& config) const;
75
76 /// \brief Applies the corresponding set of CDM codes to the DM-RS sequence.
77 ///
78 /// \param[out] sequence The DM-RS sequence after the application of CDM.
79 /// \param[in] base_sequence The original DM-RS sequence.
80 /// \param[in] dmrs_port DM-RS port number, determining the CDM code to apply.
81 /// \param[in] symbol OFDM symbol corresponding to the input DM-RS sequence.
82 /// \param[in] config PDSCH DM-RS configuration parameters.
83 static void apply_cdm(span<cf_t> sequence,
84 span<const cf_t> base_sequence,
85 unsigned dmrs_port,
86 unsigned symbol,
87 const config_t& config);
88
89 /// Temporal resource element storage.
90 static_re_buffer<MAX_PORTS, MAX_DMRS_PER_SYMBOL * MAX_DMRS_SYMBOLS> temp_re;
91
92public:
93 dmrs_pdsch_processor_impl(std::unique_ptr<pseudo_random_generator> pseudo_random_generator) :
94 prg(std::move(pseudo_random_generator))
95 {
96 srsran_assert(prg, "Invalid PRG.");
97 }
98
99 // See interface for documentation.
100 void map(resource_grid_mapper& mapper, const config_t& config) override;
101};
102
103} // namespace srsran
