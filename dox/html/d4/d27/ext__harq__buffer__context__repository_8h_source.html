34/// Fixed CB-offset increment used in the accelerator's HARQ memory.
35/// Note that it is assumed that the HARQ memory is organized in N slots of HARQ_INCR bytes.
36static constexpr unsigned HARQ_INCR = 32768;
37
38/// External HARQ buffer context.
39struct ext_harq_buffer_context_entry {
40 /// HARQ soft-data length of the CB stored in that segment of the buffer.
41 unsigned soft_data_len = 0;
42 /// Flag indicating if the entry is empty or not.
43 bool empty = true;
44};
45
46/// Abstracted interfacing to bbdev-based hardware-accelerators.
47class ext_harq_buffer_context_repository
48{
49public:
50 /// Creates a harq buffer context repository.
51 /// \param[in] nof_codeblocks_ Indicates the number of codeblocks to store in the repository.
52 /// \param[in] ext_harq_buff_size Size of the external HARQ buffer (in bytes).
53 /// \param[in] debug_mode_ Requests to implement the debug mode (meant for unittesting).
54 explicit ext_harq_buffer_context_repository(unsigned nof_codeblocks_, unsigned ext_harq_buff_size, bool debug_mode_) :
55 nof_codeblocks(nof_codeblocks_)
56 {
57 unsigned requested_size = nof_codeblocks_ * HARQ_INCR;
58 srsran_assert(requested_size <= ext_harq_buff_size,
59 "Requested size ({} bytes) for {} codeblocks exceeds external HARQ buffer capacity ({} bytes).",
60 requested_size,
61 nof_codeblocks_,
62 ext_harq_buff_size);
63 repo.resize(nof_codeblocks);
64 debug_mode = debug_mode_;
65 }
66
67 /// Get the entry for the provided absolute codeblock identifier.
68 ext_harq_buffer_context_entry* get(unsigned absolute_codeblock_id, bool new_data)
69 {
70 srsran_assert(absolute_codeblock_id < nof_codeblocks,
71 "Absolute CB index {} out of bounds - HARQ buffer context has capacity for {} CBs.",
72 absolute_codeblock_id,
73 nof_codeblocks);
74 // Initialize the entry if not already created (or in first transmissions).
75 if (repo[absolute_codeblock_id].empty || new_data) {
76 repo[absolute_codeblock_id].soft_data_len = 0;
77 repo[absolute_codeblock_id].empty = false;
78 }
79
80 return &repo[absolute_codeblock_id];
81 }
82
83 /// Free the entry reserved for the provided absolute codeblock identifier.
84 void free(unsigned absolute_codeblock_id)
85 {
86 srsran_assert(absolute_codeblock_id < nof_codeblocks,
87 "Absolute CB index {} out of bounds - HARQ buffer context has capacity for {} CBs.",
88 absolute_codeblock_id,
89 nof_codeblocks);
90 // Free the entry.
91 // Note that in debug mode, the entry won't be freed to enable HARQ unitesting.
92 if (!debug_mode) {
93 repo[absolute_codeblock_id].empty = true;
94 }
95 }
96
97private:
98 // HARQ buffer context repository.
99 std::vector<ext_harq_buffer_context_entry> repo;
100 // Number of codeblocks that can be stored in the repository.
101 unsigned nof_codeblocks;
102 // Enables an optional debug mode implementation.
103 bool debug_mode;
104};
