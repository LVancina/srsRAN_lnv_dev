260 {
261 // Pop enqueued tasks until number of enqueued jobs is zero.
262 // Note: Since we acquired the task_strand, the task enqueued in this call should always be the first being
263 // popped. Note: If there is a single producer, only the task enqueued in this call will be popped. Note: As we
264 // currently hold the strand, there is no concurrent thread popping tasks.
265 uint32_t queue_size = this->job_count.load(std::memory_order_acquire);
266 unique_task dropped_task;
267 while (queue_size > 0) {
268 unsigned run_count = 0;
269 for (; run_count != queue_size and this->queue.queue.try_pop(dropped_task); ++run_count) {
270 // do nothing with popped task.
271 }
272 queue_size = this->job_count.fetch_sub(run_count, std::memory_order_acq_rel) - run_count;
