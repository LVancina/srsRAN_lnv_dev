36/// \brief Collects the necessary parameters to calculate the Uplink Shared Channel (UL-SCH) information.
37///
38/// The parameters are described in TS38.212 Section 6.3.2.4.1.
39struct ulsch_configuration {
40 /// Transport block size. Set to zero if no shared channel is multiplexed.
41 units::bits tbs;
42 /// Modulation and coding scheme.
43 sch_mcs_description mcs_descr;
44 /// Number of HARQ-ACK bits to multiplex in the transmission. Parameter \f$O_\textup{ACK}\f$.
45 units::bits nof_harq_ack_bits;
46 /// Number of CSI Part 1 bits to multiplex in the transmission. Parameter \f$O_\textup{CSI-1}\f$.
47 units::bits nof_csi_part1_bits;
48 /// Number of CSI Part 2 bits to multiplex in the transmission. Parameter \f$O_\textup{CSI-2}\f$.
49 units::bits nof_csi_part2_bits;
50 /// \brief Parameter \f$\alpha\f$ as per TS38.212 Section 6.3.2.4.
51 ///
52 /// Provided by the higher layer parameter \e scaling in TS38.331 Section 6.3.2, Information Element \e UCI-OnPUSCH.
53 float alpha_scaling;
54 /// Parameter \f$\beta _\textup{offset} ^\textup{HARQ-ACK}\f$.
55 float beta_offset_harq_ack;
56 /// Parameter \f$\beta _\textup{offset} ^\textup{CSI-1}\f$.
57 float beta_offset_csi_part1;
58 /// Parameter \f$\beta _\textup{offset} ^\textup{CSI-2}\f$.
59 float beta_offset_csi_part2;
60 /// Transmission bandwidth in resource blocks.
61 unsigned nof_rb;
62 /// First OFDM index for the transmission within the slot.
63 unsigned start_symbol_index;
64 /// Number of OFDM symbols used for the transmission.
65 unsigned nof_symbols;
66 /// DM-RS Type.
67 dmrs_config_type dmrs_type;
68 /// Boolean mask indicating which OFDM symbols in the slot contain DM-RS.
69 bounded_bitset<MAX_NSYMB_PER_SLOT> dmrs_symbol_mask;
70 /// Number of CDM groups without data.
71 unsigned nof_cdm_groups_without_data;
72 /// Number of transmission layers.
73 unsigned nof_layers;
74};

76/// \brief Collects Uplink Shared Channel (UL-SCH) derived parameters.
77///
78/// The parameters are described in TS38.212 Section 6.3.2.4.1.
79struct ulsch_information {
80 /// Shared channel (SCH) parameters.
81 optional<sch_information> sch;
82 /// Number of encoded and rate-matched UL-SCH data bits. Parameter \f$G^\textup{UL-SCH}\f$.
83 units::bits nof_ul_sch_bits;
84 /// Number of encoded and rate-matched HARQ-ACK data bits. Parameter \f$G^\textup{HARQ-ACK}\f$.
85 units::bits nof_harq_ack_bits;
86 /// Number of reserved bits for HARQ-ACK. Parameter \f$G^\textup{HARQ-ACK}_\textup{rvd}\f$.
87 units::bits nof_harq_ack_rvd;
88 /// Number of encoded and rate-matched CSI Part 1 data bits. Parameter \f$G^\textup{CSI-part1}\f$.
89 units::bits nof_csi_part1_bits;
90 /// Number of encoded and rate-matched CSI Part 2 data bits. Parameter \f$G^\textup{CSI-part2}\f$.
91 units::bits nof_csi_part2_bits;
92 /// Number of resource elements occupied by HARQ-ACK information. Parameter \f$Q'_\textup{ACK}\f$.
93 unsigned nof_harq_ack_re;
94 /// Number of resource elements occupied by CSI Part 1 information. Parameter \f$Q'_\textup{CSI-1}\f$.
95 unsigned nof_csi_part1_re;
96 /// Number of resource elements occupied by CSI Part 1 information. Parameter \f$Q'_\textup{CSI-2}\f$.
97 unsigned nof_csi_part2_re;
98
99 /// \brief Calculates the effective code rate normalized between 0 and 1.
100 ///
101 /// The effective code rate is determined as the quotient of the number of information bits plus CRCs and the total
102 /// number of channel bits.
103 ///
104 /// An assertion is triggered if:
105 /// - UL-SCH is not present;
106 /// - the resultant codeblock size is not greater than the number of filler bits; or
107 /// - the number of rate matched bits for HARQ-ACK feedback is not smaller than the number of rate matched UL-SCH
108 /// bits.
109 float get_effective_code_rate() const
110 {
111 using namespace units::literals;
112
113 // Ensure UL-SCH is present and the number of bits per CB is greater than filler bits.
114 srsran_assert(sch.has_value(), "SCH information is not present.");
115 srsran_assert(sch.value().nof_bits_per_cb.value() > sch.value().nof_filler_bits_per_cb.value(),
116 "The number of bits per CB must be greater than the number of filler bits.");
117
118 // Return 0 if no UL-SCH bits are present.
119 if (nof_ul_sch_bits.value() == 0) {
120 return 0;
121 }
122
123 // Select the number of effective rate matched bits carrying UL-SCH.
124 unsigned effective_ul_sch_bits = nof_ul_sch_bits.value();
125
126 // Adjust the effective rate matched UL-SCH bits considering HARQ-ACK feedback bits.
127 if (nof_harq_ack_rvd > 0_bits) {
128 // Ensure the subtraction of HARQ-ACK bits does not result in zero or negative.
129 srsran_assert(nof_ul_sch_bits > nof_harq_ack_bits,
130 "UL-SCH rate match length ({} bits) must be greater than the HARQ-ACK rate match length ({} bits).",
131 nof_ul_sch_bits,
132 nof_harq_ack_bits);
133
134 effective_ul_sch_bits -= nof_harq_ack_bits.value();
135 }
136
137 // Calculate the exact number of bits to encode, including payload, transport block CRC, and codeblock CRC.
138 unsigned nof_effective_payload_bits =
139 (sch.value().nof_bits_per_cb.value() - sch.value().nof_filler_bits_per_cb.value()) * sch.value().nof_cb;
140
141 // Calculate the effective code rate as the quotient of the effective payload bits to effective rate matched bits.
142 return static_cast<float>(nof_effective_payload_bits) / static_cast<float>(effective_ul_sch_bits);
143 }
144};
