36/**
37 * Concurrent memory pool of memory blocks of equal size. This pool is thread-safe.
38 * Each worker keeps a separate thread-local memory block cache that it uses for fast, non-contended allocation and
39 * deallocation. When accessing a thread-local cache, no locks are required.
40 *
41 * When the local cache gets depleted, the worker tries to obtain a batch of segments from a central memory block cache.
42 * If the central cache is also depleted, the allocation fails.
43 *
44 * Since there is no stealing of segments between workers' local caches, it is possible that a worker cannot allocate
45 * while another worker still has blocks in its own cache. To minimize the impact of this event, an upper bound is
46 * placed on a worker local cache size. Once a worker reaches that upper bound, it sends half of its stored blocks to
47 * the central cache.
48 *
49 * Note1: Notice that the same memory block might be allocated and deallocated in different workers. So, it can happen
50 * that a worker is allocating many blocks and another worker just deallocating. The latter worker has to keep
51 * returning blocks back to the central cache every time its local cache grows beyond a given upper bound.
52 *
53 * Note2: Taking into account the usage of thread_local, this class is made a singleton. To be able to instantiate
54 * different pools, the user should use different IdTag types.
55 *
56 * Note3: No considerations were made regarding false sharing between workers. It is assumed that the blocks are big
57 * enough to fill a cache line.
58 *
59 * \tparam IdTag We use a ID type tag to be able to intantiate different pool objects.
60 * \tparam DebugSanitizeAddress when set to true, the pool verifies that the addresses allocated and deallocated are
61 * valid.
62 */
63template <typename IdTag, bool DebugSanitizeAddress = false>
64class fixed_size_memory_block_pool
65{
66 using pool_type = fixed_size_memory_block_pool<IdTag, DebugSanitizeAddress>;
67
68 /// The number of blocks in batch that the worker can steal from the central cache.
69 constexpr static size_t block_batch_size = 32U;
70
71 /// The number of batches of blocks that a worker can store in its own thread for non-contended access.
72 constexpr static size_t MAX_LOCAL_BATCH_CAPACITY = 64U;
73
74 /// A batch of memory blocks that is exchanged in bulk between the central and local caches.
75 using memory_block_batch = free_memory_block_list;
76
77 /// Thread-local cache that stores a list of batches of memory blocks.
78 using local_cache_type = static_vector<memory_block_batch, MAX_LOCAL_BATCH_CAPACITY>;
79
80 // Given that we use the MPMC queue in https://github.com/cameron314/concurrentqueue, we have to over-dimension it
81 // to account the potential number of producers. The way to exactly over-dimension this queue is inconvenient, so
82 // we just try to conservatively ensure it can accommodate up to 32 producers for a block size of 32. If this is
83 // not enough, the queue will resize itself and malloc in the process.
84 const static size_t OVER_DIM_CENTRAL_CACHE = 2 * 32 * 32;
85
86 /// Ctor of the memory pool. It is set as private because the class works as a singleton.
87 explicit fixed_size_memory_block_pool(size_t nof_blocks_, size_t memory_block_size_) :
88 // Make sure that there are no gaps between blocks when they are allocated as paret of a single array.
89 mblock_size(align_next(memory_block_size_, alignof(std::max_align_t))),
90 // Make sure all batches are filled with block_batch_size blocks.
91 nof_blocks(ceil(nof_blocks_ / (double)block_batch_size) * block_batch_size),
92 // Calculate the maximum number of batches that can be stored in the local cache.
93 max_local_batches(
94 std::max(std::min((size_t)MAX_LOCAL_BATCH_CAPACITY, static_cast<size_t>(nof_blocks / block_batch_size / 32U)),
95 static_cast<size_t>(2U))),
96 // Allocate the required memory for the given number of segments and segment size.
97 allocated_memory(mblock_size * nof_blocks),
98 // Pre-reserve space in the central cache to hold all batches and avoid reallocations.
99 central_mem_cache(nof_total_batches() + OVER_DIM_CENTRAL_CACHE)
100 {
101 srsran_assert(nof_blocks > max_local_cache_size(),
102 "The number of segments in the pool must be much larger than the thread cache size ({} <= {})",
103 nof_blocks,
104 max_local_cache_size());
105 srsran_assert(mblock_size > free_memory_block_list::min_memory_block_align(),
106 "Segment size is too small ({} <= {})",
107 mblock_size,
108 free_memory_block_list::min_memory_block_align());
109
110 // Push all memory blocks to the central cache in batches.
111 const unsigned nof_batches = nof_total_batches();
112 for (unsigned i = 0; i != nof_batches; ++i) {
113 free_memory_block_list batch;
114 for (unsigned j = 0; j != block_batch_size; ++j) {
115 batch.push(allocated_memory.data() + (i * block_batch_size + j) * mblock_size);
116 }
117 report_fatal_error_if_not(central_mem_cache.enqueue(batch), "Failed to push batch to central cache");
118 }
119 }

122 fixed_size_memory_block_pool(const fixed_size_memory_block_pool&) = delete;

124 fixed_size_memory_block_pool& operator=(const fixed_size_memory_block_pool&) = delete;
125 fixed_size_memory_block_pool& operator=(fixed_size_memory_block_pool&&) = delete;
126

129 /// \brief Get instance of a memory pool singleton.
130 static fixed_size_memory_block_pool<IdTag, DebugSanitizeAddress>& get_instance(size_t nof_blocks = 0,
131 size_t mem_block_size = 0)
132 {
133 static fixed_size_memory_block_pool<IdTag, DebugSanitizeAddress> pool(nof_blocks, mem_block_size);
134 return pool;
135 }
136
137 /// Memory block size in bytes.
138 size_t memory_block_size() const { return mblock_size; }
139
140 /// Number of memory blocks contained in this memory pool.
141 size_t nof_memory_blocks() const { return nof_blocks; }
142
143 /// Maximum number of blocks that can be stored in the thread-local memory block cache.
144 size_t max_local_cache_size() const { return max_local_batches * block_batch_size; }
145
146 /// Allocate a node from the memory pool with the maximum size.
147 void* allocate_node() noexcept { return allocate_node(memory_block_size()); }
148
149 /// Allocate a node from the memory pool with the provided size.
150 void* allocate_node(size_t sz) noexcept
151 {
152 srsran_assert(sz <= mblock_size, "Allocated node size={} exceeds max object size={}", sz, mblock_size);
153 worker_ctxt* w_ctx = get_worker_cache();
154
155 // Attempt memory block pop from local cache.
156 void* node = nullptr;
157 while (not w_ctx->local_cache.empty()) {
158 node = w_ctx->local_cache.back().try_pop();
159 if (node != nullptr) {
160 return node;
161 }
162 w_ctx->local_cache.pop_back();
163 }
164
165 // Local cache is empty. Attempt memory block pop from central cache.
166 free_memory_block_list batch;
167 if (central_mem_cache.try_dequeue(w_ctx->consumer_token, batch)) {
168 w_ctx->local_cache.push_back(batch);
169 node = w_ctx->local_cache.back().try_pop();
170 }
171
172 return node;
173 }
174
175 /// Deallocate node by returning it back to the memory pool.
176 void deallocate_node(void* p)
177 {
178 srsran_assert(p != nullptr, "Deallocated nodes must have valid address");
179
180 worker_ctxt* w_ctx = get_worker_cache();
181
182 if (DebugSanitizeAddress) {
183 // For debug purposes.
184 std::lock_guard<std::mutex> lock(debug_mutex);
185 bool found = false;
186 for (unsigned i = 0; i != nof_blocks; ++i) {
187 if (allocated_memory.data() + i * mblock_size == static_cast<uint8_t*>(p)) {
188 found = true;
189 }
190 }
191 if (not found) {
192 report_fatal_error("Error dealloccating block with address {:#x}", p);
193 }
194 }
195
196 // Verify if new batch needs to be created in local cache.
197 if (w_ctx->local_cache.empty() or w_ctx->local_cache.back().size() >= block_batch_size) {
198 w_ctx->local_cache.emplace_back();
199 }
200
201 // Push block to local cache.
202 w_ctx->local_cache.back().push(p);
203
204 if (w_ctx->local_cache.size() >= max_local_batches and w_ctx->local_cache.back().size() >= block_batch_size) {
205 // Local cache is full. Rebalance by sending batches of blocks to central cache.
206 // We leave one batch in the local cache.
207 for (unsigned i = 0; i != max_local_batches - 1; ++i) {
208 report_fatal_error_if_not(central_mem_cache.enqueue(w_ctx->producer_token, w_ctx->local_cache.back()),
209 "Failed to push allocated batch back to central cache");
210 w_ctx->local_cache.pop_back();
211 }
212 }
213 }

215 void print_all_buffers()
216 {
217 auto* worker = get_worker_cache();
218 unsigned count = 0;
219 for (const auto& l : worker->local_cache) {
220 count += l.size();
221 }
222
223 fmt::print("There are {}/{} buffers in central memory block cache. This thread contains {} in its local cache.\n",
224 central_mem_cache.size_approx() * block_batch_size,
225 nof_memory_blocks(),
226 count);
227 }

229private:
230 struct worker_ctxt {
231 /// Thread ID of the worker.
232 std::thread::id id;
233 /// Thread-local cache of memory blocks.
234 local_cache_type local_cache;
235 /// Producer Token for fast enqueueing to the central cache.
236 moodycamel::ProducerToken producer_token;
237 /// Consumer Token for fast dequeueing to the central cache.
238 moodycamel::ConsumerToken consumer_token;
239
240 worker_ctxt(fixed_size_memory_block_pool& parent) :
241 id(std::this_thread::get_id()), producer_token(parent.central_mem_cache), consumer_token(parent.central_mem_cache)
242 {
243 }
244 ~worker_ctxt()
245 {
246 pool_type& pool = pool_type::get_instance();
247 while (not local_cache.empty()) {
248 if (local_cache.back().size() < block_batch_size) {
249 // Batch is incomplete. We combine it with any other existing incomplete batch.
250 {
251 std::lock_guard<std::mutex> lock(pool.incomplete_batch_mutex);
252 while (not local_cache.back().empty()) {
253 pool.incomplete_batch.push(local_cache.back().try_pop());
254 if (pool.incomplete_batch.size() >= block_batch_size) {
255 // The incomplete batch is now complete and can be pushed to the central cache.
256 report_error_if_not(pool.central_mem_cache.enqueue(producer_token, pool.incomplete_batch),
257 "Failed to push blocks to central cache");
258 pool.incomplete_batch.clear();
259 }
260 }
261 }
262 local_cache.pop_back();
263 continue;
264 }
265 report_error_if_not(pool.central_mem_cache.enqueue(producer_token, local_cache.back()),
266 "Failed to push blocks back to central cache");
267 local_cache.pop_back();
268 }
269 }
270 };
271
272 worker_ctxt* get_worker_cache()
273 {
274 thread_local worker_ctxt worker_cache{*this};
275 return &worker_cache;
276 }
277
278 /// Number of batches of memory blocks stored in the pool.
279 size_t nof_total_batches() const { return (nof_blocks + block_batch_size - 1) / block_batch_size; }
280
281 const size_t mblock_size;
282 const size_t nof_blocks;
283 const size_t max_local_batches;
284
285 std::vector<uint8_t> allocated_memory;
286
287 moodycamel::ConcurrentQueue<free_memory_block_list> central_mem_cache;
288
289 // When workers get deleted, some local batches may be still incomplete. We collect them here to form full batches
290 // when other workers get deleted as well.
291 std::mutex incomplete_batch_mutex;
292 free_memory_block_list incomplete_batch;
293
294 std::mutex debug_mutex;
295};
