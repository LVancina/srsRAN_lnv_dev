42/// Context of a UE in the MAC DL.
43class mac_dl_ue_context
44{
45public:
46 explicit mac_dl_ue_context(const mac_ue_create_request& req);
47 // expensive copy due to HARQ buffers. Cheap move.
48 mac_dl_ue_context(const mac_dl_ue_context&) = delete;
49 mac_dl_ue_context(mac_dl_ue_context&&) noexcept = default;
50 mac_dl_ue_context& operator=(const mac_dl_ue_context&) = delete;
51 mac_dl_ue_context& operator=(mac_dl_ue_context&&) noexcept = default;
52
53 du_ue_index_t get_ue_index() const { return ue_index; }
54
55 // HARQ buffer methods.
56 span<uint8_t> dl_harq_buffer(harq_id_t h_id) { return harq_buffers[h_id]; }
57 span<const uint8_t> dl_harq_buffer(harq_id_t h_id) const { return harq_buffers[h_id]; }

59 // DL Logical Channel methods.
60 const slotted_id_vector<lcid_t, mac_sdu_tx_builder*>& logical_channels() const { return dl_bearers; }

62 void remove_logical_channels(span<const lcid_t> lcids_to_remove);

64 const ue_con_res_id_t& get_con_res_id() const { return msg3_subpdu; }

66private:

68 std::vector<std::vector<uint8_t>> harq_buffers;
69 slotted_id_vector<lcid_t, mac_sdu_tx_builder*> dl_bearers;
70 ue_con_res_id_t msg3_subpdu = {};

73/// Repository of UE MAC DL contexts.
74class mac_dl_ue_manager
75{
76public:
77 mac_dl_ue_manager(du_rnti_table& rnti_table_);
78
79 /// Check if UE with provided C-RNTI exists.
80 /// \param rnti C-RNTI of the UE.
81 /// \return True if UE exists. False, otherwise.
82 bool contains_rnti(rnti_t rnti) const
83 {
84 const du_ue_index_t ue_index = rnti_table[rnti];
85 if (is_du_ue_index_valid(ue_index)) {
86 std::lock_guard<std::mutex> lock(ue_mutex[ue_index]);
87 return ue_db.contains(ue_index);
88 }
89 return false;
90 }
91
92 /// Checks if Logical Channel with provided C-RNTI and LCID exists.
93 bool contains_lcid(rnti_t rnti, lcid_t lcid) const
94 {
95 const du_ue_index_t ue_index = rnti_table[rnti];
96 if (is_du_ue_index_valid(ue_index)) {
97 std::lock_guard<std::mutex> lock(ue_mutex[ue_index]);
98 return ue_db.contains(ue_index) and ue_db[ue_index].logical_channels().contains(lcid);
99 }
100 return false;
101 }
102
103 du_ue_index_t get_ue_index(rnti_t rnti) const
104 {
105 du_ue_index_t ue_index = rnti_table[rnti];
106 if (not is_du_ue_index_valid(ue_index)) {
107 return INVALID_DU_UE_INDEX;
108 }
109 return ue_index;
110 }
111
112 mac_sdu_tx_builder* get_bearer(rnti_t rnti, lcid_t lcid)
113 {
114 du_ue_index_t ue_index = rnti_table[rnti];
115 if (not is_du_ue_index_valid(ue_index)) {
116 return nullptr;
117 }
118 std::lock_guard<std::mutex> lock(ue_mutex[ue_index]);
119 if (not ue_db.contains(ue_index)) {
120 return nullptr;
121 }
122 auto& u = ue_db[ue_index];
123 return u.logical_channels().contains(lcid) ? u.logical_channels()[lcid] : nullptr;
124 }
125
126 bool add_ue(mac_dl_ue_context ue_to_add);
127
128 bool remove_ue(du_ue_index_t ue_index);
129
130 bool addmod_bearers(du_ue_index_t ue_index, span<const mac_logical_channel_config> dl_logical_channels);
131
132 bool remove_bearers(du_ue_index_t ue_index, span<const lcid_t> lcids);
133
134 /// \brief Returns UE Contention Resolution Id, which is derived from Msg3 bytes.
135 ue_con_res_id_t get_con_res_id(rnti_t rnti);
136
137 span<uint8_t> get_dl_harq_buffer(rnti_t rnti, harq_id_t h_id, unsigned tb_idx)
138 {
139 du_ue_index_t ue_index = rnti_table[rnti];
140 if (ue_index == INVALID_DU_UE_INDEX) {
141 return {};
142 }
143 std::lock_guard<std::mutex> lock(ue_mutex[ue_index]);
144 if (not ue_db.contains(ue_index)) {
145 return {};
146 }
147 return ue_db[ue_index].dl_harq_buffer(h_id);
148 }
149
150private:
151 du_rnti_table& rnti_table;
152
153 mutable std::array<std::mutex, MAX_NOF_DU_UES> ue_mutex;
154
155 du_ue_list<mac_dl_ue_context> ue_db;
156};
