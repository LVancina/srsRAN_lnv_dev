36/// \brief Unpacks an unsigned integer value into bits.
37///
38/// The first bit in the sequence corresponds to the bit of index <tt> nof_bits - 1 </tt> of \c value. The LSB of \c
39/// value is mapped to the last element in the sequence.
40///
41/// \param[out] bits Unpacked bits sequence.
42/// \param[in] value Provides the value to unpack.
43/// \param[in] nof_bits Indicates the number of bits to unpack.
44/// \return A view of the remaining (behind \c nof_bits) unpacked bits of \c value.
45span<uint8_t> bit_unpack(span<uint8_t> bits, unsigned value, unsigned nof_bits);
46
47/// \brief Unpacks a bit buffer into bits.
48/// \param[out] unpacked View of the unpacked bits.
49/// \param[in] packed Bit buffer to unpack.
50/// \remark The number of unpacked elements must be equal to the packed number of bits.
51void bit_unpack(span<uint8_t> unpacked, const bit_buffer& packed);
52
53/// \brief Unpacks a bit buffer into bits.
54/// \param[out] unpacked View of the unpacked bits.
55/// \param[in] packed Bit buffer to unpack.
56/// \param[in] offset Packed initial bit index.
57void bit_unpack(span<uint8_t> unpacked, const bit_buffer& packed, unsigned offset);
58
59/// \brief Packs a number of bits into an integer value.
60/// \param[in,out] bits View of unpacked bits.
61/// \param[in] nof_bits Indicates the number of bits.
62/// \return An integer containing the packed bits.
63/// \remark The number of elements in bits must be greater than or equal to \c nof_bits.
64/// \remark \c nof_bits must not exceed 32 bits.
65/// \remark After execution, \c bits will contain only the unused bits.
66unsigned bit_pack(span<const uint8_t>& bits, unsigned nof_bits);
67
68/// \brief Packs the input bits into an integer value.
69///
70/// The first bit in the sequence corresponds to the bit of index <tt> bits.size() - 1 </tt> of the returned value. The
71/// last value in the sequence corresponds to LSB of the returned value.
72///
73/// \param[in] bits View of unpacked bits.
74/// \return An integer containing the packed bits.
75/// \remark The number of elements must not exceed 32 bits.
76unsigned bit_pack(span<const uint8_t> bits);
77
78/// \brief Packs a number of bits into a bit buffer.
79/// \param[out] packed Destination bit buffer.
80/// \param[in] unpacked View of unpacked bits.
81/// \remark The number of unpacked elements must be equal to the maximum number of bits supported by the bit buffer.
82void bit_pack(bit_buffer& packed, span<const uint8_t> unpacked);
83
84/// \brief Packs a number of bits into a bit buffer.
85/// \param[out] packed Destination bit buffer.
86/// \param[in] offset Packed initial bit index.
87/// \param[in] unpacked View of unpacked bits.
88/// \remark The number of unpacked elements must be smaller than or equal to the maximum number of bits supported by the
89/// bit buffer minus \c offset.
90void bit_pack(bit_buffer& packed, unsigned offset, span<const uint8_t> unpacked);
91
92/// \brief Copies \c output.size() bits from \c input, starting at \c startpos, into \c output.
93/// \param[out] output Destination of the copy.
94/// \param[in] input Data source to copy.
95/// \param[in] startpos Bit position index to start the copy.
96void copy_offset(bit_buffer& output, span<const uint8_t> input, unsigned startpos);
97
98/// \brief Copies bits from one buffer to another.
99///
100/// Copies \c nof_bits bits from \c input, starting at \c in_offset, into \c output, starting at \c out_offset.
101/// \param[out] output Destination of the copy.
102/// \param[in] out_offset Output bit position index to start the copy.
103/// \param[in] input Data source to copy.
104/// \param[in] in_offset Input bit position index to start the copy.
105/// \param[in] nof_bits Number of bits to copy.
106void copy_offset(bit_buffer& output,
107 unsigned out_offset,
108 const bit_buffer& input,
109 unsigned in_offset,
110 unsigned nof_bits);
