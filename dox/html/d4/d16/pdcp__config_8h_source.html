41/// PDCP NR sequence number field.
42enum class pdcp_sn_size : uint8_t { size12bits = 12, size18bits = 18 };
43inline bool pdcp_sn_size_from_uint(pdcp_sn_size& sn_size, uint16_t num)
44{
45 if (num == 12) {
46 sn_size = pdcp_sn_size::size12bits;
47 return true;
48 }
49 if (num == 18) {
50 sn_size = pdcp_sn_size::size18bits;
51 return true;
52 }
53 return false;
54}
55
56/// \brief Convert PDCP SN size from enum to unsigned integer.
57constexpr uint8_t pdcp_sn_size_to_uint(pdcp_sn_size sn_size)
58{
59 return static_cast<uint8_t>(sn_size);
60}
61
62/// \brief Returns the value range of the sequence numbers
63/// \param sn_size Length of the sequence number field in bits
64/// \return cardinality of sn_size
65constexpr uint32_t pcdp_sn_cardinality(uint16_t sn_size)
66{
67 srsran_assert(sn_size < 32, "Cardinality of sn_size={} exceeds return type 'uint32_t'", sn_size);
68 return (1 << sn_size);
69}
70
71/// \brief Returns the PDCP window size
72/// \param sn_size Length of the sequence number field in bits
73/// \return size of the window
74constexpr uint32_t pdcp_window_size(uint16_t sn_size)
75{
76 return pcdp_sn_cardinality(sn_size - 1);
77}

85/// PDCP NR discard timer values.
86/// This timer is configured only for DRBs. In the transmitter,
87/// a new timer is started upon reception of an SDU from upper layer.
88/// See TS 38.322 for timer description and TS 38.331 for valid timer durations.
89enum class pdcp_discard_timer {
90 ms10 = 10,
91 ms20 = 20,
92 ms30 = 30,
93 ms40 = 40,
94 ms50 = 50,
95 ms60 = 60,
96 ms75 = 75,
97 ms100 = 100,
98 ms150 = 150,
99 ms200 = 200,
100 ms250 = 250,
101 ms300 = 300,
102 ms500 = 500,
103 ms750 = 750,
104 ms1500 = 1500,
105 infinity = -1
106};
107inline bool pdcp_discard_timer_from_int(pdcp_discard_timer& discard_timer, int num)
108{
109 switch (num) {
110 case 10:
111 case 20:
112 case 30:
113 case 40:
114 case 50:
115 case 60:
116 case 75:
117 case 100:
118 case 150:
119 case 200:
120 case 250:
121 case 300:
122 case 500:
123 case 750:
124 case 1500:
125 case -1:
126 discard_timer = static_cast<pdcp_discard_timer>(num);
127 return true;
128 default:
129 return false;
130 }
131}

139/// The PDCP cannot re-use COUNTs, see TS 38.331, section 5.3.1.2.
140/// To avoid this, we define two thresholds, one where we accept messages
141/// but notify the RRC and another where we no longer accept messages.
142/// Here, we define some default values for this, both for TX and RX.
143const uint32_t pdcp_tx_default_max_count_notify = 0xc0000000;
144const uint32_t pdcp_tx_default_max_count_hard = UINT32_MAX - 1;
145const uint32_t pdcp_rx_default_max_count_notify = 0xc0000000;
146const uint32_t pdcp_rx_default_max_count_hard = 0xd0000000;
147struct pdcp_max_count {
148 uint32_t notify;
149 uint32_t hard;
150};

203/// \brief Configurable parameters for PDCP
204///
205/// Parameters and valid values for them are taken from
206/// the RRC-NR PDCP-Config Information Element.
207///
208/// Ref: 3GPP TS 38.331 version 15.2.1
209struct pdcp_config {
210 pdcp_rb_type rb_type;
211 pdcp_rlc_mode rlc_mode;
212 bool integrity_protection_required;
213 bool ciphering_required;
214 struct {
215 pdcp_sn_size sn_size;
216 pdcp_security_direction direction;
217 optional<pdcp_discard_timer> discard_timer;
218 bool status_report_required;
219 } tx;
220 struct {
221 pdcp_sn_size sn_size;
222 pdcp_security_direction direction;
223 bool out_of_order_delivery;
224 pdcp_t_reordering t_reordering;
225 } rx;
226 pdcp_custom_config custom;
227
228 pdcp_tx_config get_tx_config() const
229 {
230 pdcp_tx_config cfg;
231 cfg.rb_type = rb_type;
232 cfg.rlc_mode = rlc_mode;
233 cfg.integrity_protection_required = integrity_protection_required;
234 cfg.sn_size = tx.sn_size;
235 cfg.direction = tx.direction;
236 cfg.discard_timer = tx.discard_timer;
237 cfg.status_report_required = tx.status_report_required;
238 cfg.custom = custom.tx;
239 return cfg;
240 }
241 pdcp_rx_config get_rx_config() const
242 {
243 pdcp_rx_config cfg;
244 cfg.rb_type = rb_type;
245 cfg.rlc_mode = rlc_mode;
246 cfg.integrity_protection_required = integrity_protection_required;
247 cfg.sn_size = rx.sn_size;
248 cfg.direction = rx.direction;
249 cfg.out_of_order_delivery = rx.out_of_order_delivery;
250 cfg.t_reordering = rx.t_reordering;
251 cfg.custom = custom.rx;
252 return cfg;
253 }
254};

256/// \brief Make default SRB parameters for PDCP
257/// Ref: 3GPP TS 38.331, section 9.2.1
258inline pdcp_config pdcp_make_default_srb_config()
259{
260 pdcp_config config = {};
261 // common TX/RX parameters
262 config.rb_type = pdcp_rb_type::srb;
263 config.rlc_mode = pdcp_rlc_mode::am;
264 config.integrity_protection_required = true;
265 config.ciphering_required = true;
266
267 // Tx config
268 config.tx.sn_size = pdcp_sn_size::size12bits;
269 config.tx.direction = pdcp_security_direction::downlink;
270 config.tx.status_report_required = false;
271
272 // Rx config
273 config.rx.sn_size = pdcp_sn_size::size12bits;
274 config.rx.direction = pdcp_security_direction::uplink;
275 config.rx.out_of_order_delivery = false;
276 config.rx.t_reordering = pdcp_t_reordering::infinity;
277
278 // Custom config
279 config.custom = {};
280 config.custom.tx.rlc_sdu_queue = 512;
281
282 return config;
283}
