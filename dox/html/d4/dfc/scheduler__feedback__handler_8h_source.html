79/// \brief UCI indication for a given UE.
80struct uci_indication {
81 struct uci_pdu {
82 struct uci_pucch_f0_or_f1_pdu {
83 constexpr static size_t NOF_HARQS_PER_UCI = 2;
84 bool sr_detected;
85 static_vector<mac_harq_ack_report_status, NOF_HARQS_PER_UCI> harqs;
86 /// \brief Metric of channel quality that ranges from -65.534 to 65.534 dBs.
87 optional<float> ul_sinr;
88 /// \brief Timing Advance Offset measured for the UE.
89 optional<phy_time_unit> time_advance_offset;
90 };
91 struct uci_pusch_pdu {
92 static_vector<mac_harq_ack_report_status, uci_constants::MAX_NOF_HARQ_BITS> harqs;
93 optional<csi_report_data> csi;
94 };
95 struct uci_pucch_f2_or_f3_or_f4_pdu {
96 /// Maximum number of SR bits expected on the PUCCH transmission.
97 static const unsigned MAX_SR_PAYLOAD_SIZE_BITS = 4;
98
99 bounded_bitset<MAX_SR_PAYLOAD_SIZE_BITS> sr_info;
100 static_vector<mac_harq_ack_report_status, uci_constants::MAX_NOF_HARQ_BITS> harqs;
101 optional<csi_report_data> csi;
102 /// \brief Metric of channel quality that ranges from -65.534 to 65.534 dBs.
103 optional<float> ul_sinr;
104 /// \brief Timing Advance Offset measured for the UE.
105 optional<phy_time_unit> time_advance_offset;
106 };
107
108 du_ue_index_t ue_index;
109 rnti_t crnti;
110 variant<uci_pucch_f0_or_f1_pdu, uci_pusch_pdu, uci_pucch_f2_or_f3_or_f4_pdu> pdu;
111 };
112 using uci_pdu_list = static_vector<uci_pdu, MAX_UCI_PDUS_PER_UCI_IND>;
113
114 // Note: user-defined ctor to avoid zero-initialization of uci_pdu_list.
115 uci_indication() {}
116
117 du_cell_index_t cell_index;
118 slot_point slot_rx;
119 uci_pdu_list ucis;
120};
