39[[gnu::noinline, noreturn]] inline void print_and_abort(const char* filename,
40 int line,
41 const char* funcname,
42 const char* condstr = nullptr,
43 const std::string& msg = "") noexcept
44{
45 fmt::memory_buffer fmtbuf;
46 fmt::format_to(fmtbuf, "{}:{}: {}: \n", filename, line, funcname);
47 if (condstr == nullptr) {
48 fmt::format_to(fmtbuf, "Assertion failed");
49 } else {
50 fmt::format_to(fmtbuf, "Assertion `{}' failed", condstr);
51 }
52 if (not msg.empty()) {
53 fmt::format_to(fmtbuf, " - {}", msg);
54 }
55 if (msg.back() != '.') {
56 fmt::format_to(fmtbuf, ".");
57 }
58 fmt::format_to(fmtbuf, "\n");
59 fmtbuf.push_back('\0'); // make it a c-string
60
61 srslog::flush();
62 fmt::print(stderr, "{}", fmtbuf.data());
63 std::abort();
64}

73#define SRSRAN_ASSERT_FAILURE__(condmessage, fmtstr, ...) \
74 srsran::detail::print_and_abort( \
75 __FILE__, __LINE__, __PRETTY_FUNCTION__, condmessage, fmt::format(FMT_STRING(fmtstr), ##__VA_ARGS__))

79#define SRSRAN_ALWAYS_ASSERT__(condition, fmtstr, ...) \
80 (void)((condition) || (SRSRAN_ASSERT_FAILURE__((#condition), fmtstr, ##__VA_ARGS__), 0))

83#define SRSRAN_ALWAYS_ASSERT_IFDEF__(enable_check, condition, fmtstr, ...) \
84 (void)((not SRSRAN_IS_DEFINED(enable_check)) || (SRSRAN_ALWAYS_ASSERT__(condition, fmtstr, ##__VA_ARGS__), 0))

87#define srsran_assertion_failure(fmtstr, ...) \
