38class pdxch_processor_baseband_spy : public pdxch_processor_baseband
39{
40public:
41 struct entry_t {
42 baseband_gateway_buffer_read_only samples;
43 symbol_context context;
44 };
45
46 // Constructs a PDxCH processor baseband spy that processes all slots and symbols.
47 pdxch_processor_baseband_spy() : random_data(1009)
48 {
49 std::mt19937 rgen;
50 std::uniform_real_distribution<float> sample_dist(-1, 1);
51 std::generate(random_data.begin(), random_data.end(), [&sample_dist, &rgen]() {
52 return cf_t(sample_dist(rgen), sample_dist(rgen));
53 });
54 }
55
56 // Constructs a PDxCH processor baseband spy that processes only the symbols belonging to the slots in
57 // frame_slots_to_process.
58 pdxch_processor_baseband_spy(std::vector<unsigned> frame_slots_to_process_) :
59 random_data(1009), frame_slots_to_process(std::move(frame_slots_to_process_))
60 {
61 std::mt19937 rgen;
62 std::uniform_real_distribution<float> sample_dist(-1, 1);
63 std::generate(random_data.begin(), random_data.end(), [&sample_dist, &rgen]() {
64 return cf_t(sample_dist(rgen), sample_dist(rgen));
65 });
66 }
67
68 bool process_symbol(baseband_gateway_buffer_writer& samples, const symbol_context& context) override
69 {
70 if (!frame_slots_to_process.empty()) {
71 // Skip processing if the slot is not to be processed.
72 if (std::find(frame_slots_to_process.begin(), frame_slots_to_process.end(), context.slot.slot_index()) ==
73 frame_slots_to_process.end()) {
74 return false;
75 }
76 }
77
78 // Counter for the remaining samples to generate.
79 std::size_t remaining = samples.get_nof_samples();
80 span<cf_t> random_data_view(random_data);
81
82 while (remaining > 0) {
83 // Number of samples to copy for this iteration.
84 unsigned nof_copied_samples = std::min(random_data.size() - random_data_index, remaining);
85 span<cf_t> random_samples = random_data_view.subspan(random_data_index, nof_copied_samples);
86
87 for (unsigned i_ch = 0, i_channel_end = samples.get_nof_channels(); i_ch != i_channel_end; ++i_ch) {
88 span<cf_t> channel_samples =
89 samples.get_channel_buffer(i_ch).subspan(samples.get_nof_samples() - remaining, nof_copied_samples);
90
91 srsvec::copy(channel_samples, random_samples);
92 }
93
94 // Update remaining samples and buffer index.
95 remaining -= nof_copied_samples;
96 random_data_index = (random_data_index + nof_copied_samples) % random_data.size();
97 }
98
99 entries.emplace_back();
100 entry_t& entry = entries.back();
101 entry.context = context;
102 entry.samples = samples;
103
104 return true;
105 }
106
107 const std::vector<entry_t>& get_entries() const { return entries; }
108
109 void clear() { entries.clear(); }
110
111private:
112 unsigned random_data_index = 0;
113 std::vector<cf_t> random_data;
114 std::vector<entry_t> entries;
115 std::vector<unsigned> frame_slots_to_process;
116};
