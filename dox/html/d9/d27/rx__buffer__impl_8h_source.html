40enum class rx_buffer_status : uint8_t { successful = 0, already_in_use, insufficient_cb };
41
42/// Implements a receiver buffer interface.
43class rx_buffer_impl : public unique_rx_buffer::callback
44{
45private:
46 /// Internal buffer states.
47 enum class state : uint8_t {
48 /// \brief The buffer has not been configured with any codeblock.
49 ///
50 /// It is available for being reserved.
51 available,
52 /// \brief The buffer is reserved with a number of codeblocks.
53 ///
54 /// It allows:
55 /// -\c expire: called when the buffer context has expired, it transitions to \c available.
56 /// -\c reserve: makes a new reservation.
57 reserved,
58 /// \brief The buffer is reserved and locked in a context. It does not accept new reservations or retransmissions,
59 /// and prevents the buffer from expiring.
60 ///
61 /// It allows:
62 /// - \c unlock: a transmission did not match the CRC, it transitions to \c reserved; and
63 /// - \c release: a transmission matched the CRC and the buffer releases all its resources, it transitions to \c
64 /// available.
65 locked
66 };
67
68 /// Current buffer state.
69 std::atomic<state> current_state = {state::available};
70 /// Reference to the codeblock pool.
71 rx_buffer_codeblock_pool& codeblock_pool;
72 /// Stores codeblocks CRCs.
73 static_vector<bool, MAX_NOF_SEGMENTS> crc;
74 /// Stores codeblock identifiers.
75 static_vector<unsigned, MAX_NOF_SEGMENTS> codeblock_ids;
76
77 /// Frees reserved codeblocks. The codeblocks are returned to the pool.
78 void free()
79 {
80 // Free all codeblocks.
81 for (unsigned cb_id : codeblock_ids) {
82 codeblock_pool.free(cb_id);
83 }
84
85 // Indicate the buffer is available by clearing the codeblocks identifiers.
86 codeblock_ids.clear();
87 }
88
89public:
90 /// \brief Creates a receive buffer.
91 /// \param pool Codeblock buffer pool.
92 explicit rx_buffer_impl(rx_buffer_codeblock_pool& pool) : codeblock_pool(pool)
93 {
94 // Do nothing.
95 }
96
97 /// Copy constructor - creates another buffer with the same codeblock pool.
98 rx_buffer_impl(const rx_buffer_impl& other) noexcept : codeblock_pool(other.codeblock_pool) {}
99
100 /// Move constructor - creates another buffer with the same codeblock pool.
101 rx_buffer_impl(rx_buffer_impl&& other) noexcept : codeblock_pool(other.codeblock_pool) {}
102
103 /// \brief Reserves a number of codeblocks from the pool.
104 ///
105 /// It optionally resets the CRCs.
106 ///
107 /// \param nof_codeblocks Number of codeblocks to reserve.
108 /// \param reset_crc Set to true for reset the codeblock CRCs.
109 /// \return The reservation status.
110 rx_buffer_status reserve(unsigned nof_codeblocks, bool reset_crc)
111 {
112 // It cannot reserve resources if it is locked.
113 if (current_state.load() == state::locked) {
114 return rx_buffer_status::already_in_use;
115 }
116
117 // If the current number of codeblocks is larger than required, free the excess of codeblocks.
118 while (codeblock_ids.size() > nof_codeblocks) {
119 // Get the codeblock identifier at the back and remove from the list.
120 unsigned cb_id = codeblock_ids.back();
121 codeblock_ids.pop_back();
122
123 // Free the codeblock.
124 codeblock_pool.free(cb_id);
125 }
126
127 // If the current number of codeblocks is less than required, reserve the remaining codeblocks.
128 while (codeblock_ids.size() < nof_codeblocks) {
129 // Reserve codeblock.
130 optional<unsigned> cb_id = codeblock_pool.reserve();
131
132 // Free the entire buffer if one codeblock cannot be reserved.
133 if (!cb_id.has_value()) {
134 free();
135 current_state = state::available;
136 return rx_buffer_status::insufficient_cb;
137 }
138
139 // Append the codeblock identifier to the list.
140 codeblock_ids.push_back(cb_id.value());
141 }
142
143 // Resize CRCs.
144 crc.resize(nof_codeblocks);
145
146 // Reset CRCs if necessary.
147 if (reset_crc) {
148 reset_codeblocks_crc();
149 }
150 current_state = state::reserved;
151
152 return rx_buffer_status::successful;
153 }
154
155 // See interface for documentation.
156 unsigned get_nof_codeblocks() const override { return crc.size(); }
157
158 // See interface for documentation.
159 void reset_codeblocks_crc() override
160 {
161 // Set all codeblock CRC to false.
162 for (bool& cb_crc : crc) {
163 cb_crc = false;
164 }
165 }
166
167 // See interface for documentation.
168 span<bool> get_codeblocks_crc() override { return crc; }
169
170 // See interface for documentation.
171 unsigned get_absolute_codeblock_id(unsigned codeblock_id) const override
172 {
173 srsran_assert(codeblock_id < codeblock_ids.size(),
174 "Codeblock index ({}) is out of range ({}).",
175 codeblock_id,
176 codeblock_ids.size());
177 return codeblock_ids[codeblock_id];
178 }
179
180 // See interface for documentation.
181 span<log_likelihood_ratio> get_codeblock_soft_bits(unsigned codeblock_id, unsigned codeblock_size) override
182 {
183 srsran_assert(codeblock_id < codeblock_ids.size(),
184 "Codeblock index ({}) is out of range ({}).",
185 codeblock_id,
186 codeblock_ids.size());
187 unsigned cb_id = codeblock_ids[codeblock_id];
188 unsigned cb_max_size = codeblock_pool.get_soft_bits(cb_id).size();
189 srsran_assert(
190 codeblock_size <= cb_max_size, "Codeblock size {} exceeds maximum size {}.", codeblock_size, cb_max_size);
191 return codeblock_pool.get_soft_bits(cb_id).first(codeblock_size);
192 }
193
194 // See interface for documentation.
195 bit_buffer get_codeblock_data_bits(unsigned codeblock_id, unsigned data_size) override
196 {
197 srsran_assert(codeblock_id < codeblock_ids.size(),
198 "Codeblock index ({}) is out of range ({}).",
199 codeblock_id,
200 codeblock_ids.size());
201 unsigned cb_id = codeblock_ids[codeblock_id];
202 unsigned data_max_size = codeblock_pool.get_data_bits(cb_id).size();
203 srsran_assert(
204 data_size <= data_max_size, "Codeblock data size {} exceeds maximum size {}.", data_size, data_max_size);
205 return codeblock_pool.get_data_bits(cb_id).first(data_size);
206 }
207
208 // See interface for documentation.
209 void lock() override
210 {
211 state previous_state = current_state.exchange(state::locked);
212 srsran_assert(previous_state == state::reserved, "Failed to lock. Invalid state.");
213 }
214
215 // See interface for documentation.
216 void unlock() override
217 {
218 state previous_state = current_state.exchange(state::reserved);
219 srsran_assert(previous_state == state::locked, "Failed to unlock. Invalid state.");
220 }
221
222 // See interface for documentation.
223 void release() override
224 {
225 // Release all reserved codeblocks.
226 free();
227
228 // The buffer can now be reused again.
229 state previous_state = current_state.exchange(state::available);
230 srsran_assert(previous_state == state::locked, "Failed to release. Invalid state.");
231 }
232
233 /// Returns true if the buffer is free.
234 bool is_free() const { return current_state == state::available; }
235
236 /// Returns true if the buffer is locked.
237 bool is_locked() const { return current_state == state::locked; }
238
239 /// \brief Expires the buffer.
240 ///
241 /// The buffer pool shall use this method when the buffer expires. The buffer frees the reserved codeblocks if it is
242 /// not blocked.
243 ///
244 /// \return \c true if the buffer is not locked.
245 bool expire()
246 {
247 state expected_state = state::reserved;
248 bool from_reserved = current_state.compare_exchange_weak(expected_state, state::available);
249
250 // The buffer cannot be freed if it is locked.
251 if (!from_reserved) {
252 return false;
253 }
254
255 // Release resources.
256 free();
257
258 return true;
259 }
260};
