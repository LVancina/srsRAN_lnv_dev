35/// \brief Keeps track of sample statistics.
36///
37/// \tparam T Type (must be arithmetic) of the sample values.
38template <typename T>
39class sample_statistics
40{
41public:
42 /// Estimator types depending on bias.
43 enum class bias {
44 /// Denotes unbiased estimators.
45 UNBIASED,
46 /// Denotes biased estimators.
47 BIASED
48 };
49
50 /// Default constructor.
51 sample_statistics() = default;
52
53 /// Clears the content of the internal registers.
54 void reset() { *this = sample_statistics(); }
55
56 /// Updates the statistics by adding an observation to the sample.
57 void update(T value)
58 {
59 float delta = value - mean;
60 ++nof_observations;
61 mean += delta / nof_observations;
62
63 // The following update formulas are inspired by
64 // Philippe Pebay, "Formulas for Robust, One-Pass Parallel Computation of Covariances and Arbitrary-Order
65 // Statistical Moments," SANDIA REPORT SAND2008-6212 (2008).
66 // [Online]. Available: https://www.osti.gov/servlets/purl/1028931
67 float nof_obs_sqrd = static_cast<float>(nof_observations) * static_cast<float>(nof_observations);
68 float delta_sqrd = delta * delta;
69 centered_power_sum_ord4 += -4 * centered_power_sum_ord3 / nof_observations * delta +
70 6 * centered_power_sum_ord2 / nof_obs_sqrd * delta_sqrd +
71 (nof_observations - 1.0F) * (nof_obs_sqrd - 3.0F * nof_observations + 3.0F) /
72 (nof_obs_sqrd * nof_observations) * delta_sqrd * delta_sqrd;
73
74 centered_power_sum_ord3 += -3 * centered_power_sum_ord2 / nof_observations * delta +
75 (nof_obs_sqrd - 3.0F * nof_observations + 2.0F) / nof_obs_sqrd * delta_sqrd * delta;
76
77 centered_power_sum_ord2 += (nof_observations - 1.0F) / nof_observations * delta_sqrd;
78
79 max = std::max(max, value);
80 min = std::min(min, value);
81 }
82
83 /// \brief Returns the maximum sample value.
84 /// \remark The function returns std::numeric_limits<T>::min() when the sample is empty.
85 T get_max() const { return max; }
86
87 /// \brief Returns the minimum sample value.
88 /// \remark The function returns std::numeric_limits<T>::max() when the sample is empty.
89 T get_min() const { return min; }
90
91 /// Returns the sample size, i.e. the number of observations.
92 size_t get_nof_observations() const { return nof_observations; }
93
94 /// \brief Returns the sample mean.
95 ///
96 /// This is simply the arithmetic mean of the \f$n\f$ values in the sample.
97 /// \f[
98 /// \mu = \frac{1}{n} \sum_{i=0}^{n-1} x_i.
99 /// \f]
100 /// \remark The function returns NAN when the sample is empty.
101 float get_mean() const
102 {
103 if (nof_observations < 1) {
104 return NAN;
105 }
106 return mean;
107 }
108
109 /// \brief Returns the sample variance.
110 ///
111 /// The sample variance is given by
112 /// \f[
113 /// \sigma^2 = \frac{1}{m} \sum_{i=0}^{n-1} (x_i - \mu)^2
114 /// \f]
115 /// where \f$\mu\f$ is the [sample mean](#sample_statistics::get_mean) and where \f$m = n\f$ or \f$m = n-1\f$ for the
116 /// biased and unbiased estimators, respectively, of the statistical variance
117 /// \f$\EV\Bigl[\bigl(X - \EV[X]\bigr)^2\Bigr]\f$.
118 /// \remark The function returns NAN when the sample consists of less than two observations.
119 float get_variance(bias biased = bias::UNBIASED) const
120 {
121 if (nof_observations < 2) {
122 return NAN;
123 }
124
125 if (biased == bias::UNBIASED) {
126 return centered_power_sum_ord2 / (nof_observations - 1);
127 }
128 return centered_power_sum_ord2 / nof_observations;
129 }
130
131 /// \brief Returns the sample standard deviation.
132 ///
133 /// This is simply the squared root of the [sample covariance](#sample_statistics::get_variance). The \c biased flag
134 /// simply denotes how the sample covariance is computed.
135 /// \remark The function returns NAN when the sample consists of less than two observations.
136 float get_std(bias biased = bias::UNBIASED) const
137 {
138 if (nof_observations < 2) {
139 return NAN;
140 }
141 return std::sqrt(get_variance(biased));
142 }
143
144 /// \brief Returns the standard error of the sample mean.
145 ///
146 /// The standard error of the mean is an estimation of the standard deviation of the
147 /// [sample mean](#sample_statistics::get_mean) (seen itself as a random quantity):
148 /// \f[
149 /// \sigma_{\mu} = \frac{\sigma}{\sqrt{n}}
150 /// \f]
151 /// where \f$n\f$ is the sample size and \f$\sigma\f$ is the [sample covariance](#sample_statistics::get_variance).
152 /// The \c biased flag simply denotes how the sample covariance is computed.
153 /// \note It is implicitly assumed that the observations are statistically independent.
154 /// \remark The function returns NAN when the sample consists of less than two observations.
155 float get_sem(bias biased = bias::UNBIASED) const
156 {
157 if (nof_observations < 2) {
158 return NAN;
159 }
160 return get_std(biased) / std::sqrt(nof_observations);
161 }
162
163 /// \brief Returns the sample skewness.
164 ///
165 /// The skewness of a distribution with mean \f$\mu_*\f$ and standard deviation \f$\sigma_*\f$ is defined as
166 /// \f[
167 /// s_* = \frac{\EV\bigl[(X-\mu_*)^3\bigr]}{\sigma_*^3}.
168 /// \f]
169 /// For a sample of \f$n\f$ observations with [sample mean](#sample_statistics::get_mean) \f$\mu\f$, the \em biased
170 /// sample skewness is given by the Fisher&ndash;Pearson coefficient
171 /// \f[
172 /// g_1 = \frac{\frac{1}{n}\sum_{i=0}^{n-1}(x_i - \mu)^3}{\Bigl[\frac{1}{n}\sum_{i=0}^{n-1}(x_i - \mu)^2\Bigr]^{3/2}}.
173 /// \f]
174 /// The \em unbiased sample skewness is obtained by applying the adjustment
175 /// \f[
176 /// G_1 = \frac{\sqrt{n(n-1)}}{n-2}g_1.
177 /// \f]
178 /// \remark The function returns NAN when the sample consists of less than three observations.
179 /// \note The skewness provides a rough indication of the symmetry of the distribution of the sample around its mean.
180 /// More specifically, for unimodal distributions, a negative-valued skewness means that the left tail of the
181 /// distribution is longer than the right one and most of the mass is concentrated to the right of the mean. In other
182 /// words, the median is higher than the mean. A positive skewness means the opposite. Symmetric distributions have
183 /// null skewness. However, the opposite is not always true as, e.g., a long left tail and a fat right tail may
184 /// compensate each other and give skewness equal to zero.
185 float get_skewness(bias biased = bias::UNBIASED) const
186 {
187 if (nof_observations < 3) {
188 return NAN;
189 }
190 float num = centered_power_sum_ord3 / nof_observations;
191 float den_tmp = std::sqrt(centered_power_sum_ord2 / nof_observations);
192 float den = den_tmp * den_tmp * den_tmp;
193
194 float result = num / den;
195
196 if (biased == bias::UNBIASED) {
197 result *= std::sqrt(nof_observations * (nof_observations - 1.0F)) / (nof_observations - 2.0F);
198 }
199 return result;
200 }
201
202 /// \brief Returns the sample kurtosis.
203 ///
204 /// The kurtosis of a distribution with mean \f$\mu_*\f$ and standard deviation \f$\sigma_*\f$ is defined as
205 /// \f[
206 /// k_* = \frac{\EV\bigl[(X-\mu_*)^4\bigr]}{\sigma_*^4}.
207 /// \f]
208 /// For a sample of \f$n\f$ observations with [sample mean](#sample_statistics::get_mean) \f$\mu\f$, the \em biased
209 /// sample kurtosis is given by the expression
210 /// \f[
211 /// g_2 = \frac{\frac{1}{n}\sum_{i=0}^{n-1}(x_i - \mu)^4}{\Bigl[\frac{1}{n}\sum_{i=0}^{n-1}(x_i - \mu)^2\Bigr]^{2}}.
212 /// \f]
213 /// The \em unbiased sample kurtosis is obtained by applying the adjustment
214 /// \f[
215 /// G_2 = \frac{n-1}{(n-2)(n-3)}\bigl[(n+1)g_1 - 3(n-1)\bigr] + 3.
216 /// \f]
217 /// \remark The returns NAN when the sample consists of less than four observations.
218 /// \note The normal distribution has kurtosis equal to 3 (for this reason, sometimes the kurtosis is defined as
219 /// \f$g_2 - 3\f$ to get a value of zero with the normal distribution).
220 /// \note Distributions with tails that decrease slower
221 /// (faster, respectively) than those of the Gaussian distribution have a kurtosis that is larger
222 /// (smaller, respectively) than 3. In other words, samples with a high kurtosis suggest that the underlying
223 /// distribution has fat tails and/or that there is a significant number of outliers outside the interval \f$\mu \pm
224 /// \sigma\f$.
225 float get_kurtosis(bias biased = bias::UNBIASED) const
226 {
227 if (nof_observations < 4) {
228 return NAN;
229 }
230 float num = centered_power_sum_ord4 / nof_observations;
231 float den_tmp = centered_power_sum_ord2 / nof_observations;
232 float den = den_tmp * den_tmp;
233
234 float result = num / den;
235
236 if (biased == bias::UNBIASED) {
237 result = result * (nof_observations + 1) - 3.0F * (nof_observations - 1);
238 result = result * (nof_observations - 1) / ((nof_observations - 2.0F) * (nof_observations - 3.0F)) + 3.0F;
239 }
240 return result;
241 }
242
243private:
244 /// Ensure the sample type is arithmetic.
245 static_assert(std::is_arithmetic<T>::value, "Invalid type.");
246 /// Records the maximum observed value.
247 T max = std::numeric_limits<T>::min();
248 /// Records the minimum observed value.
249 T min = std::numeric_limits<T>::max();
250 /// Records the sample size.
251 size_t nof_observations = 0;
252 /// Records the mean of the sample.
253 float mean = 0;
254 /// Records the centered power sum of order 2 of all observations, i.e. \f$\sum_n (X_n - \bar{X})^2\f$.
255 float centered_power_sum_ord2 = 0;
256 /// Records the centered power sum of order 3 of all observations, i.e. \f$\sum_n (X_n - \bar{X})^3\f$.
257 float centered_power_sum_ord3 = 0;
258 /// Records the centered power sum of order 4 of all observations, i.e. \f$\sum_n (X_n - \bar{X})^4\f$.
259 float centered_power_sum_ord4 = 0;
260};
