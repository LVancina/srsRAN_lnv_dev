41/// \brief PUSCH demodulator interface.
42///
43/// The demodulation of a PUSCH consists of:
44/// - extracting allocated REs from the resource grid,
45/// - equalizing of the extracted RE,
46/// - soft-demodulation of the complex data, and
47/// - descrambling.
48class pusch_demodulator
49{
50public:
51 /// Parameters defining the demodulation procedure of a PUSCH transmission.
52 struct configuration {
53 /// Radio Network Temporary Identifier, see parameter \f$n_{RNTI}\f$ in TS38.211 Section 6.3.1.1.
54 uint16_t rnti;
55 /// Allocation RB list: the entries set to true are used for transmission.
56 bounded_bitset<MAX_RB> rb_mask;
57 /// Modulation scheme used for transmission.
58 modulation_scheme modulation;
59 /// Time domain allocation within a slot: start symbol index {0, ..., 12}.
60 unsigned start_symbol_index;
61 /// Time domain allocation within a slot: number of symbols {1, ..., 14}.
62 unsigned nof_symbols;
63 /// OFDM symbols containing DM-RS: boolean mask.
64 symbol_slot_mask dmrs_symb_pos;
65 /// DM-RS configuration type.
66 dmrs_type dmrs_config_type;
67 /// Number of DM-RS CDM groups without data.
68 unsigned nof_cdm_groups_without_data;
69 /// Scrambling identifier, see parameter \f$n_{ID}\f$ in TS38.211 Section 6.3.1.1. Range is {0, ..., 1023}.
70 unsigned n_id;
71 /// Number of transmit layers.
72 unsigned nof_tx_layers;
73 /// Receive antenna port indices the PUSCH transmission is mapped to.
74 static_vector<uint8_t, MAX_PORTS> rx_ports;
75 };
76
77 /// Default destructor.
78 virtual ~pusch_demodulator() = default;
79
80 /// \brief Demodulates a PUSCH transmission.
81 ///
82 /// Computes log-likelihood ratios from channel samples by reversing all the operations described in TS38.211 Section
83 /// 6.3.1.
84 ///
85 /// \remarks
86 /// - The size of \c data determines the codeword size in soft bits.
87 /// - The total number of LLR must be consistent with the number of RE allocated to the transmission.
88 ///
89 /// \param[out] codeword_buffer Codeword buffer.
90 /// \param[in] notifier Demodulation statistics notifier.
91 /// \param[in] grid Resource grid for the current slot.
92 /// \param[in] estimates Channel estimates for the REs allocated to the PUSCH transmission.
93 /// \param[in] config Configuration parameters.
94 virtual void demodulate(pusch_codeword_buffer& codeword_buffer,
95 pusch_demodulator_notifier& notifier,
96 const resource_grid_reader& grid,
97 const channel_estimate& estimates,
98 const configuration& config) = 0;
99};
