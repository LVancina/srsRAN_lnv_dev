28/**
29 *
30 * @file circular_array.h
31 *
32 * @brief Helper class to safely access elements of a std::array
33 *
34 * Protects from out-of-bounds access by applying modulo of it's length
35 * when using the [] operator for element access.
36 *
37 * This is useful for circular data structures, like TTIs or SNs.
38 */
39
40namespace srsran {
41
42template <typename T, std::size_t N>
43class circular_array
44{
45 std::array<T, N> data;
46
47public:
48 using iterator = T*;
49 using const_iterator = const T*;
50
51 T& operator[](std::size_t pos) { return data[pos % N]; }
52 const T& operator[](std::size_t pos) const { return data[pos % N]; }
53
54 T* begin() { return data.begin(); }
55 const T* begin() const { return data.begin(); }
56
57 T* end() { return data.end(); }
58 const T* end() const { return data.end(); }
59
60 size_t size() const { return N; }
61};
62
63} // namespace srsran
