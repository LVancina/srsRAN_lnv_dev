35/// \brief This class represents and encodes a MAC DL-SCH PDU that may contain multiple subPDUs.
36/// Each subPDU is composed of a MAC subheader and MAC CE or MAC SDU payload.
37class dl_sch_pdu
38{
39public:
40 /// Encoder of a MAC SDU.
41 class mac_sdu_encoder
42 {
43 public:
44 mac_sdu_encoder() = default;
45 mac_sdu_encoder(dl_sch_pdu& pdu_, lcid_t lcid_, unsigned subhdr_len_, unsigned max_sdu_size_);
46
47 bool valid() const { return pdu != nullptr; }
48
49 /// \brief Returns the space that is reserved for this MAC SDU payload.
50 span<uint8_t> sdu_space() const;
51
52 /// \brief Updates the DL-SCH PDU with the encoded MAC SDU subheader and payload.
53 /// \return Number of bytes written to the DL-SCH PDU (MAC SDU subheader + payload).
54 unsigned encode_sdu(unsigned sdu_bytes_written);
55
56 private:
57 dl_sch_pdu* pdu = nullptr;
58 lcid_t lcid;
59 unsigned subhr_len;
60 unsigned max_sdu_size;
61 };
62
63 explicit dl_sch_pdu(span<uint8_t> pdu_buffer_) : pdu(pdu_buffer_) {}
64
65 /// \brief Gets a MAC SDU encoder for a given LCID and scheduler grant size.
66 mac_sdu_encoder get_sdu_encoder(lcid_t lcid, unsigned sdu_payload_len_estimate);
67
68 /// Adds a MAC SDU as a subPDU.
69 unsigned add_sdu(lcid_t lcid_, byte_buffer_chain&& sdu);
70
71 /// Adds a UE Contention Resolution CE as a subPDU.
72 void add_ue_con_res_id(const ue_con_res_id_t& con_res_payload);
73
74 /// Adds a Timing Advance Command CE as a subPDU.
75 void add_tag_cmd(const ta_cmd_ce_payload& ce_payload);
76
77 /// Adds a padding CE as a subPDU.
78 void add_padding(unsigned len);
79
80 /// Number of bytes of the MAC PDU.
81 unsigned nof_bytes() const { return byte_offset; }
82
83 /// Remaining space in number of bytes in the PDU.
84 unsigned nof_empty_bytes() const { return pdu.size() - byte_offset; }
85
86 /// Gets the held MAC PDU bytes.
87 span<uint8_t> get() { return pdu.first(byte_offset); }
88
89private:
90 void encode_subheader(bool F_bit, lcid_dl_sch_t lcid, unsigned header_len, unsigned payload_len);
91
92 span<uint8_t> pdu;
93 unsigned byte_offset = 0;
94};

96/// \brief Class that manages the encoding of DL-SCH MAC PDUs that will be stored in Transport Blocks.
97class dl_sch_pdu_assembler
98{
99public:
100 explicit dl_sch_pdu_assembler(mac_dl_ue_manager& ue_mng_);
101
102 /// \brief Encodes a MAC DL-SCH PDU with the provided scheduler information.
103 /// \param rnti RNTI for which the MAC PDU was allocated.
104 /// \param h_id HARQ-Id of the HARQ process used for this PDU transmission.
105 /// \param tb_idx Transport block index of the HARQ process used for this PDU transmission.
106 /// \param tb_info The information relative to the transport block allocated by the scheduler. This class contains
107 /// a list of LCIDs of the subPDUs to allocated together with how many bytes each subPDU should take.
108 /// \param tb_size_bytes Number of bytes allocated for the Transport Block.
109 /// \return Byte container with assembled PDU. This container length should be lower or equal to \c tb_size_bytes.
110 span<const uint8_t> assemble_newtx_pdu(rnti_t rnti,
111 harq_id_t h_id,
112 unsigned tb_idx,
113 const dl_msg_tb_info& tb_info,
114 unsigned tb_size_bytes);
115
116 /// \brief Fetches and assembles MAC DL-SCH PDU that corresponds to a HARQ retransmission.
117 /// \param rnti RNTI for which the MAC PDU was allocated.
118 /// \param h_id HARQ-Id of the HARQ process used for this PDU transmission.
119 /// \param tb_idx Transport block index of the HARQ process used for this PDU transmission.
120 /// \param tb_size_bytes Number of bytes allocated for the Transport Block.
121 /// \return Byte container with assembled PDU.
122 span<const uint8_t> assemble_retx_pdu(rnti_t rnti, harq_id_t harq_id, unsigned tb_idx, unsigned tbs_bytes);
123
124private:
125 class pdu_log_builder;
126
127 /// Assemble MAC SDUs for a given LCID.
128 void assemble_sdus(dl_sch_pdu& ue_pdu, rnti_t rnti, const dl_msg_lc_info& subpdu, pdu_log_builder& pdu_logger);
129
130 /// Assemble MAC subPDU with a CE.
131 void assemble_ce(dl_sch_pdu& ue_pdu, rnti_t rnti, const dl_msg_lc_info& subpdu, pdu_log_builder& pdu_logger);
132
133 mac_dl_ue_manager& ue_mng;
134
135 srslog::basic_logger& logger;
136 // memory buffer to avoid allocations during formatting of pdus
137 fmt::memory_buffer fmtbuf;
138};
