31/// \brief Determines whether a frequency is valid within a range.
32///
33/// A frequency is considered valid within a range if the range clips the frequency value within 1 Hz error.
34static bool radio_uhd_device_validate_freq_range(const uhd::freq_range_t& range, double freq)
35{
36 double clipped_freq = range.clip(freq);
37 return std::abs(clipped_freq - freq) < 1.0;
38}
39
40/// \brief Determines whether a gain is valid within a range.
41///
42/// A gain is considered valid within a range if the range clips the frequency value within 0.01 error.
43static bool radio_uhd_device_validate_gain_range(const uhd::gain_range_t& range, double gain)
44{
45 int64_t clipped_gain = static_cast<uint64_t>(std::round(range.clip(gain, true) * 100));
46 int64_t uint_gain = static_cast<uint64_t>(gain * 100);
47
48 return (clipped_gain == uint_gain);
49}

51static double to_MHz(double value_Hz)
52{
53 return value_Hz * 1e-6;
54}

65 bool usrp_make(const std::string& device_address)
66 {
67 // Parse args into dictionary.
68 uhd::device_addr_t device_addr;
69 if (!safe_execution([device_address, &device_addr]() { device_addr = uhd::device_addr_t(device_address); })) {
70 return false;
71 }
72
73 // Destroy any previous USRP instance
74 usrp = nullptr;
75
76 // If device type or name not given in args, select device from found list.
77 if (not device_addr.has_key("type")) {
78 // Find available devices that match the given device address.
79 uhd::device_addrs_t devices = uhd::device::find(device_addr);
80
81 // Stop if no device is found.
82 if (devices.empty()) {
83 printf("Error: no radio devices found.\n");
84 return false;
85 }
86
87 // Select the first available device.
88 uhd::device_addr_t first_device_addr = devices.front();
89 fmt::print("No device type given, found device with address '{}'\n", first_device_addr.to_string());
90
91 // Append to the device address the device type.
92 if (first_device_addr.has_key("type")) {
93 device_addr.set("type", first_device_addr.get("type"));
94 }
95 }
96
97 // If device type is known, parse and select default address parameters.
98 if (device_addr.has_key("type")) {
99 // Get the device type.
100 type = radio_uhd_device_type(device_addr.get("type"));
101
102 switch (type) {
103 case radio_uhd_device_type::types::E3xx:
104 // Set the default master clock rate.
105 if (!device_addr.has_key("master_clock_rate")) {
106 device_addr.set("master_clock_rate", "30.72e6");
107 }
108 break;
109 case radio_uhd_device_type::types::X3x0:
110 // Set the default master clock rate.
111 if (!device_addr.has_key("master_clock_rate")) {
112 device_addr.set("master_clock_rate", "184.32e6");
113 }
114 break;
115 case radio_uhd_device_type::types::X410:
116 // Set the default master clock rate.
117 if (!device_addr.has_key("master_clock_rate")) {
118 device_addr.set("master_clock_rate", "245.76e6");
119 }
120 break;
121 case radio_uhd_device_type::types::N3x0:
122 // Set the default master clock rate.
123 if (!device_addr.has_key("master_clock_rate")) {
124 device_addr.set("master_clock_rate", "122.88e6");
125 }
126 break;
127 case radio_uhd_device_type::types::N32x:
128 // Set the default master clock rate.
129 if (!device_addr.has_key("master_clock_rate")) {
130 device_addr.set("master_clock_rate", "245.76e6");
131 }
132 break;
133 case radio_uhd_device_type::types::B2xx:
134 case radio_uhd_device_type::types::UNKNOWN:
135 default:
136 // No default parameters are required.
137 break;
138 }
139
140 // Set frame sizes for network based USRP devices.
141 if ((type == radio_uhd_device_type::types::N32x) || (type == radio_uhd_device_type::types::N3x0) ||
142 (type == radio_uhd_device_type::types::X3x0) || (type == radio_uhd_device_type::types::X410)) {
143 // Set the default send frame size.
144 if (!device_addr.has_key("send_frame_size")) {
145 device_addr.set("send_frame_size", "8000");
146 }
147 // Set the default receive frame size.
148 if (!device_addr.has_key("recv_frame_size")) {
149 device_addr.set("recv_frame_size", "8000");
150 }
151 }
152 }
153
154 fmt::print("Making USRP object with args '{}'\n", device_addr.to_string());
155
156 return safe_execution([this, &device_addr]() { usrp = uhd::usrp::multi_usrp::make(device_addr); });
157 }
158 bool is_connection_valid()
159 {
160 // If the device is a B2xx, check if the USB is version 3.
161 if (type == radio_uhd_device_type::types::B2xx) {
162 // Get USB version.
163 unsigned usb_version = 0;
164 bool success = safe_execution([this, &usb_version]() {
165 usb_version = usrp->get_device()->get_tree()->access<unsigned>("/mboards/0/usb_version").get();
166 });
167
168 // The USB version is invalid if the USB version is accessed successfully and the version is not 3.
169 if (success && (usb_version != 3)) {
170 fmt::print("USRP operating over USB version {}, USB 3 is required.\n", usb_version);
171 return false;
172 }
173 }
174
175 return true;
176 }

221 bool set_sync_source(const radio_configuration::clock_sources& config)
222 {
223 // Convert clock source to string.
224 std::string clock_src;
225 switch (config.clock) {
226 case radio_configuration::clock_sources::source::DEFAULT:
227 case radio_configuration::clock_sources::source::INTERNAL:
228 clock_src = "internal";
229 break;
230 case radio_configuration::clock_sources::source::EXTERNAL:
231 clock_src = "external";
232 break;
233 case radio_configuration::clock_sources::source::GPSDO:
234 clock_src = "gpsdo";
235 break;
236 }
237
238 // Convert sync source to string.
239 std::string sync_src;
240 switch (config.sync) {
241 case radio_configuration::clock_sources::source::DEFAULT:
242 case radio_configuration::clock_sources::source::INTERNAL:
243 sync_src = "internal";
244 break;
245 case radio_configuration::clock_sources::source::EXTERNAL:
246 sync_src = "external";
247 break;
248 case radio_configuration::clock_sources::source::GPSDO:
249 sync_src = "gpsdo";
250 break;
251 }
252
253 logger.debug("Setting PPS source to '{}' and clock source to '{}'.", sync_src, clock_src);
254#if UHD_VERSION < 3140099
255 return safe_execution([this, &sync_src, &clock_src]() {
256 std::vector<std::string> time_sources = usrp->get_time_sources(0);
257 if (std::find(time_sources.begin(), time_sources.end(), sync_src) == time_sources.end()) {
258 on_error("Invalid time source {}. Supported: {}", sync_src, span<const std::string>(time_sources));
259 return;
260 }
261 std::vector<std::string> clock_sources = usrp->get_clock_sources(0);
262 if (std::find(clock_sources.begin(), clock_sources.end(), clock_src) == clock_sources.end()) {
263 on_error("Invalid clock source {}. Supported: {}", clock_src, span<const std::string>(clock_sources));
264 return;
265 }
266
267 usrp->set_time_source(sync_src);
268 usrp->set_clock_source(clock_src);
269 });
270#else
271 return safe_execution([this, &sync_source, &clock_source]() { usrp->set_sync_source(clock_source, sync_source); });
272#endif
273 }

375 bool set_tx_freq(uint32_t ch, const radio_configuration::lo_frequency& config)
376 {
377 logger.debug("Setting channel {} Tx frequency to {} MHz.", ch, to_MHz(config.center_frequency_hz));
378
379 return safe_execution([this, ch, &config]() {
380 uhd::freq_range_t range = usrp->get_tx_freq_range(ch);
381
382 if (!radio_uhd_device_validate_freq_range(range, config.center_frequency_hz)) {
383 on_error("Tx RF frequency {} MHz is out-of-range. Range is {}.",
384 to_MHz(config.center_frequency_hz),
385 range.to_pp_string());
386 return;
387 }
388
389 // Create tune request.
390 uhd::tune_request_t tune_request(config.center_frequency_hz);
391
392 // If the LO frequency is defined, force a LO frequency and set the DSP frequency to auto.
393 if (std::isnormal(config.lo_frequency_hz)) {
394 tune_request.rf_freq = config.lo_frequency_hz;
395 tune_request.rf_freq_policy = uhd::tune_request_t::POLICY_MANUAL;
396 tune_request.dsp_freq_policy = uhd::tune_request_t::POLICY_AUTO;
397 }
398
399 usrp->set_tx_freq(tune_request, ch);
400 });
401 }
