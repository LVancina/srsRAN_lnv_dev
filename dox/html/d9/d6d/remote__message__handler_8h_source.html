30/// This implementation forwards all F1AP messages through a network gateway. It is useful for distributed scenarios
31/// where the F1 interfaces reside in different machines.
32class remote_f1ap_pdu_handler : public f1ap_message_handler
33{
34 network_gateway_data_handler& network;
35
36public:
37 explicit remote_f1ap_pdu_handler(network_gateway_data_handler& network) : network(network) {}
38
39 void handle_message(const asn1::f1ap::f1ap_pdu_c& msg) override
40 {
41 // Pack PDU into a temporary buffer.
42 byte_buffer tx_pdu;
43 asn1::bit_ref bref(tx_pdu);
44 if (msg.pack(bref) != asn1::SRSASN_SUCCESS) {
45 return;
46 }
47
48 network.handle_pdu(tx_pdu);
49 }
50};
