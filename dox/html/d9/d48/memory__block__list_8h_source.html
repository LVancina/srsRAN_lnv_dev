34/// \brief This class represents an intrusive list of memory blocks.
35class intrusive_memory_block_list
36{
37public:
38 struct node {
39 node* next;
40 explicit node(node* prev_) : next(prev_) {}
41 };
42
43 node* head = nullptr;
44 node* tail = nullptr;
45 std::size_t count = 0;
46
47 constexpr static std::size_t min_memory_block_size() { return sizeof(node); }
48 constexpr static std::size_t min_memory_block_align() { return alignof(node); }
49
50 /// Pushes a new memory block to the linked list.
51 void push(void* block) noexcept
52 {
53 srsran_assert(is_aligned(block, min_memory_block_align()), "The provided memory block is not aligned");
54 node* ptr = ::new (block) node(head);
55 head = ptr;
56 if (tail == nullptr) {
57 tail = ptr;
58 }
59 count++;
60 }
61
62 /// Pops a memory block from the linked list.
63 void* pop() noexcept
64 {
65 srsran_assert(not empty(), "pop() called on empty list");
66 node* last_head = head;
67 head = head->next;
68 if (head == nullptr) {
69 tail = nullptr;
70 }
71 last_head->~node();
72 count--;
73 return static_cast<void*>(last_head);
74 }
75
76 void* try_pop() noexcept { return empty() ? nullptr : pop(); }
77
78 /// Steals the top pointer from another memory block list.
79 void steal_top(intrusive_memory_block_list& other) noexcept
80 {
81 srsran_assert(not other.empty(), "Trying to steal from empty memory_block list");
82 node* other_head = other.head;
83 other.head = other.head->next;
84 if (other.head == nullptr) {
85 other.tail = nullptr;
86 }
87 other_head->next = head;
88 head = other_head;
89 if (tail == nullptr) {
90 head = tail;
91 }
92 other.count--;
93 count++;
94 }
95
96 intrusive_memory_block_list try_pop_batch(unsigned n)
97 {
98 intrusive_memory_block_list new_list{};
99 if (n >= size()) {
100 new_list.head = std::exchange(head, nullptr);
101 new_list.tail = std::exchange(tail, nullptr);
102 new_list.count = std::exchange(count, 0);
103 } else if (n > 0) {
104 node* prev = head;
105 for (unsigned i = 0; i != n - 1; ++i) {
106 prev = prev->next;
107 }
108 new_list.head = head;
109 new_list.tail = prev;
110 new_list.count = n;
111 head = std::exchange(prev->next, nullptr);
112 count -= n;
113 }
114 return new_list;
115 }
116
117 void steal_blocks(intrusive_memory_block_list& list)
118 {
119 if (list.empty()) {
120 return;
121 }
122 if (empty()) {
123 head = std::exchange(list.head, nullptr);
124 } else {
125 tail->next = std::exchange(list.head, nullptr);
126 }
127 tail = std::exchange(list.tail, nullptr);
128 count += std::exchange(list.count, 0);
129 }
130
131 bool empty() const noexcept { return head == nullptr; }
132
133 std::size_t size() const { return count; }
134
135 void clear() noexcept
136 {
137 head = nullptr;
138 tail = nullptr;
139 count = 0;
140 }
141};
142
143} // namespace detail

145/// \brief List of memory blocks. It overwrites bytes of blocks passed via push(void*). Thus, it is not safe to use
146/// for any pool of initialized objects.
147using free_memory_block_list = detail::intrusive_memory_block_list;
148
149/// memory_block stack that mutexes pushing/popping
150class concurrent_free_memory_block_list
151{
152public:
153 concurrent_free_memory_block_list() = default;
154 concurrent_free_memory_block_list(const concurrent_free_memory_block_list&) = delete;
155 concurrent_free_memory_block_list(concurrent_free_memory_block_list&& other) noexcept
156 {
157 std::unique_lock<std::mutex> lk1(other.mutex, std::defer_lock);
158 std::unique_lock<std::mutex> lk2(mutex, std::defer_lock);
159 std::lock(lk1, lk2);
160 stack = other.stack;
161 }
162 concurrent_free_memory_block_list& operator=(const concurrent_free_memory_block_list&) = delete;
163 concurrent_free_memory_block_list& operator=(concurrent_free_memory_block_list&& other) noexcept
164 {
165 std::unique_lock<std::mutex> lk1(other.mutex, std::defer_lock);
166 std::unique_lock<std::mutex> lk2(mutex, std::defer_lock);
167 std::lock(lk1, lk2);
168 stack = other.stack;
169 return *this;
170 }
171
172 void push(void* block) noexcept
173 {
174 std::lock_guard<std::mutex> lock(mutex);
175 stack.push(block);
176 }
177
178 void steal_blocks(free_memory_block_list& other) noexcept
179 {
180 std::lock_guard<std::mutex> lock(mutex);
181 stack.steal_blocks(other);
182 }
183
184 void steal_blocks(free_memory_block_list& other, size_t max_n) noexcept
185 {
186 if (max_n >= other.size()) {
187 steal_blocks(other);
188 return;
189 }
190 std::lock_guard<std::mutex> lock(mutex);
191 for (size_t i = 0; i < max_n; ++i) {
192 stack.push(other.try_pop());
193 }
194 }
195
196 void* try_pop() noexcept
197 {
198 std::lock_guard<std::mutex> lock(mutex);
199 void* block = stack.try_pop();
200 return block;
201 }
202
203 free_memory_block_list try_pop_list(unsigned n) noexcept
204 {
205 std::lock_guard<std::mutex> lock(mutex);
206 return stack.try_pop_batch(n);
207 }
208
209 template <size_t N>
210 size_t try_pop(std::array<void*, N>& result) noexcept
211 {
212 std::lock_guard<std::mutex> lock(mutex);
213 size_t i = 0;
214 for (; i < N; ++i) {
215 result[i] = stack.try_pop();
216 if (result[i] == nullptr) {
217 break;
218 }
219 }
220 return i;
221 }
222
223 bool empty() const noexcept { return stack.empty(); }
224
225 size_t size() const noexcept
226 {
227 std::lock_guard<std::mutex> lock(mutex);
228 return stack.size();
229 }
230
231 void clear()
232 {
233 std::lock_guard<std::mutex> lock(mutex);
234 stack.clear();
235 }
236
237private:
238 free_memory_block_list stack;
239 mutable std::mutex mutex;
240};
