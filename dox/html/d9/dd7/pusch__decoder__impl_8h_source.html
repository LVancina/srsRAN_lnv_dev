42/// Implementation of the PUSCH decoder.
43class pusch_decoder_impl : public pusch_decoder, private pusch_decoder_buffer
44{
45public:
46 /// Code block decoder pool type.
47 using codeblock_decoder_pool = concurrent_thread_local_object_pool<pusch_codeblock_decoder>;
48
49 /// CRC calculators used in shared channels.
50 struct sch_crc {
51 /// For short TB checksums.
52 std::unique_ptr<crc_calculator> crc16;
53 /// For long TB checksums.
54 std::unique_ptr<crc_calculator> crc24A;
55 /// For segment-specific checksums.
56 std::unique_ptr<crc_calculator> crc24B;
57 };
58
59 /// \brief PUSCH decoder constructor.
60 ///
61 /// Sets up the internal components, namely LDPC segmenter, LDPC rate dematcher, LDPC decoder and all the CRC
62 /// calculators.
63 ///
64 /// \param[in] segmenter_ LDPC segmenter.
65 /// \param[in] decoder_pool_ Codeblock decoder.
66 /// \param[in] crc_set_ Structure with pointers to three CRC calculator objects, with generator
67 /// polynomials of type \c CRC16, \c CRC24A and \c CRC24B.
68 /// \param[in] executor_ Task executor for asynchronous PUSCH code block decoding.
69 /// \param[in] nof_prb Number of PRBs.
70 /// \param[in] nof_layers Number of layers.
71
72 pusch_decoder_impl(std::unique_ptr<ldpc_segmenter_rx> segmenter_,
73 std::shared_ptr<codeblock_decoder_pool> decoder_pool_,
74 sch_crc crc_set_,
75 task_executor* executor_,
76 unsigned nof_prb,
77 unsigned nof_layers) :
78
79 segmenter(std::move(segmenter_)),
80 decoder_pool(std::move(decoder_pool_)),
81 crc_set(std::move(crc_set_)),
82 executor(executor_),
83 softbits_buffer(pusch_constants::get_max_codeword_size(nof_prb, nof_layers).value()),
84 cb_stats(MAX_NOF_SEGMENTS)
85 {
86 srsran_assert(segmenter, "Invalid segmenter.");
87 srsran_assert(decoder_pool, "Invalid codeblock decoder pool.");
88 srsran_assert(crc_set.crc16, "Invalid CRC16 calculator.");
89 srsran_assert(crc_set.crc24A, "Invalid CRC24A calculator.");
90 srsran_assert(crc_set.crc24B, "Invalid CRC24B calculator.");
91 srsran_assert(crc_set.crc16->get_generator_poly() == crc_generator_poly::CRC16, "Wrong TB CRC calculator.");
92 srsran_assert(crc_set.crc24A->get_generator_poly() == crc_generator_poly::CRC24A, "Wrong TB CRC calculator.");
93 srsran_assert(crc_set.crc24B->get_generator_poly() == crc_generator_poly::CRC24B, "Wrong TB CRC calculator.");
94 if (executor != nullptr) {
95 srsran_assert(decoder_pool->capacity() > 1,
96 "The number of PUSCH code block decoder in the pool (i.e., {}) must be greater than one.",
97 decoder_pool->capacity());
98 }
99 };

136 segmenter_config segmentation_config;

139 /// Counts the number of remaining CB decoding tasks.
140 std::atomic<unsigned> cb_task_counter;
