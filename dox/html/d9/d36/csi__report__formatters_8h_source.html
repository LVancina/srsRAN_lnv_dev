30/// Custom formatter for \c srsran::csi_report_configuration.
31template <>
32struct fmt::formatter<srsran::csi_report_configuration> {
33 /// Helper used to parse formatting options and format fields.
34 srsran::delimited_formatter helper;
35
36 /// Default constructor.
37 formatter() = default;
38
39 template <typename ParseContext>
40 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
41 {
42 return helper.parse(ctx);
43 }
44
45 template <typename FormatContext>
46 auto format(const srsran::csi_report_configuration& config, FormatContext& ctx)
47 -> decltype(std::declval<FormatContext>().out())
48 {
49 helper.format_always(ctx, "nof_csi_rs_resources={}", config.nof_csi_rs_resources);
50 helper.format_always(ctx, "pmi_codebook={}", to_string(config.pmi_codebook));
51 helper.format_always(ctx, "ri_restriction={}", config.ri_restriction);
52
53 helper.format_always(ctx, "quantities={}", to_string(config.quantities));
54
55 return ctx.out();
56 }
57};

118 helper.format_always(ctx, "ri={}", data.ri.value());

120
121 if (data.li.has_value()) {
122 helper.format_always(ctx, "li={}", data.li.value());
123 }
124
125 if (data.pmi.has_value()) {
126 helper.format_always(ctx, "{}", data.pmi.value());
127 }
128
129 if (data.first_tb_wideband_cqi.has_value()) {
130 helper.format_always(ctx, "cqi1={}", data.first_tb_wideband_cqi.value());
131 }
132
133 if (data.second_tb_wideband_cqi.has_value()) {
134 helper.format_always(ctx, "cqi2={}", data.second_tb_wideband_cqi.value());
135 }
136
137 return ctx.out();
138 }
