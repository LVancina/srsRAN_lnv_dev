31/// Maximum number of UEs supported by DU (implementation-defined).
32enum du_ue_index_t : uint16_t {
33 MIN_DU_UE_INDEX = 0,
34 MAX_DU_UE_INDEX = 1023,
35 MAX_NOF_DU_UES = 1024,
36 INVALID_DU_UE_INDEX = 1024
37};
38
39/// Maximum number of cells supported by DU (implementation-defined).
40enum du_cell_index_t : uint16_t {
41 MIN_DU_CELL_INDEX = 0,
42 MAX_DU_CELL_INDEX = 15,
43 MAX_NOF_DU_CELLS = 16,
44 INVALID_DU_CELL_INDEX = MAX_NOF_DU_CELLS
45};
46
47/// Convert integer to DU UE index type.
48constexpr inline du_ue_index_t to_du_ue_index(std::underlying_type_t<du_ue_index_t> idx)
49{
50 return static_cast<du_ue_index_t>(idx);
51}
52
53constexpr inline bool is_du_ue_index_valid(du_ue_index_t ue_idx)
54{
55 return ue_idx < MAX_NOF_DU_UES;
56}
57
58/// Convert integer to DU cell index type.
59inline du_cell_index_t to_du_cell_index(std::underlying_type_t<du_cell_index_t> idx)
60{
61 return static_cast<du_cell_index_t>(idx);
62}
63
64/// Maximum number of cells supported by a single UE.
65enum ue_cell_index_t : uint16_t {
66 PCELL_INDEX = 0,
67 MAX_UE_SCELL_INDEX = 15,
68 MAX_NOF_SCELL_INDEXES = 16,
69 INVALID_UE_CELL_INDEX = MAX_NOF_SCELL_INDEXES
70};
71
72inline ue_cell_index_t to_ue_cell_index(std::underlying_type_t<ue_cell_index_t> idx)
73{
74 return static_cast<ue_cell_index_t>(idx);
75}

77/// \brief DU-specific index to group of cells that might be aggregated into a UE-specific CellGroup, if the UE

81 INVALID_DU_CELL_GROUP_INDEX = MAX_DU_CELL_GROUPS
82};
