30/// Executes a task n times, where "n" increases with the failures to dispatch the task.
31/// Dispatches a task to an executor, and if it fails to dispatch (e.g. due to the task queue being full), it reattempts
32/// on the next call.
33template <typename Executor>
34class task_redispatcher
35{
36public:
37 template <typename Exec, typename CallToExec>
38 task_redispatcher(Exec&& executor_, CallToExec&& task) :
39 executor(std::forward<Exec>(executor_)), dispatch_task(get_dispatch_task(std::forward<CallToExec>(task)))
40 {
41 }
42
43 SRSRAN_NODISCARD bool execute()
44 {
45 if (not dispatch_task(true)) {
46 dispatch_fail_count.fetch_add(1U, std::memory_order_relaxed);
47 return false;
48 }
49 return true;
50 }
51
52 SRSRAN_NODISCARD bool defer()
53 {
54 if (not dispatch_task(false)) {
55 dispatch_fail_count.fetch_add(1U, std::memory_order_relaxed);
56 return false;
57 }
58 return true;
59 }
60
61private:
62 template <typename CallToExec>
63 void run_task_n_times(CallToExec& task)
64 {
65 // we invoke the "task" a number of times equal to "1 + failed dispatches"
66 task();
67 uint32_t remaining_calls = dispatch_fail_count.load(std::memory_order_relaxed);
68 while (remaining_calls > 0) {
69 for (unsigned i = 0; i != remaining_calls; ++i) {
70 task();
71 }
72 remaining_calls = dispatch_fail_count.fetch_sub(remaining_calls, std::memory_order_acq_rel) - remaining_calls;
73 }
74 }
75
76 template <typename CallToExec>
77 unique_function<bool(bool)> get_dispatch_task(CallToExec&& task)
78 {
79 return [this, t = std::forward<CallToExec>(task)](bool is_execute) {
80 if (is_execute) {
81 return executor.execute([this, &t]() { run_task_n_times(t); });
82 }
83 return executor.defer([this, &t]() { run_task_n_times(t); });
84 };
85 }
86
87 Executor executor;
88 unique_function<bool(bool)> dispatch_task;
89
90 std::atomic<uint32_t> dispatch_fail_count{0};
91};
