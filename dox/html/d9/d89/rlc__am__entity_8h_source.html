57 ue_index_,
58 rb_id_,
59 config.tx,
60 tx_upper_dn,
61 tx_upper_cn,
62 tx_lower_dn,
63 timer_factory{timers, pcell_executor},
64 pcell_executor,
65 ue_executor,
66 metrics_period.count() != 0,

80 rx_am->set_status_handler(tx_am.get());
81 rx_am->set_status_notifier(tx_am.get());
82
83 // Store entities
84 tx = std::move(tx_am);
85 rx = std::move(rx_am);
