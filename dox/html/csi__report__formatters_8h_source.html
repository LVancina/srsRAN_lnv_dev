80 helper.format_always(ctx, "pmi={}", value.pmi);

82

85 srsran::variant_get<srsran::csi_report_pmi::typeI_single_panel_4ports_mode1>(pmi.type);
86 helper.format_always(ctx, "i_1_1={}", value.i_1_1);
87 helper.format_always(ctx, "i_1_3={}", value.i_1_3);
88 helper.format_always(ctx, "i_2={}", value.i_2);
89 }
90
91 return ctx.out();

96template <>
97struct fmt::formatter<srsran::csi_report_data> {
99 srsran::delimited_formatter helper;
100
102 formatter() = default;
103
104 template <typename ParseContext>
105 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
106 {
107 return helper.parse(ctx);
108 }
109
110 template <typename FormatContext>
111 auto format(const srsran::csi_report_data& data, FormatContext& ctx) -> decltype(std::declval<FormatContext>().out())
112 {
113 if (data.cri.has_value()) {
114 helper.format_always(ctx, "cri={}", data.cri.value());
115 }
116
117 if (data.ri.has_value()) {
118 helper.format_always(ctx, "ri={}", data.ri.value());
119 }
120
121 if (data.li.has_value()) {
122 helper.format_always(ctx, "li={}", data.li.value());
123 }
124
125 if (data.pmi.has_value()) {
126 helper.format_always(ctx, "{}", data.pmi.value());
127 }
128
129 if (data.first_tb_wideband_cqi.has_value()) {
130 helper.format_always(ctx, "cqi1={}", data.first_tb_wideband_cqi.value());
131 }
132
133 if (data.second_tb_wideband_cqi.has_value()) {
134 helper.format_always(ctx, "cqi2={}", data.second_tb_wideband_cqi.value());
135 }
136
137 return ctx.out();
138 }
139};
