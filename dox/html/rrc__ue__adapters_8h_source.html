168class rrc_ue_cu_cp_adapter : public rrc_ue_context_update_notifier, public rrc_ue_measurement_notifier
169{
170public:
171 void connect_cu_cp(cu_cp_rrc_ue_interface& cu_cp_rrc_ue_,
172 cu_cp_ue_removal_handler& ue_removal_handler_,
173 cu_cp_controller& ctrl_,
174 cu_cp_measurement_handler& meas_handler_)
175 {
176 cu_cp_rrc_ue_handler = &cu_cp_rrc_ue_;
177 ue_removal_handler = &ue_removal_handler_;
178 controller = &ctrl_;
179 meas_handler = &meas_handler_;
180 }
181
182 bool on_ue_setup_request() override
183 {
184 srsran_assert(controller != nullptr, "CU-CP controller must not be nullptr");
185 return controller->request_ue_setup();
186 }
187
188 rrc_reestablishment_ue_context_t
189 on_rrc_reestablishment_request(pci_t old_pci, rnti_t old_c_rnti, ue_index_t ue_index) override
190 {
191 srsran_assert(cu_cp_rrc_ue_handler != nullptr, "CU-CP handler must not be nullptr");
192 return cu_cp_rrc_ue_handler->handle_rrc_reestablishment_request(old_pci, old_c_rnti, ue_index);
193 }
194
195 async_task<bool> on_ue_transfer_required(ue_index_t ue_index, ue_index_t old_ue_index) override
196 {
197 srsran_assert(cu_cp_rrc_ue_handler != nullptr, "CU-CP handler must not be nullptr");
198 return cu_cp_rrc_ue_handler->handle_ue_context_transfer(ue_index, old_ue_index);
199 }
200
201 void on_ue_removal_required(ue_index_t ue_index) override
202 {
203 srsran_assert(ue_removal_handler != nullptr, "CU-CP UE removal handler must not be nullptr");
204 return ue_removal_handler->handle_ue_removal_request(ue_index);
205 }
206
207 optional<rrc_meas_cfg> on_measurement_config_request(nr_cell_id_t nci,
208 optional<rrc_meas_cfg> current_meas_config = {}) override
209 {
210 srsran_assert(meas_handler != nullptr, "Measurement handler must not be nullptr");
211 return meas_handler->handle_measurement_config_request(nci, current_meas_config);
212 }
213
214 void on_measurement_report(const ue_index_t ue_index, const rrc_meas_results& meas_results) override
215 {
216 srsran_assert(meas_handler != nullptr, "Measurement handler must not be nullptr");
217 meas_handler->handle_measurement_report(ue_index, meas_results);
218 }
219
220private:
221 cu_cp_rrc_ue_interface* cu_cp_rrc_ue_handler = nullptr;
222 cu_cp_ue_removal_handler* ue_removal_handler = nullptr;
223 cu_cp_controller* controller = nullptr;
224 cu_cp_measurement_handler* meas_handler = nullptr;
225};
