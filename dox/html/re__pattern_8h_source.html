35struct re_pattern {
37 bounded_bitset<MAX_RB> prb_mask;
39 re_prb_mask re_mask{};
41 symbol_slot_mask symbols{};
42
44 re_pattern() = default;
45
55 re_pattern(unsigned rb_begin,
56 unsigned rb_end,
57 unsigned rb_stride,
58 const re_prb_mask& re_mask_,
59 const symbol_slot_mask& symbols_) :
60 prb_mask(rb_end), re_mask(re_mask_), symbols(symbols_)
61 {
62 static constexpr interval<unsigned, true> rb_begin_interval = {0, MAX_RB - 1};
63 static constexpr interval<unsigned, true> rb_stride_interval = {1, MAX_RB - 1};
64 static constexpr interval<unsigned, true> rb_end_interval = {1, MAX_RB};
65
66 srsran_assert(rb_begin_interval.contains(rb_begin),
67 "The RB begin (i.e., {}) is out of range {}.",
68 rb_begin,
69 rb_begin_interval);
70 srsran_assert(
71 rb_end_interval.contains(rb_end), "The RB end (i.e., {}) is out of range {}.", rb_end, rb_end_interval);
72 srsran_assert(rb_stride_interval.contains(rb_stride),
73 "The RB stride (i.e., {}) is out of range {}.",
74 rb_stride,
75 rb_stride_interval);
76 srsran_assert(
77 rb_begin < rb_end, "The RB begin (i.e., {}) must be smaller than RB end (i.e., {})", rb_begin, rb_end);
78
79 if (rb_stride == 1) {
80 prb_mask.fill(rb_begin, rb_end);
81 } else {
82 for (unsigned i_prb = rb_begin; i_prb < rb_end; i_prb += rb_stride) {
83 prb_mask.set(i_prb);
84 }
85 }
86 }
87
91 re_pattern(const re_pattern& other) : prb_mask(other.prb_mask), re_mask(other.re_mask), symbols(other.symbols)
92 {
93 // Do nothing.
94 }
95

114 void get_exclusion_mask(bounded_bitset<MAX_RB * NRE>& mask, unsigned symbol) const;

120inline bool operator==(const re_pattern& pattern1, const re_pattern& pattern2)
121{
122 if (pattern1.symbols != pattern2.symbols) {
123 return false;
124 }
125
126 if (pattern1.prb_mask != pattern2.prb_mask) {
127 return false;
128 }
129
130 if (pattern1.re_mask != pattern2.re_mask) {
131 return false;
132 }
133
134 return true;
135}
