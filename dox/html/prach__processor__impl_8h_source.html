37class prach_processor_impl : public prach_processor,
38 private prach_processor_baseband,
39 private prach_processor_request_handler
40{
41private:
42 std::vector<std::unique_ptr<prach_processor_worker>> workers;
43 prach_processor_notifier* notifier = nullptr;
44
45 // See prach_processor_request_handler for documentation.
46 void handle_request(prach_buffer& buffer, const prach_buffer_context& context) override
47 {
48 srsran_assert(notifier, "Notifier has not been connected.");
49
50 // Iterate all workers...
51 for (std::unique_ptr<prach_processor_worker>& worker : workers) {
52 // Select the first worker available.
53 if (worker->is_available()) {
54 // Set request.
55 worker->handle_request(buffer, context);
56
57 // Stop iterating.
58 return;
59 }
60 }
61
62 // If no available workers, notify error and the buffer is discarded.
63 notifier->on_prach_request_overflow(context);
64 }
65
66 // See prach_processor_baseband for documentation.
67 void process_symbol(const baseband_gateway_buffer_reader& samples, const symbol_context& context) override
68 {
69 // Iterate all workers...
70 for (std::unique_ptr<prach_processor_worker>& worker : workers) {
71 // Process baseband.
72 worker->process_symbol(samples, context);
73 }
74 }
75
76public:
78 explicit prach_processor_impl(std::vector<std::unique_ptr<prach_processor_worker>>&& workers_) :
79 workers(std::move(workers_))
80 {
81 srsran_assert(!workers.empty(), "No workers are available.");
82 for (const std::unique_ptr<prach_processor_worker>& worker : workers) {
83 srsran_assert(worker, "Invalid worker.");
84 }
85 }
86
87 // See prach_processor for documentation.
88 void connect(prach_processor_notifier& notifier_) override
89 {
90 for (std::unique_ptr<prach_processor_worker>& worker : workers) {
91 worker->connect(notifier_);
92 }
93 notifier = &notifier_;
94 }
95
96 // See prach_processor for documentation.
97 prach_processor_request_handler& get_request_handler() override { return *this; }
98
99 // See prach_processor for documentation.
100 prach_processor_baseband& get_baseband() override { return *this; }
101};
