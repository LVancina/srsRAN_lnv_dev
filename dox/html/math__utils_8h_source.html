186template <class T, class CompareLess>
187constexpr const T& clamp(const T& value, const T& lower_bound, const T& upper_bound, CompareLess comp)
188{
189 return comp(value, lower_bound) ? lower_bound : (comp(upper_bound, value) ? upper_bound : value);
190}
191
195template <class T>
196constexpr const T& clamp(const T& value, const T& lower_bound, const T& upper_bound)
197{
198 return clamp(value, lower_bound, upper_bound, std::less<T>());
199}
