31class uci_decoder_spy : public uci_decoder
32{
33public:
34 struct entry_t {
35 std::vector<log_likelihood_ratio> llr;
36 configuration config;
37 std::vector<uint8_t> message;
38 uci_status status;
39 };
40
41 uci_decoder_spy() : rgen(0), bin_dist(0, 1)
42 {
43 // Do nothing.
44 }
45
46 uci_status decode(span<uint8_t> message, span<const log_likelihood_ratio> llr, const configuration& config) override
47 {
48 entries.emplace_back();
49 entry_t& entry = entries.back();
50 entry.config = config;
51
52 // Copy input soft bits.
53 entry.llr.resize(llr.size());
54 srsvec::copy(entry.llr, llr);
55
56 // Generate random decoded message.
57 std::generate(message.begin(), message.end(), [this]() { return rgen() & 1; });
58 entry.message.resize(message.size());
59 srsvec::copy(entry.message, message);
60
61 // Generate random status.
62 entry.status = (bin_dist(rgen) == 1) ? uci_status::valid : uci_status::invalid;
63
64 return entry.status;
65 }
66
67 const std::vector<entry_t>& get_entries() const { return entries; }
68
69 void clear() { entries.clear(); }
70
71private:
72 std::vector<entry_t> entries;
73 std::mt19937 rgen;
74 std::uniform_int_distribution<unsigned> bin_dist;
75};
