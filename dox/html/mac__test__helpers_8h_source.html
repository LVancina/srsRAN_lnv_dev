42inline mac_cell_creation_request make_default_mac_cell_config(const cell_config_builder_params& params = {})
43{
44 mac_cell_creation_request req{};
45
46 req.pci = params.pci;
47 req.cell_index = to_du_cell_index(0);
48 req.scs_common = params.scs_common;
49 req.dl_carrier = config_helpers::make_default_dl_carrier_configuration(params);
50 req.ul_carrier = config_helpers::make_default_ul_carrier_configuration(params);
51 req.ssb_cfg = config_helpers::make_default_ssb_config(params);
52 req.cell_barred = false;
53 req.intra_freq_resel = false;
54 req.sched_req = {}; // Note: MAC should not touch this field.
55
56 byte_buffer dummy_sib1;
57 for (unsigned i = 0; i != 100; ++i) {
58 report_fatal_error_if_not(dummy_sib1.append(i), "Failed to append to create dummy SIB1");
59 }
60 req.bcch_dl_sch_payloads.push_back(std::move(dummy_sib1));
61 return req;
62}

75inline mac_ue_create_request make_default_ue_creation_request(const cell_config_builder_params& params = {})
76{
77 mac_ue_create_request msg{};
78
79 msg.ue_index = to_du_ue_index(0);
80 msg.crnti = to_rnti(0x4601);
81 msg.cell_index = to_du_cell_index(0);
82
83 msg.rlf_notifier = nullptr;
84
85 msg.mac_cell_group_cfg = config_helpers::make_initial_mac_cell_group_config();
86
87 physical_cell_group_config& pcg_cfg = msg.phy_cell_group_cfg;
88 pcg_cfg.pdsch_harq_codebook = pdsch_harq_ack_codebook::dynamic;
89
90 msg.sched_cfg.cells.emplace();
91 msg.sched_cfg.cells->push_back(config_helpers::create_default_initial_ue_spcell_cell_config(params));
92
93 return msg;
94}

158class mac_sdu_tx_builder_test_notifier : public mac_sdu_tx_builder
159{
160public:
161 unsigned next_bs = 0;
162 byte_buffer previous_tx_sdu;
163
164 size_t on_new_tx_sdu(span<uint8_t> mac_sdu_buf) override
165 {
166 previous_tx_sdu = test_rgen::random_vector<uint8_t>(mac_sdu_buf.size());
167 auto out_it = mac_sdu_buf.begin();
168 for (span<const uint8_t> seg : previous_tx_sdu.segments()) {
169 out_it = std::copy(seg.begin(), seg.end(), out_it);
170 }
171 return mac_sdu_buf.size();
172 }
173
174 unsigned on_buffer_state_update() override { return next_bs; }
