177class rlc_tx_mac_buffer_state_updater : public rlc_tx_lower_layer_notifier
178{
179public:
180 void connect(du_ue_index_t ue_index_, lcid_t lcid_, mac_ue_control_information_handler& mac_)
181 {
182 srsran_assert(ue_index_ != INVALID_DU_UE_INDEX, "Invalid UE index");
183 srsran_assert(lcid_ != INVALID_LCID, "Invalid UE index");
184 ue_index = ue_index_;
185 lcid = lcid_;
186 mac = &mac_;
187 }
188
189 void disconnect() { lcid = INVALID_LCID; }
190
191 void on_buffer_state_update(unsigned bsr) override
192 {
193 srsran_assert(mac != nullptr, "RLC Tx Buffer State notifier is disconnected");
194 mac_dl_buffer_state_indication_message bs{};
195 bs.ue_index = ue_index;
196 bs.lcid = lcid.load(std::memory_order_relaxed);
197 bs.bs = bsr;
198 if (SRSRAN_UNLIKELY(bs.lcid == INVALID_LCID)) {
199 // Discard.
200 return;
201 }
202 mac->handle_dl_buffer_state_update(bs);
203 }
204
205private:
206 du_ue_index_t ue_index = INVALID_DU_UE_INDEX;
207 std::atomic<lcid_t> lcid{INVALID_LCID};
208 mac_ue_control_information_handler* mac = nullptr;
209};
