53struct search_space_info {
54 const search_space_configuration* cfg = nullptr;
55 const coreset_configuration* coreset = nullptr;
56 const bwp_info* bwp = nullptr;
57 crb_interval dl_crb_lims;
58 crb_interval ul_crb_lims;
59 span<const pdsch_time_domain_resource_allocation> pdsch_time_domain_list;
60 span<const pusch_time_domain_resource_allocation> pusch_time_domain_list;
61 dci_size_config dci_sz_cfg;
62 dci_sizes dci_sz;
63
66 dci_dl_format get_dl_dci_format() const { return pdcch_helper::get_dl_dci_format(*cfg); }
67
70 dci_ul_format get_ul_dci_format() const { return pdcch_helper::get_ul_dci_format(*cfg); }
71
73 span<const uint8_t> get_k1_candidates() const;
74
76 span<const uint8_t> get_pdcch_candidates(aggregation_level aggr_lvl, slot_point pdcch_slot) const
77 {
78 return ss_pdcch_candidates[pdcch_slot.to_uint() % ss_pdcch_candidates.size()][to_aggregation_level_index(aggr_lvl)];
79 }
80
82 span<const crb_index_list> get_crb_list_of_pdcch_candidates(aggregation_level aggr_lvl, slot_point pdcch_slot) const
83 {
84 return crbs_of_candidates[pdcch_slot.to_uint() % crbs_of_candidates.size()][to_aggregation_level_index(aggr_lvl)];
85 }
86
88 void update_pdcch_candidates(const std::vector<std::array<pdcch_candidate_list, NOF_AGGREGATION_LEVELS>>& candidates,
89 pci_t pci);
90
91private:
92 // PDCCH candidates of the SearchSpace for different slot offsets and aggregation levels. Indexed by
93 // ss_pddch_candidates[slot_offset % ss_pdcch_candidates.size()][aggr_level_index][candidate_index].
94 // We need to keep separate lists for different slot offsets because PDCCH candidates change with the slot index,
95 // may have a monitoring periodicity above 1 slot, and may be affected by the candidates of other search spaces.
96 std::vector<std::array<pdcch_candidate_list, NOF_AGGREGATION_LEVELS>> ss_pdcch_candidates;
97
98 // List of CRBs used by each PDCCH candidate.
99 std::vector<std::array<std::vector<crb_index_list>, NOF_AGGREGATION_LEVELS>> crbs_of_candidates;
100};
