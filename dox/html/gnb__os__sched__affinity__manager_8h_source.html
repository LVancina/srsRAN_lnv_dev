67struct gnb_os_sched_affinity_config {
69 gnb_sched_affinity_mask_types type = gnb_sched_affinity_mask_types::last;
71 os_sched_affinity_bitmask mask;
73 gnb_sched_affinity_mask_policy pinning_policy = gnb_sched_affinity_mask_policy::last;
74};

80class gnb_os_sched_affinity_manager
81{
83 class affinity_entry
84 {
85 public:
86 explicit affinity_entry(const os_sched_affinity_bitmask& mask_) : mask(mask_) {}
87 virtual ~affinity_entry() = default;
88
89 virtual os_sched_affinity_bitmask calculate_mask() = 0;
90
91 protected:
92 os_sched_affinity_bitmask mask;
93 };
94
96 class affinity_entry_rr : public affinity_entry
97 {
98 public:
99 explicit affinity_entry_rr(const os_sched_affinity_bitmask& mask_) : affinity_entry(mask_) {}
100
103 os_sched_affinity_bitmask calculate_mask() override
104 {
105 // No CPU is selected.
106 if (mask.none()) {
107 return {};
108 }
109
110 int pos = -1;
111
112 while (pos < 0) {
113 pos = mask.find_lowest(last_pos_used, mask.size());
114
115 last_pos_used = (pos < 0) ? 0 : pos + 1;
116 }
117
118 return os_sched_affinity_bitmask(mask.size(), pos);
119 }
120
121 private:
122 unsigned last_pos_used = 0;
123 };
124
126 class affinity_entry_mask : public affinity_entry
127 {
128 public:
129 explicit affinity_entry_mask(const os_sched_affinity_bitmask& mask_) : affinity_entry(mask_) {}
130
131 os_sched_affinity_bitmask calculate_mask() override { return mask; }
132 };
133
134public:
135 gnb_os_sched_affinity_manager(const std::vector<gnb_os_sched_affinity_config>& entries) :
136 cpu_masks(to_unsigned(gnb_sched_affinity_mask_types::last))
137 {
138 for (const auto& entry : entries) {
139 srsran_assert(entry.type != gnb_sched_affinity_mask_types::last, "Invalid type '{}'", to_unsigned(entry.type));
140 switch (entry.pinning_policy) {
141 case gnb_sched_affinity_mask_policy::round_robin:
142 cpu_masks[to_unsigned(entry.type)] = std::make_unique<affinity_entry_rr>(entry.mask);
143 break;
144 case gnb_sched_affinity_mask_policy::mask:
145 cpu_masks[to_unsigned(entry.type)] = std::make_unique<affinity_entry_mask>(entry.mask);
146 break;
147 default:
148 cpu_masks[to_unsigned(entry.type)] = nullptr;
149 }
150 }
151 }
152
154 os_sched_affinity_bitmask calcute_affinity_mask(gnb_sched_affinity_mask_types type)
155 {
156 srsran_assert(type != gnb_sched_affinity_mask_types::last, "Invalid type '{}'", to_unsigned(type));
157 srsran_assert(cpu_masks[to_unsigned(type)], "Invalid manager for type '{}'", to_unsigned(type));
158
159 return cpu_masks[to_unsigned(type)]->calculate_mask();
160 }
161
162private:
163 std::vector<std::unique_ptr<affinity_entry>> cpu_masks;
164};
