38inline rach_indication_message
39create_rach_indication(slot_point next_slot_rx, std::initializer_list<rach_indication_message::preamble> preambles)
40{
41 rach_indication_message rach_ind{};
42 rach_ind.cell_index = to_du_cell_index(0);
43 rach_ind.slot_rx = next_slot_rx;
44 if (preambles.size() == 0) {
45 return rach_ind;
46 }
47 rach_ind.occasions.emplace_back();
48 rach_ind.occasions.back().start_symbol = 0;
49 rach_ind.occasions.back().frequency_index = 0;
50
51 for (const auto& preamble : preambles) {
52 rach_ind.occasions.back().preambles.push_back(preamble);
53 }
54 return rach_ind;
55}

57inline uci_indication create_uci_indication(slot_point uci_sl, du_ue_index_t ue_idx, const pucch_info& pucch_pdu)
58{
59 uci_indication uci_ind{};
60 uci_ind.cell_index = to_du_cell_index(0);
61 uci_ind.slot_rx = uci_sl;
62 uci_ind.ucis.resize(1);
63 uci_ind.ucis[0].crnti = pucch_pdu.crnti;
64 uci_ind.ucis[0].ue_index = ue_idx;
65 switch (pucch_pdu.format) {
66 case pucch_format::FORMAT_1: {
67 uci_indication::uci_pdu::uci_pucch_f0_or_f1_pdu f1{};
68 f1.harqs.resize(pucch_pdu.format_1.harq_ack_nof_bits);
69 std::fill(f1.harqs.begin(), f1.harqs.end(), mac_harq_ack_report_status::ack);
70 uci_ind.ucis[0].pdu = f1;
71 } break;
72 case pucch_format::FORMAT_2: {
73 uci_indication::uci_pdu::uci_pucch_f2_or_f3_or_f4_pdu f2{};
74 f2.harqs.resize(pucch_pdu.format_2.harq_ack_nof_bits);
75 std::fill(f2.harqs.begin(), f2.harqs.end(), mac_harq_ack_report_status::ack);
76 uci_ind.ucis[0].pdu = f2;
77 if (pucch_pdu.format_2.csi_part1_bits > 0) {
78 f2.csi.emplace();
79 f2.csi->first_tb_wideband_cqi = 15;
80 }
81 } break;
82 default:
83 report_fatal_error("Unsupported PUCCH format");
84 }
85 return uci_ind;
86}
