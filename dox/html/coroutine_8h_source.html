93 template <typename Awaitable, typename EventType = detail::awaitable_result_t<Awaitable>>
94 detail::enable_if_nonvoid<EventType> on_await_resume()
95 {
96 using ForwardedAwaitable =
97 std::conditional_t<std::is_rvalue_reference<Awaitable>::value, std::decay_t<Awaitable>, Awaitable>;
98 srsran_assert(is_suspended(), "Trying to resume non-suspended coroutine");
99
100 // Extract result from awaiter and clear awaiter from memory buffer
101 auto& awaiter = mem_buffer.get_awaiter<ForwardedAwaitable>();
102 EventType ev = awaiter.await_resume();
103 mem_buffer.clear<ForwardedAwaitable>();
104 return ev;
105 }

132template <typename Promise>
133struct base_coro_frame : public base_coro_frame<void>, private Promise {
134 using promise_type = Promise;

147 template <typename Awaitable>
148 bool on_await_call(Awaitable&& awaitable_obj)
149 {
150 using ForwardedAwaitable =
151 std::conditional_t<std::is_rvalue_reference<Awaitable>::value, std::decay_t<Awaitable>, Awaitable>;
152 srsran_assert(not is_suspended(), "Trying to suspend already suspended coroutine");
153
154 // store awaitable/awaiter in memory buffer
155 mem_buffer.emplace(std::forward<Awaitable>(awaitable_obj));
156
157 // check if awaiter is ready. If not, suspend coroutine
158 awaiter_t<ForwardedAwaitable>& awaiter_obj = mem_buffer.get_awaiter<ForwardedAwaitable>();
159 if (not awaiter_obj.await_ready()) {
160 // Tag-based dispatch - Check whether awaiter uses symmetric transfer / tail-resume on await_suspend.
161 using suspend_sig = decltype(&std::decay_t<awaiter_t<ForwardedAwaitable>>::await_suspend);
162 using suspend_return_t = function_return_t<suspend_sig>;
163 using no_tail_resume_tag = std::is_same<void, suspend_return_t>;
164
165 on_await_suspend<awaiter_t<ForwardedAwaitable>>(awaiter_obj, no_tail_resume_tag{});
166 return false;
167 }
168 return true;
169 }
170
171private:
173 template <typename Awaiter>
174 void on_await_suspend(Awaiter& awaiter_obj, std::false_type)
175 {
176 // Store current caller/suspending coroutine as a continuation in the awaiter object. "This" coroutine will be
177 // called at final_suspend.
178 auto ch = awaiter_obj.await_suspend(this);
179
180 // Tail resume returned coroutine.
181 ch.resume();
182 }
183 template <typename Awaiter>
184 void on_await_suspend(Awaiter& awaiter_obj, std::true_type)
185 {
186 awaiter_obj.await_suspend(this);
187 }
188};

191template <typename FunT>
192struct coro_frame : public base_coro_frame<detail::promise_of<FunT>> {
193 template <typename... Args>
194 explicit coro_frame(Args&&... args)
195 {
196 // Create coroutine object in embedded memory buffer
197 new (&task_storage) FunT(std::forward<Args>(args)...);
198 }
199
201 void resume() final { (*get_impl())(*this); }
202
205 void destroy() noexcept final
206 {
207 srsran_sanity_check(not this->mem_buffer.empty(), "coroutine that is not suspended should not be destroyed");
208
209 // resume via cancel path until final suspension point
210 cancel();
211 srsran_sanity_check(this->state_index == coro_state_tag_t::tag_cancelled, "Invalid coroutine state");
212
213 // called after final suspension in order to destroy & deallocate coroutine frame
214 delete this;
215 }
216
218 void on_return() final { get_impl()->~FunT(); }
219
220private:
221 FunT* get_impl() { return reinterpret_cast<FunT*>(&task_storage); }
222 void cancel()
223 {
224 if (this->state_index < 0) {
225 // already cancelled or finished
226 if (this->state_index == coro_state_tag_t::tag_final_suspend) {
227 this->template on_await_cancel<decltype(this->promise().final_suspend())>();
228 this->state_index = coro_state_tag_t::tag_cancelled;
229 }
230 return;
231 }
232 this->state_index = -this->state_index;
233 resume();
234 }
235
236 std::aligned_storage_t<sizeof(FunT), alignof(FunT)> task_storage;
237};
