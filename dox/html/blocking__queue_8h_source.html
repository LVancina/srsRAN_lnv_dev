72 void stop()
73 {
74 std::unique_lock<std::mutex> lock(mutex);
75 if (active) {
76 active = false;
77 if (nof_waiting > 0) {
78 // Stop pending pushing/popping threads
79 do {
80 lock.unlock();
81 cvar_empty.notify_all();
82 cvar_full.notify_all();
83 std::this_thread::yield();
84 lock.lock();
85 } while (nof_waiting > 0);
86 }
87
88 // Empty queue
89 ring_buf.clear();
90 }
91 }

285 // Blocks producer while task queue is full.
286 bool wait_push_possible(std::unique_lock<std::mutex>& lock, blocking_tag )
287 {
288 if (active and ring_buf.full()) {
289 nof_waiting++;
290 while (ring_buf.full() and active) {
291 cvar_full.wait(lock);
292 }
293 nof_waiting--;
294 }
295 return active;
296 }

298 // Returns true if there is space in the task queue.
299 bool wait_push_possible(std::unique_lock<std::mutex>& , non_blocking_tag )
300 {
301 return active and not ring_buf.full();
302 }

330 template <typename It, typename BlockingMode>
331 unsigned push_(It b, It e, const BlockingMode& mode)
332 {
333 unsigned count = 0;
334 for (auto it = b; it != e;) {
335 std::unique_lock<std::mutex> lock(mutex);
336 if (not wait_push_possible(lock, mode)) {
337 return count;
338 }
339 unsigned n = ring_buf.try_push(it, e);
340 if (n == 0) {
341 break;
342 }
343 for (unsigned i = 0; i != n; ++i) {
344 push_func(*it);
345 ++it;
346 }
347 count += n;
348 cvar_empty.notify_one();
349 }
350 return count;
351 }

377 template <typename BlockingMode>
378 bool wait_pop_possible(std::unique_lock<std::mutex>& lock, const BlockingMode& mode)
379 {
380 if (active and ring_buf.empty()) {
381 nof_waiting++;
382 wait_pop_possible_(lock, mode);
383 nof_waiting--;
384 }
385 return active and not ring_buf.empty();

388 template <typename BlockingMode>
389 result pop_(T& obj, const BlockingMode& mode)
390 {
391 std::unique_lock<std::mutex> lock(mutex);
392 if (wait_pop_possible(lock, mode)) {
393 obj = std::move(ring_buf.top());
394 pop_func(obj);
395 ring_buf.pop();
396 cvar_full.notify_one();
397 return result::success;
398 }
399 return active ? result::timeout : result::failed;
400 }
401
402 template <typename It, typename BlockingMode>
403 unsigned pop_(It b, It e, const BlockingMode& mode)
404 {
405 unsigned count = 0;
406 for (auto it = b; it != e;) {
407 std::unique_lock<std::mutex> lock(mutex);
408 if (not wait_pop_possible(lock, mode)) {
409 break;
410 }
411 unsigned n = ring_buf.pop_into(it, e);
412 if (n == 0) {
413 break;
414 }
415 for (unsigned i = 0; i != n; ++i) {
416 pop_func(*it);
417 ++it;
418 }
419 count += n;
420 lock.unlock();
421 // notify all waiting producers because we may have emptied more than one slot of the queue.
422 cvar_full.notify_all();
423 }
424 return count;
425 }

427 template <typename StopCondition>
428 optional<T> pop_and_discard_until_(const StopCondition& cond)
429 {
430 optional<T> ret{};
431 bool notify_needed = false;
432 std::unique_lock<std::mutex> lock(mutex);
433 while (wait_pop_possible(lock, non_blocking_tag{})) {
434 pop_func(ring_buf.top());
435 notify_needed = true;
436 if (cond(ring_buf.top())) {
437 ret = std::move(ring_buf.top());
438 ring_buf.pop();
439 break;
440 }
441 ring_buf.pop();
442 }
443 if (notify_needed) {
444 cvar_full.notify_one();
445 }
446 return ret;
447 }
