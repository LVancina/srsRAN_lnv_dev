53template <typename Integer, Integer MIN_VALUE, Integer MAX_VALUE>
54class bounded_integer : public detail::bounded_integer_base<Integer>
55{
56 static_assert(std::is_integral<Integer>::value, "Template argument must be an integer");
57 static_assert(MIN_VALUE <= MAX_VALUE, "Provided bounds for bounded_integer are not valid");
58
59 using base_class = detail::bounded_integer_base<Integer>;
60
61public:
62 using base_class::base_class;
63 using difference_type = std::conditional_t<sizeof(Integer) == sizeof(int64_t), int64_t, int32_t>;
64
65 constexpr bounded_integer() = default;
66 constexpr bounded_integer(bounded_integer_invalid_tag ) : base_class(MAX_VALUE + 1) {}
67 constexpr bounded_integer(Integer v) : base_class(v) { assert_bounds(v); }
68
69 bounded_integer& operator=(Integer v)
70 {
71 assert_bounds(v);
72 base_class::value() = v;
73 return *this;
74 }
75
76 static constexpr Integer min() { return MIN_VALUE; }
77 static constexpr Integer max() { return MAX_VALUE; }
78
80 constexpr bool valid() const { return base_class::value() >= MIN_VALUE && base_class::value() <= MAX_VALUE; }
81
83 explicit constexpr operator Integer() const { return base_class::value(); }
