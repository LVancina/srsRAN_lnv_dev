33class quantizer
34{
35public:
37 explicit quantizer(unsigned bit_width_) :
38 bit_width(bit_width_), nbits_shift_sign(16U - bit_width), gain((1 << (bit_width - 1)) - 1.0f)
39 {
40 }
41
46 int16_t to_fixed_point(float value) const
47 {
48 float clipped = value;
49 if (std::abs(value) > 1.0) {
50 clipped = std::copysign(1.0, value);
51 }
52 return static_cast<int16_t>(std::round(clipped * gain));
53 }
54
62 void to_fixed_point(span<int16_t> z, span<const float> x, float in_scale) const
63 {
64 float scale = gain * in_scale;
65 srsvec::convert_round(x, scale, z);
66 }
67
72 float to_float(int p) const { return (p / gain); }
73
82 void to_float(span<cf_t> z, span<const int16_t> x, int16_t in_scale) const
83 {
84 float scale = gain / in_scale;
85 srsvec::convert(x, scale, z);
86 }
87
89 int16_t sign_extend(int16_t value) const
90 {
91 int16_t shifted = value << nbits_shift_sign;
92 return shifted >> nbits_shift_sign;
93 }
94
95private:
97 const unsigned bit_width;
99 const unsigned nbits_shift_sign;
101 const float gain;
102};
