31template <typename FutureType>
32struct base_resumable_procedure {
33 using future_type = FutureType;
34 using promise_type = typename FutureType::promise_type;
35 using result_type = typename FutureType::result_type;
36
38 virtual void start() = 0;
39
41 void operator()(detail::base_coro_frame<promise_type>& ctx)
42 {
43 if (frame_ptr == nullptr) {
44 srsran_sanity_check(ctx.state_index == detail::tag_init, "Invalid coro state");
45 ctx.state_index = 10;
46 frame_ptr = &ctx;
47 async_await(suspend_always{}, &base_resumable_procedure<FutureType>::start);
48 return;
49 }
50 resume_method();
51 }
52
53protected:
55 template <typename Awaitable, typename Derived, typename... Args>
56 void async_await(Awaitable&& a, void (Derived::*action)(Args...))
57 {
58 auto resume_func = [this, action]() {
59 if (this->frame_ptr->state_index > 0) {
60 await_resume_helper<Awaitable>(action);
61 } else {
62 srsran_sanity_check(frame_ptr->state_index != detail::coro_state_tag_t::tag_cancelled,
63 "Calling resume on cancelled task");
64 srsran_sanity_check(frame_ptr->state_index != detail::coro_state_tag_t::tag_final_suspend,
65 "Calling on task on suspended point");
66 // cancelled
67 frame_ptr->template on_await_cancel<Awaitable>();
68 frame_ptr->state_index = detail::coro_state_tag_t::tag_cancelled;
69 frame_ptr->on_return();
70 // Note: Do not touch any member at this point
71 }
72 };
73
74 // Note: we have to store the resume point before, in case the coroutine changes execution context and calls resume
75 // before on_await_call is complete
76 resume_method = resume_func;
77 if (frame_ptr->on_await_call(std::forward<Awaitable>(a))) {
78 resume_func();
79 }
80 }
81
83 template <typename R = result_type>
84 detail::enable_if_void<R> async_return()
85 {
86 frame_ptr->promise().return_value();
87 final_await();
88 }
89
91 template <typename U, typename R = result_type>
92 detail::enable_if_nonvoid<R, void> async_return(U&& r)
93 {
94 static_assert(std::is_convertible<U, result_type>::value, "Invalid value returned in async_return(R)");
95 frame_ptr->promise().return_value(std::forward<U>(r));
96 final_await();
97 }
98
99 ~base_resumable_procedure() = default;
100
101private:
102 void final_await()
103 {
104 // destroy procedure object including its members (without destroying the rest of the frame)
105 auto* local_ptr = frame_ptr;
106 frame_ptr->state_index = detail::coro_state_tag_t::tag_final_suspend;
107 frame_ptr->on_return();
108 // Note: Do not touch any procedure variable from now on.
109
110 // suspend on final awaiter
111 if (local_ptr->on_await_call(local_ptr->promise().final_suspend())) {
112 // destroy coroutine frame
113 local_ptr->destroy();
114 }
115 }
116
118 template <typename Awaitable, typename Derived, typename Arg>
119 detail::enable_if_nonvoid<detail::awaitable_result_t<Awaitable>, void>
120 await_resume_helper(void (Derived::*action)(Arg))
121 {
122 (static_cast<Derived*>(this)->*action)(frame_ptr->template on_await_resume<Awaitable>());
123 }
124
126 template <typename Awaitable, typename Derived>
127 detail::enable_if_void<detail::awaitable_result_t<Awaitable>> await_resume_helper(void (Derived::*action)())
128 {
129 frame_ptr->template on_await_resume<Awaitable>();
130 (static_cast<Derived*>(this)->*action)();
131 }
132
133 detail::base_coro_frame<promise_type>* frame_ptr = nullptr;
134 std::function<void()> resume_method;
135};

138template <typename R>
139using async_procedure = base_resumable_procedure<async_task<R>>;
