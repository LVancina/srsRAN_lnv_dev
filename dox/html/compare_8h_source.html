35template <typename T1, typename T2>
36bool equal(const T1& s1, const T2& s2)
37{
38 static_assert(detail::is_span_compatible<T1>::value, "Template type is not compatible with a span");
39 static_assert(detail::is_span_compatible<T2>::value, "Template type is not compatible with a span");
40 srsran_srsvec_assert_size(s1, s2);
41
42 return std::equal(s1.begin(), s1.end(), s2.begin(), s2.end());
43}
