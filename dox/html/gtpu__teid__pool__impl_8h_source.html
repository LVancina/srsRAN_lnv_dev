33class gtpu_teid_pool_impl final : public gtpu_teid_pool
34{
35public:
36 explicit gtpu_teid_pool_impl(uint32_t max_teids_) : max_teids(max_teids_), teid_pool(max_teids_) {}
37
38 SRSRAN_NODISCARD expected<gtpu_teid_t> request_teid() override
39 {
40 expected<gtpu_teid_t> teid;
41 teid.set_error(default_error_t{});
42
43 if (full()) {
44 return teid;
45 }
46
47 // Find a free teid
48 bool found = false;
49 uint16_t tmp = next_teid;
50 for (uint16_t n = 0; n < max_teids; n++) {
51 tmp = (next_teid + n) % max_teids;
52 if (not teid_pool[tmp]) {
53 teid_pool[tmp] = true;
54 found = true;
55 break;
56 }
57 }
58
59 if (not found) {
60 return teid;
61 }
62 next_teid = (tmp + 1) % max_teids;
63 teid = gtpu_teid_t{tmp};
64 nof_teids++;
65 return teid;
66 }
67
68 SRSRAN_NODISCARD bool release_teid(gtpu_teid_t teid) override
69 {
70 if (not teid_pool[teid.value()]) {
71 // trying to free non-allocated TEID
72 return false;
73 }
74 teid_pool[teid.value()] = false;
75 nof_teids--;
76 return true;
77 }
78
79 bool full() const override { return nof_teids >= max_teids; }
80
81 uint32_t get_max_teids() override { return max_teids; }
82
83private:
84 uint32_t next_teid = 0;
85 uint32_t nof_teids = 0;
86 const uint32_t max_teids;
87
88 std::vector<bool> teid_pool;
89};
90} // namespace srsran
