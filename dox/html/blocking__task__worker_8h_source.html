32class blocking_task_worker final : public task_executor
33{
34public:
35 blocking_task_worker(size_t q_size) : pending_tasks(q_size) {}
36
37 bool execute(unique_task task) override
38 {
39 pending_tasks.push_blocking(std::move(task));
40 return true;
41 }
42
43 bool defer(unique_task task) override
44 {
45 execute(std::move(task));
46 return true;
47 }
48
49 void request_stop()
50 {
51 execute([this]() {
52 if (not pending_tasks.is_stopped()) {
53 pending_tasks.stop();
54 }
55 });
56 }
57
58 void run()
59 {
60 while (true) {
61 bool success;
62 unique_task t = pending_tasks.pop_blocking(&success);
63 if (not success) {
64 break;
65 }
66 t();
67 }
68 }
69
70private:
71 blocking_queue<unique_task> pending_tasks;
72};
