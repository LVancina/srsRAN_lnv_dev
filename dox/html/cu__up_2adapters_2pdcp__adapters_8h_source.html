82class pdcp_f1u_adapter : public pdcp_tx_lower_notifier
83{
84public:
85 pdcp_f1u_adapter() = default;
86 ~pdcp_f1u_adapter() override {}
87
88 void connect_f1u(f1u_tx_sdu_handler& f1u_handler_) { f1u_handler = &f1u_handler_; }
89 void disconnect_f1u() { f1u_handler = nullptr; }
90
91 void on_new_pdu(pdcp_tx_pdu pdu) override
92 {
93 if (f1u_handler == nullptr) {
94 srslog::fetch_basic_logger("PDCP").warning("Unconnected F1-U handler. Dropping PDCP PDU");
95 } else {
96 f1u_handler->handle_sdu(std::move(pdu));
97 }
98 }
99
100 void on_discard_pdu(uint32_t pdcp_sn) override
101 {
102 srsran_assert(f1u_handler != nullptr, "F1-U handler must not be nullptr");
103 f1u_handler->discard_sdu(pdcp_sn);
104 }
105
106private:
107 f1u_tx_sdu_handler* f1u_handler = nullptr;
108};
