32/// Adapter between E1AP and CU-UP
33class e1ap_cu_up_adapter : public e1ap_cu_up_notifier
34{
35public:
36 void connect_cu_up(cu_up_e1ap_interface& cu_up_handler_) { cu_up_handler = &cu_up_handler_; }
37
38 e1ap_bearer_context_setup_response
39 on_bearer_context_setup_request_received(const e1ap_bearer_context_setup_request& msg) override
40 {
41 srsran_assert(cu_up_handler != nullptr, "CU-UP handler must not be nullptr");
42 return cu_up_handler->handle_bearer_context_setup_request(msg);
43 }
44
45 e1ap_bearer_context_modification_response
46 on_bearer_context_modification_request_received(const e1ap_bearer_context_modification_request& msg) override
47 {
48 srsran_assert(cu_up_handler != nullptr, "CU-UP handler must not be nullptr");
49 return cu_up_handler->handle_bearer_context_modification_request(msg);
50 }
51
52 void on_bearer_context_release_command_received(const e1ap_bearer_context_release_command& msg) override
53 {
54 srsran_assert(cu_up_handler != nullptr, "CU-UP handler must not be nullptr");
55 return cu_up_handler->handle_bearer_context_release_command(msg);
56 }
57
58private:
59 cu_up_e1ap_interface* cu_up_handler = nullptr;
60};
