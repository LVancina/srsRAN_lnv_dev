30namespace srsran {

32class dmrs_pbch_processor_impl : public dmrs_pbch_processor
33{
34private:
35 /// DMRS PBCH Sequence length in the SSB
36 static const unsigned NOF_RE = 144;
37
38 std::unique_ptr<pseudo_random_generator> prg;
39
40 /// \brief Computes the initial pseudo-random state
41 /// \param [in] config provides the required parameters to calculate the value
42 /// \return the initial pseudo-random state
43 static unsigned c_init(const config_t& config);
44
45 /// \brief Implements TS 38.211 section 7.4.1.4.1 Sequence generation
46 /// \param [out] sequence provides the destination of the sequence generation
47 /// \param [in] config provides the required fields to generate the signal
48 void generation(std::array<cf_t, NOF_RE>& sequence, const config_t& config) const;
49
50 /// \brief Implements TS 38.211 section 7.4.1.4.2 Mapping to physical resources
51 /// \param [in] sequence provides the source of the sequence
52 /// \param [out] grid provides the grid destination to map the signal
53 /// \param [in] config provides the required fields to map the signal
54 void mapping(const std::array<cf_t, NOF_RE>& sequence, resource_grid_writer& grid, const config_t& config) const;
55
56public:
57 dmrs_pbch_processor_impl(std::unique_ptr<pseudo_random_generator> pseudo_random_generator) :
58 prg(std::move(pseudo_random_generator))
59 {
60 srsran_assert(prg, "Invalid PRG.");
61 }
62
63 // See interface for documentation.
64 void map(resource_grid_writer& grid, const config_t& config) override;
65};
