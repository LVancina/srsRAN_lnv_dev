34/// \brief Circularly shifts a sequence in the forward direction.
35///
36/// The element at position \c i is moved to position <tt>i + shift<\tt>. Elements that fall beyond the end of the
37/// sequence are reintroduced at its start.
38/// \param[out] out Shifted output sequence.
39/// \param[in] in Original input sequence.
40/// \param[in] shift The number of positions the sequence is shifted by.
41/// \remark Cannot be used to override memory.
42template <typename T, typename U>
43void circ_shift_forward(T&& out, const U& in, unsigned shift)
44{
45 static_assert(detail::is_span_compatible<T>::value, "Template type is not compatible with a span");
46 static_assert(detail::is_span_compatible<U>::value, "Template type is not compatible with a span");
47 srsran_srsvec_assert_size(out, in);
48
49 unsigned length = out.size();
50 srsran_assert(std::abs(in.data() - out.data()) >= length, "Input and output memory overlap.");
51
52 copy(out.first(shift), in.last(shift));
53 copy(out.subspan(shift, length - shift), in.first(length - shift));
54}
55
56/// \brief Circularly shifts a sequence in the backward direction.
57///
58/// The element at position \c i is moved to position <tt>i - shift<\tt>. Elements that fall beyond the beginning of the
59/// sequence are reintroduced at its end.
60/// \param[out] out Shifted output sequence.
61/// \param[in] in Original input sequence.
62/// \param[in] shift The number of positions the sequence is shifted by.
63/// \remark Cannot be used to override memory.
64template <typename T, typename U>
65void circ_shift_backward(T&& out, const U& in, unsigned shift)
66{
67 static_assert(detail::is_span_compatible<T>::value, "Template type is not compatible with a span");
68 static_assert(detail::is_span_compatible<U>::value, "Template type is not compatible with a span");
69 srsran_srsvec_assert_size(out, in);
70
71 unsigned length = out.size();
72 srsran_assert(std::abs(in.data() - out.data()) >= length, "Input and output memory overlap.");
73
74 copy(out.first(length - shift), in.last(length - shift));
75 copy(out.subspan(length - shift, shift), in.first(shift));
76}
