41/// \brief Represents log-likelihood ratio (LLR) values.
42///
43/// Log-likelihood ratios are represented as \c int8_t values between <tt>-LLR_MAX</tt> and \c LLR_MAX. The values
44/// <tt>&plusmn;LLR_INFTY = &plusmn;127</tt> are reserved for denoting fixed bits, that is bits that are surely set to
45/// zero (log-likelihood ratio equal to <tt>+LLR_INFTY</tt>) or one (log-likelihood ratio equal to <tt>-LLR_INFTY</tt>).
46class log_likelihood_ratio
47{
48public:
49 /// Alias for the underlying data type used for representing the LLR value.
50 using value_type = int8_t;
51
52 /// Default constructor.
53 constexpr log_likelihood_ratio() = default;
54
55 /// \brief Constructor from integral type.
56 ///
57 /// Creates a new LLR set to the provided value if the latter is in the range <tt>-LLR_MAX</tt> to \c LLR_MAX, or
58 /// if it is <tt>&plusmn;LLR_INFTY</tt>. Otherwise, terminates the program execution.
59 /// \tparam T The input type. Must be an integral type.
60 /// \param[in] val The value the LLR is set to.
61 /// \remark Implicit conversions are allowed on purpose.
62 template <typename T>
63 constexpr log_likelihood_ratio(T val)
64 {
65 static_assert(std::is_integral<T>::value, "LLRs must be initialized with integer values.");
66 srsran_assert(((val <= LLR_MAX) && (val >= -LLR_MAX)) || (val == LLR_INFTY) || (val == -LLR_INFTY),
67 "Invalid LLR value.");
68 value = static_cast<value_type>(val);
69 }
70
71 /// \name Conversion operators.
72 ///@{
73
74 /// Converts the LLR to a plain \c value_type value.
75 explicit constexpr operator value_type() const { return value; };
76 /// Converts the LLR to a plain \c value_type value.
77 constexpr value_type to_value_type() const { return value; };
78
79 /// Converts the LLR to a plain \c int value.
80 explicit constexpr operator int() const { return value; };
81 /// Converts the LLR to a plain \c int value.
82 constexpr int to_int() const { return value; };
83
84 /// \brief Converts the LLR to a hard bit.
85 /// \note By convention, the null LLR indeterminate case returns a 1 bit.
86 constexpr uint8_t to_hard_bit() const { return static_cast<uint8_t>(value <= 0); }
87 ///@}
88
89 /// Default assignment operator.
90 constexpr log_likelihood_ratio& operator=(const log_likelihood_ratio& other)
91 {
92 value = other.value;
93 return *this;
94 }
95
96 /// Negation (additive inverse).
97 constexpr log_likelihood_ratio operator-() const { return -value; }
98
99 /// \brief Saturated sum.
100 ///
101 /// The result is clipped if larger (in absolute value) than log_likelihood_ratio::LLR_MAX. Then, for instance,
102 /// <tt>LLR_MAX + 2 = LLR_MAX</tt> and <tt>-LLR_MAX - 2 = -LLR_MAX</tt>.
103 log_likelihood_ratio operator+(log_likelihood_ratio rhs) const
104 {
105 rhs += *this;
106 return rhs;
107 }
108
109 /// \brief Addition assignment with saturated sum.
110 log_likelihood_ratio operator+=(log_likelihood_ratio rhs);
111
112 /// \brief Saturated difference.
113 ///
114 /// Follows naturally from the saturated sum: <tt>a - b = a + (-b)</tt>.
115 log_likelihood_ratio operator-(log_likelihood_ratio rhs) const { return *this + (-rhs); }
116
117 /// \brief Subtraction assignment with saturated sum.
118 log_likelihood_ratio operator-=(log_likelihood_ratio rhs)
119 {
120 *this += (-rhs);
121 return *this;
122 }
123
124 /// \name Comparison operators.
125 ///@{
126
127 /// Plain comparison operator.
128 constexpr bool operator==(log_likelihood_ratio rhs) const { return value == rhs.value; }
129 constexpr bool operator!=(log_likelihood_ratio rhs) const { return value != rhs.value; }
130 constexpr bool operator>(log_likelihood_ratio rhs) const { return value > rhs.value; }
131 constexpr bool operator<(log_likelihood_ratio rhs) const { return value < rhs.value; }
132 constexpr bool operator>=(log_likelihood_ratio rhs) const { return value >= rhs.value; }
133 constexpr bool operator<=(log_likelihood_ratio rhs) const { return value <= rhs.value; }
134 ///@}
135
136 /// \brief Clips and quantizes a real-valued (\c float), continuous log-likelihood ratio to the discrete
137 /// representation of type \c log_likelihood_ratio.
138 ///
139 /// \param[in] value The real-valued log-likelihood ratio to quantize.
140 /// \param[in] range_limit The input value mapped to \c LLR_MAX.
141 /// \return A quantized representation of \c value as a \c log_likelihood_ratio quantity.
142 /// \note The quantization in the range <tt>(-range_limit, range_limit)</tt> is [mid-tread
143 /// uniform](https://en.wikipedia.org/wiki/Quantization_(signal_processing)#Mid-riser_and_mid-tread_uniform_quantizers),
144 /// with quantization step <tt> range_limit / LLR_MAX </tt>.
145 /// \note All values larger (in magnitude) than \c range_limit, will be clipped and mapped to
146 /// <tt>&plusmn;LLR_MAX</tt>, depending on their sign.
147 static log_likelihood_ratio quantize(float value, float range_limit);
148
149 /// Returns the special value "positive infinity".
150 static constexpr log_likelihood_ratio infinity() { return LLR_INFTY; }
151
152 /// Returns the maximum finite LLR value.
153 static constexpr log_likelihood_ratio max() { return LLR_MAX; }
154
155 /// Returns the minimum finite LLR value.
156 static constexpr log_likelihood_ratio min() { return -LLR_MAX; }
157
158 /// Checks whether a log-likelihood ratio has a finite value.
159 static constexpr bool isfinite(log_likelihood_ratio llr)
160 {
161 return ((llr.value >= -LLR_MAX) && (llr.value <= LLR_MAX));
162 }
163
164 /// Checks whether a log-likelihood ratio has an infinite value.
165 static constexpr bool isinf(log_likelihood_ratio llr) { return !isfinite(llr); }
166
167 /// \brief Promotion sum.
168 ///
169 /// As opposed to the saturated sum (i.e., log_likelihood_ratio::operator+), results larger (in absolute value) than
170 /// \c LLR_MAX are set to <tt>&plusmn;LLR_INFTY</tt>.
171 /// \param[in] a First summand.
172 /// \param[in] b Second summand.
173 /// \return The promotion sum of \c a and \c b.
174 static log_likelihood_ratio promotion_sum(log_likelihood_ratio a, log_likelihood_ratio b);
175
176 /// \brief Copy sign from a number.
177 ///
178 /// \tparam T An arithmetic type.
179 /// \param[in] a A log-likelihood ratio.
180 /// \param[in] b A real number.
181 /// \return A log-likelihood ratio with the magnitude of \c a and the sign of \c b.
182 /// \remark A zero in the second argument is considered as a positive number, following the convention of \c
183 /// std::copysign.
184 template <typename T>
185 static constexpr log_likelihood_ratio copysign(log_likelihood_ratio a, T b)
186 {
187 static_assert(std::is_arithmetic<T>::value, "Template type is not an arithmetic type.");
188 if (b < 0) {
189 return -std::abs(a.value);
190 }
191 return std::abs(a.value);
192 }
193
194 /// Absolute value.
195 static value_type abs(log_likelihood_ratio a) { return static_cast<value_type>(std::abs(a.to_int())); }
196
197 /// \brief Soft XOR operation.
198 ///
199 /// The soft XOR between two LLRs \c x and \c y is the log-likelihood ratio corresponding to the logical XOR operation
200 /// \f$a \oplus b\f$, where \f$a\f$ and \f$b\f$ are the bits whose soft representation is \c x and \c y, respectively.
201 ///
202 /// \remark For complexity reasons, the true soft xor function
203 /// \f[
204 /// \text{soft_xor}(x, y) = \sign(x y) \cdot \Phi\Bigl(\Phi(\abs{x}) +
205 /// \Phi(\abs{y})\Bigr) \qquad\text{with }
206 /// \Phi: \mathbb{R}_+ \to \mathbb{R}_+; u \mapsto \Phi(u) = -\ln\Bigl(\tanh\Bigl(\frac{u}{2}\Bigr)\Bigr)
207 /// \f]
208 /// is replaced by the common approximation
209 /// \f[
210 /// \text{soft_xor}(x, y) \approx \sign(x y) \cdot \min\bigl\{\abs{x}, \abs{y}\bigr\}.
211 /// \f]
212 static log_likelihood_ratio soft_xor(log_likelihood_ratio x, log_likelihood_ratio y)
213 {
214 int abs_x = std::abs(x.value);
215 int abs_y = std::abs(y.value);
216 if (x.value * y.value < 0) {
217 return -std::min(abs_x, abs_y);
218 }
219 return std::min(abs_x, abs_y);
220 }
221
222 // Documentation at definition (see below).
223 template <typename T>
224 static int norm_squared(const T& x);
225
226 // Documentation at definition (see below).
227 template <typename T, typename U, typename V>
228 static V dot_prod(const T& x, const U& y, V init);
229
230private:
231 /// Actual LLR value.
232 value_type value = 0;
233
234 /// \brief Represents a log-likelihood ratio set to infinity.
235 ///
236 /// An LLR takes the value \c LLR_INFTY when the corresponding bit is certainly zero. Similarly, an LLR takes the
237 /// value <tt>-LLR_INFTY</tt> when the corresponding bit is certainly one.
238 static constexpr value_type LLR_INFTY = 127;
239
240 /// \brief Maximum (finite) value of a log-likelihood ratio.
241 ///
242 /// Finite LLRs take values between <tt>-LLR_MAX</tt> and \c LLR_MAX (both included).
243 static constexpr value_type LLR_MAX = 120;
244};

278/// \brief Dot product of a sequence of LLRs and a sequence of values of an arithmetic type.
279///
280/// Computes the dot product (a.k.a. inner product or scalar product) of the two input sequences, adding an initial
281/// offset.
282/// \tparam T A span (or any container convertible to a span) of log-likelihood ratios (either constant or
283/// volatile).
284/// \tparam U A span (or any container convertible to a span) of an arithmetic type.
285/// \tparam V Output type (must be compatible with the product of objects of type \c T and \c U).
286/// \param[in] x A sequence of (possibly constant) log-likelihood ratios.
287/// \param[in] y A second sequence of values.
288/// \param[in] init Initialization value.
289/// \return The dot product between the two spans plus \c init, i.e. \f$ x \cdot y + \textup{init} = \sum_i x_i y_i +
290/// \textup{init}\f$.
291/// \remark The two input spans must have the same length.
292/// \remark This function treats the LLRs as real values, thus neither saturation nor promotion apply.
293template <typename T, typename U, typename V>
294V log_likelihood_ratio::dot_prod(const T& x, const U& y, V init)
295{
296 static_assert(detail::is_llr_span_compatible<T>::value, "Template type is not compatible with a span of LLRs");
297 static_assert(srsvec::detail::is_arithmetic_span_compatible<U>::value,
298 "Template type is not compatible with a span of arithmetics");
299 srsran_assert(x.size() == y.size(), "Input spans must have identical sizes: '{}' vs '{}'", x.size(), y.size());
300 return std::inner_product(
301 x.begin(), x.end(), y.begin(), init, std::plus<V>(), [](log_likelihood_ratio a, log_likelihood_ratio b) {
302 return a.to_int() * b.to_int();
303 });
304}

306/// \brief Squared Euclidean norm of a sequence of LLRs.
307///
308/// Computes the squared Euclidean norm (sum of the squares of the elements) of the input sequence of LLRs.
309///
310/// \tparam T A span (or any container convertible to a span) of a log-likelihood ratios (either constant or
311/// volatile).
312/// \param[in] x A sequence of (possibly constant) log-likelihood ratios.
313/// \return The squared Euclidean norm of \c x as an integer.
314/// \remark This function treats the LLRs as real values, thus neither saturation nor promotion apply.
315template <typename T>
316inline int log_likelihood_ratio::norm_squared(const T& x)
317{
318 static_assert(detail::is_llr_span_compatible<T>::value, "Template type is not compatible with a span of LLRs");
319 return std::accumulate(
320 x.begin(), x.end(), 0, [](int a, log_likelihood_ratio b) { return a + b.to_int() * b.to_int(); });
321}
