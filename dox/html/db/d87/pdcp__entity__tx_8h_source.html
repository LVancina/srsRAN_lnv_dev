41/// PDCP TX state variables,
42/// TS 38.323, section 7.1
43struct pdcp_tx_state {
44 /// This state variable indicates the COUNT value of the next PDCP SDU to be transmitted. The initial value is 0,
45 /// except for SRBs configured with state variables continuation.
46 uint32_t tx_next = 0;
47 /// This state variable indicates the next COUNT value for which we will
48 /// receive a transmission notification from the F1/RLC. If TX_TRANS == TX_NEXT,
49 /// it means we are not currently waiting for any TX notification.
50 /// NOTE: This is a custom state variable, not specified by the standard.
51 uint32_t tx_trans = 0;
52 /// This state variable indicates the lower edge of the TX window, i.e. the COUNT value of the oldest PDCP SDU in the
53 /// TX window for which the discard timer is expected to expire soonest. If TX_NEXT_ACK == TX_NEXT, it means we are
54 /// not currently having any discard timers running.
55 /// For UM bearers this value is the same as TX_TRANS.
56 /// NOTE: This is a custom state variable, not specified by the standard.
57 uint32_t tx_next_ack = 0;
58
59 bool operator==(const pdcp_tx_state& other) const
60 {
61 return tx_next == other.tx_next && tx_trans == other.tx_trans && tx_next_ack == other.tx_next_ack;
62 }
63};

75 pdcp_entity_tx(uint32_t ue_index,
76 rb_id_t rb_id_,
77 pdcp_tx_config cfg_,
78 pdcp_tx_lower_notifier& lower_dn_,
79 pdcp_tx_upper_control_notifier& upper_cn_,
80 timer_factory timers_) :
81 pdcp_entity_tx_rx_base(rb_id_, cfg_.rb_type, cfg_.rlc_mode, cfg_.sn_size),
82 logger("PDCP", {ue_index, rb_id_, "DL"}),
83 cfg(cfg_),
84 lower_dn(lower_dn_),
85 upper_cn(upper_cn_),
86 timers(timers_),
87 tx_window(create_tx_window(cfg.sn_size))
88 {
89 // Validate configuration
90 if (is_srb() && (cfg.sn_size != pdcp_sn_size::size12bits)) {
91 report_error("PDCP SRB with invalid sn_size. {}", cfg);
92 }
93 if (is_srb() && is_um()) {
94 report_error("PDCP SRB cannot be used with RLC UM. {}", cfg);
95 }
96 if (is_srb() && cfg.discard_timer.has_value()) {
97 logger.log_error("Invalid SRB config with discard_timer={}", cfg.discard_timer);
98 }
99 if (is_drb() && !cfg.discard_timer.has_value()) {
100 logger.log_error("Invalid DRB config, discard_timer is not configured");
101 }
102
103 direction = cfg.direction == pdcp_security_direction::uplink ? security::security_direction::uplink
104 : security::security_direction::downlink;
105 logger.log_info("PDCP configured. {}", cfg);
106 }

154 /*
155 * Security configuration
156 */
157 void configure_security(security::sec_128_as_config sec_cfg_) final
158 {
159 srsran_assert((is_srb() && sec_cfg_.domain == security::sec_domain::rrc) ||
160 (is_drb() && sec_cfg_.domain == security::sec_domain::up),
161 "Invalid sec_domain={} for {} in {}",
162 sec_cfg.domain,
163 rb_type,
164 rb_id);
165 // The 'NULL' integrity protection algorithm (nia0) is used only for SRBs and for the UE in limited service mode,
166 // see TS 33.501 [11] and when used for SRBs, integrity protection is disabled for DRBs. In case the ′NULL'
167 // integrity protection algorithm is used, 'NULL' ciphering algorithm is also used.
168 // Ref: TS 38.331 Sec. 5.3.1.2
169 if ((sec_cfg_.integ_algo == security::integrity_algorithm::nia0) &&
170 (is_drb() || (is_srb() && sec_cfg_.cipher_algo != security::ciphering_algorithm::nea0))) {
171 logger.log_error(
172 "Integrity algorithm NIA0 is only permitted for SRBs configured with NEA0. is_srb={} NIA{} NEA{}",
173 is_srb(),
174 sec_cfg_.integ_algo,
175 sec_cfg_.cipher_algo);
176 }
177
178 sec_cfg = sec_cfg_;
179 logger.log_info(
180 "Security configured: NIA{} NEA{} domain={}", sec_cfg.integ_algo, sec_cfg.cipher_algo, sec_cfg.domain);
181 if (sec_cfg.k_128_int.has_value()) {
182 logger.log_info(sec_cfg.k_128_int.value().data(), 16, "128 K_int");
183 }
184 logger.log_info(sec_cfg.k_128_enc.data(), 16, "128 K_enc");
185 };

187 void set_integrity_protection(security::integrity_enabled integrity_enabled_) final
188 {
189 if (integrity_enabled_ == security::integrity_enabled::on) {
190 if (!sec_cfg.k_128_int.has_value()) {
191 logger.log_error("Cannot enable integrity protection: Integrity key is not configured.");
192 return;
193 }
194 if (!sec_cfg.integ_algo.has_value()) {
195 logger.log_error("Cannot enable integrity protection: Integrity algorithm is not configured.");
196 return;
197 }
198 }
199 integrity_enabled = integrity_enabled_;
200 logger.log_info("Set integrity_enabled={}", integrity_enabled);
201 }
202 void set_ciphering(security::ciphering_enabled ciphering_enabled_) final
203 {
204 ciphering_enabled = ciphering_enabled_;
205 logger.log_info("Set ciphering_enabled={}", ciphering_enabled);
