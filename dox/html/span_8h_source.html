96template <typename T>
97class span
98{
99public:
101 using element_type = T;
102 using value_type = typename std::remove_cv<T>::type;
103 using size_type = std::size_t;
104 using difference_type = std::ptrdiff_t;
105 using pointer = element_type*;
106 using const_pointer = const element_type*;
107 using reference = element_type&;
108 using const_reference = const element_type&;
109 using iterator = pointer;
110 using reverse_iterator = std::reverse_iterator<iterator>;
111
113 constexpr span() noexcept = default;
114
116 constexpr span(pointer ptr_, size_type len_) noexcept : ptr(ptr_), len(len_) {}
117
119 constexpr span(pointer first, pointer last) noexcept : ptr(first), len(last - first) {}

122 template <std::size_t N>
123 constexpr span(element_type (&arr)[N]) noexcept : ptr(arr), len(N)
124 {
125 }

130 typename std::enable_if<std::is_convertible<U (*)[], element_type (*)[]>::value, int>::type = 0>
131 constexpr span(std::array<U, N>& arr) noexcept : ptr(arr.data()), len(N)
132 {
133 }

138 typename std::enable_if<std::is_convertible<const U (*)[], element_type (*)[]>::value, int>::type = 0>
139 constexpr span(const std::array<U, N>& arr) noexcept : ptr(arr.data()), len(N)
140 {

144 template <typename Container,
145 typename std::enable_if<detail::is_container_compatible<Container, element_type>::value, int>::type = 0>
146 constexpr span(Container& container) noexcept : ptr(container.data()), len(container.size())
147 {
148 }

151 template <
152 typename Container,
153 typename std::enable_if<detail::is_container_compatible<const Container, element_type>::value, int>::type = 0>
154 constexpr span(const Container& container) noexcept : ptr(container.data()), len(container.size())
155 {
156 }
157
158 template <typename U, typename std::enable_if<std::is_convertible<U (*)[], element_type (*)[]>::value, int>::type = 0>
159 constexpr span(const span<U>& other) noexcept : ptr(other.data()), len(other.size())
160 {
161 }
162
163 span& operator=(const span& other) noexcept = default;
164
165 ~span() noexcept = default;
166
168 constexpr size_type size() const noexcept { return len; }
169
171 constexpr size_type size_bytes() const noexcept { return len * sizeof(element_type); }
172
174 constexpr bool empty() const noexcept { return size() == 0; }
175
178 reference front() const
179 {
180 srsran_assert(!empty(), "Called front() with empty span");
181 return *data();
182 }
183
186 reference back() const
187 {
188 srsran_assert(!empty(), "called back with empty span");
189 return *(data() + size() - 1);
190 }
191
194 reference operator[](size_type idx) const
195 {
196 srsran_assert(idx < len, "Index out of bounds!");
197 return ptr[idx];
198 }
199
201 constexpr pointer data() const noexcept { return ptr; }
202
204 constexpr iterator begin() const noexcept { return data(); }
205
207 constexpr iterator end() const noexcept { return data() + size(); }
208
210 constexpr reverse_iterator rbegin() const noexcept { return reverse_iterator(end()); }
211
213 constexpr reverse_iterator rend() const noexcept { return reverse_iterator(begin()); }
214
217 span<element_type> first(size_type count) const
218 {
219 srsran_assert(count <= size(), "Count is out of range");
220 return subspan(0, count);
221 }
222
225 span<element_type> last(size_type count) const
226 {
227 srsran_assert(count <= size(), "Count is out of range");
228 return subspan(size() - count, count);
229 }
230
232 span<element_type> subspan(size_type offset, size_type count) const
233 {
234 srsran_assert(count <= size() - offset, "Size out of bounds!");
235 return {data() + offset, count};
236 }
237
239 bool equals(span rhs) const { return std::equal(begin(), end(), rhs.begin(), rhs.end()); }

241private:
242 pointer ptr = nullptr;
243 size_type len = 0;
244};

269template <typename T>
270struct formatter<srsran::span<T>> {
271 // Stores parsed format string.
272 memory_buffer format_buffer;
273
274 // Stores parsed delimiter string.
275 memory_buffer delimiter_buffer;
276
277 formatter()
278 {
279 static const string_view DEFAULT_FORMAT = "{}";
280 static const string_view DEFAULT_DELIMITER = " ";
281 format_buffer.append(DEFAULT_FORMAT.begin(), DEFAULT_FORMAT.end());
282 delimiter_buffer.append(DEFAULT_DELIMITER.begin(), DEFAULT_DELIMITER.end());
283 }
284
285 template <typename ParseContext>
286 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
287 {
288 static const string_view PREAMBLE_FORMAT = "{:";
289 static const string_view COMMA_DELIMITER = ", ";
290
291 // Skip if context is empty and use default format.
292 if (ctx.begin() == ctx.end()) {
293 return ctx.end();
294 }
295
296 // Store the format string.
297 format_buffer.clear();
298 format_buffer.append(PREAMBLE_FORMAT.begin(), PREAMBLE_FORMAT.end());
299 for (auto& it : ctx) {
300 // Detect if comma is in the context.
301 if (it == ',') {
302 delimiter_buffer.clear();
303 delimiter_buffer.append(COMMA_DELIMITER.begin(), COMMA_DELIMITER.end());
304 continue;
305 }
306
307 // Detect if the hash sign is in the context. This indicates no delimiter between entries.
308 if (it == '#') {
309 delimiter_buffer.clear();
310 continue;
311 }
312
313 format_buffer.push_back(it);
314
315 // Found the end of the context.
316 if (it == '}') {
317 return &it;
318 }
319 }
320
321 // No end of context was found.
322 return ctx.end();
323 }
324
325 template <typename FormatContext>
326 auto format(srsran::span<T> buf, FormatContext& ctx)
327 {
328 string_view format_str = string_view(format_buffer.data(), format_buffer.size());
329 string_view delimiter_str = string_view(delimiter_buffer.data(), delimiter_buffer.size());
330 return format_to(ctx.out(), format_str, fmt::join(buf.begin(), buf.end(), delimiter_str));
331 }
332};
