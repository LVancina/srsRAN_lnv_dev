36class rlc_pdu_recycler
37{
38public:
39 rlc_pdu_recycler(size_t rlc_window_size, rlc_bearer_logger& logger_) : logger(logger_)
40 {
41 for (std::vector<byte_buffer>& recycle_bin : recycle_bins) {
42 recycle_bin.reserve(rlc_window_size);
43 }
44 }
45
49 bool add_discarded_pdu(byte_buffer pdu)
50 {
51 if (recycle_bin_to_fill->size() >= recycle_bin_to_fill->capacity()) {
52 // recycle bin is full; delete the PDUs upon return
53 return false;
54 }
55 recycle_bin_to_fill->push_back(std::move(pdu));
56 return true;
57 }
58
60 void clear_by_executor(task_executor& executor)
61 {
62 // Swap recycle bins under a lock
63 {
64 std::lock_guard<std::mutex> recycle_bin_swap_lock(recycle_bin_swap_mutex);
65 std::swap(recycle_bin_to_fill, recycle_bin_to_swap);
66 }
67 // Redirect recycling of unused byte_buffers to ue_executor
68 auto handle_func = [this]() mutable { clear(); };
69 if (not executor.execute(std::move(handle_func))) {
70 logger.log_error("Failed to delegate deletion of PDUs to given executor.");
71 }
72 }
73
76 std::array<std::vector<byte_buffer>, 3>& get_recycle_bins() { return recycle_bins; }
77
78private:
80 void clear()
81 { // swap recycle bins under a lock
82 trace_point clear_tp = up_tracer.now();
83 {
84 std::lock_guard<std::mutex> lock(recycle_bin_swap_mutex);
85 std::swap(recycle_bin_to_swap, recycle_bin_to_dump);
86 }
87 // delete all PDUs to return their memory segments to the pool
88 recycle_bin_to_dump->clear();
89
90 up_tracer << trace_event{"rlc_clear_pdus", clear_tp};
91 }
92
93 rlc_bearer_logger& logger;
94
96 std::array<std::vector<byte_buffer>, 3> recycle_bins;
97
99 std::vector<byte_buffer>* recycle_bin_to_fill = &recycle_bins[0];
101 std::vector<byte_buffer>* recycle_bin_to_swap = &recycle_bins[1];
103 std::vector<byte_buffer>* recycle_bin_to_dump = &recycle_bins[2];
105 std::mutex recycle_bin_swap_mutex;
106};
