43enum coreset_id : uint8_t { MAX_CORESET_ID = 11, MAX_NOF_CORESETS = 12 };

45constexpr inline coreset_id to_coreset_id(unsigned cs_id)
46{
47 return static_cast<coreset_id>(cs_id);
48}

52struct coreset_configuration {
53 struct interleaved_mapping_type {
55 unsigned reg_bundle_sz;
57 unsigned interleaver_sz;
59 unsigned shift_index;
60
61 bool operator==(const interleaved_mapping_type& rhs) const
62 {
63 return reg_bundle_sz == rhs.reg_bundle_sz and interleaver_sz == rhs.interleaver_sz and
64 shift_index == rhs.shift_index;
65 }
66 };
67 enum class precoder_granularity_type { same_as_reg_bundle, all_contiguous_rbs };
68
69 coreset_id id;
71 unsigned duration;
72 optional<interleaved_mapping_type> interleaved;
73 precoder_granularity_type precoder_granurality;
75 optional<unsigned> pdcch_dmrs_scrambling_id;
76
77 void set_coreset0_crbs(crb_interval crbs)
78 {
79 srsran_assert(id == to_coreset_id(0), "Invalid access to CORESET#0 RBs for a coresetId>0");
80 coreset0_rbs = crbs;
81 }
82 const crb_interval& coreset0_crbs() const
83 {
84 srsran_assert(id == to_coreset_id(0), "Invalid access to CORESET#0 RBs for a coresetId>0");
85 return coreset0_rbs;
86 }
87
88 void set_freq_domain_resources(freq_resource_bitmap res_bitmap)
89 {
90 srsran_assert(id != to_coreset_id(0), "Invalid access to CORESET#0 RBs for a coresetId>0");
91 other_coreset_freq_resources = res_bitmap;
92 }
93 const freq_resource_bitmap& freq_domain_resources() const
94 {
95 srsran_assert(id != to_coreset_id(0), "Invalid access to CORESET#0 RBs for a coresetId>0");
96 return other_coreset_freq_resources;
97 }
98
99 bool operator==(const coreset_configuration& rhs) const
100 {
101 if (std::tie(id, duration, interleaved, precoder_granurality, pdcch_dmrs_scrambling_id) !=
102 std::tie(rhs.id, rhs.duration, rhs.interleaved, rhs.precoder_granurality, rhs.pdcch_dmrs_scrambling_id)) {
103 return false;
104 }
105 if (id == to_coreset_id(0)) {
106 return (coreset0_crbs() == rhs.coreset0_crbs());
107 }
108 return (freq_domain_resources() == rhs.freq_domain_resources());
109 }
110
113 unsigned get_nof_cces() const
114 {
115 // For \c frequencyDomainResources, TS 38.331, each 1 in the bitset corresponds to 6 RBs.
116 unsigned nof_rbs = id == to_coreset_id(0)
117 ? coreset0_rbs.length()
118 : (other_coreset_freq_resources.count() * pdcch_constants::NOF_RB_PER_FREQ_RESOURCE);
119 // A CCE corresponds to 6 REGs, where a REG is 1 RB x 1 symbol.
120 return (nof_rbs * duration) / pdcch_constants::NOF_REG_PER_CCE;
121 }
122
124 unsigned get_coreset_start_crb() const
125 {
126 static constexpr unsigned NOF_RBS_PER_GROUP = 6U;
127 if (id == to_coreset_id(0)) {
128 return coreset0_crbs().start();
129 }
130 const uint64_t lowest_bit = other_coreset_freq_resources.find_lowest(0, other_coreset_freq_resources.size());
131 return lowest_bit * NOF_RBS_PER_GROUP;
132 }
133
134private:
135 crb_interval coreset0_rbs;
136 freq_resource_bitmap other_coreset_freq_resources;
137};
