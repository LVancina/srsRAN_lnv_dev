67public:
69 explicit crc_calculator_lut_impl(crc_generator_poly poly_);

72 void reset(uint32_t crc_ = 0) { crc = crc_; }
73
75 void put_byte(uint8_t byte)
76 {
77 unsigned idx;
78 if (order > 8) {
79 // For more than 8 bits
80 unsigned ord = order - 8U;
81 idx = ((crc >> (ord)) & 0xffU) ^ byte;
82 } else {
83 // For 8 bits or less
84 unsigned ord = 8U - order;
85 idx = ((crc << (ord)) & 0xffU) ^ byte;
86 }
87
88 crc = (crc << 8U) ^ table[idx];
89 }
90
92 void reversecrcbit(unsigned nbits)
93 {
94 uint64_t m, rmask = 0x1;
95
96 for (m = 0; m < nbits; m++) {
97 if ((rmask & crc) == 0x01)
98 crc = (crc ^ table.polynom) >> 1;
99 else
100 crc = crc >> 1;
101 }
102 crc = (crc & table.crcmask);
103 }
104
106 crc_calculator_checksum_t get_checksum() const { return static_cast<crc_calculator_checksum_t>(crc & crcmask); }
