255 std::max(SCHEDULER_MAX_K1, SCHEDULER_MAX_K2 + MAX_MSG3_DELTA));

268

273 slot_point slot_tx() const { return last_slot_ind; }

280 const cell_slot_resource_allocator& operator[](unsigned slot_delay) const
281 {
282 assert_valid_sl(slot_delay);
283 slot_point sl_tx = last_slot_ind + slot_delay;
284 const cell_slot_resource_allocator& r = *slots[sl_tx.to_uint() % slots.size()];
285 srsran_assert(r.slot == sl_tx, "Bad access to uninitialized cell_resource_grid");
286 return r;
287 }
288 cell_slot_resource_allocator& operator[](unsigned slot_delay)
289 {
290 assert_valid_sl(slot_delay);
291 slot_point sl_tx = last_slot_ind + slot_delay;
292 cell_slot_resource_allocator& r = *slots[sl_tx.to_uint() % slots.size()];
293 srsran_assert(r.slot == sl_tx, "Bad access to uninitialized cell_resource_grid");
294 return r;
295 }
296
297 cell_slot_resource_allocator& operator[](slot_point slot) { return this->operator[](slot - last_slot_ind); }
298 const cell_slot_resource_allocator& operator[](slot_point slot) const
299 {
300 return this->operator[](slot - last_slot_ind);
301 }
302
304 const cell_slot_resource_allocator* get_history(slot_point slot) const
305 {
306 int diff = last_slot_ind - slot;
307 if (diff < 0 or diff >= static_cast<int>(RING_MAX_HISTORY_SIZE)) {
308 return nullptr;
309 }
310 const cell_slot_resource_allocator& r = *slots[slot.to_uint() % slots.size()];
311 srsran_assert(r.slot == slot, "Bad access to uninitialized cell_resource_grid");
312 return &r;
313 }
314
315private:
317 void assert_valid_sl(unsigned slot_delay) const
318 {
319 srsran_sanity_check(slot_delay <= max_ul_slot_alloc_delay,
320 "The cell resource pool is too small for accessing a slot with delay: {}",
321 slot_delay);
322 }
323
325 slot_point last_slot_ind;
326
328 std::vector<std::unique_ptr<cell_slot_resource_allocator>> slots;
329};
