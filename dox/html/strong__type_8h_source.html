35template <typename T, typename U>
36class strong_equality_with_type
37{
38public:
39 friend constexpr bool operator==(const T& lhs, const U& rhs) { return lhs.value() == rhs; }
40 friend constexpr bool operator==(const U& lhs, const T& rhs) { return lhs == rhs.value(); }
41 friend constexpr bool operator!=(const T& lhs, const U& rhs) { return lhs.value() != rhs; }
42 friend constexpr bool operator!=(const U& lhs, const T& rhs) { return lhs != rhs.value(); }
43};

46template <typename T, typename U>
47class strong_comparison_with_type
48{
49public:
50 friend constexpr bool operator<(const T& lhs, const U& rhs) { return lhs.value() < rhs; }
51 friend constexpr bool operator<(const U& lhs, const T& rhs) { return lhs < rhs.value(); }
52 friend constexpr bool operator>(const T& lhs, const U& rhs) { return lhs.value() > rhs; }
53 friend constexpr bool operator>(const U& lhs, const T& rhs) { return lhs > rhs.value(); }
54 friend constexpr bool operator<=(const T& lhs, const U& rhs) { return lhs.value() <= rhs; }
55 friend constexpr bool operator<=(const U& lhs, const T& rhs) { return lhs <= rhs.value(); }
56 friend constexpr bool operator>=(const T& lhs, const U& rhs) { return lhs.value() >= rhs; }
57 friend constexpr bool operator>=(const U& lhs, const T& rhs) { return lhs >= rhs.value(); }
58};

98template <typename T, typename Tag, typename... Properties>
99class strong_type : public detail::strong_property_t<Properties, strong_type<T, Tag, Properties...>>...
100{
101 static_assert(std::is_arithmetic<T>::value, "T should be an arithmetic type");
102 T val;
103
104public:
105 using value_type = T;
106 using tag_type = Tag;
107
109 constexpr strong_type() = default;
110 explicit constexpr strong_type(T val_) : val(val_) {}
111 constexpr strong_type(const strong_type&) = default;
112 constexpr strong_type& operator=(const strong_type&) = default;
113
115 constexpr T& value() { return val; }
116 constexpr const T& value() const { return val; }
117};
