212 static log_likelihood_ratio soft_xor(log_likelihood_ratio x, log_likelihood_ratio y)
213 {
214 int abs_x = std::abs(x.value);
215 int abs_y = std::abs(y.value);
216 if (x.value * y.value < 0) {
217 return -std::min(abs_x, abs_y);
218 }
219 return std::min(abs_x, abs_y);

293template <typename T, typename U, typename V>
294V log_likelihood_ratio::dot_prod(const T& x, const U& y, V init)
295{
296 static_assert(detail::is_llr_span_compatible<T>::value, "Template type is not compatible with a span of LLRs");
297 static_assert(srsvec::detail::is_arithmetic_span_compatible<U>::value,
298 "Template type is not compatible with a span of arithmetics");
299 srsran_assert(x.size() == y.size(), "Input spans must have identical sizes: '{}' vs '{}'", x.size(), y.size());
300 return std::inner_product(
301 x.begin(), x.end(), y.begin(), init, std::plus<V>(), [](log_likelihood_ratio a, log_likelihood_ratio b) {
302 return a.to_int() * b.to_int();
303 });
304}

315template <typename T>
316inline int log_likelihood_ratio::norm_squared(const T& x)
317{
318 static_assert(detail::is_llr_span_compatible<T>::value, "Template type is not compatible with a span of LLRs");
319 return std::accumulate(
320 x.begin(), x.end(), 0, [](int a, log_likelihood_ratio b) { return a + b.to_int() * b.to_int(); });
321}
