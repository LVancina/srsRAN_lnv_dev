31struct pusch_results_wrapper {
32 srsran::optional<srsran::pusch_processor_result_control> uci;
33 srsran::optional<srsran::pusch_processor_result_data> sch;
34};
35
36/// \brief Custom formatter for \c pusch_results_wrapper.
37template <>
38struct formatter<pusch_results_wrapper> {
39 /// Helper used to parse formatting options and format fields.
40 srsran::delimited_formatter helper;
41
42 /// Default constructor.
43 formatter() = default;
44
45 template <typename ParseContext>
46 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
47 {
48 return helper.parse(ctx);
49 }
50
51 template <typename FormatContext>
52 auto format(const pusch_results_wrapper& result, FormatContext& ctx) -> decltype(std::declval<FormatContext>().out())
53 {
54 // Format SCH message.
55 if (result.sch.has_value()) {
56 helper.format_always(ctx, result.sch.value());
57 }
58
59 // Format UCI message.
60 if (result.uci.has_value()) {
61 helper.format_always(ctx, result.uci.value());
62 }
63
64 // Format channel state information.
65 if (result.sch.has_value()) {
66 helper.format_always(ctx, result.sch.value().csi);
67 } else if (result.uci.has_value()) {
68 helper.format_always(ctx, result.uci.value().csi);
69 }
70
71 return ctx.out();
72 }
73};

79class logging_pusch_processor_decorator : public pusch_processor, private pusch_processor_result_notifier
80{
81public:
82 logging_pusch_processor_decorator(srslog::basic_logger& logger_, std::unique_ptr<pusch_processor> processor_) :
83 logger(logger_), processor(std::move(processor_))
84 {
85 srsran_assert(processor, "Invalid processor.");
86 }
87
88 void process(span<uint8_t> data_,
89 unique_rx_buffer rm_buffer,
90 pusch_processor_result_notifier& notifier_,
91 const resource_grid_reader& grid,
92 const pdu_t& pdu_) override
93 {
94 notifier = &notifier_;
95 data = data_;
96 pdu = pdu_;
97 time_start = std::chrono::steady_clock::now();
98 time_uci = std::chrono::time_point<std::chrono::steady_clock>();
99 time_return = std::chrono::time_point<std::chrono::steady_clock>();
100
101 // Clear processor results.
102 results.sch.reset();
103 results.uci.reset();
104
105 processor->process(data, std::move(rm_buffer), *this, grid, pdu);
106 time_return = std::chrono::steady_clock::now();
107 }
108
109private:
110 void on_uci(const pusch_processor_result_control& uci) override
111 {
112 srsran_assert(notifier, "Invalid notifier");
113 time_uci = std::chrono::steady_clock::now();
114 results.uci = uci;
115 notifier->on_uci(uci);
116 }
117
118 void on_sch(const pusch_processor_result_data& sch) override
119 {
120 srsran_assert(notifier, "Invalid notifier");
121
122 // Data size in bytes for printing hex dump only if SCH is present and CRC is passed.
123 unsigned data_size = 0;
124 if (results.sch.has_value() && results.sch->data.tb_crc_ok) {
125 data_size = data.size();
126 }
127
128 // Save SCH results.
129 results.sch = sch;
130
131 std::chrono::time_point<std::chrono::steady_clock> time_end = std::chrono::steady_clock::now();
132
133 // Calculate the UCI report latency if available.
134 std::chrono::nanoseconds time_uci_ns(0);
135 if (time_uci != std::chrono::time_point<std::chrono::steady_clock>()) {
136 time_uci_ns = time_uci - time_start;
137 }
138
139 // Calculate the return latency if available.
140 std::chrono::nanoseconds time_return_ns(0);
141 if (time_return != std::chrono::time_point<std::chrono::steady_clock>()) {
142 time_return_ns = time_return - time_start;
143 }
144
145 // Calculate the final time.
146 std::chrono::nanoseconds time_ns = time_end - time_start;
147
148 if (logger.debug.enabled()) {
149 // Detailed log information, including a list of all PDU fields.
150 logger.debug(pdu.slot.sfn(),
151 pdu.slot.slot_index(),
152 data.data(),
153 data_size,
154 "PUSCH: {:s} tbs={} {:s} {} uci_{} ret_{}\n {:n}\n {:n}",
155 pdu,
156 data.size(),
157 results,
158 time_ns,
159 time_uci_ns,
160 time_return_ns,
161 pdu,
162 results);
163 } else {
164 // Single line log entry.
165 logger.info(pdu.slot.sfn(),
166 pdu.slot.slot_index(),
167 data.data(),
168 data_size,
169 "PUSCH: {:s} tbs={} {:s} {} uci_{} ret_{}",
170 pdu,
171 data.size(),
172 results,
173 time_ns,
174 time_uci_ns,
175 time_return_ns);
176 }
177
178 // Exchanges the notifier before notifying the reception of SCH.
179 pusch_processor_result_notifier* notifier_ = nullptr;
180 std::exchange(notifier_, notifier);
181
182 // Notify the SCH reception.
183 notifier_->on_sch(sch);
184 }
185
186 srslog::basic_logger& logger;
187 std::unique_ptr<pusch_processor> processor;
188 span<uint8_t> data;
189 pdu_t pdu;
190 pusch_processor_result_notifier* notifier;
191 std::chrono::time_point<std::chrono::steady_clock> time_start;
192 std::chrono::time_point<std::chrono::steady_clock> time_uci;
193 std::chrono::time_point<std::chrono::steady_clock> time_return;
194 fmt::pusch_results_wrapper results;
195};
