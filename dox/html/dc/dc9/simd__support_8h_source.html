36/// Number of bytes in an AVX2 register.
37constexpr unsigned AVX2_SIZE_BYTE = 32;
38/// Number of bytes in an AVX512 register.
39constexpr unsigned AVX512_SIZE_BYTE = 64;
40/// Number of bytes in a NEON register.
41constexpr unsigned NEON_SIZE_BYTE = 16;

45/// \brief Mimics a span of SIMD registers.
46/// \tparam simdWrapper Wrapper for an SIMD vector type.
47/// \tparam storageType Underlying data type of the \c simd_span (either \c int8_t or \c const \c int8_t).
48template <typename simdWrapper, typename storageType>
49class simd_span
50{
51 static_assert(std::is_same<typename std::remove_cv<storageType>::type, int8_t>::value,
52 "simd_span can only have (const) int8_t as underlying storage type.");
53
54 using pointer = storageType*;
55
56 /// Helper type for method overloading.
57 template <typename HelpType>
58 struct help_type {
59 // Intentionally empty.
60 };
61
62public:
63 using simdType = typename simdWrapper::simdType;
64 static constexpr size_t SIMD_SIZE_BYTE = sizeof(simdType);
65
66 /// Default constructor: empty view to a null pointer.
67 simd_span() : array_ptr(nullptr), view_length(0) {}
68
69 /// \brief Constructs a span from a standard array.
70 ///
71 /// \tparam IntType Type of the standard array - must be an integer type of 8 bits.
72 /// \tparam N Number of elements of the standard array.
73 /// \param[in] arr Array the span is a view of.
74 /// \param[in] offset Offset from the beginning of the array (as a number of SIMD registers).
75 /// \param[in] length Number of elements spanned by the view (as a number of SIMD registers).
76 template <typename IntType, size_t N>
77 simd_span(std::array<IntType, N>& arr, size_t offset, size_t length) :
78 array_ptr(reinterpret_cast<pointer>(arr.data()) + offset * SIMD_SIZE_BYTE), view_length(length)
79 {
80 static_assert(sizeof(IntType) == sizeof(storageType),
81 "simd_span can only be created from arrays of 1-byte integer types.");
82 srsran_assert((offset + view_length) * SIMD_SIZE_BYTE <= N, "Cannot take a span longer than the array.");
83 }
84
85 /// \brief Constructs a span from a standard array.
86 ///
87 /// \tparam IntType Type of the standard array - must be an integer type of 8 bits.
88 /// \tparam N Number of elements of the standard array.
89 /// \param[in] arr Array the span is a view of.
90 /// \param[in] length Number of elements spanned by the view (as a number of SIMD registers).
91 template <typename IntType, size_t N>
92 simd_span(std::array<IntType, N>& arr, size_t length) : simd_span(arr, 0, length)
93 {
94 }
95
96 /// \brief Constructs an \c simd_span from a \c srsran::span.
97 ///
98 /// \tparam IntType Type of the span - must be an integer type of 8 bits.
99 /// \param[in] sp Original span.
100 /// \param[in] length Number of elements viewed by the \c simd_span (as a number of SIMD registers).
101 template <typename IntType>
102 simd_span(span<IntType> sp, size_t length) : array_ptr(reinterpret_cast<pointer>(sp.data())), view_length(length)
103 {
104 static_assert(sizeof(IntType) == sizeof(storageType),
105 "simd_span can only be created from arrays of 1-byte integer types.");
106 srsran_assert(view_length * SIMD_SIZE_BYTE <= sp.size(),
107 "Cannot create an simd_span longer than the original span.");
108 }
109
110 /// Returns a pointer to the \c pos SIMD register inside the array.
111 simdType* data_at(unsigned pos)
112 {
113 srsran_assert(pos < view_length, "Index {} out of bound.", pos);
114 return reinterpret_cast<simdType*>(array_ptr) + pos;
115 }
116
117 /// Returns a read-only pointer to the \c pos SIMD register inside the array.
118 const simdType* data_at(unsigned pos) const
119 {
120 srsran_assert(pos < view_length, "Index {} out of bound.", pos);
121 return reinterpret_cast<const simdType*>(array_ptr) + pos;
122 }
123
124 /// Returns a pointer to the byte at position <tt>pos * SIMD_SIZE_BYTE + byte</tt> inside the array.
125 pointer data_at(unsigned pos, unsigned byte)
126 {
127 unsigned index = pos * SIMD_SIZE_BYTE + byte;
128 srsran_assert(index < view_length * SIMD_SIZE_BYTE, "Index ({}, {}) out of bound.", pos, byte);
129 return (array_ptr + index);
130 }
131
132 /// Returns a read-only pointer to the \c pos SIMD register inside the array.
133 const pointer data_at(unsigned pos, unsigned byte) const
134 {
135 unsigned index = pos * SIMD_SIZE_BYTE + byte;
136 srsran_assert(index < view_length * SIMD_SIZE_BYTE, "Index ({}, {}) out of bound.", pos, byte);
137 return (array_ptr + index);
138 }
139
140 // Unfortunately, we can't work with the array subscript operator [] since there seems to be no easy way to access a
141 // simdType object by reference.
142
143 /// Sets the \c pos AVX2 register to \c val.
144 void set_at(unsigned pos, simd256_type val);
145
146 /// Sets the \c pos AVX512 register to \c val.
147 void set_at(unsigned pos, simd512_type val);
148
149 /// Sets the \c pos NEON register to \c val.
150 void set_at(unsigned pos, simd128_type val);
151
152 /// Gets the value stored in the \c pos SIMD register.
153 auto get_at(unsigned pos) const { return get_at(help_type<simdWrapper>(), pos); }
154
155 /// Returns the number of SIMD registers viewed by the span.
156 size_t size() const { return view_length; }
157
158 /// Returns a view to bytes from the specified offset.
159 ///
160 /// \param[in] offset Starting point of the view (as a number of SIMD registers).
161 /// \param[in] length Length of the view (as a number of bytes).
162 span<storageType> plain_span(size_t offset, size_t length) const
163 {
164 unsigned offset_byte = offset * SIMD_SIZE_BYTE;
165 srsran_assert(offset_byte + length <= view_length * SIMD_SIZE_BYTE, "Size out of bound.");
166 return span<storageType>(array_ptr + offset_byte, length);
167 }
168
169private:
170 /// Pointer to the first element viewed by the span.
171 pointer array_ptr;
172 /// Number of elements viewed by the span.
173 size_t view_length;
174
175 /// Overload of the get method for AVX2.
176 simd256_type get_at(help_type<simd256_wrapper> /**/, unsigned pos) const;
177
178 /// Overload of the get method for AVX512.
179 simd512_type get_at(help_type<simd512_wrapper> /**/, unsigned pos) const;
180
181 /// Overload of the get method for NEON.
182 simd128_type get_at(help_type<simd128_wrapper> /**/, unsigned pos) const;
183};
