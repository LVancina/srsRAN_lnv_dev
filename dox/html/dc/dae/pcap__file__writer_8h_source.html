35/// This structure gets written to the start of the file
36struct pcap_hdr_t {
37 unsigned int magic_number; /// magic number
38 unsigned short version_major; /// major version number
39 unsigned short version_minor; /// minor version number
40 unsigned int thiszone; /// GMT to local correction
41 unsigned int sigfigs; /// accuracy of timestamps
42 unsigned int snaplen; /// max length of captured packets, in octets
43 unsigned int network; /// data link type
44};

46/// This structure precedes each packet
47struct pcaprec_hdr_t {
48 unsigned int ts_sec; /// timestamp seconds
49 unsigned int ts_usec; /// timestamp microseconds
50 unsigned int incl_len; /// number of octets of packet saved in file
51 unsigned int orig_len; /// actual length of packet
52};
