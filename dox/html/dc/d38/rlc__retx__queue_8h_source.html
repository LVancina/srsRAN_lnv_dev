33struct rlc_tx_amd_retx {
34 uint32_t sn = INVALID_RLC_SN; ///< sequence number
35 uint32_t so = 0; ///< offset to first byte of this ReTx within the associated SDU
36 uint32_t length = 0; ///< length of this ReTx
37
38 /// \brief Implements a check whether the range of this retransmission object includes the given segment range
39 /// \param segment_offset The segment offset to check
40 /// \param segment_length Length of the segment to check
41 /// \return true if the segment offset is covered by the retransmission object. Otherwise false
42 bool includes(uint32_t segment_offset, uint32_t segment_length) const
43 {
44 return (segment_offset >= so) && (segment_offset + segment_length <= so + length);
45 }
46};
47
48class rlc_retx_queue_state
49{
50 uint32_t retx_bytes = 0; ///< Total amount of queued PDU payload bytes across all queued ReTx
51 uint32_t n_retx_so_zero = 0; ///< Number of queued ReTx with zero segment offset, i.e. need small PDU header
52 uint32_t n_retx_so_nonzero = 0; ///< Number of queued ReTx with non-zero segment offset, i.e. need large PDU header
53
54public:
55 uint32_t get_retx_bytes() const { return retx_bytes; }
56 uint32_t get_n_retx_so_zero() const { return n_retx_so_zero; }
57 uint32_t get_n_retx_so_nonzero() const { return n_retx_so_nonzero; }
58
59 void add(const rlc_tx_amd_retx& elem)
60 {
61 retx_bytes += elem.length;
62 if (elem.so == 0) {
63 n_retx_so_zero++;
64 } else {
65 n_retx_so_nonzero++;
66 }
67 }
68
69 void subtract(const rlc_tx_amd_retx& elem)
70 {
71 if (retx_bytes >= elem.length) {
72 retx_bytes -= elem.length;
73 } else {
74 retx_bytes = 0;
75 }
76 if (elem.so == 0) {
77 if (n_retx_so_zero > 0) {
78 n_retx_so_zero--;
79 }
80 } else {
81 if (n_retx_so_nonzero > 0) {
82 n_retx_so_nonzero--;
83 }
84 }
85 }
86};

88struct rlc_retx_queue_item {
89 bool invalid = false; ///< Marks element as virtually removed "zombie" which can be skipped on pop
90 rlc_tx_amd_retx retx = {}; ///< The actual ReTx info
91};

93class rlc_retx_queue
94{
95 ring_buffer<rlc_retx_queue_item> queue;
96 rlc_retx_queue_state st;
97
98public:
99 rlc_retx_queue(size_t capacity) : queue(capacity) {}
100 ~rlc_retx_queue() = default;
101
102 /// \brief Tries to push a ReTx to the queue.
103 /// \param retx The ReTx to be added.
104 /// \return True if the ReTx was added successfully; False if the queue was full.
105 bool try_push(const rlc_tx_amd_retx& retx)
106 {
107 rlc_retx_queue_item elem = {};
108 elem.retx = retx;
109 bool success = queue.try_push(elem);
110 if (success) {
111 st.add(retx);
112 }
113 return success;
114 }
115
116 /// \brief Removes the first element from the front together with any following invalid elements such that front is
117 /// either a valid element or the queue is empty.
118 void pop()
119 {
120 if (not queue.empty()) {
121 st.subtract(queue.top().retx);
122 queue.pop();
123 clean_invalid_front();
124 }
125 }
126
127 /// \brief Access the first valid element of the queue unless the queue is empty.
128 /// \return First valid element in the queue
129 const rlc_tx_amd_retx& front() const
130 {
131 srsran_assert(not queue.empty(), "Cannot return front element of empty queue.");
132 srsran_assert(not queue.top().invalid, "Cannot return front element. The front is invalid.");
133 return queue.top().retx;
134 }
135
136 /// \brief Replaces the first valid element of the queue.
137 /// \param retx The ReTx to be written.
138 void replace_front(const rlc_tx_amd_retx& retx)
139 {
140 srsran_assert(not queue.empty(), "Cannot replace front element of empty queue.");
141 srsran_assert(not queue.top().invalid, "Cannot replace front element. The front is invalid.");
142 st.subtract(queue.top().retx);
143 st.add(retx);
144 rlc_retx_queue_item elem = {};
145 elem.retx = retx;
146 queue.top() = elem;
147 }
148
149 /// \brief Clear the container and reset state to the initial state.
150 void clear()
151 {
152 st = {};
153 queue.clear();
154 }
155
156 /// \brief Total number of elements (valid + invalid) in the queue.
157 /// \return Number of any elements in the queue.
158 size_t size() const { return queue.size(); }
159
160 /// \brief Checks the presence of at least one valid element.
161 /// \return True if no valid element is queued; False if at least one valid element is queued.
162 bool empty() const { return queue.empty(); }
163
164 /// \brief Access to the internal state of the queue.
165 /// \return The internal state of the queue.
166 rlc_retx_queue_state state() const { return st; }
167
168 /// Checks if the queue contains an element with a given SN.
169 /// \param sn The sequence number to look up.
170 /// \return true if queue contains the SN, false otherwise.
171 bool has_sn(uint32_t sn) const
172 {
173 if (queue.empty()) {
174 return false;
175 }
176 for (auto elem : queue) {
177 if (elem.invalid == false && elem.retx.sn == sn) {
178 return true;
179 }
180 }
181 return false;
182 };
183
184 /// Checks if the queue contains an element with a given SN and also includes a specified segment interval given by a
185 /// segment offset and segment length.
186 /// \param sn The sequence number to look up.
187 /// \param so The segment offset to check.
188 /// \param length The segment length to check.
189 /// \return true if queue contains the SN and the element includes the given segment interval, false otherwise.
190 bool has_sn(uint32_t sn, uint32_t so, uint32_t length) const
191 {
192 if (queue.empty()) {
193 return false;
194 }
195 for (auto elem : queue) {
196 if (elem.invalid == false && elem.retx.sn == sn) {
197 if (elem.retx.includes(so, length)) {
198 return true;
199 }
200 }
201 }
202 return false;
203 };
204
205 /// \brief Removes all elements with a given SN from queue and removes any invalid elements at the front.
206 ///
207 /// The ReTx for matching SN will be marked as invalid and its content will be subtracted from the state.
208 /// Subsequently, any invalid elements are removed from the front.
209 ///
210 /// \param sn sequence number to be removed from queue.
211 /// \return true if at least one element was removed, false if no element to remove was found.
212 bool remove_sn(uint32_t sn)
213 {
214 if (queue.empty()) {
215 return false;
216 }
217 auto iter = queue.begin();
218 bool result = false;
219 while (iter != queue.end()) {
220 if (iter->invalid == false && iter->retx.sn == sn) {
221 st.subtract(iter->retx);
222 iter->invalid = true;
223 result = true;
224 } else {
225 ++iter;
226 }
227 }
228 clean_invalid_front();
229 return result;
230 }
231
232 /// \brief Advances the front of the queue to the first valid element or to the end of the queue.
233 void clean_invalid_front()
234 {
235 while (!queue.empty()) {
236 if (queue.top().invalid == true) {
237 // remove invalid element
238 queue.pop();
239 } else {
240 // stop at valid element
241 return;
242 }
243 }
244 }
245};
