35/// Type of AVX2 intrinsic calculating min or max values between two vectors.
36using _mm256_comparator = __m256i (*)(__m256i a, __m256i b);

38/// \brief Finds the maximum (or minimum) 16bit sample in each of the two input resource blocks, returns its absolute
39/// value.
40///
41/// Finds absolute maximum (or minimum, depending on the comparator passed as the template argument) values
42/// for each of the two RBs passed in a three AVX2 registers.
43///
44/// Registers are evenly divided between two RBs so that IQ samples of each RB occupy one 128bit lane.
45/// The following diagram shows the input format. Here RBx stands for one unique RE (a pair of IQ samples, 32 bits long)
46/// pertaining to a respective RB:
47/// | | | | | |
48/// | ----- | ------- | ------- | ------- | ------- |
49/// | \c v0_epi16: | RB0 RB0 | RB0 RB0 | RB1 RB1 | RB1 RB1 |
50/// | \c v1_epi16: | RB0 RB0 | RB0 RB0 | RB1 RB1 | RB1 RB1 |
51/// | \c v2_epi16: | RB0 RB0 | RB0 RB0 | RB1 RB1 | RB1 RB1 |
52///
53/// The function first performs vertical min/max search across the three registers (comparing 16bit samples), then
54/// horizontal min/max across elements in each 128bit lane of the resulting vector. Finally, the function takes the
55/// absolute value of the two 16bit samples computed thus far and stores them in the output span.
56///
57/// \tparam COMPARE AVX2 intrinsic used for comparison, either \c mm256_min_epi16 or \c _mm256_max_epi16.
58/// \param[out] abs A span of two absolute values of max (or min) samples in each of the two input RBs.
59/// \param[in] v0_epi16 AVX2 register storing first four IQ pairs of each of the two RB.
60/// \param[in] v1_epi16 AVX2 register storing second four IQ pairs of each of the two RB.
61/// \param[in] v2_epi16 AVX2 register storing last four IQ pairs of each of the two RB.
62template <_mm256_comparator COMPARE>
63inline void find_rbs_abs_min_max_values(span<uint16_t> abs, __m256i v0_epi16, __m256i v1_epi16, __m256i v2_epi16)
64{
65 // Find vertical min/max across every 3 16bit values in each resource block.
66 __m256i cmp_8val_epi16 = COMPARE(COMPARE(v0_epi16, v1_epi16), v2_epi16);
67 // Now 16bit vector elements look as follows:
68 // [max(I00, I08, I04), max(Q00, Q08, Q04), max(I01, I09, I05), max(Q01, Q09, Q05), ...].
69
70 // Swap 64bit words inside 128bit lanes and find 4 min/max values.
71 __m256i cmp_8val_swp_epi16 = _mm256_shuffle_epi32(cmp_8val_epi16, 0x4e);
72 __m256i cmp_4val_epi16 = COMPARE(cmp_8val_epi16, cmp_8val_swp_epi16);
73 // Swap 32bit words and find 2 min/max values.
74 __m256i cmp_4val_swp_epi16 = _mm256_shuffle_epi32(cmp_4val_epi16, 0x1b);
75 __m256i cmp_2val_epi16 = COMPARE(cmp_4val_epi16, cmp_4val_swp_epi16);
76 // Swap 16bit words and find final min/max values for each RB.
77 const __m256i shuffle_mask = _mm256_setr_epi8(
78 2, 3, 0, 1, 6, 7, 4, 5, 8, 9, 10, 11, 14, 15, 12, 13, 2, 3, 0, 1, 6, 7, 4, 5, 8, 9, 10, 11, 14, 15, 12, 13);
79 __m256i cmp_2val_swp_epi16 = _mm256_shuffle_epi8(cmp_2val_epi16, shuffle_mask);
80 __m256i cmp_rb0_rb1_epi16 = COMPARE(cmp_2val_epi16, cmp_2val_swp_epi16);
81 __m256i cmp_rb0_rb1_epu16 = _mm256_abs_epi16(cmp_rb0_rb1_epi16);
82
83 // Extract absolute minimum/maximum values for each RB.
84 abs[0] = _mm256_extract_epi16(cmp_rb0_rb1_epu16, 0);
85 abs[1] = _mm256_extract_epi16(cmp_rb0_rb1_epu16, 8);
86}

88/// \brief Finds maximum absolute value across 16bit IQ samples in each of the two input resource blocks passed in three
89/// AVX2 registers.
90///
91/// The content of the three input AVX2 registers is represented in the table below. Here RBx means one unique RE
92/// (pair of IQ samples, 32 bits long) pertaining to a respective RB:
93/// | | | | | |
94/// | ----- | ------- | ------- | ------- | ------- |
95/// | \c rb0_epi16: | RB0 RB0 | RB0 RB0 | RB0 RB0 | RB0 RB0 |
96/// | \c rb01_epi16: | RB0 RB0 | RB0 RB0 | RB1 RB1 | RB1 RB1 |
97/// | \c rb1_epi16: | RB1 RB1 | RB1 RB1 | RB1 RB1 | RB1 RB1 |
98///
99/// \param[out] max_abs A span of two maximum absolute values in the two input RBs.
100/// \param[in] rb0_epi16 AVX2 register storing 16bit IQ pairs of the first RB.
101/// \param[in] rb01_epi16 AVX2 register storing 16bit IQ pairs of the first and second RBs.
102/// \param[in] rb1_epi16 AVX2 register storing 16bit IQ pairs of the second RB.
103inline void calculate_max_abs(span<unsigned> max_abs, __m256i rb0_epi16, __m256i rb01_epi16, __m256i rb1_epi16)
104{
105 std::array<uint16_t, 2> abs_min_values;
106 std::array<uint16_t, 2> abs_max_values;
107
108 // Reorganize vectors to be able to vertically compare 16bit samples pertaining to the same resource block:
109 // [ RB0 RB0 | RB0 RB0 | RB1 RB1 | RB1 RB1 ]
110 // [ RB0 RB0 | RB0 RB0 | RB1 RB1 | RB1 RB1 ]
111 // [ RB0 RB0 | RB0 RB0 | RB1 RB1 | RB1 RB1 ]
112 __m256i v0_epi16 = _mm256_permute2f128_si256(rb0_epi16, rb1_epi16, 0x20);
113 __m256i v2_epi16 = _mm256_permute2f128_si256(rb0_epi16, rb1_epi16, 0x31);
114 __m256i v1_epi16 = rb01_epi16;
115
116 find_rbs_abs_min_max_values<_mm256_min_epi16>(abs_min_values, v0_epi16, v1_epi16, v2_epi16);
117 find_rbs_abs_min_max_values<_mm256_max_epi16>(abs_max_values, v0_epi16, v1_epi16, v2_epi16);
118
119 max_abs[0] = std::max<unsigned>(abs_max_values[0], abs_min_values[0] - 1U);
120 max_abs[1] = std::max<unsigned>(abs_max_values[1], abs_min_values[1] - 1U);
121}
