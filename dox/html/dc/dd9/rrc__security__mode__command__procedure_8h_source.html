76/// \brief Fills ASN.1 RRC Security Mode Command struct.
77/// \param[out] rrc_smc The RRC security mode command ASN.1 struct to fill.
78/// \param[in] int_algo The selected integrity protection algorithm.
79/// \param[in] ciph_algo The selected ciphering algorithm.
80/// \param[in] rrc_transaction_id The RRC transaction id.
81inline void fill_asn1_rrc_smc_msg(asn1::rrc_nr::security_mode_cmd_s& rrc_smc,
82 const security::integrity_algorithm& int_algo,
83 const security::ciphering_algorithm& ciph_algo,
84 uint8_t rrc_transaction_id)
85{
86 using namespace asn1::rrc_nr;
87 security_mode_cmd_ies_s& smc_ies = rrc_smc.crit_exts.set_security_mode_cmd();
88 rrc_smc.rrc_transaction_id = rrc_transaction_id;
89
90 // Set security algorithms
91 security_cfg_smc_s& security_cfg_smc = smc_ies.security_cfg_smc;
92 security_algorithm_cfg_s& security_algorithm_cfg = security_cfg_smc.security_algorithm_cfg;
93
94 security_algorithm_cfg.integrity_prot_algorithm_present = true;
95 switch (int_algo) {
96 case security::integrity_algorithm::nia0:
97 security_algorithm_cfg.integrity_prot_algorithm = integrity_prot_algorithm_e::nia0;
98 break;
99 case security::integrity_algorithm::nia1:
100 security_algorithm_cfg.integrity_prot_algorithm = integrity_prot_algorithm_e::nia1;
101 break;
102 case security::integrity_algorithm::nia2:
103 security_algorithm_cfg.integrity_prot_algorithm = integrity_prot_algorithm_e::nia2;
104 break;
105 case security::integrity_algorithm::nia3:
106 security_algorithm_cfg.integrity_prot_algorithm = integrity_prot_algorithm_e::nia3;
107 break;
108 }
109 switch (ciph_algo) {
110 case security::ciphering_algorithm::nea0:
111 security_algorithm_cfg.ciphering_algorithm = ciphering_algorithm_e::nea0;
112 break;
113 case security::ciphering_algorithm::nea1:
114 security_algorithm_cfg.ciphering_algorithm = ciphering_algorithm_e::nea1;
115 break;
116 case security::ciphering_algorithm::nea2:
117 security_algorithm_cfg.ciphering_algorithm = ciphering_algorithm_e::nea2;
118 break;
119 case security::ciphering_algorithm::nea3:
120 security_algorithm_cfg.ciphering_algorithm = ciphering_algorithm_e::nea3;
121 break;
122 }
123}
