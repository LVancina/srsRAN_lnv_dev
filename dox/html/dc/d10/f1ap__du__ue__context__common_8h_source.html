45/// \brief Creates a \c f1ap_drb_to_setup from ASN.1 type
46///
47/// This function is used by the following procedures
48/// - f1ap_du_ue_context_setup_procedure
49/// - f1ap_du_ue_context_modification_procedure
50///
51/// \param drb_item ASN.1 item with DRB-specific parameters to be setup
52/// \return f1ap_drb_to_setup object
53template <typename Asn1Type>
54f1ap_drb_to_setup make_drb_to_setup(const Asn1Type& drb_item)
55{
56 f1ap_drb_to_setup drb_obj;
57 drb_obj.drb_id = (drb_id_t)drb_item.drb_id;
58 drb_obj.mode = (drb_rlc_mode)(unsigned)drb_item.rlc_mode;
59 drb_obj.uluptnl_info_list.resize(drb_item.ul_up_tnl_info_to_be_setup_list.size());
60 for (unsigned j = 0; j != drb_obj.uluptnl_info_list.size(); ++j) {
61 drb_obj.uluptnl_info_list[j] =
62 asn1_to_up_transport_layer_info(drb_item.ul_up_tnl_info_to_be_setup_list[j].ul_up_tnl_info);
63 }
64 drb_obj.five_qi = uint_to_five_qi(
65 drb_item.qos_info.choice_ext().value().drb_info().drb_qos.qos_characteristics.non_dyn_5qi().five_qi);
66 return drb_obj;
67}
