61/// Interface to trigger bind/listen/connect operations on gateway socket.
62class sctp_network_gateway_controller : public network_gateway_controller
63{
64public:

66
67 /// \brief Create and connect socket to given address.

71 virtual bool recreate_and_reconnect() = 0;
72
73 /// \brief Start listening on socket.
74 virtual bool listen() = 0;
75
76 /// \brief Return the port on which the socket is listening.
