34constexpr unsigned gtpu_sn_mod = 65536;
35constexpr unsigned gtpu_rx_window_size = 32768;

37/// GTP-U RX state variables
38struct gtpu_rx_state {
39 /// RX_NEXT indicates the SN value of the next GTP-U SDU expected to be received.
40 uint16_t rx_next;
41 /// RX_DELIV indicates the SN value of the first GTP-U SDU not delivered to the lower layers, but still
42 /// waited for.
43 uint16_t rx_deliv;
44 /// RX_REORD indicates the SN value following the SN value associated with the GTP-U PDU which
45 /// triggered t-Reordering.
46 uint16_t rx_reord;
47};
48
49struct gtpu_rx_sdu_info {
50 byte_buffer sdu = {};
51 qos_flow_id_t qos_flow_id = qos_flow_id_t::invalid;
52 optional<uint16_t> sn = {};
53};

55/// Class used for receiving GTP-U bearers.
56class gtpu_tunnel_ngu_rx : public gtpu_tunnel_base_rx

86protected:
87 // domain-specific PDU handler
88 void handle_pdu(gtpu_dissected_pdu&& pdu, const sockaddr_storage& src_addr) final
89 {
90 gtpu_teid_t teid = pdu.hdr.teid;
91 psup_dl_pdu_session_information pdu_session_info = {};
92 bool have_pdu_session_info = false;
93 for (auto ext_hdr : pdu.hdr.ext_list) {
94 switch (ext_hdr.extension_header_type) {
95 case gtpu_extension_header_type::pdu_session_container:
96 if (!have_pdu_session_info) {
97 have_pdu_session_info = psup_packer.unpack(pdu_session_info, ext_hdr.container);
98 if (!have_pdu_session_info) {
99 logger.log_error("Failed to unpack PDU session container.");
100 }
101 } else {
102 logger.log_warning("Ignoring multiple PDU session container.");
103 }
104 break;
105 default:
106 logger.log_warning("Ignoring unexpected extension header at NG-U interface. type={}",
107 ext_hdr.extension_header_type);
108 }
109 }
110 if (!have_pdu_session_info) {
111 logger.log_warning(
112 "Incomplete PDU at NG-U interface: missing or invalid PDU session container. pdu_len={} teid={}",
113 pdu.buf.length(),
114 teid);
115 // As per TS 29.281 Sec. 5.2.2.7 the (...) PDU Session Container (...) shall be transmitted in a G-PDU over the
116 // N3 and N9 user plane interfaces (...).
117 return;
118 }
119
120 logger.log_debug(pdu.buf.begin(), pdu.buf.end(), "RX PDU. sdu_len={} {}", pdu.buf.length(), st);
121
122 if (!pdu.hdr.flags.seq_number || config.t_reordering.count() == 0) {
123 // Forward this SDU straight away.
124 byte_buffer rx_sdu = gtpu_extract_t_pdu(std::move(pdu)); // header is invalidated after extraction
125 gtpu_rx_sdu_info rx_sdu_info = {std::move(rx_sdu), pdu_session_info.qos_flow_id};
126 deliver_sdu(rx_sdu_info);
127 return;
128 }
129
130 uint16_t sn = pdu.hdr.seq_number;
131 byte_buffer rx_sdu = gtpu_extract_t_pdu(std::move(pdu)); // header is invalidated after extraction
132
133 // Check out-of-window
134 if (!inside_rx_window(sn)) {
135 logger.log_warning("SN falls out of Rx window. sn={} {}", sn, st);
136 gtpu_rx_sdu_info rx_sdu_info = {std::move(rx_sdu), pdu_session_info.qos_flow_id, sn};
137 deliver_sdu(rx_sdu_info);
138 return;
139 }
140
141 // Check late SN
142 if (rx_mod_base(sn) < rx_mod_base(st.rx_deliv)) {
143 logger.log_debug("Out-of-order after timeout or duplicate. sn={} {}", sn, st);
144 gtpu_rx_sdu_info rx_sdu_info = {std::move(rx_sdu), pdu_session_info.qos_flow_id, sn};
145 deliver_sdu(rx_sdu_info);
146 return;
147 }
148
149 // Check if PDU has been received
150 if (rx_window->has_sn(sn)) {
151 logger.log_warning("Duplicate PDU dropped. sn={}", sn);
152 return;
153 }
154
155 gtpu_rx_sdu_info& rx_sdu_info = rx_window->add_sn(sn);
156 rx_sdu_info.sdu = std::move(rx_sdu);
157 rx_sdu_info.qos_flow_id = pdu_session_info.qos_flow_id;
158 rx_sdu_info.sn = sn;
159
160 // Update RX_NEXT
161 if (rx_mod_base(sn) >= rx_mod_base(st.rx_next)) {
162 st.rx_next = sn + 1;
163 }
164
165 if (rx_mod_base(sn) == rx_mod_base(st.rx_deliv)) {
166 // Deliver all consecutive SDUs in ascending order of associated SN
167 deliver_all_consecutive_sdus();
168 }
169
170 // Stop re-ordering timer if we advanced the window past RX_REORD
171 if (reordering_timer.is_running() and (not inside_rx_window(st.rx_reord))) {
172 reordering_timer.stop();
173 logger.log_debug("Stopped t-Reordering. {}", st);
174 }
175
176 if (config.t_reordering.count() == 0) {
177 st.rx_reord = st.rx_next;
178 handle_t_reordering_expire();
179 } else if (!reordering_timer.is_running() and rx_mod_base(st.rx_deliv) < rx_mod_base(st.rx_next)) {
180 st.rx_reord = st.rx_next;
181 reordering_timer.run();
182 logger.log_debug("Started t-Reordering. {}", st);
183 }
184 }
185
186 void deliver_sdu(gtpu_rx_sdu_info& sdu_info)
187 {
188 logger.log_info(sdu_info.sdu.begin(),
189 sdu_info.sdu.end(),
190 "RX SDU. sdu_len={} qos_flow={} sn={}",
191 sdu_info.sdu.length(),
192 sdu_info.qos_flow_id,
193 sdu_info.sn);
194 lower_dn.on_new_sdu(std::move(sdu_info.sdu), sdu_info.qos_flow_id);
195 }
196
197 void deliver_all_consecutive_sdus()
198 {
199 while (st.rx_deliv != st.rx_next && rx_window->has_sn(st.rx_deliv)) {
200 gtpu_rx_sdu_info& sdu_info = (*rx_window)[st.rx_deliv];
201 deliver_sdu(sdu_info);
202 rx_window->remove_sn(st.rx_deliv);
203
204 // Update RX_DELIV
205 st.rx_deliv = st.rx_deliv + 1;
206 }
207 }
208
209 void handle_t_reordering_expire()
210 {
211 // Check if timer has been restarted by the PDU handling routine between expiration and execution of this handler.
212 if (reordering_timer.is_running()) {
213 logger.log_info("reordering timer has been already restarted. Skipping outdated event. {}", st);
214 return;
215 }
216 if (not inside_rx_window(st.rx_reord)) {
217 logger.log_info("rx_reord is outside RX window. Skipping outdated event. {}", st);
218 return;
219 }
220
221 while (st.rx_deliv != st.rx_reord) {
222 if (rx_window->has_sn(st.rx_deliv)) {
223 gtpu_rx_sdu_info& sdu_info = (*rx_window)[st.rx_deliv];
224 deliver_sdu(sdu_info);
225 rx_window->remove_sn(st.rx_deliv);
226 }
227
228 // Update RX_DELIV
229 st.rx_deliv = st.rx_deliv + 1;
230 }
231
232 deliver_all_consecutive_sdus();
233
234 if (rx_mod_base(st.rx_deliv) < rx_mod_base(st.rx_next)) {
235 if (config.t_reordering.count() == 0) {
236 logger.log_error("reordering timer expired after 0ms and rx_deliv < rx_next. {}", st);
237 return;
238 }
239 logger.log_debug("updating rx_reord to rx_next. {}", st);
240 st.rx_reord = st.rx_next;
241 reordering_timer.run();
242 }
243 }

279 /// \brief Helper function for arithmetic comparisons of state variables or SN values
280 ///
281 /// When performing arithmetic comparisons of state variables or SN values, a modulus base shall be used.
282 /// This is adapted from RLC AM, TS 38.322 Sec. 7.1
283 ///
284 /// \param sn The sequence number to be rebased from RX_Deliv, as this is the lower-edge of the window.
285 /// \return The rebased value of sn
286 constexpr uint16_t rx_mod_base(uint16_t sn) const { return (sn - st.rx_deliv) % gtpu_sn_mod; }
287
288 /// Checks whether a sequence number is inside the current Rx window
289 /// \param sn The sequence number to be checked
290 /// \return True if sn is inside the Rx window, false otherwise
291 constexpr bool inside_rx_window(uint16_t sn) const
292 {
293 // RX_Deliv <= SN < RX_Deliv + Window_Size
294 return rx_mod_base(sn) < gtpu_rx_window_size;
295 }
