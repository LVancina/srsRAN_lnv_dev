32class sdap_session_trx_log_prefix
33{
34public:
35 sdap_session_trx_log_prefix(uint32_t ue_index,
36 pdu_session_id_t psi,
37 optional<qos_flow_id_t> qfi,
38 drb_id_t drb_id,
39 const char* dir)
40 {
41 fmt::memory_buffer buffer;
42 if (qfi.has_value()) {
43 fmt::format_to(buffer, "ue={} {} {} {} {}: ", ue_index, psi, qfi.value(), drb_id, dir);
44 } else {
45 fmt::format_to(buffer, "ue={} {} {} {}: ", ue_index, psi, drb_id, dir);
46 }
47 prefix = srsran::to_c_str(buffer);
48 }
49 const char* to_c_str() const { return prefix.c_str(); }
50
51private:
52 std::string prefix;
53};
