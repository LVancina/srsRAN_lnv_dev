33/// Class used for receiving GTP-U bearers.
34class gtpu_tunnel_base_rx : public gtpu_tunnel_rx_upper_layer_interface
35{
36public:
37 gtpu_tunnel_base_rx(gtpu_tunnel_log_prefix log_prefix) : logger("GTPU", log_prefix) {}
38 ~gtpu_tunnel_base_rx() = default;
39
40 /*
41 * SDU/PDU handlers
42 */
43 void handle_pdu(byte_buffer buf, const sockaddr_storage& src_addr) final
44 {
45 gtpu_dissected_pdu dissected_pdu;
46 bool read_ok = gtpu_dissect_pdu(dissected_pdu, std::move(buf), logger);
47 if (!read_ok) {
48 logger.log_error("Dropped PDU, error reading GTP-U header. pdu_len={}", dissected_pdu.buf.length());
49 return;
50 }
51
52 // continue processing in domain-specific subclass
53 handle_pdu(std::move(dissected_pdu), src_addr);
54 }
55
56protected:
57 virtual void handle_pdu(gtpu_dissected_pdu&& pdu, const sockaddr_storage& src_addr) = 0;
58
59 gtpu_tunnel_logger logger;
60};
