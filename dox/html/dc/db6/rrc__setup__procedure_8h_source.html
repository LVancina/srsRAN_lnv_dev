36/// \brief Handles the creation of SRBs in the RRC UE.
37/// \startuml
38/// queue DU
39/// participant F1AP
40/// participant PDCP
41/// participant RRC
42/// participant "DU processor" as DUMNG
43/// DU->F1AP: initial UL RRC message
44/// activate F1AP
45/// F1AP->DUMNG: ue_creation_request
46/// activate DUMNG
47/// Note over DUMNG: Allocate UE index
48/// Note over DUMNG: Create UE object
49/// Note over DUMNG: Create SRB0 notifiers
50/// DUMNG->RRC: create_ue (with SRB0 handle)
51/// DUMNG-->F1AP: return of "ue_creation_request" (with UE index and SRB0 handle)
52/// deactivate DUMNG
53/// Note over F1AP: Allocate F1AP-CU-UE-ID
54/// Note over F1AP: Association of UE index\nwith F1AP-CU-UE-ID
55/// F1AP->RRC: notify_init_ul_rrc_msg
56/// deactivate F1AP
57/// Note over RRC: Decide RRC Setup vs Reject
58/// activate RRC
59/// Note over RRC: Create SRB1 (SRB1)
60/// Note over RRC: Initiate RRCSetup procedure
61/// RRC->F1AP: send_rrc_setup
62/// F1AP->DU: DL RRC Message Transfer
63/// DU->F1AP: UL RRC Message Transfer
64/// F1AP->RRC: rrc_setup_complete
65/// deactivate RRC
66/// Note over RRC: Decide next step
67/// \enduml
68class rrc_setup_procedure
69{
70public:
71 rrc_setup_procedure(rrc_ue_context_t& context_,

80 void operator()(coro_context<async_task<void>>& ctx);
