124/// Basic class for HARQ processes: will be extended by DL and UL HARQ process classes
125template <bool IsDownlink>
126class harq_process
127{
128 /// Type used to store the DCI format and RNTI type of the TB stored in the HARQ.
129 using dci_rnti_config_type = std::conditional_t<IsDownlink, dci_dl_rnti_config_type, dci_ul_rnti_config_type>;
130
131public:
132 /// \brief Default timeout in slots after which the HARQ process assumes that the CRC/ACK went missing
133 /// (implementation-defined).
134 constexpr static unsigned DEFAULT_ACK_TIMEOUT_SLOTS = 256U;
135
136 constexpr static unsigned SHORT_ACK_TIMEOUT_DTX = 4U;
137
138 /// Maximum number of Transport Blocks as per TS38.321, 5.3.2.1 and 5.4.2.1.
139 constexpr static size_t MAX_NOF_TBS = IsDownlink ? 2 : 1;
140
141 struct transport_block {
142 /// State of the Transport Block.
143 enum class state_t { empty, pending_retx, waiting_ack } state = state_t::empty;
144
145 /// New Data Indicator. Its value should flip for every new Tx.
146 bool ndi = false;
147
148 /// Number of retransmissions that took place for the current Transport Block.
149 unsigned nof_retxs = 0;
150
151 /// Maximum number of retransmission before Transport Block is reset.
152 unsigned max_nof_harq_retxs = 0;
153
154 /// HARQ-bit index corresponding to this HARQ process in the UCI PDU indication.
155 uint8_t harq_bit_idx = 0;
156
157 /// Whether to set the HARQ as ACKed or NACKed when the timeout expires.
158 bool ack_on_timeout = false;
159
160 bool empty() const { return state == state_t::empty; }
161 };
162
163 /// HARQ process id.
164 const harq_id_t id;
165
166 /// \brief HARQ process constructor.
167 /// \param h_id HARQ process ID.
168 /// \param max_ack_wait_in_slots_ number of slots above which the scheduler considers that the ACK/CRC went missing.
169 explicit harq_process(harq_id_t h_id,
170 harq_logger& logger_,
171 ue_harq_timeout_notifier timeout_notif = {},
172 unsigned max_ack_wait_in_slots_ = DEFAULT_ACK_TIMEOUT_SLOTS);
173
174 /// \brief Indicate the beginning of a new slot.
175 void slot_indication(slot_point slot_tx);
176
177 /// \brief Last slot when a SCH grant was scheduled for this HARQ process.
178 slot_point slot_tx() const { return last_slot_tx; }
179
180 /// \brief Slot when the last SCH grant of this HARQ process is supposed to be ACKed.
181 slot_point slot_ack() const { return last_slot_ack; }
182
183 /// \brief Checks whether a specified TB is empty.
184 bool empty(unsigned tb_idx) const
185 {
186 srsran_assert(tb_idx < tb_array.size(), "Invalid TB index={}", tb_idx);
187 return tb_array[tb_idx].empty();
188 }
189
190 /// \brief Checks whether specified TB has pending new_retx.
191 bool has_pending_retx(unsigned tb_idx) const
192 {
193 srsran_assert(tb_idx < tb_array.size(), "Invalid TB index ({} >= {})", tb_idx, tb_array.size());
194 return tb_array[tb_idx].state == transport_block::state_t::pending_retx;
195 }
196
197 /// \brief Checks whether specified TB is expecting an ACK.
198 bool is_waiting_ack(unsigned tb_idx) const
199 {
200 srsran_assert(tb_idx < tb_array.size(), "Invalid TB index={}", tb_idx);
201 return tb_array[tb_idx].state == transport_block::state_t::waiting_ack;
202 }
203
204 unsigned max_nof_harq_retxs(unsigned tb_idx) const
205 {
206 srsran_assert(tb_idx < tb_array.size(), "Invalid TB index ({} >= {})", tb_idx, tb_array.size());
207 return tb_array[tb_idx].max_nof_harq_retxs;
208 }
209
210 /// \brief Getter for TB parameters.
211 const transport_block& tb(unsigned tb_idx) const { return tb_array[tb_idx]; }
212
213 /// \brief Resets HARQ process state.
214 void reset();
215
216 /// \brief Cancels the HARQ process by stopping retransmissions of the currently held TB.
217 void cancel_harq_retxs(unsigned tb_idx);
218
219 /// \brief Getter of the slot when the HARQ process will assume that the ACK/CRC went missing.
220 slot_point get_slot_ack_timeout() const { return slot_ack_timeout; }
221
222protected:
223 void tx_common(slot_point slot_tx, slot_point slot_ack);
224 void new_tx_tb_common(unsigned tb_idx, unsigned max_nof_harq_retxs, uint8_t harq_bit_idx);
225 void new_retx_tb_common(unsigned tb_idx, uint8_t harq_bit_idx);
226
227 /// \brief Updates the ACK state of the TB of the HARQ process.
228 bool ack_info_common(uint32_t tb_idx, bool ack);
229
230 /// \brief Clear the contents of a specific TB of this HARQ process and resets it back to empty state.
231 void reset_tb(unsigned tb_idx);
232
233 /// HARQ entity logger used by this HARQ process.
234 harq_logger& logger;
235
236 /// Notifier used by this HARQ process when there is a timeout.
237 ue_harq_timeout_notifier timeout_notifier;
238
239 /// Maximum value of time interval, in slots, before the HARQ process assumes that the ACK/CRC went missing.
240 const unsigned max_ack_wait_in_slots;
241
242 /// Last slot indication.
243 slot_point last_slot_ind;
244
245 /// For DL, slot_tx corresponds to the slot when the TB in the HARQ process is going to be transmitted by the gNB.
246 /// For UL, slot_tx corresponds to the slot when the TB in the HARQ process is going to be transmitted by the UE.
247 slot_point last_slot_tx;
248
249 /// For DL, slot_ack is the slot at which gNB is expected to receive the ACK via UCI.
250 /// For UL, slot_ack is the slot when the PUSCH CRC will be received by the gNB. It coincides with slot_tx of UL.
251 slot_point last_slot_ack;
252
253 /// Slot when the HARQ process will assume that the ACK/CRC went missing.
254 slot_point slot_ack_timeout;
255
256 std::array<transport_block, MAX_NOF_TBS> tb_array;
257};

554 template <typename HarqVector>
555 harq_id_t find_oldest_harq_retx(const HarqVector& harqs) const
556 {
557 harq_id_t oldest_h = harq_id_t::INVALID_HARQ_ID;
558 for (unsigned i = 0; i != harqs.size(); ++i) {
