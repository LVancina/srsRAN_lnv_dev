30/// Encoder of bits in a byte buffer.
31class bit_encoder
32{
33public:
34 explicit bit_encoder(byte_buffer& bytes) : writer(bytes) {}
35
36 /// Get a view to held byte buffer.
37 srsran::byte_buffer_view data() const { return writer.view(); }
38
39 /// Gets number of written bytes.
40 unsigned nof_bytes() const { return writer.length(); }
41
42 /// Gets number of written bits.
43 unsigned nof_bits() const { return 8U * (writer.length() - (offset == 0U ? 0U : 1U)) + offset; }
44
45 /// \brief Get offset of the next bit to be set. Offset 0 corresponds to the MSB of the current byte being packed,
46 /// while offset 7 corresponds to LSB.
47 unsigned next_bit_offset() const { return offset; }
48
49 /// \brief Append provided bitmap into byte_buffer held by bit_encoder.
50 /// The bitmap to pack is left aligned (i.e. MSB of bitmap will be the last bit to be packed in underlying buffer).
51 /// Example:
52 /// - Before: [...][11______]
53 /// - pack(0b1011, 4): [...][111011__]
54 /// - pack(0b01100, 5): [...][11101101][100_____]
55 /// \param val bitmap to be packed.
56 /// \param n_bits number of bits to pack.
57 /// \return success or failure.
58 bool pack(uint64_t val, uint32_t n_bits);
59
60 /// Append range of bytes into byte_buffer held by bit_encoder.
61 /// \param bytes span of bytes.
62 bool pack_bytes(srsran::span<const uint8_t> bytes);
63
64 /// Append bytes of a byte_buffer into byte_buffer held by bit_encoder.
65 bool pack_bytes(srsran::byte_buffer_view bytes);
66
67 /// Pads held buffer with zeros until the next byte.
68 void align_bytes_zero();
69
70private:
71 /// Interface to byte buffer where bits are going to be appended.
72 srsran::byte_buffer_writer writer;
73 /// Offset of the next bit to be set. Values: (0..7).
74 uint8_t offset = 0;
75};

77/// Decoder of bits stored in a byte_buffer.
78class bit_decoder
79{
80public:
81 explicit bit_decoder(srsran::byte_buffer_view buffer_) : buffer(buffer_), it(buffer.begin()) {}
82
83 /// Get a view to held byte buffer.
84 byte_buffer_view data() const { return buffer; }
85
86 /// Gets number of read bytes.
87 unsigned nof_bytes() const { return (it - buffer.begin()) + (offset != 0 ? 1U : 0U); }
88
89 /// Gets number of read bits.
90 unsigned nof_bits() const { return offset + 8U * static_cast<unsigned>(it - buffer.begin()); }
91
92 /// Get offset of the next bit to be set.
93 unsigned next_bit_offset() const { return offset; }
94
95 /// Advance reader by n_bits bits.
96 /// \param n_bits number of bits to advance the reader.
97 /// \return true if bit jump did not exceed the end of the byte buffer. False, otherwise.
98 bool advance_bits(uint32_t n_bits);
99
100 /// Read bits from underlying byte_buffer pointed by bit_decoder.
101 /// \param val bitmap of bits read.
102 /// \param n_bits number of bits to read.
103 /// \return true if read is successful. False, if the number of bits read exceeded the byte_buffer size.
104 template <class T>
105 bool unpack(T& val, uint32_t n_bits);
106
107 /// Read unaligned bytes from underlying byte_buffer.
108 /// \param bytes span of bytes where the result is stored. The span size defines the number of bytes to be read.
109 /// \return true if successful. False if the number of bytes to be read exceeds the end of the byte_buffer.
110 bool unpack_bytes(srsran::span<uint8_t> bytes);
111
112 /// Create a view across \c n_bytes bytes of the underlying byte_buffer, starting at the first full byte from the
113 /// current location. For this purpose, the decoder is first automatically aligned via \c align_bytes.
114 /// \param n_bytes Number of bytes to read.
115 /// \return byte_buffer_view across \c n_bytes bytes of the underlying byte_buffer.
116 /// An empty view is returned if the number of bytes to be read exceeds the end of the byte_buffer.
117 byte_buffer_view unpack_aligned_bytes(size_t n_bytes);
118
119 /// Skip bits until the beginning of the next byte.
120 void align_bytes();
121
122private:
123 srsran::byte_buffer_view buffer;
124 srsran::byte_buffer::const_iterator it;
125 uint8_t offset = 0;
126};

132/// \brief Custom formatter for bit_encoder
133template <>
134struct formatter<srsran::bit_encoder> {
135 template <typename ParseContext>
136 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
137 {
138 return ctx.begin();
139 }
140
141 template <typename FormatContext>
142 auto format(const srsran::bit_encoder& s, FormatContext& ctx) -> decltype(std::declval<FormatContext>().out())
143 {
144 if (s.next_bit_offset() == 0) {
145 fmt::format_to(ctx.out(), "{:b}", s.data());
146 } else {
147 fmt::format_to(ctx.out(), "{:b}", s.data().view(0, s.nof_bytes() - 1));
148 uint8_t last_byte = s.data()[s.nof_bytes() - 1];
149 last_byte = last_byte >> (8U - s.next_bit_offset());
150 fmt::format_to(ctx.out(), " {:0>{}b}", last_byte, s.next_bit_offset());
151 }
152 return ctx.out();
153 }
154};
