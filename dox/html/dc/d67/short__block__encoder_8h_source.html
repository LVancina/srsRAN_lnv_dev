48/// Short-block encoder interface.
49class short_block_encoder
50{
51public:
52 /// Default destructor.
53 virtual ~short_block_encoder() = default;
54
55 /// \brief Encodes and rate-matches a small number (less than 12) bits.
56 ///
57 /// Encodes a short sequence of bits (length 1 to 11, both included), following the specifications of TS38.212
58 /// Section 5.3.3, and rate-matches the encoded bits according to TS38.212 Section 5.4.3.
59 /// \param[out] output Sequence of encoded and rate-matched bits (i.e., \f$f_0, f_1, \dots, f_{E-1}\f$ in TS38.212).
60 /// \param[in] input Sequence of bits to encode (i.e., \f$c_0, c_1, \dots, c_{K-1}\f$ in TS38.212).
61 /// \param[in] mod Modulation scheme.
62 /// \remark Both \c output and \c input follow an unpacked representation, that is one bit per entry.
63 /// \remark The size \f$E\f$ of \c output should be as follows:
64 /// - \f$E \ge Q_m\f$, where \f$Q_m\f$ is the modulation order, if \c input is a single bit;
65 /// - \f$E \ge 3Q_m\f$ if \c input consists of 2 bits;
66 /// - \f$E \ge 32\f$ otherwise (\c input size from 3 to 11 bits).
67 virtual void encode(span<uint8_t> output, span<const uint8_t> input, modulation_scheme mod) = 0;
68};
