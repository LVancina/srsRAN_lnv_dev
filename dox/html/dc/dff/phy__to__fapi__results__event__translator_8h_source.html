32/// \brief PHY-to-FAPI uplink results events translator.
33///
34/// This class listens to upper PHY uplink result events and translates them into FAPI indication messages that are sent
35/// through the FAPI data-specific message notifier.
36class phy_to_fapi_results_event_translator : public upper_phy_rx_results_notifier
37{
38public:
39 explicit phy_to_fapi_results_event_translator(srslog::basic_logger& logger_);
40
41 // See interface for documentation.
42 void on_new_prach_results(const ul_prach_results& result) override;
43
44 // See interface for documentation.
45 void on_new_pusch_results_control(const ul_pusch_results_control& result) override;
46
47 // See interface for documentation.
48 void on_new_pusch_results_data(const ul_pusch_results_data& result) override;
49
50 // See interface for documentation.
51 void on_new_pucch_results(const ul_pucch_results& result) override;
52
53 /// Configures the FAPI slot-based, data-specific notifier to the given one.
54 void set_slot_data_message_notifier(fapi::slot_data_message_notifier& fapi_data_slot_notifier)
55 {
56 data_notifier = std::ref(fapi_data_slot_notifier);
57 }
58
59private:
60 /// Notifies a new FAPI \e CRC.indication through the data notifier.
61 void notify_crc_indication(const ul_pusch_results_data& result);
62
63 /// Notifies a new FAPI \e Rx_Data.indication through the data notifier.
64 void notify_rx_data_indication(const ul_pusch_results_data& result);
65
66 /// Notifies a new FAPI \e UCI.indication through the data notifier that carries a PUSCH PDU.
67 void notify_pusch_uci_indication(const ul_pusch_results_control& result);
68
69private:
70 /// FAPI logger.
71 srslog::basic_logger& logger;
72 /// FAPI slot-based, data-specific message notifier.
73 std::reference_wrapper<fapi::slot_data_message_notifier> data_notifier;
74};
