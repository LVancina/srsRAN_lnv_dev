31/// eCPRI packet decoder implementation.
32class packet_decoder_impl : public packet_decoder
33{
34public:
35 explicit packet_decoder_impl(srslog::basic_logger& logger_) : logger(logger_) {}
36
37 // See interface for documentation.
38 span<const uint8_t> decode(span<const uint8_t> packet, packet_parameters& params) override;
39
40private:
41 /// Decodes an eCPRI common header from the packet.
42 ///
43 /// \param packet eCPRI packet to decode.
44 /// \param params eCPRI decoded parameters.
45 /// \return A span to the eCPRI parameters and payload on success, otherwise an empty span.
46 span<const uint8_t> decode_header(span<const uint8_t> packet, packet_parameters& params);
47
48 /// Decodes eCPRI parameters and extracts eCPRI payload from the given packet.
49 /// \param packet eCPRI parameters and payload to decode.
50 /// \param params eCPRI decoded parameters.
51 /// \return A span to the eCPRI payload on success, otherwise an empty span.
52 virtual span<const uint8_t> decode_payload(span<const uint8_t> packet, packet_parameters& params) = 0;
53
54protected:
55 srslog::basic_logger& logger;
56};
