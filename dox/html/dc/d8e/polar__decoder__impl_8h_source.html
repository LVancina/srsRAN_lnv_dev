34/// \brief Polar decoder implementation.
35///
36/// Generic polar decoder according to the specifications of TS38.212 Section 5.3.1.
37class polar_decoder_impl : public polar_decoder
38{
39private:
40 /// Types of node in an SSC decoder.
41 enum node_rate : uint8_t {
42 /// See function rate_0_node().
43 RATE_0 = 0,
44 /// See function rate_r_node().
45 RATE_R = 2,
46 /// See function rate_1_node().
47 RATE_1 = 3,
48 };
49
50 /// Collection of SSC polar decoder parameters.
51 struct params_s {
52 /// \f$log_2\f$ of the code size.
53 uint8_t code_size_log;
54 /// Number of bits of the encoder input/output vector, for all stages.
55 srsvec::aligned_vec<uint16_t> code_stage_size;
56 /// Stores the type of all nodes in stage 0.
57 std::vector<node_rate> node_type_alloc;
58 /// Stores the type of all nodes, stage by stage (e.g., <tt>node_type[0]</tt> is the same as \c node_type_alloc).
59 std::vector<node_rate*> node_type;
60 };
61
62 /// Describes the SSC polar decoder inner state.
63 struct state_s {
64 /// Current stage of the decoding algorithm (0,...,code_size_log).
65 uint8_t stage;
66 /// True if the last bit is decoded. False otherwise.
67 bool flag_finished;
68 /// Indicates the active node at each stage of the algorithm at a given moment.
69 srsvec::aligned_vec<uint16_t> active_node_per_stage;
70 };
71
72 /// Structure with pointers needed to obtain the node_type.
73 struct tmp_node_s {
74 /// \brief Denotes whether a node is of type [RATE_0](#polar_decoder_impl::node_rate) (value 0) or of another
75 /// type (value 1).
76 srsvec::aligned_vec<uint8_t> is_not_rate_0;
77 /// \brief Denotes whether a node is of type [RATE_1](#polar_decoder_impl::node_rate) (value 1) or of another type
78 /// (value 0).
79 span<uint8_t> is_rate_1;
80 /// List of even-valued node indices.
81 srsvec::aligned_vec<uint16_t> i_even;
82 /// List of odd-valued node indices.
83 srsvec::aligned_vec<uint16_t> i_odd;
84
85 /// \brief Allocates memory resources for the computation of the node_type.
86 /// \param[in] nMax \f$log_2\f$ of the maximum number of bits in the codeword.
87 explicit tmp_node_s(uint8_t nMax);
88
89 /// \brief Computes node types for the decoding tree associated to the given frozen set.
90 /// \param[out] node_type Two-dimensional pointer containing the node type at each stage of the decoding tree.
91 /// \param[in] frozen_set Position of the frozen bits in the codeword.
92 /// \param[in] code_size_log \f$log_2\f$ of code size.
93 void compute(std::vector<node_rate*>& node_type, span<const uint16_t> frozen_set, uint16_t code_size_log);
94 };
95
96 /// LLR values for stage 0 (i.e., received LLRs).
97 srsvec::aligned_vec<log_likelihood_ratio> llr_alloc;
98 /// Pointers to the upper half of the LLR values for all stages.
99 std::vector<span<log_likelihood_ratio>> llr0;
100 /// Pointers to the lower half of LLR values for all stages.
101 std::vector<span<log_likelihood_ratio>> llr1;
102 /// Temporary estimated bits.
103 srsvec::aligned_vec<uint8_t> est_bit;
104 /// Decoder inner parameters.
105 params_s param;
106 /// Decoder state.
107 state_s state;
108 /// Node type manager.
109 tmp_node_s tmp_node_type;
110 /// A polar encoder.
111 std::unique_ptr<polar_encoder> enc;
112
113 /// \brief Polar decoder initialization.
114 ///
115 /// Initializes internal parameters and registers.
116 /// \param[in] data_decoded Register for writing the decoded data.
117 /// \param[in] input_llr Input codeblock as a sequence of log-likelihood ratios.
118 /// \param[in] code_size_log Codeblock size (binary logarithm thereof).
119 /// \param[in] frozen_set Indices of the frozen bits.
120 void init(span<uint8_t> data_decoded,
121 span<const log_likelihood_ratio> input_llr,
122 uint8_t code_size_log,
123 span<const uint16_t> frozen_set);
124
125 /// \brief Updates a RATE_0 node.
126 ///
127 /// All decoded bits below a [RATE_0](#polar_decoder_impl::node_rate) node are 0.
128 /// The function updates the \a p->state->active_node_per_stage
129 /// pointer to point to the next active node. It is assumed that message bits are initialized to 0.
130 void rate_0_node();
131
132 /// \brief Updates a RATE_1 node.
133 ///
134 /// [RATE_1](#polar_decoder_impl::node_rate) nodes at stage \f$ s \f$ return the associated \f$2^s\f$ estimated bits
135 /// by making a hard decision on them. RATE_1 nodes also update message bits vector.
136 void rate_1_node(span<uint8_t> data_decoded);
137
138 /// \brief Updates a RATE_R node.
139 ///
140 /// RATE_R nodes at stage \f$ s \f$ return the associated \f$2^s\f$ decoded bit by calling
141 /// the child nodes to the right and left of the decoding tree and then polar encoding (xor) their output.
142 void rate_r_node(span<uint8_t> data_decoded);
143
144 /// \brief Node processing.
145 ///
146 /// Switches between the different [types of node](#polar_decoder_impl::node_rate) for the SSC algorithm.
147 /// Nodes in the decoding tree at stage \f$ s\f$ get the \f$2^s\f$ LLRs from the parent node and
148 /// return the associated \f$2^s\f$ estimated bits.
149 void simplified_node(span<uint8_t> data_decoded);
150
151public:
152 /// \brief Polar decoder initialization.
153 ///
154 /// Initializes all the polar decoder variables according to the Simplified Successive
155 /// Cancellation (SSC) decoder algorithm and the maximum given code size.
156 ///
157 /// \param[in] enc A polar encoder.
158 /// \param[in] code_size_log The \f$ log_2\f$ of the number of bits of the decoder input/output vector.
159 /// \return A unique pointer to the initialized SSC polar decoder.
160 polar_decoder_impl(std::unique_ptr<polar_encoder> enc_, uint8_t nMax);
161
162 // See interface for the documentation.
163 void decode(span<uint8_t> data_decoded, span<const log_likelihood_ratio> input_llr, const polar_code& code) override;
164};
165
