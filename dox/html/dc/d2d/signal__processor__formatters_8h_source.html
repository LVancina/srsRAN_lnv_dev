32namespace fmt {
33/// \brief Custom formatter for \c nzp_csi_rs_generator::config_t.
34template <>
35struct formatter<srsran::nzp_csi_rs_generator::config_t> {
36 /// Helper used to parse formatting options and format fields.
37 srsran::delimited_formatter helper;
38
39 /// Default constructor.
40 formatter() = default;
41
42 template <typename ParseContext>
43 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
44 {
45 return helper.parse(ctx);
46 }
47
48 template <typename FormatContext>
49 auto format(const srsran::nzp_csi_rs_generator::config_t& nzp_csi_config, FormatContext& ctx)
50 -> decltype(std::declval<FormatContext>().out())
51 {
52 helper.format_always(ctx, "prb=[{}, {})", nzp_csi_config.start_rb, nzp_csi_config.start_rb + nzp_csi_config.nof_rb);
53 helper.format_always(ctx, "row={}", nzp_csi_config.csi_rs_mapping_table_row);
54 helper.format_always(ctx, "Ki={}", srsran::span<const unsigned>(nzp_csi_config.freq_allocation_ref_idx));
55 helper.format_always(ctx, "l0={}", nzp_csi_config.symbol_l0);
56 helper.format_always(ctx, "l1={}", nzp_csi_config.symbol_l1);
57 helper.format_always(ctx, "CDM={}", to_string(nzp_csi_config.cdm));
58 helper.format_always(ctx, "density={}", to_string(nzp_csi_config.freq_density));
59 helper.format_if_verbose(ctx, "slot={}", nzp_csi_config.slot);
60 helper.format_if_verbose(ctx, "cp={}", nzp_csi_config.cp.to_string());
61 helper.format_if_verbose(ctx, "sc_id={}", nzp_csi_config.scrambling_id);
62 helper.format_if_verbose(ctx, "amplitude={}", nzp_csi_config.amplitude);
63 helper.format_if_verbose(ctx, "precoding={}", nzp_csi_config.precoding);
64
65 return ctx.out();
66 }
67};
