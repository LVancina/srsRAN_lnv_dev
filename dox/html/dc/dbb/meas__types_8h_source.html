35/// Maximum number of measurements.
36const uint8_t MAX_NOF_MEAS = 64;

38/// Meas index.
39enum class meas_id_t : uint8_t { min = 1, max = MAX_NOF_MEAS, invalid = MAX_NOF_MEAS + 1 };
