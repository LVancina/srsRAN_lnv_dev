40class radio_uhd_tx_stream_fsm
41{
42private:
43 /// Wait for end-of-burst acknowledgement timeout in seconds.
44 static constexpr double WAIT_EOB_ACK_TIMEOUT_S = 0.01;
45
46 /// Defines the Tx stream internal states.
47 enum class states {
48 /// Indicates the stream was not initialized successfully.
49 UNINITIALIZED = 0,
50 /// Indicates the stream is ready to start burst.
51 START_BURST,
52 /// Indicates the stream is transmitting a burst.
53 IN_BURST,
54 /// Indicates an end-of-burst must be transmitted and abort any transmission.
55 END_OF_BURST,
56 /// Indicates wait for end-of-burst acknowledgement.
57 WAIT_END_OF_BURST,
58 /// Signals a stop to the asynchronous thread.
59 WAIT_STOP,
60 /// Indicates the asynchronous thread is notify_stop.
61 STOPPED
62 };
63
64 /// Indicates the current state.
65 states state;
66
67 /// Protects the class concurrent access.
68 mutable std::mutex mutex;
69 /// Condition variable to wait for certain states.
70 std::condition_variable cvar;
71
72 uhd::time_spec_t wait_eob_timeout = uhd::time_spec_t();
73
74public:
75 /// \brief Notifies that the transmit stream has been initialized successfully.
76 void init_successful()
77 {
78 std::unique_lock<std::mutex> lock(mutex);
79 state = states::START_BURST;
80 }
81
82 /// \brief Notifies a late or an underflow event.
83 /// \remark Transitions state end of burst if it is in a burst.
84 /// \param[in] time_spec Indicates the time the underflow event occurred.
85 void async_event_late_underflow(const uhd::time_spec_t& time_spec)
86 {
87 std::unique_lock<std::mutex> lock(mutex);
88 if (state == states::IN_BURST) {
89 state = states::END_OF_BURST;
90 wait_eob_timeout = time_spec;
91 wait_eob_timeout += WAIT_EOB_ACK_TIMEOUT_S;
92 }
93 }
94
95 /// \brief Notifies an end-of-burst acknowledgement.
96 /// \remark Transitions state to start burst if it is waiting for the end-of-burst.
97 void async_event_end_of_burst_ack()
98 {
99 std::unique_lock<std::mutex> lock(mutex);
100 if (state == states::WAIT_END_OF_BURST) {
101 state = states::START_BURST;
102 }
103 }
104
105 /// \brief Handles a new transmission.
106 /// \param[out] metadata Destination of the required metadata.
107 /// \param[in] time_spec Transmission time of the first sample.
108 /// \param[in] is_empty Empty buffer flag.
109 /// \param[in] tail_padding Tail padding flag, indicating the last transmission in the burst.
110 /// \return True if the block shall be transmitted. False if the block shall be ignored.
111 bool on_transmit(uhd::tx_metadata_t& metadata, uhd::time_spec_t& time_spec, bool is_empty, bool tail_padding)
112 {
113 std::unique_lock<std::mutex> lock(mutex);
114 switch (state) {
115 case states::WAIT_END_OF_BURST:
116 // Do nothing if the wait for end-of-burst timeout has not expired.
117 if (wait_eob_timeout.get_real_secs() >= time_spec.get_real_secs()) {
118 return false;
119 }
120 // Otherwise go into start burst state and handle the state.
121 state = states::START_BURST;
122 case states::START_BURST:
123 // Set start of burst flag and time spec.
124 if (!is_empty) {
125 metadata.has_time_spec = true;
126 metadata.start_of_burst = true;
127 metadata.end_of_burst = tail_padding;
128 metadata.time_spec = time_spec;
129
130 // Transition to in-burst.
131 if (!tail_padding) {
132 state = states::IN_BURST;
133 }
134
135 return true;
136 }
137 return false;
138 case states::IN_BURST:
139 if (is_empty || tail_padding) {
140 // Signal end of burst to UHD.
141 metadata.end_of_burst = true;
142
143 // Transition to start burst without waiting for the EOB ACK.
144 state = states::START_BURST;
145 }
146 break;
147 case states::END_OF_BURST:
148 // Flag end-of-burst.
149 metadata.end_of_burst = true;
150 state = states::WAIT_END_OF_BURST;
151 if (wait_eob_timeout == uhd::time_spec_t()) {
152 wait_eob_timeout = time_spec;
153 wait_eob_timeout += WAIT_EOB_ACK_TIMEOUT_S;
154 }
155 break;
156 case states::UNINITIALIZED:
157 case states::WAIT_STOP:
158 case states::STOPPED:
159 // Ignore transmission.
160 return false;
161 }
162
163 // Transmission shall not be ignored.
164 return true;
165 }
166
167 void stop(uhd::tx_metadata_t& metadata)
168 {
169 std::unique_lock<std::mutex> lock(mutex);
170 if (state == states::IN_BURST) {
171 metadata.end_of_burst = true;
172 }
173 state = states::WAIT_STOP;
174 }
175
176 bool is_stopping() const
177 {
178 std::unique_lock<std::mutex> lock(mutex);
179 return state == states::WAIT_STOP;
180 }
181
182 void wait_stop()
183 {
184 std::unique_lock<std::mutex> lock(mutex);
185 while (state != states::STOPPED) {
186 cvar.wait(lock);
187 }
188 }
189
190 /// Notifies the asynchronous task has notify_stop.
191 void async_task_stopped()
192 {
193 std::unique_lock<std::mutex> lock(mutex);
194 state = states::STOPPED;
195 cvar.notify_all();
196 }
197};
