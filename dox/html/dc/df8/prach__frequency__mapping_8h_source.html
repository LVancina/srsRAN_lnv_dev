30/// Collects PRACH frequency mapping information as per TS38.211 Section 5.3.2 and Section 6.3.3.2.
31struct prach_frequency_mapping_information {
32 /// Number of resource blocks per PRACH occasion in the frequency domain, relative to the PUSCH subcarrier spacing
33 /// (\f$\Delta f\f$).
34 unsigned nof_rb_ra;
35 /// Parameter \f$\bar{k}\f$.
36 unsigned k_bar;
37};
38
39/// Reserved PRACH frequency-domain mapping information.
40static constexpr prach_frequency_mapping_information PRACH_FREQUENCY_MAPPING_INFORMATION_RESERVED = {0, 0};
41
42/// \brief Gets the PRACH frequency mapping information contained in TS38.211 Table 6.3.3.2-1.
43/// \param[in] prach_scs_Hz PRACH subcarrier spacing (parameter \f$\Delta f_{RA}\f$).
44/// \param[in] pusch_scs_Hz PUSCH subcarrier spacing (parameter \f$\Delta f\f$).
45/// \return The requested PRACH frequency mapping information if the combination of input parameters is valid, \c
46/// PRACH_FREQUENCY_MAPPING_INFORMATION_RESERVED otherwise.
47prach_frequency_mapping_information prach_frequency_mapping_get(prach_subcarrier_spacing prach_scs,
48 subcarrier_spacing pusch_scs);
