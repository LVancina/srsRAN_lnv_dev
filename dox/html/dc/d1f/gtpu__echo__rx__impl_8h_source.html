31/// Class used for reception of GTP-U path management messages.
32class gtpu_echo_rx : public gtpu_tunnel_base_rx
33{
34public:
35 gtpu_echo_rx(gtpu_echo_tx_interface& tx_) :
36 gtpu_tunnel_base_rx(gtpu_tunnel_log_prefix{{}, GTPU_PATH_MANAGEMENT_TEID, "DL"}),
37 tx(tx_),
38 psup_packer(logger.get_basic_logger())
39 {
40 }
41 ~gtpu_echo_rx() = default;
42
43protected:
44 // domain-specific PDU handler
45 void handle_pdu(gtpu_dissected_pdu&& pdu, const sockaddr_storage& src_addr) final
46 {
47 // TEID sanity check
48 if (pdu.hdr.teid != GTPU_PATH_MANAGEMENT_TEID) {
49 logger.log_error("Discarded PDU. Cause: Invalid TEID for path management message. teid={} msg_type={:#x}",
50 pdu.hdr.teid,
51 pdu.hdr.message_type);
52 return;
53 }
54 // SN sanity check
55 if (!pdu.hdr.flags.seq_number) {
56 logger.log_error("Discarded PDU. Cause: Missing sequence number in path management message. msg_type={:#x}",
57 pdu.hdr.message_type);
58 return;
59 }
60 switch (pdu.hdr.message_type) {
61 case GTPU_MSG_ECHO_REQUEST:
62 logger.log_info("Rx echo request. sn={}", pdu.hdr.seq_number);
63 tx.send_echo_response(src_addr, pdu.hdr.seq_number);
64 break;
65 case GTPU_MSG_ECHO_RESPONSE:
66 logger.log_info("Rx echo response. sn={}", pdu.hdr.seq_number);
67 tx.handle_echo_response(src_addr, pdu.hdr.seq_number);
68 break;
69 case GTPU_MSG_SUPPORTED_EXTENSION_HEADERS_NOTIFICATION:
70 logger.log_warning("Discarded PDU. Cause: 'Supported extension headers notification' not supported. sn={}",
71 pdu.hdr.seq_number);
72 // TS 29.281 Sec. 5.1:
73 // For Supported Extension Headers Notification [...], the Sequence Number shall be ignored by the receiver,
74 // even though the S flag is set to '1'.
75 return;
76 case GTPU_MSG_ERROR_INDICATION:
77 // TODO: unpack and print information elements; add handling
78 logger.log_error(pdu.buf.begin(), pdu.buf.end(), "Received error indication from peer");
79 // TS 29.281 Sec. 7.3.1: Error Indication
80 // When a GTP-U node receives a G-PDU for which no EPS Bearer context, PDP context, PDU Session, MBMS Bearer
81 // context, or RAB exists, the GTP-U node shall discard the G - PDU.If the TEID of the incoming G-PDU is
82 // different from the value 'all zeros' the GTP-U node shall also return a GTP error indication to the
83 // originating node.
84 return;
85 case GTPU_MSG_END_MARKER:
86 // TODO: unpack and print information elements; add handling
87 logger.log_warning("Discarded PDU. Cause: 'End marker' not supported");
88 return;
89 default:
90 logger.log_error(
91 "Discarded PDU. Cause: Invalid message type for path or tunnel management message. teid={} msg_type={:#x}",
92 pdu.hdr.teid,
93 pdu.hdr.message_type);
94 return;
95 }
96 }
97
98private:
99 gtpu_echo_tx_interface& tx;
100 psup_packing psup_packer;
101};
102} // namespace srsran
