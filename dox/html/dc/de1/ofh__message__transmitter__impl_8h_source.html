33/// \brief Transmits enqueued Open Fronthaul messages through an Ethernet gateway.
34///
35/// Message transmission is managed according the given transmission window.
36class message_transmitter_impl : public ota_symbol_boundary_notifier
37{
38 /// Internal structure used to store transmission window timing parameters expressed in a number of symbols.
39 struct tx_timing_parameters {
40 /// Offset from the current OTA symbol to the first symbol at which DL Control-Plane message can be sent, or in
41 /// other words it is the offset to the start of DL Control-Plane transmission window. Must be calculated based on
42 /// \c T1a_max_cp_dl parameter.
43 unsigned sym_cp_dl_start;
44 /// Offset from the current OTA symbol to the last symbol at which DL Control-Plane message can be sent within its
45 /// transmission window. Must be calculated based on \c T1a_min_cp_dl parameter.
46 unsigned sym_cp_dl_end;
47 /// Offset from the current OTA symbol to the first symbol at which UL Control-Plane message can be sent within its
48 /// transmission window. Must be calculated based on \c T1a_max_cp_ul parameter.
49 unsigned sym_cp_ul_start;
50 /// Offset from the current OTA symbol to the last symbol at which UL Control-Plane message can be sent within its
51 /// transmission window. Must be calculated based on \c T1a_min_cp_ul parameter.
52 unsigned sym_cp_ul_end;
53 /// Offset from the current OTA symbol to the first symbol at which DL User-Plane message can be sent within its
54 /// transmission window. Must be calculated based on \c T1a_max_up parameter.
55 unsigned sym_up_dl_start;
56 /// Offset from the current OTA symbol to the last symbol at which DL User-Plane message can be sent within its
57 /// transmission window. Must be calculated based on \c T1a_min_up parameter.
58 unsigned sym_up_dl_end;
59
60 tx_timing_parameters(const du_tx_window_timing_parameters& params,
61 std::chrono::duration<double, std::nano> symbol_duration) :
62 sym_cp_dl_start(std::floor(params.T1a_max_cp_dl / symbol_duration)),
63 sym_cp_dl_end(std::ceil(params.T1a_min_cp_dl / symbol_duration)),
64 sym_cp_ul_start(std::floor(params.T1a_max_cp_ul / symbol_duration)),
65 sym_cp_ul_end(std::ceil(params.T1a_min_cp_ul / symbol_duration)),
66 sym_up_dl_start(std::floor(params.T1a_max_up / symbol_duration)),
67 sym_up_dl_end(std::ceil(params.T1a_min_up / symbol_duration))
68 {
69 }
70 };
71
72 /// Logger.
73 srslog::basic_logger& logger;
74 /// Ethernet frame pool.
75 std::shared_ptr<ether::eth_frame_pool> pool_ptr;
76 ether::eth_frame_pool& pool;
77 /// Gateway handling message transmission.
78 std::unique_ptr<ether::gateway> gateway;
79 /// Internal representation of timing parameters.
80 const tx_timing_parameters timing_params;
81
82public:
83 message_transmitter_impl(srslog::basic_logger& logger_,
84 const symbol_handler_config& cfg,
85 std::unique_ptr<ether::gateway> gw,
86 std::shared_ptr<ether::eth_frame_pool> frame_pool);
87
88 // See interface for documentation.
89 void on_new_symbol(slot_symbol_point symbol_point) override;
90
91private:
92 /// Transmits enqueued messages for the given interval of slot symbol points.
93 void transmit_enqueued_messages(const ether::frame_pool_interval& interval);
94};
