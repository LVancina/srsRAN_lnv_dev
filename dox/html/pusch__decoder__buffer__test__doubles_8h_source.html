29class pusch_decoder_buffer_spy : public pusch_decoder_buffer
30{
31public:
32 pusch_decoder_buffer_spy() = default;
33
34 pusch_decoder_buffer_spy(unsigned buffer_size) { data.resize(buffer_size); }
35
36 // See interface for documentation.
37 span<log_likelihood_ratio> get_next_block_view(unsigned block_size) override
38 {
39 data.resize(block_size + count);
40 return span<log_likelihood_ratio>(data).subspan(count, block_size);
41 }
42
43 // See interface for documentation.
44 void on_new_softbits(span<const log_likelihood_ratio> softbits) override
45 {
46 span<log_likelihood_ratio> block = get_next_block_view(softbits.size());
47
48 if (block.data() != softbits.data()) {
49 srsvec::copy(block, softbits);
50 }
51
52 count += softbits.size();
53 }
54
55 // See interface for documentation.
56 void on_end_softbits() override
57 {
58 // Ignore.
59 }
60
61 span<const log_likelihood_ratio> get_data() const { return span<const log_likelihood_ratio>(data).first(count); }
62
63private:
64 unsigned count = 0;
65 std::vector<log_likelihood_ratio> data;
66};
