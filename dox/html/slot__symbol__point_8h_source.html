41 slot_symbol_point(uint32_t numerology_, uint32_t count_, unsigned nof_symbols_) :
42 nof_symbols(nof_symbols_), numerology(numerology_), count_val(count_)

48 slot_point get_slot() const

50 return numerology < NOF_NUMEROLOGIES ? slot_point(numerology, count_val / nof_symbols) : slot_point();
51 }

66 slot_symbol_point& operator+=(int jump)
67 {
68 const int nof_symbols_per_slot_wrap = NOF_SFNS * NOF_SUBFRAMES_PER_FRAME *
69 get_nof_slots_per_subframe(to_subcarrier_spacing(numerology)) * nof_symbols;
70
71 int tmp = (static_cast<int>(count_val) + jump) % nof_symbols_per_slot_wrap;
72 count_val = tmp + (tmp < 0 ? nof_symbols_per_slot_wrap : 0);
73 return *this;
74 }
75
77 slot_symbol_point& operator-=(int jump) { return this->operator+=(-jump); }
78
81 bool operator==(const slot_symbol_point& other) const
82 {
83 return other.count_val == count_val && other.numerology == numerology && other.nof_symbols == nof_symbols;
84 }
85
87 bool operator!=(const slot_symbol_point& other) const { return !(*this == other); }
88
92 bool operator<(const slot_symbol_point& other) const
93 {
94 srsran_assert(numerology == other.numerology, "Comparing slots symbol point of different numerologies");
95 srsran_assert(get_nof_symbols() == other.get_nof_symbols(),
96 "Comparing slots symbol point with different number of symbols");
97
98 const int nof_symbols_per_slot_wrap = NOF_SFNS * NOF_SUBFRAMES_PER_FRAME *
99 get_nof_slots_per_subframe(to_subcarrier_spacing(numerology)) * nof_symbols;
100
101 int v = static_cast<int>(other.count_val) - static_cast<int>(count_val);
102 if (v > 0) {
103 return (v < nof_symbols_per_slot_wrap / 2);
104 }
105 return (v < -nof_symbols_per_slot_wrap / 2);
106 }
107
109 bool operator<=(const slot_symbol_point& other) const { return (*this == other) || (*this < other); }
110 bool operator>=(const slot_symbol_point& other) const { return !(*this < other); }
111 bool operator>(const slot_symbol_point& other) const { return (*this != other) && *this >= other; }
112
113private:
114 uint32_t nof_symbols : 4;
115 uint32_t numerology : 3;
116 uint32_t count_val : 25;
117};

145inline int operator-(slot_symbol_point lhs, slot_symbol_point rhs)
146{
147 srsran_assert(rhs.get_numerology() == lhs.get_numerology(),
148 "Cannot calculate the distance of two slot symbol points that have different numerologies");
149 srsran_assert(rhs.get_nof_symbols() == lhs.get_nof_symbols(),
150 "Cannot calculate the distance of two slot symbol points that have a different number of symbols");
151
152 const int nof_symbols_per_slot_wrap = NOF_SFNS * NOF_SUBFRAMES_PER_FRAME *
153 get_nof_slots_per_subframe(to_subcarrier_spacing(lhs.get_numerology())) *
154 lhs.get_nof_symbols();
155
156 int tmp = static_cast<int>(lhs.system_slot()) - static_cast<int>(rhs.system_slot());
157 if (tmp > (nof_symbols_per_slot_wrap / 2)) {
158 return (tmp - nof_symbols_per_slot_wrap);
159 }
160
161 if (tmp < -(nof_symbols_per_slot_wrap / 2)) {
162 return tmp + nof_symbols_per_slot_wrap;
163 }
164 return tmp;
165}
