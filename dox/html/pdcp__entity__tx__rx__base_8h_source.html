35// Header length
36constexpr size_t pdcp_data_pdu_header_size_12bit = 2;
37constexpr size_t pdcp_data_pdu_header_size_18bit = 3;
38constexpr size_t pdcp_data_pdu_header_size_max = pdcp_data_pdu_header_size_18bit;
39constexpr size_t pdcp_data_pdu_header_size(pdcp_sn_size sn_size)
40{
41 switch (sn_size) {
42 case pdcp_sn_size::size12bits:
43 return pdcp_data_pdu_header_size_12bit;
44 case pdcp_sn_size::size18bits:
45 return pdcp_data_pdu_header_size_18bit;
46 }
47 srsran_assertion_failure("Cannot determine PDCP data PDU header size: unsupported sn_size={}",
48 pdcp_sn_size_to_uint(sn_size));
49 return pdcp_data_pdu_header_size_12bit;
50}
51
52// Window size
53constexpr size_t pdcp_window_size_12bit = 2048;
54constexpr size_t pdcp_window_size_18bit = 131072;
55constexpr uint32_t pdcp_window_size(pdcp_sn_size sn_size)
56{
57 switch (sn_size) {
58 case pdcp_sn_size::size12bits:
59 return pdcp_window_size_12bit;
60 case pdcp_sn_size::size18bits:
61 return pdcp_window_size_18bit;
62 }
63 srsran_assertion_failure("Cannot determine PDCP window size: unsupported sn_size={}", pdcp_sn_size_to_uint(sn_size));
64 return pdcp_window_size_12bit;
65}

69class pdcp_entity_tx_rx_base
70{
71protected:
72 explicit pdcp_entity_tx_rx_base(rb_id_t rb_id_,
73 pdcp_rb_type rb_type_,
74 pdcp_rlc_mode rlc_mode_,
75 pdcp_sn_size sn_size_) :
76 rb_id(rb_id_),
77 rb_type(rb_type_),
78 rlc_mode(rlc_mode_),
79 hdr_len_bytes((pdcp_data_pdu_header_size(sn_size_))),
80 window_size(pdcp_window_size(sn_size_)),
81 sn_size(sn_size_)
82 {
83 if (rb_id_.is_drb()) {
84 bearer_id = drb_id_to_uint(rb_id_.get_drb_id()) - 1;
85 } else if (rb_id.is_srb()) {
86 bearer_id = srb_id_to_uint(rb_id_.get_srb_id()) - 1;
87 } else {
88 srsran_assertion_failure("Invalid RB identity: {}", rb_id_);
89 }
90 }
91
94 uint8_t bearer_id;
95 rb_id_t rb_id;
96
97 /*
98 * RB helpers
99 */
100 const pdcp_rb_type rb_type;
101 const pdcp_rlc_mode rlc_mode;
102 bool is_srb() const { return rb_type == pdcp_rb_type::srb; }
103 bool is_drb() const { return rb_type == pdcp_rb_type::drb; }
104 bool is_um() const { return rlc_mode == pdcp_rlc_mode::um; }
105 bool is_am() const { return rlc_mode == pdcp_rlc_mode::am; }
106
107 /*
108 * Header and window helpers
109 */

111 const uint32_t window_size;
112
113 /*
114 * COUNT overflow helpers

116 bool max_count_notified = false;
117 bool max_count_overflow = false;
118
119 /*
120 * SN, HFN and COUNT helpers
121 */
122 const pdcp_sn_size sn_size;
123 uint32_t SN(uint32_t count) const { return pdcp_compute_sn(count, sn_size); }
124 uint32_t HFN(uint32_t count) const { return pdcp_compute_hfn(count, sn_size); }
125 uint32_t COUNT(uint32_t hfn, uint32_t sn) const { return pdcp_compute_count(hfn, sn, sn_size); }
