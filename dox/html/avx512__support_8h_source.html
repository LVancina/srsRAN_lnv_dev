65inline __m512i scale_epi8(__m512i a, float sf, uint8_t max)
66{
67 srsran_assert((sf > 0) && (sf <= 1), "Scaling factor out of range.");
68 srsran_assert(max < 127, "Parameter max out of range.");
69
70 if (sf >= .9999) {
71 return a;
72 }
73
74 static const __m512i mask_even_epi8 = _mm512_set1_epi16(0x00ff);
75 const __m512i MAX_epi8 = _mm512_set1_epi8(max);
76 const __m512i MIN_epi8 = _mm512_set1_epi8(-max);
77 // FLOAT2INT = 2^16 = 65536
78 static constexpr unsigned FLOAT2INT = 1U << 16U;
79
80 // Create a register filled with copies of the scaling factor times FLOAT2INT. Note that the upper bound on sf implies
81 // that (sf * FLOAT2INT) can be represented over 16 bits.
82 __m512i sf_epi16 = _mm512_set1_epi16(static_cast<uint16_t>(sf * FLOAT2INT));
83 // Even bytes in a, alternated with a zero-valued byte.
84 __m512i even_epi16 = _mm512_and_si512(a, mask_even_epi8);
85 // Odd bytes in a, moved to the even positions and alternated with a zero-valued byte.
86 __m512i odd_epi16 = _mm512_srli_epi16(a, 8);
87
88 // This is equivalent to multiply each byte by sf: indeed, we multiply each byte by (sf * FLOAT2INT), store the result
89 // in over 32 bit and pick only the 16 most significant bits (i.e., we divide by FLOAT2INT).
90 __m512i p_even_epi16 = _mm512_mulhi_epu16(even_epi16, sf_epi16);
91 __m512i p_odd_epi16 = _mm512_mulhi_epu16(odd_epi16, sf_epi16);
92
93 // Move the odd bits back to their original positions.
94 p_odd_epi16 = _mm512_slli_epi16(p_odd_epi16, 8);
95
96 // Combine even and odd bits. Note that the "odd" bytes of p_even_epi16 are equal to 0 (the result of multiplying a
97 // byte by a 16-bit value will occupy at most 24 of the 32 bits).
98 __m512i product_epi8 = _mm512_xor_si512(p_even_epi16, p_odd_epi16);
99
100 // Replace values corresponding to "large" inputs with their original values.
101 __mmask64 mask_epi8 = _mm512_cmpgt_epi8_mask(a, MAX_epi8);
102 product_epi8 = _mm512_mask_blend_epi8(mask_epi8, product_epi8, a);
103 mask_epi8 = _mm512_cmpgt_epi8_mask(MIN_epi8, a);
104 product_epi8 = _mm512_mask_blend_epi8(mask_epi8, product_epi8, a);
105
106 return product_epi8;
107}
