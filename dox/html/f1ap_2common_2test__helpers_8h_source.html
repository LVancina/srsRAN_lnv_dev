32inline bool is_f1ap_pdu_packable(const asn1::f1ap::f1ap_pdu_c& pdu)
33{
34 byte_buffer buffer;
35 asn1::bit_ref bref(buffer);
36 return pdu.pack(bref) == asn1::SRSASN_SUCCESS;
37}
