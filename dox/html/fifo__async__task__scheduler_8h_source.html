34class fifo_async_task_scheduler
35{
36public:
37 fifo_async_task_scheduler(size_t queue_size) : queue(queue_size) { run(); }
38 fifo_async_task_scheduler(const fifo_async_task_scheduler&) = delete;
39 fifo_async_task_scheduler& operator=(const fifo_async_task_scheduler&) = delete;
40
41 template <typename R>
42 bool schedule(async_task<R>&& t)
43 {
44 return queue.try_push(std::move(t));
45 }
46
47 template <typename AsyncTask, typename... Args>
48 bool schedule(Args&&... args)
49 {
50 return queue.try_push(launch_async<AsyncTask>(std::forward<Args>(args)...));
51 }
52
53 template <typename AsyncFunc>
54 bool schedule(AsyncFunc&& async_func)
55 {
56 return queue.try_push(launch_async(std::forward<AsyncFunc>(async_func)));
57 }
58
59 void clear_pending_tasks() { queue.clear(); }
60
61 eager_async_task<void> request_stop()
62 {
63 // Enqueue task in case main loop is waiting for new procedure
64 running = false;
65 queue.clear();
66 queue.try_push(launch_async([](coro_context<async_task<void>>& ctx) {
67 CORO_BEGIN(ctx);
68 CORO_RETURN();
69 }));
70 return std::move(loop_task);
71 }
72
73 size_t nof_pending_tasks() const { return queue.size(); }
74
75 bool empty() const { return queue.size() == 0; }
76
77 bool is_stopped() const { return loop_task.empty(); }
78
79private:
80 void run()
81 {
82 loop_task = launch_async([this](coro_context<eager_async_task<void>>& ctx) {
83 CORO_BEGIN(ctx);
84
85 // runs until requested to stop.
86 while (running) {
87 // Wait for new procedure to be enqueued.
88 CORO_AWAIT_VALUE(next_task, queue);
89
90 // Await for popped task to complete
91 CORO_AWAIT(std::move(next_task));
92 }
93
94 CORO_RETURN();
95 });
96 }
97
98 bool running = true;
99 async_queue<async_task<void>> queue;
100 eager_async_task<void> loop_task;
101 async_task<void> next_task;
102};

113template <typename Callback,
114 typename ReturnType = detail::function_return_t<decltype(&std::decay_t<Callback>::operator())>,
115 std::enable_if_t<std::is_same<ReturnType, void>::value, int> = 0>
116async_task<bool> when_completed_on_task_sched(fifo_async_task_scheduler& task_sched, Callback&& task_to_run)
117{
118 struct task_offloader {
119 task_offloader(fifo_async_task_scheduler& task_sched_, Callback&& callback_) :
120 task_sched(task_sched_), callback(std::forward<Callback>(callback_))
121 {
122 }
123
124 void operator()(coro_context<async_task<bool>>& ctx)
125 {
126 CORO_BEGIN(ctx);
127
128 task_sched.schedule(dispatched_task());
129
130 CORO_AWAIT_VALUE(const bool result, rx);
131
132 CORO_RETURN(result);
133 }
134
135 private:
136 async_task<void> dispatched_task()
137 {
138 return launch_async([this, tx = rx.get_sender()](coro_context<async_task<void>>& ctx) mutable {
139 CORO_BEGIN(ctx);
140
141 callback();
142
143 tx.set();
144
145 CORO_RETURN();
146 });
147 }
148
149 fifo_async_task_scheduler& task_sched;
150 Callback callback;
151 event_receiver<void> rx;
152 };
153
154 return launch_async<task_offloader>(task_sched, std::forward<Callback>(task_to_run));
155}
