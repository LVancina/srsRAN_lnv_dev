58struct dlsch_information {
60 sch_information sch;

63
68 float get_effective_code_rate() const
69 {
70 srsran_assert(nof_dl_sch_bits.value() != 0, "DL-SCH number of bits must not be zero.");
71 srsran_assert(sch.nof_bits_per_cb.value() > sch.nof_filler_bits_per_cb.value(),
72 "The number of bits per CB must be greater than the number of filler bits.");
73 return static_cast<float>((sch.nof_bits_per_cb.value() - sch.nof_filler_bits_per_cb.value()) * sch.nof_cb) /
74 static_cast<float>(nof_dl_sch_bits.value());
75 }
76};
