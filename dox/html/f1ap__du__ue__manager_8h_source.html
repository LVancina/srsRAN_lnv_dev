48 f1ap_du_ue& add_ue(du_ue_index_t ue_index)
49 {
50 srsran_assert(not ues.contains(ue_index), "Duplicate ueId={} detected", ue_index);
51
52 gnb_du_ue_f1ap_id_t f1ap_id = static_cast<gnb_du_ue_f1ap_id_t>(next_gnb_f1ap_du_ue_id++);
53 ues.emplace(
54 ue_index, ue_index, f1ap_id, du_handler, f1ap_msg_notifier, ctrl_exec, ue_exec_mapper.ctrl_executor(ue_index));
55
56 {
57 std::lock_guard<std::mutex> lock(map_mutex);
58 f1ap_ue_id_to_du_ue_id_map[f1ap_id] = ue_index;
59 }
60
61 return ues[ue_index];
62 }
63
64 bool remove_ue(du_ue_index_t ue_index)
65 {
66 if (not ues.contains(ue_index)) {
67 return false;
68 }
69 {
70 std::lock_guard<std::mutex> lock(map_mutex);
71 f1ap_ue_id_to_du_ue_id_map.erase(ues[ue_index].context.gnb_du_ue_f1ap_id);
72 }
73 ues.erase(ue_index);
74 return true;
75 }

83 f1ap_du_ue* find(gnb_du_ue_f1ap_id_t du_ue_f1ap_id)
84 {
85 std::lock_guard<std::mutex> lock(map_mutex);
86 auto it = f1ap_ue_id_to_du_ue_id_map.find(du_ue_f1ap_id);
87 return it != f1ap_ue_id_to_du_ue_id_map.end() ? find(it->second) : nullptr;
88 }
89 const f1ap_du_ue* find(gnb_du_ue_f1ap_id_t du_ue_f1ap_id) const
90 {
91 std::lock_guard<std::mutex> lock(map_mutex);
92 auto it = f1ap_ue_id_to_du_ue_id_map.find(du_ue_f1ap_id);
93 return it != f1ap_ue_id_to_du_ue_id_map.end() ? find(it->second) : nullptr;
94 }
95 const f1ap_du_ue* find(gnb_cu_ue_f1ap_id_t ue_cu_f1ap_id) const
96 {
97 std::lock_guard<std::mutex> lock(map_mutex);
98 auto it = std::find_if(ues.begin(), ues.end(), [ue_cu_f1ap_id](const f1ap_du_ue& e) {
99 return e.context.gnb_cu_ue_f1ap_id == ue_cu_f1ap_id;
100 });
101 return it != ues.end() ? &*it : nullptr;
102 }
103
104 bool contains(du_ue_index_t ue_index) const { return find(ue_index) != nullptr; }
105 bool contains(gnb_du_ue_f1ap_id_t du_ue_f1ap_id) const { return find(du_ue_f1ap_id) != nullptr; }
106
107 du_ue_index_t get_ue_index(gnb_du_ue_f1ap_id_t du_ue_f1ap_id) const
108 {
109 std::lock_guard<std::mutex> lock(map_mutex);
110 auto it = f1ap_ue_id_to_du_ue_id_map.find(du_ue_f1ap_id);
111 return it == f1ap_ue_id_to_du_ue_id_map.end() ? INVALID_DU_UE_INDEX : it->second;
112 }
113
