31class pdcch_encoder_spy : public pdcch_encoder
32{
33private:
34 struct entry_t {
35 config_t config;
36 std::vector<uint8_t> data;
37 std::vector<uint8_t> encoded;
38 };
39 std::vector<entry_t> entries;
40
41public:
42 void encode(span<uint8_t> encoded, span<const uint8_t> data, const config_t& config) override
43 {
44 // Generate new entry.
45 entry_t entry = {};
46 entry.config = config;
47 entry.encoded.resize(encoded.size());
48
49 // Fill data.
50 entry.data.resize(data.size());
51 std::copy(data.begin(), data.end(), entry.data.begin());
52
53 // Generate encoded message.
54 for (unsigned i = 0, len = encoded.size(); i != len; ++i) {
55 encoded[i] = static_cast<uint8_t>(i & 0xff);
56 entry.encoded[i] = encoded[i];
57 }
58
59 // Append entry.
60 entries.emplace_back(entry);
61 }
62
63 void reset() { entries.clear(); }
64 unsigned get_nof_entries() const { return entries.size(); }
65 const std::vector<entry_t>& get_entries() const { return entries; }
66};
