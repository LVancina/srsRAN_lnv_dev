45class mac_ul_processor final : public mac_ul_configurator, public mac_pdu_handler
46{
47public:
48 mac_ul_processor(const mac_ul_config& cfg_) :
49 cfg(cfg_),
50 logger(srslog::fetch_basic_logger("MAC")),
51 ue_manager(cfg.rnti_table),
52 pdu_handler(cfg.ul_ccch_notifier, cfg.ue_exec_mapper, cfg.sched, ue_manager, cfg.rnti_table, cfg.pcap)
53 {
54 }
55
56 async_task<bool> add_ue(const mac_ue_create_request& request) override
57 {
58 // Update UE executor to match new PCell.
59 cfg.ue_exec_mapper.rebind_executors(request.ue_index, request.cell_index);
60 task_executor& ul_exec = cfg.ue_exec_mapper.ctrl_executor(request.ue_index);
61
62 // Dispatch UE creation task to new UL executor.
63 return dispatch_and_resume_on(ul_exec, cfg.ctrl_exec, [this, request]() { return ue_manager.add_ue(request); });
64 }
65
66 virtual async_task<bool> addmod_bearers(du_ue_index_t ue_index,
67 const std::vector<mac_logical_channel_config>& ul_logical_channels) override
68 {
69 return dispatch_and_resume_on(
70 cfg.ue_exec_mapper.ctrl_executor(ue_index), cfg.ctrl_exec, [this, ue_index, ul_logical_channels]() {
71 return ue_manager.addmod_bearers(ue_index, ul_logical_channels);
72 });
73 }
74
75 async_task<bool> remove_bearers(du_ue_index_t ue_index, span<const lcid_t> lcids_to_rem) override
76 {
77 std::vector<lcid_t> lcids(lcids_to_rem.begin(), lcids_to_rem.end());
78 return dispatch_and_resume_on(
79 cfg.ue_exec_mapper.ctrl_executor(ue_index), cfg.ctrl_exec, [this, ue_index, lcids = std::move(lcids)]() {
80 return ue_manager.remove_bearers(ue_index, lcids);
81 });
82 }
83
84 async_task<void> remove_ue(const mac_ue_delete_request& msg) override
85 {
86 return dispatch_and_resume_on(cfg.ue_exec_mapper.ctrl_executor(msg.ue_index),
87 cfg.ctrl_exec,
88 [this, ue_index = msg.ue_index]() { ue_manager.remove_ue(ue_index); });
89 }
90
91 bool flush_ul_ccch_msg(du_ue_index_t ue_index, byte_buffer ccch_pdu) override
92 {
93 if (not cfg.ue_exec_mapper.ctrl_executor(ue_index).execute([this, ue_index, pdu = std::move(ccch_pdu)]() mutable {
94 pdu_handler.push_ul_ccch_msg(ue_index, std::move(pdu));
95 })) {
96 logger.warning("ue={}: Unable to forward UL-CCCH message to upper layers. Cause: task queue is full.", ue_index);
97 // Note: The UE is not yet created in the CU, so there in no inactivity timer.
98 return false;
99 }
100 return true;
101 }
102
105 void handle_rx_data_indication(mac_rx_data_indication msg) override
106 {
107 for (mac_rx_pdu& pdu : msg.pdus) {
108 if (pdu.pdu.empty()) {
109 logger.error("cell={} slot_rx={} rnti={}: Received empty MAC RX PDU from lower layers",
110 msg.cell_index,
111 msg.sl_rx,
112 pdu.rnti);
113 continue;
114 }
115
116 // > Convert C-RNTI to DU-specific UE index.
117 // Note: for Msg3, the UE context is not yet created, and ue_index will be an invalid index. This situation is
118 // handled inside the pdu_handler.
119 du_ue_index_t ue_index = cfg.rnti_table[pdu.rnti];
120
121 // > Fork each PDU handling to different executors based on the PDU RNTI.
122 if (not cfg.ue_exec_mapper.mac_ul_pdu_executor(ue_index).execute(
123 [this, slot_rx = msg.sl_rx, cell_idx = msg.cell_index, pdu = std::move(pdu)]() mutable {
124 // > Decode Rx PDU and handle respective subPDUs.
125 pdu_handler.handle_rx_pdu(slot_rx, cell_idx, std::move(pdu));
126 })) {
127 logger.warning(
128 "cell={} slot_rx={}: Discarding Rx PDU. Cause: Rx task queue is full.", msg.cell_index, msg.sl_rx);
129 }
130 }
131 }
132
133private:
134 const mac_ul_config cfg;
135 srslog::basic_logger& logger;
136
137 mac_ul_ue_manager ue_manager;
138
140 pdu_rx_handler pdu_handler;
141};
