125 void remove_ue(ue_index_t ue_index)
126 {
127 srsran_assert(ue_index != ue_index_t::invalid, "Invalid ue_index={}", ue_index);
128
129 if (ue_index_to_ue_e1ap_id.find(ue_index) == ue_index_to_ue_e1ap_id.end()) {
130 logger.warning("ue={}: GNB-CU-CP-UE-E1AP-ID not found", ue_index);
131 return;
132 }
133
134 // Remove UE from lookup
135 gnb_cu_cp_ue_e1ap_id_t cu_cp_ue_e1ap_id = ue_index_to_ue_e1ap_id.at(ue_index);
136 ue_index_to_ue_e1ap_id.erase(ue_index);
137
138 if (ues.find(cu_cp_ue_e1ap_id) == ues.end()) {
139 logger.warning("cu_cp_ue_e1ap_id={}: UE context not found", cu_cp_ue_e1ap_id);
140 return;
141 }
142
143 ues.at(cu_cp_ue_e1ap_id).logger.log_debug("Removing UE context");
144 ues.erase(cu_cp_ue_e1ap_id);
145 }

150 gnb_cu_cp_ue_e1ap_id_t allocate_gnb_cu_cp_ue_e1ap_id()
151 {
152 if (ue_index_to_ue_e1ap_id.size() == max_nof_supported_ues) {
153 return gnb_cu_cp_ue_e1ap_id_t::invalid;
154 }
155
156 // Check if the next_cu_cp_ue_e1ap_id is available
157 if (ues.find(next_cu_cp_ue_e1ap_id) == ues.end()) {
158 gnb_cu_cp_ue_e1ap_id_t ret = next_cu_cp_ue_e1ap_id;
159 // increase the next cu-cp ue e1ap id
160 increase_next_cu_cp_ue_e1ap_id();
161 return ret;
162 }
163
164 // Find holes in the allocated IDs by iterating over all ids starting with the next_cu_cp_ue_e1ap_id to find the
165 // available id
166 while (true) {
167 // Only iterate over ue_index_to_ue_e1ap_id (size=MAX NOF CU UEs)
168 // to avoid iterating over all possible values of gnb_cu_cp_ue_e1ap_id_t (size=2^32-1)
169 auto it = std::find_if(ue_index_to_ue_e1ap_id.begin(), ue_index_to_ue_e1ap_id.end(), [this](auto& u) {
170 return u.second == next_cu_cp_ue_e1ap_id;
171 });
172
173 // return the id if it is not already used
174 if (it == ue_index_to_ue_e1ap_id.end()) {
175 gnb_cu_cp_ue_e1ap_id_t ret = next_cu_cp_ue_e1ap_id;
176 // increase the next cu-cp ue e1ap id
177 increase_next_cu_cp_ue_e1ap_id();
178 return ret;
179 }
180
181 // increase the next cu-cp ue e1ap id and try again
182 increase_next_cu_cp_ue_e1ap_id();
183 }
184
185 return gnb_cu_cp_ue_e1ap_id_t::invalid;
186 }

188 void update_ue_index(ue_index_t new_ue_index, ue_index_t old_ue_index)
189 {
190 srsran_assert(new_ue_index != ue_index_t::invalid, "Invalid new_ue_index={}", new_ue_index);
191 srsran_assert(old_ue_index != ue_index_t::invalid, "Invalid old_ue_index={}", old_ue_index);
192 srsran_assert(ue_index_to_ue_e1ap_id.find(old_ue_index) != ue_index_to_ue_e1ap_id.end(),
193 "ue={}: GNB-CU-CP-UE-E1AP-ID not found",
194 old_ue_index);
195
196 // no need to update if the ue indexes are equal
197 if (new_ue_index == old_ue_index) {
198 return;
199 }
200
201 gnb_cu_cp_ue_e1ap_id_t cu_cp_ue_e1ap_id = ue_index_to_ue_e1ap_id.at(old_ue_index);
202
203 srsran_assert(
204 ues.find(cu_cp_ue_e1ap_id) != ues.end(), "cu_cp_ue_e1ap_id={}: UE context not found", cu_cp_ue_e1ap_id);
205
206 // Update UE context
207 ues.at(cu_cp_ue_e1ap_id).ue_ids.ue_index = new_ue_index;
208
209 // Update lookup
210 ue_index_to_ue_e1ap_id.emplace(new_ue_index, cu_cp_ue_e1ap_id);
211 ue_index_to_ue_e1ap_id.erase(old_ue_index);
212
213 // Update logger
214 ues.at(cu_cp_ue_e1ap_id)
215 .logger.set_prefix({ues.at(cu_cp_ue_e1ap_id).ue_ids.ue_index,
216 ues.at(cu_cp_ue_e1ap_id).ue_ids.cu_cp_ue_e1ap_id,
217 ues.at(cu_cp_ue_e1ap_id).ue_ids.cu_up_ue_e1ap_id});
218
219 ues.at(cu_cp_ue_e1ap_id).logger.log_debug("Updated UE index from ue_index={}", old_ue_index);
220 }

230 inline void increase_next_cu_cp_ue_e1ap_id()
231 {
232 if (next_cu_cp_ue_e1ap_id == gnb_cu_cp_ue_e1ap_id_t::max) {
233 // reset cu-cp ue e1ap id counter
234 next_cu_cp_ue_e1ap_id = gnb_cu_cp_ue_e1ap_id_t::min;
235 } else {
236 // increase cu-cp ue e1ap id counter
237 next_cu_cp_ue_e1ap_id = int_to_gnb_cu_cp_ue_e1ap_id(gnb_cu_cp_ue_e1ap_id_to_uint(next_cu_cp_ue_e1ap_id) + 1);
