31class ulsch_demultiplex_spy : public ulsch_demultiplex
32{
33public:
34 struct demultiplex_entry {
35 std::vector<log_likelihood_ratio> sch_data;
36 std::vector<log_likelihood_ratio> harq_ack;
37 std::vector<log_likelihood_ratio> csi_part1;
38 std::vector<log_likelihood_ratio> csi_part2;
39 unsigned nof_csi_part2_bits;
40 pusch_codeword_buffer_spy input;

44 ulsch_demultiplex_spy() : softbit_dist(log_likelihood_ratio::min().to_int(), log_likelihood_ratio::max().to_int())
45 {
46 // Do nothing.
47 }

49 void set_ulsch_config(unsigned nof_codeword_llr_,
50 unsigned nof_ul_sch_bits_,

52 unsigned nof_csi_part1_enc_bits_,
53 unsigned nof_csi_part2_enc_bits_)

55 nof_codeword_llr = nof_codeword_llr_;
56 nof_ul_sch_bits = nof_ul_sch_bits_;
57 nof_harq_ack_enc_bits = nof_harq_ack_enc_bits_;
58 nof_csi_part1_enc_bits = nof_csi_part1_enc_bits_;

62 void
63 set_csi_part2(pusch_decoder_buffer& csi_part2, unsigned nof_csi_part2_bits, unsigned nof_csi_part2_enc_bits_) override
64 {
65 srsran_assert(nof_csi_part2_enc_bits == nof_csi_part2_enc_bits_,
66 "Number of CSI Part 2 encoded bits do not match (i.e., {} != {}).",
67 nof_csi_part2_enc_bits_,
68 nof_csi_part2_enc_bits);
69
70 demultiplex_entry& entry = demultiplex_entries.back();
71 entry.nof_csi_part2_bits = nof_csi_part2_bits;
72 entry.csi_part2.resize(nof_csi_part2_enc_bits_);
73 generate_softbits(entry.csi_part2);
74
75 csi_part2.on_new_softbits(entry.csi_part2);
76 csi_part2.on_end_softbits();
77 }
78
79 pusch_codeword_buffer& demultiplex(pusch_decoder_buffer& sch_data,
80 pusch_decoder_buffer& harq_ack,
81 pusch_decoder_buffer& csi_part1,
82 const configuration& config) override
83 {
84 demultiplex_entries.emplace_back();
85
86 demultiplex_entry& entry = demultiplex_entries.back();
87 entry.config = config;
88
89 // Set codeword input.
90 entry.input = pusch_codeword_buffer_spy(nof_codeword_llr);
91
92 // Set SCH data.
93 entry.sch_data.resize(nof_ul_sch_bits);
94 generate_softbits(entry.sch_data);
95 sch_data.on_new_softbits(entry.sch_data);
96 sch_data.on_end_softbits();
97
98 // Set HARQ-ACK data.
99 entry.harq_ack.resize(nof_harq_ack_enc_bits);
100 generate_softbits(entry.harq_ack);
101 harq_ack.on_new_softbits(entry.harq_ack);
102 harq_ack.on_end_softbits();
103
104 // Set CSI Part 1 data.
105 entry.csi_part1.resize(nof_csi_part1_enc_bits);
106 generate_softbits(entry.csi_part1);
107 csi_part1.on_new_softbits(entry.csi_part1);
108 csi_part1.on_end_softbits();
109
110 return entry.input;
111 }
112
113 const std::vector<demultiplex_entry>& get_demultiplex_entries() const { return demultiplex_entries; }
114
115 void clear() { demultiplex_entries.clear(); }
116
117private:
118 void generate_softbits(span<log_likelihood_ratio> softbits)
119 {
120 std::generate(softbits.begin(), softbits.end(), [this]() { return log_likelihood_ratio(softbit_dist(rgen)); });
121 }
122
123 unsigned nof_codeword_llr = 0;
124 unsigned nof_ul_sch_bits = 0;
125 unsigned nof_harq_ack_enc_bits = 0;
126 unsigned nof_csi_part1_enc_bits = 0;
127 unsigned nof_csi_part2_enc_bits = 0;
128 std::mt19937 rgen;
129 std::uniform_int_distribution<int8_t> softbit_dist;
130 std::vector<demultiplex_entry> demultiplex_entries;
131};
