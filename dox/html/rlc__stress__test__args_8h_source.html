86inline bool parse_args(stress_test_args& args, int argc, char* argv[])
87{
88 static const struct option long_options[] = {
89 {"help", no_argument, nullptr, 'h'},
90 {"mode", required_argument, nullptr, 'm'},
91 {"sdu_size", required_argument, nullptr, 's'},
92 {"min_sdu_size", required_argument, nullptr, 'z'},
93 {"max_sdu_size", required_argument, nullptr, 'Z'},
94 {"const_opp", no_argument, nullptr, 'o'},
95 {"avg_opp_size", required_argument, nullptr, 'p'},
96 {"pdu_drop_rate", required_argument, nullptr, 'd'},
97 {"pdu_cut_rate", required_argument, nullptr, 'c'},
98 {"pdu_duplicate_rate", required_argument, nullptr, 'D'},
99 {"nof_ttis", required_argument, nullptr, 't'},
100 {"pdcp_sn_size", required_argument, nullptr, to_number(long_only::pdcp_sn_size)},
101 {"pdcp_t_reordering", required_argument, nullptr, to_number(long_only::pdcp_t_reordering)},
102 {"pdcp_integrity_algo", required_argument, nullptr, to_number(long_only::pdcp_integrity_algo)},
103 {"pdcp_ciphering_algo", required_argument, nullptr, to_number(long_only::pdcp_ciphering_algo)},
104 {"log_filename", required_argument, nullptr, 'l'},
105 {"log_stack_level", required_argument, nullptr, to_number(long_only::log_stack_level)},
106 {"log_traffic_level", required_argument, nullptr, to_number(long_only::log_traff_level)},
107 {"log_pdcp_level", required_argument, nullptr, to_number(long_only::log_pdcp_level)},
108 {"log_f1_level", required_argument, nullptr, to_number(long_only::log_f1_level)},
109 {"log_rlc_level", required_argument, nullptr, to_number(long_only::log_rlc_level)},
110 {"log_mac_level", required_argument, nullptr, to_number(long_only::log_mac_level)},
111 {"log_hex_limit", required_argument, nullptr, 'H'},
112 {"seed", required_argument, nullptr, 'S'},
113 {"nof_pdu_tti", required_argument, nullptr, 'T'},
114 {nullptr, 0, nullptr, 0}};
115 // clang-format off
116 static const char usage[] =
117 "Usage: rlc_stress_test [options]\n"
118 "\n"
119 " -h, --help Show help message and quit.\n"
120 " -m, --mode <RLC mode> Set RLC mode {TM, UM6, UM12, AM12, AM18}.\n"
121 " -s, --sdu_size <size> Set SDU size (-1 means random).\n"
122 " -z, --min_sdu_size <size> Set minimum SDU size (for randomly selected SDU size).\n"
123 " -Z, --max_sdu_size <size> Set maximum SDU size (for randomly selected SDU size).\n"
124 " -o, --const_opp Generate constant-size MAC opportunities (default: random).\n"
125 " -p, --avg_opp_size <size> Set average MAC opportunity size.\n"
126 " -d, --pdu_drop_rate <rate> Set rate at which RLC PDUs are dropped.\n"
127 " -c, --pdu_cut_rate <rate> Set rate at which RLC PDUs are chopped in length.\n"
128 " -D, --pdu_duplicate_rate <rate> Set rate at which RLC PDUs are dropped.\n"
129 " -l, --nof_ttis <ttis> Set number of TTIs to emulate.\n"
130 " -l, --log_filename <filename> Set log filename. Use 'stdout' to print to console.\n"
131 " --pdcp_sn_size <level> Set PDCP SN size.\n"
132 " --pdcp_t_reordering <timeout> Set PDCP t-Reordering timeout (ms).\n"
133 " --pdcp_integrity_algo <algo> Set PDCP NIA algo (default NIA2).\n"
134 " --pdcp_ciphering_algo <algo> Set PDCP NEA algo (default NEA2).\n"
135 " --log_stack_level <level> Set STACK log level (default: info).\n"
136 " --log_traffic_level <level> Set STACK log level (default: info).\n"
137 " --log_pdcp_level <level> Set PDCP log level (default: error).\n"
138 " --log_f1_level <level> Set F1 log level (default: error).\n"
139 " --log_rlc_level <level> Set RLC log level (default: error).\n"
140 " --log_mac_level <level> Set MAC log level (default: error).\n"
141 " --log_hex_limit <hex_limit> Set log limit for hex dumps (default: 32 bytes).\n"
142 " -S, --seed <seed> Set seed to use in run. 0 means the seed is randomly generated.\n"
143 " -T, --nof_pdu_tti <num> Set number of PDUs processed in a TTI.\n"
144 "\n";
145 // clang-format on
146
147 // Parse arguments
148 while (true) {
149 int option_index = 0;
150 int c = getopt_long(argc, argv, "hm:s:z:Z:op:d:c:D:t:l:H:S:T:", long_options, &option_index);
151 if (c == -1) {
152 break;
153 }
154
155 switch (c) {
156 case 'h':
157 fprintf(stdout, "%s", usage);
158 exit(0);
159 case 'm':
160 args.mode = std::string(optarg);
161 fprintf(stdout, "RLC Mode %s\n", args.mode.c_str());
162 break;
163 case 's':
164 args.sdu_size = std::strtol(optarg, nullptr, 10);
165 fprintf(stdout, "SDU size %d\n", args.sdu_size);
166 break;
167 case 'z':
168 args.min_sdu_size = std::strtol(optarg, nullptr, 10);
169 fprintf(stdout, "Minimum SDU size %d\n", args.min_sdu_size);
170 break;
171 case 'Z':
172 args.max_sdu_size = std::strtol(optarg, nullptr, 10);
173 fprintf(stdout, "Maximum SDU size %d\n", args.max_sdu_size);
174 break;
175 case 'o':
176 args.const_opp = true;
177 fprintf(stdout, "Using constant-size MAC opportunities\n");
178 break;
179 case 'p':
180 args.avg_opp_size = std::strtol(optarg, nullptr, 10);
181 fprintf(stdout, "Average MAC opportunity size %d\n", args.avg_opp_size);
182 break;
183 case 'd':
184 args.pdu_drop_rate = std::strtod(optarg, nullptr);
185 fprintf(stdout, "RLC PDU drop rate %f\n", args.pdu_drop_rate);
186 break;
187 case 'c':
188 args.pdu_cut_rate = std::strtod(optarg, nullptr);
189 fprintf(stdout, "RLC PDU cut rate %f\n", args.pdu_cut_rate);
190 break;
191 case 'D':
192 args.pdu_duplicate_rate = std::strtod(optarg, nullptr);
193 fprintf(stdout, "RLC PDU duplicate rate %f\n", args.pdu_duplicate_rate);
194 break;
195 case 't':
196 args.nof_ttis = std::strtol(optarg, nullptr, 10);
197 fprintf(stdout, "Number of TTIs to emulate %u\n", args.nof_ttis);
198 break;
199 case 'l':
200 args.log_filename = std::string(optarg);
201 fprintf(stdout, "Log filename %s\n", args.log_filename.c_str());
202 break;
203 case to_number(long_only::pdcp_sn_size):
204 args.pdcp_sn_size = std::strtol(optarg, nullptr, 10);
205 fprintf(stdout, "PDCP SN Size %s\n", optarg);
206 break;
207 case to_number(long_only::pdcp_t_reordering):
208 args.pdcp_t_reordering = std::strtol(optarg, nullptr, 10);
209 fprintf(stdout, "PDCP t-Reordering %s\n", optarg);
210 break;
211 case to_number(long_only::pdcp_integrity_algo):
212 args.pdcp_integrity_algo = std::strtol(optarg, nullptr, 10);
213 fprintf(stdout, "PDCP integrity algorithm %s\n", optarg);
214 break;
215 case to_number(long_only::pdcp_ciphering_algo):
216 args.pdcp_ciphering_algo = std::strtol(optarg, nullptr, 10);
217 fprintf(stdout, "PDCP ciphering algorithm %s\n", optarg);
218 break;
219 case to_number(long_only::log_stack_level):
220 args.log_level_stack = srslog::str_to_basic_level(std::string(optarg));
221 fprintf(stdout, "STACK log level %s\n", optarg);
222 break;
223 case to_number(long_only::log_traff_level):
224 args.log_level_traff = srslog::str_to_basic_level(std::string(optarg));
225 fprintf(stdout, "TRAFF log level %s\n", optarg);
226 break;
227 case to_number(long_only::log_pdcp_level):
228 args.log_level_pdcp = srslog::str_to_basic_level(std::string(optarg));
229 fprintf(stdout, "PDCP log level %s\n", optarg);
230 break;
231 case to_number(long_only::log_f1_level):
232 args.log_level_f1 = srslog::str_to_basic_level(std::string(optarg));
233 fprintf(stdout, "F1 log level %s\n", optarg);
234 break;
235 case to_number(long_only::log_rlc_level):
236 args.log_level_rlc = srslog::str_to_basic_level(std::string(optarg));
237 fprintf(stdout, "RLC log level %s\n", optarg);
238 break;
239 case to_number(long_only::log_mac_level):
240 args.log_level_mac = srslog::str_to_basic_level(std::string(optarg));
241 fprintf(stdout, "MAC log level %s\n", optarg);
242 break;
243 case 'H':
244 args.log_hex_limit = std::strtol(optarg, nullptr, 10);
245 fprintf(stdout, "Log hex limit %d\n", args.log_hex_limit);

249 fprintf(stdout, "Seed %d\n", args.seed);

255 default:
256 fprintf(stderr, "error parsing arguments\n");

258 }
259 }
260 return true;

263inline security::sec_128_as_config get_security_config_from_args(const stress_test_args& args)
264{
265 security::sec_128_as_config sec_cfg = {};
266
267 // Set security domain

269
270 // Set security keys

277

281inline pdcp_config get_pdcp_config_from_args(uint32_t id, const stress_test_args& args)
282{
283 pdcp_config cnfg = {};
284 cnfg.rb_type = pdcp_rb_type::drb;
285 if (args.pdcp_sn_size == 12) {
286 cnfg.tx.sn_size = srsran::pdcp_sn_size::size12bits;
287 cnfg.rx.sn_size = srsran::pdcp_sn_size::size12bits;
288 } else if (args.pdcp_sn_size == 18) {
289 cnfg.tx.sn_size = srsran::pdcp_sn_size::size18bits;
290 cnfg.rx.sn_size = srsran::pdcp_sn_size::size18bits;
291 } else {
292 fprintf(stderr, "Unsupported PDCP SN %d, exiting.\n", args.pdcp_sn_size);
293 exit(-1);
294 }
295 if (id == 0) {
296 cnfg.tx.direction = srsran::pdcp_security_direction::downlink;
297 cnfg.rx.direction = srsran::pdcp_security_direction::uplink;
298 } else {
299 cnfg.tx.direction = srsran::pdcp_security_direction::uplink;
300 cnfg.rx.direction = srsran::pdcp_security_direction::downlink;
301 }
302 cnfg.rx.t_reordering = static_cast<pdcp_t_reordering>(args.pdcp_t_reordering);
303 return cnfg;
304}

306inline rlc_config get_rlc_config_from_args(const stress_test_args& args)
307{
308 rlc_config cnfg = {};
309 if (args.mode == "TM") {
310 cnfg.mode = srsran::rlc_mode::tm;
311 } else if (args.mode == "UM6") {
312 cnfg.mode = srsran::rlc_mode::um_bidir;
313 cnfg.um.rx.sn_field_length = rlc_um_sn_size::size6bits;
314 cnfg.um.tx.sn_field_length = rlc_um_sn_size::size6bits;
315 } else if (args.mode == "UM12") {
316 cnfg.mode = srsran::rlc_mode::um_bidir;
317 cnfg.um.rx.sn_field_length = rlc_um_sn_size::size12bits;
318 cnfg.um.tx.sn_field_length = rlc_um_sn_size::size12bits;
319 } else if (args.mode == "AM12") {
320 cnfg.mode = srsran::rlc_mode::am;
321 cnfg.am.rx.sn_field_length = rlc_am_sn_size::size12bits;
322 cnfg.am.tx.sn_field_length = rlc_am_sn_size::size12bits;
323 cnfg.am.tx.t_poll_retx = 50;
324 cnfg.am.tx.poll_pdu = 10;
325 cnfg.am.tx.poll_byte = 5000;
326 cnfg.am.rx.t_reassembly = 40;
327 cnfg.am.rx.t_status_prohibit = 8;
328 } else if (args.mode == "AM18") {
329 cnfg.mode = srsran::rlc_mode::am;
330 cnfg.am.rx.sn_field_length = rlc_am_sn_size::size18bits;
331 cnfg.am.tx.sn_field_length = rlc_am_sn_size::size18bits;
332 cnfg.am.tx.t_poll_retx = 50;
333 cnfg.am.tx.poll_pdu = 10;
334 cnfg.am.tx.poll_byte = 5000;
335 cnfg.am.rx.t_reassembly = 40;
336 cnfg.am.rx.t_status_prohibit = 8;
337 } else {
338 fprintf(stderr, "Unsupported RLC mode %s, exiting.\n", args.mode.c_str());
339 exit(-1);
340 }
341 cnfg.metrics_period = std::chrono::milliseconds(1000);
342 return cnfg;
343}

345inline void init_log_from_args(const stress_test_args& args)
346{
347 srslog::init();
348
349 srslog::sink* log_sink =
350 (args.log_filename == "stdout") ? srslog::create_stdout_sink() : srslog::create_file_sink(args.log_filename);
351 if (log_sink == nullptr) {
352 return;
353 }
354 srslog::log_channel* chan = srslog::create_log_channel("main_channel", *log_sink);
355 if (chan == nullptr) {
356 return;
357 }
358 srslog::set_default_sink(*log_sink);
359
360 auto& log_stack = srslog::fetch_basic_logger("STACK", false);
361 log_stack.set_level(args.log_level_stack);
362 log_stack.set_hex_dump_max_size(args.log_hex_limit);
363
364 auto& log_traff = srslog::fetch_basic_logger("TRAFF", false);
365 log_traff.set_level(args.log_level_traff);
366 log_traff.set_hex_dump_max_size(args.log_hex_limit);
367
368 auto& log_pdcp = srslog::fetch_basic_logger("PDCP", false);
369 log_pdcp.set_level(args.log_level_pdcp);
370 log_pdcp.set_hex_dump_max_size(args.log_hex_limit);
371
372 auto& log_f1 = srslog::fetch_basic_logger("F1AP", false);
373 log_f1.set_level(args.log_level_pdcp);
374 log_f1.set_hex_dump_max_size(args.log_hex_limit);
375
376 auto& log_rlc = srslog::fetch_basic_logger("RLC", false);
377 log_rlc.set_level(args.log_level_rlc);
378 log_rlc.set_hex_dump_max_size(args.log_hex_limit);
379
380 auto& log_mac = srslog::fetch_basic_logger("MAC", false);
381 log_mac.set_level(args.log_level_mac);
382 log_mac.set_hex_dump_max_size(args.log_hex_limit);
383}
