31class lcid_ul_sch_t
32{
33 using underlying_type = std::underlying_type_t<lcid_t>;
34
35public:
37 enum options : underlying_type {
39 CCCH_SIZE_64 = 0b000000,
40
42 LCID1 = 1,
43 // ...
44 LCID32 = 32,
45
47 MIN_RESERVED = 33,
48 MAX_RESERVED = 51,
49
51 CCCH_SIZE_48 = 0b110100,
52
53 BIT_RATE_QUERY = 0b110101,
54
55 SE_PHR = 0b111001, // Single Entry PHR
56 CRNTI = 0b111010,
57 SHORT_TRUNC_BSR = 0b111011,
58 LONG_TRUNC_BSR = 0b111100,
59 SHORT_BSR = 0b111101,
60 LONG_BSR = 0b111110,
61 PADDING = 0b111111
62 };
63
64 lcid_ul_sch_t(underlying_type lcid_ = PADDING) : lcid_val(lcid_) { srsran_assert(lcid_ <= PADDING, "Invalid LCID"); }
65 lcid_ul_sch_t& operator=(underlying_type lcid)
66 {
67 srsran_assert(lcid <= PADDING, "Invalid LCID");
68 lcid_val = lcid;
69 return *this;
70 }
71
72 explicit operator underlying_type() const { return lcid_val; }
73 underlying_type value() const { return lcid_val; }
74
76 bool is_ccch() const { return (lcid_val == CCCH_SIZE_48 || lcid_val == CCCH_SIZE_64); }
77
79 bool is_ce() const { return lcid_val <= PADDING and lcid_val >= BIT_RATE_QUERY; }
80
82 bool is_sdu() const { return lcid_val <= LCID32 and lcid_val >= LCID1; }
83
85 bool is_valid_lcid() const { return lcid_val <= PADDING and (lcid_val < MIN_RESERVED or lcid_val > MAX_RESERVED); }
86
88 bool has_length_field() const
89 {
90 // CCCH (both versions) don't have a length field in the UL
91 if (is_ccch()) {
92 return false;
93 }
94 return (is_sdu() || is_var_len_ce());
95 }
96
97 bool is_var_len_ce() const
98 {
99 switch (lcid_val) {
100 case LONG_TRUNC_BSR:
101 case LONG_BSR:
102 return true;
103 default:
104 return false;
105 }
106 }
107
108 uint32_t sizeof_ce() const
109 {
110 switch (lcid_val) {
111 case CCCH_SIZE_48:
112 return 6;
113 case CCCH_SIZE_64:
114 return 8;
115 case CRNTI:
116 return 2;
117 case SHORT_BSR:
118 case SHORT_TRUNC_BSR:
119 return 1;
120 case SE_PHR:
121 return 2;
122 // NOTE: LONG_BSR and LONG_TRUNC_BSR are variable-sized MAC CE, not fixed-sized. Right now this function is not
123 // called for these two cases.
124 // TODO: Verify if these two cases will be used in the future by this function.
125 case LONG_BSR:
126 case LONG_TRUNC_BSR:
127 return 1; // minimum size, could be more than that
128 case PADDING:
129 return 0;
130 default:
131 break;
132 }
133 return 0;
134 }
135
136 bool operator==(lcid_ul_sch_t other) const { return lcid_val == other.lcid_val; }
137 bool operator!=(lcid_ul_sch_t other) const { return lcid_val != other.lcid_val; }
138
139private:
140 underlying_type lcid_val;
141};
