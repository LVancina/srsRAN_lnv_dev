40enum class rx_buffer_status : uint8_t { successful = 0, already_in_use, insufficient_cb };
41
43class rx_buffer_impl : public unique_rx_buffer::callback
44{
45private:
47 enum class state : uint8_t {
51 available,
57 reserved,
65 locked
66 };
67
69 std::atomic<state> current_state = {state::available};
71 rx_buffer_codeblock_pool& codeblock_pool;
73 static_vector<bool, MAX_NOF_SEGMENTS> crc;
75 static_vector<unsigned, MAX_NOF_SEGMENTS> codeblock_ids;
76
78 void free()
79 {
80 // Free all codeblocks.
81 for (unsigned cb_id : codeblock_ids) {
82 codeblock_pool.free(cb_id);
83 }
84
85 // Indicate the buffer is available by clearing the codeblocks identifiers.
86 codeblock_ids.clear();
87 }
88
89public:
92 explicit rx_buffer_impl(rx_buffer_codeblock_pool& pool) : codeblock_pool(pool)
93 {
94 // Do nothing.
95 }
96
98 rx_buffer_impl(const rx_buffer_impl& other) noexcept : codeblock_pool(other.codeblock_pool) {}
99
101 rx_buffer_impl(rx_buffer_impl&& other) noexcept : codeblock_pool(other.codeblock_pool) {}

110 rx_buffer_status reserve(unsigned nof_codeblocks, bool reset_crc)
111 {
112 // It cannot reserve resources if it is locked.
113 if (current_state.load() == state::locked) {
114 return rx_buffer_status::already_in_use;
115 }
116
117 // If the current number of codeblocks is larger than required, free the excess of codeblocks.
118 while (codeblock_ids.size() > nof_codeblocks) {
119 // Get the codeblock identifier at the back and remove from the list.
120 unsigned cb_id = codeblock_ids.back();
121 codeblock_ids.pop_back();
122
123 // Free the codeblock.
124 codeblock_pool.free(cb_id);
125 }
126
127 // If the current number of codeblocks is less than required, reserve the remaining codeblocks.
128 while (codeblock_ids.size() < nof_codeblocks) {
129 // Reserve codeblock.
130 optional<unsigned> cb_id = codeblock_pool.reserve();
131
132 // Free the entire buffer if one codeblock cannot be reserved.
133 if (!cb_id.has_value()) {
134 free();
135 current_state = state::available;
136 return rx_buffer_status::insufficient_cb;
137 }
138
139 // Append the codeblock identifier to the list.
140 codeblock_ids.push_back(cb_id.value());
141 }
142
143 // Resize CRCs.
144 crc.resize(nof_codeblocks);
145
146 // Reset CRCs if necessary.
147 if (reset_crc) {
148 reset_codeblocks_crc();
149 }
150 current_state = state::reserved;
151
152 return rx_buffer_status::successful;
153 }

199 codeblock_id,

203 srsran_assert(

205 return codeblock_pool.get_data_bits(cb_id).first(data_size);
206 }

208 // See interface for documentation.
209 void lock() override
210 {
211 state previous_state = current_state.exchange(state::locked);
212 srsran_assert(previous_state == state::reserved, "Failed to lock. Invalid state.");
213 }
214
215 // See interface for documentation.
216 void unlock() override
217 {
218 state previous_state = current_state.exchange(state::reserved);
219 srsran_assert(previous_state == state::locked, "Failed to unlock. Invalid state.");
220 }
221
222 // See interface for documentation.
223 void release() override
224 {
225 // Release all reserved codeblocks.
226 free();
227
228 // The buffer can now be reused again.
229 state previous_state = current_state.exchange(state::available);
230 srsran_assert(previous_state == state::locked, "Failed to release. Invalid state.");
231 }
232
234 bool is_free() const { return current_state == state::available; }
235
237 bool is_locked() const { return current_state == state::locked; }
238
245 bool expire()
246 {
247 state expected_state = state::reserved;
248 bool from_reserved = current_state.compare_exchange_weak(expected_state, state::available);
249
250 // The buffer cannot be freed if it is locked.
251 if (!from_reserved) {
252 return false;
253 }
254
255 // Release resources.
256 free();
257
258 return true;
259 }
260};
