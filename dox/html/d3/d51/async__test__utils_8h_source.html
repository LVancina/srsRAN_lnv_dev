33/// Factory of async tasks that await on an external event.
34template <typename Result>
35class wait_manual_event_tester
36{
37public:
38 wait_manual_event_tester() {}
39 wait_manual_event_tester(const Result& r) : result(r) {}
40 wait_manual_event_tester(Result&& r) : result(std::move(r)) {}
41
42 async_task<Result> launch()
43 {
44 return launch_async([this](coro_context<async_task<Result>>& ctx) {
45 CORO_BEGIN(ctx);
46 CORO_AWAIT(ready_ev);
47 CORO_RETURN(result);
48 });
49 }
50
51 manual_event_flag ready_ev;
52 Result result;
53};
