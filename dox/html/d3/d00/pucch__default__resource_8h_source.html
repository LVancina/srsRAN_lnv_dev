29/// \file
30/// \brief Procedures for selecting PUCCH default resource.
31///
32/// Helper files for selecting the PUCCH resource before dedicated resources are configured as described in TS38.213
33/// Section 9.2.1.
34///
35namespace srsran {
36
37/// Collects the default PUCCH resource parameters given by TS38.213 Table 9.2.1-1.
38struct pucch_default_resource {
39 /// PUCCH format.
40 pucch_format format;
41 /// First symbol index within the slot.
42 unsigned first_symbol_index;
43 /// Number of symbols.
44 unsigned nof_symbols;
45 /// PRB offset, parameter \f$RB_{BWP}^{offset}\f$ in TS38.213 Section 9.2.1.
46 unsigned rb_bwp_offset;
47 /// Set of initial cyclic shift indexes. The number of elements provides the parameter \f$N_{CS}\f$ in TS38.213
48 /// Section 9.2.1.
49 static_vector<uint8_t, 4> cs_indexes;
50};
51
52/// \brief Calculates the default PUCCH resource index \f$r_{PUCCH}\f$.
53///
54/// It implemented is according to TS38.213 Section 9.2.1.
55///
56/// \param[in] n_cce Index of the first CCE for the PDCCH reception, parameter \f$n_{CCE,0}\f$.
57/// \param[in] nof_cce Number of CCEs in a CORESET of the PDCCH reception with DCI associated to the PUCCH
58/// transmission, Parameter \f$N_{CCE}\f$.
59/// \param[in] delta_pri PUCCH resource indicator field in the associated DCI, parameter \f$\Delta _{PRI}\f$.
60/// \return PUCCH resource index in with a range of {0, ..., 15}.
61unsigned get_pucch_default_resource_index(unsigned n_cce, unsigned nof_cce, unsigned delta_pri);
62
63/// \brief Gets the PRB indexes for PUCCH resource before dedicated PUCCH as per TS38.213 Section 9.2.1.
64/// \param[in] r_pucch PUCCH resource index, parameter \f$r_{PUCCH}\f$.
65/// \param[in] rb_bwp_offset PRB offset, parameter \f$RB_{BWP}^{offset}\f$.
66/// \param[in] nof_cs Number of cyclic shifts, parameter \f$N_{CS}\f$.
67/// \param[in] N_bwp_size BWP size, parameter \f$N_{BWP}^{size}\f$.
68/// \return A pair of values containing the PRB index for the first and second hops.
69std::pair<unsigned, unsigned>
70get_pucch_default_prb_index(unsigned r_pucch, unsigned rb_bwp_offset, unsigned nof_cs, unsigned N_bwp_size);
71
72/// \brief Gets the PUCCH cyclic shift for PUCCH resource before dedicated PUCCH as per TS38.213 Section 9.2.1.
73/// \param[in] r_pucch PUCCH resource index, parameter \f$r_{PUCCH}\f$.
74/// \param[in] nof_cs Number of cyclic shifts, parameter \f$N_{CS}\f$.
75/// \return The PUCCH resource cyclic shift.
76unsigned get_pucch_default_cyclic_shift(unsigned r_pucch, unsigned nof_cs);
77
78/// \brief Gets the PUCCH resource parameters before dedicated PUCCH resources are configured.
79///
80/// The values are given by TS38.213 Table 9.2.1-1.
81///
82/// \param[in] row_index PUCCH default row index, higher layer parameter \e pucch-ResourceCommon (TS38.331
83/// Section 6.3.2, Information Element \e PUCCH-ConfigCommon). Possible values are {0, ... , 15}.
84/// \param[in] bwp_size Active BWP size, parameter \f$N_{BWP}^{size}\f$ in TS38.213 Section 9.2.1.
85/// \return A valid default PUCCH resource.
86/// \remark An assertion is triggered if the row index exceeds its boundaries.
87pucch_default_resource get_pucch_default_resource(unsigned row_index, unsigned N_bwp_size);
