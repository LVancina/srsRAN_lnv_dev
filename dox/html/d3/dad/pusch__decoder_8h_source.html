41/// \brief PUSCH decoder interface.
42///
43/// Recovers a UL-SCH transport block from a PUSCH codeword. Reverting the encoding operations described in TS38.212
44/// Sections 6.2.1-6.2.6, the codeword is first split into rate-matched codeblocks. Then, each codeblock is restored
45/// to its base rate, combined with previous retransmissions, and decoded. Finally, if all blocks pass the CRC check,
46/// the data bits from all codeblocks are concatenated to form the UL-SCH transport block. If applicable, a last
47/// transport-block CRC is computed and verified.
48///
49/// The PUSCH decoder is configured using the method \ref new_data. This method returns a \ref pusch_decoder_buffer
50/// which is used to write softbits into the decoder.
51///
52/// The decoder could potentially start the code block processing if a codeword length is provided (see
53/// \ref set_codeword_length) prior \ref pusch_decoder_buffer::on_end_softbits.
54class pusch_decoder
55{
56public:
57 /// Collects the parameters necessary for decoding a PUSCH transport block.
58 struct configuration {
59 /// Code base graph.
60 ldpc_base_graph_type base_graph = ldpc_base_graph_type::BG1;
61 /// Redundancy version, values in {0, 1, 2, 3}.
62 unsigned rv = 0;
63 /// Modulation scheme.
64 modulation_scheme mod = modulation_scheme::BPSK;
65 /// \brief Limited buffer rate matching length in bits, as per TS38.212 Section 5.4.2.
66 /// \note Set to zero for unlimited buffer length.
67 unsigned Nref = 0;
68 /// Number of transmission layers the transport block is mapped onto.
69 unsigned nof_layers = 0;
70 /// Maximum number of iterations of the LDPC decoder.
71 unsigned nof_ldpc_iterations = 6;
72 /// Flag for LDPC decoder early stopping: \c true to activate.
73 bool use_early_stop = true;
74 /// Flag to denote new data (first HARQ transmission).
75 bool new_data = true;
76 };
77
78 /// Default destructor.
79 virtual ~pusch_decoder() = default;
80
81 /// \brief Decodes a PUSCH codeword.
82 /// \param[out] transport_block The decoded transport block, with packed (8 bits per entry) representation.
83 /// \param[in,out] rm_buffer A buffer for combining log-likelihood ratios from different retransmissions.
84 /// \param[in] notifier Interface for notifying the completion of the TB processing.
85 /// \param[in] cfg Decoder configuration parameters.
86 /// \return A \ref pusch_decoder_buffer, used to write softbits into the decoder.
87 virtual pusch_decoder_buffer& new_data(span<uint8_t> transport_block,
88 unique_rx_buffer rm_buffer,
89 pusch_decoder_notifier& notifier,
90 const configuration& cfg) = 0;
91
92 /// \brief Sets the number of UL-SCH codeword softbits expected by the PUSCH decoder.
93 ///
94 /// It allows the decoder to start decoding codeblocks before receiving the entire codeword. If it is called before
95 /// \ref new_data or after decoding has started, it has no effect.
96 ///
97 /// \param[in] nof_softbits Number of codeword softbits, parameter \f$G^\textup{UL-SCH}\f$ in TS38.212 Section 6.2.7.
98 virtual void set_nof_softbits(units::bits nof_softbits) = 0;
99};
