32/// \brief Implement baseband gateway interface for UHD.
33///
34/// It contains a transmit stream and a receive stream.
35class radio_uhd_baseband_gateway : public baseband_gateway
36{
37public:
38 radio_uhd_baseband_gateway(radio_uhd_device& device,
39 task_executor& async_executor,
40 radio_notification_handler& notifier,
41 const radio_uhd_tx_stream::stream_description& tx_stream_config,
42 const radio_uhd_rx_stream::stream_description& rx_stream_config) :
43 tx_stream(device.create_tx_stream(async_executor, notifier, tx_stream_config)),
44 rx_stream(device.create_rx_stream(notifier, rx_stream_config))
45 {
46 }
47
48 // See interface for documentation.
49 baseband_gateway_transmitter& get_transmitter() override { return *tx_stream; }
50
51 // See interface for documentation.
52 baseband_gateway_receiver& get_receiver() override { return *rx_stream; }
53
54 // See interface for documentation.
55 unsigned get_transmitter_optimal_buffer_size() const override { return tx_stream->get_buffer_size(); }
56
57 // See interface for documentation.
58 unsigned get_receiver_optimal_buffer_size() const override { return rx_stream->get_buffer_size(); }
59
60 /// Gets the transmit stream instance.
61 radio_uhd_tx_stream& get_tx_stream() { return *tx_stream; }
62
63 /// Gets the transmit stream.
64 radio_uhd_rx_stream& get_rx_stream() { return *rx_stream; }
65
66 /// Checks if the streams are successfully initialized.
67 bool is_successful() const
68 {
69 if (!tx_stream || !rx_stream) {
70 return false;
71 }
72
73 return tx_stream->is_successful() && rx_stream->is_successful();
74 }
75
76private:
77 /// Transmit stream.
78 std::unique_ptr<radio_uhd_tx_stream> tx_stream;
79 /// Receive stream.
80 std::unique_ptr<radio_uhd_rx_stream> rx_stream;
81};
