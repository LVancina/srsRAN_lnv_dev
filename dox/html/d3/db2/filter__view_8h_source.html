33/// \brief This class provides a range (with begin and end) that filters out elements from another range for which
34/// FilterFunc returns false.
35/// \tparam Range
36/// \tparam FilterFunc
37template <typename Range, typename FilterFunc>
38class filter_view
39{
40 template <typename Data>
41 class iter_impl
42 {
43 using range_it_t = std::conditional_t<std::is_const<Data>::value,
44 typename std::decay_t<Range>::const_iterator,
45 typename std::decay_t<Range>::iterator>;
46
47 public:
48 using value_type = std::remove_const_t<Data>;
49 using reference = Data&;
50 using pointer = Data*;
51 using difference_type = std::ptrdiff_t;
52 using iterator_category = std::forward_iterator_tag;
53
54 iter_impl(filter_view<Range, FilterFunc>& parent_, range_it_t it_) : parent(&parent_), it(it_) { next_valid_it(); }
55
56 iter_impl<Data>& operator++()
57 {
58 srsran_sanity_check(it != parent->r.end(), "Iterator incremented past end() point");
59 ++it;
60 next_valid_it();
61 return *this;
62 }
63
64 value_type& operator*() { return *it; }
65 value_type* operator->() { return &(*it); }
66
67 bool operator==(const iter_impl<Data>& other) const { return it == other.it; }
68 bool operator!=(const iter_impl<Data>& other) const { return it != other.it; }
69
70 private:
71 void next_valid_it()
72 {
73 while (it != parent->r.end() and not parent->pred(*it)) {
74 ++it;
75 }
76 }
77
78 filter_view<Range, FilterFunc>* parent;
79 range_it_t it;
80 };
81
82public:
83 using value_type = typename std::decay_t<Range>::value_type;
84 using iterator = iter_impl<value_type>;
85 using const_iterator = iter_impl<const value_type>;
86
87 filter_view(Range r_, FilterFunc pred_) : r(std::forward<Range>(r_)), pred(std::forward<FilterFunc>(pred_)) {}
88
89 iterator begin() { return iterator{*this, r.begin()}; }
90 iterator end() { return iterator{*this, r.end()}; }
91 const_iterator begin() const { return iterator{*this, r.begin()}; }
92 const_iterator end() const { return iterator{*this, r.end()}; }
93
94private:
95 Range r;
96 FilterFunc pred;
97};
98
99template <typename Range, typename Pred>
100filter_view<Range, Pred> filter(Range&& r, Pred&& p)
101{
102 return {std::forward<Range>(r), std::forward<Pred>(p)};
103}
