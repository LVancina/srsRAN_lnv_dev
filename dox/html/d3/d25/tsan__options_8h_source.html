25// Options taken from Mozilla project
26// abort_on_error=1 - Causes TSan to abort instead of using exit().
27// halt_on_error=1 - Causes TSan to stop on the first race detected.
28//
29// report_signal_unsafe=0 - Required to avoid TSan deadlocks when
30// receiving external signals (e.g. SIGINT manually on console).
31//
32// allocator_may_return_null=1 - Tell TSan to return NULL when an allocation
33// fails instead of aborting the program. This allows us to handle failing
34// allocations the same way we would handle them with a regular allocator and
35// also uncovers potential bugs that might occur in these situations.
36
37#ifdef __cplusplus
38extern "C" {
39#endif
40
41const char* __tsan_default_options()
42{
43 return "halt_on_error=1:abort_on_error=1:report_signal_unsafe=0"
44 ":allocator_may_return_null=1";
45}
46
47const char* __tsan_default_suppressions()
48{
49 return
50 // External uninstrumented libraries
51 "called_from_lib:libzmq.so\n"
52 "called_from_lib:libpgm-5.2.so\n"
53 "called_from_lib:libusb*\n"
54 "called_from_lib:libuhd*\n"
55 // Races detected inside uninstrumented libraries. This may hide legit races if any of the libraries appear in the
56 // backtrace
57 "race:libusb*\n"
58 "race:libuhd*\n";
59}
