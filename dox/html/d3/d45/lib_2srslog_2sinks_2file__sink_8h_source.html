30/// This sink implementation writes to files. Includes the optional feature of
31/// file rotation: a new file is created when file size exceeds an established
32/// threshold.
33class file_sink : public sink
34{
35public:
36 file_sink(std::string name, size_t max_size_, bool force_flush_, std::unique_ptr<log_formatter> f) :
37 sink(std::move(f)),
38 max_size((max_size_ == 0) ? 0 : std::max<size_t>(max_size_, 4 * 1024)),
39 force_flush(force_flush_),
40 base_filename(std::move(name))
41 {
42 }
43
44 file_sink(const file_sink& other) = delete;
45 file_sink& operator=(const file_sink& other) = delete;
46
47 detail::error_string write(detail::memory_buffer buffer) override
48 {
49 // Create a new file the first time we hit this method.
50 if (is_first_write()) {
51 assert(!handler && "No handler should be created yet");
52 if (auto err_str = create_file()) {
53 return err_str;
54 }
55 }
56
57 // Do not bother doing any work when the file was closed on a previous
58 // error.
59 if (!handler) {
60 return {};
61 }
62
63 if (auto err_str = handle_rotation(buffer.size())) {
64 return err_str;
65 }
66
67 if (force_flush) {
68 flush();
69 }
70
71 return handler.write(buffer);
72 }
73
74 detail::error_string write_error(detail::memory_buffer input_buffer) override { return write(input_buffer); }
75
76 detail::error_string flush() override { return handler.flush(); }
77
78protected:
79 /// Returns the current file index.
80 uint32_t get_file_index() const { return file_index; }
81
82private:
83 /// Returns true when the sink has never written data to a file, otherwise
84 /// returns false.
85 bool is_first_write() const { return file_index == 0; }
86
87 /// Creates a new file and increments the file index counter.
88 detail::error_string create_file()
89 {
90 return handler.create(file_utils::build_filename_with_index(base_filename, file_index++));
91 }
92
93 /// Handles the file rotation feature when it is activated.
94 /// NOTE: The file handler must be valid.
95 detail::error_string handle_rotation(size_t size)
96 {
97 assert(handler && "Expected a valid file handle");
98 current_size += size;
99 if (max_size && current_size >= max_size) {
100 current_size = size;
101 return create_file();
102 }
103 return {};
104 }
105
106private:
107 const size_t max_size;
108 const bool force_flush;
109 const std::string base_filename;
110 file_utils::file handler;
111 size_t current_size = 0;
112 uint32_t file_index = 0;
113};
114
115} // namespace srslog
