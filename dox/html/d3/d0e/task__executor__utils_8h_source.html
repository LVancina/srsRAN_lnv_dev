32template <typename T>
33class is_task_executor_ptr
34{
35 template <typename U>
36 static auto test(U&& ptr) -> decltype(ptr->execute(std::declval<unique_task>()), std::true_type{});
37 template <typename U>
38 static auto test(const U* ptr)
39 -> decltype(std::declval<U*>()->execute(std::declval<unique_task>()), std::true_type{});
40
41 static std::false_type test(...);
42
43public:
44 static constexpr bool value = decltype(test(std::declval<T>()))::value;
45};
46
47// Helper function to convert a task executor object, that can be a raw pointer or unique_ptr to a reference.
48template <typename T, std::enable_if_t<is_task_executor_ptr<T>::value, int> = 0>
49auto& get_task_executor_ref(T&& exec)
50{
51 return *exec;
52}
53template <typename T, std::enable_if_t<not is_task_executor_ptr<T>::value, int> = 0>
54auto get_task_executor_ref(T&& exec)
55{
56 return std::forward<T>(exec);
57}

59template <typename T>
60class task_executor_has_can_run_task_inline
61{
62 template <typename U>
63 static auto test(const U& u) -> decltype(get_task_executor_ref(u).can_run_task_inline(), std::true_type{});
64
65 static std::false_type test(...);
66
67public:
68 static constexpr bool value = decltype(test(std::declval<T>()))::value;
69};
70
71// Helper function to determine whether an executor allows a task to be run inline, given the caller thread.
72template <typename Executor, std::enable_if_t<task_executor_has_can_run_task_inline<Executor>::value, int> = 0>
73bool executor_lets_run_task_inline(const Executor& exec)
74{
75 return get_task_executor_ref(exec).can_run_task_inline();
76}
77template <typename Executor, std::enable_if_t<not task_executor_has_can_run_task_inline<Executor>::value, int> = 0>
78bool executor_lets_run_task_inline(const Executor& exec)
79{
80 return false;
81}
