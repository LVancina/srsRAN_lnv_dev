29/// \brief This class assists with the search of PDSCH parameters given an UE config that ensures a valid UE PDSCH
30/// resource allocation in the Cell Resource Grid. The search iterates through different HARQ, SearchSpace and PDSCH
31/// time-domain resource candidates, returning only the valid ones. This class operates as a range with begin()
32/// and end() iterators, and only searches for valid candidates in a lazy fashion. That means that we only compute
33/// all valid candidates if we iterate from begin() to end().
34///
35/// To be a valid candidate, the following conditions must be met:
36/// - The HARQ process state must match the \c is_retx state passed in the constructor. If \c is_retx is false, there
37/// should be at most one empty HARQ candidate returned during the whole search. If \c is_retx is true, all the HARQs
38/// of the candidates returned must have a pending retransmission.
39/// - We avoid SearchSpace#0 for UE PDSCH allocation.
40/// - The slot must have sufficient DL symbols, given the PDSCH time-domain resource.
41/// - For \c is_retx equal to true, we ensure the candidates will lead to allocations with the same number of DL
42/// symbols as the previous HARQ allocation.
43class ue_pdsch_param_candidate_searcher
44{
45 using search_space_candidate_list = static_vector<const search_space_info*, MAX_NOF_SEARCH_SPACE_PER_BWP>;
46 using dl_harq_iter = static_vector<const dl_harq_process*, MAX_NOF_HARQS>::const_iterator;
47 using ss_iter = static_vector<const search_space_info*, MAX_NOF_SEARCH_SPACE_PER_BWP>::const_iterator;
48
49public:
50 struct iterator;
51
52 /// Parameters for a given PDSCH candidate.
53 struct candidate {
54 /// DL HARQ process (may be newTx or reTx).
55 const dl_harq_process& harq() const { return **harq_it; }
56
57 /// Chosen Search Space.
58 const search_space_info& ss() const { return **ss_it; }
59
60 /// Chosen PDSCH Time Domain Resource.
61 const pdsch_time_domain_resource_allocation& pdsch_td_res() const
62 {
63 return (*ss_it)->pdsch_time_domain_list[time_res];
64 }
65
66 /// Chosen PDSCH Time Domain Resource Index.
67 unsigned pdsch_td_res_index() const { return time_res; }
68
69 /// Chosen DCI DL RNTI Config Type.
70 dci_dl_rnti_config_type dci_dl_rnti_cfg_type() const
71 {
72 if (not harq().empty()) {
73 return harq().last_alloc_params().dci_cfg_type;
74 }
75 return ss().get_dl_dci_format() == dci_dl_format::f1_0 ? dci_dl_rnti_config_type::c_rnti_f1_0
76 : dci_dl_rnti_config_type::c_rnti_f1_1;
77 }
78
79 bool operator==(const candidate& other) const
80 {
81 return harq_it == other.harq_it and ss_it == other.ss_it and time_res == other.time_res;
82 }
83 bool operator!=(const candidate& other) const { return not(*this == other); }
84 /// A candidate being less than another means that the first has higher priority than the second.
85 bool operator<(const candidate& other) const
86 {
87 return harq_it < other.harq_it or (harq_it == other.harq_it and ss_it < other.ss_it) or
88 (harq_it == other.harq_it and ss_it == other.ss_it and time_res < other.time_res);
89 }
90 bool operator>=(const candidate& other) const { return not(*this < other); }
91 bool operator<=(const candidate& other) const { return *this < other or *this == other; }
92 bool operator>(const candidate& other) const { return not(*this < other) and *this != other; }
93
94 private:
95 friend struct iterator;
96 friend class ue_pdsch_param_candidate_searcher;
97
98 candidate(dl_harq_iter h, ss_iter ss, unsigned time_res_) : harq_it(h), ss_it(ss), time_res(time_res_) {}
99
100 dl_harq_iter harq_it;
101 ss_iter ss_it;
102 unsigned time_res;
103 };
104
105 /// Iterator for list of candidates.
106 struct iterator {
107 using value_type = candidate;
108 using reference = value_type&;
109 using pointer = value_type*;
110
111 iterator(ue_pdsch_param_candidate_searcher& parent_, dl_harq_iter h_it, ss_iter ss_it, unsigned time_res_) :
112 parent(&parent_), current(h_it, ss_it, time_res_)
113 {
114 parent->iterate_until_valid_candidate_found(current);
115 }
116
117 candidate& operator*() { return current; }
118 const candidate& operator*() const { return current; }
119 candidate* operator->() { return &current; }
120 const candidate* operator->() const { return &current; }
121
122 /// Incrementing the iterator means moving up one position in the list of candidates.
123 iterator& operator++()
124 {
125 ++current.time_res;
126 // When we increment the time_res, it may be pointing to the end of the PDSCH time domain resource list.
127 // So, we may need to move to the next SearchSpace or HARQ candidate.
128 parent->iterate_until_valid_candidate_found(current);
129 return *this;
130 }
131
132 bool operator==(const iterator& other) const { return current == other.current; }
133 bool operator!=(const iterator& other) const { return not(*this == other); }
134
135 private:
136 ue_pdsch_param_candidate_searcher* parent;
137 candidate current;
138 };

140 /// Create a searcher for UE PDSCH parameters.
141 ue_pdsch_param_candidate_searcher(const ue& ue_ref_,
142 ue_cell_index_t cell_index,
143 bool is_retx_,
144 slot_point pdcch_slot_) :
145 ue_ref(ue_ref_), ue_cc(ue_ref.get_cell(cell_index)), is_retx(is_retx_), pdcch_slot(pdcch_slot_)
146 {
147 if (is_retx) {
148 // Create list of DL HARQ processes with pending retx, sorted from oldest to newest.
149 for (unsigned i = 0; i != ue_cc.harqs.nof_dl_harqs(); ++i) {
150 const dl_harq_process& h = ue_cc.harqs.dl_harq(i);
151 if (h.has_pending_retx()) {
152 dl_harq_candidates.push_back(&h);
153 }
154 }
155 std::sort(
156 dl_harq_candidates.begin(),
157 dl_harq_candidates.end(),
158 [](const dl_harq_process* lhs, const dl_harq_process* rhs) { return lhs->slot_ack() < rhs->slot_ack(); });
159 } else if (ue_ref.get_cell(cell_index).is_active()) {
160 // If there are no pending new Tx bytes, return.
161 if (not ue_ref.has_pending_dl_newtx_bytes()) {
162 return;
163 }
164
165 // Find empty HARQ. Create a list with a single position.
166 const dl_harq_process* h = ue_cc.harqs.find_empty_dl_harq();
167 if (h != nullptr) {
168 dl_harq_candidates.push_back(h);
169 }
170 }
171 }
172
173 /// Get begin to the list of candidates.
174 iterator begin() { return iterator{*this, dl_harq_candidates.begin(), ss_candidate_list.begin(), 0}; }
175 iterator end() { return iterator{*this, dl_harq_candidates.end(), nullptr, 0}; }
176
177 /// List of DL HARQ candidates.
178 const static_vector<const dl_harq_process*, MAX_NOF_HARQS>& dl_harqs() const { return dl_harq_candidates; }
179
180private:
181 // Generate Search Space candidates for a given HARQ.
182 void generate_ss_candidates(dl_harq_iter current_harq_it)
183 {
184 if (harq_of_ss_list == *current_harq_it) {
185 // The HARQ candidate didn't change. We can early exit.
186 return;
187 }
188 srsran_assert(is_retx or harq_of_ss_list == nullptr, "Regenerating SS candidates should only be needed for reTxs");
189
190 // New HARQ. Search Space candidates are recomputed.
191 const auto* prev_h = harq_of_ss_list;
192 harq_of_ss_list = *current_harq_it;
193
194 // Check which RNTI Type is preferred for this UE and HARQ.
195 optional<dci_dl_rnti_config_type> preferred_rnti_type;
196 if (is_retx) {
197 preferred_rnti_type = harq_of_ss_list->last_alloc_params().dci_cfg_type;
198 } else if (ue_ref.is_conres_ce_pending()) {
199 preferred_rnti_type = dci_dl_rnti_config_type::tc_rnti_f1_0;
200 }
201
202 if (prev_h != nullptr and preferred_rnti_type == current_rnti_type) {
203 // It is the same RNTI Type as the previous HARQ candidate. Search Space Candidate doesn't need to be regenerated,
204 // and we can use the previous list.
205 return;
206 }
207
208 // Update alloc_params list.
209 ss_candidate_list = ue_cc.get_active_dl_search_spaces(pdcch_slot, preferred_rnti_type);
210 current_rnti_type = preferred_rnti_type;
211 }
212
213 // Check if a candidate has valid parameters for an allocation.
214 bool is_candidate_valid(const candidate& current) const
215 {
216 if (ue_cc.cfg().cell_cfg_common.get_nof_dl_symbol_per_slot(pdcch_slot + current.pdsch_td_res().k0) <
217 current.pdsch_td_res().symbols.stop()) {
218 // Check whether PDSCH time domain resource fits in DL symbols of the slot.
219 return false;
220 }
221
222 // Check whether PDSCH time domain resource does not overlap with CORESET.
223 if (current.pdsch_td_res().symbols.start() <
224 current.ss().cfg->get_first_symbol_index() + current.ss().coreset->duration) {
225 return false;
226 }
227
228 // If it is a retx, we need to ensure we use a time_domain_resource with the same number of symbols as used for
229 // the first transmission.
230 if (is_retx and current.pdsch_td_res().symbols.length() != current.harq().last_alloc_params().nof_symbols) {
231 return false;
232 }
233
234 return true;
235 }
236
237 // Iterate over the list of candidates until a valid one is found.
238 void iterate_until_valid_candidate_found(candidate& current)
239 {
240 for (; current.harq_it != dl_harq_candidates.end(); ++current.harq_it) {
241 // If the HARQ candidate changed, generate new list of Search Spaces.
242 generate_ss_candidates(current.harq_it);
243
244 for (; current.ss_it != ss_candidate_list.end(); ++current.ss_it) {
245 if (not ue_cc.is_in_fallback_mode() and current_rnti_type != dci_dl_rnti_config_type::tc_rnti_f1_0 and
246 (*current.ss_it)
247 ->get_pdcch_candidates(ue_cc.get_aggregation_level(
248 ue_cc.channel_state_manager().get_wideband_cqi(), **current.ss_it, true),
249 pdcch_slot)
250 .empty()) {
251 // For the case when dedicated UE config is used, skip SearchSpaces without PDCCH candidates being monitored
252 // in this slot.
253 continue;
254 }
255
256 for (; current.time_res < (*current.ss_it)->pdsch_time_domain_list.size(); ++current.time_res) {
257 if (is_candidate_valid(current)) {
258 // Valid candidate found.
259 return;
260 }
261 }
262 current.time_res = 0;
263 }
264 current.ss_it = ss_candidate_list.begin();
265 }
266
267 // Iteration finished.
268 ss_candidate_list.clear();
269 harq_of_ss_list = nullptr;
270 current.ss_it = nullptr;
271 current.time_res = 0;
272 }
273
274 // UE being allocated.
275 const ue& ue_ref;
276 // UE cell being allocated.
277 const ue_cell& ue_cc;
278 // Whether the current search is for a newTx or a reTx.
279 const bool is_retx;
280
281 // List of HARQ candidates considered for allocation.
282 static_vector<const dl_harq_process*, MAX_NOF_HARQS> dl_harq_candidates;
283
284 // List of Search Space candidates for the current HARQ candidate being iterated over.
285 search_space_candidate_list ss_candidate_list;
286 // HARQ candidate for which the Search Space candidate list was generated.
287 const dl_harq_process* harq_of_ss_list = nullptr;
288 // RNTI type used to generate ss_candidate_list.
289 optional<dci_dl_rnti_config_type> current_rnti_type;
290
291 // PDCCH slot point used to verify if the PDSCH fits a DL slot.
292 slot_point pdcch_slot;
293};
