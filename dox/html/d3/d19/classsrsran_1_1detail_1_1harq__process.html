59{
60 last_slot_ind = slot_tx;
61 for (transport_block& tb : tb_array) {
62 if (tb.state == transport_block::state_t::empty) {
63 continue;
64 }
65 if (slot_ack_timeout > last_slot_ind) {
66 // Wait more slots for ACK/NACK to arrive.
67 return;
68 }
69
70 if (tb.state == transport_block::state_t::pending_retx) {
71 // [Implementation-defined] Maximum time we give to the scheduler policy to retransmit a HARQ process.
72 const unsigned max_nof_slots_for_retx = last_slot_ack.nof_slots_per_system_frame() / 4;
73 if (slot_tx >= (last_slot_ack + max_nof_slots_for_retx)) {
74 // If a HARQ retx is never scheduled, the HARQ process will never be cleared. This is a safety mechanism to
75 // account for a potential bug or limitation in the scheduler policy that is leaving HARQ processes with
76 // pending reTxs for too long.
77 tb.state = transport_block::state_t::empty;
78 logger.warning(
79 id,
80 "Discarding HARQ. Cause: Too much time has passed since the last HARQ transmission. The scheduler "
81 "policy is likely not prioritizing retransmissions of old HARQ processes.");
82 }
83 continue;
84 }
85
86 // HARQ-ACK is waiting for ACK.
87
88 if (tb.ack_on_timeout) {
89 // Case: Not all HARQ-ACKs were received, but at least one positive ACK was received.
90 tb.state = transport_block::state_t::empty;
91 logger.debug(id,
92 "Setting HARQ to \"ACKed\" state. Cause: HARQ-ACK wait timeout ({} slots) was reached with still "
93 "missing PUCCH HARQ-ACKs. However, one positive ACK was received.",
94 slot_ack_timeout - last_slot_ack);
95 continue;
96 }
97
98 // At least one of the expected ACKs went missing and we haven't received any positive ACK.
99 tb.state = transport_block::state_t::empty;
100 if (max_ack_wait_in_slots != 1) {
101 // Only in non-NTN case, we log a warning.
102 logger.warning(id,
103 "Discarding HARQ. Cause: HARQ-ACK wait timeout ({} slots) was reached, but there are still "
104 "missing HARQ-ACKs and none of the received ones are positive.",
105 slot_ack_timeout - last_slot_ack);
106 }
107
108 // Report timeout with NACK.

◆ last_slot_ack

template<bool IsDownlink>
slot_point srsran::detail::harq_process< IsDownlink >::last_slot_ack
protected

For DL, slot_ack is the slot at which gNB is expected to receive the ACK via UCI. For UL, slot_ack is the slot when the PUSCH CRC will be received by the gNB. It coincides with slot_tx of UL.

◆ last_slot_tx

For DL, slot_tx corresponds to the slot when the TB in the HARQ process is going to be transmitted by the gNB. For UL, slot_tx corresponds to the slot when the TB in the HARQ process is going to be transmitted by the UE.
