32/// Describes a resource grid coordinate as symbol index and carrier.
33struct resource_grid_coordinate {
34 /// Symbol index (0...13).
35 uint8_t symbol = 0;
36 /// Subcarrier index (0...3299).
37 uint16_t subcarrier = 0;
38
39 /// Default destructor
40 resource_grid_coordinate() = default;
41
42 /// \brief Constructor from \c unsigned values.
43 /// \param[in] symbol_ Provides the \c symbol value.
44 /// \param[in] subcarrier_ Provides the \c subcarrier value.
45 resource_grid_coordinate(unsigned symbol_, unsigned subcarrier_) :
46 symbol(static_cast<uint8_t>(symbol_)), subcarrier(static_cast<uint16_t>(subcarrier_))
47 {
48 }
49
50 /// \brief Overloads equal comparison operator.
51 ///
52 /// Two resource grid coordinates are equal when symbol and subcarrier indexes are equal.
53 /// \param[in] other The comparing coordinate.
54 bool operator==(const resource_grid_coordinate& other) const
55 {
56 return (other.symbol == symbol) && (subcarrier == other.subcarrier);
57 }
58};

60/// \brief Resource grid writer interface.
61///
62/// Contains the necessary functions to write resource elements in a resource grid.
63///
64/// \remark All the methods contained in this interface must not result in writing outside of the resource grid region.
65class resource_grid_writer : public resource_grid_base
66{
67public:
68 /// Default destructor
69 virtual ~resource_grid_writer() = default;
70
71 /// \brief Puts a number of elements in the grid for a given port according to a list of coordinates.
72 ///
73 /// \param[in] port Port index.
74 /// \param[in] coordinates List of grid symbol&ndash;subcarrier coordinates.
75 /// \param[in] symbols List of symbols to written at the specified coordinates.
76 /// \note The number of elements in \c coordinates and \c symbols shall be the same.
77 virtual void put(unsigned port, span<const resource_grid_coordinate> coordinates, span<const cf_t> symbols) = 0;
78
79 /// \brief Puts a number of resource elements in the resource grid at the given port and symbol using a mask to
80 /// indicate which subcarriers are allocated and which are not.
81 ///
82 /// \param[in] port Port index.
83 /// \param[in] l Symbol index.
84 /// \param[in] k_init Initial subcarrier index.
85 /// \param[in] mask Boolean mask denoting the subcarriers to be written (if \c true), starting from \c k_init.
86 /// \param[in] symbols Symbols to be written into the resource grid.
87 /// \return A view to the unused entries of \c symbols.
88 /// \note The number of elements of \c mask shall be equal to or greater than the resource grid number of subcarriers.
89 /// \note The number of elements of \c symbols shall be equal to or greater than the number of true elements in
90 /// \c mask.
91 virtual span<const cf_t>
92 put(unsigned port, unsigned l, unsigned k_init, span<const bool> mask, span<const cf_t> symbols) = 0;
93
94 /// \brief Puts a number of resource elements in the resource grid at the given port and symbol using a bounded bitset
95 /// to indicate which subcarriers are allocated and which are not.
96 ///
97 /// \param[in] port Port index.
98 /// \param[in] l Symbol index.
99 /// \param[in] k_init Initial subcarrier index.
100 /// \param[in] mask Bitset denoting the subcarriers to be written (if \c true), starting from \c k_init.
101 /// \param[in] symbols Symbols to be written into the resource grid.
102 /// \return A view to the unused entries of \c symbols.
103 /// \note The number of elements of \c mask shall be equal to or greater than the resource grid number of subcarriers.
104 /// \note The number of elements of \c symbols shall be equal to or greater than the number of true elements in
105 /// \c mask.
106 virtual span<const cf_t> put(unsigned port,
107 unsigned l,
108 unsigned k_init,
109 const bounded_bitset<NRE * MAX_RB>& mask,
110 span<const cf_t> symbols) = 0;
111
112 /// \brief Puts a consecutive number of resource elements for the given \c port and symbol \c l, starting at \c
113 /// k_init.
114 ///
115 /// \param[in] port Port index.
116 /// \param[in] l Symbol index.
117 /// \param[in] k_init Initial subcarrier index.
118 /// \param[in] symbols Symbols to be written into the resource grid.
119 /// \note The sum of \c k_init and the number of elements in \c symbols shall not exceed the resource grid number of
120 /// subcarriers.
121 virtual void put(unsigned port, unsigned l, unsigned k_init, span<const cf_t> symbols) = 0;
122
123 /// \brief Puts a number of resource elements for the given \c port and symbol \c l, starting at \c
124 /// k_init and at a distance of \c stride.
125 ///
126 /// \param[in] port Port index.
127 /// \param[in] l Symbol index.
128 /// \param[in] k_init Initial subcarrier index.
129 /// \param[in] stride Distance between adjacent symbols. A stride of 1 means that the allocated REs are contiguous.
130 /// \param[in] symbols Symbols to be written into the resource grid.
131 /// \note The RE positions given \c k_init, the number of elements in \c symbols and the \c stride shall be within the
132 /// resource grid number of subcarriers.
133 virtual void put(unsigned port, unsigned l, unsigned k_init, unsigned stride, span<const cf_t> symbols) = 0;
134};
