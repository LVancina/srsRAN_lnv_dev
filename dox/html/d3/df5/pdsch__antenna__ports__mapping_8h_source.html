30/// Antenna ports mapping information for PDSCH when using DCI format 1_1. See TS 38.212 Tables 7.3.1.2.2-1/2/3/4.
31struct pdsch_antenna_ports_mapping {
32 /// Number of DMRS CDM group(s) without data.
33 unsigned nof_dmrs_cdm_groups_without_data;
34 /// Contains list of DMRS ports used.
35 /// Example 1: For a cell with 4 DL antenna ports and if the value of DMRS ports list is {0, 3}, then the DMRS ports
36 /// (1000 + 0) = 1000 and (1000 + 3) = 1003 are used.
37 /// Example 2: For a cell with 2 DL antenna ports and if the value of DMRS ports list is {0, 1}, then the DMRS ports
38 /// (1000 + 0) = 1000 and (1000 + 1) = 1001 are used.
39 std::vector<unsigned> dmrs_ports;
40 /// Number of DMRS front-load symbols. Number of front-load symbols equates to maxLength in DMRS-DownlinkConfig whose
41 /// default value is Len1 i.e. 1 symbol.
42 unsigned nof_front_load_symbols = 1;
43};

45/// \brief Returns the PDSCH antenna ports mapping from TS 38.212, tables 7.3.1.2.2-1/2/3/4 based on input
46/// configuration.
47/// \param[in] nof_layers Number of layers.
48/// \param[in] nof_dl_antenna_ports Number of DL antenna ports supported by gNB.
49/// \param[in] dmrs_cfg_type PDSCH DMRS configuration type.
50/// \param[in] dmrs_max_len PDSCH DMRS maximum number of OFDM symbols.
51/// \param[in] are_both_cws_enabled Flag indicating whether Codeword 0 and Codeword 1 are enabled or not.
52/// \return The PDSCH antenna ports mapping from TS 38.212, tables 7.3.1.2.2-1/2/3/4 based on DMRS configuration, nof.
53/// layers etc. If the given input configuration does not match any of the rows in the table, the function
54/// triggers an asserts. But if input configuration matches more than one row in the table, the first matching
55/// row information is returned.
56pdsch_antenna_ports_mapping get_pdsch_antenna_port_mapping(unsigned nof_layers,
57 unsigned nof_dl_antenna_ports,
58 dmrs_config_type dmrs_cfg_type,

60 bool are_both_cws_enabled);
