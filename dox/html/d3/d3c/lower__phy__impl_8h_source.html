51public:

54 /// Downlink processor.

56 /// Uplink processor.
57 std::unique_ptr<lower_phy_uplink_processor> uplink_proc;
58 /// Baseband controller.
59 std::unique_ptr<lower_phy_controller> controller;
60 /// Symbol handler to notify the reception of symbols.
61 lower_phy_rx_symbol_notifier* rx_symbol_notifier;
62 /// The timing handler to notify the timing boundaries.
63 lower_phy_timing_notifier* timing_notifier;
64 /// Error handler to notify runtime errors.
65 lower_phy_error_notifier* error_notifier;
66 };
67
68 /// Constructs a generic lower physical layer.

77 // See interface for documentation.

80 /// Downlink processor.
81 std::unique_ptr<lower_phy_downlink_processor> downlink_proc;

84 /// Processor handler adaptor.

86 /// Processor notification adaptor.
87 processor_notifier_adaptor notification_adaptor;
