47 bool is_expired() const
48 {
49 // No expiration time configured.
50 if (wait_expire_time == std::chrono::time_point<std::chrono::steady_clock>{}) {
51 return false;
52 }
53
54 // Get current time.
55 std::chrono::time_point<std::chrono::steady_clock> now = std::chrono::steady_clock::now();
56
57 // Check if the timer has been expired.
58 if (now < wait_expire_time) {
59 return false;
60 }
61
62 return true;
63 }
