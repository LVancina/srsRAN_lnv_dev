130inline clock_sources::source to_clock_source(const std::string& str)
131{
132 if (str == "internal") {
133 return clock_sources::source::INTERNAL;
134 }
135 if (str == "external") {
136 return clock_sources::source::EXTERNAL;
137 }
138 if (str == "gpsdo") {
139 return clock_sources::source::GPSDO;
140 }
141 if (str == "default") {
142 return clock_sources::source::DEFAULT;
143 }
144 report_error("Invalid clock/sync source '{}'.", str);
145}
146
148inline over_the_wire_format to_otw_format(const std::string& str)
149{
150 if (str == "sc8") {
151 return over_the_wire_format::SC8;
152 }
153 if (str == "sc12") {
154 return over_the_wire_format::SC12;
155 }
156 if (str == "sc16") {
157 return over_the_wire_format::SC16;
158 }
159 if (str == "default") {
160 return over_the_wire_format::DEFAULT;
161 }
162 report_error("Invalid OTW format '{}'.", str);
163}
