75 void one()
76 {
77 unsigned nof_full = nof_full_words();
78 word_t val = (1U << bits_per_word) - 1;
79 std::fill_n(buffer.begin(), nof_full, val);
80
81 unsigned tail_bits = size() - nof_full * bits_per_word;
82 if (tail_bits == 0) {
83 return;
84 }
85 val = (1U << tail_bits) - 1;
86 buffer[nof_full] = val << (bits_per_word - tail_bits);
87 }

97 template <typename Integer>
98 void insert(Integer value_, unsigned startpos, unsigned count)
99 {
100 srsran_assert(count <= bits_per_word,
101 "The number of bits to insert (i.e., {}) exceeds the number of bits per word (i.e., {}).",
102 count,
103 static_cast<unsigned>(bits_per_word));
104 srsran_assert(value_ < (1U << count),
105 "The value (i.e., {}) number of bits exceeds the number of bits to insert (i.e., {}).",
106 value_,
107 count);
108 srsran_assert(startpos + count <= size(),
109 "The bit range starting at {} for {} bits exceed the buffer size (i.e., {}).",
110 startpos,
111 count,
112 size());
113 unsigned start_word = startpos / bits_per_word;
114 unsigned start_mod = startpos % bits_per_word;
115 word_t value = static_cast<word_t>(value_);
116
117 srsran_assert(
118 value_ <= mask_lsb_ones<word_t>(count),
119 "The value (i.e., {}) number of bits exceeds the number of bits that can be fitted inside a word (i.e., {}).",
120 value,
121 mask_lsb_ones<word_t>(count));
122
123 // If the start bit is aligned with the beginning of the word.
124 if (start_mod == 0) {
125 buffer[start_word] = (buffer[start_word] & mask_msb_zeros<word_t>(count)) | (value << (bits_per_word - count));
126 return;
127 }
128
129 // If the insertion only affects to one word.
130 if (start_mod + count <= bits_per_word) {
131 // Extract word.
132 word_t word = buffer[start_word];
133 // Mask bits that are used.
134 word &= static_cast<word_t>(mask_msb_ones<word_t>(start_mod) |
135 mask_lsb_ones<word_t>(bits_per_word - start_mod - count));
136 // Insert bits.
137 word |= value << (bits_per_word - start_mod - count);
138 // Insert word.
139 buffer[start_word] = word;
140 return;
141 }
142
143 buffer[start_word] =
144 (buffer[start_word] & mask_msb_ones<word_t>(start_mod)) | (value >> (start_mod + count - bits_per_word));
145 buffer[start_word + 1] = (buffer[start_word + 1] & mask_msb_zeros<word_t>(start_mod + count - bits_per_word)) |
146 (value << (2 * bits_per_word - start_mod - count));
147 }

159 template <typename Integer = uint8_t>
160 Integer extract(unsigned startpos, unsigned count) const
161 {
162 srsran_assert(count <= bits_per_word,
163 "The number of bits to insert (i.e., {}) exceeds the number of bits per word (i.e., {}).",
164 count,
165 static_cast<unsigned>(bits_per_word));
166 srsran_assert(startpos + count <= size(),
167 "The bit range starting at {} for {} bits exceed the buffer size (i.e., {}).",
168 startpos,
169 count,
170 size());
171 unsigned start_word = startpos / bits_per_word;
172 unsigned start_mod = startpos % bits_per_word;
173
174 // If the operation does not cross boundaries between words.
175 if (start_mod == 0) {
176 return buffer[start_word] >> (bits_per_word - count);
177 }
178
179 // If the insertion only affects to one word.
180 if (start_mod + count <= bits_per_word) {
181 return (buffer[start_word] >> (bits_per_word - start_mod - count)) & mask_lsb_ones<Integer>(count);
182 }
183
184 // Concatenates two bytes in a 32-bit register and then extracts a word with the requested number of bits.
185 using extended_word_t = uint32_t;
186 extended_word_t word = static_cast<extended_word_t>(buffer[start_word]) << bits_per_word;
187 word |= static_cast<extended_word_t>(buffer[start_word + 1]);
188 word = word >> (2 * bits_per_word - start_mod - count);
189 word &= mask_lsb_ones<extended_word_t>(count);
190
191 return static_cast<Integer>(word);
192 }

238 bit_buffer last(unsigned count)
239 {
240 srsran_assert((size() - count) % bits_per_word == 0, "Only bit word boundaries are supported.");

263 template <typename OutputIt>
264 OutputIt to_bin_string(OutputIt&& mem_buffer) const
265 {
266 unsigned byte_remainder = current_size % bits_per_word;
267 if (byte_remainder == 0) {
268 fmt::format_to(mem_buffer, "{:08B}", buffer.first(nof_words()));
269 return mem_buffer;
270 }
271
272 if (nof_words() > 1) {
273 fmt::format_to(mem_buffer, "{:08B}", buffer.first(nof_words() - 1));
274 }
275
276 fmt::format_to(mem_buffer, " {:0{}B}", buffer[nof_words() - 1] >> (bits_per_word - byte_remainder), byte_remainder);
277 return mem_buffer;
278 }

300 bool operator==(const bit_buffer& other) const
301 {
302 if (size() != other.size()) {
303 return false;
304 }
305
306 // Compare full words.
307 unsigned nof_complete_words = size() / bits_per_word;
308 if (!std::equal(buffer.begin(), buffer.begin() + nof_complete_words, other.buffer.begin())) {
309 return false;
310 }
311
312 // Check the remaining number of bits to compare.
313 unsigned remainder = size() - nof_complete_words * bits_per_word;
314 if (remainder == 0) {
315 return true;
316 }
317
318 // Compare the remaining number of bits.
319 return extract(bits_per_word * nof_complete_words, remainder) ==
320 other.extract(bits_per_word * nof_complete_words, remainder);
321 }
322
324 bool operator!=(const bit_buffer& other) const { return !(*this == other); }

396 enum { hexadecimal, binary } mode = binary;
397
398 template <typename ParseContext>
399 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
400 {
401 auto it = ctx.begin();
402 while (it != ctx.end() and *it != '}') {
403 if (*it == 'x') {
404 mode = hexadecimal;
405 }
406 ++it;
407 }
408
409 return it;
410 }
411
412 template <typename FormatContext>
413 auto format(const srsran::bit_buffer& s, FormatContext& ctx) -> decltype(std::declval<FormatContext>().out())
414 {
415 if (mode == hexadecimal) {
416 return s.template to_hex_string(ctx.out());
417 }
418 return s.template to_bin_string(ctx.out());
419 }
