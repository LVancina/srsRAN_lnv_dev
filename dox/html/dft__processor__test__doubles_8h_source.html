31class dft_processor_spy : public dft_processor
32{
33public:
34 struct entry {
35 std::vector<cf_t> input;
36 std::vector<cf_t> output;
37 };
38
39 dft_processor_spy(unsigned seed, const configuration& config) :
40 size(config.size), dir(config.dir), input_buffer(config.size), output_buffer(config.size), rgen(seed), dist(-1, 1)
41 {
42 // Do nothing.
43 }
44 direction get_direction() const override { return dir; }
45 unsigned get_size() const override { return size; }
46 span<cf_t> get_input() override { return input_buffer; }
47 span<const cf_t> run() override
48 {
49 // Generate some random output if ouput buffer was not set.
50 if (generate_random_output) {
51 for (cf_t& value : output_buffer) {
52 value = {dist(rgen), dist(rgen)};
53 }
54 }
55
56 entries.emplace_back();
57 entry& e = entries.back();
58 e.input = input_buffer;
59 e.output = output_buffer;
60
61 return output_buffer;
62 }
63
64 void clear_entries() { entries.clear(); }
65
66 const std::vector<entry>& get_entries() const { return entries; }
67
68 void set_output_buffer(const std::vector<cf_t>& buffer)
69 {
70 report_fatal_error_if_not(
71 buffer.size() == size, "Buffer size {} is not equal to the DFT size {}.", buffer.size(), size);
72 generate_random_output = false;
73 output_buffer = buffer;
74 }
75
76private:
77 bool generate_random_output = true;
78 unsigned size;
79 direction dir;
80 std::vector<cf_t> input_buffer;
81 std::vector<cf_t> output_buffer;
82 std::mt19937 rgen;
83 std::uniform_real_distribution<float> dist;
84 std::vector<entry> entries;
85};
