167class dynamic_re_buffer : public re_buffer_reader, public re_buffer_writer
168{
169public:
173 dynamic_re_buffer(unsigned max_nof_slices_, unsigned max_nof_re_) :
174 max_nof_slices(max_nof_slices_), max_nof_re(max_nof_re_), data({max_nof_re, max_nof_slices})
175 {
176 }
177
183 void resize(unsigned nof_slices, unsigned nof_re)
184 {
185 srsran_assert(nof_slices <= max_nof_slices,
186 "The number of slices (i.e., {}) exceeds the maximum (i.e., {}).",
187 nof_slices,
188 max_nof_slices);
189 srsran_assert(nof_re <= max_nof_re,
190 "The number of resource elements (i.e., {}) exceeds the maximum (i.e., {}).",
191 nof_re,
192 max_nof_re);
193 data.resize({nof_re, nof_slices});
194 }
195
196 // See interface for documentation.
197 unsigned get_nof_slices() const override { return data.get_dimension_size(dims::slice); }
198
199 // See interface for documentation.
200 unsigned get_nof_re() const override { return data.get_dimension_size(dims::re); }
201
202 // See interface for documentation.
203 span<cf_t> get_slice(unsigned i_slice) override { return data.get_view<1>({i_slice}); }
204
205 // See interface for documentation.
206 span<const cf_t> get_slice(unsigned i_slice) const override { return data.get_view<1>({i_slice}); }
207
208private:
210 enum class dims : unsigned { re = 0, slice = 1, all = 2 };
211
213 unsigned max_nof_slices;
215 unsigned max_nof_re;
216
218 dynamic_tensor<static_cast<unsigned>(dims::all), cf_t, dims> data;
219};
