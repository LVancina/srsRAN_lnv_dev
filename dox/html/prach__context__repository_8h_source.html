40class prach_context
41{
43 struct prach_buffer_writer_stats {
44 prach_buffer_writer_stats(unsigned nof_ports, size_t nof_res) :
45 re_written(nof_ports, bounded_bitset<prach_constants::LONG_SEQUENCE_LENGTH>(nof_res))
46 {
47 }
48
50 static_vector<bounded_bitset<prach_constants::LONG_SEQUENCE_LENGTH>, MAX_NOF_SUPPORTED_EAXC> re_written;
51
53 bool have_all_res_been_written() const
54 {
55 return std::all_of(re_written.begin(), re_written.end(), [&](const auto& port) { return port.all(); });
56 }
57 };
58
59public:
60 struct prach_context_information {
62 prach_buffer_context context;
64 prach_buffer* buffer = nullptr;
65 };
66
68 prach_context() = default;
69
71 prach_context(const prach_buffer_context& context, prach_buffer& buffer) :
72 context_info({context, &buffer}), nof_symbols(get_preamble_duration(context.format))
73 {
74 srsran_assert(context.nof_fd_occasions == 1, "Only supporting one frequency domain occasion");
75 srsran_assert(context.nof_td_occasions == 1, "Only supporting one time domain occasion");
76
77 // Get preamble information.
78 preamble_info =
79 is_long_preamble(context.format)
80 ? get_prach_preamble_long_info(context.format)
81 : get_prach_preamble_short_info(context.format, to_ra_subcarrier_spacing(context.pusch_scs), true);
82
83 freq_mapping_info = prach_frequency_mapping_get(preamble_info.scs, context.pusch_scs);
84
85 if (!nof_symbols) {
86 nof_symbols = 1;
87 }
88 // Initialize statistic.
89 for (unsigned i = 0; i != nof_symbols; ++i) {
90 buffer_stats.push_back({buffer.get_max_nof_ports(), preamble_info.sequence_length});
91 }
92 }
93
95 bool empty() const { return context_info.buffer == nullptr; }
96
98 const prach_buffer_context& get_context() const { return context_info.context; }
99
101 unsigned get_prach_nof_re() const { return empty() ? 0U : preamble_info.sequence_length; }
102
104 unsigned get_prach_offset_to_first_re() const { return freq_mapping_info.k_bar; }
105
107 unsigned get_max_nof_ports() const { return empty() ? 0U : context_info.buffer->get_max_nof_ports(); }
108
111 span<const bounded_bitset<prach_constants::LONG_SEQUENCE_LENGTH>> get_symbol_re_written(unsigned symbol) const
112 {
113 srsran_assert(symbol < nof_symbols, "Invalid symbol index");
114 return buffer_stats[symbol].re_written;
115 }
116
118 void write_iq(unsigned port, unsigned symbol, unsigned re_start, span<const cf_t> iq_buffer)
119 {
120 symbol -= context_info.context.start_symbol;
121
122 srsran_assert(context_info.buffer, "No valid PRACH buffer in the context");
123 srsran_assert(symbol < nof_symbols, "Invalid symbol index");
124
125 // Skip writing if the given port does not fit in the PRACH buffer.
126 if (port >= context_info.buffer->get_max_nof_ports()) {
127 return;
128 }
129
130 // Update the buffer.
131 span<cf_t> prach_out_buffer = context_info.buffer->get_symbol(
132 port, context_info.context.nof_fd_occasions - 1, context_info.context.nof_td_occasions - 1, symbol);
133
134 srsran_assert(prach_out_buffer.last(prach_out_buffer.size() - re_start).size() >= iq_buffer.size(),
135 "Invalid IQ data buffer size to copy as it does not fit into the PRACH buffer");
136
137 std::copy(iq_buffer.begin(), iq_buffer.end(), prach_out_buffer.begin() + re_start);
138
139 // Update statistics.
140 buffer_stats[symbol].re_written[port].fill(re_start, re_start + iq_buffer.size());
141 }
142
145 expected<prach_context_information> try_getting_complete_prach_buffer() const
146 {
147 if (!context_info.buffer) {
148 return default_error_t({});
149 }
150
151 if (!std::all_of(buffer_stats.begin(), buffer_stats.end(), [&](const auto& symbol) {
152 return symbol.have_all_res_been_written();
153 })) {
154 return default_error_t({});
155 }
156
157 return {context_info};
158 }
159
160private:
162 prach_context_information context_info;
164 static_vector<prach_buffer_writer_stats, prach_constants::SHORT_SEQUENCE_MAX_NOF_SYMBOLS> buffer_stats;
166 prach_preamble_information preamble_info;
168 prach_frequency_mapping_information freq_mapping_info;
170 unsigned nof_symbols;
171};

174class prach_context_repository
175{
177 static constexpr unsigned SFN_MAX_VALUE = 1U << 8;
178
179 srslog::basic_logger* logger;
180 std::vector<prach_context> buffer;
181 //: TODO: make this lock free
182 mutable std::mutex mutex;
183
185 prach_context& entry(slot_point slot)
186 {
187 slot_point entry_slot(slot.numerology(), slot.sfn() % SFN_MAX_VALUE, slot.slot_index());
188 unsigned index = entry_slot.system_slot() % buffer.size();
189 return buffer[index];
190 }
191
193 const prach_context& entry(slot_point slot) const
194 {
195 slot_point entry_slot(slot.numerology(), slot.sfn() % SFN_MAX_VALUE, slot.slot_index());
196 unsigned index = entry_slot.system_slot() % buffer.size();
197 return buffer[index];
198 }
199
200public:
201 explicit prach_context_repository(unsigned size_, srslog::basic_logger* logger_ = nullptr) :
202 logger(logger_), buffer(size_)
203 {
204 }
205
207 void add(const prach_buffer_context& context, prach_buffer& buffer_, slot_point slot = slot_point())
208 {
209 std::lock_guard<std::mutex> lock(mutex);
210
211 slot_point current_slot = slot.valid() ? slot : context.slot;
212
213 if (logger) {
214 if (!entry(current_slot).empty()) {
215 const prach_buffer_context& previous_context = entry(current_slot).get_context();
216 logger->warning("Missed incoming User-Plane PRACH messages for slot '{}' and sector#{}",
217 previous_context.slot,
218 previous_context.sector);
219 }
220 }
221
222 entry(current_slot) = prach_context(context, buffer_);
223 }
224
226 void write_iq(slot_point slot, unsigned port, unsigned symbol, unsigned re_start, span<const cf_t> iq_buffer)
227 {
228 std::lock_guard<std::mutex> lock(mutex);
229 entry(slot).write_iq(port, symbol, re_start, iq_buffer);
230 }
231
233 prach_context get(slot_point slot) const
234 {
235 std::lock_guard<std::mutex> lock(mutex);
236 return entry(slot);
237 }
238
240 void clear(slot_point slot)
241 {
242 std::lock_guard<std::mutex> lock(mutex);
243 entry(slot) = {};
244 }
245};
