38class pucch_uci_message
39{
40public:
42 pucch_uci_message() = default;
43
45 struct configuration {
47 unsigned nof_sr;
49 unsigned nof_harq_ack;
51 unsigned nof_csi_part1;
53 unsigned nof_csi_part2;
54 };
55
60 explicit pucch_uci_message(const configuration& config) :
61 nof_sr_bits(config.nof_sr),
62 nof_harq_ack_bits(config.nof_harq_ack),
63 nof_csi_part1_bits(config.nof_csi_part1),
64 nof_csi_part2_bits(config.nof_csi_part2)
65 {
66 srsran_assert(config.nof_sr + config.nof_harq_ack + config.nof_csi_part1 + config.nof_csi_part2 <=
67 uci_constants::MAX_NOF_PAYLOAD_BITS,
68 "The total number of payload bits (i.e. {}) shall not exceed {}.",
69 config.nof_sr + config.nof_harq_ack + config.nof_csi_part1 + config.nof_csi_part2,
70 uci_constants::MAX_NOF_PAYLOAD_BITS);
71 }
72
74 void set_status(uci_status status_) { status = status_; }
75
77 uci_status get_status() const { return status; }
78
81 span<uint8_t> get_full_payload()
82 {
83 return span<uint8_t>(data).first(nof_sr_bits + nof_harq_ack_bits + nof_csi_part1_bits + nof_csi_part2_bits);
84 }
85
88 span<const uint8_t> get_full_payload() const
89 {
90 return span<const uint8_t>(data).first(nof_sr_bits + nof_harq_ack_bits + nof_csi_part1_bits + nof_csi_part2_bits);
91 }
92
94 unsigned get_expected_nof_bits_full_payload() const
95 {
96 return nof_sr_bits + nof_harq_ack_bits + nof_csi_part1_bits + nof_csi_part2_bits;
97 }
98
100 unsigned get_expected_nof_sr_bits() const { return nof_sr_bits; }
101
103 unsigned get_expected_nof_harq_ack_bits() const { return nof_harq_ack_bits; }
104
106 unsigned get_expected_nof_csi_part1_bits() const { return nof_csi_part1_bits; }
107
109 unsigned get_expected_nof_csi_part2_bits() const { return nof_csi_part2_bits; }
110
112 span<uint8_t> get_sr_bits() { return span<uint8_t>(data).subspan(nof_harq_ack_bits, nof_sr_bits); }
113
115 span<const uint8_t> get_sr_bits() const { return span<const uint8_t>(data).subspan(nof_harq_ack_bits, nof_sr_bits); }
116
118 span<uint8_t> get_harq_ack_bits() { return span<uint8_t>(data).subspan(0, nof_harq_ack_bits); }
119
121 span<const uint8_t> get_harq_ack_bits() const { return span<const uint8_t>(data).subspan(0, nof_harq_ack_bits); }
122
124 span<uint8_t> get_csi_part1_bits()
125 {
126 return span<uint8_t>(data).subspan(nof_sr_bits + nof_harq_ack_bits, nof_csi_part1_bits);
127 }
128
130 span<const uint8_t> get_csi_part1_bits() const
131 {
132 return span<const uint8_t>(data).subspan(nof_harq_ack_bits + nof_sr_bits, nof_csi_part1_bits);
133 }
134
136 span<uint8_t> get_csi_part2_bits()
137 {
138 return span<uint8_t>(data).subspan(nof_harq_ack_bits + nof_sr_bits + nof_csi_part1_bits, nof_csi_part2_bits);
139 }
140
142 span<const uint8_t> get_csi_part2_bits() const
143 {
144 return span<const uint8_t>(data).subspan(nof_harq_ack_bits + nof_sr_bits + nof_csi_part1_bits, nof_csi_part2_bits);
145 }
146
147private:
149 std::array<uint8_t, uci_constants::MAX_NOF_PAYLOAD_BITS> data;
151 unsigned nof_sr_bits = 0;
153 unsigned nof_harq_ack_bits = 0;
155 unsigned nof_csi_part1_bits = 0;
157 unsigned nof_csi_part2_bits = 0;
159 uci_status status = uci_status::invalid;
160};
