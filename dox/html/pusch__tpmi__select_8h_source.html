35class pusch_tpmi_select_info
36{
37public:
38 struct tpmi_info {
40 unsigned tpmi;
42 float sinr_dB;
43 };
44
46 unsigned get_max_nof_layers() const { return info.size(); }
47
49 bool is_valid() const { return !info.empty(); }
50
54 const tpmi_info& get_tpmi_select(unsigned nof_layers) const
55 {
56 interval<unsigned, true> nof_layers_range(1, get_max_nof_layers());
57 srsran_assert(nof_layers_range.contains(nof_layers),
58 "The number of layers (i.e., {}) is out-of-range {}",
59 nof_layers,
60 nof_layers_range);
61
62 return info[nof_layers - 1];
63 }
64
66 pusch_tpmi_select_info(const std::initializer_list<tpmi_info>& info_) : info(info_.begin(), info_.end()) {}
67
68private:
70 static_vector<tpmi_info, pusch_constants::MAX_NOF_LAYERS> info;
71};
