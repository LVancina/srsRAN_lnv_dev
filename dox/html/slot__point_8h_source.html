55public:
57 constexpr slot_point() : numerology_val(NOF_NUMEROLOGIES), count_val(0) {}

60 constexpr slot_point(uint32_t numerology, uint32_t count) : numerology_val(numerology), count_val(count)
61 {
62 srsran_assert(numerology < NOF_NUMEROLOGIES, "Invalid numerology idx={} passed", numerology);
63 srsran_assert(count < nof_slots_per_system_frame(), "Invalid slot count={} passed", count);

70 constexpr slot_point(uint32_t numerology, uint32_t sfn_val, uint32_t slot_radio_frame_idx_) :
71 numerology_val(numerology), count_val(slot_radio_frame_idx_ + sfn_val * nof_slots_per_frame())
72 {
73 srsran_assert(numerology < NOF_NUMEROLOGIES, "Invalid numerology idx={} passed", numerology);
74 srsran_assert(sfn_val < NOF_SFNS, "Invalid SFN={} provided", sfn_val);
75 srsran_assert(slot_radio_frame_idx_ < nof_slots_per_frame(),
76 "Slot index={} exceeds maximum number of slots={}",
77 slot_radio_frame_idx_,
78 nof_slots_per_frame());
79 }

82 constexpr slot_point(subcarrier_spacing scs, uint32_t sfn_val, uint32_t slot_radio_frame_idx_) :

88 constexpr slot_point(uint32_t numerology, uint32_t sfn_val, uint32_t subframe_index_, uint32_t slot_subframe_index_) :
89 numerology_val(numerology),
90 count_val(slot_subframe_index_ + subframe_index_ * nof_slots_per_subframe() + sfn_val * nof_slots_per_frame())
91 {
92 srsran_assert(numerology < NOF_NUMEROLOGIES, "Invalid numerology idx={} passed", numerology);
93 srsran_assert(sfn_val < NOF_SFNS, "Invalid SFN={} provided", sfn_val);
94 srsran_assert(subframe_index_ < NOF_SUBFRAMES_PER_FRAME, "Invalid Subframe index={} provided", subframe_index_);
95 srsran_assert(slot_subframe_index_ < nof_slots_per_subframe(),
96 "Slot index={} exceeds maximum number of slots in subframe={}",
97 slot_subframe_index_,
98 nof_slots_per_subframe());
99 }

151 constexpr bool operator==(const slot_point& other) const
152 {
153 return other.count_val == count_val and other.numerology_val == numerology_val;
154 }

157 constexpr bool operator!=(const slot_point& other) const { return not(*this == other); }

162 constexpr bool operator<(const slot_point& other) const
163 {
164 srsran_assert(numerology() == other.numerology(), "Comparing slots of different numerologies");
165 int v = static_cast<int>(other.count_val) - static_cast<int>(count_val);
166 if (v > 0) {
167 return (v < (int)nof_slots_per_system_frame() / 2);
168 }
169 return (v < -(int)nof_slots_per_system_frame() / 2);
170 }
171
173 constexpr bool operator<=(const slot_point& other) const { return (*this == other) or (*this < other); }
174 constexpr bool operator>=(const slot_point& other) const { return not(*this < other); }
175 constexpr bool operator>(const slot_point& other) const { return (*this != other) and *this >= other; }
176
179 constexpr slot_difference operator-(const slot_point& other) const
180 {
181 int a = static_cast<int>(count_val) - static_cast<int>(other.count_val);
182 if (a >= (int)nof_slots_per_system_frame() / 2) {
183 return a - nof_slots_per_system_frame();
184 }
185 if (a < -(int)nof_slots_per_system_frame() / 2) {
186 return a + nof_slots_per_system_frame();
187 }
188 return a;
189 }
190
192 slot_point& operator++()
193 {
194 count_val++;
195 if (count_val == nof_slots_per_system_frame()) {
196 count_val = 0;
197 }
198 return *this;
199 }
200 slot_point operator++(int)
201 {
202 slot_point ret{*this};
203 this-> operator++();
204 return ret;
205 }

208 template <typename Unsigned, std::enable_if_t<std::is_unsigned<Unsigned>::value, int> = 0>
209 slot_point& operator+=(Unsigned jump)
210 {
211 count_val = (count_val + jump) % nof_slots_per_system_frame();
212 return *this;
213 }
214 slot_point& operator+=(int jump)
215 {
216 const int nof_slots = static_cast<int>(nof_slots_per_system_frame());
217 int tmp = (static_cast<int>(count_val) + jump) % nof_slots;
218 count_val = tmp + (tmp < 0 ? nof_slots : 0);
219 return *this;
220 }
221
223 slot_point& operator-=(uint32_t jump) { return this->operator+=(-static_cast<int>(jump)); }
224 slot_point& operator-=(int jump) { return this->operator+=(-jump); }

283inline slot_point set_slot_numerology(slot_point sl, unsigned new_numerology)
284{
285 unsigned old_numerology = sl.numerology();
286 if (old_numerology > new_numerology) {
287 return slot_point{new_numerology, sl.to_uint() >> (old_numerology - new_numerology)};
288 }
289 if (old_numerology < new_numerology) {
290 return slot_point{new_numerology, sl.to_uint() << (new_numerology - old_numerology)};
291 }
292 return sl;
293}
