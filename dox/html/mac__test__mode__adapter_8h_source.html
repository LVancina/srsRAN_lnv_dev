61class test_ue_info_manager
62{
63public:
64 test_ue_info_manager(rnti_t rnti_start_, uint16_t nof_ues_) : rnti_start(rnti_start_), nof_ues(nof_ues_) {}
65
66 du_ue_index_t rnti_to_du_ue_idx(rnti_t rnti) const
67 {
68 if (rnti_to_ue_info_lookup.count(rnti) == 0) {
69 return INVALID_DU_UE_INDEX;
70 }
71 return rnti_to_ue_info_lookup.at(rnti).ue_idx;
72 }
73
74 bool is_test_ue(du_ue_index_t ue_idx) const { return ue_idx < nof_ues; }
75
76 bool is_test_ue(rnti_t rnti) const
77 {
78 return (rnti >= rnti_start) and (rnti < to_rnti(to_value(rnti_start) + nof_ues));
79 }
80
81 void add_ue(rnti_t rnti, du_ue_index_t ue_idx_, const sched_ue_config_request& sched_ue_cfg_req_)
82 {
83 rnti_to_ue_info_lookup[rnti] =
84 test_ue_info{.ue_idx = ue_idx_, .sched_ue_cfg_req = sched_ue_cfg_req_, .msg4_rx_flag = false};
85 }
86
87 void remove_ue(rnti_t rnti)
88 {
89 if (rnti_to_ue_info_lookup.count(rnti) > 0) {
90 rnti_to_ue_info_lookup.erase(rnti);
91 }
92 }
93
94 sched_ue_config_request get_sched_ue_cfg_request(rnti_t rnti) const
95 {
96 return rnti_to_ue_info_lookup.at(rnti).sched_ue_cfg_req;
97 }
98
99 bool is_msg4_rxed(rnti_t rnti) const
100 {
101 if (rnti_to_ue_info_lookup.count(rnti) > 0) {
102 return rnti_to_ue_info_lookup.at(rnti).msg4_rx_flag;
103 }
104 return false;
105 }
106
107 void msg4_rxed(rnti_t rnti, bool msg4_rx_flag_)
108 {
109 if (rnti_to_ue_info_lookup.count(rnti) > 0) {
110 rnti_to_ue_info_lookup.at(rnti).msg4_rx_flag = msg4_rx_flag_;
111 }
112 }
113
114private:
115 struct test_ue_info {
116 du_ue_index_t ue_idx;
117 sched_ue_config_request sched_ue_cfg_req;
118 bool msg4_rx_flag;
119 };
120
122 rnti_t rnti_start;
123 uint16_t nof_ues;
124
126 std::unordered_map<rnti_t, test_ue_info> rnti_to_ue_info_lookup;
127};
