39class pdsch_processor_wrapper : private pdsch_processor, private pdsch_processor_notifier
40{
41public:
43 explicit pdsch_processor_wrapper(unsigned index_,
44 pdsch_processor_free_list& free_list_,
45 std::unique_ptr<pdsch_processor> processor_) :
46 index(index_), free_list(free_list_), notifier(nullptr), processor(std::move(processor_))
47 {
48 srsran_assert(processor, "Invalid PDSCH processor.");
49 }
50
52 pdsch_processor_wrapper(pdsch_processor_wrapper&& other) :
53 index(other.index), free_list(other.free_list), notifier(other.notifier), processor(std::move(other.processor))
54 {
55 other.notifier = nullptr;
56 }
57
58 // See pdsch_processor interface for documentation.
59 void process(resource_grid_mapper& mapper,
60 pdsch_processor_notifier& notifier_,
61 static_vector<span<const uint8_t>, pdsch_processor::MAX_NOF_TRANSPORT_BLOCKS> data,
62 const pdsch_processor::pdu_t& pdu) override
63 {
64 // Save original notifier.
65 notifier = &notifier_;
66
67 // Process.
68 processor->process(mapper, *this, data, pdu);
69 }
70
71private:
72 // See pdsch_processor_notifier for documentation.
73 void on_finish_processing() override
74 {
75 srsran_assert(notifier != nullptr, "Invalid notifier.");
76
77 // Notify the completion of the processing.
78 notifier->on_finish_processing();
79
80 // Return the PDSCH processor identifier to the free list.
81 free_list.push_blocking(index);
82 }
83
85 unsigned index;
87 pdsch_processor_free_list& free_list;
89 pdsch_processor_notifier* notifier = nullptr;
91 std::unique_ptr<pdsch_processor> processor;
92};

100class pdsch_processor_pool : public pdsch_processor
101{
102public:
103 explicit pdsch_processor_pool(span<std::unique_ptr<pdsch_processor>> processors_) : free_list(processors_.size())
104 {
105 unsigned index = 0;
106 for (std::unique_ptr<pdsch_processor>& processor : processors_) {
107 free_list.push_blocking(index);
108 processors.emplace_back(detail::pdsch_processor_wrapper(index++, free_list, std::move(processor)));
109 }
110 }
111
112 void process(resource_grid_mapper& mapper,
113 pdsch_processor_notifier& notifier,
114 static_vector<span<const uint8_t>, MAX_NOF_TRANSPORT_BLOCKS> data,
115 const pdu_t& pdu) override
116 {
117 // Try to get a worker.
118 optional<unsigned> index = free_list.try_pop();
119
120 // If no worker is available.
121 if (!index.has_value()) {
122 srslog::fetch_basic_logger("PHY").warning("Insufficient number of PDSCH processors. Dropping PDSCH {:s}.", pdu);
123 notifier.on_finish_processing();
124 return;
125 }
126
127 // Process PDSCH.
128 processors[index.value()].process(mapper, notifier, data, pdu);
129 }
130
131private:
132 std::vector<detail::pdsch_processor_wrapper> processors;
133 detail::pdsch_processor_free_list free_list;
134};
