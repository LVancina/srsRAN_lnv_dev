54
55 void handle_sdu(byte_buffer buf, qos_flow_id_t qfi) final
56 {
57 gtpu_header hdr = {};
58 hdr.flags.version = GTPU_FLAGS_VERSION_V1;
59 hdr.flags.protocol_type = GTPU_FLAGS_GTP_PROTOCOL;
60 hdr.flags.ext_hdr = true;
61 hdr.message_type = GTPU_MSG_DATA_PDU;
62 hdr.length = buf.length() + 4 + 4;
63 hdr.teid = cfg.peer_teid;
64 hdr.next_ext_hdr_type = gtpu_extension_header_type::pdu_session_container;
65
66 byte_buffer ext_buf;
67 bit_encoder encoder{ext_buf};
68 encoder.pack(1, 4); // PDU type
69 encoder.pack(0, 4); // unused options
70 encoder.pack(0, 1); // spare
71 encoder.pack(qos_flow_id_to_uint(qfi), 7); // QFI
72
73 gtpu_extension_header ext;
74 ext.extension_header_type = gtpu_extension_header_type::pdu_session_container;
75 ext.container = ext_buf;
76
77 hdr.ext_list.push_back(ext);
78
79 bool write_ok = gtpu_write_header(buf, hdr, logger);
80
81 if (!write_ok) {
82 logger.log_error("Dropped SDU, error writing GTP-U header. teid={}", hdr.teid);
83 return;
84 }
85 logger.log_info(buf.begin(), buf.end(), "TX PDU. pdu_len={} teid={} qfi={}", buf.length(), hdr.teid, qfi);
86 send_pdu(std::move(buf), peer_sockaddr);
87 }
88
89private:
90 const gtpu_config::gtpu_tx_config cfg;
91 sockaddr_storage peer_sockaddr;
92};
