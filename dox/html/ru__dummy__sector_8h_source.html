77 // See ru_downlink_plane_handler interface for documentation.
78 void handle_dl_data(const resource_grid_context& context, const resource_grid_reader& grid) override
79 {
80 std::lock_guard<std::mutex> lock(dl_request_mutex);
81 resource_grid_context prev_context = std::exchange(dl_request[context.slot.system_slot()], context);
82
83 // If the previous slot is valid, it is a late.
84 if (prev_context.slot.valid()) {
85 logger.warning("Real-time failure in RU: received late DL request from slot {} in sector {}.",
86 prev_context.slot,
87 prev_context.sector);
88 }
89 }

91 // See ru_uplink_plane_handler interface for documentation.
92 void handle_prach_occasion(const prach_buffer_context& context, prach_buffer& buffer) override
93 {
94 std::lock_guard<std::mutex> lock(prach_request_mutex);
95 prach_buffer_context prev_context = std::exchange(prach_request[context.slot.system_slot()], context);
96
97 // Detect if there is an unhandled request from a different slot.
98 if (prev_context.slot.valid()) {
99 logger.warning(context.slot.sfn(),
100 context.slot.slot_index(),
101 "Real-time failure in RU: received late PRACH request from slot {} in sector {}.",
102 prev_context.slot,
103 prev_context.sector);
104 }
105 }

124 void new_slot_boundary(slot_point slot)
125 {
126 // Process DL request for this slot.
127 slot_point current_dl_slot = slot + dl_data_margin;
128
129 resource_grid_context context = {slot_point(), 0};
130 {
131 std::lock_guard<std::mutex> lock(dl_request_mutex);
132 context = std::exchange(dl_request[current_dl_slot.system_slot()], context);
133 }
134
135 // Notify with a warning message if the DL previous saved context do not match with the current slot.
136 if (context.slot.valid() && (context.slot != current_dl_slot)) {
137 logger.warning(current_dl_slot.sfn(),
138 current_dl_slot.slot_index(),
139 "Real-time failure in RU: detected late DL request from slot {} in sector {}.",
140 context.slot,
141 context.sector);
142 }
143
144 // Process UL request for this slot.
145 resource_grid_context ul_context = {slot_point(), 0};
146 {
147 std::lock_guard<std::mutex> lock(ul_request_mutex);
148 ul_context = std::exchange(ul_request[slot.system_slot()], ul_context);
149 }
150
151 // Check if the UL context from the request list is valid.
152 if (ul_context.slot.valid()) {
153 if (ul_context.slot == slot) {
154 // Prepare receive symbol context.
155 ru_uplink_rx_symbol_context rx_context;
156 rx_context.slot = ul_context.slot;
157 rx_context.sector = ul_context.sector;
158 rx_context.symbol_id = MAX_NSYMB_PER_SLOT - 1;
159
160 // Notify received resource grid.
161 symbol_notifier.on_new_uplink_symbol(rx_context, rx_symbols_resource_grid);
162 } else {
163 // Notify with a warning message if the UL previous saved context do not match with the current slot.
164 logger.warning(slot.sfn(),
165 slot.slot_index(),
166 "Real-time failure in RU: detected late UL request from slot {} in sector {}.",
167 ul_context.slot,
168 ul_context.sector);
169 }
170 }
171
172 // Process PRACH request for this slot.
173 prach_buffer_context prach_context{};
174 {
175 std::lock_guard<std::mutex> lock(prach_request_mutex);
176 prach_context = std::exchange(prach_request[slot.system_slot()], prach_context);
177 }
178
179 // Check if the UL context from the request list is valid.
180 if (prach_context.slot.valid()) {
181 if (prach_context.slot == slot) {
182 // Notify received PRACH buffer.
183 symbol_notifier.on_new_prach_window_data(prach_context, rx_symbols_prach_buffer.get_buffer(prach_context));
184 } else {
185 // Notify with a warning message if the UL previous saved context do not match with the current slot.
186 logger.warning(slot.sfn(),
187 slot.slot_index(),
188 "Real-time failure in RU: detected late PRACH request from slot {} in sector {}.",
189 prach_context.slot,
190 prach_context.sector);
191 }
192 }
193 }
