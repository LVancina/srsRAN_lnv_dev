33/// \brief Collects PRACH preamble information parameters.
34///
35/// The parameters are used and described in TS38.211 Section 6.3.3.1.
36struct prach_preamble_information {
37 /// Sequence length in frequency domain, parameter \f$L_{RA}\f$.
38 unsigned sequence_length;
39 /// Parameter \f$\Delta f^{RA}\f$.
40 prach_subcarrier_spacing scs;
41 /// Parameter \f$N_u\f$. Expressed in units of the reference symbol time \f$\kappa\f$.
42 phy_time_unit symbol_length;
43 /// Parameter \f$N_{CP}^{RA}\f$. Expressed in units of the reference symbol time \f$\kappa\f$.
44 phy_time_unit cp_length;
45 /// Flag: true if the preamble supports the restricted sets A and B.
46 bool support_restricted_sets;
47};
48
49/// \brief Get long PRACH preamble information as per TS38.211 Table 6.3.3.1-1.
50///
51/// An assertion is triggered if the PRACH preamble format is short.
52///
53/// \param[in] format PRACH preamble format.
54/// \return PRACH preamble information.
55prach_preamble_information get_prach_preamble_long_info(prach_format_type format);
56
57/// \brief Get short PRACH preamble information as per TS38.211 Table 6.3.3.1-2.
58///
59/// An assertion is triggered if:
60/// - the PRACH preamble format is long, or
61/// - the random access subcarrier spacing does not satisfy \f$\Delta f ^\textup{RA} = 15 \cdot 2^\mu\f$ where \f$\mu
62/// \in \{0, 1, 2, 3\}\f$.
63///
64/// \param[in] format PRACH preamble format.
65/// \param[in] ra_scs Random access subcarrier spacing \f$\Delta f ^\textup{RA}\f$.
66/// \param[in] last_occasion Set to true if \f$n_\textup{t}^\textup{RA}=N_\textup{t}^\textup{RA,slot}-1\f$
67/// \return PRACH preamble information.
68prach_preamble_information
69get_prach_preamble_short_info(prach_format_type format, prach_subcarrier_spacing ra_scs, bool last_occasion);
70
71/// \brief Gets the PRACH window duration.
72/// \param[in] format PRACH preamble format.
73/// \param[in] pusch_scs Uplink resource grid subcarrier spacing. Determines the numerology \f$\mu\f$.
74/// \param[in] start_symbol_index OFDM symbol index within the slot that marks the start of the acquisition window for
75/// the first time-domain PRACH occasion.
76/// \param[in] nof_td_occasions Number of time-domain occasions within a PRACH window.
77/// \return The PRACH window duration from the beginning of the slot to the end of the last time-domain occasion.
78/// \note The RA subcarrier spacing for short preambles is assumed to be equal to the uplink resource grid subcarrier
79/// spacing.
80phy_time_unit get_prach_window_duration(prach_format_type format,
81 subcarrier_spacing pusch_scs,
82 unsigned start_symbol_index,
83 unsigned nof_td_occasions);
84
85/// \brief Collects PRACH preamble duration information.
86struct prach_symbols_slots_duration {
87 /// Duration of the PRACH Preamble in slots, with reference to the PUSCH SCS.
88 unsigned prach_length_slots;
89 /// PRACH starting slot within the subframe, with reference to the PUSCH SCS. For 15kHz PUSCH SCS, values: {0}; for
90 /// 30kHz PUSCH SCS, values: {0, 1}
91 unsigned start_slot_pusch_scs;
92 /// PRACH duration in symbols, with reference to the PUSCH SCS.
93 unsigned nof_symbols;
94 /// PRACH starting symbol within the slot, with reference to the PUSCH SCS.
95 unsigned start_symbol_pusch_scs;
96};
97
98/// \brief Compute PRACH preamble duration information.
99///
100/// \param[in] prach_cfg PRACH preamble configuration.
101/// \param[in] pusch_scs PUSCH SCS that is used as a reference for symbol and slot unit.
102/// \return PRACH preamble duration information.
103prach_symbols_slots_duration get_prach_duration_info(const prach_configuration& prach_cfg,
104 subcarrier_spacing pusch_scs);
