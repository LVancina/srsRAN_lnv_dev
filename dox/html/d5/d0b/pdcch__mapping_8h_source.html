33inline prb_index_list cce_to_prb_mapping(const bwp_configuration& bwp_cfg,
34 const coreset_configuration& cs_cfg,
35 pci_t pci,
36 aggregation_level aggr_lvl,
37 unsigned ncce)
38{
39 if (cs_cfg.id == 0) {
40 return cce_to_prb_mapping_coreset0(cs_cfg.get_coreset_start_crb(),
41 get_coreset_nof_prbs(cs_cfg),
42 cs_cfg.duration,
43 pci,
44 to_nof_cces(aggr_lvl),
45 ncce);
46 }
47 if (cs_cfg.interleaved.has_value()) {
48 return cce_to_prb_mapping_interleaved(bwp_cfg.crbs.start(),
49 cs_cfg.freq_domain_resources(),
50 cs_cfg.duration,
51 cs_cfg.interleaved->reg_bundle_sz,
52 cs_cfg.interleaved->interleaver_sz,
53 cs_cfg.interleaved->shift_index,
54 to_nof_cces(aggr_lvl),
55 ncce);
56 }
57 return cce_to_prb_mapping_non_interleaved(
58 bwp_cfg.crbs.start(), cs_cfg.freq_domain_resources(), cs_cfg.duration, to_nof_cces(aggr_lvl), ncce);
59}
