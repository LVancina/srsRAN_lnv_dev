48/// Maximum number of UEs per DU (implementation-defined).
49const uint16_t MAX_NOF_UES_PER_DU = 1024;
50/// Maximum number of DUs supported by CU-CP (implementation-defined).
51const uint16_t MAX_NOF_DUS = 65535;
52/// Maximum number of UEs supported by CU-CP (implementation-defined).
53#define MAX_NOF_CU_UES (MAX_NOF_DUS * MAX_NOF_UES_PER_DU)
54/// Maximum number of CU-UPs supported by CU-CP (implementation-defined).
55const uint16_t MAX_NOF_CU_UPS = 65535;
56/// Maximum number of cells per DU supported by CU-CP (implementation-defined).
57const uint16_t MAX_NOF_DU_CELLS = 16;
58
59/// \brief ue_index internally used to identify the UE CU-CP-wide.
60/// \remark The ue_index is derived from the DU index and the maximum number of UEs per DU.
61enum class ue_index_t : uint64_t { min = 0, max = MAX_NOF_CU_UES - 1, invalid = MAX_NOF_CU_UES };
62
63/// Convert ue_index type to integer.
64inline uint64_t ue_index_to_uint(ue_index_t index)
65{
66 return static_cast<uint64_t>(index);
67}
68
69/// Convert integer to ue_index type.
70inline ue_index_t uint_to_ue_index(std::underlying_type_t<ue_index_t> index)
71{
72 return static_cast<ue_index_t>(index);
73}
74
75/// Maximum number of DUs supported by CU-CP (implementation-defined).
76enum class du_index_t : uint16_t { min = 0, max = MAX_NOF_DUS - 1, invalid = MAX_NOF_DUS };
77
78/// Convert integer to DU index type.
79constexpr inline du_index_t uint_to_du_index(std::underlying_type_t<du_index_t> index)
80{
81 return static_cast<du_index_t>(index);
82}
83
84/// Convert DU index type to integer.
85constexpr inline std::underlying_type_t<du_index_t> du_index_to_uint(du_index_t du_index)
86{
87 return static_cast<std::underlying_type_t<du_index_t>>(du_index);
88}
89
90/// Get DU index from UE index.
91inline du_index_t get_du_index_from_ue_index(ue_index_t index)
92{
93 if (index == ue_index_t::invalid) {
94 return du_index_t::invalid;
95 }
96 return uint_to_du_index((ue_index_to_uint(index) / MAX_NOF_UES_PER_DU));
97}
98
99/// Generate a UE index from DU index and an index.
100inline ue_index_t generate_ue_index(du_index_t du_index, uint16_t index)
101{
102 return uint_to_ue_index(du_index_to_uint(du_index) * MAX_NOF_UES_PER_DU + index);
103}
104
105/// Maximum number of CU-UPs supported by CU-CP (implementation-defined).
106enum class cu_up_index_t : uint16_t { min = 0, max = MAX_NOF_CU_UPS - 1, invalid = MAX_NOF_CU_UPS };
107
108/// Convert integer to CU-UP index type.
109constexpr inline cu_up_index_t uint_to_cu_up_index(std::underlying_type_t<cu_up_index_t> index)
110{
111 return static_cast<cu_up_index_t>(index);
112}
113
114/// Convert CU-UP index type to integer.
115constexpr inline std::underlying_type_t<cu_up_index_t> cu_up_index_to_uint(cu_up_index_t cu_up_index)
116{
117 return static_cast<std::underlying_type_t<cu_up_index_t>>(cu_up_index);
118}
119
120/// Maximum number of cells per DU supported by CU-CP (implementation-defined).
121enum class du_cell_index_t : uint16_t { min = 0, max = MAX_NOF_DU_CELLS - 1, invalid = MAX_NOF_DU_CELLS };
122
123/// Convert integer to DU cell index type.
124inline du_cell_index_t uint_to_du_cell_index(std::underlying_type_t<du_cell_index_t> index)
125{
126 return static_cast<du_cell_index_t>(index);
127}
128
129/// Convert DU cell index type to integer.
130constexpr inline std::underlying_type_t<du_cell_index_t> du_cell_index_to_uint(du_cell_index_t du_cell_index)
131{
132 return static_cast<std::underlying_type_t<du_cell_index_t>>(du_cell_index);
133}
