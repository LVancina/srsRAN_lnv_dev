31/// Radio Unit radio notification logger.
32class ru_radio_notification_handler_logger : public radio_notification_handler
33{
34public:
35 ru_radio_notification_handler_logger(std::unique_ptr<radio_notification_handler> handler_,
36 srslog::basic_logger& logger_) :
37 handler(std::move(handler_)), logger(logger_)
38 {
39 }
40
41 // See interface for documentation.
42 void on_radio_rt_event(const event_description& description) override
43 {
44 // Forward event.
45 if (handler) {
46 handler->on_radio_rt_event(description);
47 }
48
49 if (description.type == event_type::LATE or description.type == event_type::UNDERFLOW or
50 description.type == event_type::OVERFLOW) {
51 logger.warning("Real-time failure in RF: {}", description.type.to_string());
52 }
53
54 static const auto& log_format_debug = "Real-time failure in RF: Type={} Source={} Timestamp={}";
55
56 if (description.timestamp.has_value()) {
57 logger.debug(log_format_debug,
58 description.type.to_string(),
59 description.source.to_string(),
60 description.timestamp.value());
61 } else {
62 logger.debug(log_format_debug, description.type.to_string(), description.source.to_string(), "na");
63 }
64 }
65
66private:
67 std::unique_ptr<radio_notification_handler> handler;
68 srslog::basic_logger& logger;
69};

71/// Radio Unit radio notification counter.
72class ru_radio_notification_handler_counter : public radio_notification_handler
73{
74public:
75 explicit ru_radio_notification_handler_counter(std::unique_ptr<radio_notification_handler> handler_,
76 task_executor& executor_,
77 unsigned print_interval_in_slots_) :
78 handler(std::move(handler_)),
79 executor(executor_),
80 print_interval_in_slots(print_interval_in_slots_),
81 slot_counter(0)
82 {
83 }
84
85 // See interface for documentation.
86 void on_radio_rt_event(const event_description& description) override
87 {
88 if (handler) {
89 handler->on_radio_rt_event(description);
90 }
91
92 // Count events.
93 switch (description.type) {
94 case event_type::LATE:
95 ++late_count;
96 break;
97 case event_type::UNDERFLOW:
98 ++underflow_count;
99 break;
100 case event_type::OVERFLOW:
101 ++overflow_count;
102 break;
103 case event_type::UNDEFINED:
104 case event_type::START_OF_BURST:
105 case event_type::END_OF_BURST:
106 case event_type::OTHER:
107 // Uncounted.
108 break;
109 }
110 }
111
112 /// Enqueue the task of printing statistics.
113 void print()
114 {
115 ++slot_counter;
116
117 if (slot_counter == print_interval_in_slots) {
118 slot_counter = 0;
119 bool dispatched = executor.defer([this]() { print_statistics(); });
120 // TODO: handle failure to dispatch print statistics task.
121 (void)dispatched;
122 }
123 }
124
125private:
126 void print_statistics()
127 {
128 unsigned temp_late_count = late_count.exchange(0);
129 unsigned temp_underflow_count = underflow_count.exchange(0);
130 unsigned temp_overflow_count = overflow_count.exchange(0);
131
132 // Print late, underflow and overflow counts since last print.
133 if ((temp_late_count != 0) || (temp_underflow_count != 0) || (temp_overflow_count != 0)) {
134 fmt::print(
135 "Late: {}; Underflow: {}; Overflow: {};\n", temp_late_count, temp_underflow_count, temp_overflow_count);
136 }
137 }
138
139private:
140 std::atomic<unsigned> late_count = {};
141 std::atomic<unsigned> underflow_count = {};
142 std::atomic<unsigned> overflow_count = {};
143 std::unique_ptr<radio_notification_handler> handler;
144 task_executor& executor;
145 const unsigned print_interval_in_slots;
146 unsigned slot_counter;
147};
