38/// Internal PDSCH subprocessor.
39class pdsch_block_processor : public resource_grid_mapper::symbol_buffer
40{
41public:
42 pdsch_block_processor(ldpc_segmenter_tx& segmenter_,
43 ldpc_encoder& encoder_,
44 ldpc_rate_matcher& rate_matcher_,
45 pseudo_random_generator& scrambler_,
46 modulation_mapper& modulator_) :
47 segmenter(segmenter_), encoder(encoder_), rate_matcher(rate_matcher_), scrambler(scrambler_), modulator(modulator_)
48 {
49 }
50
51 /// \brief Configures a new transmission.
52 /// \param[in] data Transport block data.
53 /// \param[in] i_cw Codeword index.
54 /// \param[in] pdu PDSCH transmission parameters.
55 void configure_new_transmission(span<const uint8_t> data, unsigned i_cw, const pdsch_processor::pdu_t& pdu);
56
57 // See interface for documentation.
58 unsigned get_max_block_size() const override;
59
60 bool empty() const override;
61
62 // See interface for documentation.
63 span<const ci8_t> pop_symbols(unsigned block_size) override;
64
65private:
66 /// \brief Determines the maximum block size that can be processed.
67 ///
68 /// Set it to fit a complex float block in a cache page. Typically a cache page is 4096 bytes. It can be checked
69 /// running the command <tt>$ getconf -a | grep PAGESIZE</tt>.
70 static constexpr unsigned max_block_size = 4096 / sizeof(ci8_t);
71
72 /// LDPC segmenter.
73 ldpc_segmenter_tx& segmenter;
74 /// LDPC encoder.
75 ldpc_encoder& encoder;
76 /// LDPC rate matcher.
77 ldpc_rate_matcher& rate_matcher;
78 /// Scrambling pseudo-random generator.
79 pseudo_random_generator& scrambler;
80 /// Modulation mapper.
81 modulation_mapper& modulator;
82 /// Buffer for storing data segments obtained after transport block segmentation.
83 static_vector<described_segment, MAX_NOF_SEGMENTS> d_segments;
84 /// Temporary packed bits.
85 static_bit_buffer<ldpc::MAX_CODEBLOCK_RM_SIZE> temp_codeblock;
86 /// Current transmission modulation.
87 modulation_scheme modulation;
88 /// Current codeblock index.
89 unsigned next_i_cb = 0;
90 /// Temporary storage of codeblock symbols.
91 static_bit_buffer<ldpc::MAX_CODEBLOCK_RM_SIZE> rm_buffer;
92 /// Temporary storage of codeblock symbols.
93 std::array<ci8_t, ldpc::MAX_CODEBLOCK_RM_SIZE> temp_codeblock_symbols;
94 /// Current view of the codeblock modulated symbols.
95 span<ci8_t> codeblock_symbols;
96
97 /// Processes a new codeblock and writes the new data in \ref encoded_bit_buffer.
98 void new_codeblock();
99};

101/// Implements a PDSCH processor with low memory footprint codeblock processing.
102class pdsch_processor_lite_impl : public pdsch_processor
103{
104public:
105 /// Creates a concurrent PDSCH processor with all the dependencies.
106 pdsch_processor_lite_impl(std::unique_ptr<ldpc_segmenter_tx> segmenter_,
107 std::unique_ptr<ldpc_encoder> encoder_,
108 std::unique_ptr<ldpc_rate_matcher> rate_matcher_,
109 std::unique_ptr<pseudo_random_generator> scrambler_,
110 std::unique_ptr<modulation_mapper> modulator_,
111 std::unique_ptr<dmrs_pdsch_processor> dmrs_) :
112 segmenter(std::move(segmenter_)),
113 encoder(std::move(encoder_)),
114 rate_matcher(std::move(rate_matcher_)),
115 scrambler(std::move(scrambler_)),
116 modulator(std::move(modulator_)),
117 dmrs(std::move(dmrs_)),
118 subprocessor(*segmenter, *encoder, *rate_matcher, *scrambler, *modulator)
119 {
120 srsran_assert(segmenter != nullptr, "Invalid segmenter pointer.");
121 srsran_assert(scrambler != nullptr, "Invalid scrambler pointer.");
122 srsran_assert(dmrs != nullptr, "Invalid dmrs pointer.");
123 }
124
125 // See interface for documentation.
126 void process(resource_grid_mapper& mapper,
127 pdsch_processor_notifier& notifier,
128 static_vector<span<const uint8_t>, MAX_NOF_TRANSPORT_BLOCKS> data,
129 const pdu_t& pdu) override;
130
131private:
132 /// \brief Processes DM-RS.
133 /// \param[out] mapper Resource grid mapper interface.
134 /// \param[in] pdu Necessary parameters to process the DM-RS.
135 void process_dmrs(resource_grid_mapper& mapper, const pdu_t& pdu);
136
137 /// Pointer to an LDPC segmenter.
138 std::unique_ptr<ldpc_segmenter_tx> segmenter;
139 /// Pointer to an LDPC encoder.
140 std::unique_ptr<ldpc_encoder> encoder;
141 /// Pointer to an LDPC rate matcher.
142 std::unique_ptr<ldpc_rate_matcher> rate_matcher;
143 /// Pointer to scrambling pseudo-random generator.
144 std::unique_ptr<pseudo_random_generator> scrambler;
145 /// Pointer to modulation mapper.
146 std::unique_ptr<modulation_mapper> modulator;
147 /// Pointer to DM-RS processor.
148 std::unique_ptr<dmrs_pdsch_processor> dmrs;
149 /// Internal block processor.
150 pdsch_block_processor subprocessor;
151};
