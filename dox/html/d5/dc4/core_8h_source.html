341/**
342 An implementation of ``std::basic_string_view`` for pre-C++17. It provides a
343 subset of the API. ``fmt::basic_string_view`` is used for format strings even
344 if ``std::string_view`` is available to prevent issues when a library is
345 compiled with a different ``-std`` option than the client code (which is not
346 recommended).
347 */
348template <typename Char> class basic_string_view {
349 private:
350 const Char* data_;
351 size_t size_;
352
353 public:
354 using value_type = Char;
355 using iterator = const Char*;
356
357 constexpr basic_string_view() FMT_NOEXCEPT : data_(nullptr), size_(0) {}
358
359 /** Constructs a string reference object from a C string and a size. */
360 constexpr basic_string_view(const Char* s, size_t count) FMT_NOEXCEPT
361 : data_(s),
362 size_(count) {}
363
364 /**
365 \rst
366 Constructs a string reference object from a C string computing
367 the size with ``std::char_traits<Char>::length``.
368 \endrst
369 */
370#if __cplusplus >= 201703L // C++17's char_traits::length() is constexpr.
371 FMT_CONSTEXPR
372#endif
373 basic_string_view(const Char* s)
374 : data_(s), size_(std::char_traits<Char>::length(s)) {}
375
376 /** Constructs a string reference from a ``std::basic_string`` object. */
377 template <typename Traits, typename Alloc>
378 FMT_CONSTEXPR basic_string_view(
379 const std::basic_string<Char, Traits, Alloc>& s) FMT_NOEXCEPT
380 : data_(s.data()),
381 size_(s.size()) {}
382
383 template <typename S, FMT_ENABLE_IF(std::is_same<
384 S, detail::std_string_view<Char>>::value)>
385 FMT_CONSTEXPR basic_string_view(S s) FMT_NOEXCEPT : data_(s.data()),
386 size_(s.size()) {}
387
388 /** Returns a pointer to the string data. */
389 constexpr const Char* data() const { return data_; }
390
391 /** Returns the string size. */
392 constexpr size_t size() const { return size_; }
393
394 constexpr iterator begin() const { return data_; }
395 constexpr iterator end() const { return data_ + size_; }
396
397 constexpr const Char& operator[](size_t pos) const { return data_[pos]; }
398
399 FMT_CONSTEXPR void remove_prefix(size_t n) {
400 data_ += n;
401 size_ -= n;
402 }
403
404 // Lexicographically compare this string reference to other.
405 int compare(basic_string_view other) const {
406 size_t str_size = size_ < other.size_ ? size_ : other.size_;
407 int result = std::char_traits<Char>::compare(data_, other.data_, str_size);
408 if (result == 0)
409 result = size_ == other.size_ ? 0 : (size_ < other.size_ ? -1 : 1);
410 return result;
411 }
412
413 friend bool operator==(basic_string_view lhs, basic_string_view rhs) {
414 return lhs.compare(rhs) == 0;
415 }
416 friend bool operator!=(basic_string_view lhs, basic_string_view rhs) {
417 return lhs.compare(rhs) != 0;
418 }
419 friend bool operator<(basic_string_view lhs, basic_string_view rhs) {
420 return lhs.compare(rhs) < 0;
421 }
422 friend bool operator<=(basic_string_view lhs, basic_string_view rhs) {
423 return lhs.compare(rhs) <= 0;
424 }
425 friend bool operator>(basic_string_view lhs, basic_string_view rhs) {
426 return lhs.compare(rhs) > 0;
427 }
428 friend bool operator>=(basic_string_view lhs, basic_string_view rhs) {
429 return lhs.compare(rhs) >= 0;
430 }
431};

524/**
525 \rst
526 Parsing context consisting of a format string range being parsed and an
527 argument counter for automatic indexing.
528
529 You can use one of the following type aliases for common character types:
530
531 +-----------------------+-------------------------------------+
532 | Type | Definition |
533 +=======================+=====================================+
534 | format_parse_context | basic_format_parse_context<char> |
535 +-----------------------+-------------------------------------+
536 | wformat_parse_context | basic_format_parse_context<wchar_t> |
537 +-----------------------+-------------------------------------+
538 \endrst
539 */
540template <typename Char, typename ErrorHandler = detail::error_handler>
541class basic_format_parse_context : private ErrorHandler {
542 private:
543 basic_string_view<Char> format_str_;
544 int next_arg_id_;
545
546 public:
547 using char_type = Char;
548 using iterator = typename basic_string_view<Char>::iterator;
549
550 explicit constexpr basic_format_parse_context(
551 basic_string_view<Char> format_str, ErrorHandler eh = {})
552 : ErrorHandler(eh), format_str_(format_str), next_arg_id_(0) {}
553
554 /**
555 Returns an iterator to the beginning of the format string range being
556 parsed.
557 */
558 constexpr iterator begin() const FMT_NOEXCEPT { return format_str_.begin(); }
559
560 /**
561 Returns an iterator past the end of the format string range being parsed.
562 */
563 constexpr iterator end() const FMT_NOEXCEPT { return format_str_.end(); }
564
565 /** Advances the begin iterator to ``it``. */
566 FMT_CONSTEXPR void advance_to(iterator it) {
567 format_str_.remove_prefix(detail::to_unsigned(it - begin()));
568 }
569
570 /**
571 Reports an error if using the manual argument indexing; otherwise returns
572 the next argument index and switches to the automatic indexing.
573 */
574 FMT_CONSTEXPR int next_arg_id() {
575 // Don't check if the argument id is valid to avoid overhead and because it
576 // will be checked during formatting anyway.
577 if (next_arg_id_ >= 0) return next_arg_id_++;
578 on_error("cannot switch from manual to automatic argument indexing");
579 return 0;
580 }
581
582 /**
583 Reports an error if using the automatic argument indexing; otherwise
584 switches to the manual indexing.
585 */
586 FMT_CONSTEXPR void check_arg_id(int) {
587 if (next_arg_id_ > 0)
588 on_error("cannot switch from automatic to manual argument indexing");
589 else
590 next_arg_id_ = -1;
591 }
592
593 FMT_CONSTEXPR void check_arg_id(basic_string_view<Char>) {}
594
595 FMT_CONSTEXPR void on_error(const char* message) {
596 ErrorHandler::on_error(message);
597 }
598
599 constexpr ErrorHandler error_handler() const { return *this; }
600};

625 \rst

627 class and shouldn't be used directly, only via `~fmt::basic_memory_buffer`.
628 \endrst
629 */
630template <typename T> class buffer {
631 private:
632 T* ptr_;
633 size_t size_;
634 size_t capacity_;
635
636 protected:
637 // Don't initialize ptr_ since it is not accessed to save a few cycles.
638 FMT_SUPPRESS_MSC_WARNING(26495)
639 buffer(size_t sz) FMT_NOEXCEPT : size_(sz), capacity_(sz) {}
640
641 buffer(T* p = nullptr, size_t sz = 0, size_t cap = 0) FMT_NOEXCEPT
642 : ptr_(p),
643 size_(sz),
644 capacity_(cap) {}
645
646 /** Sets the buffer data and capacity. */
647 void set(T* buf_data, size_t buf_capacity) FMT_NOEXCEPT {
648 ptr_ = buf_data;
649 capacity_ = buf_capacity;
650 }
651
652 /** Increases the buffer capacity to hold at least *capacity* elements. */
653 virtual void grow(size_t capacity) = 0;
654
655 public:
656 using value_type = T;
657 using const_reference = const T&;
658
659 buffer(const buffer&) = delete;
660 void operator=(const buffer&) = delete;
661 virtual ~buffer() = default;
662
663 T* begin() FMT_NOEXCEPT { return ptr_; }
664 T* end() FMT_NOEXCEPT { return ptr_ + size_; }
665
666 const T* begin() const FMT_NOEXCEPT { return ptr_; }
667 const T* end() const FMT_NOEXCEPT { return ptr_ + size_; }
668
669 /** Returns the size of this buffer. */
670 size_t size() const FMT_NOEXCEPT { return size_; }
671
672 /** Returns the capacity of this buffer. */
673 size_t capacity() const FMT_NOEXCEPT { return capacity_; }
674
675 /** Returns a pointer to the buffer data. */
676 T* data() FMT_NOEXCEPT { return ptr_; }
677
678 /** Returns a pointer to the buffer data. */
679 const T* data() const FMT_NOEXCEPT { return ptr_; }
680
681 /**
682 Resizes the buffer. If T is a POD type new elements may not be initialized.
683 */
684 void resize(size_t new_size) {
685 reserve(new_size);
686 size_ = new_size;
687 }
688
689 /** Clears this buffer. */
690 void clear() { size_ = 0; }

1150/**
1151 \rst
1152 Visits an argument dispatching to the appropriate visit method based on
1153 the argument type. For example, if the argument type is ``double`` then
1154 ``vis(value)`` will be called with the value of type ``double``.
1155 \endrst
1156 */
1157template <typename Visitor, typename Context>
1158FMT_CONSTEXPR_DECL FMT_INLINE auto visit_format_arg(
1159 Visitor&& vis, const basic_format_arg<Context>& arg) -> decltype(vis(0)) {
1160 using char_type = typename Context::char_type;
1161 switch (arg.type_) {
1162 case detail::type::none_type:
1163 break;
1164 case detail::type::int_type:
1165 return vis(arg.value_.int_value);
1166 case detail::type::uint_type:
1167 return vis(arg.value_.uint_value);
1168 case detail::type::long_long_type:
1169 return vis(arg.value_.long_long_value);
1170 case detail::type::ulong_long_type:
1171 return vis(arg.value_.ulong_long_value);
1172#if FMT_USE_INT128
1173 case detail::type::int128_type:
1174 return vis(arg.value_.int128_value);
1175 case detail::type::uint128_type:
1176 return vis(arg.value_.uint128_value);
1177#else
1178 case detail::type::int128_type:
1179 case detail::type::uint128_type:
1180 break;
1181#endif
1182 case detail::type::bool_type:
1183 return vis(arg.value_.bool_value);
1184 case detail::type::char_type:
1185 return vis(arg.value_.char_value);
1186 case detail::type::float_type:
1187 return vis(arg.value_.float_value);
1188 case detail::type::double_type:
1189 return vis(arg.value_.double_value);
1190 case detail::type::long_double_type:
1191 return vis(arg.value_.long_double_value);
1192 case detail::type::cstring_type:
1193 return vis(arg.value_.string.data);
1194 case detail::type::string_type:
1195 return vis(basic_string_view<char_type>(arg.value_.string.data,
1196 arg.value_.string.size));
1197 case detail::type::pointer_type:
1198 return vis(arg.value_.pointer);
1199 case detail::type::custom_type:
1200 return vis(typename basic_format_arg<Context>::handle(arg.value_.custom));
1201 }
1202 return vis(monostate());
1203}
