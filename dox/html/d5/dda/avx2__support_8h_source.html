58/// \brief Scales packed 8-bit integers in \c a by the scaling factor \c sf.
59///
60/// Values of \c a larger than \c max or smaller than \c -max are forwarded unaltered.
61/// \param[in] a Vector of packed 8-bit integers.
62/// \param[in] sf Scaling factor (0, 1].
63/// \param[in] max Maximum input value (in absolute value) to which the scaling is applied [0, 127).
64/// \return Vector of packed 8-bit integers with the scaling result.
65inline __m256i scale_epi8(__m256i a, float sf, uint8_t max)
66{
67 srsran_assert((sf > 0) && (sf <= 1), "Scaling factor out of range.");
68 srsran_assert(max < 127, "Parameter max out of range.");
69
70 if (sf >= .9999) {
71 return a;
72 }
73
74 static const __m256i mask_even_epi8 = _mm256_set1_epi16(0x00ff);
75 // FLOAT2INT = 2^16 = 65536
76 static constexpr unsigned FLOAT2INT = 1U << 16U;
77 const __m256i MAX_epi8 = _mm256_set1_epi8(max);
78 const __m256i MIN_epi8 = _mm256_set1_epi8(-max);
79
80 // Create a register filled with copies of the scaling factor times FLOAT2INT. Note that the upper bound on sf implies
81 // that (sf * FLOAT2INT) can be represented over 16 bits.
82 __m256i sf_epi16 = _mm256_set1_epi16(static_cast<uint16_t>(sf * FLOAT2INT));
83 // Even bytes in a, alternated with a zero-valued byte.
84 __m256i even_epi16 = _mm256_and_si256(a, mask_even_epi8);
85 // Odd bytes in a, moved to the even positions and alternated with a zero-valued byte.
86 __m256i odd_epi16 = _mm256_srli_epi16(a, 8);
87
88 // This is equivalent to multiply each byte by sf: indeed, we multiply each byte by (sf * FLOAT2INT), store the result
89 // in over 32 bit and pick only the 16 most significant bits (i.e., we divide by FLOAT2INT).
90 __m256i p_even_epi16 = _mm256_mulhi_epu16(even_epi16, sf_epi16);
91 __m256i p_odd_epi16 = _mm256_mulhi_epu16(odd_epi16, sf_epi16);
92
93 // Move the odd bits back to their original positions.
94 p_odd_epi16 = _mm256_slli_epi16(p_odd_epi16, 8);
95
96 // Combine even and odd bits. Note that the "odd" bytes of p_even_epi16 are equal to 0 (the result of multiplying a
97 // byte by a 16-bit value will occupy at most 24 of the 32 bits).
98 __m256i product_epi8 = _mm256_xor_si256(p_even_epi16, p_odd_epi16);
99
100 // Replace values corresponding to "large" inputs with their original values.
101 __m256i mask_epi8 = _mm256_cmpgt_epi8(a, MAX_epi8);
102 product_epi8 = _mm256_blendv_epi8(product_epi8, a, mask_epi8);
103 mask_epi8 = _mm256_cmpgt_epi8(MIN_epi8, a);
104 product_epi8 = _mm256_blendv_epi8(product_epi8, a, mask_epi8);
105 return product_epi8;
106}
