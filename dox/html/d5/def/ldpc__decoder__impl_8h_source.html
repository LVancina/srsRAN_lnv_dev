37/// Template LDPC decoder
38class ldpc_decoder_impl : public ldpc_decoder
39{
40public:
41 /// \brief Maximum degree of a check node.
42 ///
43 /// In the base graph, each check node is connected, at most, to all variable nodes in the high-rate region
44 /// (of maximum length max_BG_K + 4) and an extra variable node in the extension region.
45 static constexpr unsigned MAX_CHECK_NODE_DEGREE = ldpc::MAX_BG_K + 5;
46
47 /// Default constructor.
48 ldpc_decoder_impl() = default;
49
50 // See interface for the documentation.
51 optional<unsigned> decode(bit_buffer& output,
52 span<const log_likelihood_ratio> input,
53 crc_calculator* crc,
54 const configuration& cfg) override;
55
56private:
57 /// Initializes the decoder inner variables.
58 void init(const configuration& cfg);
59
60 /// Initializes implementation-specific inner variables.
61 virtual void specific_init() = 0;
62
63 /// Loads the input log-likelihood ratios into the soft-bit buffers.
64 void load_soft_bits(span<const log_likelihood_ratio> llrs);
65
66 /// \brief Updates the messages going from variable nodes to check nodes.
67 /// \param[in] check_node The check node (in the base graph) the messages are directed to.
68 void update_variable_to_check_messages(unsigned check_node);
69
70 /// \brief Updates the messages going from check nodes to variable nodes.
71 /// \param[in] check_node The check node (in the base graph) the messages are coming from.
72 void update_check_to_variable_messages(unsigned check_node);
73
74 /// Gets a view to a node soft bits.
75 span<log_likelihood_ratio> get_soft_bits(unsigned i_node)
76 {
77 return span<log_likelihood_ratio>(soft_bits).subspan(i_node * node_size_byte, node_size_byte);
78 }
79
80 /// Gets a view to the check-to-var message from layer (check node) \c i_layer to variable node \c i_node.
81 span<log_likelihood_ratio> get_check_to_var(unsigned i_layer, unsigned i_node)
82 {
83 return span<log_likelihood_ratio>(check_to_var[i_layer]).subspan(i_node * node_size_byte, node_size_byte);
84 }
85
86 /// Gets a view to the variable-to-check message from variable node \c i_node to the current layer (check node) after
87 /// applying a \c shift.
88 span<log_likelihood_ratio> get_var_to_check(unsigned i_node, unsigned shift)
89 {
90 return span<log_likelihood_ratio>(var_to_check).subspan(2 * i_node * node_size_byte + shift, node_size_byte);
91 }
92
93 /// \brief Helper function for \ref update_check_to_variable_messages - Analyzes the variable-to-check messages for a
94 /// check node.
95 ///
96 /// Finds the two absolute minimums of all the arriving variable-to-check messages and the product of their signs.
97 /// The function operates on all the messages corresponding to the variable-to-check edges obtained from lifting a
98 /// single variable-to-check edge in the base graph. Therefore, all buffers below have a size equal to the lifting
99 /// size.
100 /// \param[out] min_var_to_check Buffer to store the value of the absolute minimum variable-to-check message.
101 /// \param[out] second_min_var_to_check Buffer to store the value of the second absolute minimum variable-to-check
102 /// message.
103 /// \param[out] min_var_to_check_index Buffer to store the index of the absolute minimum variable-to-check message.
104 /// \param[out] sign_prod_var_to_check Buffer to store the sign of the product of all the messages reaching the check
105 /// node.
106 /// \param[in] rotated_node Buffer with the cyclically-shifted version of the considered variable-to-check
107 /// messages.
108 /// \param[in] var_node Counting index of the variable node, that is it can only take values between
109 /// zero and the degree of the check node minus one.
110 virtual void analyze_var_to_check_msgs(span<log_likelihood_ratio> min_var_to_check,
111 span<log_likelihood_ratio> second_min_var_to_check,
112 span<uint8_t> min_var_to_check_index,
113 span<uint8_t> sign_prod_var_to_check,
114 span<const log_likelihood_ratio> rotated_node,
115 unsigned var_node) = 0;
116
117 /// \brief Helper function for \ref update_check_to_variable_messages - Computes the new check-to-variable messages.
118 ///
119 /// The function operates on all the messages corresponding to the variable-to-check edges obtained from lifting a
120 /// single variable-to-check edge in the base graph. Therefore, all buffers below have a size equal to the lifting
121 /// size.
122 /// \param[out] this_check_to_var Buffer to store the check-to-variable messages for the considered check node.
123 /// \param[in] this_var_to_check Buffer with the variable-to-check messages for the considered check node.
124 /// \param[in] rotated_node Buffer containing a cyclically-shifted version of the variable-to-check
125 /// messages.
126 /// \param[in] min_var_to_check Buffer containing the value of the absolute minimum variable-to-check message.
127 /// \param[in] second_min_var_to_check Buffer containing the value of the second absolute minimum variable-to-check
128 /// message.
129 /// \param[in] min_var_to_check_index Buffer containing the index of the absolute minimum variable-to-check message.
130 /// \param[in] sign_prod_var_to_check Buffer containing the sign of the product of all the messages reaching the
131 /// check node.
132 /// \param[in] shift Shift applied to the variable-to-check edges when lifting the base graph.
133 /// \param[in] var_node Counting index of the variable node, that is it can only take values between
134 /// zero and the degree of the check node minus one.
135 virtual void compute_check_to_var_msgs(span<log_likelihood_ratio> this_check_to_var,
136 span<const log_likelihood_ratio> this_var_to_check,
137 span<const log_likelihood_ratio> rotated_node,
138 span<const log_likelihood_ratio> min_var_to_check,
139 span<const log_likelihood_ratio> second_min_var_to_check,
140 span<const uint8_t> min_var_to_check_index,
141 span<const uint8_t> sign_prod_var_to_check,
142 unsigned shift,
143 unsigned var_node) = 0;
144
145 /// \brief Updates the soft bits corresponding to the variable nodes connected to the same base graph check node.
146 /// \param[in] check_node The check node (in the base graph) the variables nodes are connected to.
147 void update_soft_bits(unsigned check_node);
148
149 /// \brief Helper function for \ref update_soft_bits().
150 ///
151 /// Updates the soft bits corresponding to one variable node of the base graph (that is, the number of updated soft
152 /// bits is equal to the lifting size).
153 /// \param[out] this_soft_bits View to the updated soft bits.
154 /// \param[in] this_var_to_check View to the input variable-to-check messages.
155 /// \param[in] this_check_to_var View to the input check-to-variable messages.
156 virtual void compute_soft_bits(span<log_likelihood_ratio> this_soft_bits,
157 span<const log_likelihood_ratio> this_var_to_check,
158 span<const log_likelihood_ratio> this_check_to_var) = 0;
159
160 /// Converts soft bits into hard bits and returns the decoded message.
161 virtual void get_hard_bits(bit_buffer& out) = 0;
162
163 /// \brief Helper function for \ref update_variable_to_check_messages().
164 ///
165 /// Computes the updated value of the messages between a lifted variable node and a lifted check node (therefore, the
166 /// number of computed messages is equal to the lifting size).
167 /// \param[out] this_var_to_check Buffer to store the updated variable-to-check messages.
168 /// \param[in] this_soft_bits Current soft bits.
169 /// \param[in] this_check_to_var Current check-to-variable messages.
170 /// \remard An exception is raised if the size of the three spans is not the same.
171 virtual void compute_var_to_check_msgs(span<log_likelihood_ratio> this_var_to_check,
172 span<const log_likelihood_ratio> this_soft_bits,
173 span<const log_likelihood_ratio> this_check_to_var) = 0;
174
175protected:
176 /// Number of base graph variable nodes corresponding to information bits.
177 uint16_t bg_K = 22;
178 /// Lifting size as a natural number (as opposed to an element from srsran::ldpc::lifting_size_t).
179 uint16_t lifting_size = 2;
180 /// Number of bytes used to store a lifted node.
181 unsigned node_size_byte = 2;
182 /// Scaling factor of the normalized min-sum algorithm.
183 float scaling_factor = 0.8;
184
185 /// Buffer to store the current value of the soft bits.
186 std::array<log_likelihood_ratio, static_cast<size_t>(ldpc::MAX_BG_N_FULL* ldpc::MAX_LIFTING_SIZE)> soft_bits;
187
188private:
189 /// Pointer to the Tanner graph (~ parity check matrix) used by the encoding algorithm.
190 const ldpc_graph_impl* current_graph = nullptr;
191 /// Total number of base graph variable nodes in the current graph.
192 uint16_t bg_N_full = 68;
193 /// Number of base graph variable nodes after shortening.
194 uint16_t bg_N_short = 66;
195 /// Number of base graph variable nodes in the high-rate region.
196 uint16_t bg_N_high_rate = 26;
197 /// Number of base graph check nodes.
198 uint16_t bg_M = 46;
199 /// \brief Number of used variable nodes.
200 ///
201 /// Instead of using all the variable nodes and setting to zero all the punctured LLRs, the decoder will work only
202 /// with the enough variable nodes to accommodate all LLRs provided by the input vector. It is a multiple of the
203 /// lifting size.
204 uint16_t codeblock_length = 52;
205 /// Number of significant (not filler) bits.
206 uint16_t nof_significant_bits = 44;
207
208 /// Maximum number of iterations
209 unsigned max_iterations = 6;
210
211 /// \brief Buffer to store the current value of the variable-to-check messages.
212 ///
213 /// Implementing a layered-based algorithm, we only need to store the variable-to-check messages corresponding
214 /// to the current (base graph) check node.
215 /// \remark The factor 2 stems from the fact that we store two copies of all messages to facilitate their circular
216 /// shift.
217 std::array<log_likelihood_ratio, static_cast<size_t>(2 * MAX_CHECK_NODE_DEGREE * ldpc::MAX_LIFTING_SIZE)>
218 var_to_check = {};
219
220 /// Buffer to store the current value of the check-to-variable messages.
221 std::array<std::array<log_likelihood_ratio, static_cast<size_t>(MAX_CHECK_NODE_DEGREE* ldpc::MAX_LIFTING_SIZE)>,
222 ldpc::MAX_BG_M>
223 check_to_var;
224
225 /// Initialization flags of check-to-variable messages: true if initialized.
226 std::array<bool, ldpc::MAX_BG_M> is_check_to_var_initialized;
227};
