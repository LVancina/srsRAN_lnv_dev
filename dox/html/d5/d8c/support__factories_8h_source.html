58/// \brief Creates a generic resource grid pool.
59///
60/// It selects a different resource grid every time a resource grid is requested. The resource grid is repeated every
61/// \c grids.size() requests.
62///
63/// \param[in] grids Resource grids, ownerships are transferred to the pool.
64/// \return A generic resource grid pool.
65std::unique_ptr<resource_grid_pool>
66create_generic_resource_grid_pool(std::vector<std::unique_ptr<resource_grid>> grids);
67
68/// \brief Creates an asynchronous resource grid pool.
69///
70/// It selects a different resource grid every time a resource grid is requested. The resource grid expires
71/// \c expire_timeout_slots after it is requested. When a resource grid expires, it is asynchronously set to zero.
72///
73/// The resource grid repetition is not deterministic but it is guaranteed that it is repeated after
74/// \c expire_timeout_slots.
75///
76/// \param[in] expire_timeout_slots Number of slots after which resource grids are expired.
77/// \param[in] async_executor Asynchronous task executor for setting the grid to zero.
78/// \param[in] grids Resource grids, ownerships are transferred to the pool.
79/// \return An asynchronous resource grid pool.
80std::unique_ptr<resource_grid_pool>
81create_asynchronous_resource_grid_pool(unsigned expire_timeout_slots,
82 task_executor& async_executor,
83 std::vector<std::unique_ptr<resource_grid>> grids);
84
85/// \brief Creates a long PRACH sequence buffer.
86///
87/// Long buffers contain 839-element PRACH sequences for up to 4 OFDM symbols and a given maximum number of
88/// frequency-domain occasions.
89///
90/// \param[in] max_nof_antennas Maximum number of antennas.
91/// \param[in] max_nof_fd_occasions Maximum number of frequency-domain occasions.
92/// \return A long preamble sequence buffer.
93std::unique_ptr<prach_buffer> create_prach_buffer_long(unsigned max_nof_antennas, unsigned max_nof_fd_occasions);
94
95/// \brief Creates a short PRACH sequence buffer.
96///
97/// Short buffers contain 139-element PRACH sequences for up to \ref prach_constants::SHORT_SEQUENCE_MAX_NOF_SYMBOLS
98/// symbols per occasion.
99///
100/// \param[in] max_nof_antennas Maximum number of antennas.
101/// \param[in] max_nof_td_occasions Maximum number of time-domain occasions.
102/// \param[in] max_nof_fd_occasions Maximum number of frequency-domain occasions.
103/// \return A short preamble sequence buffer containing PRACH sequence buffers for the number of selected occasions.
104std::unique_ptr<prach_buffer>
105create_prach_buffer_short(unsigned max_nof_antennas, unsigned max_nof_td_occasions, unsigned max_nof_fd_occasions);
106
107/// \brief Returns a pool for the given PRACH buffers.
108std::unique_ptr<prach_buffer_pool> create_prach_buffer_pool(std::vector<std::unique_ptr<prach_buffer>>&& elements);
