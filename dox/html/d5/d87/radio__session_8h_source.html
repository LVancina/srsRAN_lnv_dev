33/// Describes a radio baseband unit session interface.
34class radio_session : radio_base
35{
36public:
37 /// \brief Gets the management plane.
38 /// \return The reference to the management plane for this radio session.
39 virtual radio_management_plane& get_management_plane() = 0;
40
41 /// \brief Gets the data plane.
42 /// \param[in] stream_id Stream identifier.
43 /// \return The reference to the data plane for this radio session.
44 /// \remark An assertion is triggered if the stream identifier is invalid.
45 virtual baseband_gateway& get_baseband_gateway(unsigned stream_id) = 0;
46
47 /// \brief Gets the current time.
48 virtual baseband_gateway_timestamp read_current_time() = 0;
49
50 /// \brief Starts the radio operation.
51 ///
52 /// It requests all radio streams to start simultaneously.
53 ///
54 /// \param[in] init_time Initial time in which streaming shall start.
55 /// \remark Radio streams run until halted by a call of the stop() method.
56 /// \remark Calling start() more than once results in undefined behavior, even after calling stop().
57 virtual void start(baseband_gateway_timestamp init_time) = 0;
58
59 /// \brief Stops the radio session operation.
60 ///
61 /// Any call to radio operation after calling stop() will return instantly without interacting with the physical
62 /// radio.
63 ///
64 /// \remark Calling stop() without having called start() results in undefined behavior.
65 /// \remark Calling start() after stop() results in undefined behavior.
66 virtual void stop() = 0;
67};
