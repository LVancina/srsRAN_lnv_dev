33/// \brief E2AP bridge that uses the IO broker to handle the SCTP connection
34class e2ap_network_adapter : public e2_message_notifier,
35 public e2_message_handler,
36 public sctp_network_gateway_control_notifier,
37 public network_gateway_data_notifier
38{
39public:
40 e2ap_network_adapter(io_broker& broker_, dlt_pcap& pcap_) : broker(broker_), pcap(pcap_) {}
41
42 void connect_gateway(std::unique_ptr<sctp_network_gateway> gateway_)
43 {
44 gateway = std::move(gateway_);
45 gateway_ctrl_handler = gateway.get();
46 gateway_data_handler = gateway.get();
47
48 packer = create_e2ap_asn1_packer(*gateway_data_handler, *this, pcap);
49
50 if (!gateway_ctrl_handler->create_and_connect()) {
51 report_error("Failed to create SCTP gateway.\n");
52 }
53 bool success =
54 broker.register_fd(gateway_ctrl_handler->get_socket_fd(), [this](int fd) { gateway_ctrl_handler->receive(); });
55 if (!success) {
56 report_fatal_error("Failed to register E2 (SCTP) network gateway at IO broker. socket_fd={}",
57 gateway_ctrl_handler->get_socket_fd());
58 }
59 }
60
61 void bind_and_listen(std::unique_ptr<sctp_network_gateway> gateway_)
62 {
63 gateway = std::move(gateway_);
64 gateway_ctrl_handler = gateway.get();
65 gateway_data_handler = gateway.get();
66
67 packer = create_e2ap_asn1_packer(*gateway_data_handler, *this, pcap);
68
69 if (!gateway_ctrl_handler->create_and_bind()) {
70 report_error("Failed to create SCTP gateway.\n");
71 }
72
73 if (!gateway_ctrl_handler->listen()) {
74 report_error("Failed to listen SCTP gateway.\n");
75 }
76
77 bool success =
78 broker.register_fd(gateway_ctrl_handler->get_socket_fd(), [this](int fd) { gateway_ctrl_handler->receive(); });
79 if (!success) {
80 report_fatal_error("Failed to register E2 (SCTP) network gateway at IO broker. socket_fd={}",
81 gateway_ctrl_handler->get_socket_fd());
82 }
83 }
84
85 /// \brief Return the port on which the gateway is listening.
86 ///
87 /// In case the gateway was configured to listen on port 0, i.e. the operating system shall pick a random free port,
88 /// this function can be used to get the actual port number.
89 optional<uint16_t> get_listen_port()
90 {
91 if (gateway_ctrl_handler == nullptr) {
92 return {};
93 }
94 return gateway_ctrl_handler->get_listen_port();
95 }
96
97 void connect_e2ap(e2_message_handler* e2ap_msg_handler_, e2_event_handler* event_handler_)
98 {
99 e2ap_msg_handler = e2ap_msg_handler_;
100 event_handler = event_handler_;
101 }
102
103 void disconnect_gateway()
104 {
105 srsran_assert(gateway_ctrl_handler, "Gateway handler not set.");
106 bool success = broker.unregister_fd(gateway_ctrl_handler->get_socket_fd());
107 if (!success) {
108 report_fatal_error("Failed to unregister E2 (SCTP) network gateway at IO broker. socket_fd={}",
109 gateway_ctrl_handler->get_socket_fd());
110 }
111
112 gateway_ctrl_handler = nullptr;
113 gateway_data_handler = nullptr;
114
115 packer.reset();
116 }
117
118private:
119 // E2AP calls interface to send (unpacked) E2AP PDUs
120 void on_new_message(const e2_message& msg) override
121 {
122 srsran_assert(packer, "E2AP ASN1 packer disconnected");
123 packer->handle_message(msg);
124 }
125
126 // SCTP network gateway calls interface to inject received PDUs (ASN1 packed)
127 void on_new_pdu(byte_buffer pdu) override
128 {
129 srsran_assert(packer, "E2AP ASN1 packer disconnected");
130 packer->handle_packed_pdu(pdu);
131 }
132
133 // The packer calls this interface to inject unpacked E2AP PDUs
134 void handle_message(const e2_message& msg) override
135 {
136 srsran_assert(e2ap_msg_handler, "E2AP msg handler not set.");
137 e2ap_msg_handler->handle_message(msg);
138 }
139
140 void on_connection_loss() override
141 {
142 srsran_assert(event_handler, "E2AP event handler not set.");
143 event_handler->handle_connection_loss();
144 }
145
146 void on_connection_established() override
147 {
148 // TODO: extend event interface to inform about connection establishment
149 logger.debug("on_connection_established");
150 }
151
152 io_broker& broker;
153 dlt_pcap& pcap;
154 std::unique_ptr<e2ap_packer> packer;
155 std::unique_ptr<sctp_network_gateway> gateway;
156 sctp_network_gateway_controller* gateway_ctrl_handler;
157 sctp_network_gateway_data_handler* gateway_data_handler;
158 e2_message_handler* e2ap_msg_handler;
159 e2_event_handler* event_handler;
160
161 srslog::basic_logger& logger = srslog::fetch_basic_logger("SCTP-GW");
162};
163
164}; // namespace srsran
