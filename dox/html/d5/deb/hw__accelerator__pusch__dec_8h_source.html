38/// Hardware-accelerated PUSCH decoder configuration parameters.
39struct hw_pusch_decoder_configuration {
40 /// Base graph used for encoding/decoding the current transport block.
41 ldpc_base_graph_type base_graph_index;
42 /// Modulation scheme.
43 modulation_scheme modulation;
44 /// Number of segments in the transport block.
45 unsigned nof_segments;
46 /// Redundancy version, values in {0, 1, 2, 3}.
47 unsigned rv;
48 /// Total codeword length in bits (E).
49 unsigned cw_length;
50 /// LDPC lifting size (z_c), as per TS38.212, Section 5.2.2.
51 unsigned lifting_size;
52 /// Length of the circular buffer in bits.
53 unsigned Ncb;
54 /// Limited buffer rate matching length in bits.
55 unsigned Nref;
56 /// Segment size expressed as an integer number of bits.
57 unsigned nof_segment_bits;
58 /// Number of Filler bits (n_filler = K – K’, as per TS38.212 Section 5.2.2.
59 unsigned nof_filler_bits;
60 /// Maximum number of iterations of the LDPC decoder.
61 unsigned max_nof_ldpc_iterations;
62 /// Flag for LDPC decoder early stopping: \c true to activate.
63 bool use_early_stop;
64 /// Flag to denote new data (first HARQ transmission).
65 bool new_data;
66 /// CB CRC length in bits.
67 unsigned cb_crc_len;
68 /// CB CRC type.
69 hw_dec_cb_crc_type cb_crc_type;
70 /// Unique absolute CB ID for the HARQ buffer utilized by the decoding task.
71 unsigned absolute_cb_id;
72};

74/// Hardware-accelerated PUSCH decoder output parameters.
75struct hw_pusch_decoder_outputs {
76 /// CRC value of the current decoding LDPC decoder operation.
77 bool CRC_pass;
78 /// Number of used iterations in the current decoding LDPC decoder operation.
79 unsigned nof_ldpc_iterations;
80};

82/// Generic hardware accelerated PUSCH decoder functions.
83class hw_accelerator_pusch_dec : public hw_accelerator<int8_t, uint8_t>
84{
85public:
86 /// Default destructor.
87 virtual ~hw_accelerator_pusch_dec() = default;
88
89 /// Configures the decoding operation given the common HW-oriented PUSCH decoder configuration.
90 /// \param[in] config Structure providing the configuration parameters of the PUSCH decoder.
91 /// \param[in] cb_index Optional. Index of the CB for which the PUSCH decoding operation is being configured.
92 virtual void configure_operation(const hw_pusch_decoder_configuration& config, unsigned cb_index = 0) = 0;
93
94 /// Reads the decoding operation status outputs.
95 /// \param[out] out Output results associated to the PUSCH decoding operation.
96 /// \param[in] cb_index Optional. Index of the decoded CB.
97 /// \param[in] absolute_cb_id Optional. Unique ID of associated to the CB in the soft-buffer and HARQ ACK context
98 /// entry.
99 virtual void
100 read_operation_outputs(hw_pusch_decoder_outputs& out, unsigned cb_index = 0, unsigned absolute_cb_id = 0) = 0;
101
102 /// Free the external HARQ buffer context repository entry for the associated operation.
103 /// \param[in] absolute_cb_id Unique ID of associated to the CB in the soft-buffer and HARQ ACK context entry.
104 virtual void free_harq_context_entry(unsigned absolute_cb_id) = 0;
105
106 /// Checks if the hardware-accelerated PUSCH decoder uses an external HARQ buffer.
107 /// \return True if an external HARQ buffer is used, false otherwise.
108 virtual bool is_external_harq_supported() const = 0;
109};
