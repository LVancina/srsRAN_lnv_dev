100class base_task_strand
101{
102public:
103 virtual ~base_task_strand() = default;
104
106 bool handle_enqueued_task(bool is_execute)
107 {
108 // If the task_strand is acquired, it means that no other thread is running the pending tasks and we need to
109 // dispatch a job to run them to the wrapped executor.
110 bool strand_acquired = job_count.fetch_add(1U, std::memory_order_acq_rel) == 0U;
111 if (not strand_acquired) {
112 // If we failed to acquire the task_strand, it means that another call has already acquired it and scheduled
113 // the running of all enqueued jobs, including the one we just enqueued in this call.
114 return true;
115 }
116 return this->dispatch_strand(is_execute);
117 }
118
119protected:
120 virtual bool dispatch_strand(bool is_execute) = 0;
121
122 // Number of jobs currently enqueued in the strand.
123 std::atomic<uint32_t> job_count{0};
124};

128template <concurrent_queue_policy QueuePolicy, typename StrandPtr>
129class task_strand_executor_impl final : public task_executor
130{
131public:
132 template <typename U>
133 task_strand_executor_impl(non_blocking_enqueuer<unique_task, QueuePolicy> enqueuer_, U&& strand_) :
134 enqueuer(enqueuer_), strand(std::forward<U>(strand_))
135 {
136 }
137
138 SRSRAN_NODISCARD bool execute(unique_task task) override
139 {
140 // Enqueue task in task_strand queue.
141 if (not enqueuer.try_push(std::move(task))) {
142 // strand queue is full.
143 return false;
144 }
145 return strand->handle_enqueued_task(true);
146 }
147
148 SRSRAN_NODISCARD bool defer(unique_task task) override
149 {
150 // Enqueue task in task_strand queue.
151 if (not enqueuer.try_push(std::move(task))) {
152 // strand queue is full.
153 return false;
154 }
155 return strand->handle_enqueued_task(false);
156 }
157
158private:
159 non_blocking_enqueuer<unique_task, QueuePolicy> enqueuer;
160 StrandPtr strand;
161};

164class task_strand_with_queue : public base_task_strand

223 template <size_t... Is>
224 std::unique_ptr<task_executor> get_executor_ptr_impl(enqueue_priority priority, std::index_sequence<Is...> /*unused*/)
225 {
226 const size_t idx = detail::enqueue_priority_to_queue_index(priority, sizeof...(QueuePolicies));
227 std::unique_ptr<task_executor> exec;
228 (void)std::initializer_list<int>{[this, idx, &exec]() {
229 if (idx == Is) {
230 exec = std::make_unique<executor<detail::queue_index_to_enqueue_priority(Is, sizeof...(QueuePolicies))>>(
231 this->queue.template get_enqueuer(), *this);
232 }
233 return 0;
234 }()...};
235 return exec;
236 }
237
238 // Run all tasks that have been enqueued in the strand, and unlocks the strand.
239 void run_enqueued_tasks()
240 {
241 unique_task task;
242 uint32_t queue_size = this->job_count.load(std::memory_order_acquire);
243 while (queue_size > 0) {
244 unsigned run_count = 0;
245 for (; run_count != queue_size and queue.queue.try_pop(task); ++run_count) {
246 task();
247 }
248 if (run_count != queue_size) {
249 // Unexpected failure to pop enqueued tasks. It might be due to queue shutdown.
250 srslog::fetch_basic_logger("ALL").info("Couldn't run all pending tasks in strand");
251 }
252
253 // We have run all the tasks that were enqueued since when we computed queue_size.
254 // Recompute the queue_size to check if there are tasks that were enqueued in the meantime.
255 queue_size = this->job_count.fetch_sub(run_count, std::memory_order_acq_rel) - run_count;
256 }
257 }
258
259 void handle_failed_task_dispatch()
260 {
261 // Pop enqueued tasks until number of enqueued jobs is zero.
262 // Note: Since we acquired the task_strand, the task enqueued in this call should always be the first being
263 // popped. Note: If there is a single producer, only the task enqueued in this call will be popped. Note: As we
264 // currently hold the strand, there is no concurrent thread popping tasks.
265 uint32_t queue_size = this->job_count.load(std::memory_order_acquire);
266 unique_task dropped_task;
267 while (queue_size > 0) {
268 unsigned run_count = 0;
269 for (; run_count != queue_size and this->queue.queue.try_pop(dropped_task); ++run_count) {
270 // do nothing with popped task.
271 }
272 queue_size = this->job_count.fetch_sub(run_count, std::memory_order_acq_rel) - run_count;
273 }
274 }
275
276 unique_task get_run_strand_task()
277 {
278 return [this]() { run_enqueued_tasks(); };
279 }
280
281 // Queue of pending tasks waiting for their turn.
282 detail::strand_queue<QueuePolicies...> queue;
283};

291template <typename Executor, concurrent_queue_policy... QueuePolicies>
292class task_strand final : public detail::task_strand_with_queue<QueuePolicies...>
293{
294 using base_type = detail::task_strand_with_queue<QueuePolicies...>;
295
296public:
297 template <enqueue_priority Priority>
298 using executor = typename base_type::template executor<Priority>;
299
300 template <typename Exec, typename... Args>
301 explicit task_strand(Exec&& executor_, Args&&... queue_params) :
302 base_type(std::forward<Args>(queue_params)...), out_exec(std::forward<Exec>(executor_))
303 {
304 }
305
306private:
307 bool dispatch_strand(bool is_execute) final
308 {
309 // Check if the adapted executor gives us permission to run pending tasks inline. An example of when this may
310 // happen is when the caller is running in the same execution context of the underlying task worker or task worker
311 // pool. However, this permission is still not a sufficient condition to simply call the task inline. For
312 // instance, if the execution context is a thread pool, we have to ensure that the task_strand is not already
313 // acquired by another thread of the same pool. If we do not do this, running the task inline would conflict with
314 // the task_strand strict serialization requirements. For this reason, we will always enqueue the task and try to
315 // acquire the task_strand.
316 bool dispatch_successful;
317 if (is_execute) {
318 dispatch_successful = detail::get_task_executor_ref(out_exec).execute(this->get_run_strand_task());
319 } else {
320 dispatch_successful = detail::get_task_executor_ref(out_exec).defer(this->get_run_strand_task());
321 }
322 if (not dispatch_successful) {
323 // Unable to dispatch executor job to run enqueued tasks.
324 this->handle_failed_task_dispatch();
325 return false;
326 }
327 return true;
328 }
329
330 // Executor to which tasks are dispatched in serialized manner.
331 Executor out_exec;
332};

400constexpr size_t MAX_QUEUES_PER_STRAND = 3;
401
402// Special case to stop recursion for task queue policies.
403template <typename OutExec,
404 concurrent_queue_policy... QueuePolicies,
405 std::enable_if_t<(sizeof...(QueuePolicies) > MAX_QUEUES_PER_STRAND), int> = 0>
406std::vector<std::unique_ptr<task_executor>>
407make_strand_executors_iter_helper(OutExec&& out_exec, span<const concurrent_queue_params> strand_queues)
408{
409 report_fatal_error("Strands with equal or more than {} queues are not supported", MAX_QUEUES_PER_STRAND);
410 return {};
411};
412
413template <typename OutExec,
414 concurrent_queue_policy... QueuePolicies,
415 std::enable_if_t<sizeof...(QueuePolicies) <= MAX_QUEUES_PER_STRAND, int> = 0>
416std::vector<std::unique_ptr<task_executor>>
417make_strand_executors_iter_helper(OutExec&& out_exec, span<const concurrent_queue_params> strand_queues)
418{
419 constexpr static size_t Is = sizeof...(QueuePolicies);
420 if (strand_queues.empty() or Is > strand_queues.size()) {
421 return {};
422 }
423 if (Is == strand_queues.size()) {
424 std::array<unsigned, Is> strand_queue_sizes;
425 for (unsigned i = 0; i != strand_queues.size(); ++i) {
426 strand_queue_sizes[i] = strand_queues[i].size;
427 }
428 return make_strand_executor_ptrs<QueuePolicies...>(std::forward<OutExec>(out_exec),
429 span<const unsigned>(strand_queue_sizes));
430 }
431
432 switch (strand_queues[Is].policy) {
433 case concurrent_queue_policy::lockfree_mpmc: {
434 return make_strand_executors_iter_helper<OutExec, QueuePolicies..., concurrent_queue_policy::lockfree_mpmc>(
435 std::forward<OutExec>(out_exec), strand_queues);
436 } break;
437 case concurrent_queue_policy::lockfree_spsc: {
438 return make_strand_executors_iter_helper<OutExec, QueuePolicies..., concurrent_queue_policy::lockfree_spsc>(
439 std::forward<OutExec>(out_exec), strand_queues);
440 } break;
441 default:
442 report_fatal_error("Unsupported queue policy.");
443 }
444 return {};
445}
