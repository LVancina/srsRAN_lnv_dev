33class ul_logical_channel_manager
34{
35public:
36 ul_logical_channel_manager();
37
39 void set_status(lcg_id_t lcg_id, bool active) { groups[lcg_id].active = active; }
40
42 void configure(span<const logical_channel_config> log_channels_configs);
43
45 void deactivate();
46
48 bool is_active(lcg_id_t lcg_id) const { return groups[lcg_id].active; }
49
51 bool is_active() const
52 {
53 return std::any_of(groups.begin(), groups.end(), [](const auto& g) { return g.active; });
54 }
55
57 bool has_pending_bytes() const
58 {
59 return std::any_of(groups.begin(), groups.end(), [](const auto& ch) { return ch.active and ch.buf_st > 0; });
60 }
61
63 unsigned pending_bytes() const
64 {
65 unsigned bytes = 0;
66 for (unsigned i = 0; i <= MAX_LCG_ID; ++i) {
67 bytes += pending_bytes(uint_to_lcg_id(i));
68 }
69 return bytes;
70 }
71
74 unsigned pending_bytes(lcg_id_t lcg_id) const
75 {
76 // Note: TS38.321, 6.1.3.1 - The size of the RLC and MAC headers are not considered in the buffer size computation.
77 return is_active(lcg_id) ? get_mac_sdu_required_bytes(add_upper_layer_header_bytes(lcg_id, groups[lcg_id].buf_st))
78 : 0;
79 }
80
82 bool has_pending_sr() const { return sr_pending.load(std::memory_order_relaxed); }
83
85 void handle_bsr_indication(const ul_bsr_indication_message& msg)
86 {
87 for (const auto& lcg_report : msg.reported_lcgs) {
88 groups[lcg_report.lcg_id].buf_st = lcg_report.nof_bytes;
89 }
90 }
91
93 void handle_sr_indication()
94 {
95 if (not is_active()) {
96 // Ignore SR indication if the UL has been deactivated.
97 return;
98 }
99 sr_pending.store(true, std::memory_order::memory_order_relaxed);
100 // TODO: handle SR indication content.
101 }
102
103 void reset_sr_indication() { sr_pending.store(false, std::memory_order::memory_order_relaxed); }
104
105private:
106 struct channel_group_context {
107 bool active = false;
109 unsigned buf_st = 0;
110 };
111
113 static unsigned add_upper_layer_header_bytes(lcg_id_t lcgid, unsigned payload_bytes)
114 {
115 // Estimate of the number of bytes required for the upper layer header.
116 constexpr static unsigned RLC_HEADER_SIZE_ESTIMATE = 3U;
117
118 if (payload_bytes == 0 or lcgid == 0) {
119 // In case of no payload or LCG-ID == 0, there is no need to account for upper layer header.
120 return payload_bytes;
121 }
122 return payload_bytes + RLC_HEADER_SIZE_ESTIMATE;
123 }
124
125 // This state variable tells whether there is a pending SR. Note: It is an atomic variable because SR indications
126 // can be received from different cells (in different threads).
127 std::atomic<bool> sr_pending{false};
128
129 std::array<channel_group_context, MAX_NOF_LCGS> groups;
130};
131
132} // namespace srsran
