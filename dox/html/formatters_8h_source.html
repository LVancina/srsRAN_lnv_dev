116template <>
117struct formatter<srsran::pusch_processor::pdu_t> {
119 srsran::delimited_formatter helper;
120
122 formatter() = default;
123
124 template <typename ParseContext>
125 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
126 {
127 return helper.parse(ctx);
128 }
129
130 template <typename FormatContext>
131 auto format(const srsran::pusch_processor::pdu_t& pdu, FormatContext& ctx)
132 -> decltype(std::declval<FormatContext>().out())
133 {
134 if (pdu.context.has_value()) {
135 helper.format_always(ctx, pdu.context.value());
136 } else {
137 helper.format_always(ctx, "rnti=0x{:04x}", pdu.rnti);
138 }
139 helper.format_if_verbose(ctx, "bwp=[{}, {})", pdu.bwp_start_rb, pdu.bwp_start_rb + pdu.bwp_size_rb);
140 helper.format_always(ctx, "prb={}", pdu.freq_alloc);
141 helper.format_always(ctx, "symb=[{}, {})", pdu.start_symbol_index, pdu.start_symbol_index + pdu.nof_symbols);
142
143 // UCI description.
144 helper.format_always(ctx, pdu.uci);
145
146 // Verbose parameters.
147 helper.format_always(ctx, "mod={}", to_string(pdu.mcs_descr.modulation));
148 helper.format_if_verbose(ctx, "tcr={}", pdu.mcs_descr.get_normalised_target_code_rate());
149
150 // PUSCH data codeword if available.
151 if (pdu.codeword.has_value()) {
152 helper.format_always(ctx, pdu.codeword.value());
153 }
154
155 helper.format_if_verbose(ctx, "n_id={}", pdu.n_id);
156 helper.format_if_verbose(ctx, "dmrs_mask={}", pdu.dmrs_symbol_mask);
157 helper.format_if_verbose(ctx, "n_scr_id={}", pdu.scrambling_id);
158 helper.format_if_verbose(ctx, "n_scid={}", pdu.n_scid);
159 helper.format_if_verbose(ctx, "n_cdm_g_wd={}", pdu.nof_cdm_groups_without_data);
160 helper.format_if_verbose(ctx, "dmrs_type={}", (pdu.dmrs == srsran::dmrs_type::TYPE1) ? 1 : 2);
161 helper.format_if_verbose(ctx, "lbrm={}bytes", pdu.tbs_lbrm_bytes);
162 helper.format_if_verbose(ctx, "slot={}", pdu.slot);
163 helper.format_if_verbose(ctx, "cp={}", pdu.cp.to_string());
164 helper.format_if_verbose(ctx, "nof_layers={}", pdu.nof_tx_layers);
165 helper.format_if_verbose(ctx, "ports={}", srsran::span<const uint8_t>(pdu.rx_ports));
166 helper.format_if_verbose(ctx, "dc_position={}", pdu.dc_position);
167
168 return ctx.out();
169 }
170};

203template <>
204struct formatter<srsran::channel_state_information> {
206 srsran::delimited_formatter helper;
207
209 formatter() = default;
210
211 template <typename ParseContext>
212 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
213 {
214 return helper.parse(ctx);
215 }
216
217 template <typename FormatContext>
218 auto format(const srsran::channel_state_information& csi, FormatContext& ctx)
219 -> decltype(std::declval<FormatContext>().out())
220 {
221 if (helper.is_verbose()) {
222 // Verbose representation prints all available SINR parameters.
223 if (csi.sinr_report_type == srsran::channel_state_information::sinr_type::channel_estimator) {
224 helper.format_if_verbose(ctx, "sinr_ch_est[sel]={:.1f}dB", csi.get_sinr_dB());
225 } else if (csi.sinr_ch_estimator_dB.has_value()) {
226 helper.format_if_verbose(ctx, "sinr_ch_est={:.1f}dB", csi.sinr_ch_estimator_dB.value());
227 }
228
229 if (csi.sinr_report_type == srsran::channel_state_information::sinr_type::post_equalization) {
230 helper.format_if_verbose(ctx, "sinr_eq[sel]={:.1f}dB", csi.get_sinr_dB());
231 } else if (csi.sinr_post_eq_dB.has_value()) {
232 helper.format_if_verbose(ctx, "sinr_eq={:.1f}dB", csi.sinr_post_eq_dB.value());
233 }
234
235 if (csi.sinr_report_type == srsran::channel_state_information::sinr_type::evm) {
236 helper.format_if_verbose(ctx, "sinr_evm[sel]={:.1f}dB", csi.get_sinr_dB());
237 } else if (csi.sinr_evm_dB.has_value()) {
238 helper.format_if_verbose(ctx, "sinr_evm={:.1f}dB", csi.sinr_evm_dB.value());
239 }
240
241 if (csi.evm.has_value()) {
242 helper.format_if_verbose(ctx, "evm={:.2f}", csi.evm.value());
243 }
244
245 helper.format_if_verbose(ctx, "epre={:+.1f}dB", csi.get_epre_dB());
246 helper.format_if_verbose(ctx, "rsrp={:+.1f}dB", csi.get_rsrp_dB());
247 helper.format_if_verbose(ctx, "t_align={:.1f}us", csi.get_time_alignment().to_seconds() * 1e6);
248 } else {
249 // Short representation only prints the SINR selected for CSI reporting to higher layers.
250 helper.format_always(ctx, "sinr={:.1f}dB", csi.get_sinr_dB());
251 }
252 return ctx.out();
253 }

281template <>
282struct formatter<srsran::pusch_processor_result_control> {
284 srsran::delimited_formatter helper;
285
287 formatter() = default;
288
289 template <typename ParseContext>
290 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
291 {
292 return helper.parse(ctx);
293 }
294
295 template <typename FormatContext>
296 auto format(const srsran::pusch_processor_result_control& result, FormatContext& ctx)
297 -> decltype(std::declval<FormatContext>().out())
298 {
299 if ((!result.harq_ack.payload.empty())) {
300 if (result.harq_ack.status == srsran::uci_status::valid) {
301 helper.format_always(ctx, "ack={:b}", result.harq_ack.payload);
302 } else {
303 helper.format_always(ctx, "ack={:#}", srsran::detail::uci_bad_payload(result.harq_ack.payload.size()).get());
304 }
305 }
306 if ((!result.csi_part1.payload.empty())) {
307 if (result.csi_part1.status == srsran::uci_status::valid) {
308 helper.format_always(ctx, "csi1={:b}", result.csi_part1.payload);
309 } else {
310 helper.format_always(ctx, "csi1={:#}", srsran::detail::uci_bad_payload(result.csi_part1.payload.size()).get());
311 }
312 }
313 if ((!result.csi_part2.payload.empty())) {
314 if (result.csi_part2.status == srsran::uci_status::valid) {
315 helper.format_always(ctx, "csi2={:b}", result.csi_part2.payload);
316 } else {
317 helper.format_always(ctx, "csi2={:#}", srsran::detail::uci_bad_payload(result.csi_part2.payload.size()).get());
318 }
319 }
320 return ctx.out();
321 }
