62 static unsigned determine_exponent(uint16_t x, unsigned data_width)
63 {
64 srsran_assert(data_width != 0, "Invalid data width");
65 srsran_assert(data_width <= MAX_IQ_WIDTH, "Passed IQ data width exceeds 16 bits");
66
67 unsigned max_shift = MAX_IQ_WIDTH - data_width;
68 unsigned lz_without_sign = max_shift;
69
70 if ((x > 0) && (max_shift > 0)) {
71 // TODO: use a wrapper that checks whether this builtin is actually available and provides a fallback option.
72 lz_without_sign = __builtin_clz(x) - 16U - 1U;
73 }
74 int raw_exp = std::min(max_shift, lz_without_sign);
75 return std::max(0, static_cast<int>(MAX_IQ_WIDTH - data_width) - raw_exp);
76 }
