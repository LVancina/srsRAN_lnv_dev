33class upper_phy_rx_symbol_handler_printer_decorator : public upper_phy_rx_symbol_handler
34{
35public:
36 upper_phy_rx_symbol_handler_printer_decorator(std::unique_ptr<upper_phy_rx_symbol_handler> handler_,
37 srslog::basic_logger& logger_,
38 const std::string& filename,
39 unsigned nof_rb,
40 interval<unsigned> ul_print_ports,
41 bool print_prach_) :
42 handler(std::move(handler_)),
43 logger(logger_),
44 worker("rx_symb_print", 40),
45 temp_buffer(nof_rb * NRE),
46 nof_symbols(MAX_NSYMB_PER_SLOT),
47 start_port(ul_print_ports.start()),
48 end_port(ul_print_ports.stop()),
49 print_prach(print_prach_)
50 {
51 if (!logger.info.enabled()) {
52 logger.error("Receive symbol enabled but logger level not enabled. No symbols will be printed.");
53 return;
54 }
55
56 file = std::ofstream(filename, std::ios_base::binary | std::ios_base::trunc);
57 if (!file.is_open()) {
58 logger.error("RX_SYMBOL: failed to open file {}.", filename);
59 }
60
61 if (end_port <= start_port) {
62 logger.error("End port {} is not larger than start port {}.", end_port, start_port);
63 }
64 }

66 void handle_rx_symbol(const upper_phy_rx_symbol_context& context, const resource_grid_reader& grid) override
67 {
68 // Handle Rx symbol.
69 handler->handle_rx_symbol(context, grid);
70
71 // Early return if the number of symbols does not reach the configured one or the file is not open.
72 if ((context.symbol != (nof_symbols - 1)) || !file.is_open()) {
73 return;
74 }
75
76 // Queue write request.
77 if (not worker.push_task([this, context, &grid]() {
78 // Save the resource grid.
79 for (unsigned i_port = start_port; i_port != end_port; ++i_port) {
80 for (unsigned symbol_idx = 0; symbol_idx != nof_symbols; ++symbol_idx) {
81 grid.get(temp_buffer, i_port, symbol_idx, 0);
82 file.write(reinterpret_cast<const char*>(temp_buffer.data()), temp_buffer.size() * sizeof(cf_t));
83 }
84 }
85
86 // Log the resource grid information.
87 unsigned nof_complex_floats = temp_buffer.size() * nof_symbols * (end_port - start_port);
88 logger.info(context.slot.sfn(),
89 context.slot.slot_index(),
90 "RX_SYMBOL: sector={} offset={} size={}",
91 context.sector,
92 file_offset,
93 nof_complex_floats);
94
95 // Advance file offset.
96 file_offset += nof_complex_floats;
97 })) {
98 logger.warning(context.slot.sfn(),
99 context.slot.slot_index(),
100 "RX_SYMBOL: Failed to write UL samples. Cause: task worker queue is full");
101 }
102 }
103
104 void handle_rx_prach_window(const prach_buffer_context& context, const prach_buffer& buffer) override
105 {
106 handler->handle_rx_prach_window(context, buffer);
107
108 // Queue write request.
109 if (print_prach && !worker.push_task([this, context, &buffer]() {
110 unsigned nof_replicas = buffer.get_max_nof_symbols();
111 unsigned prach_start = 0;
112 unsigned prach_stop = buffer.get_max_nof_ports();
113 for (unsigned i_port = prach_start; i_port != prach_stop; ++i_port) {
114 for (unsigned i_replica = 0; i_replica != nof_replicas; ++i_replica) {
115 span<const cf_t> samples = buffer.get_symbol(i_port, 0, 0, i_replica);
116 file.write(reinterpret_cast<const char*>(samples.data()), samples.size() * sizeof(cf_t));
117 }
118 }
119
120 unsigned nof_complex_floats = buffer.get_sequence_length() * nof_replicas * (prach_stop - prach_start);
121 // Log the resource grid information.
122 logger.info(context.slot.sfn(),
123 context.slot.slot_index(),
124 "RX_PRACH: sector={} offset={} size={}",
125 context.sector,
126 file_offset,
127 nof_complex_floats);
128
129 // Advance file offset.
130 file_offset += nof_complex_floats;
131 })) {
132 logger.warning(context.slot.sfn(),
133 context.slot.slot_index(),
134 "RX_PRACH: Failed to write PRACH samples. Cause: task worker queue is full");
135 }
136 }
137
138 void handle_rx_srs_symbol(const upper_phy_rx_symbol_context& context) override
139 {
140 handler->handle_rx_srs_symbol(context);
141 }
142
143private:
144 size_t file_offset = 0;
145 std::unique_ptr<upper_phy_rx_symbol_handler> handler;
146 srslog::basic_logger& logger;
147 std::ofstream file;
148 task_worker worker;
149 srsvec::aligned_vec<cf_t> temp_buffer;
150 unsigned nof_symbols;
151 unsigned start_port;
152 unsigned end_port;
153 bool print_prach;
154};
