128inline constexpr units::bits compute_tb_crc_size(units::bits tbs)
129{
130 using namespace units::literals;
131 constexpr units::bits MAX_BITS_CRC16 = 3824_bits;
132 return ((tbs <= MAX_BITS_CRC16) ? 16_bits : 24_bits);
133}

140inline constexpr unsigned compute_nof_codeblocks(units::bits tbs, ldpc_base_graph_type bg)
141{
142 using namespace units::literals;
143 constexpr units::bits CBLOCK_CRC_LENGTH = 24_bits;
144 units::bits tb_and_crc_bits = tbs + compute_tb_crc_size(tbs);
145
146 units::bits max_segment_length = (bg == ldpc_base_graph_type::BG1) ? 8448_bits : 3840_bits;
147
148 return ((tb_and_crc_bits <= max_segment_length)
149 ? 1
150 : divide_ceil(tb_and_crc_bits.value(), (max_segment_length - CBLOCK_CRC_LENGTH).value()));
151}

157inline unsigned compute_lifting_size(units::bits tbs, ldpc_base_graph_type base_graph, unsigned nof_segments)
158{
159 using namespace units::literals;
160 units::bits nof_tb_bits_in = tbs + compute_tb_crc_size(tbs);
161
162 unsigned ref_length = 22;
163 if (base_graph == ldpc_base_graph_type::BG2) {
164 if (nof_tb_bits_in > 640_bits) {
165 ref_length = 10;
166 } else if (nof_tb_bits_in > 560_bits) {
167 ref_length = 9;
168 } else if (nof_tb_bits_in > 192_bits) {
169 ref_length = 8;
170 } else {
171 ref_length = 6;
172 }
173 }
174
175 unsigned total_ref_length = nof_segments * ref_length;
176
177 // Calculate the number of bits per transport block including codeblock CRC.
178 units::bits nof_tb_bits_out = nof_tb_bits_in;
179 if (nof_segments > 1) {
180 nof_tb_bits_out += units::bits(24 * nof_segments);
181 }
182
183 unsigned lifting_size = 0;
184 for (auto ls : all_lifting_sizes) {
185 if (ls * total_ref_length >= nof_tb_bits_out.value()) {
186 lifting_size = ls;
187 break;
188 }
189 }
190 srsran_assert(lifting_size != 0, "Lifting size cannot be 0");
191
192 return lifting_size;
193}

200inline units::bits compute_codeblock_size(ldpc_base_graph_type base_graph, unsigned lifting_size)
201{
202 constexpr unsigned base_length_BG1 = 22;
203 constexpr unsigned base_length_BG2 = 10;
204 unsigned base_length = (base_graph == ldpc_base_graph_type::BG1) ? base_length_BG1 : base_length_BG2;
205
206 return units::bits(base_length * lifting_size);
207}

210inline units::bits compute_full_codeblock_size(ldpc_base_graph_type base_graph, units::bits codeblock_size)
211{
212 // BG1 has rate 1/3 and BG2 has rate 1/5.
213 constexpr unsigned INVERSE_BG1_RATE = 3;
214 constexpr unsigned INVERSE_BG2_RATE = 5;
215 unsigned inverse_rate = (base_graph == ldpc_base_graph_type::BG1) ? INVERSE_BG1_RATE : INVERSE_BG2_RATE;
216
217 return codeblock_size * inverse_rate;
218}
