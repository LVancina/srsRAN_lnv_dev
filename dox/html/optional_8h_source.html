47struct base_optional_storage {
48 using storageT = std::remove_const_t<T>;

50 // Storage for trivially destructible type.
51 template <typename U, bool = std::is_trivially_destructible<U>::value>
52 union storage_type {
53 constexpr storage_type() noexcept : dummy() {}
54 template <typename... Args>
55 explicit storage_type(in_place_t, Args&&... args) : val(std::forward<Args>(args)...)
56 {
57 }
58
59 char dummy;
60 U val;
61 };

63 // Storage for non-trivially destructible type.
64 template <typename U>
65 union storage_type<U, false> {
66 constexpr storage_type() noexcept : dummy() {}
67 template <typename... Args>
68 explicit storage_type(in_place_t, Args&&... args) : val(std::forward<Args>(args)...)
69 {
70 }
71
72 // User-defined destructor in case T is not trivially destructible.
73 ~storage_type() {}
74
75 char dummy;
76 U val;
77 };

79 constexpr base_optional_storage() noexcept : payload(), has_val(false) {}
80 template <typename... Args>
81 explicit base_optional_storage(in_place_t tag, Args&&... args) :
82 payload(tag, std::forward<Args>(args)...), has_val(true)
83 {
84 }
85 explicit base_optional_storage(const base_optional_storage<T>& other, non_trivial_oper_tag ) noexcept :
86 has_val(other.has_val)
87 {
88 if (other.has_val) {
89 construct_(other.payload.val);
90 }
91 }
92 explicit base_optional_storage(base_optional_storage<T>&& other, non_trivial_oper_tag ) noexcept :
93 has_val(other.has_val)

97 }
98 }

100 base_optional_storage(const base_optional_storage<T>&) = default;
101 base_optional_storage(base_optional_storage<T>&&) noexcept = default;
102 base_optional_storage& operator=(const base_optional_storage<T>&) = default;
103 base_optional_storage& operator=(base_optional_storage<T>&&) noexcept = default;
104
105 template <typename... Args>
106 constexpr void construct_(Args&&... args) noexcept(std::is_nothrow_constructible<storageT, Args...>::value)
107 {
108 new (&payload.val) storageT(std::forward<Args>(args)...);
109 has_val = true;
110 }
111
112 constexpr void destroy_() noexcept
113 {
114 has_val = false;
115 payload.val.~storageT();
116 }
117
118 constexpr void reset_() noexcept
119 {
120 if (has_val) {
121 destroy_();
122 }
123 }
124
125 constexpr void copy_assign_(const base_optional_storage<T>& other)
126 {
127 if (this == &other) {
128 return;
129 }
130 if (other.has_val and has_val) {
131 this->payload.val = other.payload.val;
132 } else {
133 if (other.has_val) {
134 this->construct_(other.payload.val);
135 } else {
136 this->reset_();
137 }
138 }
139 }
140
141 constexpr void move_assign_(base_optional_storage<T>&& other) noexcept(
142 std::is_nothrow_move_constructible<T>::value and std::is_nothrow_move_assignable<T>::value)
143 {
144 if (other.has_val and has_val) {
145 this->payload.val = std::move(other.payload.val);
146 } else {
147 if (other.has_val) {
148 this->construct_(std::move(other.payload.val));
149 } else {
150 this->reset_();
151 }
152 }
153 }
154
155 storage_type<storageT> payload;
156 bool has_val = false;
157};

279template <typename T>
280class optional
281{
282 // SFINAE helpers.
283 template <typename U>
284 using is_self = std::is_same<optional<T>, std::remove_const_t<std::decay_t<U>>>;
285
286public:
287 using value_type = T;
288
289 constexpr optional() = default;
290 constexpr optional(nullopt_t ) : optional() {}
291 template <typename U = T, std::enable_if_t<not is_self<U>::value, int> = 0>
292 constexpr optional(U&& u) : storage(in_place_t{}, std::forward<U>(u))
293 {
294 }
295
298 constexpr bool has_value() const noexcept { return storage.has_val; }
299 constexpr explicit operator bool() const noexcept { return has_value(); }
300
301 constexpr T& value() & noexcept
302 {
303 srsran_assert(has_value(), "Invalid optional<T> access");
304 return storage.payload.val;
305 }
306 constexpr T&& value() && noexcept
307 {
308 srsran_assert(has_value(), "Invalid optional<T> access");
309 return std::move(storage.payload.val);
310 }
311 constexpr const T& value() const& noexcept
312 {
313 srsran_assert(has_value(), "Invalid optional<T> access");
314 return storage.payload.val;
315 }
316 constexpr const T&& value() const&& noexcept
317 {
318 srsran_assert(has_value(), "Invalid optional<T> access");
319 return std::move(storage.payload.val);
320 }
321 constexpr T* operator->() noexcept { return &value(); }
322 constexpr const T* operator->() const noexcept { return &value(); }
323 constexpr T& operator*() & noexcept { return value(); }
324 constexpr T&& operator*() && noexcept { return value(); }
325 constexpr const T& operator*() const& noexcept { return value(); }
326 constexpr const T&& operator*() const&& noexcept { return value(); }
327
328 optional& operator=(nullopt_t ) noexcept
329 {
330 reset();
331 return *this;
332 }
333
335 template <typename... Args>
336 constexpr T& emplace(Args&&... args) noexcept(std::is_nothrow_constructible<T, Args...>::value)
337 {
338 storage.reset_();
339 storage.construct_(std::forward<Args>(args)...);
340 return value();
341 }
342
344 constexpr void reset() noexcept { storage.reset_(); }
345
346private:
347 detail::optional_storage<T> storage;
348};

350template <typename T>
351bool operator==(const optional<T>& lhs, const optional<T>& rhs) noexcept
352{
353 return lhs.has_value() == rhs.has_value() and (not lhs.has_value() or lhs.value() == rhs.value());
354}

356template <typename T>
357bool operator==(const optional<T>& lhs, const T& rhs) noexcept
