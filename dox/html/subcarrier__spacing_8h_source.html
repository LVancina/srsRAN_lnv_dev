34enum class subcarrier_spacing { kHz15 = 0, kHz30, kHz60, kHz120, kHz240, invalid };

37constexpr inline bool is_scs_valid(subcarrier_spacing scs)
38{
39 return scs <= subcarrier_spacing::kHz240;
40}

43constexpr inline bool is_scs_valid(subcarrier_spacing scs, frequency_range fr)
44{
45 return is_scs_valid(scs) and (((fr == frequency_range::FR2) and scs >= subcarrier_spacing::kHz60) or
46 ((fr == frequency_range::FR1) and scs <= subcarrier_spacing::kHz60));
47}

50constexpr inline unsigned to_numerology_value(subcarrier_spacing scs)

56constexpr inline unsigned scs_to_khz(subcarrier_spacing scs)
57{
58 srsran_assert(is_scs_valid(scs), "Invalid SCS.");
59 return 15U << to_numerology_value(scs);
60}

63constexpr inline subcarrier_spacing to_subcarrier_spacing(unsigned numerology)

69inline subcarrier_spacing to_subcarrier_spacing(const std::string& str)
70{
71 unsigned in_scs = static_cast<unsigned>(std::strtof(str.data(), nullptr));
72 for (unsigned numerology = 0, numerology_end = to_numerology_value(subcarrier_spacing::invalid);
73 numerology != numerology_end;
74 ++numerology) {
75 subcarrier_spacing scs = to_subcarrier_spacing(numerology);
76 unsigned scs_kHz = scs_to_khz(scs);
77
78 if ((in_scs == scs_kHz) || (in_scs == scs_kHz * 1000)) {
79 return scs;
80 }
81 }
82 return subcarrier_spacing::invalid;
83}

86inline const char* to_string(subcarrier_spacing scs)
87{
88 switch (scs) {

95 case subcarrier_spacing::kHz120:
96 return "120kHz";
97 case subcarrier_spacing::kHz240:
98 return "240kHz";
99 case subcarrier_spacing::invalid:
100 default:
101 return "invalid";
102 }
103}
