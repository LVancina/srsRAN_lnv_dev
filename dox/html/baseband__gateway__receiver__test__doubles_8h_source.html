30class baseband_gateway_receiver_spy : public baseband_gateway_receiver
31{
32public:
33 struct entry_t {
34 baseband_gateway_receiver::metadata metadata;
35 baseband_gateway_buffer_read_only data;
36 };
37
38 baseband_gateway_receiver_spy() : sample_dist(-1.0F, +1.0F) {}
39
40 // See interface for documentation.
41 metadata receive(baseband_gateway_buffer_writer& data) override
42 {
43 // Fill data with random samples.
44 for (unsigned channel = 0; channel != data.get_nof_channels(); ++channel) {
45 span<cf_t> buffer = data[channel];
46 std::generate(buffer.begin(), buffer.end(), [this]() { return cf_t(sample_dist(rgen), sample_dist(rgen)); });
47 }
48
49 entries.emplace_back();
50 entry_t& entry = entries.back();
51 entry.metadata.ts = current_timestamp;
52 entry.data = data;
53
54 current_timestamp += data.get_nof_samples();
55
56 return entry.metadata;
57 }
58
59 void set_current_timestamp(baseband_gateway_timestamp timestamp) { current_timestamp = timestamp; }
60
62 const std::vector<entry_t>& get_entries() const { return entries; }
63
65 void clear() { entries.clear(); }
66
67private:
68 std::mt19937 rgen;
69 std::uniform_real_distribution<float> sample_dist;
70 baseband_gateway_timestamp current_timestamp = 0;
71 std::vector<entry_t> entries;
72};
