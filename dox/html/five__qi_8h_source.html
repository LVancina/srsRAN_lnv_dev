28namespace srsran {

30// See TS 38.463: Five QI valid values: (0..255)
31constexpr static uint16_t MAX_FIVEQI = 255;
32
34enum class five_qi_t : uint16_t { min = 0, max = MAX_FIVEQI, invalid = MAX_FIVEQI + 1 };
35
37constexpr inline uint16_t five_qi_to_uint(five_qi_t five_qi)
38{
39 return static_cast<uint16_t>(five_qi);
40}
41
43constexpr inline five_qi_t uint_to_five_qi(uint16_t five_qi)
44{
45 srsran_assert(
46 five_qi < MAX_FIVEQI, "Invalid 5QI {} - Must be between [{}..{}]", five_qi, five_qi_t::min, five_qi_t::max);
47 return static_cast<five_qi_t>(five_qi);
48}
49
50} // namespace srsran

52// Formatters
53namespace fmt {
54template <>
55struct formatter<srsran::five_qi_t> {
56 template <typename ParseContext>

62 template <typename FormatContext>
63 auto format(const srsran::five_qi_t& five_qi, FormatContext& ctx) -> decltype(std::declval<FormatContext>().out())
64 {
65 return format_to(ctx.out(), "5QI={:#x}", five_qi_to_uint(five_qi));
66 }
67};
