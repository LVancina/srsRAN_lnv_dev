113class lower_phy_rx_symbol_notifier_spy : public lower_phy_rx_symbol_notifier
114{
115private:
116 srslog::basic_logger& logger;
117
118 struct rx_symbol_event {
119 lower_phy_rx_symbol_context context;
120 const resource_grid_reader* grid;
121 };
122 std::vector<rx_symbol_event> rx_symbol_events;
123
124 struct rx_prach_event {
125 prach_buffer_context context;
126 const prach_buffer* buffer;
127 };
128 std::vector<rx_prach_event> rx_prach_events;
129
130public:
132 explicit lower_phy_rx_symbol_notifier_spy(const std::string& log_level = "warning") :
133 logger(srslog::fetch_basic_logger("Rx Notifier"))
134 {
135 logger.set_level(srslog::str_to_basic_level(log_level));
136 }
137
138 // See interface for documentation.
139 void on_rx_symbol(const lower_phy_rx_symbol_context& context, const resource_grid_reader& grid) override
140 {
141 logger.debug(context.slot.sfn(),
142 context.slot.slot_index(),
143 "Sector {} - On Rx Symbol {}.",
144 context.sector,
145 context.nof_symbols);
146 rx_symbol_events.emplace_back();
147 rx_symbol_event& event = rx_symbol_events.back();
148 event.context = context;
149 event.grid = &grid;
150 }
151
152 // See interface for documentation.
153 void on_rx_prach_window(const prach_buffer_context& context, const prach_buffer& buffer) override
154 {
155 logger.debug(context.slot.sfn(), context.slot.slot_index(), "Sector {} - On Rx PRACH Window.", context.sector);
156 rx_prach_events.emplace_back();
157 rx_prach_event& event = rx_prach_events.back();
158 event.context = context;
159 event.buffer = &buffer;
160 }
161
162 // See interface for documentation.
163 void on_rx_srs_symbol(const lower_phy_rx_symbol_context& context) override
164 {
165 logger.debug(context.slot.sfn(), context.slot.slot_index(), "Sector {} - On Rx SRS Symbol.", context.sector);
166 }
167
176 unsigned get_nof_events() const { return rx_symbol_events.size() + rx_prach_events.size(); }
177
183 const std::vector<rx_symbol_event>& get_rx_symbol_events() const { return rx_symbol_events; }
184
190 const std::vector<rx_prach_event>& get_rx_prach_events() const { return rx_prach_events; }
191
193 void clear_all_events()
194 {
195 rx_symbol_events.clear();
196 rx_prach_events.clear();
197 }
198};
