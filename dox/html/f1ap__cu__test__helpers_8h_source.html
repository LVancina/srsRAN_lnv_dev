57 void request_new_du_connection()
58 {
59 class sink_f1ap_message_notifier : public f1ap_message_notifier
60 {
61 public:
62 void on_new_message(const f1ap_message& msg) override {}
63 };
64
65 auto notifier = local_f1c_gw.handle_du_connection_request(std::make_unique<sink_f1ap_message_notifier>());
66 if (notifier != nullptr) {
67 du_tx_notifiers.push_back(std::move(notifier));
68 }
69 }
70
71 void remove_du_connection(size_t connection_idx) { du_tx_notifiers.erase(du_tx_notifiers.begin() + connection_idx); }
72
73 span<const f1ap_message> last_rx_pdus(size_t connection_idx) const
74 {
75 return local_f1c_gw.get_last_cu_cp_rx_pdus(connection_idx);
76 }
77 span<const f1ap_message> last_tx_pdus(size_t connection_idx) const
78 {
79 return local_f1c_gw.get_last_cu_cp_tx_pdus(connection_idx);
80 }
81
82 void push_cu_cp_rx_pdu(size_t du_connectin_idx, const f1ap_message& msg)
83 {
84 du_tx_notifiers[du_connectin_idx]->on_new_message(msg);
85 }
86
87 size_t nof_connections() const { return du_tx_notifiers.size(); }
88
89private:
90 srslog::basic_logger& logger;
91 f1c_test_local_gateway local_f1c_gw;
92
