79 unsigned pending_bytes() const
80 {
81 unsigned bytes = pending_ce_bytes();
82 // Skip index 0 ==> SRB0.
83 for (unsigned i = 1; i <= MAX_LCID; ++i) {
84 bytes += pending_bytes((lcid_t)i);
85 }
86 return bytes;
87 }
88
90 unsigned pending_ce_bytes() const
91 {
92 unsigned bytes = pending_ue_con_res_id_ce_bytes();
93 for (const auto& ce : pending_ces) {
94 bytes += ce.ce_lcid.is_var_len_ce() ? get_mac_sdu_required_bytes(ce.ce_lcid.sizeof_ce())
95 : FIXED_SIZED_MAC_CE_SUBHEADER_SIZE + ce.ce_lcid.sizeof_ce();
96 }
97 return bytes;
98 }
99
101 unsigned pending_ue_con_res_id_ce_bytes() const
102 {
103 static const auto ce_size = lcid_dl_sch_t{lcid_dl_sch_t::UE_CON_RES_ID}.sizeof_ce();
104 return is_con_res_id_pending() ? FIXED_SIZED_MAC_CE_SUBHEADER_SIZE + ce_size : 0;
105 }

108 unsigned pending_bytes(lcid_t lcid) const
109 {
110 return is_active(lcid) ? get_mac_sdu_required_bytes(channels[lcid].buf_st) : 0;
111 }

120 void handle_mac_ce_indication(const mac_ce_info& ce)
121 {

123 pending_con_res_id = true;

125 pending_ces.push_back(ce);
126 }
