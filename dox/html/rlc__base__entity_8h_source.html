39class rlc_base_entity : public rlc_entity
40{
41public:
42 rlc_base_entity(uint32_t du_index_,
43 du_ue_index_t ue_index_,
44 rb_id_t rb_id_,
45 timer_duration metrics_period_,
46 rlc_metrics_notifier* rlc_metrics_notifier_,
47 timer_factory timers) :
48 logger("RLC", {du_index_, ue_index_, rb_id_, "DL/UL"}),
49 ue_index(ue_index_),
50 rb_id(rb_id_),
51 metrics_period(metrics_period_),
52 metrics_timer(timers.create_timer())
53 {
54 rlc_metrics_notif = rlc_metrics_notifier_;
55 if (metrics_period.count() != 0) {
56 metrics_timer.set(metrics_period, [this](timer_id_t /*tid*/) { push_metrics(); });
57 metrics_timer.run();
58 }
59 }
60 ~rlc_base_entity() override = default;
61 rlc_base_entity(const rlc_base_entity&) = delete;
62 rlc_base_entity& operator=(const rlc_base_entity&) = delete;
63 rlc_base_entity(const rlc_base_entity&&) = delete;
64 rlc_base_entity& operator=(const rlc_base_entity&&) = delete;
65
66 rlc_tx_upper_layer_data_interface* get_tx_upper_layer_data_interface() final { return tx.get(); };
67 rlc_tx_lower_layer_interface* get_tx_lower_layer_interface() final { return tx.get(); };
68 rlc_rx_lower_layer_interface* get_rx_lower_layer_interface() final { return rx.get(); };
69
70 rlc_metrics get_metrics() final
71 {
72 rlc_metrics metrics = {};
73 metrics.ue_index = ue_index;
74 metrics.rb_id = rb_id;
75 metrics.tx = tx->get_metrics();
76 metrics.rx = rx->get_metrics();
77 return metrics;
78 }
79
80 void reset_metrics()
81 {
82 tx->reset_metrics();
83 rx->reset_metrics();
84 }
85
86protected:
87 rlc_bearer_logger logger;
88 du_ue_index_t ue_index;
89 rb_id_t rb_id;
90 std::unique_ptr<rlc_tx_entity> tx = {};
91 std::unique_ptr<rlc_rx_entity> rx = {};
92 timer_duration metrics_period;
93
94private:
95 unique_timer metrics_timer;
96 rlc_metrics_notifier* rlc_metrics_notif;
97
98 void push_metrics()
99 {
100 rlc_metrics m = get_metrics();
101 if (rlc_metrics_notif) {
102 rlc_metrics_notif->report_metrics(m);
103 }
104 if (m.tx.mode == rlc_mode::am) {
105 logger.log_info(
106 "TX metrics period={}ms num_sdus={} sdu_rate={}kbps, dropped_sdus={} discarded_sdus={} "
107 "num_pdus_no_segm={} pdu_rate_no_segm={}kbps num_pdus_with_segm={} pdu_rate_with_segm={}kbps num_retx={} "
108 "retx_rate={}kbps ctrl_pdus={} ctrl_rate={}kbps",
109 metrics_period.count(),
110 m.tx.num_sdus,
111 (double)m.tx.num_sdu_bytes * 8 / (double)metrics_period.count(),
112 m.tx.num_dropped_sdus,
113 m.tx.num_discarded_sdus,
114 m.tx.num_pdus_no_segmentation,
115 (double)m.tx.num_pdu_bytes_no_segmentation * 8 / (double)metrics_period.count(),
116 m.tx.mode_specific.am.num_pdus_with_segmentation,
117 (double)m.tx.mode_specific.am.num_pdu_bytes_with_segmentation * 8 / (double)metrics_period.count(),
118 m.tx.mode_specific.am.num_retx_pdus,
119 (double)m.tx.mode_specific.am.num_retx_pdu_bytes * 8 / (double)metrics_period.count(),
120 m.tx.mode_specific.am.num_ctrl_pdus,
121 (double)m.tx.mode_specific.am.num_ctrl_pdu_bytes * 8 / (double)metrics_period.count());
122 logger.log_info("RX metrics period={}ms num_sdus={} sdu_rate={}kbps num_pdus={} pdu_rate={}kbps "
123 "ctrl_pdus={}, ctrl_rate={}kbps",
124 metrics_period.count(),
125 m.rx.num_sdus,
126 (double)m.rx.num_sdu_bytes * 8 / (double)metrics_period.count(),
127 m.rx.num_pdus,
128 (double)m.rx.num_pdu_bytes * 8 / (double)metrics_period.count(),
129 m.tx.mode_specific.am.num_ctrl_pdus,
130 (double)m.rx.mode_specific.am.num_ctrl_pdu_bytes * 8 / (double)metrics_period.count());
131 }
132 reset_metrics();
133 metrics_timer.run();
134 }
135};
