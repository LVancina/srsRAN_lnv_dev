42template <unsigned NDIMS, typename Type, typename Index_type = unsigned>
43class tensor
44{
45public:
47 using dims = Index_type;
48
50 using dimensions_size_type = std::array<unsigned, NDIMS>;
51
53 virtual ~tensor() = default;
54
59 template <unsigned N = 1>
60 span<Type> get_view(const std::array<unsigned, NDIMS - N>& indices)
61 {
62 span<Type> data = get_data();
63 const dimensions_size_type& dimensions_size = get_dimensions_size();
64
65 // Calculate the view size for the first N dimensions.
66 unsigned view_size = std::accumulate(dimensions_size.begin(), dimensions_size.begin() + N, 1, std::multiplies<>());
67
68 return data.subspan(get_view_offset<N>(indices), view_size);
69 }
70
75 template <unsigned N = 1>
76 span<const Type> get_view(const std::array<unsigned, NDIMS - N>& indices) const
77 {
78 span<const Type> data = get_data();
79 const dimensions_size_type& dimensions_size = get_dimensions_size();
80
81 // Calculate the view size for the first N dimensions.
82 unsigned view_size = std::accumulate(dimensions_size.begin(), dimensions_size.begin() + N, 1, std::multiplies<>());
83
84 return data.subspan(get_view_offset<N>(indices), view_size);
85 }
86
88 Type& operator[](const dimensions_size_type& indices)
89 {
90 span<Type> data = get_data();
91 return data[get_view_offset<0>(indices)];
92 }
93
95 const Type& operator[](const dimensions_size_type& indices) const
96 {
97 span<const Type> data = get_data();
98 return data[get_view_offset<0>(indices)];
99 }
100
102 virtual const dimensions_size_type& get_dimensions_size() const = 0;
103
107 unsigned get_dimension_size(Index_type i_dimension) const
108 {
109 return get_dimensions_size()[static_cast<unsigned>(i_dimension)];
110 }
111
113 virtual void resize(const dimensions_size_type& dimensions) = 0;
114
115protected:
117 virtual span<Type> get_data() = 0;
119 virtual span<const Type> get_data() const = 0;
120
121private:
123 template <unsigned N>
124 unsigned get_view_offset(const std::array<unsigned, NDIMS - N>& indices) const
125 {
126 // Get dimensions.
127 const dimensions_size_type& dimensions_size = get_dimensions_size();
128 // Dimension initial stride.
129 unsigned stride = std::accumulate(dimensions_size.begin(), dimensions_size.begin() + N, 1, std::multiplies<>());
130 // Current offset of the dimension.
131 unsigned offset = 0;
132 // For each dimension, add its offset.
133 for (unsigned dimension_index = 0; dimension_index != NDIMS - N; ++dimension_index) {
134 srsran_assert(indices[dimension_index] < dimensions_size[dimension_index + N],
135 "Dimension {} index {} exceeds the maximum index {}.",
136 dimension_index + N,
137 indices[dimension_index],
138 dimensions_size[dimension_index + N]);
139 offset += stride * indices[dimension_index];
140 stride *= dimensions_size[dimension_index + N];
141 }
142 return offset;
143 }
144};

152template <unsigned NDIMS, typename Type, unsigned MAX_ELEMENTS, typename Index_type = unsigned>
153class static_tensor : public tensor<NDIMS, Type, Index_type>
154{
155public:
156 using dims = Index_type;
157 using dimensions_size_type = std::array<unsigned, NDIMS>;
158
160 static_tensor() : dimensions_size{} {}
161
163 explicit static_tensor(const dimensions_size_type& dimensions) { resize(dimensions); }
164
166 explicit static_tensor(const tensor<NDIMS, Type>& other) noexcept :
167 dimensions_size(other.get_dimensions_size()), elements(other.get_data())
168 {
169 // Do nothing.
170 }
171
173 static_tensor(static_tensor&& other) noexcept :
174 dimensions_size(std::move(other.dimensions_size)), elements(std::move(other.elements))
175 {
176 // Do nothing.
177 }
178
183 void resize(const dimensions_size_type& dimensions) override
184 {
185 dimensions_size = dimensions;
186 unsigned total_size = to_size(dimensions);
187 srsran_assert(total_size <= MAX_ELEMENTS,
188 "The total number of elements {} (dimensions: {}) exceeds the maximum number of elements {}.",
189 total_size,
190 span<const unsigned>(dimensions),
191 MAX_ELEMENTS);
192 }
193
194 // See interface for documentation.
195 const dimensions_size_type& get_dimensions_size() const override { return dimensions_size; }
196
197 // See interface for documentation.
198 span<Type> get_data() override
199 {
200 return span<Type>(reinterpret_cast<Type*>(elements.data()), MAX_ELEMENTS).first(to_size(dimensions_size));
201 }
202
203 // See interface for documentation.
204 span<const Type> get_data() const override
205 {
206 return span<const Type>(reinterpret_cast<const Type*>(elements.data()), MAX_ELEMENTS)
207 .first(to_size(dimensions_size));
208 }
209
210private:
212 static unsigned to_size(const dimensions_size_type& dimensions_size)
213 {
214 return std::accumulate(begin(dimensions_size), end(dimensions_size), 1, std::multiplies<>());
215 }
216
218 dimensions_size_type dimensions_size;
220 std::array<uint8_t, sizeof(Type) * MAX_ELEMENTS> elements;
221};

228template <unsigned NDIMS, typename Type, typename Index_type = unsigned>
229class dynamic_tensor : public tensor<NDIMS, Type, Index_type>
230{
231public:
232 using dims = Index_type;
233 using dimensions_size_type = std::array<unsigned, NDIMS>;
234
236 dynamic_tensor() = default;
237
239 explicit dynamic_tensor(const dimensions_size_type& dimensions) { reserve(dimensions); }
240
242 explicit dynamic_tensor(const tensor<NDIMS, Type>& other) noexcept :
243 dimensions_size(other.get_dimensions_size()), elements(other.get_data())
244 {
245 // Do nothing.
246 }
247
249 dynamic_tensor(dynamic_tensor&& other) noexcept :
250 dimensions_size(std::move(other.dimensions_size)), elements(std::move(other.elements))
251 {
252 other.dimensions_size = {};
253 }
254
256 void reserve(const dimensions_size_type& dimensions)
257 {
258 dimensions_size = dimensions;
259 unsigned total_size = std::accumulate(begin(dimensions_size), end(dimensions_size), 1, std::multiplies<>());
260 elements.resize(total_size);
261 }
262
268 void resize(const dimensions_size_type& dimensions) override
269 {
270 dimensions_size = dimensions;
271 unsigned total_size = size();
272 if (total_size > elements.size()) {
273 elements.resize(total_size);
274 }
275 }
276
277 // See interface for documentation.
278 const dimensions_size_type& get_dimensions_size() const override { return dimensions_size; }
279
280 // See interface for documentation.
281 span<Type> get_data() override { return span<Type>(elements).first(size()); }
282
283 // See interface for documentation.
284 span<const Type> get_data() const override { return span<const Type>(elements).first(size()); }
285
287 constexpr unsigned size() const
288 {
289 return std::accumulate(begin(dimensions_size), end(dimensions_size), 1, std::multiplies<>());
290 }
291
293 bool empty() const { return size() == 0; }
294
295private:
297 dimensions_size_type dimensions_size = {};
299 std::vector<Type> elements = {};
300};
