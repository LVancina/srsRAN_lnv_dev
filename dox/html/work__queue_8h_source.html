36template <typename T, size_t capacity = SRSLOG_QUEUE_CAPACITY>
37class work_queue
38{
39 srsran::ring_buffer<T> queue;
40 mutable mutex m;
41 static constexpr size_t threshold = capacity * 0.98;
42
43public:
44 work_queue() : queue(capacity) {}
45
46 work_queue(const work_queue&) = delete;
47 work_queue& operator=(const work_queue&) = delete;
48
51 bool push(const T& value)
52 {
53 m.lock();
54 // Discard the new element if we reach the maximum capacity.
55 if (queue.full()) {
56 m.unlock();
57 return false;
58 }
59 queue.push(value);
60 m.unlock();
61
62 return true;
63 }
64
67 bool push(T&& value)
68 {
69 m.lock();
70 // Discard the new element if we reach the maximum capacity.
71 if (queue.full()) {
72 m.unlock();
73 return false;
74 }
75 queue.push(std::move(value));
76 m.unlock();
77
78 return true;
79 }
80
83 std::pair<bool, T> try_pop()
84 {
85 m.lock();
86
87 if (queue.empty()) {
88 m.unlock();
89 return {false, T()};
90 }
91
92 // Here we have been woken up normally.
93 T Item = std::move(queue.top());
94 queue.pop();
95
96 m.unlock();
97
98 return {true, std::move(Item)};
99 }
100
102 size_t get_capacity() const { return capacity; }
103
105 bool is_almost_full() const
106 {
107 scoped_lock lock(m);
108
109 return queue.size() > threshold;
110 }
111};
