97static constexpr unsigned get_cdm_group_size(csi_rs_cdm_type cdm)
98{
99 // Determine the CDM group size.
100 unsigned cdm_group_size = 1;
101 if (cdm == csi_rs_cdm_type::fd_CDM2) {
102 cdm_group_size = 2;
103 } else if (cdm == csi_rs_cdm_type::cdm4_FD2_TD2) {
104 cdm_group_size = 4;
105 } else if (cdm == csi_rs_cdm_type::cdm8_FD2_TD4) {
106 cdm_group_size = 8;
107 }
108 return cdm_group_size;
109}
