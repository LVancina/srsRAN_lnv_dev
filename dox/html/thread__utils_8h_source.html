34class mutex
35{
36public:
37 mutex(const mutex&) = delete;
38 mutex& operator=(const mutex&) = delete;
39
40 mutex()
41 {
42 ::pthread_mutexattr_t mutex_attr;
43 ::pthread_mutexattr_init(&mutex_attr);
44 ::pthread_mutexattr_setprotocol(&mutex_attr, PTHREAD_PRIO_INHERIT);
45 ::pthread_mutex_init(&m, &mutex_attr);
46 }
47
48 ~mutex() { ::pthread_mutex_destroy(&m); }
49
51 void lock() { ::pthread_mutex_lock(&m); }
52
54 void unlock() { ::pthread_mutex_unlock(&m); }
55
57 bool try_lock() { return (::pthread_mutex_trylock(&m) == 0); }
58
60 pthread_mutex_t* raw() { return &m; }
61 const pthread_mutex_t* raw() const { return &m; }
62
63private:
64 pthread_mutex_t m;

133 void broadcast() { ::pthread_cond_broadcast(&cond_var); }

137 bool wait(timespec ts) { return (::pthread_cond_timedwait(&cond_var, m.raw(), &ts) == ETIMEDOUT); }
138
141 static timespec build_timeout(unsigned timeout_ms)
142 {
143 timespec ts;
144 clock_gettime(CLOCK_REALTIME, &ts);
145
146 ts.tv_sec += timeout_ms / 1000;
147 ts.tv_nsec += (timeout_ms % 1000) * 1000000;
148 if (ts.tv_nsec >= 1000000000) {
149 ++ts.tv_sec;
150 ts.tv_nsec -= 1000000000;
151 }
152
153 return ts;
154 }
155
156private:
157 mutable mutex m;
158 pthread_cond_t cond_var;
159};
