36class phase_compensation_lut
37{
38private:
40 static_vector<cf_t, MAX_NSYMB_PER_SLOT * get_nof_slots_per_subframe(subcarrier_spacing::kHz240)> coefficients;
41
42public:
49 phase_compensation_lut(subcarrier_spacing scs,
50 cyclic_prefix cp,
51 unsigned dft_size,
52 double center_frequency_Hz,
53 bool is_tx)
54 {
55 double sampling_rate_Hz = to_sampling_rate_Hz(scs, dft_size);
56 srsran_assert(std::isnormal(sampling_rate_Hz),
57 "Invalid sampling rate from SCS {} kHz and DFT size {}.",
58 scs_to_khz(scs),
59 dft_size);
60
61 unsigned nslot_per_subframe = get_nof_slots_per_subframe(scs);
62 unsigned nsymb_per_slot = get_nsymb_per_slot(cp);
63 double sign_two_pi = ((is_tx) ? -1 : 1) * 2.0 * M_PI;
64
65 // For each symbol in a subframe.
66 for (unsigned symbol = 0, symbol_offset = 0; symbol != nslot_per_subframe * nsymb_per_slot; ++symbol) {
67 // Add cyclic prefix length to the symbol offset.
68 symbol_offset += cp.get_length(symbol, scs).to_samples(sampling_rate_Hz);
69
70 // Calculate the time between the start of the subframe and the start of the symbol.
71 double start_time_s = static_cast<double>(symbol_offset) / sampling_rate_Hz;
72
73 // Calculate the phase in radians.
74 double symbol_phase = sign_two_pi * center_frequency_Hz * start_time_s;
75
76 // Calculate phase compensation.
77 coefficients.emplace_back(static_cast<cf_t>(std::polar(1.0, symbol_phase)));
78
79 // Advance the symbol size.
80 symbol_offset += dft_size;
81 }
82 }
83
88 cf_t get_coefficient(unsigned symbol_index) const
89 {
90 srsran_assert(symbol_index < coefficients.size(),
91 "The symbol index within a subframe {} exceeds the number of symbols in the subframe {}.",
92 symbol_index,
93 coefficients.size());
94 return coefficients[symbol_index];
95 }
96};
