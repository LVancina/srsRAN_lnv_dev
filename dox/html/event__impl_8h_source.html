33template <typename AwaiterType>
34void flush_awaiter_list(AwaiterType* current)
35{
36 while (current != nullptr) {
37 auto* next = current->next;
38 current->suspended_handle.resume();
39 current = next;
40 }
41}

46class manual_event_common
47{
48public:
49 struct awaiter_common {
50 explicit awaiter_common(manual_event_common& event_) : event(&event_) {}
51 bool await_ready() const { return event->is_set(); }
52 void await_suspend(coro_handle<> c)
53 {
54 // Store suspending coroutine.
55 suspended_handle = c;
56
57 // enqueue awaiter
58 next = static_cast<awaiter_common*>(std::exchange(event->state, static_cast<void*>(this)));
59 }
60
61 manual_event_common* event;
62 awaiter_common* next;
63 coro_handle<> suspended_handle;
64 };
65
66 manual_event_common() = default;
67 manual_event_common(const manual_event_common& event_) = delete;
68 manual_event_common& operator=(const manual_event_common& event_) = delete;
69 ~manual_event_common() {}
70
72 void reset()
73 {
74 if (is_set()) {
75 state = nullptr;
76 }
77 }
78
80 bool is_set() const { return state == this; }
81
82protected:
83 // Event state. Can be in 3 states:
84 // - this - event is set
85 // - nullptr - event is not set and with no awaiters pending
86 // - other - state is not set. Points to first element in linked list of awaiters
87 void* state = nullptr;
88
90 void flush()
91 {
92 if (is_set()) {
93 return;
94 }
95 auto old_state = std::exchange(state, this);
96 flush_awaiter_list(static_cast<awaiter_common*>(old_state));
97 }
98};
