36/// \brief Describes the PDCCH processor interface.
37///
38/// The PDCCH processor shall:
39/// - encode DCI, as per TS38.212 Section 7.3, where the DCI payload is given in \c dci_description::payload,
40/// - modulate PDCCH, as per TS38.211 Section 7.3.2, and
41/// - generate DMRS for PDCCH, as per TS38.211 Section 7.4.1.3.
42///
43/// \remark One Resource Element Group (REG) is equivalent to one Resource Block (RB), that is 12 Resource Elements
44/// (REs).
45/// \remark One Control-Channel Element (CCE) is equivalent to 6 REGs.
46/// \remark REGs are indexed first in time (for each symbol) and then in frequency.
47class pdcch_processor
48{
49public:
50 /// Describes a DCI transmission.
51 struct dci_description {
52 /// Parameter \f$x_{rnti, k}\f$ as per TS38.211 section 7.3.2 {0...65535}.
53 unsigned rnti;
54 /// Parameter \f$n_{ID}\f$ used for DMRS scrambling as per TS38.211 Section 7.4.1.3.1 {0...65535}.
55 unsigned n_id_pdcch_dmrs;
56 /// Parameter \f$n_{ID}\f$ used in data scrambling as per TS38.211 Section 7.3.2.3 {0...65535}.
57 unsigned n_id_pdcch_data;
58 /// Parameter \f$n_{RNTI}\f$ used for PDCCH data scrambling in TS38.211 Section 7.3.2.3 {1...65535}.
59 unsigned n_rnti;
60 /// CCE start index used to send the DCI {0, 135}.
61 unsigned cce_index;
62 /// Indicates the number of CCE used by the PDCCH transmission as per TS38.211 Section 7.3.2.1 {1, 2, 4, 8, 16}.
63 unsigned aggregation_level;
64 /// Ratio of PDCCH DM-RS EPRE to SSS EPRE in decibels.
65 float dmrs_power_offset_dB;
66 /// Ratio of PDCCH Data EPRE to SSS EPRE in decibels.
67 float data_power_offset_dB;
68 /// DCI payload as unpacked bits.
69 static_vector<uint8_t, pdcch_constants::MAX_DCI_PAYLOAD_SIZE> payload;
70 /// Precoding configuration.
71 precoding_configuration precoding;
72 };
73
74 /// CCE-to-REG mapping types as per TS38.211 Section 7.3.2.2.
75 enum class cce_to_reg_mapping_type {
76 /// CORESET is configured by the PBCH or SIB1, (subcarrier 0 of the CORESET)
77 CORESET0 = 0,
78 /// Not configured by PBCH or SIB 1, and non-interleaved.
79 NON_INTERLEAVED,
80 /// Not configured by PBCH or SIB 1, and interleaved.
81 INTERLEAVED
82 };
83
84 /// \brief Describes CORESET parameters.
85 ///
86 /// Provides the Control Resource Set (CORESET, TS38.331 \c ControlResourceSet for more information) related
87 /// parameters common for up to \c MAX_NOF_DCI Downlink Control Information (DCI) transmissions.
88 ///
89 /// For \c controlResourceSetZero (CORESET 0, TS38.213 Section 13 for more information) there are some exceptions:
90 /// - \c cce_to_reg_mapping_type shall be set to \c CORESET0.
91 /// - \c bwp_start_rb and \c bwp_size_rb indicate the CORESET start point and contiguous resource blocks.
92 /// - \c shift_index shall be equal to the physical cell identifier {0...1007}.
93 /// - \c cce_to_reg_mapping_type, \c frequency_resources, \c reg_bundle_size, \c interleaver_size are ignored.
94 struct coreset_description {
95 /// Number of contiguous PRBs allocated to the BWP {1, ..., 275}.
96 unsigned bwp_size_rb;
97 /// BWP start RB index from Point A {0, ..., 274}.
98 unsigned bwp_start_rb;
99 /// Starting OFDM symbol for the CORESET.
100 unsigned start_symbol_index;
101 /// Contiguous time duration of the CORESET in number of symbols {1,2,3}. Corresponds to L1 parameter
102 /// \f$N^{CORESET}_{symb}\f$ as per RS 38.211 Section 7.3.2.2.
103 unsigned duration;
104 /// \brief Frequency domain resources.
105 ///
106 /// This is a bitmap defining non-overlapping groups of 6 PRBs in ascending order. Defined as \f$N_{CORESET}^RB\f$
107 /// as per TS38.211 Section 7.3.2.2.
108 freq_resource_bitmap frequency_resources;
109 /// CCE-to-REG mapping.
110 cce_to_reg_mapping_type cce_to_reg_mapping;
111 /// \brief The number of REGs in a bundle. Corresponds to parameter \f$L\f$ in TS38.211 7.3.2.2.
112 ///
113 /// Ignored for \c cce_to_reg_mapping_type set to \c INTERLEAVED or \c CORESET0. It must be:
114 /// - {2,6} for duration of 1 or 2 symbols, and
115 /// - {3,6} for duration of 3 symbols.
116 unsigned reg_bundle_size;
117 /// \brief The CCE-to-REG interleaver size. Corresponds to parameter \f$R\f$ in TS38.211 7.3.2.2.
118 ///
119 /// Ignored for \c cce_to_reg_mapping_type set to \c INTERLEAVED or \c CORESET0. It must be {2,3,6}.
120 unsigned interleaver_size;
121 /// \brief Shift index \f$n_{shift}\f$ in TS38.211 Section 7.3.2.2.
122 ///
123 /// The value is:
124 /// - set to the physical cell identifier for a transmission in CORESET0,
125 /// - ignored for non-interleaved mapping, and
126 /// - set to {0,275} for non-interleaved CCE-to-REG mapping.
127 unsigned shift_index;
128 };
129
130 /// Collects the PDCCH parameters for a transmission.
131 struct pdu_t {
132 /// Context information.
133 optional<pdcch_context> context;
134 /// Indicates the slot and numerology.
135 slot_point slot;
136 /// Cyclic prefix type.
137 cyclic_prefix cp;
138 /// Provides CORESET description.
139 coreset_description coreset;
140 /// Downlink Control Information.
141 dci_description dci;
142 };
143
144 /// Default detsructor.
145 virtual ~pdcch_processor() = default;
146
147 /// \brief Processes a PDCCH transmission.
148 ///
149 /// \param[out] mapper Resource grid mapper interface.
150 /// \param[in] pdu Necessary parameters to process the PDCCH transmission.
151 virtual void process(resource_grid_mapper& mapper, const pdu_t& pdu) = 0;
152};
