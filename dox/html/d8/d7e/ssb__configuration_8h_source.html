30/// SSB-Index identifies an SS-Block within an SS-Burst.
31/// \remark See TS 38.331, "SSB-Index" and "maxNrofSSBs". See also, TS 38.213, clause 4.1.
32enum ssb_id_t : uint8_t { MIN_SSB_ID = 0, MAX_SSB_ID = 63, MAX_NOF_SSB_RESOURCES = 64 };
33
34/// Maximum number of SS/PBCH beams in a SS/PBCH period as per TS38.213 Section 4.1.
35constexpr size_t NOF_BEAMS = 64;
36
37/// SSB Configuration.
38struct ssb_configuration {
39 /// SSB subcarrier spacing.
40 subcarrier_spacing scs;
41 /// Represents the offset to Point A in PRBs as per TS38.331 Section 6.3.2 IE offsetToPointA.
42 /// Possible values: {0, ..., 2199}.
43 ssb_offset_to_pointA offset_to_point_A;
44 /// SSB periodicity.
45 ssb_periodicity ssb_period;
46 /// k_ssb or SSB SubcarrierOffest, as per TS38.211 Section 7.4.3.1. Possible values: {0, ..., 23}.
47 ssb_subcarrier_offset k_ssb;
48 /// Each bit in this bitmap represents if a beam is active or not as per TS38.331 Section 6.3.2 IE
49 /// ssb-PositionsInBurst.
50 uint64_t ssb_bitmap;
51 /// The n-th element of the array indicates what Beam ID to use for the n-th SSB occasion in \c ssb_bitmap. Only
52 /// relevant if the n-th bit of \c ssb_bitmap is set.
53 std::array<uint8_t, NOF_BEAMS> beam_ids;
54 /// PSS EPRE to SSS EPRE for SSB. TS 38.213, Section 4.1 gives an explanation of this measure, but doesn't provide a
55 /// name for this parameter. Nor is there a name in the TS 38.331.
56 ssb_pss_to_sss_epre pss_to_sss_epre;
57 /// Average EPRE of the resources elements that carry secondary synchronization signals in dBm, as per
58 /// ServingCellConfigCommonSIB, TS 38.331. Possible values: {-60, ..., 50}
59 int ssb_block_power;
60};
