75/// Open Fronthaul message receiver.
76///
77/// This class listens to incoming Ethernet frames and decodes them as Open Fronthaul messages. Once a new message is
78/// detected, is it handled to the corresponding data flow for further processing.
79class message_receiver : public ether::frame_notifier
80{
81public:
82 message_receiver(const message_receiver_config& config, message_receiver_dependencies&& dependencies);
83
84 // See interface for documentation.
85 void on_new_frame(span<const uint8_t> payload) override;
86
87 /// Returns the Ethernet receiver of this Open Fronthaul message receiver.
88 ether::receiver& get_ethernet_receiver() { return *eth_receiver; }
89
90private:
91 /// Returns true if the ethernet frame represented by the given eth parameters should be filtered, otherwise false.
92 bool should_ethernet_frame_be_filtered(const ether::vlan_frame_params& eth_params) const;
93
94 /// Returns true if the eCPRI packet represented by the given eCPRI parameters should be filtered, otherwise false.
95 bool should_ecpri_packet_be_filtered(const ecpri::packet_parameters& ecpri_params) const;
96
97private:
98 srslog::basic_logger& logger;
99 const ether::vlan_frame_params vlan_params;
100 const static_vector<unsigned, MAX_NOF_SUPPORTED_EAXC> ul_prach_eaxc;
101 const static_vector<unsigned, MAX_NOF_SUPPORTED_EAXC> ul_eaxc;
102 rx_window_checker& window_checker;
103 std::unique_ptr<sequence_id_checker> seq_id_checker;
104 std::unique_ptr<ether::vlan_frame_decoder> vlan_decoder;
105 std::unique_ptr<ecpri::packet_decoder> ecpri_decoder;
106 std::unique_ptr<uplane_message_decoder> uplane_decoder;
107 std::unique_ptr<data_flow_uplane_uplink_data> data_flow_uplink;
108 std::unique_ptr<data_flow_uplane_uplink_prach> data_flow_prach;
109 std::unique_ptr<ether::receiver> eth_receiver;
110};
