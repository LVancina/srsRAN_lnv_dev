29/// \brief Downlink processor state management class.
30///
31/// Represents the internal state of the upper PHY downlink processor single executor implementation.
32class downlink_processor_single_executor_state
33{
34public:
35 downlink_processor_single_executor_state() : state(states::idle), pending_pdus(0)
36 {
37 // Do nothing.
38 }
39
40 /// Possible states of the downlink proecsosr.
41 enum class states {
42 /// The processor is ready for a new resource grid.
43 idle,
44 /// The DL PDUs are accepted and processed.
45 processing,
46 /// No new DL PDUs are accepted. The resource grid is sent when there are no ongoing processing tasks.
47 finishing
48 };
49
50 /// Resource grid configured callback. It is meant to be called when a new resource grid has been accepted.
51 void on_resource_grid_configured()
52 {
53 srsran_assert(
54 (state == states::idle), "DL processor grid was configured in an invalid state, i.e., {}.", to_string(state));
55 srsran_assert((pending_pdus == 0), "DL processor grid was configured with {} pending PDUs.", pending_pdus);
56 state = states::processing;
57 }
58
59 /// \brief Finish processing PDUs callback.
60 ///
61 /// \return \c true if the resource grid can be immediately sent, \c false otherwise.
62 bool on_finish_requested()
63 {
64 srsran_assert(
65 (state != states::idle), "DL processor finish was requested in an invalid state, i.e., {}.", to_string(state));
66 state = states::finishing;
67 return (pending_pdus == 0);
68 }
69
70 /// Resource grid sent callback.
71 void on_grid_sent()
72 {
73 srsran_assert((pending_pdus == 0), "Grid was sent with {} pending PDUs.", pending_pdus);
74 srsran_assert(
75 (state == states::finishing), "DL processor grid was sent in an invalid state, i.e., {}.", to_string(state));
76 state = states::idle;
77 }
78
79 /// Task creation callback. Increases the pending task counter.
80 void on_task_creation()
81 {
82 srsran_assert(
83 (state == states::processing), "DL processor task created in an invalid state, i.e., {}.", to_string(state));
84 increase_pending_pdus();
85 }
86
87 /// Task completion callback. Decreases the pending task counter and checks wether the resource grid can be sent.
88 ///
89 /// \return \c true if the resource grid can be immediately sent, \c false otherwise.
90 bool on_task_completion()
91 {
92 decrease_pending_pdus();
93 return (state == states::finishing) && (pending_pdus == 0);
94 }
95
96 /// Checks wether the current state allows for a new resource grid to be configured.
97 bool is_idle() const { return state == states::idle; }
98
99 /// Checks wether the current state allows for a new task to be enqueued.
100 bool is_processing() const { return state == states::processing; }
101
102private:
103 /// Increases the pending PDU counter.
104 void increase_pending_pdus() { ++pending_pdus; }
105
106 /// Decreases the pending PDU counter.
107 void decrease_pending_pdus()
108 {
109 if (pending_pdus > 0) {
110 --pending_pdus;
111 }
112 }
113
114 static std::string to_string(states state_)
115 {
116 switch (state_) {
117 case states::idle:
118 return "idle";
119 case states::processing:
120 return "processing";
121 case states::finishing:
122 default:
123 return "finishing";
124 }
125 }
126
127 /// Downlink processor current state.
128 states state;
129 /// Number of pending PDUs to process before sending the resource grid.
130 unsigned pending_pdus;
131};
132} // namespace srsran
