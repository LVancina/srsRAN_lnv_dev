39/// \brief MAC-to-FAPI translator.
40///
41/// This class listens to cell-specific MAC events carrying scheduling results and translates them into FAPI messages
42/// that are sent through the FAPI message gateway.
43class mac_to_fapi_translator : public mac_cell_result_notifier
44{
45public:
46 mac_to_fapi_translator(srslog::basic_logger& logger_,
47 fapi::slot_message_gateway& msg_gw_,
48 fapi::slot_last_message_notifier& last_msg_notifier_,
49 std::unique_ptr<precoding_matrix_mapper> pm_mapper_,
50 std::unique_ptr<uci_part2_correspondence_mapper> part2_mapper_,
51 unsigned cell_nof_prbs_) :
52 logger(logger_),
53 msg_gw(msg_gw_),
54 last_msg_notifier(last_msg_notifier_),
55 pm_mapper(std::move(pm_mapper_)),
56 part2_mapper(std::move(part2_mapper_)),
57 cell_nof_prbs(cell_nof_prbs_)
58 {
59 srsran_assert(pm_mapper, "Invalid precoding matrix mapper");
60 srsran_assert(part2_mapper, "Invalid Part2 mapper");
61 }
62
63 // See interface for documentation.
64 void on_new_downlink_scheduler_results(const mac_dl_sched_result& dl_res) override;
65
66 // See interface for documentation.
67 void on_new_downlink_data(const mac_dl_data_result& dl_data) override;
68
69 // See interface for documentation.
70 void on_new_uplink_scheduler_results(const mac_ul_sched_result& ul_res) override;
71
72 // See interface for documentation.
73 void on_cell_results_completion(slot_point slot) override;
74
75private:
76 /// Handles the UL_DCI.request message.
77 void handle_ul_dci_request(span<const pdcch_ul_information> pdcch_info,
78 span<const dci_payload> payloads,
79 slot_point slot,
80 bool is_last_message_in_slot);
81
82private:
83 /// FAPI logger.
84 srslog::basic_logger& logger;
85 /// FAPI message gateway to the outside world.
86 fapi::slot_message_gateway& msg_gw;
87 /// Slot-specific last message notifier.
88 fapi::slot_last_message_notifier& last_msg_notifier;
89 /// Precoding matrix mapper.
90 std::unique_ptr<precoding_matrix_mapper> pm_mapper;
91 /// UCI Part2 correspondence mapper.
92 std::unique_ptr<uci_part2_correspondence_mapper> part2_mapper;
93 /// Cell number of resource blocks.
94 const unsigned cell_nof_prbs;
95};
