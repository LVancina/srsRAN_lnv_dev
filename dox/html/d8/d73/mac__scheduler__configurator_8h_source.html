32/// \brief Adapter interface between MAC and scheduler that allows the srsGNB MAC to configure different scheduler
33/// implementations.
34/// Note: This adapter interface makes scheduler configuration methods awaitable.
35class mac_scheduler_configurator
36{
37public:
38 virtual ~mac_scheduler_configurator() = default;
39
40 /// \brief Adds a new cell configuration to the scheduler and activates it.
41 ///
42 /// \param msg New cell configuration.
43 virtual void add_cell(const mac_cell_creation_request& msg) = 0;
44
45 /// \brief Removes an existing cell from the scheduler.
46 ///
47 /// \param cell_index DU-specific index of the cell to remove.
48 virtual void remove_cell(du_cell_index_t cell_index) = 0;
49
50 /// \brief Adds a new UE configuration to the scheduler.
51 ///
52 /// \param msg new UE configuration.
53 /// \return Asynchronous task handle that represents the state and outcome of the UE reconfiguration task.
54 virtual async_task<bool> handle_ue_creation_request(const mac_ue_create_request& msg) = 0;
55
56 /// \brief Reconfigures an existing UE configuration in the scheduler.
57 ///
58 /// \param msg new configuration for existing UE.
59 /// \return Asynchronous task handle that represents the state and outcome of the UE reconfiguration task.
60 virtual async_task<bool> handle_ue_reconfiguration_request(const mac_ue_reconfiguration_request& msg) = 0;
61
62 /// \brief Removes UE from MAC scheduler in an asynchronous manner.
63 /// The scheduler shouldn't allocate more grants directed at the UE being removed after this procedure is complete.
64 virtual async_task<bool> handle_ue_removal_request(const mac_ue_delete_request& msg) = 0;
65};
