35/// Concurrent queue type used for storing the PUSCH processor result notifier adaptors.
36using free_adaptor_queue =
37 concurrent_queue<unsigned, concurrent_queue_policy::lockfree_mpmc, concurrent_queue_wait_policy::sleep>;

39/// \brief Adapts the PUSCH processor result notifier to the upper PHY receive results notifier.
40///
41/// It collects the Channel State Information (CSI), control and data decoding information from a PUSCH processor object
42/// and notifies them through the upper PHY notification interface.
43///
44/// \remark The order of the calls matters. Method on_csi() must be called before on_uci() or on_sch(). Otherwise, an
45/// assertion is triggered.
46class pusch_processor_result_notifier_adaptor : private pusch_processor_result_notifier
47{
48public:
49 /// \brief Constructs an adaptor.
50 ///
51 /// The PUSCH adaptor queue identifier is set to the current size of the queue.
52 ///
53 /// \param[in] queue_ Free adaptor queue instance.
54 pusch_processor_result_notifier_adaptor(free_adaptor_queue& queue_) : queue(queue_), queue_identifier(queue.size())
55 {
56 queue.push_blocking(queue_identifier);
57 }
58
59 /// \brief Creates a PUSCH processor result notifier adaptor.
60 /// \param[in] notifier_ Upper physical layer result notifier.
61 /// \param[in] rnti_ User RNTI.
62 /// \param[in] slot_ Current slot.
63 /// \param[in] harq_id_ User HARQ process identifier.
64 /// \param[in] payload_ View to the data payload.
65 pusch_processor_result_notifier& configure(upper_phy_rx_results_notifier& notifier_,
66 rnti_t rnti_,
67 slot_point slot_,
68 harq_id_t harq_id_,
69 span<const uint8_t> payload_)
70 {
71 notifier = &notifier_;
72 rnti = rnti_;
73 slot = slot_;
74 harq_id = harq_id_;
75 payload = payload_;
76 return *this;
77 }
78
79private:
80 // See interface for documentation.
81 void on_uci(const pusch_processor_result_control& uci) override
82 {
83 srsran_assert(notifier != nullptr, "Invalid notifier.");
84
85 ul_pusch_results_control result;
86 result.rnti = rnti;
87 result.slot = slot;
88 result.csi = uci.csi;
89
90 if (!uci.harq_ack.payload.empty()) {
91 result.harq_ack.emplace(uci.harq_ack);
92 }
93
94 if (!uci.csi_part1.payload.empty()) {
95 result.csi1.emplace(uci.csi_part1);
96 }
97
98 if (!uci.csi_part2.payload.empty()) {
99 result.csi2.emplace(uci.csi_part2);
100 }
101
102 notifier->on_new_pusch_results_control(result);
103
104 l1_tracer << instant_trace_event("on_uci", instant_trace_event::cpu_scope::thread);
105 }
106
107 // See interface for documentation.
108 void on_sch(const pusch_processor_result_data& sch) override
109 {
110 // Gets the notifier and invalidates. It makes sure the notifier is no longer available after the exchange.
111 upper_phy_rx_results_notifier* notifier_ = nullptr;
112 std::exchange(notifier_, notifier);
113 srsran_assert(notifier_ != nullptr, "Invalid notifier.");
114
115 // Notify the completion of PUSCH data.
116 ul_pusch_results_data result;
117 result.rnti = rnti;
118 result.slot = slot;
119 result.csi = sch.csi;
120 result.harq_id = harq_id;
121 result.decoder_result = sch.data;
122 result.payload = (sch.data.tb_crc_ok) ? payload : span<const uint8_t>();
123 notifier_->on_new_pusch_results_data(result);
124
125 // Return the adaptor identifier to the queue.
126 queue.push_blocking(queue_identifier);
127
128 l1_tracer << instant_trace_event("on_sch", instant_trace_event::cpu_scope::thread);
129 }
130
131 free_adaptor_queue& queue;
132 unsigned queue_identifier;
133 upper_phy_rx_results_notifier* notifier = nullptr;
134 rnti_t rnti = rnti_t::INVALID_RNTI;
135 slot_point slot = slot_point();
136 harq_id_t harq_id = harq_id_t::INVALID_HARQ_ID;
137 span<const uint8_t> payload;
138};
