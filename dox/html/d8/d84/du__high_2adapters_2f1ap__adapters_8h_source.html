56class f1ap_du_configurator_adapter : public f1ap_du_configurator
57{
58public:
59 explicit f1ap_du_configurator_adapter(timer_factory timers_) : timers(timers_)
60 {
61 for (unsigned i = 0; i != MAX_NOF_DU_UES; ++i) {
62 ues.emplace(i, to_du_ue_index(i), timers);
63 }
64 }
65
66 void connect(du_manager_configurator& du_mng_)
67 {
68 du_mng = &du_mng_;
69 for (unsigned i = 0; i != MAX_NOF_DU_UES; ++i) {
70 ues[i].connect(*du_mng);
71 }
72 }
73
74 timer_factory& get_timer_factory() override { return timers; }
75
76 void schedule_async_task(async_task<void>&& task) override { du_mng->schedule_async_task(std::move(task)); }
77
78 f1ap_ue_task_scheduler& get_ue_handler(du_ue_index_t ue_index) override { return ues[ue_index]; }
79
80 du_ue_index_t find_free_ue_index() override { return du_mng->find_unused_du_ue_index(); }
81
82 async_task<f1ap_ue_context_creation_response>
83 request_ue_creation(const f1ap_ue_context_creation_request& request) override
84 {
85 return du_mng->handle_ue_context_creation(request);
86 }
87
88 async_task<f1ap_ue_context_update_response>
89 request_ue_context_update(const f1ap_ue_context_update_request& request) override
90 {
91 return du_mng->handle_ue_context_update(request);
92 }
93
94 async_task<void> request_ue_removal(const f1ap_ue_delete_request& request) override
95 {
96 return du_mng->handle_ue_delete_request(request);
97 }
98
99 void notify_reestablishment_of_old_ue(du_ue_index_t new_ue_index, du_ue_index_t old_ue_index) override
100 {
101 du_mng->handle_ue_reestablishment(new_ue_index, old_ue_index);
102 }
103
104private:
105 timer_factory timers;
106 du_manager_configurator* du_mng = nullptr;
107 slotted_array<f1ap_ue_task_scheduler_adapter, MAX_NOF_DU_UES> ues;
108};
