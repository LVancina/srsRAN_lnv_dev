35/// \brief PUSCH code block decoder.
36///
37/// Reverts the rate matching, LDPC encoding and CRC check.
38class pusch_codeblock_decoder
39{
40public:
41 /// CRC calculators used in shared channels.
42 struct sch_crc {
43 /// For short TB checksums.
44 std::unique_ptr<crc_calculator> crc16;
45 /// For long TB checksums.
46 std::unique_ptr<crc_calculator> crc24A;
47 /// For segment-specific checksums.
48 std::unique_ptr<crc_calculator> crc24B;
49 };
50
51 /// \brief PUSCH code block decoder constructor.
52 ///
53 /// Sets up the internal components, namely LDPC rate dematcher, LDPC decoder and all the CRC calculators.
54 ///
55 /// \param[in] rdem Pointer to an LDPC rate dematcher object.
56 /// \param[in] dec Pointer to an LDPC decoder object.
57 /// \param[in] crcs Structure with pointers to three CRC calculator objects, with generator polynomials of type \c
58 /// CRC16, \c CRC24A and \c CRC24B.
59 pusch_codeblock_decoder(std::unique_ptr<ldpc_rate_dematcher> rdem, std::unique_ptr<ldpc_decoder> dec, sch_crc& crcs) :
60 dematcher(std::move(rdem)),
61 decoder(std::move(dec)),
62 crc_set({std::move(crcs.crc16), std::move(crcs.crc24A), std::move(crcs.crc24B)})
63 {
64 srsran_assert(dematcher, "Invalid dematcher.");
65 srsran_assert(crc_set.crc16, "Invalid CRC16 calculator.");
66 srsran_assert(crc_set.crc24A, "Invalid CRC24A calculator.");
67 srsran_assert(crc_set.crc24B, "Invalid CRC24B calculator.");
68 srsran_assert(crc_set.crc16->get_generator_poly() == crc_generator_poly::CRC16, "Wrong TB CRC calculator.");
69 srsran_assert(crc_set.crc24A->get_generator_poly() == crc_generator_poly::CRC24A, "Wrong TB CRC calculator.");
70 srsran_assert(crc_set.crc24B->get_generator_poly() == crc_generator_poly::CRC24B, "Wrong TB CRC calculator.");
71 };
72
73 /// Selects the CRC calculator from a CRC polynomial.
74 crc_calculator* select_crc(crc_generator_poly poly)
75 {
76 srsran_assert((poly == crc_generator_poly::CRC16) || (poly == crc_generator_poly::CRC24A) ||
77 (poly == crc_generator_poly::CRC24B),
78 "Invalid CRC polynomial.");
79 if (poly == crc_generator_poly::CRC16) {
80 return crc_set.crc16.get();
81 }
82 if (poly == crc_generator_poly::CRC24A) {
83 return crc_set.crc24A.get();
84 }
85 if (poly == crc_generator_poly::CRC24B) {
86 return crc_set.crc24B.get();
87 }
88 return nullptr;
89 }

104 /// \brief Rate matches.
105 ///
106 /// Reverts the rate matching process and applies LDPC decoding.
107 ///
108 /// \param[out] cb_data Code block data after decoding.
109 /// \param[in,out] rm_buffer Rate matching buffer.
110 /// \param[in] cb_llrs New soft bits to write in the soft buffer.
111 /// \param[in] new_data Set to true for indicating a new transmission.
112 /// \param[in] crc_poly CRC polynomial used for the code block.
113 /// \param[in] use_early_stop Set to true to allow the LDPC decoder to stop decoding when the CRC matches.
114 /// \param[in] nof_ldpc_iterations Number of LDPC decoder iterations.
115 /// \param[in] metadata Code block metadata.
116 /// \return The number of iterations if the CRC matches after the LDPC decoder. Otherwise, \c nullopt.
117 optional<unsigned> decode(bit_buffer cb_data,
118 span<log_likelihood_ratio> rm_buffer,
119 span<const log_likelihood_ratio> cb_llrs,
120 bool new_data,
121 srsran::crc_generator_poly crc_poly,
122 bool use_early_stop,
123 unsigned nof_ldpc_iterations,
124 const codeblock_metadata& metadata);
125
126private:
127 /// Pointer to an LDPC rate-dematcher.
128 std::unique_ptr<ldpc_rate_dematcher> dematcher;
129 /// Pointer to an LDPC decoder.
130 std::unique_ptr<ldpc_decoder> decoder;
131 /// \brief Pointer to a CRC calculator for TB-wise checksum.
132 ///
133 /// Only the CRC calculator with generator polynomial crc_generator_poly::CRC24A, used for long transport blocks, is
134 /// needed. Indeed, if a transport block is short enough not to be segmented, the CRC is verified by the decoder.
135 sch_crc crc_set;
136};
