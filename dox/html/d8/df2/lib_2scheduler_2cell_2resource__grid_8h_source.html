38/// Parameters of a PDSCH or PUSCH grant allocation within a BWP.
39struct bwp_sch_grant_info {
40 const bwp_configuration* bwp_cfg;
41 ofdm_symbol_range symbols;
42 /// PRB interval within respective BWP.
43 prb_interval prbs;
44
45 bwp_sch_grant_info(const bwp_configuration& bwp_, ofdm_symbol_range symbols_, prb_interval prbs_) :
46 bwp_cfg(&bwp_), symbols(symbols_), prbs(prbs_)
47 {
48 srsran_sanity_check(symbols.stop() <= get_nsymb_per_slot(bwp_cfg->cp), "OFDM symbols do not fit slot");
49 srsran_sanity_check(prbs.stop() <= bwp_cfg->crbs.length(), "PRBs={} do not fit BWP={}", prbs, bwp_cfg->crbs);
50 }
51};

77/// \brief Represents the Symbol x CRB resource grid of a DL/UL Carrier. The number of CRBs of the grid will depend on
78/// the carrier bandwidth and numerology used.
79class carrier_subslot_resource_grid
80{
81public:
82 explicit carrier_subslot_resource_grid(const scs_specific_carrier& carrier_cfg);
83
84 subcarrier_spacing scs() const { return carrier_cfg.scs; }
85 unsigned nof_rbs() const { return carrier_cfg.carrier_bandwidth; }
86 unsigned offset() const { return carrier_cfg.offset_to_carrier; }
87 crb_interval rb_dims() const { return get_carrier_rb_dims(carrier_cfg); }
88
89 /// Clearer Carrier Resource Grid.
90 void clear();
91
92 /// Allocates the symbol x CRB range in the carrier resource grid.
93 /// \param symbols OFDM symbol interval of the allocation. Interval must fall within [0, 14).
94 /// \param crbs CRB interval, where CRB=0 corresponds to the CRB closest to pointA.
95 void fill(ofdm_symbol_range symbols, crb_interval crbs);
96
97 /// Allocates the symbol x CRB list in the carrier resource grid.
98 /// \param symbols OFDM symbol interval of the allocation. Interval must fall within [0, 14).
99 /// \param crbs List of CRBs, where CRB=0 corresponds to the CRB closest to pointA.
100 void fill(ofdm_symbol_range symbols, span<const uint16_t> crb_list);
101
102 /// Checks whether the provided symbol x CRB range collides with any other allocation in the carrier resource grid.
103 /// \param symbols OFDM symbol interval of the allocation. Interval must fall within [0, 14).
104 /// \param crbs CRB interval, where CRB=0 corresponds to the CRB closest to pointA.
105 /// \return true if a collision was detected. False otherwise.
106 bool collides(ofdm_symbol_range symbols, crb_interval crbs) const;
107
108 /// Checks whether the provided symbol x CRB list collides with any other allocation in the carrier resource grid.
109 /// \param symbols OFDM symbol interval of the allocation. Interval must fall within [0, 14).
110 /// \param crbs List of CRBs, where CRB=0 corresponds to the CRB closest to pointA.
111 /// \return true if a collision was detected. False otherwise.
112 bool collides(ofdm_symbol_range symbols, span<const uint16_t> crb_list) const;
113
114 /// \brief Calculates a bitmap where each bit set one represents a CRB that is occupied or unavailable.
115 /// A CRB is considered occupied if it is outside of the provided BWP CRB boundaries or if it is already allocated
116 /// in at least one OFDM symbol of the provided OFDM symbol range.
117 /// \param[in] bwp_crb_lims CRB range where the BWP is located in the frequency domain, and used for the CRB to PRB
118 /// conversion.
119 /// \param[in] symbols Range of OFDM symbols, where the search for available CRBs is carrier out.
120 /// \return an CRB bitmap with bits set to one for unavailable CRBs.
121 crb_bitmap used_crbs(crb_interval bwp_crb_lims, ofdm_symbol_range symbols) const;
122
123 /// Checks whether the provided symbol x CRB range in the carrier resource grid is set.
124 /// \param symbols OFDM symbol interval of the allocation. Interval must fall within [0, 14).
125 /// \param crbs CRB interval, where CRB=0 corresponds to the CRB closest to pointA.
126 /// \return true if all symbols and CRBs of the provided range are set. False otherwise.
127 bool all_set(ofdm_symbol_range symbols, crb_interval crbs) const;
128
129private:
130 /// Represents a matrix of symbol index x carrier RB index. The matrix dimensions get scaled based on the number
131 /// of carrier RBs. RB index=0 corresponds to the carrier offset. Resources in the bitset are represented in the
132 /// following order: [{symb=0,RB=0}, {symb=0,RB=1}, ..., {symb=1,RB=0}, ..., {symb=14,RB=carrierRBs}]
133 using slot_rb_bitmap = bounded_bitset<NOF_OFDM_SYM_PER_SLOT_NORMAL_CP * MAX_NOF_PRBS, true>;
134
135 /// Carrier configuration containining numerology, carrier offset and carrier bandwidth.
136 scs_specific_carrier carrier_cfg;
137
138 /// symbol x RB bitmap of this carrier.
139 slot_rb_bitmap slot_rbs;
140};

245/// Circular Ring of cell_slot_resource_grid objects. This class manages the automatic resetting of
246/// cell_slot_resource_grid objects, once they become old.
247struct cell_resource_allocator {
248 /// \brief Number of previous slot results to keep in history before they get deleted.
249 ///
250 /// Having access to past decisions is useful during the handling of error indications.
251 static const size_t RING_MAX_HISTORY_SIZE = 8;
252 /// Number of slots managed by this container.
253 static const size_t RING_ALLOCATOR_SIZE =
254 get_allocator_ring_size_gt_min(RING_MAX_HISTORY_SIZE + SCHEDULER_MAX_K0 + NTN_CELL_SPECIFIC_KOFFSET_MAX +
255 std::max(SCHEDULER_MAX_K1, SCHEDULER_MAX_K2 + MAX_MSG3_DELTA));
256
257 /// Cell configuration
258 const cell_configuration& cfg;
259
260 /// Maximum number of slots that can be allocated in advance.
261 const unsigned max_dl_slot_alloc_delay;
262 const unsigned max_ul_slot_alloc_delay;
263
264 explicit cell_resource_allocator(const cell_configuration& cfg_);
265
266 /// Indicate the processing of a new slot in the scheduler.
267 void slot_indication(slot_point sl_tx);
268
269 /// Cell index of the resource grid.
270 du_cell_index_t cell_index() const { return cfg.cell_index; }
271
272 /// Last slot indicated to the scheduler.
273 slot_point slot_tx() const { return last_slot_ind; }
274
275 /// Access resource allocator for a given {slot, cell}. The accessed slot corresponds to the latest slot indication
276 /// provided to the cell_resource_grid_manager + slot_delay provided as argument.
277 /// Given that slot_delay is unsigned, this class can only access the present and future slots.
278 /// \param slot_delay delay in #slots added to the last slot indication value.
279 /// \return the cell resource grid corresponding to the accessed slot.
280 const cell_slot_resource_allocator& operator[](unsigned slot_delay) const
281 {
282 assert_valid_sl(slot_delay);
283 slot_point sl_tx = last_slot_ind + slot_delay;
284 const cell_slot_resource_allocator& r = *slots[sl_tx.to_uint() % slots.size()];
285 srsran_assert(r.slot == sl_tx, "Bad access to uninitialized cell_resource_grid");
286 return r;
287 }
288 cell_slot_resource_allocator& operator[](unsigned slot_delay)
289 {
290 assert_valid_sl(slot_delay);
291 slot_point sl_tx = last_slot_ind + slot_delay;
292 cell_slot_resource_allocator& r = *slots[sl_tx.to_uint() % slots.size()];
293 srsran_assert(r.slot == sl_tx, "Bad access to uninitialized cell_resource_grid");
294 return r;
295 }
296
297 cell_slot_resource_allocator& operator[](slot_point slot) { return this->operator[](slot - last_slot_ind); }
298 const cell_slot_resource_allocator& operator[](slot_point slot) const
299 {
300 return this->operator[](slot - last_slot_ind);
301 }
302
303 /// \brief Access a past slot decision made by the scheduler for the given cell.
304 const cell_slot_resource_allocator* get_history(slot_point slot) const
305 {
306 int diff = last_slot_ind - slot;
307 if (diff < 0 or diff >= static_cast<int>(RING_MAX_HISTORY_SIZE)) {
308 return nullptr;
309 }
310 const cell_slot_resource_allocator& r = *slots[slot.to_uint() % slots.size()];
311 srsran_assert(r.slot == slot, "Bad access to uninitialized cell_resource_grid");
312 return &r;
313 }
314
315private:
316 /// Ensure we are not overflowing the ring.
317 void assert_valid_sl(unsigned slot_delay) const
318 {
319 srsran_sanity_check(slot_delay <= max_ul_slot_alloc_delay,
320 "The cell resource pool is too small for accessing a slot with delay: {}",
321 slot_delay);
322 }
323
324 /// The latest slot value indicated by the PHY to the MAC/scheduler.
325 slot_point last_slot_ind;
326
327 /// Circular pool of cell resource grids, where each entry represents a separate slot.
328 std::vector<std::unique_ptr<cell_slot_resource_allocator>> slots;
329};
