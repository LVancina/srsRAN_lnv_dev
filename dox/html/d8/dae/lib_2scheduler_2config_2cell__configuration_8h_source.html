79 /// Checks if DL/UL is active for current slot
80 bool is_fully_dl_enabled(slot_point sl) const
81 {
82 if (dl_symbols_per_slot_lst.empty()) {
83 // Note: dl_enabled_slot_lst is empty in the FDD case.
84 return true;
85 }
86 if (sl.numerology() != to_numerology_value(tdd_cfg_common->ref_scs)) {
87 // Convert slot into equivalent reference SCS.
88 sl = set_slot_numerology(sl, to_numerology_value(tdd_cfg_common->ref_scs));
89 }
90 return dl_symbols_per_slot_lst[sl.to_uint() % dl_symbols_per_slot_lst.size()] ==
91 get_nsymb_per_slot(dl_cfg_common.init_dl_bwp.generic_params.cp);
92 }
93 bool is_fully_ul_enabled(slot_point sl) const
94 {
95 if (ul_symbols_per_slot_lst.empty()) {
96 // Note: ul_enabled_slot_lst is empty in the FDD case.
97 return true;
98 }
99 if (sl.numerology() != to_numerology_value(tdd_cfg_common->ref_scs)) {
100 // Convert slot into equivalent reference SCS.
101 sl = set_slot_numerology(sl, to_numerology_value(tdd_cfg_common->ref_scs));
102 }
103 return ul_symbols_per_slot_lst[sl.to_uint() % ul_symbols_per_slot_lst.size()] ==
104 get_nsymb_per_slot(ul_cfg_common.init_ul_bwp.generic_params.cp);
105 }

107 bool is_dl_enabled(slot_point sl) const
108 {
109 if (dl_symbols_per_slot_lst.empty()) {
110 // Note: dl_enabled_slot_lst is empty in the FDD case.
111 return true;
112 }
113 if (sl.numerology() != to_numerology_value(tdd_cfg_common->ref_scs)) {
114 // Convert slot into equivalent reference SCS.
115 sl = set_slot_numerology(sl, to_numerology_value(tdd_cfg_common->ref_scs));
116 }
117 return dl_symbols_per_slot_lst[sl.to_uint() % dl_symbols_per_slot_lst.size()] > 0;
118 }
119 bool is_ul_enabled(slot_point sl) const
120 {
121 if (ul_symbols_per_slot_lst.empty()) {
122 // Note: ul_enabled_slot_lst is empty in the FDD case.
123 return true;
124 }
125 if (sl.numerology() != to_numerology_value(tdd_cfg_common->ref_scs)) {
126 // Convert slot into equivalent reference SCS.
127 sl = set_slot_numerology(sl, to_numerology_value(tdd_cfg_common->ref_scs));
128 }
129 return ul_symbols_per_slot_lst[sl.to_uint() % ul_symbols_per_slot_lst.size()] > 0;
130 }
131 unsigned get_nof_dl_symbol_per_slot(slot_point sl) const
132 {
133 if (dl_symbols_per_slot_lst.empty()) {
134 // Note: dl_enabled_slot_lst is empty in the FDD case.
135 return get_nsymb_per_slot(dl_cfg_common.init_dl_bwp.generic_params.cp);
136 }
137 if (sl.numerology() != to_numerology_value(tdd_cfg_common->ref_scs)) {
138 // Convert slot into equivalent reference SCS.
139 sl = set_slot_numerology(sl, to_numerology_value(tdd_cfg_common->ref_scs));
140 }
141 return dl_symbols_per_slot_lst[sl.to_uint() % dl_symbols_per_slot_lst.size()];
142 }
143 unsigned get_nof_ul_symbol_per_slot(slot_point sl) const
144 {
145 if (ul_symbols_per_slot_lst.empty()) {
146 // Note: ul_enabled_slot_lst is empty in the FDD case.
147 return get_nsymb_per_slot(ul_cfg_common.init_ul_bwp.generic_params.cp);
148 }
149 if (sl.numerology() != to_numerology_value(tdd_cfg_common->ref_scs)) {
150 // Convert slot into equivalent reference SCS.
151 sl = set_slot_numerology(sl, to_numerology_value(tdd_cfg_common->ref_scs));
152 }
153 return ul_symbols_per_slot_lst[sl.to_uint() % ul_symbols_per_slot_lst.size()];
154 }
155 const coreset_configuration& get_common_coreset(coreset_id cs_id) const
156 {
157 // The existence of the Coreset (either CommonCoreset or Coreset0) has been verified by the validator.
158 return dl_cfg_common.init_dl_bwp.pdcch_common.common_coreset.has_value() and
159 dl_cfg_common.init_dl_bwp.pdcch_common.common_coreset.value().id == cs_id
160 ? dl_cfg_common.init_dl_bwp.pdcch_common.common_coreset.value()
161 : dl_cfg_common.init_dl_bwp.pdcch_common.coreset0.value();
162 }
163
164private:
165 /// Vector circularly indexed by slot with the list of nof active DL/UL symbols per slot.
166 std::vector<unsigned> dl_symbols_per_slot_lst;
167 std::vector<unsigned> ul_symbols_per_slot_lst;
168};
