35/// Implements a PRACH buffer reader that contains always the same random data.
36class ru_dummy_rx_prach_buffer : private prach_buffer
37{
38public:
39 ru_dummy_rx_prach_buffer(unsigned seed, unsigned max_nof_ports)
40 {
41 // Resize PRACH buffer data tp the maximum buffer size.
42 data.resize({prach_constants::SHORT_SEQUENCE_LENGTH,
43 prach_constants::SHORT_SEQUENCE_MAX_NOF_SYMBOLS,
44 prach_constants::MAX_NOF_PRACH_FD_OCCASIONS,
45 prach_constants::MAX_NOF_PRACH_TD_OCCASIONS,
46 max_nof_ports});
47
48 // Fill PRACH buffer with random data.
49 complex_normal_distribution<cf_t> dist(0, 1);
50 std::mt19937 rgen(seed);
51 span<cf_t> data_view = data.get_data();
52 std::generate(data_view.begin(), data_view.end(), [&dist, &rgen]() { return dist(rgen); });
53 }
54
55 /// Gets the the PRACH buffer.
56 prach_buffer& get_buffer(const prach_buffer_context& context)
57 {
58 unsigned nof_symbols, sequence_length;
59
60 // Derive parameters from the PRACH format.
61 if (is_long_preamble(context.format)) {
62 prach_preamble_information info = get_prach_preamble_long_info(context.format);
63 nof_symbols = static_cast<unsigned>(info.symbol_length.to_seconds() * ra_scs_to_Hz(info.scs));
64 sequence_length = info.sequence_length;
65 } else {
66 prach_preamble_information info =
67 get_prach_preamble_short_info(context.format, to_ra_subcarrier_spacing(context.pusch_scs), false);
68 nof_symbols = static_cast<unsigned>(info.symbol_length.to_seconds() * ra_scs_to_Hz(info.scs));
69 sequence_length = info.sequence_length;
70 }
71
72 // Derive other parameters.
73 unsigned nof_ports = context.ports.size();
74 unsigned nof_td_occasions = context.nof_td_occasions;
75 unsigned nof_fd_occasions = context.nof_fd_occasions;
76
77 // Resize data to match context specifications.
78 if ((sequence_length != data.get_dimension_size(dims::re)) ||
79 (nof_symbols != data.get_dimension_size(dims::symbol)) ||
80 (nof_fd_occasions != data.get_dimension_size(dims::fd_occasion)) ||
81 (nof_td_occasions != data.get_dimension_size(dims::td_occasion)) ||
82 (nof_ports != data.get_dimension_size(dims::port))) {
83 data.resize({sequence_length, nof_symbols, nof_fd_occasions, nof_td_occasions, nof_ports});
84 }
85
86 return *this;
87 }
88
89private:
90 // See interface for documentation.
91 unsigned int get_max_nof_ports() const override { return data.get_dimension_size(dims::port); }
92
93 // See interface for documentation.
94 unsigned get_max_nof_td_occasions() const override { return data.get_dimension_size(dims::td_occasion); }
95
96 // See interface for documentation.
97 unsigned get_max_nof_fd_occasions() const override { return data.get_dimension_size(dims::fd_occasion); }
98
99 // See interface for documentation.
100 unsigned get_max_nof_symbols() const override { return data.get_dimension_size(dims::symbol); }
101
102 // See interface for documentation.
103 unsigned get_sequence_length() const override { return data.get_dimension_size(dims::re); }
104
105 // See interface for documentation.
106 span<cf_t> get_symbol(unsigned i_port, unsigned i_td_occasion, unsigned i_fd_occasion, unsigned i_symbol) override
107 {
108 srsran_assert(i_port < get_max_nof_ports(),
109 "The port index (i.e., {}) exceeds the maximum number of ports (i.e., {}).",
110 i_port,
111 get_max_nof_ports());
112 srsran_assert(i_td_occasion < get_max_nof_td_occasions(),
113 "The time-domain occasion (i.e., {}) exceeds the maximum number of time-domain occasions (i.e., {}).",
114 i_td_occasion,
115 get_max_nof_td_occasions());
116 srsran_assert(
117 i_fd_occasion < get_max_nof_fd_occasions(),
118 "The frequency-domain occasion (i.e., {}) exceeds the maximum number of frequency-domain occasions (i.e., {}).",
119 i_fd_occasion,
120 get_max_nof_fd_occasions());
121 srsran_assert(i_symbol < get_max_nof_symbols(),
122 "The symbol index (i.e., {}) exceeds the maximum number of symbols (i.e., {}).",
123 i_symbol,
124 get_max_nof_symbols());
125 return data.get_view({i_symbol, i_fd_occasion, i_td_occasion, i_port});
126 }

129 span<const cf_t>
130 get_symbol(unsigned i_port, unsigned i_td_occasion, unsigned i_fd_occasion, unsigned i_symbol) const override

134 i_port,

142 "The frequency-domain occasion (i.e., {}) exceeds the maximum number of frequency-domain occasions (i.e., {}).",
143 i_fd_occasion,

145 srsran_assert(i_symbol < get_max_nof_symbols(),
146 "The symbol index (i.e., {}) exceeds the maximum number of symbols (i.e., {}).",
147 i_symbol,
148 get_max_nof_symbols());
149 return data.get_view({i_symbol, i_fd_occasion, i_td_occasion, i_port});
150 }

152 /// Data storage dimensions.
153 enum class dims : unsigned {
154 re = 0,
155 symbol = 1,
156 fd_occasion = 2,
157 td_occasion = 3,
158 port = 4,
159 count = 5,
160 };
161
162 /// Data storage.
163 dynamic_tensor<static_cast<std::underlying_type_t<dims>>(dims::count), cf_t, dims> data;
164};
