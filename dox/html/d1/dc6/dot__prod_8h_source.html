38/// \brief Dot product of two spans.
39///
40/// Computes the dot product (a.k.a. inner product or scalar product) of the two sequences represented by the input
41/// spans, adding an initial offset.
42/// \tparam T A span of an arithmetic type.
43/// \tparam U A span of an arithmetic type.
44/// \tparam V Output type (must be compatible with the product of object of type \c T and \c U).
45/// \param[in] x First span.
46/// \param[in] y Second span.
47/// \param[in] init Initialization value.
48/// \return The dot product between the two spans plus \c init, i.e. \f$ x \cdot y + \textup{init} = \sum_i x_i y_i +
49/// \textup{init}\f$.
50/// \remark The two input spans must have the same length.
51template <typename T, typename U, typename V>
52inline V dot_prod(const T& x, const U& y, V init)
53{
54 static_assert(detail::is_arithmetic_span_compatible<T>::value,
55 "Template type is not compatible with a span of arithmetics");
56 static_assert(detail::is_arithmetic_span_compatible<U>::value,
57 "Template type is not compatible with a span of arithmetics");
58 srsran_srsvec_assert_size(x, y);
59 return std::inner_product(x.begin(), x.end(), y.begin(), init);
60}

62/// \brief Dot product of two complex spans.
63///
64/// Computes the dot product (a.k.a. inner product or scalar product) of the two complex sequences. The sequences are
65/// represented by the input spans \c x and \c y, the second sequence \c y is conjugated.
66///
67/// The implementation is equivalent to:
68/// \code
69/// cf_t dot_prod(span<const cf_t> x, span<const cf_t> y) {
70/// return std::inner_product(x.begin(), x.end(), y.begin(), cf_t(0.0F), std::plus<>(), [](cf_t a, cf_t b) { return a
71/// * std::conj(b); });
72/// }
73/// \endcode
74///
75/// \param[in] x First span.
76/// \param[in] y Second span.
77/// \return The dot product between the two spans, i.e. \f$ x \cdot \conj{y}= \sum_i x_i \conj{y}_i \f$.
78/// \remark The two input spans must have the same length.
79cf_t dot_prod(span<const cf_t> x, span<const cf_t> y);
80
81/// \brief Estimates the average power of a complex span - linear scale.
82///
83/// The average power of a span is defined as its squared Euclidean norm divided by the number of its elements, i.e.
84/// <tt>dot_prod(x, x) / x.size()</tt>.
85inline float average_power(span<const cf_t> x)
86{
87 if (x.empty()) {
88 return 0.0F;
89 }
90
91 return std::real(srsvec::dot_prod(x, x)) / static_cast<float>(x.size());
92}
