32/// Container that stores all scheduler UEs.
33class ue_repository
34{
35 using ue_list = slotted_id_table<du_ue_index_t, std::unique_ptr<ue>, MAX_NOF_DU_UES>;
36
37public:
38 using value_type = ue_list::value_type;
39 using iterator = ue_list::iterator;
40 using const_iterator = ue_list::const_iterator;
41
42 explicit ue_repository();
43
44 /// \brief Mark start of new slot and update UEs states.
45 void slot_indication(slot_point sl_tx);
46
47 /// \brief Contains ue index.
48 bool contains(du_ue_index_t ue_index) const { return ues.contains(ue_index); }
49
50 ue& operator[](du_ue_index_t ue_index) { return *ues[ue_index]; }
51 const ue& operator[](du_ue_index_t ue_index) const { return *ues[ue_index]; }
52
53 /// \brief Search UE context based on TC-RNTI/C-RNTI.
54 ue* find_by_rnti(rnti_t rnti);
55 const ue* find_by_rnti(rnti_t rnti) const;
56
57 /// \brief Add new UE in the UE repository.
58 void add_ue(std::unique_ptr<ue> u);
59
60 /// \brief Initiate removal of existing UE from the repository.
61 void schedule_ue_rem(ue_config_delete_event ev);
62
63 ue* find(du_ue_index_t ue_index) { return ues.contains(ue_index) ? ues[ue_index].get() : nullptr; }
64 const ue* find(du_ue_index_t ue_index) const { return ues.contains(ue_index) ? ues[ue_index].get() : nullptr; }
65
66 size_t size() const { return ues.size(); }
67
68 bool empty() const { return ues.empty(); }
69
70 iterator begin() { return ues.begin(); }
71 iterator end() { return ues.end(); }
72 const_iterator begin() const { return ues.begin(); }
73 const_iterator end() const { return ues.end(); }
74
75 const_iterator lower_bound(du_ue_index_t ue_index) const { return ues.lower_bound(ue_index); }
76
77private:
78 srslog::basic_logger& logger;
79
80 // Repository of UEs.
81 ue_list ues;
82
83 // Mapping of RNTIs to UE indexes.
84 std::vector<std::pair<rnti_t, du_ue_index_t>> rnti_to_ue_index_lookup;
85
86 // Queue of UEs marked for later removal.
87 ring_buffer<ue_config_delete_event> ues_to_rem{MAX_NOF_DU_UES};
88};
