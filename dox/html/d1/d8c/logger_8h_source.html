31/// The logger_impl template class contains the common functionality to loggers.
32/// Its main responsibility is to control the logging level by individually
33/// enabling or disabling the logging channels that form the logger.
34/// NOTE: Thread safe class.
35template <typename T, typename Enum>
36class logger_impl : public T
37{
38 static_assert(std::is_enum<Enum>::value, "Expected enum type");
39
40 using enum_base_type = typename std::underlying_type<Enum>::type;
41 static constexpr unsigned size = static_cast<enum_base_type>(Enum::LAST) - 1;
42
43public:
44 template <typename... Args>
45 explicit logger_impl(std::string id, Args&&... args) :
46 T{std::forward<Args>(args)...}, logger_id(std::move(id)), channels{&args...}
47 {
48 static_assert(sizeof...(args) == size, "Number of levels in enum does not match number of log channels");
49 }
50
51 logger_impl(const logger_impl& other) = delete;
52 logger_impl& operator=(const logger_impl& other) = delete;
53
54 /// Returns the id string of the logger.
55 const std::string& id() const { return logger_id; }
56
57 /// Change the logging level.
58 void set_level(Enum lvl)
59 {
60 detail::scoped_lock lock(m);
61
62 // Disable all channels ...
63 if (lvl == Enum::none) {
64 for (unsigned i = 0, e = channels.size(); i != e; ++i) {
65 channels[i]->set_enabled(false);
66 }
67 return;
68 }
69
70 // ... otherwise do it selectively.
71 for (unsigned i = 0, e = channels.size(); i != e; ++i) {
72 channels[i]->set_enabled(static_cast<Enum>(i + 1) <= lvl);
73 }
74 }
75
76 /// Set the specified context value to all the channels of the logger.
77 void set_context(uint32_t a, uint32_t b)
78 {
79 detail::scoped_lock lock(m);
80 for (auto channel : channels) {
81 channel->set_context(a, b);
82 }
83 }
84
85 /// Set the maximum number of bytes to can be printed in a hex dump to all the
86 /// channels of the logger.
87 void set_hex_dump_max_size(int x)
88 {
89 detail::scoped_lock lock(m);
90 for (auto channel : channels) {
91 channel->set_hex_dump_max_size(x);
92 }
93 }
94
95private:
96 /// Comparison operator for enum types, used by the set_level method.
97 friend bool operator<=(Enum lhs, Enum rhs)
98 {
99 return static_cast<enum_base_type>(lhs) <= static_cast<enum_base_type>(rhs);
100 }
101
102private:
103 const std::string logger_id;
104 const std::array<log_channel*, size> channels;
105 mutable detail::mutex m;
106};
107
108/// Type trait to detect if T is a logger.
109template <typename T>
110struct is_logger : std::false_type {
111};
112template <typename T, typename Enum>
113struct is_logger<logger_impl<T, Enum>> : std::true_type {
114};

118/// The build_logger_type type alias allows developers to define new logger
119/// types in an application.
120///
121/// To create a new logger type simply follow these steps:
122/// 1) Define an enum class where each element will represent a logging level.
123/// Order the elements from highest to lowest logging level. First element
124/// should be "none", which represents a disabled logger.
125/// The last element should be called LAST as it is a sentinel value.
126/// 2) Define a struct composed by only log_channel references. Declare the
127/// members in the same order as done in the enum.
128/// 3) Define the new logger type by using the build_logger_type alias. Pass
129/// the previous defined types as template parameters.
130///
131/// Example to declare a logger with three logging levels: error, warning and
132/// info, being error the highest logging level and info the lowest:
133/// 1) Define the logging level enum:
134/// enum class three_level_logger_levels { none, error, warning, info, LAST
135/// };
136/// 2) Define the struct of three channels (same order as in the enum):
137/// struct three_level_logger {
138/// log_channel &error;
139/// log_channel &warning;
140/// log_channel &info;
141/// };
142/// 3) Define the new logger type:
143/// using my_new_logger =
144/// build_logger_type<three_level_logger, three_level_logger_levels>;
145template <typename T, typename Enum>
146using build_logger_type = detail::logger_impl<T, Enum>;
147
148///
149/// Common logger types.
150///
151
152/// Basic logger with four levels.
153enum class basic_levels { none, error, warning, info, debug, LAST };
154struct basic_logger_channels {
155 log_channel& error;
156 log_channel& warning;
157 log_channel& info;
158 log_channel& debug;
159};
160using basic_logger = build_logger_type<basic_logger_channels, basic_levels>;
161
162/// Translates a string to the corresponding logger basic level.
163inline basic_levels str_to_basic_level(std::string s)
164{
165 std::transform(s.begin(), s.end(), s.begin(), ::toupper);
166
167 if ("NONE" == s) {
168 return basic_levels::none;
169 }
170 if ("ERROR" == s) {
171 return basic_levels::error;
172 }
173 if ("WARNING" == s) {
174 return basic_levels::warning;
175 }
176 if ("INFO" == s) {
177 return basic_levels::info;
178 }
179 if ("DEBUG" == s) {
180 return basic_levels::debug;
181 }
182 return basic_levels::none;
183}
184
185/// Translates a logger basic level to the corresponding string.
186inline const char* basic_level_to_string(basic_levels level)
187{
188 switch (level) {
189 case basic_levels::debug:
190 return "DEBUG";
191 case basic_levels::info:
192 return "INFO";
193 case basic_levels::warning:
194 return "WARNING";
195 case basic_levels::error:
196 return "ERROR";
197 default:
198 break;
199 }
200 return "NONE";
201}
