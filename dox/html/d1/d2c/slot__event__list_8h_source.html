31/// This class stores incoming events in a thread-safe queue, and pops all enqueued events in a single batch, when
32/// a new slot indication is notified.
33/// The advantage of this class over a mutexed queue is the lower contention caused when popping elements from the
34/// queue. This is particularly relevant because the events need to be processed during the scheduler slot handling.
35/// \tparam Event type of event pushed/popped from the list.
36template <typename Event>
37class slot_event_list
38{
39public:
40 void reserve(unsigned cap)
41 {
42 std::lock_guard<std::mutex> lock(mutex);
43 pending_events.reserve(cap);
44 current_events.reserve(cap);
45 }
46
47 void slot_indication()
48 {
49 current_events.clear();
50 std::lock_guard<std::mutex> lock(mutex);
51 pending_events.swap(current_events);
52 }
53
54 template <typename Ev>
55 void push(Ev&& ev)
56 {
57 static_assert(std::is_convertible<Ev, Event>::value, "Invalid type");
58 std::lock_guard<std::mutex> lock(mutex);
59 pending_events.push_back(std::forward<Ev>(ev));
60 }
61
62 template <typename... Args>
63 void emplace(Args&&... args)
64 {
65 std::lock_guard<std::mutex> lock(mutex);
66 pending_events.template emplace_back(std::forward<Args>(args)...);
67 }
68
69 span<Event> get_events() { return current_events; }
70
71private:
72 /// Stores all enqueued events that are going to be processed in the next slot_indication.
73 std::vector<Event> pending_events;
74
75 /// Contains the events being processed in the current slot.
76 /// Note: the transfer of next_events to current_events is done via a std::swap, which for std::vector is very fast.
77 std::vector<Event> current_events;
78
79 std::mutex mutex;
80};
