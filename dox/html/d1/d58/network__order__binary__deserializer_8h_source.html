33/// This class deserializes data from the configured memory buffer in network order and returns it in host endianness.
34class network_order_binary_deserializer
35{
36public:
37 explicit network_order_binary_deserializer(span<const uint8_t> buffer) : ptr(buffer.data()), size(buffer.size())
38 {
39 srsran_assert(ptr, "Invalid pointer to buffer");
40 }
41
42 /// \brief Deserializes a T from the buffer and advances the position by sizeof(T) bytes.
43 ///
44 /// \note This function only accepts POD types.
45 template <typename T>
46 T read()
47 {
48 static_assert(std::is_pod<T>::value, "Unsupported data type");
49
50 switch (sizeof(T)) {
51 case 1:
52 return read_one_byte();
53 case 2:
54 return read_two_bytes();
55 case 4:
56 return read_four_bytes();
57 default:
58 srsran_assert(0, "Deserializer does not support this data");
59 }
60 return T();
61 }
62
63 /// Deserializes a fixed amount of elements given by the size of the input span and advances the position by sizeof(T)
64 /// bytes for each element.
65 template <typename T>
66 void read(span<T> s)
67 {
68 for (auto& element : s) {
69 element = read<T>();
70 }
71 }
72
73 /// Deserializes a fixed amount of bytes given by the size of the input span and advances the position by the span
74 /// size.
75 void read(span<uint8_t> s)
76 {
77 std::memcpy(s.data(), ptr, s.size() * sizeof(uint8_t));
78 advance(s.size() * sizeof(uint8_t));
79 }
80
81 /// Advances the offset by the given amount.
82 void advance(unsigned x)
83 {
84 ptr += x;
85 offset += x;
86 }
87
88 /// Returns the current offset.
89 unsigned get_offset() const { return offset; }
90
91 /// Returns the number of unconsumed bytes of the buffer.
92 unsigned remaining_bytes() const { return size - offset; }
93
94private:
95 /// Deserializes one byte and advances the position by one byte.
96 uint8_t read_one_byte()
97 {
98 uint8_t value;
99 std::memcpy(&value, ptr, sizeof(value));
100 advance(sizeof(value));
101
102 return value;
103 }
104
105 /// Deserializes two bytes and advances the position by two bytes.
106 uint16_t read_two_bytes()
107 {
108 uint16_t value;
109 std::memcpy(&value, ptr, sizeof(value));
110 advance(sizeof(value));
111
112 return ntohs(value);
113 }
114
115 /// Deserializes four bytes and advances the position by four bytes.
116 uint32_t read_four_bytes()
117 {
118 uint32_t value;
119 std::memcpy(&value, ptr, sizeof(value));
120 advance(sizeof(value));
121
122 return ntohl(value);
123 }
124
125private:
126 const uint8_t* ptr;
127 const unsigned size;
128 unsigned offset = 0;
129};
