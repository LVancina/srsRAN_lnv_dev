31/// \brief Repository of processor pools.
32///
33/// Stores channel processor pools, organized according to the underlying numerology.
34/// \tparam T Type of the processor.
35template <typename T>
36class processor_pool_repository
37{
38 /// \brief Container of processors.
39 ///
40 /// Each entry corresponds to a different slot.
41 using processor_pool = std::vector<std::unique_ptr<T>>;
42
43 /// \brief Container of processor pools.
44 ///
45 /// Each entry corresponds to a different numerology.
46 std::array<processor_pool, NOF_NUMEROLOGIES> numerologies;
47
48public:
49 /// \brief Returns a reference to a processor using the given slot.
50 T& get_processor(slot_point slot)
51 {
52 unsigned numerology = slot.numerology();
53 srsran_assert(numerology < NOF_NUMEROLOGIES, "Invalid numerology ({}).", numerology);
54
55 processor_pool& pool = numerologies[numerology];
56 srsran_assert(!pool.empty(), "Numerology ({}) has no processors.", numerology);
57
58 unsigned index = slot.system_slot() % pool.size();
59 return *pool[index];
60 }
61
62 /// \brief Associates the given processor pool with this repository.
63 ///
64 /// \param[in] scs Subcarrier spacing.
65 /// \param[in] obj New processor pool for the given numerology.
66 void insert(subcarrier_spacing scs, processor_pool&& obj)
67 {
68 unsigned numerology = to_numerology_value(scs);
69 srsran_assert(numerology < NOF_NUMEROLOGIES, "Invalid numerology ({}).", numerology);
70
71 processor_pool& pool = numerologies[numerology];
72 srsran_assert(pool.empty(), "Numerology ({}) already has processors.", numerology);
73
74 numerologies[numerology] = std::move(obj);
75 }
76};
