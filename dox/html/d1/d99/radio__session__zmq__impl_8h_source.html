34/// Describes a ZeroMQ radio based session.
35class radio_session_zmq_impl : public radio_session, public radio_management_plane
36{
37private:
38 /// Default sockets send and receive timeout in milliseconds.
39 static constexpr unsigned DEFAULT_TRX_TIMEOUT_MS = 10;
40 /// Default linger timeout in milliseconds.
41 static constexpr unsigned DEFAULT_LINGER_TIMEOUT_MS = 0;
42 /// Default stream buffer size in samples.
43 static constexpr unsigned DEFAULT_STREAM_BUFFER_SIZE = 614400;
44
45 /// Radio session logger.
46 srslog::basic_logger& logger;
47 /// ZMQ context.
48 void* zmq_context;
49 /// Stores transmit streams.
50 std::vector<std::unique_ptr<radio_zmq_baseband_gateway>> bb_gateways;
51 /// Indicates the session has been created succesfully.
52 bool successful = false;
53
54public:
55 /// \brief Default constructor.
56 /// \param[in] config Provides the required parameters to start a ZMQ radio based session.
57 /// \param[in] async_task_executor Provides a task executor to perform asynchronous tasks.
58 /// \param[in] notification_handler Provides a radio event notification handler.
59 /// \note Use is_successful() to check that the instance was successfully initialized.
60 radio_session_zmq_impl(const radio_configuration::radio& config,
61 task_executor& async_task_executor,
62 radio_notification_handler& notification_handler);
63
64 /// Default destructor.
65 ~radio_session_zmq_impl();
66
67 /// Indicates if the instance was successfully initialized.
68 bool is_successful() const { return successful; }
69
70 // See interface for documentation.
71 radio_management_plane& get_management_plane() override { return *this; };
72
73 // See interface for documentation.
74 baseband_gateway& get_baseband_gateway(unsigned stream_id) override
75 {
76 srsran_assert(stream_id < bb_gateways.size(),
77 "Stream identifier (i.e., {}) exceeds the number of baseband gateways (i.e., {})",
78 stream_id,
79 bb_gateways.size());
80 return *bb_gateways[stream_id];
81 }
82
83 // See interface documentation.
84 baseband_gateway_timestamp read_current_time() override;
85
86 // See interface for documentation.
87 void start(baseband_gateway_timestamp init_time) override;
88
89 // See interface for documentation.
90 void stop() override;
91
92 // See interface for documentation.
93 bool set_tx_gain(unsigned port_id, double gain_dB) override;
94
95 // See interface for documentation.
96 bool set_rx_gain(unsigned port_id, double gain_dB) override;
97};
