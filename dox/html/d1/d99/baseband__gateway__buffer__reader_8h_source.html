30/// \brief Read-only baseband buffer interface.
31///
32/// It comprises get_nof_channels() channels of get_nof_samples() samples each.
33class baseband_gateway_buffer_reader
34{
35public:
36 virtual ~baseband_gateway_buffer_reader() = default;
37
38 /// Gets the number of channels.
39 virtual unsigned get_nof_channels() const = 0;
40
41 /// Gets the number of samples.
42 virtual unsigned get_nof_samples() const = 0;
43
44 /// \brief Gets a channel buffer.
45 /// \param[in] channel_idx Indicates the channel index.
46 /// \return A view to a baseband buffer.
47 /// \note An invalid channel index causes undefined behaviour.
48 virtual span<const cf_t> get_channel_buffer(unsigned channel_idx) const = 0;
49
50 /// See get_channel_buffer() documentation.
51 span<const cf_t> operator[](unsigned channel_index) const { return get_channel_buffer(channel_index); }
52};
