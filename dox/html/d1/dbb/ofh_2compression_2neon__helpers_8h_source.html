36/// \brief Determines BFP exponent for the resource block passed in a vector of three NEON registers.
37///
38/// The content of the three NEON registers is represented in the table below. Here REx stands for one component of
39/// a unique resource element (16 bits long):
40/// | | | | | |
41/// | ----- | ------- | ------- | --------- | ------- |
42/// | \c reg0: | RE0 RE0 | RE1 RE1 | RE2 RE2 | RE3 RE3 |
43/// | \c reg1: | RE4 RE4 | RE5 RE5 | RE6 RE6 | RE7 RE7 |
44/// | \c reg2: | RE8 RE8 | RE9 RE9 | RE10 RE10 | RE11 RE11 |
45///
46/// \param[in] vec_s16x3 Vector of three NEON registers storing 16bit IQ pairs of the resource block.
47/// \param[in] data_width Compressed samples bit width.
48///
49/// \return Calculated exponent value.
50inline uint8_t determine_bfp_exponent(int16x8x3_t vec_s16x3, unsigned data_width)
51{
52 // Max allowed exponent (accounts for the extra bit used for sign).
53 const uint16_t max_lzcnt = (16 - data_width) + 1;
54 uint16x4_t vec_max_lzc = vdup_n_u16(max_lzcnt);
55
56 // Calculate maximum value in the resource block (following instructions perform three pairwise comparisons between
57 // the three registers and then find a maximum value across resulting vector).
58 int16x8_t tmp_max_s16 = vmaxq_s16(vmaxq_s16(vec_s16x3.val[0], vec_s16x3.val[1]), vec_s16x3.val[2]);
59 int16_t max_s16 = vmaxvq_s16(tmp_max_s16);
60
61 // Calculate minimum value in the resource block (following instructions perform three pairwise comparisons between
62 // the three registers and then find a minimum value across resulting vector).
63 int16x8_t tmp_min_s16 = vminq_s16(vminq_s16(vec_s16x3.val[0], vec_s16x3.val[1]), vec_s16x3.val[2]);
64 int16_t min_s16 = vminvq_s16(tmp_min_s16);
65
66 // Calculate maximum absolute value.
67 uint16_t max_abs = std::max(std::abs(max_s16), std::abs(min_s16) - 1);
68
69 // Calculate number of leading zeros.
70 uint16x4_t lzc_u16 = vclz_u16(vdup_n_u16(max_abs));
71
72 // Calculate exponent.
73 return static_cast<uint8_t>(vget_lane_u16(vsub_u16(vec_max_lzc, lzc_u16), 0));
74}
