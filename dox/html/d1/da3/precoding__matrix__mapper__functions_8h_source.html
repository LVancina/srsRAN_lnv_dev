64/// Returns the precoding matrix index for the four-port PDSCH codebook using the given i_1_1, i_1_3 and i_1_2 PMI
65/// parameters.
66inline unsigned get_pdsch_four_port_precoding_matrix_index(unsigned i_1_1, unsigned i_1_3, unsigned i_2)
67{
68 static constexpr unsigned MAX_NUM_I_1_3 = 2U;
69 static constexpr unsigned MAX_NUM_I_2 = 4U;
70
71 return MAX_NUM_I_2 * MAX_NUM_I_1_3 * i_1_1 + MAX_NUM_I_2 * i_1_3 + i_2;
72}
