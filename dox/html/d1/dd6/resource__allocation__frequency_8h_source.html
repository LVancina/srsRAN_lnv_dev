27/// \brief Describes the resource allocation in frequency domain for Type1 as per TS38.214 Section 5.1.2.2.2 for
28/// Downlink and Section 6.1.2.2 for Uplink.
29///
30/// In downlink resource allocation of type 1, the resource block assignment information indicates to a scheduled UE a
31/// set of contiguously allocated non-interleaved or interleaved virtual resource blocks.
32///
33/// In uplink resource allocation of type 1, the resource block assignment information indicates to a scheduled UE a
34/// set of contiguously allocated non-interleaved virtual resource blocks.
35struct ra_frequency_type1_configuration {
36 /// \brief BWP size in resource blocks, parameter \f$N_{BWP}^{size}\f$.
37 ///
38 /// For downlink, the value must be set to:
39 /// - the size of CORESET0 if
40 /// - the DCI is transmitted in a common SS, and
41 /// - CORESET0 is configured for the cell;
42 /// - the size of the initial BWP if
43 /// - the DCI is transmitted in a common SS, and
44 /// - CORESET0 is not configured for the cell;
45 /// - the active BWP size otherwise.
46 ///
47 /// For uplink, the value must be set to:
48 /// - the size of the initial BWP \f$N_{BWP,0}^{size}\f$ if the DCI is transmitted in a common SS;
49 /// - the active BWP size otherwise.
50 unsigned N_bwp_size;
51 /// Start VRB index \f$RB_{start}\f$.
52 unsigned start_vrb;
53 /// Length in terms of contiguously allocated resource blocks \f$L_{RBs}\f$.
54 unsigned length_vrb;
55};
56
57/// \brief Calculates the resource indication value \f$RIV\f$ as per TS38.214 Section 5.1.2.2.2 for
58/// Downlink and Section 6.1.2.2 for Uplink.
59///
60/// if \f$(L_{RB} - 1) \leq \left \lfloor N_{BWP}^{size}/2 \right \rfloor\f$, then <br>
61/// \f$RIV = N_{BWP}^{size}(L_{RBs}-1)+RB_{start}\f$ <br>
62/// Otherwise, <br>
63/// \f$RIV = N_{BWP}^{size}(N_{BWP}^{size}-L_{RBs}+1)+(N_{BWP}^{size}-1-RB_{start})\f$.
64unsigned ra_frequency_type1_get_riv(const ra_frequency_type1_configuration& config);
65
66/// \brief Calculates the PRBs from \f$RIV\f$ as per TS 38.214 Section 5.1.2.2.2 for Downlink and
67/// Section 6.1.2.2 for Uplink.
68ra_frequency_type1_configuration ra_frequency_type1_from_riv(unsigned N_bwp_size, unsigned riv);

70/// \brief Describes the resource allocation in frequency domain for Type1 special case as per TS38.214
71/// Section 5.1.2.2.2 for Downlink and Section 6.1.2.2 for Uplink.
72///
73/// The special case in downlink is when the DCI size for DCI format 1_0 in USS is derived from the size of DCI format
74/// 1_0 in CSS but applied to an active BWP.
75///
76/// The special case in uplink is when the DCI size for DCI format 0_0 in USS is derived from the initial UL BWP but
77/// applied to another active BWP.
78///
79/// This special cases occur when step 3 in TS38.212 Section 7.3.1.0 fails. In this case, DCI formats 0_0 and 1_0 in USS
80/// are derived as they were in CSS (step 4).
81struct ra_frequency_type1_special_configuration {
82 /// \brief Initial BWP size in resource blocks, parameter \f$N_{BWP}^{initial}\f$.
83 ///
84 /// For downlink, the value must be set to:
85 /// - the size of CORESET 0 if CORESET 0 is configured for the cell;
86 /// - the size of initial DL bandwidth part otherwise.
87 ///
88 /// For uplink, the value must be set to the initial bandwidth part size \f$N_{BWP,0}^{size}\f$.
89 unsigned N_bwp_initial;
90 /// Active BWP size in resource blocks, parameter \f$N_{BWP}^{active}\f$.
91 unsigned N_bwp_active;
92 /// Start VRB index \f$RB_{start}\f$.
93 unsigned start_vrb;
94 /// Length in terms of contiguously allocated resource blocks \f$L_{RBs}\f$.
95 unsigned length_vrb;
96};
97
98/// \brief Calculates the resource indication value \f$RIV\f$ for special cases as per TS38.214 Section 5.1.2.2.2 for
99/// Downlink and Section 6.1.2.2 for Uplink.
100///
101/// if \f$(L'_{RB} - 1) \leq \left \lfloor N_{BWP}^{initial}/2 \right \rfloor\f$, then <br>
102/// \f$RIV = N_{BWP}^{initial}(L'_{RBs}-1)+RB'_{start}\f$ <br>
103/// Otherwise, <br>
104/// \f$RIV = N_{BWP}^{initial}(N_{BWP}^{initial}-L'_{RBs}+1)+(N_{BWP}^{initial}-1-RB'_{start})\f$. <br>
105/// Where \f$L'_{RB}=L_{RB}/K\f$, \f$RB'_{start}=RB_{start}/K\f$. <br>
106/// if \f$N_{BWP}^{active}>N_{BWP}^{initial}\f$ then \f$K\f$ is the maximum value from set {1,2,4,8} which satisfies
107/// \f$K \leq \left \lfloor N_{BWP}^{active}/ N_{BWP}^{initial}\right \rfloor\f$; otherwise \f$K=1\f$.
108unsigned ra_frequency_type1_special_get_riv(const ra_frequency_type1_special_configuration& config);
