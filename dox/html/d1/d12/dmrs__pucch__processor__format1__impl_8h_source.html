33/// Generic implementation of a DM-RS channel estimator for PUCCH Format 1.
34class dmrs_pucch_processor_format1_impl : public dmrs_pucch_processor

36public:
37 using layer_dmrs_pattern = port_channel_estimator::layer_dmrs_pattern;
38
39 dmrs_pucch_processor_format1_impl(std::unique_ptr<pseudo_random_generator> prg_,
40 std::unique_ptr<const low_papr_sequence_collection> sequence_collection_,
41 std::unique_ptr<port_channel_estimator> ch_est_) :
42 helper(std::move(prg_)), sequence_collection(std::move(sequence_collection_)), ch_estimator(std::move(ch_est_))

54 /// PUCCH Format 1 implementation expects pre-generated sequence collection on instantiation.

59 /// Buffer for DM-RS symbols.

66 // Sequence group.
67 unsigned u;
68 // Sequence number.
69 unsigned v;

74 };

76 /// \brief Generates a DM-RS sequence according to TS38.211 Section 6.4.1.3.1.1.
77 ///

84 const sequence_generation_config& gen_config,

87 /// \brief Generates the PUCCH DM-RS allocation pattern.

93};
