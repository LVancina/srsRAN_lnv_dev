32/// \brief Determine CRB limits for PUSCH grant, based on BWP configs, SearchSpace type and DCI format as per
33/// TS38.214, 6.1.2.2.2.
34///
35/// \param dci_type DL DCI format.
36/// \param init_ul_bwp Initial UL BWP configuration.
37/// \param active_ul_bwp Active UL BWP configuration.
38/// \param is_common_ss Flag indicating Common SearchSpace type or not.
39/// \return Calculated CRB limits.
40inline crb_interval get_ra_crb_limits(dci_ul_rnti_config_type dci_type,
41 const bwp_configuration& init_ul_bwp,
42 const bwp_configuration& active_ul_bwp,
43 bool is_common_ss)
44{
45 // See TS 38.214, 6.1.2.2.2, Uplink resource allocation type 1.
46 if (dci_type == dci_ul_rnti_config_type::c_rnti_f0_0 and is_common_ss) {
47 return init_ul_bwp.crbs;
48 }
49 return active_ul_bwp.crbs;
50}
