29/// Checks if alignment is power of 2.
30constexpr bool is_alignment_valid(std::size_t alignment)
31{
32 return alignment > 0 && (alignment & (alignment - 1U)) == 0U;
33}
34
35/// Checks whether the pointer is set with the correct alignment.
36constexpr inline bool is_aligned(std::uintptr_t ptr, std::size_t alignment)
37{
38 return (ptr & (alignment - 1)) == 0;
39}
40inline bool is_aligned(void* ptr, std::size_t alignment)
41{
42 return is_aligned(reinterpret_cast<std::uintptr_t>(ptr), alignment);
43}
44
45/// \brief Moves the pointer by the given size in bytes.
46constexpr inline void* advance_ptr(void* pos, std::size_t sz)
47{
48 return static_cast<char*>(pos) + sz;
49}
50
51/// Moves the pointer to the next aligned position.
52constexpr inline std::uintptr_t align_next(std::uintptr_t pos, std::size_t alignment)
53{
54 return (pos + (alignment - 1)) & ~(alignment - 1);
55}
56inline void* align_next(void* pos, std::size_t alignment)
57{
58 return reinterpret_cast<void*>(align_next(reinterpret_cast<std::uintptr_t>(pos), alignment));
59}
