33/// Adapter between CU-CP and NGAP, to initialize connection procedures
34class cu_cp_ngap_adapter : public cu_cp_ngap_control_notifier, public cu_cp_ngap_statistics_notifier
35{
36public:
37 void connect_ngap(ngap_connection_manager& ngap_conn_mng_,
38 ngap_ue_context_removal_handler& ngap_ue_handler_,
39 ngap_statistics_handler& ngap_statistic_handler_)
40 {
41 ngap_conn_mng = &ngap_conn_mng_;
42 ngap_ue_handler = &ngap_ue_handler_;
43 ngap_statistic_handler = &ngap_statistic_handler_;
44 }
45
46 async_task<ngap_ng_setup_result> on_ng_setup_request(const ngap_ng_setup_request& request) override
47 {
48 srsran_assert(ngap_conn_mng != nullptr, "NGAP handler must not be nullptr");
49 return ngap_conn_mng->handle_ng_setup_request(request);
50 }
51
52 void remove_ue(ue_index_t ue_index) override
53 {
54 srsran_assert(ngap_ue_handler != nullptr, "NGAP handler must not be nullptr");
55 return ngap_ue_handler->remove_ue_context(ue_index);
56 }
57
58 size_t get_nof_ues() const override
59 {
60 srsran_assert(ngap_statistic_handler != nullptr, "NGAP statistics handler must not be nullptr");
61 return ngap_statistic_handler->get_nof_ues();
62 }
63
64private:
65 ngap_connection_manager* ngap_conn_mng = nullptr;
66 ngap_ue_context_removal_handler* ngap_ue_handler = nullptr;
67 ngap_statistics_handler* ngap_statistic_handler = nullptr;
68};
