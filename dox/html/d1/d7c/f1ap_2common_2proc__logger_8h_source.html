61 template <typename FormatContext>
62 auto format(const srsran::f1ap_common_log_prefix& prefix, FormatContext& ctx)
63 {
64 bool first_id = true;
65 auto get_sep = [&first_id]() { return std::exchange(first_id, false) ? "" : " "; };
66 if (prefix.du_ue_id != srsran::gnb_du_ue_f1ap_id_t::invalid) {
67 format_to(ctx.out(), "{}GNB-DU-UE-F1AP-ID={}", get_sep(), prefix.du_ue_id);
68 }
69 if (prefix.cu_ue_id != srsran::gnb_cu_ue_f1ap_id_t::invalid) {
70 format_to(ctx.out(), "{}GNB-CU-UE-F1AP-ID={}", get_sep(), prefix.cu_ue_id);
71 }
72 if (prefix.proc_name != nullptr) {
73 format_to(ctx.out(), "{}proc=\"{}\"", get_sep(), prefix.proc_name);
74 }
75 return ctx.out();
76 }
