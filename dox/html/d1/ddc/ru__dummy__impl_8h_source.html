48/// \brief Implements a dummy radio unit.
49///
50/// The dummy radio unit is a radio unit designed for performance and stability testing purposes. It does not connect to
51/// any RF device.
52///
53/// It determines the timing from the system time.
54class ru_dummy_impl : public radio_unit,
55 private ru_controller,
56 private ru_downlink_plane_handler,
57 private ru_uplink_plane_handler
58{
59public:
60 /// Creates a dummy radio unit from its configuration.
61 explicit ru_dummy_impl(const ru_dummy_configuration& config, ru_dummy_dependencies dependencies) noexcept;
62
63 // See radio_unit interface for documentation.
64 ru_controller& get_controller() override { return *this; }
65
66 // See radio_unit interface for documentation.
67 ru_downlink_plane_handler& get_downlink_plane_handler() override { return *this; }
68
69 // See radio_unit interface for documentation.
70 ru_uplink_plane_handler& get_uplink_plane_handler() override { return *this; }
71
72private:
73 /// Possible internal states.
74 enum class state : uint8_t { idle = 0, running, wait_stop, stopped };
75 /// Minimum loop time.
76 const std::chrono::microseconds minimum_loop_time = std::chrono::microseconds(10);
77
78 // See ru_controller for documentation.
79 void start() override;
80
81 // See ru_controller for documentation.
82 void stop() override;
83
84 // See ru_downlink_plane_handler for documentation.
85 void handle_dl_data(const resource_grid_context& context, const resource_grid_reader& grid) override
86 {
87 interval<unsigned> sector_range(0, sectors.size());
88 srsran_assert(sector_range.contains(context.sector),
89 "Sector identifier (i.e., {}) is out-of-bounds {}.",
90 context.sector,
91 sector_range);
92 sectors[context.sector].handle_dl_data(context, grid);
93 }
94
95 // See ru_uplink_plane_handler for documentation.
96 void handle_prach_occasion(const prach_buffer_context& context, prach_buffer& buffer) override
97 {
98 interval<unsigned> sector_range(0, sectors.size());
99 srsran_assert(sector_range.contains(context.sector),
100 "Sector identifier (i.e., {}) is out-of-bounds {}.",
101 context.sector,
102 sector_range);
103 sectors[context.sector].handle_prach_occasion(context, buffer);
104 }
105
106 // See ru_uplink_plane_handler for documentation.
107 void handle_new_uplink_slot(const resource_grid_context& context, resource_grid& grid) override
108 {
109 interval<unsigned> sector_range(0, sectors.size());
110 srsran_assert(sector_range.contains(context.sector),
111 "Sector identifier (i.e., {}) is out-of-bounds {}.",
112 context.sector,
113 sector_range);
114 sectors[context.sector].handle_new_uplink_slot(context, grid);
115 }
116
117 /// Loop execution task.
118 void loop();
119
120 /// Ru logger.
121 srslog::basic_logger& logger;
122 /// Internal executor.
123 task_executor& executor;
124 /// Radio Unit timing notifier.
125 ru_timing_notifier& timing_notifier;
126 /// Internal state.
127 std::atomic<state> current_state = {state::idle};
128 /// Slot time in microseconds.
129 std::chrono::microseconds slot_duration;
130 /// Number of slots is notified in advance of the transmission time.
131 unsigned max_processing_delay_slots;
132 /// Current slot.
133 slot_point current_slot;
134 /// Radio unit sectors.
135 std::vector<ru_dummy_sector> sectors;
136};
