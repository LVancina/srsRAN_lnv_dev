32template <typename Callable>
33struct scope_exit {
34 template <typename C>
35 explicit scope_exit(C&& f_) : exit_function(std::forward<C>(f_))
36 {
37 }
38 scope_exit(scope_exit&& rhs) noexcept : exit_function(std::move(rhs.exit_function)), active(rhs.active)
39 {
40 rhs.release();
41 }
42 scope_exit(const scope_exit&) = delete;
43 scope_exit& operator=(const scope_exit&) = delete;
44 scope_exit& operator=(scope_exit&&) noexcept = delete;
45 ~scope_exit()
46 {
47 if (active) {
48 exit_function();
49 }
50 }
51
52 void release() { active = false; }
53
54private:
55 Callable exit_function;
56 bool active = true;
57};
58
59} // namespace detail

61/// \brief Defers callable call to scope exit
62/// \tparam Callable Any type with a call operator
63/// \param callable function that is called at scope exit
64/// \return object that has to be stored in a local variable
65template <typename Callable>
66detail::scope_exit<std::decay_t<Callable>> make_scope_exit(Callable&& callable)
67{
68 return detail::scope_exit<std::decay_t<Callable>>{std::forward<Callable>(callable)};
69}
