49 void compute_group_sequence(pucch_group_hopping group_hopping, unsigned n_id, unsigned& u, unsigned& v)
50 {
51 unsigned f_gh = 0;
52 unsigned f_ss = 0;
53
54 switch (group_hopping) {
55 case pucch_group_hopping::NEITHER:
56 f_ss = n_id % low_papr_sequence_collection::NOF_GROUPS;
57 break;
58 case pucch_group_hopping::ENABLE:
59 srsran_terminate("Group hopping is not implemented");
60 return;
61 case pucch_group_hopping::DISABLE:
62 srsran_terminate("Hopping is not implemented");
63 return;
64 }
65
66 u = (f_gh + f_ss) % low_papr_sequence_collection::NOF_GROUPS;
67 v = 0;
68 }

79 unsigned get_alpha_index(const slot_point& slot,
80 const cyclic_prefix& cp,
81 unsigned n_id,
82 unsigned symbol,
83 unsigned m0,
84 unsigned m_cs) const
85 {
86 // Initialize pseudo-random sequence with the seed set to nid
87 unsigned cinit = n_id;
88
89 // Pseudo-random sequence length for max numerology (i.e. 4)
90 const size_t max_seq_length = 8U * MAX_NSYMB_PER_SLOT * NOF_SUBFRAMES_PER_FRAME * (1U << 4U);
91 unsigned sequence_length = 8U * get_nsymb_per_slot(cp) * slot.nof_slots_per_frame();
92
93 // Create a zero array
94 static_bit_buffer<max_seq_length> cs(sequence_length);
95
96 prg->init(cinit);
97 prg->generate(cs);
98
99 // Slot number
100 unsigned n_slot = slot.slot_index();
101
102 // Create n_cs parameter
103 unsigned base_idx = get_nsymb_per_slot(cp) * n_slot + symbol;
104 unsigned n_cs = reverse_byte(cs.get_byte(base_idx));
105
106 unsigned alpha_idx = (m0 + m_cs + n_cs) % NRE;
107 return alpha_idx;
108 }
109};
