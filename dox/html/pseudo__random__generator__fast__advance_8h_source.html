36template <unsigned MaxNofStep>
37class pseudo_random_generator_fast_advance
38{
39private:
43 const pseudo_random_initializer_x1 x1_init;
47 const pseudo_random_initializer_x2 x2_init;
51 const pseudo_random_generator_fast_advance<MaxNofStep / 2> advance_half;
52
53public:
55 pseudo_random_generator_fast_advance() : x1_init(MaxNofStep), x2_init(MaxNofStep), advance_half() {}
56
62 void advance(uint32_t& x1_state, uint32_t& x2_state, unsigned count) const
63 {
64 if (count >= MaxNofStep) {
65 x1_state = x1_init.get(x1_state);
66 x2_state = x2_init.get(x2_state);
67 count -= MaxNofStep;
68 }
69
70 if (count == 0) {
71 return;
72 }
73
74 advance_half.advance(x1_state, x2_state, count);
75 }
76};
77

82public:
83 void advance(uint32_t& /* x1_state */, uint32_t& /* x2_state */, unsigned count) const
84 {
85 srsran_assert(count == 0, "Count must be zero.");
86 }
87};
