70 void set_error_handler(error_handler new_err_handler)
71 {
72 // Ignore new handlers when the worker is running.
73 if (is_running()) {
74 return;
75 }
76
77 // Install a dummy callback if the input one is empty. This avoids having to
78 // check for null at each call site.
79 if (!new_err_handler) {
80 err_handler = [](const std::string&) {};
81 return;
82 }
83
84 err_handler = std::move(new_err_handler);
85 }

105 bool report_queue_on_full_once()
106 {
107 if (queue.is_almost_full() && !is_err_handler_empty) {
108 err_handler(fmt::format("The backend queue size is about to reach its maximum "
109 "capacity of {} elements, new log entries will get "
110 "discarded.\nConsider increasing the queue capacity.",
111 queue.get_capacity()));
112 err_handler = [](const std::string&) {};
113 is_err_handler_empty = true;
114 return true;
115 }
116
117 return false;
118 }
