33inline unsigned
34get_biggest_min_tx_parameter_in_symbols(const du_tx_window_timing_parameters& tx_timing_params,
35 const std::chrono::duration<double, std::nano>& symbol_duration_ns)
36{
37 auto max_value =
38 std::max({tx_timing_params.T1a_min_cp_dl, tx_timing_params.T1a_min_cp_ul, tx_timing_params.T1a_min_up});
39
40 return std::floor(max_value / symbol_duration_ns);
41}

44inline unsigned calculate_nof_symbols_before_ota(cyclic_prefix cp,
45 subcarrier_spacing scs,
46 std::chrono::microseconds dl_processing_time,
47 const du_tx_window_timing_parameters tx_timing_params)
48{
49 unsigned nof_symbols = get_nsymb_per_slot(cp);
50 auto symbol_duration_ns =
51 std::chrono::duration<double, std::nano>(1e6 / (nof_symbols * get_nof_slots_per_subframe(scs)));
52 unsigned dl_processing_time_in_symbols = std::floor(dl_processing_time / symbol_duration_ns);
53
54 return dl_processing_time_in_symbols + get_biggest_min_tx_parameter_in_symbols(tx_timing_params, symbol_duration_ns);
55}
