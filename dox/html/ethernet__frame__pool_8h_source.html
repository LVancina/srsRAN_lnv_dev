42class frame_buffer
43{
45 enum class frame_buffer_status { free, reserved, used, marked_to_send };
46
47 size_t sz = 0;
48 frame_buffer_status status = frame_buffer_status::free;
49 std::vector<uint8_t> buffer;
50
51 friend class frame_buffer_array;
52
53public:
55 frame_buffer() = default;
56 explicit frame_buffer(unsigned size) : buffer(size, 0) {}
57
58 constexpr bool empty() const noexcept { return sz == 0; }
59
60 constexpr size_t size() const noexcept { return sz; }
61
62 constexpr void set_size(size_t new_size) noexcept
63 {
64 srsran_assert(new_size <= MAX_ETH_FRAME_LENGTH, "Can't set_size: new size if out-of-bounds");
65 unsigned written_size = (new_size < MIN_ETH_FRAME_LENGTH) ? MIN_ETH_FRAME_LENGTH : new_size;
66 if (new_size < MIN_ETH_FRAME_LENGTH) {
67 std::fill(buffer.begin() + new_size, buffer.begin() + MIN_ETH_FRAME_LENGTH, 0);
68 }
69 sz = written_size;
70 }
71
72 constexpr void clear() noexcept { sz = 0; }
73
74 span<uint8_t> data() noexcept { return {buffer}; }
75 span<const uint8_t> data() const noexcept { return {buffer.data(), sz}; }
76};

101class frame_buffer_array
102{
104 using storage_array_type = std::vector<frame_buffer>;
105
108 static constexpr size_t NUM_OF_ENTRIES = 2;
109
111 struct rd_wr_counter {
112 unsigned count = 0;
113 unsigned boundary = 0;
114 // Constructor receives upper boundary.
115 explicit rd_wr_counter(unsigned b) : boundary(b) {}
116 // Increment by N positions and wrap-around if needed
117 void increment(unsigned n = 1) { count += n; }
118 // Return counter value.
119 unsigned value() const { return count % boundary; }
120 };
121
122public:
123 struct used_buffer {
124 frame_buffer* buffer;
125 ofh::slot_symbol_point timestamp;
126 };
127
128 // Constructor receives number of buffers stored/read at a time, reserves storage for all eAxCs.
129 frame_buffer_array(unsigned nof_buffers_to_return, unsigned buffer_size, unsigned nof_antennas) :
130 increment_quant(nof_buffers_to_return),
131 storage_nof_buffers(nof_buffers_to_return * nof_antennas * NUM_OF_ENTRIES),
132 buffers_array(storage_nof_buffers, frame_buffer{buffer_size}),
133 write_position(storage_nof_buffers)
134 {
135 aux_array.reserve(storage_nof_buffers);
136 }
137
138 // Returns view over increment_quant buffers for writing if they are free, returns empty span otherwise.
139 // The state of returned buffers is changed to 'reserved'.
140 span<frame_buffer> reserve_buffers()
141 {
142 span<frame_buffer> wr_buffers(&buffers_array[write_position.value()], increment_quant);
143
144 bool free = std::all_of(wr_buffers.begin(), wr_buffers.end(), [](const frame_buffer& buffer) {
145 return buffer.status == frame_buffer::frame_buffer_status::free;
146 });
147 if (!free) {
148 return {};
149 }
150 // Mark buffers as reserved.
151 for (auto& buffer : wr_buffers) {
152 buffer.status = frame_buffer::frame_buffer_status::reserved;
153 }
154 write_position.increment(increment_quant);
155 return wr_buffers;
156 }
157
158 // Stores actually used buffers in a list of buffers ready for sending.
159 // Unused buffers state is changed to 'free'.
160 void push_buffers(span<frame_buffer> prepared_buffers, ofh::slot_symbol_point symbol_point)
161 {
162 for (auto& buffer : prepared_buffers) {
163 if (buffer.empty()) {
164 buffer.status = frame_buffer::frame_buffer_status::free;
165 } else {
166 buffer.status = frame_buffer::frame_buffer_status::used;
167 used_buffers.push_back({&buffer, symbol_point});
168 }
169 }
170 }
171
172 // Changed state of sent buffers to 'free'.
173 void clear_buffers()
174 {
175 for (auto& buffer : buffers_array) {
176 if (buffer.status == frame_buffer::frame_buffer_status::marked_to_send) {
177 buffer.status = frame_buffer::frame_buffer_status::free;
178 }
179 }
180 }
181
182 // Changed state of all 'used' buffers to 'free'.
183 void reset_buffers()
184 {
185 for (auto& buffer : buffers_array) {
186 if (buffer.status != frame_buffer::frame_buffer_status::marked_to_send) {
187 buffer.status = frame_buffer::frame_buffer_status::free;
188 }
189 }
190 }
191
192 // Returns a vector of pointers to the buffers ready for sending.
193 span<const frame_buffer*> find_buffers_ready_for_sending()
194 {
195 aux_array.clear();
196 for (auto& used_buf : used_buffers) {
197 used_buf.buffer->status = frame_buffer::frame_buffer_status::marked_to_send;
198 aux_array.push_back(used_buf.buffer);
199 }
200 used_buffers.clear();
201 return aux_array;
202 }
203
204 span<const used_buffer> get_prepared_buffers() const { return used_buffers; }
205
206private:
207 // Number of buffers accessed at a time.
208 unsigned increment_quant;
209 // Maximum number of buffers stored in the pool.
210 unsigned storage_nof_buffers;
211 // Data buffers.
212 storage_array_type buffers_array;
213 // Used buffers are added to this list.
214 static_vector<used_buffer, 128> used_buffers;
215 // Auxiliary array used as a list of ready-to-send buffers returned to a reader.
216 std::vector<const frame_buffer*> aux_array;
217 // Keeps track of the current write position.
218 rd_wr_counter write_position;
219};

234 struct pool_entry {
236 const size_t NUM_CP_MESSAGES_TO_RETURN = 1;
237
239 std::vector<frame_buffer_array> buffers;
240
241 pool_entry(units::bytes mtu, unsigned num_of_frames)
242 {
243 // DL C-Plane storage.
244 buffers.emplace_back(NUM_CP_MESSAGES_TO_RETURN, mtu.value(), ofh::MAX_NOF_SUPPORTED_EAXC);
245 // UL C-Plane storage.
246 buffers.emplace_back(NUM_CP_MESSAGES_TO_RETURN, mtu.value(), ofh::MAX_NOF_SUPPORTED_EAXC * 2);
247 // U-Plane storage.
248 buffers.emplace_back(num_of_frames, mtu.value(), ofh::MAX_NOF_SUPPORTED_EAXC);
249 }
250
252 frame_buffer_array& get_ofh_type_buffers(ofh::message_type type, ofh::data_direction dir)
253 {
254 unsigned index = static_cast<unsigned>(type) * 2;
255 if (dir == ofh::data_direction::uplink) {
256 index += 1;
257 }
258 return buffers[index];
259 }

262 const frame_buffer_array& get_ofh_type_buffers(ofh::message_type type, ofh::data_direction dir) const
263 {
264 unsigned index = static_cast<unsigned>(type) * 2;
265 if (dir == ofh::data_direction::uplink) {
266 index += 1;

272 span<frame_buffer> reserve_buffers(const ofh_pool_message_type& context)
273 {

364 span<const frame_buffer*> read_frame_buffers(const frame_pool_interval& interval)
365 {
366 if (interval.start > interval.end) {
367 return {};
368 }
369 // Acquire lock before accessing pool entries.
370 std::lock_guard<std::mutex> lock(mutex);
371
372 aux_array.clear();
373
374 // Extra 1 is added to include the end symbol of the interval.
375 unsigned distance = (interval.end - interval.start) + 1;
376 for (unsigned i = 0; i != distance; ++i) {
377 ofh::slot_symbol_point tmp_symbol = interval.start + i;
378
379 pool_entry& p_entry = get_pool_entry(tmp_symbol.get_slot(), tmp_symbol.get_symbol_index());
380 auto symbol_buffers = p_entry.read_buffers(interval.type);
381 for (const auto& buffer : symbol_buffers) {
382 aux_array.push_back(buffer);
383 }
384 }
385 return aux_array;
386 }

417 void clear_downlink_slot(slot_point slot_point, srslog::basic_logger& logger)
418 {
419 // Lock before changing the pool entries.
420 std::lock_guard<std::mutex> lock(mutex);
421
422 pool_entry& cp_entry = get_pool_entry(slot_point, 0);
423 // Clear buffers with DL Control-Plane messages.
424 ofh_pool_message_type msg_type{ofh::message_type::control_plane, ofh::data_direction::downlink};
425
426 auto dl_cp_buffers = cp_entry.get_prepared_buffers(msg_type);
427 for (const auto& used_buf : dl_cp_buffers) {
428 if (used_buf.timestamp.get_slot() == slot_point) {
429 continue;
430 }
431 logger.warning("Detected '{}' late downlink C-Plane messages in the transmitter queue for slot '{}'",
432 dl_cp_buffers.size(),
433 used_buf.timestamp.get_slot());
434 cp_entry.reset_buffers(msg_type);
435 break;
436 }
437
438 // Clear buffers with User-Plane messages.
439 msg_type.type = ofh::message_type::user_plane;
440 msg_type.direction = ofh::data_direction::downlink;
441 for (unsigned symbol = 0; symbol != NOF_OFDM_SYM_PER_SLOT_NORMAL_CP; ++symbol) {
442 pool_entry& up_entry = get_pool_entry(slot_point, symbol);
443
444 auto dl_up_buffers = up_entry.get_prepared_buffers(msg_type);
445 for (const auto& used_buf : dl_up_buffers) {
446 if (used_buf.timestamp.get_slot() == slot_point) {
447 continue;
448 }
449 logger.warning(
450 "Detected '{}' late downlink U-Plane messages in the transmitter queue for slot '{}', symbol '{}'",
451 dl_up_buffers.size(),
452 used_buf.timestamp.get_slot(),
453 used_buf.timestamp.get_symbol_index());
454 up_entry.reset_buffers(msg_type);
455 break;
456 }
457 }
458 }
