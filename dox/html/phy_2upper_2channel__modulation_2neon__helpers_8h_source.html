46inline float32x4_t copysign_f32(float32x4_t value0, float32x4_t value1)
47{
48 float32x4_t abs_value0 = vabsq_f32(value0);
49 float32x4_t neg_value0 = vnegq_f32(abs_value0);
50 // uint32x4_t sign_mask = vtstq_s32(vreinterpretq_s32_f32(value1), vdupq_n_s32(0x80000000));
51 uint32x4_t sign_mask = vreinterpretq_u32_s32(vshrq_n_s32(vreinterpretq_s32_f32(value1), 31));
52 return vbslq_f32(sign_mask, neg_value0, abs_value0);
53}

93inline int32x4_t check_bounds_s32(int32x4_t value, int32x4_t bound_up, int32x4_t bound_low)
94{
95 int32x4_t ZEROS = vdupq_n_s32(0);
96 uint32x4_t mask = vcgtq_s32(value, bound_up);
97 value = vbslq_s32(mask, ZEROS, value);
98
99 mask = vcgtq_s32(bound_low, value);
100 value = vbslq_s32(mask, ZEROS, value);
101
102 return value;
103}

119inline int8x16_t
120quantize_f32(float32x4_t value_0, float32x4_t value_1, float32x4_t value_2, float32x4_t value_3, float range_limit)
121{
122 // Scale.
123 float32x4_t SCALE = vdupq_n_f32(static_cast<float>(log_likelihood_ratio::max().to_int()) / range_limit);
124 value_0 = vmulq_f32(value_0, SCALE);
125 value_1 = vmulq_f32(value_1, SCALE);
126 value_2 = vmulq_f32(value_2, SCALE);
127 value_3 = vmulq_f32(value_3, SCALE);
128
129 // Clip and round to the nearest integer.
130 float32x4_t RANGE_CEIL = vdupq_n_f32(log_likelihood_ratio::max().to_int());
131 float32x4_t RANGE_FLOOR = vdupq_n_f32(log_likelihood_ratio::min().to_int());
132 value_0 = vrndnq_f32(clip_f32(value_0, RANGE_CEIL, RANGE_FLOOR));
133 value_1 = vrndnq_f32(clip_f32(value_1, RANGE_CEIL, RANGE_FLOOR));
134 value_2 = vrndnq_f32(clip_f32(value_2, RANGE_CEIL, RANGE_FLOOR));
135 value_3 = vrndnq_f32(clip_f32(value_3, RANGE_CEIL, RANGE_FLOOR));
136
137 // Convert to 32 bit integer.
138 int32x4_t llr_s32_0 = vcvtq_s32_f32(value_0);
139 int32x4_t llr_s32_1 = vcvtq_s32_f32(value_1);
140 int32x4_t llr_s32_2 = vcvtq_s32_f32(value_2);
141 int32x4_t llr_s32_3 = vcvtq_s32_f32(value_3);
142
143 // Check bounds one more time: if value_X contains a NaN, its casting to int32_t is indeterminate - set it to zero.
144 int32x4_t BOUND_UP = vdupq_n_s32(log_likelihood_ratio::max().to_int());
145 int32x4_t BOUND_LOW = vdupq_n_s32(log_likelihood_ratio::min().to_int());
146 llr_s32_0 = check_bounds_s32(llr_s32_0, BOUND_UP, BOUND_LOW);
147 llr_s32_1 = check_bounds_s32(llr_s32_1, BOUND_UP, BOUND_LOW);
148 llr_s32_2 = check_bounds_s32(llr_s32_2, BOUND_UP, BOUND_LOW);
149 llr_s32_3 = check_bounds_s32(llr_s32_3, BOUND_UP, BOUND_LOW);
150
151 // Conversion to 16 bit and cast to int8 (every odd element of the vector will be 0 or 0xff).
152 int8x8_t llr_s8_0 = vreinterpret_s8_s16(vmovn_s32(llr_s32_0));
153 int8x8_t llr_s8_1 = vreinterpret_s8_s16(vmovn_s32(llr_s32_1));
154 int8x8_t llr_s8_2 = vreinterpret_s8_s16(vmovn_s32(llr_s32_2));
155 int8x8_t llr_s8_3 = vreinterpret_s8_s16(vmovn_s32(llr_s32_3));
156
157 // Merge resulting LLRs into a single register.
158 int8x8_t out_llrs_s8_0 = vuzp1_s8(llr_s8_0, llr_s8_1);
159 int8x8_t out_llrs_s8_1 = vuzp1_s8(llr_s8_2, llr_s8_3);
160 return vcombine_s8(out_llrs_s8_0, out_llrs_s8_1);
161}

168inline uint32x4_t compute_interval_idx(float32x4_t value, float interval_width, int nof_intervals)
169{
170 // Scale.
171 value = vmulq_f32(value, vdupq_n_f32(1.0F / interval_width));
172
173 // Round to the lowest integer.
174 value = vrndmq_f32(value);
175
176 // Convert to int32.
177 int32x4_t idx = vcvtq_s32_f32(value);
178
179 // Add interval offset.
180 idx = vaddq_s32(idx, vdupq_n_s32(nof_intervals / 2));
181
182 // Clip index and return.
183 return vreinterpretq_u32_s32(clip_s32(idx, vdupq_n_s32(nof_intervals - 1), vdupq_n_s32(0)));
184}

190inline float32x4_t look_up_table(const float* table, uint32x4_t indexes)
191{
192 float32x4_t value = vdupq_n_f32(0.0);
193 value = vsetq_lane_f32(table[vgetq_lane_u32(indexes, 0)], value, 0);
194 value = vsetq_lane_f32(table[vgetq_lane_u32(indexes, 1)], value, 1);
195 value = vsetq_lane_f32(table[vgetq_lane_u32(indexes, 2)], value, 2);
196 value = vsetq_lane_f32(table[vgetq_lane_u32(indexes, 3)], value, 3);
197 return value;
198}

210template <typename Table>
211inline float32x4_t interval_function(float32x4_t value,
212 float32x4_t rcp_noise,
213 float interval_width,
214 unsigned nof_intervals,
215 const Table& slopes,
216 const Table& intercepts)
217{
218 uint32x4_t interval_index = neon::compute_interval_idx(value, interval_width, nof_intervals);
219
220 float32x4_t slope = neon::look_up_table(slopes.data(), interval_index);
221 float32x4_t intercept = neon::look_up_table(intercepts.data(), interval_index);
222
223 return vmulq_f32(vaddq_f32(vmulq_f32(slope, value), intercept), rcp_noise);
224}

229inline float32x4_t safe_div(float32x4_t dividend, float32x4_t divisor)
230{
231 static const float32x4_t all_zero = vdupq_n_f32(0.0f);
232 // Compare divisor to zero (greater than).
233 uint32x4_t mask = vcgtq_f32(divisor, all_zero);
234 float32x4_t result = vdivq_f32(dividend, divisor);
235 return vbslq_f32(mask, result, all_zero);
236}
