117template <typename measure_type, typename Tensor>
118class re_measurement_tensor_base : public re_measurement<measure_type>
119{
120public:
121 // See interface for documentation.
122 const re_measurement_dimensions& size() const override { return dimensions; }
123
124 // See interface for documentation.
125 span<measure_type> get_slice(unsigned i_slice) override
126 {
127 srsran_assert(i_slice < dimensions.nof_slices,
128 "Requested slice {}, but only {} slices are supported.",
129 i_slice,
130 dimensions.nof_slices);
131 return re_meas.template get_view<re_measurement_nof_dimensions::slice>({i_slice});
132 }
133
134 // See interface for documentation.
135 span<const measure_type> get_slice(unsigned i_slice) const override
136 {
137 srsran_assert(i_slice < dimensions.nof_slices,
138 "Requested slice {}, but only {} slices are supported.",
139 i_slice,
140 dimensions.nof_slices);
141 return re_meas.template get_view<re_measurement_nof_dimensions::slice>({i_slice});
142 }
143
144 // See interface for documentation.
145 void set_slice(span<const measure_type> data, unsigned i_slice) override
146 {
147 srsran_assert(i_slice < dimensions.nof_slices,
148 "Requested slice {}, but only {} slices are supported.",
149 i_slice,
150 dimensions.nof_slices);
151 srsran_assert(data.size() == dimensions.nof_subc * dimensions.nof_symbols,
152 "Slice size mismatch: given {}, current {}.",
153 data.size(),
154 dimensions.nof_subc * dimensions.nof_symbols);
155
156 span<measure_type> slice = re_meas.template get_view<re_measurement_nof_dimensions::slice>({i_slice});
157 srsvec::copy(slice, data);
158 }
159
160 // See interface for documentation.
161 span<measure_type> get_symbol(unsigned i_symbol, unsigned i_slice) override
162 {
163 srsran_assert(i_symbol < dimensions.nof_symbols,
164 "Requested symbol {}, but only {} symbols are supported.",
165 i_slice,
166 dimensions.nof_symbols);
167 srsran_assert(i_slice < dimensions.nof_slices,
168 "Requested slice {}, but only {} slices are supported.",
169 i_slice,
170 dimensions.nof_slices);
171 return re_meas.template get_view<re_measurement_nof_dimensions::symbol>({i_symbol, i_slice});
172 }
173
174 // See interface for documentation.
175 span<const measure_type> get_symbol(unsigned i_symbol, unsigned i_slice) const override
176 {
177 srsran_assert(i_symbol < dimensions.nof_symbols,
178 "Requested symbol {}, but only {} symbols are supported.",
179 i_slice,
180 dimensions.nof_symbols);
181 srsran_assert(i_slice < dimensions.nof_slices,
182 "Requested slice {}, but only {} slices are supported.",
183 i_slice,
184 dimensions.nof_slices);
185 return re_meas.template get_view<re_measurement_nof_dimensions::symbol>({i_symbol, i_slice});
186 }
187
188 // See interface for documentation.
189 void resize(const re_measurement_dimensions& dims) override
190 {
191 srsran_assert(dims.nof_subc <= reserved_dims.nof_subc,
192 "Requested {} subcarriers, but at most {} are allowed.",
193 dims.nof_subc,
194 reserved_dims.nof_subc);
195 srsran_assert(dims.nof_symbols <= reserved_dims.nof_symbols,
196 "Requested {} OFDM symbols, but at most {} are allowed.",
197 dims.nof_symbols,
198 reserved_dims.nof_symbols);
199 srsran_assert(dims.nof_slices <= reserved_dims.nof_slices,
200 "Requested {} slices, but at most {} are supported.",
201 dims.nof_slices,
202 reserved_dims.nof_slices);
203
204 dimensions = dims;
205 re_meas.resize({dims.nof_subc, dims.nof_symbols, dims.nof_slices});
206 }
207
208protected:
210 re_measurement_dimensions reserved_dims;
211
218 Tensor re_meas;
219
220private:
222 re_measurement_dimensions dimensions;
223};

235template <typename measure_type, unsigned MAX_NOF_SUBC, unsigned MAX_NOF_SYMBOLS, unsigned MAX_NOF_SLICES>
236class static_re_measurement
237 : public detail::re_measurement_tensor_base<measure_type,
238 static_tensor<detail::re_measurement_nof_dimensions::all,
239 measure_type,
240 MAX_NOF_SUBC * MAX_NOF_SYMBOLS * MAX_NOF_SLICES>>
241{
242 using base = detail::re_measurement_tensor_base<measure_type,
243 static_tensor<detail::re_measurement_nof_dimensions::all,
244 measure_type,
245 MAX_NOF_SUBC * MAX_NOF_SYMBOLS * MAX_NOF_SLICES>>;
246
247public:
249 static_re_measurement()
250 {
251 static_assert(std::is_same<measure_type, float>::value || std::is_same<measure_type, cf_t>::value,
252 "At the moment, this template is only available for float and cf_t.");
253 base::reserved_dims = {MAX_NOF_SUBC, MAX_NOF_SYMBOLS, MAX_NOF_SLICES};
254 }
255
257 explicit static_re_measurement(const re_measurement_dimensions& dims) : static_re_measurement()
258 {
259 static_assert(std::is_same<measure_type, float>::value || std::is_same<measure_type, cf_t>::value,
260 "At the moment, this template is only available for float and cf_t.");
261 base::resize(dims);
262 }
