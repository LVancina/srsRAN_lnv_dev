43inline scheduler_expert_config make_default_scheduler_expert_config()
44{
45 scheduler_expert_config cfg;
46 cfg.ra.max_nof_msg3_harq_retxs = 4;
47 cfg.ra.msg3_mcs_index = 0;
48 cfg.ra.rar_mcs_index = 0;
49
50 // As per TS 38.214, Section 5.1.3.1, only an MCS with modulation order 2 allowed for SIB1.
51 cfg.si.sib1_mcs_index = 5;
52 cfg.si.sib1_retx_period = sib1_rtx_periodicity::ms160;
53 cfg.si.sib1_dci_aggr_lev = aggregation_level::n4;
54
55 // As per TS 38.214, Section 5.1.3.1, only an MCS with modulation order 2 allowed for Paging.
56 cfg.pg.paging_mcs_index = 5;
57 cfg.pg.paging_dci_aggr_lev = aggregation_level::n4;
58 cfg.pg.max_paging_retries = 2;
59
60 cfg.ue.dl_mcs = {0, 28};
61 cfg.ue.initial_cqi = 3;
62 cfg.ue.pdsch_rv_sequence = {0};
63 cfg.ue.ul_mcs = {0, 28};
64 cfg.ue.pusch_rv_sequence = {0};
65 cfg.ue.max_nof_harq_retxs = 4;
66 // The UE is not expected to decode a PDSCH scheduled with P-RNTI, RA-RNTI, SI-RNTI and Qm > 2 i.e. MCS index of 9 in
67 // Table 5.1.3.1-1 of TS 38.214.
68 // Note: A MCS index of 7 can handle Msg4 of size 458 bytes for PDSCH occupying symbols (2,..,14].
69 // Note: A MCS index of 8 can handle Msg4 of size 325 bytes for PDSCH occupying symbols (3,..,14].
70 cfg.ue.max_msg4_mcs = 9;
71 cfg.ue.initial_ul_sinr = 5;
72 cfg.ue.enable_csi_rs_pdsch_multiplexing = true;
73 cfg.ue.ta_measurement_slot_period = 80;
74 cfg.ue.ta_cmd_offset_threshold = 1;
75 cfg.ue.ta_update_measurement_ul_sinr_threshold = 0.0F;
76
77 cfg.log_broadcast_messages = true;
78 cfg.metrics_report_period = std::chrono::milliseconds{1000};
79
80 return cfg;
81}

84inline du_cell_config make_default_du_cell_config(const cell_config_builder_params_extended& params = {})
85{
86 du_cell_config cfg{};
87 cfg.pci = params.pci;
88 cfg.tac = 1;
89 cfg.nr_cgi.plmn = "00101";
90 cfg.nr_cgi.nci = config_helpers::make_nr_cell_identity(411, 32, 1);
91
92 cfg.dl_carrier = make_default_dl_carrier_configuration(params);
93 cfg.ul_carrier = make_default_ul_carrier_configuration(params);
94 cfg.coreset0_idx = *params.coreset0_index;
95 cfg.searchspace0_idx = params.search_space0_index;
96 cfg.dl_cfg_common = make_default_dl_config_common(params);
97 cfg.ul_cfg_common = make_default_ul_config_common(params);
98 cfg.scs_common = params.scs_common;
99 cfg.ssb_cfg = make_default_ssb_config(params);
100 cfg.cell_barred = false;
101 cfg.intra_freq_resel = false;
102 cfg.ue_timers_and_constants = make_default_ue_timers_and_constants_config();
103
104 // The CORESET duration of 3 symbols is only permitted if dmrs-typeA-Position is set to 3. Refer TS 38.211, 7.3.2.2.
105 const pdcch_type0_css_coreset_description coreset0_desc = pdcch_type0_css_coreset_get(
106 cfg.dl_carrier.band, params.ssb_scs, params.scs_common, *params.coreset0_index, params.k_ssb->value());
107 cfg.dmrs_typeA_pos = coreset0_desc.nof_symb_coreset == 3U ? dmrs_typeA_position::pos3 : dmrs_typeA_position::pos2;
108
109 cfg.tdd_ul_dl_cfg_common = params.tdd_ul_dl_cfg_common;
110 cfg.ue_ded_serv_cell_cfg = create_default_initial_ue_serving_cell_config(params);
111
112 return cfg;
113}

135inline std::map<five_qi_t, du_qos_config> make_default_du_qos_config_list(int rlc_metrics_report)
136{
137 std::map<five_qi_t, du_qos_config> qos_list = {};
138 {
139 // 5QI=7
140 du_qos_config cfg{};
141 // RLC
142 cfg.rlc.mode = rlc_mode::um_bidir;
143 cfg.rlc.um.tx.sn_field_length = rlc_um_sn_size::size12bits;
144 cfg.rlc.um.rx.sn_field_length = rlc_um_sn_size::size12bits;
145 cfg.rlc.um.rx.t_reassembly = 100;
146 cfg.rlc.um.tx.queue_size = 4096;
147 cfg.rlc.metrics_period = std::chrono::milliseconds(rlc_metrics_report);
148 // F1-U
149 cfg.f1u.t_notify = 10;
150 // MAC
151 cfg.mac = make_default_drb_mac_lc_config();
152 cfg.mac.priority = 4;
153 cfg.mac.lcg_id = uint_to_lcg_id(1);
154
155 qos_list[uint_to_five_qi(7)] = cfg;
156 }
157 {
158 // 5QI=9
159 du_qos_config cfg{};
160 // RLC
161 cfg.rlc.mode = rlc_mode::am;
162 cfg.rlc.am.tx.sn_field_length = rlc_am_sn_size::size18bits;
163 cfg.rlc.am.tx.t_poll_retx = 20;
164 cfg.rlc.am.tx.poll_pdu = 16;
165 cfg.rlc.am.tx.poll_byte = -1;
166 cfg.rlc.am.tx.max_retx_thresh = 32;
167 cfg.rlc.am.tx.max_window = 0;
168 cfg.rlc.am.tx.queue_size = 4096;
169 cfg.rlc.am.rx.sn_field_length = rlc_am_sn_size::size18bits;
170 cfg.rlc.am.rx.t_reassembly = 20;
171 cfg.rlc.am.rx.t_status_prohibit = 10;
172 cfg.rlc.am.rx.max_sn_per_status = {};
173 cfg.rlc.metrics_period = std::chrono::milliseconds(rlc_metrics_report);
174 // F1-U
175 cfg.f1u.t_notify = 10;
176 // MAC
177 cfg.mac = make_default_drb_mac_lc_config();
178
179 qos_list[uint_to_five_qi(9)] = cfg;
180 }
181 return qos_list;
182}
