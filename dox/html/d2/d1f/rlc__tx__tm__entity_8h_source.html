31class rlc_tx_tm_entity : public rlc_tx_entity
32{
33private:

36

39
40 task_executor& pcell_executor;
41
42 pcap_rlc_pdu_context pcap_context;
43
44 /// This atomic_flag indicates whether a buffer state update task has been queued but not yet run by pcell_executor.
45 /// It helps to avoid queuing of redundant notification tasks in case of frequent changes of the buffer status.
46 /// If the flag is set, no further notification needs to be scheduled, because the already queued task will pick the
47 /// latest buffer state upon execution.
48 std::atomic_flag pending_buffer_state = ATOMIC_FLAG_INIT;
49
50public:
51 rlc_tx_tm_entity(uint32_t du_index,
52 du_ue_index_t ue_index,
53 rb_id_t rb_id,
54 const rlc_tx_tm_config& config,
55 rlc_tx_upper_layer_data_notifier& upper_dn_,
56 rlc_tx_upper_layer_control_notifier& upper_cn_,
57 rlc_tx_lower_layer_notifier& lower_dn_,
58 task_executor& pcell_executor_,
59 bool metrics_enabled_,
60 rlc_pcap& pcap_);
61
62 // Interfaces for higher layers
63 void handle_sdu(rlc_sdu sdu) override;
64 void discard_sdu(uint32_t pdcp_sn) override;
65
66 // Interfaces for lower layers
67 size_t pull_pdu(span<uint8_t> rlc_pdu_buf) override;
68 uint32_t get_buffer_state() override;
69
70private:
71 /// Called whenever the buffer state has been changed by upper layers (new SDUs or SDU discard) so that lower layers
72 /// need to be informed about the new buffer state. This function defers the actual notification \c
73 /// handle_changed_buffer_state to pcell_executor. The notification is discarded if another notification is already
74 /// queued for execution. This function should not be called from \c pull_pdu, since the lower layer accounts for the
75 /// amount of extracted data itself.
76 ///
77 /// Safe execution from: Any executor
78 void handle_changed_buffer_state();
79
80 /// Immediately informs the lower layer of the current buffer state. This function is called from pcell_executor and
81 /// its execution is queued by \c handle_changed_buffer_state.
82 ///
83 /// Safe execution from: pcell_executor
84 void update_mac_buffer_state();
85};
