31class rx_symbol_handler_example : public upper_phy_rx_symbol_handler
32{
33private:
34 srslog::basic_logger& logger;

37public:
38 rx_symbol_handler_example(std::string log_level) : logger(srslog::fetch_basic_logger("RxSyHan"))
39 {
40 logger.set_level(srslog::str_to_basic_level(log_level));
41 }
42
43 void handle_rx_symbol(const upper_phy_rx_symbol_context& context, const resource_grid_reader& grid) override
44 {
45 std::unique_lock<std::mutex> lock(mutex);
46 logger.debug(context.slot.sfn(),
47 context.slot.slot_index(),
48 "Rx symbol {} received for sector {}",
49 context.symbol,
50 context.sector);
51 }
52
53 void handle_rx_prach_window(const prach_buffer_context& context, const prach_buffer& buffer) override
54 {
55 std::unique_lock<std::mutex> lock(mutex);
56 logger.debug(context.slot.sfn(),
57 context.slot.slot_index(),
58 "PRACH symbol {} received for sector {}",
59 context.start_symbol,
60 context.sector);
61 }
62
63 void handle_rx_srs_symbol(const upper_phy_rx_symbol_context& context) override
64 {
65 std::unique_lock<std::mutex> lock(mutex);
66 logger.debug(context.slot.sfn(),
67 context.slot.slot_index(),
68 "SRS symbol {} received for sector {}",
69 context.symbol,
70 context.sector);
71 }
72};
