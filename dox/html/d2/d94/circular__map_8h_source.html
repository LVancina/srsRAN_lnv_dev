11/// Contiguous array of optional objects indexed circularly via a integer key. Random access, insertion and removal
12/// should have complexity O(1).
13/// Each container element contains 3 fields: (i) presence flag, (ii) key, (iii) stored object
14/// In case the presence flag is set to true, the stored object exists and its ctor has been called.
15/// The following steps are performed to access an element:
16/// 1. The provided key K is converted to an array index via the operation I = K % N.
17/// 2. The presence flag for the index I is checked. If present,
18/// 3. Check if the provided key K corresponds to the key stored in the element. If the keys match,
19/// 4. Retrieve the stored object.
20/// The container does not resolve collisions, i.e. it doesn't allow two objects of different keys but same index
21/// to be inserted in the container. It is up to the user to resolve such situations.
22/// There is no pointer and iterator invalidation.
23/// \tparam K type of key used to retrieve an element index. K must be an unsigned integer
24/// \tparam T type of each element
25/// \tparam N maximum number of elements
26template <typename K, typename T, size_t N>
27class circular_map

171 /// Constructs object with given key if no collision is detected.
172 /// \param key key of created object
173 /// \return true if no collision was detected and object was inserted. False, otherwise.
174 template <typename... Args>
175 bool emplace(K key, Args&&... args)
176 {
177 static_assert(std::is_constructible<T, Args...>::value, "Invalid argument types");
178 size_t idx = key % N;
179 if (present[idx]) {
180 return false;
181 }
182 buffer[idx].emplace(
183 std::piecewise_construct, std::forward_as_tuple(key), std::forward_as_tuple(std::forward<Args>(args)...));
184 present[idx] = true;
185 count++;
186 return true;
187 }
