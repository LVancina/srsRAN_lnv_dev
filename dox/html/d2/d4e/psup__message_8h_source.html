68/// DL PDU SESSION INFORMATION (PDU Type 0)
69///
70/// Ref: TS 38.415 Sec. 5.2.2.1
71struct psup_dl_pdu_session_information {
72 /// PDU Type.
73 /// The PDU Type indicates the structure of the PDU session UP frame. The field takes the value of the PDU Type it
74 /// identifies; i.e. "0" for PDU Type 0. The PDU type is in bit 4 to bit 7 in the first octet of the frame.
75 const psup_pdu_type pdu_type = psup_pdu_type::dl_pdu_session_information;
76 /// Reflective QoS Indicator (RQI).
77 /// This parameter indicates activation of the reflective QoS towards the UE for the transferred packet as described
78 /// in clause 5.4.1.1. It is used only in the downlink direction. If RQA (Reflective QoS Activation) has not been
79 /// configured for the involved QoS flow, the RQI shall be ignored by the NG-RAN node.
80 bool rqi = false;
81 /// QoS Flow Identifier (QFI).
82 /// When present this parameter indicates the QoS Flow Identifier of the QoS flow to which the transferred packet
83 /// belongs.
84 qos_flow_id_t qos_flow_id = qos_flow_id_t::invalid;
85 /// Paging Policy Indicator (PPI).
86 /// When present, the Paging Policy Indicator is used for paging policy differentiation (see 3GPP TS 23.501). This
87 /// field applies to PDU sessions of IP type.
88 optional<psup_ppi> ppi;
89 /// DL Received Time Stamp.
90 /// This field indicates the time when the UPF sends the DL PDU Session Information frame with the QMP field set to 1.
91 /// It is used only in the downlink direction and encoded in the same format as the 64 - bit timestamp format as
92 /// defined in Section 6 of IETF RFC 5905 [6].
93 optional<uint64_t> dl_sending_time_stamp;
94 /// DL QFI Sequence Number.
95 /// This parameter indicates the sequence number as assigned by the UPF/ NG-RAN node associated with a given QoS flow.
96 optional<uint32_t> dl_qfi_sn;
97
98 bool operator==(const psup_dl_pdu_session_information& other) const
99 {
100 return rqi == other.rqi && qos_flow_id == other.qos_flow_id && ppi == other.ppi &&
101 dl_sending_time_stamp == other.dl_sending_time_stamp && dl_qfi_sn == other.dl_qfi_sn;
102 }
103};

108struct psup_ul_pdu_session_information {
109 /// PDU Type.
110 /// The PDU Type indicates the structure of the PDU session UP frame. The field takes the value of the PDU Type it
111 /// identifies; i.e. "0" for PDU Type 0. The PDU type is in bit 4 to bit 7 in the first octet of the frame.

113 /// QoS Flow Identifier (QFI).
114 /// When present this parameter indicates the QoS Flow Identifier of the QoS flow to which the transferred packet
115 /// belongs.
116 qos_flow_id_t qos_flow_id = qos_flow_id_t::invalid;
117 /// DL Sending Time Stamp Repeated.
118 /// This field indicates the value of the DL Sending Time Stamp field that the NG-RAN has received in the DL PDU
119 /// Session Information frame with the QMP field set to 1 for the involved QoS flow. It is used only in the uplink
120 /// direction and encoded in the same format as the 64-bit timestamp format as defined in Section 6 of IETF RFC 5905
121 /// [6]. The UPF shall, if supported, use this information to calculate DL or RTT delay between the NG-RAN and the UPF
122 /// as specified in [5].
123 optional<uint64_t> dl_sending_time_stamp_repeated;
124 /// DL Received Time Stamp.
125 /// This field indicates the time when the NG-RAN node receives the DL PDU Session Information frame with the QMP
126 /// field set to 1 for the involved QoS flow. It is used only in the uplink direction and encoded in the same format
127 /// as the 64-bit timestamp format as defined in Section 6 of IETF RFC 5905 [6]. The UPF shall, if supported, use
128 /// this information to calculate DL or RTT delay between the NG-RAN and the UPF as specified in [5].
129 optional<uint64_t> dl_received_time_stamp;
130 /// DL Sending Time Stamp.
131 /// This field indicates the time when the UPF sends the DL PDU Session Information frame with the QMP field set
132 /// to 1. It is used only in the downlink direction and encoded in the same format as the 64-bit timestamp format as
133 /// defined in Section 6 of IETF RFC 5905 [6].
134 optional<uint64_t> ul_sending_time_stamp;
135 /// DL Delay Result.
136 /// This field indicates the downlink delay measurement result which is the sum of the delay incurred in NG-RAN
137 /// (including the delay at gNB-CU-UP, on F1-U and on gNB-DU) and the delay over Uu interface in milliseconds for the
138 /// involved QoS flow. It is used only in the uplink direction and encoded as an Unsigned32 binary integer value. The
139 /// UPF shall, if supported, use this information to calculate DL or RTT delay as specified in [5].
140 optional<uint32_t> dl_delay_result;
141 /// UL Delay Result.
142 /// This field indicates the uplnk delay measurement result which is the sum of the delay incurred in NG-RAN
143 /// (including the delay at gNB-CU-UP, on F1-U and on gNB-DU) and the delay over Uu interface in milliseconds for the
144 /// involved QoS flow. It is used only in the uplink direction and encoded as an Unsigned32 binary integer value. The
145 /// UPF shall, if supported, use this information to calculate UL or RTT delay as specified in [5].
146 optional<uint32_t> ul_delay_result;
147 /// UL QFI Sequence Number.
148 /// This parameter indicates the sequence number as assigned by the NG-RAN node associated with a given QoS flow.
149 optional<uint32_t> ul_qfi_sn;
150};
