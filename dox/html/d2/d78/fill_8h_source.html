31/// \brief Sets all elements of a sequence to a given value.
32///
33/// \tparam T Type of the sequence container, must be span-compatible.
34/// \param x Sequence container.
35/// \param value Value the sequence is set to. Must be compatible with the element type of the sequence.
36template <typename T>
37void fill(T&& x, detail::value_type_of_t<T> value)
38{
39 static_assert(detail::is_span_compatible<T>::value, "Template type is not compatible with a span");
40 std::fill(x.begin(), x.end(), value);
41}
