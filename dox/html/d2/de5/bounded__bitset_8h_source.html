36/// \brief Returns an unsigned integer with the N most significant bits (MSB) set to zero, and the remaining bits set
37/// to 1.
38/// \tparam Integer Type of unsigned integer returned by the function.
39/// \param[in] N Number of MSB bits set to zero.
40/// \return Resulting integer bitmap.
41template <typename Integer>
42constexpr Integer mask_msb_zeros(size_t N)
43{
44 static_assert(std::is_unsigned<Integer>::value, "T must be unsigned integer");
45 return (N == 0) ? static_cast<Integer>(-1) : (N == sizeof(Integer) * 8U) ? 0 : (static_cast<Integer>(-1) >> (N));
46}
47
48/// \brief Returns an unsigned integer with the N least significant bits (LSB) set to zero, and the remaining bits set
49/// to 1.
50/// \tparam Integer Type of unsigned integer returned by the function.
51/// \param[in] N Number of LSB bits set to zero.
52/// \return Resulting integer bitmap.
53template <typename Integer>
54constexpr Integer mask_lsb_ones(size_t N)
55{
56 return mask_msb_zeros<Integer>(sizeof(Integer) * 8U - N);
57}
58
59/// \brief Returns an unsigned integer with the N most significant bits (MSB) set to one, and the remaining bits set
60/// to zero.
61/// \tparam Integer Type of unsigned integer returned by the function.
62/// \param[in] N Number of MSB bits set to one.
63/// \return Resulting integer bitmap.
64template <typename Integer>
65constexpr Integer mask_msb_ones(size_t N)
66{
67 return ~mask_msb_zeros<Integer>(N);
68}

70/// \brief Returns an unsigned integer with the N least significant bits (LSB) set to one, and the remaining bits set
71/// to zero.
72/// \tparam Integer Type of unsigned integer returned by the function.
73/// \param[in] N Number of LSB bits set to one.
74/// \return Resulting integer bitmap.

83template <typename Integer>
84struct bitset_builtin_helper {
85 /// \brief Returns the number of leading 0-bits in value, starting at the most significant bit position.
86 /// If value is 0, the result is number of bits of Integer.
87 static Integer zero_msb_count(Integer value)
88 {
89 if (value == 0) {
90 return std::numeric_limits<Integer>::digits;
91 }
92 Integer ret = 0;
93 for (Integer shift = std::numeric_limits<Integer>::digits >> 1; shift != 0; shift >>= 1) {
94 Integer tmp = value >> shift;
95 if (tmp != 0) {
96 value = tmp;
97 } else {
98 ret |= shift;
99 }
100 }
101 return ret;
102 }
103 /// \brief Returns the number of trailing 0-bits in value, starting at the least significant bit position.
104 /// If value is 0, the result is number of bits of Integer.
105 static Integer zero_lsb_count(Integer value)
106 {
107 if (value == 0) {
108 return std::numeric_limits<Integer>::digits;
109 }
110 Integer ret = 0;
111 for (Integer shift = std::numeric_limits<Integer>::digits >> 1, mask = std::numeric_limits<Integer>::max() >> shift;
112 shift != 0;
113 shift >>= 1, mask >>= shift) {
114 if ((value & mask) == 0) {
115 value >>= shift;
116 ret |= shift;
117 }
118 }
119 return ret;
120 }
121
122 /// \brief Returns the number of 1-bits in value.
123 static int count_ones(Integer value)
124 {
125 // Note: use an "int" for count triggers popcount optimization if SSE instructions are enabled.
126 int c = 0;
127 for (Integer w = value; w > 0; c++) {
128 w &= w - 1;
129 }
130 return c;
131 }
132};

180/// \brief Knuth's swap of upper and lower sections of a bitset.
181/// \param m Mask of bits to swap.
182/// \param k shift amount.
183template <typename T, T m, int k>
184T swapbits(T p)
185{
186 T q = ((p >> k) ^ p) & m;
187 return p ^ q ^ (q << k);
188}

192/// \brief Knuth's 64-bit reverse. E.g. 0x0000000000000001 -> 0x8000000000000000.
193/// For more information see: https://matthewarcus.wordpress.com/2012/11/18/reversing-a-64-bit-word/
194/// \param n Number to reverse.
195/// \return Reversed number.
196inline uint64_t bit_reverse(uint64_t n)
197{
198 static const uint64_t m0 = 0x5555555555555555LLU;
199 static const uint64_t m1 = 0x0300c0303030c303LLU;
200 static const uint64_t m2 = 0x00c0300c03f0003fLLU;
201 static const uint64_t m3 = 0x00000ffc00003fffLLU;
202 n = ((n >> 1U) & m0) | (n & m0) << 1U;
203 n = detail::swapbits<uint64_t, m1, 4>(n);
204 n = detail::swapbits<uint64_t, m2, 8>(n);
205 n = detail::swapbits<uint64_t, m3, 20>(n);
206 n = (n >> 34U) | (n << 30U);
207 return n;
208}

248/// \brief Represents a dynamically-sized bitset with an upper bound capacity of N bits.
249///
250/// The bounded_bitset is represented internally via an array of uint64_t, with each integer storing a bitmap.
251/// This class also offers many standard logic manipulation methods, like ::any(), operators &=, &, |=, |, etc. and
252/// utility methods to convert the bitset into strings or integers.
253///
254/// Depending on the passed \c LowestInfoBitIsMSB template argument, this class can represent the bits of the bitset in
255/// different orders. E.g.
256///
257/// bounded_bitset<6, false> a(5); // Bitset of 5 bits. LSB is Lowest Information Bit (bit 0).
258/// a.set(0);
259/// assert(a.to_uint64() == 0b1);
260/// bounded_bitset<6, true> b(5); // Bitset of 5 bits. MSB is Lowest Information Bit (bit 0).
261/// b.set(1);
262/// assert(a.to_uint64() == 0b1000);
263///
264/// The \c LowestInfoBitIsMSB template argument also affects the default string representation of the bitset. E.g.
265/// fmt::print("{:b}", a); // prints "00001".
266/// fmt::print("{:x}", a); // prints "1".
267/// fmt::print("{:b}", b); // prints "01000".
268/// fmt::print("{:x}", b); // prints "8".
269///
270/// However, it does not affect the information bit position string representation, e.g.
271/// fmt::print("{:n}", a); // prints "0".
272/// fmt::print("{:n}", b); // prints "1".
273///
274/// \tparam N Upper bound for bitset size in number of bits.
275/// \tparam LowestInfoBitIsMSB Bit index order in memory. If set to (false / true), the bit index 0 (Lowest Information
276/// Bit) corresponds to either the LSB or MSB of the bitset. Note that this argument has an effect on the underlying
277/// bitset memory layout.
278template <size_t N, bool LowestInfoBitIsMSB = false>
279class bounded_bitset
280{
281 using word_t = uint64_t;
282 static const size_t bits_per_word = 8U * sizeof(word_t);

284public:

297 /// \brief Constructs a bitset using iterators.
298 ///
299 /// The constructed bitset size is equal to <tt> end - begin </tt> size. The values in the list are mapped one to one
300 /// starting from begin.
301 ///
302 /// \tparam Iterator Boolean iterator type.
303 /// \param[in] begin Begin iterator.
304 /// \param[in] end End iterator.
305 template <
306 typename Iterator,
307 typename std::enable_if_t<std::is_convertible<typename std::iterator_traits<Iterator>::value_type, bool>::value,
308 int> = 0>
309 constexpr bounded_bitset(Iterator begin, Iterator end)
310 {
311 resize(end - begin);
312 std::for_each(begin, end, [this, n = 0](bool value) mutable {
313 if (value) {
314 set_(n);
315 } else {
316 reset_(n);
317 }
318 ++n;
319 });
320 }

322 /// \brief Constructs a bitset from an initializer list.
323 ///
324 /// The constructed bitset size is equal to \c values size. The values in the list are mapped one to one.
325 ///
326 /// \param[in] values Boolean initializer list.
327 constexpr bounded_bitset(const std::initializer_list<const bool>& values)
328 {
329 resize(values.size());
330 std::for_each(values.begin(), values.end(), [this, n = 0](bool value) mutable {
331 assert_within_bounds_(n, true);
332 if (value) {
333 set_(n);

336 }

365 /// \brief Resize of the bounded_bitset. If <tt> new_size > max_size() </tt>, an assertion is triggered. The newly
366 /// created are set to zero.
367 void resize(size_t new_size)
368 {
369 srsran_assert(new_size <= max_size(), "ERROR: new size='{}' exceeds bitset capacity='{}'", new_size, max_size());
370 if (new_size == cur_size) {
371 return;
372 }
373 const size_t prev_nof_words = nof_words_();
374 cur_size = new_size;
375 sanitize_();
376 const size_t new_nof_words = nof_words_();
377 const size_t start = std::min(prev_nof_words, new_nof_words), end = std::max(prev_nof_words, new_nof_words);
378 // The words created/deleted are set to zero.
379 for (size_t i = start; i != end; ++i) {
380 buffer[i] = static_cast<word_t>(0);
381 }
382 }

434 /// \brief Appends a bit with value \c val to the set.
435 ///
436 /// Assertion is triggered if the resultant size exceeds the maximum size of the bitset.

444 /// \brief Appends \c nof_bits bits to the set.
445 ///
446 /// The least \c nof_bits significant bits of \c val are appended to the set, starting from the most significant bit
447 /// and finishing with the least significant bit.
448 ///
449 /// Assertion is triggered if the resultant size exceeds the maximum size of the bitset.
450 template <typename Integer>
451 void push_back(Integer val, unsigned nof_bits)
452 {
453 static_assert(std::is_unsigned<Integer>::value, "push_back only works for unsigned integers");
454 unsigned bitpos = size();
455 resize(size() + nof_bits);
456 if (LowestInfoBitIsMSB) {
457 for (unsigned bit_index = 0; bit_index != bitpos; ++bit_index) {
458 set(bit_index, test(bit_index + nof_bits));
459 }
460 }
461 for (unsigned bit_index = 0; bit_index != nof_bits; ++bit_index) {
462 set(bitpos + bit_index, (val >> (nof_bits - 1 - bit_index)) & 1U);
463 }
464 }

466 /// \brief Extracts \c nof_bits starting from \c startpos.
467 ///
468 /// \return An unsigned integer containing \c nof_bits of the set where starting with the most significant bit and
469 /// finishing with the most significant bit.
470 /// \remark An assertion is triggered if the bit range exceed the set size.
471 template <typename Integer = unsigned>
472 Integer extract(unsigned startpos, unsigned nof_bits) const
473 {
474 static_assert(std::is_unsigned<Integer>::value, "Extract only works for unsigned integers");
475 srsran_assert(nof_bits <= sizeof(Integer) * 8,
476 "The number of bits (i.e., {}) exceeds the destination bit-width (i.e., {}).",
477 nof_bits,
478 sizeof(Integer) * 8);
479 srsran_assert(startpos + nof_bits <= size(),
480 "The start position (i.e., {}) plus the number of bits (i.e., {}) exceed the current size (i.e., {})",
481 startpos,
482 nof_bits,
483 size());
484
485 Integer val = 0;
486
487 for (unsigned bit_index = 0; bit_index != nof_bits; ++bit_index) {
488 if (test(startpos + bit_index)) {
489 val |= 1U << (nof_bits - 1 - bit_index);
490 }
491 }
492
493 return val;
494 }

496 /// \brief Kronecker product of the bitset with another bitset.
497 ///
498 /// Expands the bitset by a factor of \c other.size() replacing every \c true bit with the contents of \c other and
499 /// every \c false bit with \c other.size() \c false bits.
500 ///
501 /// \tparam Factor Maximum expansion factor.
502 /// \param[in] other Bitset used for expansion.
503 /// \return The result of the bitset product.
504 /// \remark The current implementation supports only a bitset containing one word. An assertion is triggered if \c
505 /// other contains more than one word.
506 template <unsigned Factor>
507 bounded_bitset<Factor * N> kronecker_product(const bounded_bitset<Factor>& other) const
508 {
509 static_assert(Factor <= bits_per_word,
510 "The current algorithm does not support a filter containing more than one word.");
511
512 // Prepare an empty result.
513 bounded_bitset<Factor * N> result(size() * other.size());
514
515 // Places the contents of other centered at the positions indicated by the true bits.
516 std::function<void(unsigned)> kronecker_expansion = [&other, &result](unsigned bit_index) {
517 unsigned bitpos = bit_index * Factor;
518 word_t word = other.buffer[0];
519
520 unsigned bit_offset = bitpos % bits_per_word;
521 unsigned word_index = bitpos / bits_per_word;
522
523 result.buffer[word_index] |= (word << bit_offset);
524 if (bit_offset && (bit_offset + other.size() > bits_per_word)) {
525 result.buffer[word_index + 1] |= (word >> (bits_per_word - bit_offset));
526 }
527 };
528
529 if (is_contiguous(true)) {
530 int i_begin = find_lowest(true);
531 int i_end = find_highest(true) + 1;
532
533 if ((i_begin < 0) || (i_end <= 0)) {
534 // Empty bitset.
535 return result;
536 }
537
538 // If the bitset in contiguous and the other bitset is all set, then use fill.
539 if (other.all()) {
540 result.fill(i_begin * other.size(), i_end * other.size());
541 } else {
542 // Otherwise, place the contents of other into contiguous bit positions.
543 for (int i_bit = i_begin; i_bit != i_end; ++i_bit) {
544 kronecker_expansion(i_bit);
545 }
546 }
547
548 srsran_assert(count() * other.count() == result.count(),
549 "The resultant number of ones is not consistent with inputs. It expected {} but got {}.",
550 count() * other.count(),
551 result.count());
552 return result;
553 }
554
555 // Place the contents of other into arbitrary bit positions.
556 for_each(0, size(), kronecker_expansion);
557
558 srsran_assert(count() * other.count() == result.count(),
559 "The resultant number of ones is not consistent with inputs. It expected {} but got {}.",
560 count() * other.count(),
561 result.count());
562
563 return result;
564 }

604 /// \brief Returns bounded_bitset<> that represents a slice or subview of the original bounded_bitset. Unless
605 /// it is specified, the returned slice has the same template parameters "N" and "LowestInfoBitIsMSB" of "this".
606 ///
607 /// \param[in] startpos The bit index where the subview starts.
608 /// \param[in] endpos The bit index where the subview stops.
609 template <size_t N2 = N>
610 bounded_bitset<N2, LowestInfoBitIsMSB> slice(size_t startpos, size_t endpos) const
611 {
612 bounded_bitset<N2, LowestInfoBitIsMSB> sliced(endpos - startpos);
613 if (LowestInfoBitIsMSB) {
614 std::swap(startpos, endpos);
615 startpos = get_bitidx_(startpos) + 1;
616 endpos = get_bitidx_(endpos) + 1;
617 }
618 unsigned start_word = startpos / bits_per_word;
619 unsigned start_mod = startpos % bits_per_word;
620
621 if (start_mod != 0) {
622 word_t left_mask = mask_lsb_ones<word_t>(bits_per_word - start_mod);
623 word_t right_mask = mask_lsb_ones<word_t>(start_mod);
624 for (unsigned i = 0; i != sliced.nof_words_(); ++i) {
625 sliced.buffer[i] = (buffer[i + start_word] >> start_mod) & left_mask;
626 if (i + start_word + 1 < nof_words_()) {
627 sliced.buffer[i] |= (buffer[i + start_word + 1] & right_mask) << (bits_per_word - start_mod);
628 }
629 }
630 } else {
631 for (unsigned i = 0; i != sliced.nof_words_(); ++i) {
632 sliced.buffer[i] = buffer[i + start_word];
633 }
634 }
635 sliced.sanitize_();
636 return sliced;
637 }

640 /// \param[in] value The bit value to find, either true (1) or false (0).
641 /// \return Returns the lowest found bit index or -1 in case no bit was found with the provided value argument.
642 int find_lowest(bool value = true) const noexcept { return find_lowest(0, size(), value); }

644 /// \brief Finds, within a range of bit indexes, the lowest bit with value set to the value passed as argument.
645 /// \param[in] startpos Starting bit index for the search.
646 /// \param[in] endpos End bit index for the search.
647 /// \param[in] value The bit value to find, either true (1) or false (0).
648 /// \return Returns the lowest found bit index or -1 in case no bit was found with the provided value argument.
649 int find_lowest(size_t startpos, size_t endpos, bool value = true) const noexcept
650 {
651 assert_range_bounds_(startpos, endpos);
652 if (startpos == endpos) {
653 return -1;
654 }
655
656 if (not LowestInfoBitIsMSB) {
657 return find_first_(startpos, endpos, value);
658 }
659 return find_first_reversed_(startpos, endpos, value);
660 }

662 /// \brief Executes a function for all \c true (or all \c false) bits in the given bitset interval.
663 ///
664 /// \param[in] startpos Smallest bit index considered for the function execution (included).
665 /// \param[in] endpos Largest bit index considered for the function execution (excluded).
666 /// \param[in] function Function to execute - the signature should be compatible with <tt>void ()(unsigned)</tt>.
667 /// \param[in] value Bit value that triggers the function execution.
668 template <class T>
669 inline void for_each(size_t startpos, size_t endpos, T&& function, bool value = true) const noexcept
670 {
671 static_assert(std::is_convertible<T, std::function<void(size_t)>>::value,
672 "The function must have void(size_t) signature.");
673 static_assert(!LowestInfoBitIsMSB, "The for_each method is not yet available for reversed bitsets.");
674
675 assert_range_bounds_(startpos, endpos);
676
677 if (startpos == endpos) {
678 return;
679 }
680
681 if ((value && all(startpos, endpos)) || (!value && none(startpos, endpos))) {
682 for (size_t bitpos = startpos; bitpos != endpos; ++bitpos) {
683 function(bitpos);
684 }
685 return;
686 }
687
688 size_t startword = startpos / bits_per_word;
689 size_t lastword = (endpos + bits_per_word - 1) / bits_per_word;
690 for (size_t i = startword; i != lastword; ++i) {
691 word_t w = buffer[i];
692 if (not value) {
693 w = ~w;
694 }
695
696 if (w == 0) {
697 continue;
698 }
699
700 if (i == startword) {
701 w &= mask_lsb_zeros<word_t>(startpos % bits_per_word);
702 }
703
704 if ((i == lastword - 1) && (endpos % bits_per_word != 0)) {
705 w &= mask_lsb_ones<word_t>(endpos % bits_per_word);
706 }
707
708 // Process presets of 4 bits.
709 unsigned bitpos = i * bits_per_word;
710 for (; w != 0; w = w >> 4, bitpos += 4) {
711 switch (w & 0xf) {
712 case 0B0000:
713 // All bits are false, skip.
714 break;
715 case 0B0001:
716 function(bitpos + 0);
717 break;
718 case 0B0010:
719 function(bitpos + 1);
720 break;
721 case 0B0011:
722 function(bitpos + 0);
723 function(bitpos + 1);
724 break;
725 case 0B0100:
726 function(bitpos + 2);
727 break;
728 case 0B0101:
729 function(bitpos + 0);
730 function(bitpos + 2);
731 break;
732 case 0B0110:
733 function(bitpos + 1);
734 function(bitpos + 2);
735 break;
736 case 0B0111:
737 function(bitpos + 0);
738 function(bitpos + 1);
739 function(bitpos + 2);
740 break;
741 case 0B1000:
742 function(bitpos + 3);
743 break;
744 case 0B1001:
745 function(bitpos + 0);
746 function(bitpos + 3);
747 break;
748 case 0B1010:
749 function(bitpos + 1);
750 function(bitpos + 3);
751 break;
752 case 0B1011:
753 function(bitpos + 0);
754 function(bitpos + 1);
755 function(bitpos + 3);
756 break;
757 case 0B1100:
758 function(bitpos + 2);
759 function(bitpos + 3);
760 break;
761 case 0B1101:
762 function(bitpos + 0);
763 function(bitpos + 2);
764 function(bitpos + 3);
765 break;
766 case 0B1110:
767 function(bitpos + 1);
768 function(bitpos + 2);
769 function(bitpos + 3);
770 break;
771 case 0B1111:
772 default:
773 function(bitpos + 0);
774 function(bitpos + 1);
775 function(bitpos + 2);
776 function(bitpos + 3);
777 break;
778 }
779 }
780 }
781 }

783 /// \brief Checks if all bits in the bitset are set to 1.
784 /// \return Returns true if all bits are 1.
785 bool all() const noexcept
786 {

789 return true;

795 }
796 }

864 /// \brief Determines whether all bits with value set to \c value are contiguous.
865 ///
866 /// Bits with the same value are contiguous if:
867 /// 1. one bit with \c value is found,
868 /// 2. no bit with \c value is found, and
869 /// 3. all bits with \c value are consecutive.
870 ///
871 /// \param[in] startpos Starting bit index for the search.
872 /// \param[in] endpos End bit index for the search.
873 /// \param[in] value The bit value to find, either true (1) or false (0).
874 /// \return Returns the lowest found bit index or -1 in case no bit was found with the provided value argument.
875 bool is_contiguous(bool value = true) const noexcept
876 {
877 // Find the lowest value.
878 int startpos = find_lowest(0, size(), value);
879
880 // Condition 1. No value is found.
881 if (startpos == -1) {
882 return true;
883 }
884
885 // Find the highest value.
886 int endpos = find_highest(startpos + 1, size(), value);
887
888 // Condition 2. There is only one bit with the value (in startpos).
889 if (endpos == -1) {
890 return true;
891 }
892
893 // Count the number of elements set to value.
894 size_t value_count = count();
895 if (not value) {
896 value_count = size() - value_count;
897 }
898
899 // Condition 3. The number of elements must match with the start to end number of elements.
900 return (value_count == static_cast<size_t>((endpos + 1) - startpos));
901 }

989 /// \brief Converts the bitset to an array of packed bits. Each element of the resulting array will contain a bitmap.
990 /// The LowInfoBitIsMSB template parameter defines the order of bits in the resulting array.
991 /// \tparam UnsignedInteger Value type of the array where packed bits will be stored. It must be an unsigned integer.
992 /// \param[in] packed_bits Array where packed bits will be stored. The array size must be equal or larger than the
993 /// bitset size (in bits) divided by \c sizeof(UnsignedInteger) * 8U (the number of bits per integer).
994 /// \return Returns the number of positions of \c packed_bits that were written during the function call.
995 template <typename UnsignedInteger>
996 size_t to_packed_bits(span<UnsignedInteger> packed_bits) const
997 {
998 static_assert(sizeof(UnsignedInteger) <= sizeof(word_t), "ERROR: provided array type is too large");
999 static_assert(std::is_unsigned<UnsignedInteger>::value, "Only unsigned integers are supported");
1000 static constexpr size_t steps_per_word = sizeof(word_t) / sizeof(UnsignedInteger);
1001 static constexpr size_t bits_per_integer = sizeof(UnsignedInteger) * 8U;
1002 static constexpr word_t integer_mask = mask_lsb_ones<word_t>(bits_per_integer);
1003 const unsigned last_word_steps = divide_ceil(size() % bits_per_word, bits_per_integer);
1004 const unsigned nof_words = nof_words_();
1005 const unsigned nof_integers_packed = (nof_words - 1) * steps_per_word + last_word_steps;
1006 srsran_assert(
1007 packed_bits.size() >= nof_integers_packed, "ERROR: provided array size='{}' is too small", packed_bits.size());
1008
1009 size_t count = 0;
1010 if (not LowestInfoBitIsMSB) {
1011 for (unsigned i = 0; i != nof_words; ++i) {
1012 unsigned nof_steps = i == nof_words - 1 ? last_word_steps : steps_per_word;
1013 for (unsigned j = 0; j != nof_steps; ++j) {
1014 packed_bits[count++] = (buffer[i] >> (j * steps_per_word)) & integer_mask;
1015 }
1016 }
1017 } else {
1018 for (unsigned i = 0; i != nof_words; ++i) {
1019 word_t w = buffer[nof_words - i - 1];
1020 unsigned nof_steps = steps_per_word;
1021 if (i == nof_words - 1) {
1022 w <<= (bits_per_word - (size() % bits_per_word));
1023 nof_steps = last_word_steps;
1024 }
1025 for (unsigned j = 0; j != nof_steps; ++j) {
1026 packed_bits[count++] = (w >> (bits_per_word - (j + 1) * bits_per_integer)) & integer_mask;
1027 }
1028 }
1029 }
1030
1031 return nof_integers_packed;
1032 }

1034 /// \brief Converts the bitset to an array of unpacked bits, i.e. an array where each element represents a single
1035 /// boolean. The order of bits in the resulting array matches the bit information order in the bitset and the template
1036 /// parameter \c LowInfoBitIsMSB has no effect. That means that \c unpacked_bits[i] will be equal to \c
1037 /// bitset.test(i).
1038 ///
1039 /// \tparam UnsignedInteger Value type of the array where packed bits will be stored. It must be an
1040 /// unsigned integer. \param[in] unpacked_bits Array where the unpacked bits will be stored. The array size must be
1041 /// equal or larger than the bitset size (in bits). \return Returns the number of bits packed.
1042 template <typename UnsignedInteger>
1043 void to_unpacked_bits(span<UnsignedInteger> unpacked_bits) const
1044 {
1045 static_assert(std::is_unsigned<UnsignedInteger>::value, "Only unsigned integers are supported");
1046 srsran_assert(size() == unpacked_bits.size(),
1047 "ERROR: provided array size='{}' does not match bitset size='{}'",
1048 unpacked_bits.size(),
1049 size());
1050
1051 for (unsigned i = 0, ie = size(); i != ie; ++i) {
1052 unpacked_bits[i] = test(i);
1053 }
1054 }

1056 /// \brief Generates a list of bit positions corresponding to the information bits set to one or zero.
1057 ///
1058 /// The bit positions correspond to the location of each bit within the information bit word stored in the bitset,
1059 /// regardless of the bit index order in memory given by \ref LowestInfoBitIsMSB.
1060 ///
1061 /// \param[in] value Selects the bits whose positions are returned. Set to \c true to find ones, \c false for zeros.
1062 /// \return A list containing the bit positions.
1063 static_vector<size_t, N> get_bit_positions(bool value = true) const
1064 {
1065 static_vector<size_t, N> positions;
1066
1067 size_t i_bit = 0;
1068 while (i_bit < size()) {
1069 // Find the next bit position of the bit set to value.
1070 int next_position = find_lowest(i_bit, size(), value);
1071 if (next_position < 0) {
1072 break;
1073 }
1074
1075 // If a bit was found, add to the list.
1076 positions.emplace_back(static_cast<size_t>(next_position));
1077
1078 // Exclude the evaluated bit range from the next search.
1079 i_bit = next_position + 1;
1080 }
1081
1082 return positions;
1083 }

1176 int find_last_(size_t startpos, size_t endpos, bool value) const noexcept
1177 {
1178 size_t startword = startpos / bits_per_word;
1179 size_t lastword = (endpos - 1) / bits_per_word;
1180
1181 for (size_t i = lastword; i != startword - 1; --i) {
1182 word_t w = buffer[i];
1183 if (not value) {
1184 w = ~w;
1185 }
1186
1187 if (i == startword) {
1188 size_t offset = startpos % bits_per_word;
1189 w &= (LowestInfoBitIsMSB) ? mask_msb_zeros<word_t>(offset) : mask_lsb_zeros<word_t>(offset);
1190 }
1191 if (i == lastword) {
1192 size_t offset = (endpos - 1) % bits_per_word;
1193 w &= (LowestInfoBitIsMSB) ? mask_msb_ones<word_t>(offset + 1) : mask_lsb_ones<word_t>(offset + 1);
1194 }
1195 if (w != 0) {
1196 return static_cast<int>(i * bits_per_word + find_first_msb_one(w));
1197 }
1198 }
1199 return -1;
1200 }

1242 w &= mask_lsb_ones<word_t>(offset + 1);

1404/// \brief Divides a bitset of size "S" into "M" smaller bitsets, where each bitset has length "L=S/M". A bitwise-or
1405/// operation is performed across bitsets. At the end, a slice with an offset "O" and length "K" is taken from the
1406/// bitset of length "L" that resulted from the bitwise-or operation.
1407/// This operation is equivalent to reshaping an array of bits of size "S" into a matrix of dimensions "(M, L)" and
1408/// applying an "or" operation across all bits of each column. The resulting array of "L" bits, is then sliced with
1409/// an offset "O" and length "K".
1410/// The operation asserts if "S % L != 0".
1411/// E.g. Consider the bitset 1000 0100 0000 1001 (S=16), L=4, O=1, K=2. This function performs the following steps:
1412/// 1. Break the bitset into M=S/L=4 parts: {1000, 0100, 0000, 1001}.
1413/// 2. Bitwise-or all the M parts: 1101.
1414/// 3. Slice the bitset obtained in 2. with offset O=1 and slice length K=2: 10.
1415///
1416/// \tparam N2 maximum bitset size for returned bitset.
1417/// \param[in] other original bitset of length "S".
1418/// \param[in] fold_length length of each folded bitset "L".
1419/// \param[in] slice_offset offset from where to slice each fold "O".
1420/// \param[in] slice_length length of the slice taken from each fold "K".
1421/// \return bitset of size slice_length with the or-accumulated folds.
1422template <size_t N2, size_t N, bool LowestInfoBitIsMSB>
1423inline bounded_bitset<N2, LowestInfoBitIsMSB> fold_and_accumulate(const bounded_bitset<N, LowestInfoBitIsMSB>& other,
1424 size_t fold_length,
1425 size_t slice_offset,
1426 size_t slice_length) noexcept
1427{
1428 srsran_assert(
1429 other.size() % fold_length == 0, "Invalid fold length={} for bitset of size={}", fold_length, other.size());
1430 bounded_bitset<N2, LowestInfoBitIsMSB> ret(slice_length);
1431 for (size_t i = 0; i != other.size(); i += fold_length) {
1432 ret |= other.template slice<N2>(i + slice_offset, i + slice_offset + slice_length);
1433 }
1434 return ret;
1435}
