67// Specialization for lockfree SPSC without blocking mechanism.
68template <typename T>
69class queue_impl<T, concurrent_queue_policy::lockfree_spsc, concurrent_queue_wait_policy::non_blocking>
70{
71 struct custom_deleter {
72 void operator()(::rigtorp::SPSCQueue<T>* ptr) const
73 {
74 using namespace ::rigtorp;
75 if (ptr != nullptr) {
76 ptr->~SPSCQueue<T>();
77 free(ptr);
78 }
79 }
80 };
81
82public:
83 template <typename... Args>
84 explicit queue_impl(size_t qsize)
85 {
86 // Note: Pre-c++17 does not support new with alignof > alignof(max_align_t).
87 void* ptr = nullptr;
88 int ret = posix_memalign(&ptr, alignof(::rigtorp::SPSCQueue<T>), sizeof(::rigtorp::SPSCQueue<T>));
89 report_error_if_not(ret == 0, "Unable to allocate memory for SPSCQueue");
90 queue.reset(new (ptr)::rigtorp::SPSCQueue<T>(qsize));
91 }
92
93 template <typename U>
94 bool try_push(U&& elem)
95 {
96 return queue->try_push(std::forward<U>(elem));
97 }
98
99 bool try_pop(T& elem)
100 {
101 T* front = queue->front();
102 if (front != nullptr) {
103 elem = std::move(*front);
104 queue->pop();
105 return true;
106 }
107 return false;
108 }
109
110 template <typename PoppingFunc>
111 bool try_call_on_pop(PoppingFunc&& func)
112 {
113 T* front = queue->front();
114 if (front != nullptr) {
115 func(*front);
116 queue->pop();
117 return true;
118 }
119 return false;
120 }
121
122 size_t size() const { return queue->size(); }
123
124 bool empty() const { return queue->empty(); }
125
126 size_t capacity() const { return queue->capacity(); }
127
128protected:
129 std::unique_ptr<::rigtorp::SPSCQueue<T>, custom_deleter> queue;
130};

134class queue_impl<T, concurrent_queue_policy::lockfree_spsc, concurrent_queue_wait_policy::sleep>
135 : public queue_impl<T, concurrent_queue_policy::lockfree_spsc, concurrent_queue_wait_policy::non_blocking>

138
139public:
140 template <typename... Args>
141 explicit queue_impl(size_t qsize, std::chrono::microseconds sleep_time_) :
142 queue_impl<T, concurrent_queue_policy::lockfree_spsc, concurrent_queue_wait_policy::non_blocking>(qsize),
143 sleep_time(sleep_time_)
144 {
145 }
146
147 void request_stop() { running = false; }
148
149 template <typename U>
150 bool push_blocking(U&& elem)
151 {
152 while (running.load(std::memory_order_relaxed)) {
153 if (this->try_push(std::forward<U>(elem))) {
154 return true;
155 }
156 std::this_thread::sleep_for(sleep_time);
157 }
158 return false;
159 }
160
161 bool pop_blocking(T& elem) noexcept
162 {
163 T* f = front_blocking();
164 if (f != nullptr) {
165 elem = std::move(*f);
166 this->queue->pop();
167 return true;
168 }
169 return false;
170 }
171
172 template <typename PoppingFunc>
173 bool call_on_pop_blocking(PoppingFunc&& func)
174 {
175 T* f = front_blocking();
176 if (f != nullptr) {
177 func(*f);
178 this->queue->pop();
179 return true;
180 }
181 return false;
182 }
183
184private:
185 T* front_blocking()
186 {
187 while (running.load(std::memory_order_relaxed)) {
188 T* front = this->queue->front();
189 if (front != nullptr) {
190 return front;
191 }
192 std::this_thread::sleep_for(sleep_time);
193 }
194 return nullptr;
195 }
196
197 std::chrono::microseconds sleep_time;
198 std::atomic<bool> running{true};

392// Barrier implementation based on sleep.
393struct queue_sleep_barrier {
394public:
395 queue_sleep_barrier(std::chrono::microseconds sleep_time_ = std::chrono::microseconds{0}) : sleep_time(sleep_time_) {}
396
397 void request_stop() { running = false; }
398
399 void notify_push()
400 {
401 // do nothing.
402 }
403
404 void notify_pop()
405 {
406 // do nothing.
407 }
408
409 template <typename HasSpace>
410 void wait_push(std::unique_lock<std::mutex>& lock, const HasSpace& has_space)
411 {
412 while (running.load(std::memory_order_relaxed) and not has_space()) {
413 lock.unlock();
414 std::this_thread::sleep_for(sleep_time);
415 lock.lock();
416 }
417 }
418
419 template <typename HasElem>
420 void wait_pop(std::unique_lock<std::mutex>& lock, const HasElem& has_elem)
421 {
422 while (running.load(std::memory_order_relaxed) and not has_elem()) {
423 lock.unlock();
424 std::this_thread::sleep_for(sleep_time);
425 lock.lock();
426 }
427 }
428
429 bool is_running() const { return running.load(std::memory_order_relaxed); }
430
431private:
432 std::atomic<bool> running{true};
433 std::chrono::microseconds sleep_time;
434};

436// Barrier implementation based on condition variable.
437struct queue_cond_var_barrier {
438public:
439 void request_stop()
440 {
441 running = false;
442 notify_push();
443 notify_pop();
444 }
445
446 void notify_push() { cvar_push.notify_one(); }
447
448 void notify_pop() { cvar_pop.notify_one(); }
449
450 template <typename HasSpace>
451 void wait_push(std::unique_lock<std::mutex>& lock, const HasSpace& has_space)
452 {
453 cvar_pop.wait(lock, [this, &has_space]() { return not running.load(std::memory_order_relaxed) or has_space(); });
454 }
455
456 template <typename HasElem>
457 void wait_pop(std::unique_lock<std::mutex>& lock, const HasElem& has_elem)
458 {
459 cvar_push.wait(lock, [this, &has_elem]() { return not running.load(std::memory_order_relaxed) or has_elem(); });
460 }
461
462 bool is_running() const { return running.load(std::memory_order_relaxed); }
463
464private:
465 std::atomic<bool> running{true};
466 std::condition_variable cvar_pop, cvar_push;
467};

469// Specialization for lock-based MPSC, using a condition variable or sleep as blocking mechanism. The dequeues are
470// done in a batch to minimize mutex contention.
471template <typename T, concurrent_queue_wait_policy BlockingPolicy>
472class queue_impl<T, concurrent_queue_policy::locking_mpsc, BlockingPolicy>
473{
474 using queue_barrier = std::conditional_t<BlockingPolicy == concurrent_queue_wait_policy::condition_variable,
475 queue_cond_var_barrier,
476 queue_sleep_barrier>;
477
478public:
479 template <typename... Args>
480 explicit queue_impl(size_t qsize, Args&&... args) :
481 cap(qsize),
482 queues({ring_buffer<T, true>{(unsigned)qsize}, ring_buffer<T, true>{(unsigned)qsize}}),
483 barrier(std::forward<Args>(args)...)
484 {
485 }
486
487 void request_stop()
488 {
489 std::lock_guard<std::mutex> lock(mutex);
490 pushing_queue().clear();
491 barrier.request_stop();
492 }
493
494 template <typename U>
495 bool try_push(U&& elem)
496 {
497 std::unique_lock<std::mutex> lock(mutex);
498 if (barrier.is_running() and not pushing_queue().full()) {
499 pushing_queue().push(std::forward<U>(elem));
500 barrier.notify_push();
501 return true;
502 }
503 return false;
504 }
505
506 template <typename U>
507 bool push_blocking(U&& elem)
508 {
509 std::unique_lock<std::mutex> lock(mutex);
510 barrier.wait_push(lock, [this]() { return not pushing_queue().full(); });
511 if (barrier.is_running()) {
512 pushing_queue().push(std::forward<U>(elem));
513 barrier.notify_push();
514 return true;
515 }
516 return false;
517 }
518
519 bool try_pop(T& elem)
520 {
521 T* f = front<false>();
522 if (f != nullptr) {
523 elem = std::move(*f);
524 pop();
525 return true;
526 }
527 return false;
528 }
529
530 bool pop_blocking(T& elem) noexcept
531 {
532 T* f = front<true>();
533 if (f != nullptr) {
534 elem = std::move(*f);
535 pop();
536 return true;
537 }
538 return false;
539 }
540
541 template <typename PoppingFunc>
542 bool try_call_on_pop(PoppingFunc&& func)
543 {
544 T* f = front<false>();
545 if (f != nullptr) {
546 func(*f);
547 pop();
548 return true;
549 }
550 return false;
551 }
552
553 template <typename PoppingFunc>
554 bool call_on_pop_blocking(PoppingFunc&& func)
555 {
556 T* f = front<true>();
557 if (f != nullptr) {
558 func(*f);
559 pop();
560 return true;
561 }
562 return false;
563 }
564
565 size_t size() const
566 {
567 std::lock_guard<std::mutex> lock(mutex);
568 return pushing_queue().size() + count_local_objs.load(std::memory_order_relaxed);
569 }
570
571 bool empty() const
572 {
573 if (count_local_objs.load(std::memory_order_relaxed) > 0) {
574 return false;
575 }
576 std::lock_guard<std::mutex> lock(mutex);
577 return pushing_queue().empty();
578 }
579
580 size_t capacity() const { return cap; }
581
582private:
583 template <bool Blocking>
584 T* front()
585 {
586 if (not barrier.is_running()) {
587 return nullptr;
588 }
589 if (not popping_queue().empty()) {
590 // Successful pop.
591 return &popping_queue().top();
592 }
593 {
594 std::unique_lock<std::mutex> lock(mutex);
595 if (Blocking) {
596 barrier.wait_pop(lock, [this]() { return !pushing_queue().empty(); });
597 if (not barrier.is_running()) {
598 return nullptr;
599 }
600 } else if (pushing_queue().empty()) {
601 return nullptr;
602 }
603 // Swap queues.
604 index_queue_for_pop = 1 - index_queue_for_pop;
605 }
606 barrier.notify_pop();
607 count_local_objs.store(popping_queue().size(), std::memory_order_relaxed);
608 return &popping_queue().top();
609 }
610
611 void pop()
612 {
613 popping_queue().pop();
614 count_local_objs.fetch_sub(1, std::memory_order_relaxed);
615 }
616
617 ring_buffer<T, true>& popping_queue() { return queues[index_queue_for_pop]; }
618 ring_buffer<T, true>& pushing_queue() { return queues[1 - index_queue_for_pop]; }
619 const ring_buffer<T, true>& pushing_queue() const { return queues[1 - index_queue_for_pop]; }
620
621 const size_t cap;
622 std::atomic<unsigned> count_local_objs{0};
623 std::array<ring_buffer<T, true>, 2> queues;
624 unsigned index_queue_for_pop = 0;
625 mutable std::mutex mutex;
626 queue_barrier barrier;
627};

744static constexpr size_t enqueue_priority_to_queue_index(enqueue_priority prio, size_t nof_priority_levels)
745{
746 if (nof_priority_levels == 0) {
747 return 0;
748 }
749 size_t queue_idx = std::numeric_limits<size_t>::max() - static_cast<size_t>(prio);
750 return queue_idx < nof_priority_levels ? queue_idx : nof_priority_levels - 1;
751}

754{
755 return static_cast<enqueue_priority>(std::numeric_limits<size_t>::max() - queue_idx);
756}
