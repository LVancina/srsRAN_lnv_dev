102 described_segment(const codeblock_metadata& metadata_, units::bits cb_size_) : metadata(metadata_)
103 {
104 using namespace units::literals;
105 srsran_assert((cb_size_ > 0_bits) && (cb_size_ <= MAX_SEG_LENGTH),
106 "The codeblock size (i.e., {}) is invalid (max. {}).",
107 cb_size_,
108 MAX_SEG_LENGTH);
109 data.resize(cb_size_.value());
110 }
111
114 bit_buffer& get_data()
115 {
116 srsran_assert(!data.is_empty(), "The segment has not been constructed.");
117 return data;
118 }
119
122 const bit_buffer& get_data() const
123 {
124 srsran_assert(!data.is_empty(), "The segment has not been constructed.");
125 return data;
126 }
127
129 const codeblock_metadata& get_metadata() const { return metadata; }
130
131private:
133 codeblock_metadata metadata;
135 static_bit_buffer<MAX_SEG_LENGTH.value()> data;
136};
