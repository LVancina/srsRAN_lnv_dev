32struct srb_pdcp_context {
33 std::unique_ptr<pdcp_entity> entity;
34 pdcp_rrc_ue_tx_adapter pdcp_tx_notifier;
35 pdcp_tx_control_rrc_ue_adapter rrc_tx_control_notifier;
36 pdcp_rrc_ue_rx_adapter rrc_rx_data_notifier;
37 pdcp_rx_control_rrc_ue_adapter rrc_rx_control_notifier;
38
39 srb_pdcp_context(const ue_index_t ue_index, const srb_id_t srb_id, timer_factory timers)
40 {
41 // prepare PDCP creation message
42 pdcp_entity_creation_message srb_pdcp{};
43 srb_pdcp.ue_index = ue_index_to_uint(ue_index);
44 srb_pdcp.rb_id = srb_id;
45 srb_pdcp.config = pdcp_make_default_srb_config(); // TODO: allow non-default PDCP SRB configs
46 srb_pdcp.tx_lower = &pdcp_tx_notifier;
47 srb_pdcp.tx_upper_cn = &rrc_tx_control_notifier;
48 srb_pdcp.rx_upper_dn = &rrc_rx_data_notifier;
49 srb_pdcp.rx_upper_cn = &rrc_rx_control_notifier;
50 srb_pdcp.timers = timers;
51
52 // create PDCP entity
53 entity = create_pdcp_entity(srb_pdcp);
54 }
55};

66 // Setup security
67 void enable_tx_security(security::integrity_enabled int_enabled,
68 security::ciphering_enabled ciph_enabled,
69 security::sec_128_as_config sec_cfg) const
70 {
71 // Configure tx security
72 auto& pdcp_tx_ctrl = pdcp_context.entity->get_tx_upper_control_interface();
73 pdcp_tx_ctrl.configure_security(sec_cfg);
74 pdcp_tx_ctrl.set_integrity_protection(int_enabled);
75 pdcp_tx_ctrl.set_ciphering(ciph_enabled);
76 }
77
78 void enable_rx_security(security::integrity_enabled int_enabled,
79 security::ciphering_enabled ciph_enabled,
80 security::sec_128_as_config sec_cfg) const
81 {
82 // Configure tx security
83 auto& pdcp_rx_ctrl = pdcp_context.entity->get_rx_upper_control_interface();
84 pdcp_rx_ctrl.configure_security(sec_cfg);
85 pdcp_rx_ctrl.set_integrity_protection(int_enabled);
86 pdcp_rx_ctrl.set_ciphering(ciph_enabled);
87 }
88
89 // Full security setup. Used e.g., with SRB2
90 void enable_full_security(security::sec_128_as_config sec_cfg) const
91 {
92 // Configure rx security
93 auto& pdcp_tx_ctrl = pdcp_context.entity->get_tx_upper_control_interface();
94 pdcp_tx_ctrl.configure_security(sec_cfg);
95 pdcp_tx_ctrl.set_integrity_protection(security::integrity_enabled::on);
96 pdcp_tx_ctrl.set_ciphering(security::ciphering_enabled::on);
97
98 // Configure tx security
99 auto& pdcp_rx_ctrl = pdcp_context.entity->get_rx_upper_control_interface();
100 pdcp_rx_ctrl.configure_security(sec_cfg);
101 pdcp_rx_ctrl.set_integrity_protection(security::integrity_enabled::on);
102 pdcp_rx_ctrl.set_ciphering(security::ciphering_enabled::on);
103 }
104
105 // Add ciphering and integrity protection to an RRC PDU.
106 byte_buffer pack_rrc_pdu(byte_buffer rrc_pdu)
107 {
108 pdcp_context.entity->get_tx_upper_data_interface().handle_sdu(std::move(rrc_pdu));
109 return pdcp_context.pdcp_tx_notifier.get_pdcp_pdu();
110 }
111
112 // Decipher and verify integrity of an PDCP PDU.
113 byte_buffer unpack_pdcp_pdu(byte_buffer pdcp_pdu)
114 {
115 pdcp_context.entity->get_rx_lower_interface().handle_pdu(byte_buffer_chain{std::move(pdcp_pdu)});
116 return pdcp_context.rrc_rx_data_notifier.get_rrc_pdu();
117 }
118
119private:
120 srb_pdcp_context pdcp_context;
121};
