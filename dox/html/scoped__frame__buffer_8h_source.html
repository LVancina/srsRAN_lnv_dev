33class scoped_frame_buffer
34{
35 ether::eth_frame_pool& frame_pool;
36 const ether::frame_pool_context context;
37 const span<ether::frame_buffer> frames;
38
39public:
41 scoped_frame_buffer(ether::eth_frame_pool& frame_pool_,
42 slot_symbol_point symbol_point,
43 message_type type,
44 data_direction direction) :
45 frame_pool(frame_pool_), context({{type, direction}, symbol_point}), frames(frame_pool.get_frame_buffers(context))
46 {
47 }
48
50 ether::frame_buffer& get_next_frame()
51 {
52 for (auto& frame : frames) {
53 if (frame.empty()) {
54 return frame;
55 }
56 }
57
58 srsran_terminate("No empty Ethernet frame available in slot '{}' symbol '{}'\n",
59 context.symbol_point.get_slot(),
60 context.symbol_point.get_symbol_index());
61 }
62
63 bool empty() const { return frames.empty(); }
64
66 ~scoped_frame_buffer()
67 {
68 if (!frames.empty()) {
69 frame_pool.push_frame_buffers(context, frames);
70 }
71 }
72};
