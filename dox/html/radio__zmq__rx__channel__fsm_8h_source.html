23#pragma once
24
25#include "radio_zmq_timer.h"
26#include <condition_variable>
27#include <mutex>
28
29namespace srsran {
30class radio_zmq_rx_channel_fsm
31{
32private:
34 enum class states {
36 UNINITIALIZED = 0,
38 SEND_REQUEST,
40 RECEIVE_DATA,
42 WAIT_STOP,
44 STOPPED
45 };
46
48 states state = states::UNINITIALIZED;
50 mutable std::mutex mutex;
52 std::condition_variable cvar;
54 radio_zmq_timer timer;
55
57 bool is_running_unprotected() const { return state == states::SEND_REQUEST || state == states::RECEIVE_DATA; }
58
59public:
61 void init_successful()
62 {
63 std::unique_lock<std::mutex> lock(mutex);
64 state = states::SEND_REQUEST;
65 timer.start();
66 }
67
69 void on_error()
70 {
71 std::unique_lock<std::mutex> lock(mutex);
72 state = states::STOPPED;
73 cvar.notify_all();
74 }
75
77 void request_sent()
78 {
79 std::unique_lock<std::mutex> lock(mutex);
80 if (state == states::SEND_REQUEST) {
81 state = states::RECEIVE_DATA;
82 timer.start();
83 }
84 }
85
87 void data_received()
88 {
89 std::unique_lock<std::mutex> lock(mutex);
90 if (state == states::RECEIVE_DATA) {
91 state = states::SEND_REQUEST;
92 timer.start();
93 }
94 }
95
97 void stop()
98 {
99 std::unique_lock<std::mutex> lock(mutex);
100 if (is_running_unprotected()) {
101 state = states::WAIT_STOP;
102 }
103 }
104
106 void wait_stop()
107 {
108 std::unique_lock<std::mutex> lock(mutex);
109 while (state != states::STOPPED) {
110 cvar.wait(lock);
111 }
112 }
113
115 void async_task_stopped()
116 {
117 std::unique_lock<std::mutex> lock(mutex);
118 state = states::STOPPED;
119 cvar.notify_all();
120 }
121
123 bool is_running() const
124 {
125 std::unique_lock<std::mutex> lock(mutex);
126 return is_running_unprotected();
127 }
128
130 bool has_pending_response() const
131 {
132 std::unique_lock<std::mutex> lock(mutex);
133 return state == states::RECEIVE_DATA;
134 }
135
145 bool has_wait_expired()
146 {
147 std::unique_lock<std::mutex> lock(mutex);
148
149 // It can only be expired if it is currently running.
150 if (!is_running_unprotected()) {
151 return false;
152 }
153
154 // Check if the timer has not expired.
155 if (!timer.is_expired()) {
156 return false;
157 }
158
159 // Update timer with new time.
160 timer.start();
161 return true;
162 }
163};
164} // namespace srsran
