34class sequence_id_checker_impl : public sequence_id_checker
35{
36 static constexpr int NOF_SEQUENCES_IDENTIFIERS = 1u << 8;
37 static constexpr int HALF_NOF_SEQUENCES_IDENTIFIERS = NOF_SEQUENCES_IDENTIFIERS / 2;
38
39 bounded_bitset<MAX_SUPPORTED_EAXC_ID_VALUE> initialized;
40 circular_map<unsigned, uint8_t, MAX_SUPPORTED_EAXC_ID_VALUE> counters;
41
42public:
44 sequence_id_checker_impl() : initialized(MAX_SUPPORTED_EAXC_ID_VALUE)
45 {
46 for (unsigned K = 0; K != MAX_SUPPORTED_EAXC_ID_VALUE; ++K) {
47 counters.insert(K, 0);
48 }
49 }
50
51 // See interface for documentation.
52 int update_and_compare_seq_id(unsigned eaxc, uint8_t seq_id) override
53 {
54 srsran_assert(eaxc < MAX_SUPPORTED_EAXC_ID_VALUE,
55 "Invalid eAxC value '{}'. Maximum eAxC value is '{}'",
56 eaxc,
57 MAX_SUPPORTED_EAXC_ID_VALUE);
58
59 // First packet is always valid.
60 if (!initialized.test(eaxc)) {
61 initialized.set(eaxc);
62 counters[eaxc] = seq_id;
63
64 return 0;
65 }
66
67 // Get the expected sequence identifier and update its value.
68 uint8_t expected_seq_id = counters[eaxc] + 1;
69
70 if (seq_id == expected_seq_id) {
71 ++counters[eaxc];
72
73 return 0;
74 }
75
76 int nof_skipped_seq_id = get_nof_skipped_sequence_id(seq_id, expected_seq_id);
77
78 // Update the expected sequence identifier when the sequence identifier is from the future.
79 if (nof_skipped_seq_id > 0) {
80 counters[eaxc] = seq_id;
81 }
82
83 return nof_skipped_seq_id;
84 }
85
86private:
93 static int get_nof_skipped_sequence_id(uint8_t seq_id, uint8_t expected_seq_id)
94 {
95 int a = static_cast<int>(seq_id) - static_cast<int>(expected_seq_id);
96 if (a >= HALF_NOF_SEQUENCES_IDENTIFIERS) {
97 return a - NOF_SEQUENCES_IDENTIFIERS;
98 }
99 if (a < -HALF_NOF_SEQUENCES_IDENTIFIERS) {
100 return a + NOF_SEQUENCES_IDENTIFIERS;
101 }
102 return a;
103 }
104};
