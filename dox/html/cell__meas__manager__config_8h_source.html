85// Cell meas config formatter
86template <>
87struct formatter<srsran::srs_cu_cp::cell_meas_config> {
88 template <typename ParseContext>
89 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
90 {
91 return ctx.begin();
92 }
93
94 template <typename FormatContext>
95 auto format(srsran::srs_cu_cp::cell_meas_config cfg, FormatContext& ctx)
96 -> decltype(std::declval<FormatContext>().out())
97 {
98 std::string ncell_str = "[ ";
99 for (const auto& ncell : cfg.ncells) {
100 ncell_str = ncell_str + std::to_string(ncell.nci) + " ";
101 }
102 ncell_str = ncell_str + "]";
103
104 return format_to(
105 ctx.out(),
106 "nci={} complete={} band={} ssb_arfcn={} ssb_scs={} ncells={}",
107 cfg.serving_cell_cfg.nci,
108 is_complete(cfg.serving_cell_cfg) ? "yes" : "no",
109 cfg.serving_cell_cfg.band.has_value() ? to_string(nr_band_to_uint(cfg.serving_cell_cfg.band.value())) : "?",
110 cfg.serving_cell_cfg.ssb_arfcn.has_value() ? to_string(cfg.serving_cell_cfg.ssb_arfcn.value()) : "?",
111 cfg.serving_cell_cfg.ssb_scs.has_value() ? to_string(cfg.serving_cell_cfg.ssb_scs.value()) : "?",
112 ncell_str);
113 }
114};
