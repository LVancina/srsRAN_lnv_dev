46class delimited_formatter
47{
48public:
50 delimited_formatter()
51 {
52 static const fmt::string_view DEFAULT_FORMAT = "{}";
53 static const fmt::string_view DEFAULT_DELIMITER = " ";
54 format_buffer.append(DEFAULT_FORMAT.begin(), DEFAULT_FORMAT.end());
55 delimiter_buffer.append(DEFAULT_DELIMITER.begin(), DEFAULT_DELIMITER.end());
56 }
57
59 explicit delimited_formatter(const std::string& default_delimiter)
60 {
61 static const fmt::string_view DEFAULT_FORMAT = "{}";
62 static const fmt::string_view DEFAULT_DELIMITER = default_delimiter;
63 format_buffer.append(DEFAULT_FORMAT.begin(), DEFAULT_FORMAT.end());
64 delimiter_buffer.append(DEFAULT_DELIMITER.begin(), DEFAULT_DELIMITER.end());
65 }
66
76 template <typename ParseContext>
77 auto parse(ParseContext& context)
78 {
79 // Set the first field indicator.
80 first = true;
81
82 static const fmt::string_view PREAMBLE_FORMAT = "{:";
83 static const fmt::string_view NEWLINE_DELIMITER = "\n ";
84
85 // Skip if context is empty and use default format.
86 if (context.begin() == context.end()) {
87 return context.end();
88 }
89
90 format_buffer.clear();
91 format_buffer.append(PREAMBLE_FORMAT.begin(), PREAMBLE_FORMAT.end());
92
93 for (auto& it : context) {
94 switch (it) {
95 case 'n':
96 // New line delimiter.
97 delimiter_buffer.clear();
98 delimiter_buffer.append(NEWLINE_DELIMITER.begin(), NEWLINE_DELIMITER.end());
99 break;
100 case 's':
101 // Short representation.
102 verbose = false;
103 break;
104 case '}':
105 format_buffer.push_back(it);
106 return &it;
107 }
108 // Propagate formatting to underlying structures.
109 format_buffer.push_back(it);
110 }
111 // No end of context was found.
112 return context.end();
113 }

196private:
198 template <typename FormatContext, typename... Args>
199 void format_to(FormatContext& context, const char* format, Args&&... args)
200 {
201 if (!first) {
202 // Buffer to hold the formatted string.
203 fmt::memory_buffer temp_buffer;
204 fmt::format_to(temp_buffer, format, std::forward<Args>(args)...);
205
206 if (temp_buffer.size() > 0) {
207 // Prepend delimiter to the formatted output.
208 fmt::format_to(context.out(), "{}", fmt::string_view(delimiter_buffer.data(), delimiter_buffer.size()));
209 }
210
211 // Append the formatted string to the context iterator.
212 fmt::format_to(context.out(), "{}", fmt::string_view(temp_buffer.data(), temp_buffer.size()));
213
214 return;
215 }
216 // Format without prepending delimiter.
217 fmt::format_to(context.out(), format, std::forward<Args>(args)...);
218 first = false;
219 }
220
222 template <typename FormatContext, typename... Args>
223 void format_to(FormatContext& context, Args&&... args)
224 {
225 if (!first) {
226 // Buffer to hold the formatted string.
227 fmt::memory_buffer temp_buffer;
228 fmt::format_to(
229 temp_buffer, fmt::string_view(format_buffer.data(), format_buffer.size()), std::forward<Args>(args)...);
230
231 if (temp_buffer.size() > 0) {
232 // Prepend delimiter to the formatted output.
233 fmt::format_to(context.out(), "{}", fmt::string_view(delimiter_buffer.data(), delimiter_buffer.size()));
234 }
235
236 // Append the formatted string to the context iterator.
237 fmt::format_to(context.out(), "{}", fmt::string_view(temp_buffer.data(), temp_buffer.size()));
238
239 return;
240 }
241 // Format without prepending delimiter.
242 fmt::format_to(
243 context.out(), fmt::string_view(format_buffer.data(), format_buffer.size()), std::forward<Args>(args)...);
244 first = false;
245 }
246
248 bool first = true;
250 bool verbose = true;
251
253 fmt::memory_buffer delimiter_buffer;
255 fmt::memory_buffer format_buffer;
256};
