223class unique_timer
224{
225 using state_t = timer_manager::state_t;
226
227public:
228 unique_timer() = default;
229 unique_timer(const unique_timer&) = delete;
230 unique_timer& operator=(const unique_timer&) = delete;
231
232 unique_timer(unique_timer&& other) noexcept : handle(std::exchange(other.handle, nullptr)) {}
233 unique_timer& operator=(unique_timer&& other) noexcept
234 {
235 reset();
236 handle = std::exchange(other.handle, nullptr);
237 return *this;
238 }
239 ~unique_timer() { reset(); }
240
242 void reset()
243 {
244 if (is_valid()) {
245 handle->destroy();
246 handle = nullptr;
247 }
248 }
249
251 bool is_valid() const { return handle != nullptr; }
252
254 timer_id_t id() const { return is_valid() ? handle->id : timer_id_t::invalid; }
255
257 bool is_set() const { return is_valid() && handle->duration != timer_manager::INVALID_DURATION; }
258
260 bool is_running() const { return is_valid() && handle->state == timer_manager::state_t::running; }
261
263 bool has_expired() const { return is_valid() && handle->state == timer_manager::state_t::expired; }
264
266 timer_duration duration() const { return is_valid() ? handle->duration : timer_manager::INVALID_DURATION; }
267
269 void set(timer_duration duration, unique_function<void(timer_id_t)> callback)
270 {
271 srsran_assert(is_valid(), "Trying to setup empty timer pimpl");
272 handle->set(duration, std::move(callback));
273 }
274
276 void set(timer_duration duration)
277 {
278 srsran_assert(is_valid(), "Trying to setup empty timer pimpl");
279 handle->set(duration);
280 }
281
283 void run()
284 {
285 srsran_assert(is_valid(), "Starting invalid timer");
286 handle->run();
287 }
288
290 void stop()
291 {
292 if (is_running()) {
293 handle->stop();
294 }
295 }
296
297private:
298 friend class timer_manager;
299
300 explicit unique_timer(timer_manager::timer_frontend& impl) : handle(&impl) {}
301
302 timer_manager::timer_frontend* handle = nullptr;
303};
