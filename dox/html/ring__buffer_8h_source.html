58template <typename T, size_t N, bool ForcePower2Size>
59class ring_buffer_storage<std::array<T, N>, ForcePower2Size>
60{
61 static_assert(not ForcePower2Size or ((N & (N - 1)) == 0), "N must be a power of 2 if ForcePower2Size is set");
62
63public:
64 constexpr ring_buffer_storage() = default;
65 ring_buffer_storage(size_t , unsigned rpos_, unsigned count_) : rpos(rpos_), count(count_) {}
66 ring_buffer_storage(ring_buffer_storage&& other) noexcept :
67 rpos(std::exchange(other.rpos, 0)), count(std::exchange(other.count, 0))
68 {
69 static_assert(std::is_move_constructible<T>::value, "T must be move-constructible");
70 for (unsigned i = 0; i != count; ++i) {
71 unsigned idx = advance_pos(rpos, i);
72 data[idx].emplace(std::move(other.data[idx].get()));
73 other.data[idx].destroy();
74 }
75 }
76 ring_buffer_storage& operator=(ring_buffer_storage&& other) noexcept
77 {
78 for (unsigned i = 0; i != count; ++i) {
79 data[advance_pos(rpos, i)].destroy();
80 }
81 rpos = std::exchange(other.rpos, 0);
82 count = std::exchange(other.count, 0);
83 for (unsigned i = 0; i < count; ++i) {
84 unsigned idx = advance_pos(rpos, i);
85 data[idx].emplace(std::move(other.data[idx].get()));
86 other.data[idx].destroy();
87 }
88 return *this;
89 }
90
91 static constexpr size_t max_size() noexcept { return N; }
92
94 static unsigned advance_pos(unsigned pos, unsigned n) noexcept { return (pos + n) % max_size(); }
95
96 void set_size(unsigned sz)
97 {
98 // do nothing.
99 }
100
101 unsigned rpos = 0;
102 unsigned count = 0;
103 std::array<T, N> data;
104};

157template <typename T>
158class ring_buffer_storage<std::vector<T>, true> : private ring_buffer_storage<std::vector<T>, false>
159{
160 using super_type = ring_buffer_storage<std::vector<T>, false>;
161
162 static unsigned to_next_pow2(unsigned sz) { return pow(2, ceil(log(sz) / log(2))); }
163
164public:
165 using super_type::count;
166 using super_type::data;
167 using super_type::max_size;
168 using super_type::rpos;
169 using super_type::super_type;
170 using super_type::swap;
171 explicit ring_buffer_storage(unsigned sz) : super_type(to_next_pow2(sz)) {}
172
173 void set_size(unsigned sz)
174 {
175 // Round up to closest power of 2.
176 super_type::set_size(to_next_pow2(sz));
177 }
178
181 unsigned advance_pos(unsigned pos, unsigned n) const noexcept
182 {
183 // x mod y == x * (y - 1), when y is power of 2.
184 return (pos + n) & (this->max_size() - 1);
185 }
