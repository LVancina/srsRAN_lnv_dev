39template <typename Type>
40class concurrent_thread_local_object_pool
41{
43 using queue_type = concurrent_queue<std::unique_ptr<Type>,
44 concurrent_queue_policy::lockfree_mpmc,
45 concurrent_queue_wait_policy::sleep>;
46
47public:
51 concurrent_thread_local_object_pool(std::vector<std::unique_ptr<Type>> instances) :
52 main_thread_id(std::this_thread::get_id()), queue(instances.size())
53 {
54 srsran_assert(!instances.empty(), "Pool is empty.");
55 srsran_assert(std::none_of(instances.begin(), instances.end(), [](auto& element) { return !element; }),
56 "Invalid instance in pool.");
57
58 for (std::unique_ptr<Type>& instance : instances) {
59 queue.try_push(std::move(instance));
60 }
61 }
62
68 template <typename... Args>
69 concurrent_thread_local_object_pool(unsigned nof_elements, Args&&... args) : queue(nof_elements)
70 {
71 for (unsigned i_elem = 0; i_elem != nof_elements; ++i_elem) {
72 bool success = queue.try_push(std::make_unique<Type>(std::forward<Args>(args)...));
73 report_error_if_not(success, "Failed to push element into the queue.");
74 }
75 }
76
78 ~concurrent_thread_local_object_pool()
79 {
80 if (main_thread_ptr) {
81 report_error_if_not(queue.try_push(std::move(main_thread_ptr)),
82 "Failed to push element into the queue (destructor).");
83 }
84 report_fatal_error_if_not(queue.size() == queue.capacity(), "Not all instances are returned to the pool.");
85 }
86
88 class custom_deleter
89 {
90 public:
91 custom_deleter(queue_type& queue_) : queue(&queue_) {}
92
93 void operator()(Type* ptr)
94 {
95 // Ignore pointer if it is invalid.
96 if (ptr == nullptr) {
97 return;
98 }
99
100 bool success = queue->push_blocking(std::unique_ptr<Type>(ptr));
101 report_error_if_not(success, "Failed to push element into the queue (deleter).");
102 }
103
104 private:
105 queue_type* queue;
106 };
107
109 using unique_ptr = std::unique_ptr<Type, custom_deleter>;
110
114 Type& get()
115 {
116 // Checks if the thread is the main.
117 if (std::this_thread::get_id() == main_thread_id) {
118 if (!main_thread_ptr) {
119 report_fatal_error_if_not(queue.try_pop(main_thread_ptr), "Insufficient number of instances (main thread).");
120 }
121 return *main_thread_ptr;
122 }
123
124 thread_local unique_ptr ptr = [this]() {
125 std::unique_ptr<Type> std_ptr;
126 report_fatal_error_if_not(queue.try_pop(std_ptr), "Insufficient number of instances.");
127
128 unique_ptr custom_ptr(std_ptr.get(), custom_deleter(queue));
129
130 std_ptr.release();
131
132 return custom_ptr;
133 }();
134
135 return *ptr;
136 }
137
139 std::size_t capacity() const { return queue.capacity(); }
140
141private:
143 std::thread::id main_thread_id;
145 std::unique_ptr<Type> main_thread_ptr;
147 queue_type queue;
148};
