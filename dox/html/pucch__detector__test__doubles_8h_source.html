32class pucch_detector_spy : public pucch_detector
33{
34public:
35 struct entry_format1 {
36 const resource_grid_reader* grid;
37 const channel_estimate* estimate;
38 format1_configuration config;
39 pucch_uci_message msg;
40 };
41
42 std::pair<pucch_uci_message, channel_state_information> detect(const resource_grid_reader& grid,
43 const format0_configuration& config) override
44 {
45 return std::pair<pucch_uci_message, channel_state_information>();
46 }
47
48 pucch_detection_result detect(const resource_grid_reader& grid,
49 const channel_estimate& estimates,
50 const format1_configuration& config) override
51 {
52 entries_format1.emplace_back();
53 entry_format1& entry = entries_format1.back();
54 entry.grid = &grid;
55 entry.estimate = &estimates;
56 entry.config = config;
57 entry.msg = pucch_uci_message({0, config.nof_harq_ack, 0, 0});
58 span<uint8_t> harq_ack_bits = entry.msg.get_harq_ack_bits();
59 std::generate(harq_ack_bits.begin(), harq_ack_bits.end(), [this]() { return rgen() & 1; });
60 return {entry.msg, 1};
61 }
62
63 const std::vector<entry_format1>& get_entries_format1() const { return entries_format1; }
64
65 void clear() { entries_format1.clear(); }
66
67private:
68 std::mt19937 rgen;
69 std::vector<entry_format1> entries_format1;
70};
