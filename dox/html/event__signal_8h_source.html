36class signal_event_common
37{
38public:
39 struct awaiter_common {
40 explicit awaiter_common(signal_event_common& event_) : event(&event_) {}
41 bool await_ready() const { return false; }
42 void await_suspend(coro_handle<> c)
43 {
44 // Store suspending coroutine.
45 suspended_handle = c;
46
47 // enqueue awaiter in linked list.
48 next = static_cast<awaiter_common*>(std::exchange(event->awaiters, static_cast<void*>(this)));
49 }
50
51 signal_event_common* event;
52 awaiter_common* next;
53 coro_handle<> suspended_handle;
54 };
55
56 signal_event_common() = default;
57 signal_event_common(const signal_event_common& event_) = delete;
58 signal_event_common& operator=(const signal_event_common& event_) = delete;
59 ~signal_event_common() {}
60
61protected:
63 void* awaiters = nullptr;
64
66 void flush()
67 {
68 auto old_state = std::exchange(awaiters, nullptr);
69 flush_awaiter_list(static_cast<awaiter_common*>(old_state));
70 }
71};
