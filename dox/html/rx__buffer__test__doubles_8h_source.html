30class rx_buffer_spy : public unique_rx_buffer::callback
31{
32public:
33 rx_buffer_spy() = default;
34
35 rx_buffer_spy(unsigned max_codeblock_size, unsigned nof_codeblocks) :
36 soft_bits({max_codeblock_size, nof_codeblocks}), crc(nof_codeblocks)
37 {
38 for (unsigned i_cb = 0; i_cb != nof_codeblocks; ++i_cb) {
39 hard_bits.emplace_back(dynamic_bit_buffer(max_codeblock_size));
40 }
41 }
42
43 unsigned get_nof_codeblocks() const override
44 {
45 ++const_count;
46 return soft_bits.get_dimensions_size().back();
47 }
48 void reset_codeblocks_crc() override { ++count; }
49 span<bool> get_codeblocks_crc() override
50 {
51 ++count;
52 return span<bool>(reinterpret_cast<bool*>(crc.data()), crc.size());
53 }
54 unsigned get_absolute_codeblock_id(unsigned codeblock_id) const override
55 {
56 ++const_count;
57 return 0;
58 }
59 span<log_likelihood_ratio> get_codeblock_soft_bits(unsigned codeblock_id, unsigned data_size) override
60 {
61 ++count;
62 return soft_bits.get_view<1>({codeblock_id}).first(data_size);
63 }
64 bit_buffer get_codeblock_data_bits(unsigned codeblock_id, unsigned data_size) override
65 {
66 ++count;
67 return hard_bits[codeblock_id].first(data_size);
68 }
69
70 void lock() override { locked = true; }
71
72 void unlock() override { locked = false; }
73
74 void release() override { locked = false; }
75
77 void clear()
78 {
79 count = 0;
80 const_count = 0;
81 }
82
84 unsigned get_total_count() const { return count + const_count; }
85
87 bool is_locked() const { return locked; }
88
89private:
90 unsigned count = 0;
91 mutable unsigned const_count = 0;
92 bool locked = false;
93
94 dynamic_tensor<2, log_likelihood_ratio> soft_bits;
95 std::vector<dynamic_bit_buffer> hard_bits;
96 std::vector<uint8_t> crc;
97};
