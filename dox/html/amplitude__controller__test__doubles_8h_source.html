31class amplitude_controller_spy : public amplitude_controller
32{
33public:
34 struct entry_t {
35 std::vector<cf_t> input;
36 std::vector<cf_t> output;
37 };
38
39 amplitude_controller_spy() : random_data(1009)
40 {
41 std::mt19937 rgen;
42 std::uniform_real_distribution<float> sample_dist(-1, 1);
43 std::generate(random_data.begin(), random_data.end(), [&sample_dist, &rgen]() {
44 return cf_t(sample_dist(rgen), sample_dist(rgen));
45 });
46 }
47
48 amplitude_controller_metrics process(span<cf_t> output, span<const cf_t> input) override
49 {
50 entries.emplace_back();
51 entry_t& entry = entries.back();
52
53 entry.input = std::vector<cf_t>(input.begin(), input.end());
54
55 std::size_t remaining = output.size();
56 span<cf_t> random_data_view(random_data);
57
58 while (remaining > 0) {
59 // Number of samples to copy for this iteration.
60 unsigned nof_copied_samples = std::min(random_data.size() - random_data_index, remaining);
61 span<cf_t> random_samples = random_data_view.subspan(random_data_index, nof_copied_samples);
62 span<cf_t> out_samples = output.subspan(output.size() - remaining, nof_copied_samples);
63
64 srsvec::copy(out_samples, random_samples);
65
66 // Update remaining samples and buffer index.
67 remaining -= nof_copied_samples;
68 random_data_index = (random_data_index + nof_copied_samples) % random_data.size();
69 }
70
71 entry.output = std::vector<cf_t>(output.begin(), output.end());
72
73 return amplitude_controller_metrics();
74 }
75
76 const std::vector<entry_t>& get_entries() const { return entries; }
77
78 void clear() { entries.clear(); }
79
80private:
81 unsigned random_data_index = 0;
82 std::vector<cf_t> random_data;
83 std::uniform_real_distribution<float> output_dist;
84 std::vector<entry_t> entries;
85};
