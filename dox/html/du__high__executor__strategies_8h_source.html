31class index_based_ue_executor_mapper final : public du_high_ue_executor_mapper
32{
33public:
34 index_based_ue_executor_mapper(const std::vector<task_executor*>& ctrl_execs,
35 const std::vector<task_executor*>& ul_execs = {},
36 const std::vector<task_executor*>& dl_execs = {}) :
37 execs(ctrl_execs.size())
38 {
39 srsran_assert(ctrl_execs.size() > 0, "At least one control executor must be specified");
40 srsran_assert(ul_execs.size() == 0 or ul_execs.size() == ctrl_execs.size(),
41 "If specified, the number of UE UL executors must be equal to the number of control executors");
42 srsran_assert(dl_execs.size() == 0 or dl_execs.size() == ctrl_execs.size(),
43 "If specified, the number of UE DL executors must be equal to the number of control executors");
44
45 for (unsigned i = 0; i != ctrl_execs.size(); ++i) {
46 execs[i].ctrl_exec = ctrl_execs[i];
47 // If UL executors not specified, we use the control executor.
48 execs[i].ul_exec = ul_execs.empty() ? ctrl_execs[i] : ul_execs[i];
49 // If DL executors not specified, we use the DL executor (that might be the control executor if the DL executor
50 // was not specified).
51 execs[i].dl_exec = dl_execs.empty() ? execs[i].ul_exec : dl_execs[i];
52 }
53 }
54
55 void rebind_executors(du_ue_index_t ue_index, du_cell_index_t pcell_index) override
56 {
57 // Do nothing.
58 }
59
60 task_executor& ctrl_executor(du_ue_index_t ue_index) override
61 {
62 ue_index = ue_index < MAX_NOF_DU_UES ? ue_index : to_du_ue_index(0);
63 return *execs[ue_index % execs.size()].ctrl_exec;
64 }
65
66 task_executor& f1u_dl_pdu_executor(du_ue_index_t ue_index) override
67 {
68 ue_index = ue_index < MAX_NOF_DU_UES ? ue_index : to_du_ue_index(0);
69 return *execs[ue_index % execs.size()].dl_exec;
70 }
71
72 task_executor& mac_ul_pdu_executor(du_ue_index_t ue_index) override
73 {
74 ue_index = ue_index < MAX_NOF_DU_UES ? ue_index : to_du_ue_index(0);
75 return *execs[ue_index % execs.size()].ul_exec;
76 }
77
78private:
79 struct ue_executors {
80 task_executor* ctrl_exec;
81 task_executor* ul_exec;
82 task_executor* dl_exec;
83 };
84
85 std::vector<ue_executors> execs;
86};

89class pcell_ue_executor_mapper final : public du_high_ue_executor_mapper
90{
91public:
92 explicit pcell_ue_executor_mapper(const std::vector<task_executor*>& ctrl_execs,
93 const std::vector<task_executor*>& ul_execs = {},
94 const std::vector<task_executor*>& dl_execs = {}) :
95 execs(ctrl_execs.size())
96 {
97 srsran_assert(not ctrl_execs.empty(), "At least one control executor must be specified");
98 srsran_assert(ul_execs.empty() or ul_execs.size() == ctrl_execs.size(),
99 "If specified, the number of UE UL executors must be equal to the number of control executors");
100 srsran_assert(dl_execs.empty() or dl_execs.size() == ctrl_execs.size(),
101 "If specified, the number of UE DL executors must be equal to the number of control executors");
102
103 for (unsigned i = 0; i != ctrl_execs.size(); ++i) {
104 execs[i].ctrl_exec = ctrl_execs[i];
105 // If UL executors not specified, we use the control executor.
106 execs[i].ul_exec = ul_execs.empty() ? ctrl_execs[i] : ul_execs[i];
107 // If DL executors not specified, we use the DL executor (that might be the control executor if the DL executor
108 // was not specified).
109 execs[i].dl_exec = dl_execs.empty() ? execs[i].ul_exec : dl_execs[i];
110 }
111
112 // Initialize executors in a round-robin fashion.
113 for (unsigned i = 0; i != MAX_NOF_DU_UES; ++i) {
114 ue_idx_to_exec_index[i] = i % execs.size();
115 }
116 }
117
118 void rebind_executors(du_ue_index_t ue_index, du_cell_index_t pcell_index) override
119 {
120 srsran_sanity_check(is_du_ue_index_valid(ue_index), "Invalid ue id={}", ue_index);
121 ue_idx_to_exec_index[ue_index] = pcell_index % execs.size();
122 }
123
124 task_executor& ctrl_executor(du_ue_index_t ue_index) override
125 {
126 if (ue_index < MAX_NOF_DU_UES) {
127 return *execs[ue_idx_to_exec_index[ue_index]].ctrl_exec;
128 }
129 return *execs[0].ctrl_exec;
130 }
131
132 task_executor& f1u_dl_pdu_executor(du_ue_index_t ue_index) override
133 {
134 if (ue_index < MAX_NOF_DU_UES) {
135 return *execs[ue_idx_to_exec_index[ue_index]].dl_exec;
136 }
137 return *execs[0].dl_exec;
138 }
139
140 task_executor& mac_ul_pdu_executor(du_ue_index_t ue_index) override
141 {
142 if (ue_index < MAX_NOF_DU_UES) {
143 return *execs[ue_idx_to_exec_index[ue_index]].ul_exec;
144 }
145 return *execs[0].ul_exec;
146 }
147
148private:
149 struct ue_executors {
150 task_executor* ctrl_exec;
151 task_executor* ul_exec;
152 task_executor* dl_exec;
153 };
154
155 std::vector<ue_executors> execs;
156
158 std::array<unsigned, MAX_NOF_DU_UES> ue_idx_to_exec_index;
159};

163class cell_executor_mapper final : public du_high_cell_executor_mapper
164{
165public:
169 explicit cell_executor_mapper(const std::initializer_list<task_executor*>& cell_execs_,
170 const std::initializer_list<task_executor*>& slot_execs_ = {},
171 const std::initializer_list<task_executor*>& err_ind_execs_ = {}) :
172 cell_execs(cell_execs_.begin(), cell_execs_.end()),
173 slot_execs(slot_execs_.begin(), slot_execs_.end()),
174 err_ind_execs(cell_execs_.begin(), cell_execs_.end())
175 {
176 srsran_assert(not cell_execs.empty(), "The number of DL executors must be higher than 1");
177 if (slot_execs.empty()) {
178 slot_execs = cell_execs;
179 }
180 if (err_ind_execs.empty()) {
181 err_ind_execs = cell_execs;
182 }
183 }
184
185 task_executor& executor(du_cell_index_t cell_index) override { return *cell_execs[cell_index % cell_execs.size()]; }
186
187 task_executor& slot_ind_executor(du_cell_index_t cell_index) override
188 {
189 return *slot_execs[cell_index % slot_execs.size()];
190 }
191
192 task_executor& error_ind_executor(du_cell_index_t cell_index) override
193 {
194 return *err_ind_execs[cell_index % err_ind_execs.size()];
195 }
196
197private:
199 std::vector<task_executor*> cell_execs;
200
204 std::vector<task_executor*> slot_execs;
205
207 std::vector<task_executor*> err_ind_execs;
208};
