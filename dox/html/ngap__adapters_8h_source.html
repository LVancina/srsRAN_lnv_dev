37class ngap_to_cu_cp_task_scheduler : public ngap_ue_task_scheduler
38{
39public:
40 ngap_to_cu_cp_task_scheduler() = default;
41
42 void connect_cu_cp(ue_task_scheduler_manager& cu_cp_task_sched_) { cu_cp_task_sched = &cu_cp_task_sched_; }
43
44 void schedule_async_task(ue_index_t ue_index, async_task<void>&& task) override
45 {
46 srsran_assert(cu_cp_task_sched != nullptr, "CU-CP task scheduler handler must not be nullptr");
47 logger.debug("ue={}: Scheduling async task", ue_index);
48 cu_cp_task_sched->handle_ue_async_task(ue_index, std::move(task));
49 }
50
51 unique_timer make_unique_timer() override
52 {
53 srsran_assert(cu_cp_task_sched != nullptr, "CU-CP task scheduler handler must not be nullptr");
54 return cu_cp_task_sched->make_unique_timer();
55 }
56 timer_manager& get_timer_manager() override
57 {
58 srsran_assert(cu_cp_task_sched != nullptr, "CU-CP task scheduler handler must not be nullptr");
59 return cu_cp_task_sched->get_timer_manager();
60 }
61
62private:
63 ue_task_scheduler_manager* cu_cp_task_sched = nullptr;
64 srslog::basic_logger& logger = srslog::fetch_basic_logger("NGAP");
65};
