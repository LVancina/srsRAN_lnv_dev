30enum class sink_stream_type { stdout, stderr };
31
33class stream_sink : public sink
34{
35public:
36 stream_sink(sink_stream_type s, std::unique_ptr<log_formatter> f) :
37 sink(std::move(f)), handle((s == sink_stream_type::stdout) ? stdout : stderr)
38 {
39 }
40
41 stream_sink(const stream_sink& other) = delete;
42 stream_sink& operator=(const stream_sink& other) = delete;
43
44 detail::error_string write(detail::memory_buffer buffer) override
45 {
46 assert(handle && "Invalid stream handle");
47 std::fwrite(buffer.data(), sizeof(char), buffer.size(), handle);
48 // We want to see the output instantly.
49 std::fflush(handle);
50 return {};
51 }
52
53 detail::error_string flush() override
54 {
55 std::fflush(handle);
56 return {};
57 }
58
59private:
60 std::FILE* handle;
61};
