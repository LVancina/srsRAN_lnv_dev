29struct uci_part2_size_description {
31 static constexpr unsigned max_nof_parameters = 2;
33 static constexpr unsigned max_nof_entries = 2;
35 static constexpr unsigned max_nof_entry_bits = 4;
37 static constexpr unsigned max_size_table = 1U << max_nof_entry_bits;
38
39 uci_part2_size_description() = default;
40
45 uci_part2_size_description(unsigned fixed_size)
46 {
47 // Clear entries if no UCI Part 2 payload.
48 if (fixed_size == 0) {
49 entries.clear();
50 return;
51 }
52
53 // Create new size entry.
54 entry& e = entries.emplace_back();
55
56 // Create zero bit width parameter.
57 parameter& p = e.parameters.emplace_back();
58 p.width = 0;
59 p.offset = 0;
60
61 // Add size to table.
62 e.map.emplace_back(fixed_size);
63 }
64
66 struct parameter {
68 uint16_t offset;
70 uint8_t width;
71 };
72
74 struct entry {
77 static_vector<parameter, max_nof_parameters> parameters;
79 static_vector<uint16_t, max_size_table> map;
80 };
81
83 static_vector<entry, max_nof_entries> entries;
84
86 bool is_valid(unsigned uci_part1_nof_bits) const
87 {
88 // For each entry...
89 for (const auto& entry_ : entries) {
90 unsigned param_size = 0;
91
92 // For each parameter...
93 for (const auto& param : entry_.parameters) {
94 // Accumulate the parameter width.
95 param_size += param.width;
96
97 // Make sure the parameter offset and width does not exceed the size of CSI Part 1.
98 if ((param.width + param.offset) >= uci_part1_nof_bits) {
99 return false;
100 }
101 }
102
103 // Make sure the entry table size corresponds to the number of bits.
104 if (1U << param_size != entry_.map.size()) {
105 return false;
106 }
107 }
108
109 // The description must be valid if reached here.
110 return true;
111 }
112};
