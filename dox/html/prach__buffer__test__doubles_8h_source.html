34class prach_buffer_spy : public prach_buffer
35{
36public:
37 struct entry_t {
38 unsigned i_port;
39 unsigned i_td_occasion;
40 unsigned i_fd_occasion;
41 unsigned i_symbol;
42 };
43
44 prach_buffer_spy() : buffer(0, 0, 0, 0, 0) {}
45
46 prach_buffer_spy(span<cf_t> data_,
47 unsigned nof_td_occasions,
48 unsigned nof_fd_occasions,
49 unsigned nof_symbols,
50 unsigned sequence_length) :
51 buffer(1, nof_td_occasions, nof_fd_occasions, nof_symbols, sequence_length)
52 {
53 span<const cf_t> data = data_;
54 unsigned nof_ports = get_max_nof_ports_inner();
55
56 report_fatal_error_if_not(data.size() == nof_td_occasions * nof_fd_occasions * sequence_length,
57 "The symbols data size is not consistent with the symbol size and number of symbols.");
58
59 for (unsigned i_td_occasion = 0; i_td_occasion != nof_td_occasions; ++i_td_occasion) {
60 for (unsigned i_fd_occasion = 0; i_fd_occasion != nof_fd_occasions; ++i_fd_occasion) {
61 span<const cf_t> occasion_data = data.first(sequence_length);
62 data = data.last(data.size() - sequence_length);
63 for (unsigned i_port = 0; i_port != nof_ports; ++i_port) {
64 for (unsigned i_symbol = 0; i_symbol != nof_symbols; ++i_symbol) {
65 srsvec::copy(buffer.get_symbol(i_port, i_td_occasion, i_fd_occasion, i_symbol), occasion_data);
66 }
67 }
68 }
69 }
70 }
71
72 unsigned get_max_nof_ports() const override { return get_max_nof_ports_inner(); }
73
74 unsigned get_max_nof_td_occasions() const override
75 {
76 ++count_get_max_nof_td_occasions;
77 return buffer.get_max_nof_td_occasions();
78 }
79
80 unsigned get_max_nof_fd_occasions() const override
81 {
82 ++count_get_max_nof_fd_occasions;
83 return buffer.get_max_nof_fd_occasions();
84 }
85
86 unsigned get_max_nof_symbols() const override
87 {
88 ++count_get_max_nof_symbols;
89 return buffer.get_max_nof_symbols();
90 }
91
92 unsigned get_sequence_length() const override
93 {
94 ++count_get_sequence_length;
95 return buffer.get_sequence_length();
96 }
97
98 span<cf_t> get_symbol(unsigned i_port, unsigned i_td_occasion, unsigned i_fd_occasion, unsigned i_symbol) override
99 {
100 get_symbol_entries.emplace_back();
101 entry_t& entry = get_symbol_entries.back();
102 entry.i_port = i_port;
103 entry.i_td_occasion = i_td_occasion;
104 entry.i_fd_occasion = i_fd_occasion;
105 entry.i_symbol = i_symbol;
106 return buffer.get_symbol(i_port, i_td_occasion, i_fd_occasion, i_symbol);
107 }
108
109 span<const cf_t>
110 get_symbol(unsigned i_port, unsigned i_td_occasion, unsigned i_fd_occasion, unsigned i_symbol) const override
111 {
112 get_symbol_const_entries.emplace_back();
113 entry_t& entry = get_symbol_const_entries.back();
114 entry.i_port = i_port;
115 entry.i_td_occasion = i_td_occasion;
116 entry.i_fd_occasion = i_fd_occasion;
117 entry.i_symbol = i_symbol;
118 return buffer.get_symbol(i_port, i_td_occasion, i_fd_occasion, i_symbol);
119 }
120
121 unsigned get_total_count() const
122 {
123 return count_get_max_nof_ports + count_get_max_nof_td_occasions + count_get_max_nof_fd_occasions +
124 count_get_max_nof_symbols + count_get_sequence_length + get_symbol_entries.size() +
125 get_symbol_const_entries.size();
126 }
127
128 const std::vector<entry_t>& get_get_symbol_entries() { return get_symbol_entries; };
129 const std::vector<entry_t>& get_get_symbol_const_entries() { return get_symbol_const_entries; };
130
131 void clear()
132 {
133 count_get_max_nof_ports = 0;
134 count_get_max_nof_td_occasions = 0;
135 count_get_max_nof_fd_occasions = 0;
136 count_get_max_nof_symbols = 0;
137 count_get_sequence_length = 0;
138 }
139
140private:
142 prach_buffer_impl buffer;
143
144 mutable unsigned count_get_max_nof_ports = 0;
145 mutable unsigned count_get_max_nof_td_occasions = 0;
146 mutable unsigned count_get_max_nof_fd_occasions = 0;
147 mutable unsigned count_get_max_nof_symbols = 0;
148 mutable unsigned count_get_sequence_length = 0;
149 std::vector<entry_t> get_symbol_entries;
150 mutable std::vector<entry_t> get_symbol_const_entries;
151
152 unsigned get_max_nof_ports_inner() const
153 {
154 ++count_get_max_nof_ports;
155 return buffer.get_max_nof_ports();
156 }
157};

160class prach_buffer_tensor : public prach_buffer
161{
162public:
164 enum class dims : unsigned {
165 re = 0,
166 symbol = 1,
167 fd_occasion = 2,
168 td_occasion = 3,
169 port = 4,
170 count = 5,
171 };
172
173 using storage_type = tensor<static_cast<std::underlying_type_t<dims>>(dims::count), cf_t, dims>;
174
176 explicit prach_buffer_tensor(storage_type& data_) : data(data_) {}
177
178 // See interface for documentation.
179 unsigned get_max_nof_ports() const override { return data.get_dimension_size(dims::port); }
180
181 // See interface for documentation.
182 unsigned get_max_nof_td_occasions() const override { return data.get_dimension_size(dims::td_occasion); }
183
184 // See interface for documentation.
185 unsigned get_max_nof_fd_occasions() const override { return data.get_dimension_size(dims::fd_occasion); }
186
187 // See interface for documentation.
188 unsigned get_max_nof_symbols() const override { return data.get_dimension_size(dims::symbol); }
189
190 // See interface for documentation.
191 unsigned get_sequence_length() const override { return data.get_dimension_size(dims::re); }
192
193 // See interface for documentation.
194 span<cf_t> get_symbol(unsigned i_port, unsigned i_td_occasion, unsigned i_fd_occasion, unsigned i_symbol) override
195 {
196 srsran_assert(i_port < get_max_nof_ports(),
197 "The port index (i.e., {}) exceeds the maximum number of ports (i.e., {}).",
198 i_port,
199 get_max_nof_ports());
200 srsran_assert(i_td_occasion < get_max_nof_td_occasions(),
201 "The time-domain occasion (i.e., {}) exceeds the maximum number of time-domain occasions (i.e., {}).",
202 i_td_occasion,
203 get_max_nof_td_occasions());
204 srsran_assert(
205 i_fd_occasion < get_max_nof_fd_occasions(),
206 "The frequency-domain occasion (i.e., {}) exceeds the maximum number of frequency-domain occasions (i.e., {}).",
207 i_fd_occasion,
208 get_max_nof_fd_occasions());
209 srsran_assert(i_symbol < get_max_nof_symbols(),
210 "The symbol index (i.e., {}) exceeds the maximum number of symbols (i.e., {}).",
211 i_symbol,
212 get_max_nof_symbols());
213 return data.get_view({i_symbol, i_fd_occasion, i_td_occasion, i_port});
214 }
215
216 // See interface for documentation.
217 span<const cf_t>
218 get_symbol(unsigned i_port, unsigned i_td_occasion, unsigned i_fd_occasion, unsigned i_symbol) const override
219 {

222 i_port,
223 get_max_nof_ports());

237 return data.get_view({i_symbol, i_fd_occasion, i_td_occasion, i_port});
238 }
