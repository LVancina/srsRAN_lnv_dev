333class dummy_ngap_cu_cp_ue_creation_notifier : public ngap_cu_cp_ue_creation_notifier
334{
335public:
336 dummy_ngap_cu_cp_ue_creation_notifier(ngap_ue_manager& ue_manager_) :
337 ue_manager(ue_manager_), logger(srslog::fetch_basic_logger("TEST")){};
338
339 void add_ue(ue_index_t ue_index,
340 ngap_rrc_ue_pdu_notifier& rrc_ue_pdu_notifier,
341 ngap_rrc_ue_control_notifier& rrc_ue_ctrl_notifier,
342 ngap_du_processor_control_notifier& du_processor_ctrl_notifier)
343 {
344 ue_notifiers_map.emplace(
345 ue_index, ngap_ue_notifiers{&rrc_ue_pdu_notifier, &rrc_ue_ctrl_notifier, &du_processor_ctrl_notifier});
346 }
347
348 bool on_new_ngap_ue(ue_index_t ue_index) override
349 {
350 srsran_assert(ue_notifiers_map.find(ue_index) != ue_notifiers_map.end(), "UE context must be present");
351 auto& ue_notifier = ue_notifiers_map.at(ue_index);
352
353 srsran_assert(ue_notifier.rrc_ue_pdu_notifier != nullptr, "rrc_ue_pdu_notifier must not be nullptr");
354 srsran_assert(ue_notifier.rrc_ue_ctrl_notifier != nullptr, "rrc_ue_ctrl_notifier must not be nullptr");
355 srsran_assert(ue_notifier.du_processor_ctrl_notifier != nullptr, "du_processor_ctrl_notifier must not be nullptr");
356
357 last_ue = ue_index;
358
359 // Add NGAP UE to UE manager
360 ngap_ue* ue = ue_manager.set_ue_ng_context(ue_index,
361 *ue_notifier.rrc_ue_pdu_notifier,
362 *ue_notifier.rrc_ue_ctrl_notifier,
363 *ue_notifier.du_processor_ctrl_notifier);
364
365 if (ue == nullptr) {
366 logger.error("ue={}: Failed to create UE", ue_index);
367 return false;
368 }
369
370 logger.info("ue={}: NGAP UE was created", ue_index);
371 return true;
372 }
373
374 ue_index_t last_ue = ue_index_t::invalid;
375
376private:
377 ngap_ue_manager& ue_manager;
378
379 srslog::basic_logger& logger;
380
381 struct ngap_ue_notifiers {
382 ngap_rrc_ue_pdu_notifier* rrc_ue_pdu_notifier = nullptr;
383 ngap_rrc_ue_control_notifier* rrc_ue_ctrl_notifier = nullptr;
384 ngap_du_processor_control_notifier* du_processor_ctrl_notifier = nullptr;
385 };
386
387 std::map<ue_index_t, ngap_ue_notifiers> ue_notifiers_map;
388};
