36inline const char* get_cause_str(const asn1::ngap::cause_c& cause)
37{
38 using namespace asn1::ngap;
39 switch (cause.type()) {
40 case cause_c::types_opts::radio_network:
41 return cause.radio_network().to_string();
42 case cause_c::types_opts::transport:
43 return cause.transport().to_string();
44 case cause_c::types_opts::nas:
45 return cause.nas().to_string();
46 case cause_c::types_opts::protocol:
47 return cause.protocol().to_string();
48 case cause_c::types_opts::misc:
49 return cause.misc().to_string();
50 default:
51 break;
52 }
53 return "unknown";
54}

110inline void copy_asn1_key(security::sec_key& key_out, const asn1::fixed_bitstring<256, false, true>& key_in)
111{
112 for (uint32_t i = 0; i < key_in.nof_octets(); ++i) {
113 key_out[i] = key_in.data()[key_in.nof_octets() - 1 - i];
114 }
115}

117inline void fill_supported_algorithms(security::supported_algorithms& supported_algos_out,
118 const asn1::fixed_bitstring<16, true, true>& supported_algos_in)
119{
120 uint16_t tmp = *reinterpret_cast<const uint16_t*>(supported_algos_in.data());
121 for (uint16_t i = 0; i < 3; ++i) {
122 supported_algos_out[i] = ((tmp >> (15 - i)) & 0x01U) == 1; // interesting bits are in the second byte
123 }
124}
