33class uplink_request_handler_task_dispatcher : public uplink_request_handler
34{
35public:
36 uplink_request_handler_task_dispatcher(std::unique_ptr<uplink_request_handler> ul_handler_,
37 task_executor& executor_) :
38 ul_handler(std::move(ul_handler_)), executor(executor_)
39 {
40 srsran_assert(ul_handler, "Invalid uplink request handler");
41 }
42
43 // See interface for documentation.
44 void handle_prach_occasion(const prach_buffer_context& context, prach_buffer& buffer) override
45 {
46 if (!executor.execute([this, context, &buffer]() { ul_handler->handle_prach_occasion(context, buffer); })) {
47 srslog::fetch_basic_logger("OFH").warning("Failed to dispatch PRACH occasion for slot '{}'", context.slot);
48 }
49 }
50
51 // See interface for documentation.
52 void handle_new_uplink_slot(const resource_grid_context& context, resource_grid& grid) override
53 {
54 if (!executor.execute([this, context, &grid]() { ul_handler->handle_new_uplink_slot(context, grid); })) {
55 srslog::fetch_basic_logger("OFH").warning("Failed to dispatch new uplink slot for slot '{}'", context.slot);
56 }
57 }
58
59private:
60 std::unique_ptr<uplink_request_handler> ul_handler;
61 task_executor& executor;
62};
