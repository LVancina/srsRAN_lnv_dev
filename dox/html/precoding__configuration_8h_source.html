58 precoding_configuration(unsigned nof_layers_, unsigned nof_ports_, unsigned nof_prg_, unsigned prg_size_) :
59 prg_size(prg_size_), nof_prg(nof_prg_), nof_layers(nof_layers_), nof_ports(nof_ports_)
60 {
62 static constexpr interval<unsigned, true> nof_layers_range{1, precoding_constants::MAX_NOF_LAYERS};
64 static constexpr interval<unsigned, true> nof_ports_range{1, precoding_constants::MAX_NOF_PORTS};
66 static constexpr interval<unsigned, true> nof_prg_range{1, precoding_constants::MAX_NOF_PRG};
68 static constexpr interval<unsigned, true> prg_size_range{precoding_constants::MIN_PRG_SIZE, MAX_NOF_PRBS};
69
70 srsran_assert(nof_prg_range.contains(nof_prg_),
71 "The number of PRG (i.e., {}) is out of the range {}.",
72 nof_prg_,
73 nof_prg_range);
74
75 srsran_assert(prg_size_range.contains(prg_size_),
76 "The PRG size (i.e., {}) is out of the range {}.",
77 prg_size_,
78 prg_size_range);
79 srsran_assert(nof_layers_range.contains(nof_layers),
80 "The number of layers (i.e., {}) is outside the range {}.",
81 nof_layers,
82 nof_layers_range);
83 srsran_assert(nof_ports_range.contains(nof_ports),
84 "The number of ports (i.e., {}) is outside the range {}.",
85 nof_ports,
86 nof_ports_range);
87
88 // Construct precoding matrices.
89 for (unsigned i_prg = 0; i_prg != nof_prg; ++i_prg) {
90 weights_list.push_back(precoding_weight_matrix(nof_layers, nof_ports));
91 }

96 precoding_configuration(const precoding_configuration& other) :

131 static precoding_configuration make_wideband(const precoding_weight_matrix& precoding_matrix)
132 {
133 // Create a precoding configuration with a single PRG spanning all the available BW.
134 precoding_configuration config(
135 precoding_matrix.get_nof_layers(), precoding_matrix.get_nof_ports(), 1, MAX_NOF_PRBS);
136
137 // Set the PRG weights.
138 config.set_prg_coefficients(precoding_matrix, 0);
139
140 return config;
141 }

256 void set_prg_coefficients(const precoding_weight_matrix& prg_coefficients, unsigned i_prg)
257 {
258 srsran_assert((i_prg < get_nof_prg()),
259 "The PRG identifier, i.e., {}, must be smaller than the number of PRG, i.e., {}.",
260 i_prg,
261 get_nof_prg());
262
263 srsran_assert(prg_coefficients.get_nof_layers() == get_nof_layers(),
264 "The precoding matrix number of layers (i.e., {}) does not match the precoding configuration number "
265 "of layers (i.e., {}).",
266 prg_coefficients.get_nof_layers(),
267 get_nof_layers());
268 srsran_assert(prg_coefficients.get_nof_ports() == get_nof_ports(),
269 "The precoding matrix number of ports (i.e., {}) does not match the precoding configuration number "
270 "of ports (i.e., {}).",
271 prg_coefficients.get_nof_ports(),
272 get_nof_ports());
273
274 weights_list[i_prg] = prg_coefficients;
275 }

277 void resize(unsigned nof_prg_, unsigned prg_size_)
278 {
280 static constexpr interval<unsigned, true> nof_prg_range{1, precoding_constants::MAX_NOF_PRG};
282 static constexpr interval<unsigned, true> prg_size_range{precoding_constants::MIN_PRG_SIZE, MAX_NOF_PRBS};
283
284 srsran_assert(nof_prg_range.contains(nof_prg_),
285 "The number of PRG (i.e., {}) is out of the range {}.",
286 nof_prg_,
287 nof_prg_range);
288
289 srsran_assert(prg_size_range.contains(prg_size_),
290 "The PRG size (i.e., {}) is out of the range {}.",
291 prg_size_,
292 prg_size_range);
293
294 if (nof_prg_ <= get_nof_prg()) {
295 // Shrink the list of precoding weights.
296 weights_list.resize(nof_prg);
297 } else {
298 // Grow the list of matrices using the appropriate matrix dimensions. Append new matrices at the end of the list.
299 for (unsigned i_prg = get_nof_prg(); i_prg != nof_prg_; ++i_prg) {
300 weights_list.push_back(precoding_weight_matrix(get_nof_layers(), get_nof_ports()));
301 }
302 }
303
304 // Update the dimensions.
305 prg_size = prg_size_;
306 nof_prg = nof_prg_;
307 }
308
310 precoding_configuration& operator*=(float scale)
311 {
312 for (unsigned i_prg = 0; i_prg != nof_prg; ++i_prg) {
313 weights_list[i_prg] *= scale;
314 }
315 return *this;
316 }

318private:
320 unsigned prg_size = MAX_NOF_PRBS;
322 unsigned nof_prg = 0;
324 unsigned nof_layers = 0;
326 unsigned nof_ports = 0;
327
329 static_vector<precoding_weight_matrix, precoding_constants::MAX_NOF_PRG> weights_list;
