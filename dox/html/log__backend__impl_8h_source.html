41 log_backend_impl& operator=(const log_backend_impl& other) = delete;
42
43 void start(backend_priority priority = backend_priority::normal) override { worker.start(priority); }
44
45 bool push(detail::log_entry&& entry) override
46 {
47 auto* arg_store = entry.metadata.store;
48 if (!queue.push(std::move(entry))) {
49 arg_pool.dealloc(arg_store);
50 return false;
51 }
52 return true;
53 }
54
55 fmt::dynamic_format_arg_store<fmt::format_context>* alloc_arg_store() override { return arg_pool.alloc(); }
56
57 bool is_running() const override { return worker.is_running(); }
58
60 void set_error_handler(error_handler err_handler) { worker.set_error_handler(std::move(err_handler)); }
61
63 void stop() { worker.stop(); }
64
65private:
66 detail::work_queue<detail::log_entry> queue;
67 detail::dyn_arg_store_pool arg_pool;
68 backend_worker worker;
