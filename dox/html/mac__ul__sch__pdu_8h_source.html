90inline rnti_t decode_crnti_ce(byte_buffer_view payload)
91{
92 if (payload.length() < 2) {
93 return rnti_t::INVALID_RNTI;
94 }
95
96 // Conversion between Little Endian to RNTI value. See TS 38.321, 6.1.3.2 - C-RNTI MAC CE.
97 return to_rnti(le16toh((uint16_t)payload[0] << 8U | payload[1]));
98}

104template <>
105struct formatter<srsran::mac_ul_sch_subpdu> {
106 template <typename ParseContext>
107 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
108 {
109 return ctx.begin();
110 }
111
112 template <typename FormatContext>
113 auto format(const srsran::mac_ul_sch_subpdu& subpdu, FormatContext& ctx) -> decltype(ctx.out())
114 {
115 using namespace srsran;
116 if (subpdu.lcid().is_sdu()) {
117 format_to(ctx.out(), "lcid={}: len={}", subpdu.lcid(), subpdu.sdu_length());
118 return ctx.out();
119 }
120 switch (subpdu.lcid().value()) {
121 case lcid_ul_sch_t::CCCH_SIZE_64:
122 format_to(ctx.out(), "CCCH64: len={}", subpdu.sdu_length());
123 break;
124 case lcid_ul_sch_t::CCCH_SIZE_48:
125 format_to(ctx.out(), "CCCH48: len={}", subpdu.sdu_length());
126 break;
127 case lcid_ul_sch_t::CRNTI:
128 format_to(ctx.out(), "C-RNTI: {:#x}", to_value(decode_crnti_ce(subpdu.payload())));
129 break;
130 case lcid_ul_sch_t::SHORT_TRUNC_BSR: {
131 lcg_bsr_report sbsr = decode_sbsr(subpdu.payload());
132 format_to(ctx.out(), "SHORT_TRUNC_BSR: len={} bs={}", subpdu.total_length(), sbsr.buffer_size);
133 break;
134 }
135 case lcid_ul_sch_t::LONG_TRUNC_BSR:
136 format_to(ctx.out(), "LONG_TRUNC_BSR: len={}", subpdu.total_length());
137 break;
138 case lcid_ul_sch_t::SHORT_BSR: {
139 lcg_bsr_report sbsr = decode_sbsr(subpdu.payload());
140 format_to(ctx.out(), "SBSR: lcg={} bs={}", sbsr.lcg_id, sbsr.buffer_size);
141 break;
142 }
143 case lcid_ul_sch_t::LONG_BSR: {
144 expected<long_bsr_report> lbsr = decode_lbsr(bsr_format::LONG_BSR, subpdu.payload());
145 if (lbsr.has_value()) {
146 format_to(ctx.out(), "LBSR: ");
147 for (const auto& lcg : lbsr.value().list) {
148 format_to(ctx.out(), "lcg={} bs={} ", lcg.lcg_id, lcg.buffer_size);
149 }
150 } else {
151 format_to(ctx.out(), "LBSR: invalid");
152 }
153 break;
154 }
155 case lcid_ul_sch_t::SE_PHR: {
156 const phr_report phr = decode_se_phr(subpdu.payload());
157 format_to(ctx.out(), "SE_PHR: ph={}dB p_cmax={}dBm", phr.get_se_phr().ph, phr.get_se_phr().p_cmax);
158 break;
159 }
160 case lcid_ul_sch_t::PADDING:
161 format_to(ctx.out(), "PAD: len={}", subpdu.sdu_length());
162 break;
163 default:
164 format_to(ctx.out(), "Unrecognized CE={} total_len={}", subpdu.lcid(), subpdu.total_length());
165 break;
166 }
167 return ctx.out();
168 }
169};
