32inline bool search_space_supports_dci_dl_format(const search_space_configuration& ss_cfg, dci_dl_format dci_fmt)
33{
34 if (ss_cfg.is_common_search_space()) {
35 const auto& cmn_dci_fmt =
36 variant_get<search_space_configuration::common_dci_format>(ss_cfg.get_monitored_dci_formats());
37 switch (dci_fmt) {
38 case dci_dl_format::f1_0:
39 return cmn_dci_fmt.f0_0_and_f1_0;
40 case dci_dl_format::f1_1:
41 return false;
42 case dci_dl_format::f2_0:
43 return cmn_dci_fmt.f2_0;
44 default:
45 report_fatal_error("DCI format {} not supported for common SearchSpace", dci_fmt);
46 }
47 } else {
48 const auto& ue_dci_fmt =
49 variant_get<search_space_configuration::ue_specific_dci_format>(ss_cfg.get_monitored_dci_formats());
50 switch (dci_fmt) {
51 case dci_dl_format::f1_0:
52 return ue_dci_fmt == search_space_configuration::ue_specific_dci_format::f0_0_and_f1_0;
53 case dci_dl_format::f1_1:
54 return ue_dci_fmt == search_space_configuration::ue_specific_dci_format::f0_1_and_1_1;
55 default:
56 report_fatal_error("DCI format {} not supported for UE-dedicated SearchSpace", dci_fmt);
57 }
58 }
59 return false;
60}

65inline dci_dl_format get_dl_dci_format(const search_space_configuration& ss_cfg)
66{
67 if (ss_cfg.is_common_search_space()) {
68 const auto common_dci_fmt =
69 variant_get<search_space_configuration::common_dci_format>(ss_cfg.get_monitored_dci_formats());
70 if (common_dci_fmt.f0_0_and_f1_0) {
71 return dci_dl_format::f1_0;
72 }
73 // TODO: Handle DCI Formats 2_0, 2_1, 2_2, 2_3 under Common SearchSpace.
74 report_fatal_error("Unsupported DL DCI format");
75 }
76 const auto uss_dci_fmt =
77 variant_get<search_space_configuration::ue_specific_dci_format>(ss_cfg.get_monitored_dci_formats());
78 switch (uss_dci_fmt) {
79 case search_space_configuration::ue_specific_dci_format::f0_0_and_f1_0:
80 return dci_dl_format::f1_0;
81 case search_space_configuration::ue_specific_dci_format::f0_1_and_1_1:
82 return dci_dl_format::f1_1;
83 }
84 report_fatal_error("Unsupported DL DCI format");
85}

118inline bool is_pdcch_monitoring_active(slot_point sl, const search_space_configuration& ss_cfg)
119{
120 const unsigned slot_offset = sl.to_uint() % ss_cfg.get_monitoring_slot_periodicity();
121 const unsigned window_start = ss_cfg.get_monitoring_slot_offset();
122 const unsigned window_end = (window_start + ss_cfg.get_duration()) % ss_cfg.get_monitoring_slot_periodicity();
123
124 // Checks whether slot_offset falls [window_start, window_end), taking into account the wrap-around of "window_end".
125 if (window_start < window_end) {
126 return slot_offset >= window_start and slot_offset < window_end;
127 }
128 return slot_offset >= window_start or slot_offset < window_end;
129}
