209 unsigned get_first_symbol_index(uint8_t ssb_beam_idx = 0) const
210 {
211 monitoring_symbols_within_slot_t symbols_within_slot;
212 if (is_search_space0()) {
213 // TODO: Revise this when FR2 support is added.
214 // As per TS 38.213, Table 13-11.
215 srsran_assert(ssb_beam_idx < monitoring_symbols_within_slot.size(), "Invalid SSB beam index={}", ssb_beam_idx);
216 symbols_within_slot = monitoring_symbols_within_slot[ssb_beam_idx];
217 } else {
218 srsran_assert(ssb_beam_idx == 0, "SSB beam index is not used for SearchSpace Id > 0");
219 symbols_within_slot = monitoring_symbols_within_slot.back();
220 }
221 for (unsigned n = 0; n < symbols_within_slot.size(); ++n) {
222 if (symbols_within_slot.test(n)) {
223 return n;
224 }
225 }
226 srsran_assertion_failure("Monitoring symbols within slot for SS id {} doesn't have any symbols set to 1", id);
227 return symbols_within_slot.size();
228 }
