135template <concurrent_queue_policy QueuePolicy = concurrent_queue_policy::locking_mpsc,
136 concurrent_queue_wait_policy WaitPolicy = concurrent_queue_wait_policy::condition_variable>
137class general_task_worker_executor final : public task_executor
138{
139public:
140 general_task_worker_executor() = default;
141
142 general_task_worker_executor(general_task_worker<QueuePolicy, WaitPolicy>& worker_) : worker(&worker_) {}
143
144 SRSRAN_NODISCARD bool execute(unique_task task) override
145 {
146 if (can_run_task_inline()) {
147 // Same thread. Run task right away.
148 task();
149 return true;
150 }
151 return defer(std::move(task));
152 }
153
154 SRSRAN_NODISCARD bool defer(unique_task task) override { return worker->push_task(std::move(task)); }
155
157 bool can_run_task_inline() const { return worker->get_id() == std::this_thread::get_id(); }
158
159private:
160 general_task_worker<QueuePolicy, WaitPolicy>* worker = nullptr;
161};
