100 void resize(size_t nof_bytes)

107 void append(span<const uint8_t> bytes) { append(bytes.begin(), bytes.end()); }

110 template <typename It>
111 void append(It it_begin, It it_end)
112 {
113 static_assert(std::is_same<std::decay_t<decltype(*it_begin)>, uint8_t>::value, "Invalid value_type");
114 static_assert(
115 std::is_same<typename std::iterator_traits<It>::iterator_category, std::random_access_iterator_tag>::value,
116 "Only random access iterators allowed.");
117 srsran_assert((size_t)(it_end - it_begin) <= tailroom(), "There is not enough tailroom for append.");
118 iterator new_end = std::copy(it_begin, it_end, end());
119 payload = span<uint8_t>{payload.begin(), new_end};
120 }
121
123 void append(uint8_t byte)
124 {
125 srsran_assert(tailroom() >= 1, "There is not enough tailroom space.");
126 payload = span<uint8_t>{payload.begin(), payload.end() + 1};
127 payload.back() = byte;
128 }
129
131 void prepend(span<const uint8_t> bytes)
132 {
133 srsran_assert(headroom() >= bytes.size(), "There is not enough headroom space.");
134 payload = span<uint8_t>{payload.begin() - bytes.size(), payload.end()};
135 std::copy(bytes.begin(), bytes.end(), begin());
136 }
137
140 void reserve_prepend(size_t nof_bytes)
141 {
142 srsran_assert(headroom() >= nof_bytes, "There is not enough headroom space.");
143 payload = span<uint8_t>{payload.begin() - nof_bytes, payload.end()};
144 }
145
147 void trim_head(size_t nof_bytes)
148 {
149 srsran_assert(nof_bytes <= length(), "There is not enough headroom space.");
150 payload = span<uint8_t>{payload.begin() + nof_bytes, payload.end()};
151 }
152
154 void trim_tail(size_t nof_bytes)
155 {
156 srsran_assert(nof_bytes <= length(), "There is not enough headroom space.");
157 payload = span<uint8_t>{payload.begin(), payload.end() - nof_bytes};
158 }
159
161 iterator begin() { return payload.begin(); }
162 const_iterator begin() const { return payload.begin(); }
163
165 iterator end() { return payload.end(); }
166 const_iterator end() const { return payload.end(); }
167
169 template <typename Container>
170 bool operator==(const Container& other) const
171 {
172 static_assert(std::is_same<std::decay_t<decltype(*other.begin())>, uint8_t>::value, "Invalid value_type");
173 return std::equal(begin(), end(), other.begin(), other.end());
174 }
175 template <typename Container>
176 bool operator!=(const Container& other) const
177 {
178 return !(*this == other);
179 }
180
181private:
182 // Range of bytes spanning the beginning of the headroom and end of the tailroom.
183 span<const uint8_t> buffer;
184
185 // Range of bytes where the payload is stored.
186 span<uint8_t> payload;
187};
