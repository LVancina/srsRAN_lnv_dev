33class phy_time_unit
34{
35private:
37 using value_type = int64_t;
39 static constexpr value_type MAX_SCS_HZ = 480 * 1000;
41 static constexpr value_type N_F = 4096;
43 static constexpr double T_C = 1.0 / static_cast<double>(MAX_SCS_HZ * N_F);
45 static constexpr value_type SCS_REF_HZ = 15 * 1000;
47 static constexpr value_type N_F_REF = 2048;
49 static constexpr value_type KAPPA = 64;
50
52 value_type value = 0;
53
55 explicit constexpr phy_time_unit(value_type value_) : value(value_)
56 {
57 // Do nothing.
58 }
59
60public:
62 phy_time_unit() = default;
63
65 const value_type to_Tc() const { return value; }
66
69 template <class U = double>
70 constexpr U to_seconds() const
71 {
72 static_assert(std::is_convertible<double, U>::value && std::is_floating_point<U>::value, "Invalid type.");
73 return static_cast<U>(value) * static_cast<U>(T_C);
74 }
75
84 template <class U>
85 constexpr bool is_sample_accurate(U sampling_rate_Hz_) const
86 {
87 static_assert(std::is_convertible<U, value_type>::value, "Invalid type.");
88 value_type sampling_rate_Hz = static_cast<value_type>(sampling_rate_Hz_);
89 return ((std::abs(value) * sampling_rate_Hz) % (SCS_REF_HZ * N_F_REF * KAPPA)) == 0;
90 }
91
100 template <typename U>
101 constexpr value_type to_samples(U sampling_rate_Hz_) const
102 {
103 static_assert(std::is_convertible<U, value_type>::value, "Invalid type.");
104 value_type sampling_rate_Hz = static_cast<value_type>(sampling_rate_Hz_);
105 srsran_assert(is_sample_accurate(sampling_rate_Hz),
106 "Incompatible sampling rate {}.{:02} MHz with time {}.",
107 sampling_rate_Hz / 1000000,
108 (sampling_rate_Hz % 1000000) / 10000,
109 value);
110 return (value * sampling_rate_Hz) / (SCS_REF_HZ * N_F_REF * KAPPA);
111 }
112
119 constexpr unsigned to_Ta(subcarrier_spacing scs) const
120 {
121 return divide_round(value * pow2(to_numerology_value(scs)), 16U * KAPPA);
122 }

221 static constexpr inline phy_time_unit from_seconds(double seconds)
222 {
223 // Convert to units of Tc.
224 double tc_units_dbl = seconds / T_C;
225 // Multiply by ten and convert to value_type.
226 value_type tc_units = static_cast<value_type>(tc_units_dbl * 10.0);
227 // Round to the nearest integer avoiding using std::round.
228 return phy_time_unit(tc_units / 10 + (tc_units % 10) / 5);
229 }
