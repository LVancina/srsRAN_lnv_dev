42template <typename ComplexType = cf_t>
43class complex_normal_distribution
44{
45 // NB: complex types are always floating point types.
46 static_assert(is_complex<ComplexType>::value, "result_type must be a complex floating point type.");
47
48public:
50 using result_type = ComplexType;
52 using stddev_type = typename ComplexType::value_type;
53
58 struct param_type {
59 using distribution_type = complex_normal_distribution<ComplexType>;
60
62 param_type() : param_type(ComplexType(0, 0)) {}
63
68 explicit param_type(result_type m, stddev_type sd = stddev_type(1)) : mean_v(m), stddev_v(sd)
69 {
70 srsran_assert(stddev_v > stddev_type(0), "Standard deviation should be positive, given {}.", stddev_v);
71 }
72
74 result_type mean() const { return mean_v; }
75
77 stddev_type stddev() const { return stddev_v; }
78
80 friend bool operator==(const param_type& p1, const param_type& p2)
81 {
82 return ((p1.mean_v == p2.mean_v) && (p1.stddev_v == p2.stddev_v));
83 }
84
86 friend bool operator!=(const param_type& p1, const param_type& p2) { return !(p1 == p2); }
87
88 private:
89 result_type mean_v;
90 stddev_type stddev_v;
91 };
92
94 complex_normal_distribution() : complex_normal_distribution(0.0) {}
95
97 explicit complex_normal_distribution(result_type mean_v, stddev_type stddev_v = stddev_type(1)) :
98 parameters(mean_v, stddev_v)
99 {
100 }
101
103 explicit complex_normal_distribution(const param_type& params) : parameters(params) {}
104
106 void reset() { real_normal.reset(); }
107
109 result_type mean() const { return parameters.mean(); }
110
112 stddev_type stddev() const { return parameters.stddev(); }
113
115 param_type param() const { return parameters; }
116
118 void param(const param_type& pp) { parameters = pp; }
119
121 template <typename UniformRandomNumberGenerator>
122 result_type operator()(UniformRandomNumberGenerator& urng)
123 {
124 return this->operator()(urng, parameters);
125 }
126
128 template <typename UniformRandomNumberGenerator>
129 result_type operator()(UniformRandomNumberGenerator& urng, const param_type& pp)
130 {
131 // Square root of 1/2, for variance normalization.
132 static const stddev_type SQRT1_2 = 1 / std::sqrt(2.0);
133
134 result_type tmp(real_normal(urng), real_normal(urng));
135 return (tmp * pp.stddev() * SQRT1_2 + pp.mean());
136 }
137
139 template <typename ComplexType1>
140 friend bool operator==(const complex_normal_distribution<ComplexType1>& distr1,
141 const complex_normal_distribution<ComplexType1>& distr2);
142
143private:
145 param_type parameters;
147 std::normal_distribution<stddev_type> real_normal = {};
148};

154{

156}
