32class ofdm_symbol_demodulator_spy : public ofdm_symbol_demodulator
33{
34public:
35 struct demodulate_entry {
36 std::vector<cf_t> input;
37 const resource_grid_writer* grid;
38 unsigned port_index;
39 unsigned symbol_index;
40 };
41
42 ofdm_symbol_demodulator_spy(const ofdm_demodulator_configuration& config) :
43 rgen(0), dist(-1, +1), configuration(config)
44 {
45 // Do nothing.
46 }
47
48 unsigned get_symbol_size(unsigned symbol_index) const override
49 {
50 unsigned sampling_rate_Hz = (configuration.dft_size * 15000) << configuration.numerology;
51 return configuration.cp.get_length(symbol_index, to_subcarrier_spacing(configuration.numerology))
52 .to_samples(sampling_rate_Hz) +
53 configuration.dft_size;
54 }
55
56 void
57 demodulate(resource_grid_writer& grid, span<const cf_t> input, unsigned port_index, unsigned symbol_index) override
58 {
59 TESTASSERT_EQ(input.size(), get_symbol_size(symbol_index));
60
61 demodulate_entries.emplace_back();
62 demodulate_entry& entry = demodulate_entries.back();
63 entry.grid = &grid;
64 entry.port_index = port_index;
65 entry.symbol_index = symbol_index;
66
67 entry.input.resize(input.size());
68 srsvec::copy(entry.input, input);
69 }
70
71 void clear_demodulate_entries() { demodulate_entries.clear(); }
72
73 const ofdm_demodulator_configuration& get_configuration() const { return configuration; }
74
75 const std::vector<demodulate_entry>& get_demodulate_entries() const { return demodulate_entries; }
76
77private:
78 std::mt19937 rgen;
79 std::uniform_real_distribution<float> dist;
80 ofdm_demodulator_configuration configuration;
81 std::vector<demodulate_entry> demodulate_entries;
82};
