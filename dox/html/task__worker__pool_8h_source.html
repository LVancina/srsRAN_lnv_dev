231template <concurrent_queue_policy QueuePolicy>
232class priority_task_worker_pool_executor final : public task_executor
233{
234public:
235 priority_task_worker_pool_executor() = default;
236 priority_task_worker_pool_executor(priority_enqueuer<unique_task, QueuePolicy> enqueuer_,
237 enqueue_priority prio_,
238 detail::base_worker_pool& workers_) :
239 enqueuer(std::move(enqueuer_)), prio(prio_), workers(workers_)
240 {
241 }
242
243 SRSRAN_NODISCARD bool execute(unique_task task) override
244 {
245 if (can_run_task_inline()) {
246 task();
247 return true;
248 }
249 return enqueuer.try_push(std::move(task));
250 }
251
252 SRSRAN_NODISCARD bool defer(unique_task task) override { return enqueuer.try_push(std::move(task)); }
253
256 bool can_run_task_inline() const { return prio == enqueue_priority::max and workers.is_in_thread_pool(); }
257
258private:
259 priority_enqueuer<unique_task, QueuePolicy> enqueuer;
260 enqueue_priority prio;
261 detail::base_worker_pool& workers;
262};

283template <typename Func, concurrent_queue_policy... QueuePolicies, size_t... Is>
284void visit_executor(task_worker_pool<QueuePolicies...>& w,
285 enqueue_priority priority,
286 const Func& func,
287 std::index_sequence<Is...> /*unused*/)
288{
289 const size_t idx = detail::enqueue_priority_to_queue_index(priority, sizeof...(QueuePolicies));
290 (void)std::initializer_list<int>{[idx, &w, &func]() {
291 if (idx == Is) {
292 func(
293 make_priority_task_worker_pool_executor<detail::queue_index_to_enqueue_priority(Is, sizeof...(QueuePolicies)),
294 QueuePolicies...>(w));
295 }
296 return 0;
297 }()...};
298}
