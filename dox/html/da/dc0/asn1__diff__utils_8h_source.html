39/// Creates an functor that applies operator< comparison between objects based on their id.
40template <typename GetId>
41struct obj_id_comparator {
42 using id_type = detail::function_return_t<decltype(&std::decay_t<GetId>::operator())>;
43
44 obj_id_comparator(GetId&& c_) : func(std::forward<GetId>(c_)) {}
45
46 template <typename T, typename U>
47 bool operator()(const T& lhs, const U& rhs) const
48 {
49 return get_id(lhs) < get_id(rhs);
50 }
51
52private:
53 template <typename T>
54 id_type get_id(const T& t) const
55 {
56 return func(t);
57 }
58 id_type get_id(const id_type& t) const { return t; }
59
60 GetId func;
61};
62
63template <typename GetId>
64obj_id_comparator<GetId> make_id_comparator(GetId&& g)
65{
66 return {std::forward<GetId>(g)};
67}

69/**
70 * Apply toAddModList/toRemoveList changes
71 * @param prev_list original list of rrc fields
72 * @param add_diff_list added/modified elements
73 * @param rm_diff_list removed elements
74 * @param next_list resulting list. (Can be same as prev_list)
75 */
76template <typename AddModList, typename RemoveList, typename GetId>
77void apply_addmodremlist_diff(const AddModList& prev_list,
78 const AddModList& add_diff_list,
79 const RemoveList& rm_diff_list,
80 AddModList& next_list,
81 const GetId& id_func)
82{
83 if (add_diff_list.size() == 0 and rm_diff_list.size() == 0) {
84 if (&next_list != &prev_list) {
85 next_list = prev_list;
86 }
87 return;
88 }
89 auto id_cmp_op = make_id_comparator(id_func);
90 srsran_sanity_check(std::is_sorted(prev_list.begin(), prev_list.end(), id_cmp_op), "Expected sorted list");
91 srsran_sanity_check(std::is_sorted(add_diff_list.begin(), add_diff_list.end(), id_cmp_op), "Expected sorted list");
92 srsran_sanity_check(std::is_sorted(rm_diff_list.begin(), rm_diff_list.end()), "Expected sorted list");
93
94 AddModList tmp_lst;
95 // apply remove list
96 std::set_difference(prev_list.begin(),
97 prev_list.end(),
98 rm_diff_list.begin(),
99 rm_diff_list.end(),
100 std::back_inserter(tmp_lst),
101 id_cmp_op);
102
103 // apply toaddmodlist
104 next_list.clear();
105 std::set_union(add_diff_list.begin(),
106 add_diff_list.end(),
107 tmp_lst.begin(),
108 tmp_lst.end(),
109 std::back_inserter(next_list),
110 id_cmp_op);
111}

122/// \brief Invokes rem_func, add_func and mod_func for any detected differences between prev_list and next_list.
123/// \param rem_func Callable with signature void(const obj_type&), where obj_type is the value_type of \c List,
124/// which gets called for every object whose id exists in \c prev_list but does not exist in \c next_list.
125/// \param add_func Callable with signature void(const obj_type&), where obj_type is the value_type of \c List,
126/// which gets called for every object whose id exists in \c next_list but does not exist in \c prev_list.
127/// \param mod_func Callable with signature void(const obj_type&), where obj_type is the value_type of \c List,
128/// which gets called for every object whose id exists in both \c next_list and \c prev_list, but its parameters are
129/// different between the two lists.
130/// \tparam List List of objects, where each object has an ID.
131/// \tparam GetId Function to obtain ID of an object.
132/// \param[in] prev_list Previous list of objects.
133/// \param[in] next_list Next list of objects.
134/// \param[in] id_func function to extract the ID from an object with type equal to the value_type of \c List.
135template <typename List, typename RemFunctor, typename AddFunctor, typename ModFunctor, typename GetId>
136std::enable_if_t<not is_iterable<AddFunctor>::value> calculate_addmodremlist_diff(const RemFunctor& rem_func,
137 const AddFunctor& add_func,
138 const ModFunctor& mod_func,
139 const List& prev_list,
140 const List& next_list,
141 const GetId& id_func)
142{
143 auto id_cmp_op = make_id_comparator(id_func);
144 srsran_sanity_check(std::is_sorted(prev_list.begin(), prev_list.end(), id_cmp_op), "Expected sorted list");
145 srsran_sanity_check(std::is_sorted(next_list.begin(), next_list.end(), id_cmp_op), "Expected sorted list");
146
147 auto src_it = prev_list.begin();
148 auto target_it = next_list.begin();
149
150 bool src_left = src_it != prev_list.end();
151 bool target_left = target_it != next_list.end();
152 while (src_left or target_left) {
153 if (not target_left or (src_left and id_cmp_op(*src_it, *target_it))) {
154 rem_func(src_it++);
155 } else if (not src_left or (target_left and id_cmp_op(*target_it, *src_it))) {
156 // a new object has been added to target
157 add_func(target_it++);
158 } else {
159 mod_func(src_it++, target_it++);
160 }
161 src_left = src_it != prev_list.end();
162 target_left = target_it != next_list.end();
163 }
164}

166/// \brief Generate toAddModList and toReleaseList based on differences between prev_list and next_list.
167/// \tparam toAddModList List of objects added or modified.
168/// \tparam RemoveList List of IDs for objects removed.
169/// \tparam List List of objects, where each object has an ID.
170/// \tparam ConvertElem Function to convert element of \c List to element of \c toAddModList.
171/// \tparam GetId Function to obtain ID of an object.
172/// \param[out] add_diff_list Difference in terms of added/modified objects between \c prev_list and \c next_list.
173/// \param[out] rem_diff_list Difference in terms of removed objects between \c prev_list and \c next_list.
174/// \param[in] prev_list Previous list of objects.
175/// \param[in] next_list Next list of objects.
176/// \param[in] convert_func Callable used to convert an object with type \c List::value_type into an object with type
177/// \c toAddModList::value_type.
178/// \param[in] id_func function to extract the ID from an object with type equal to the value_type of \c List.
179template <typename List, typename toAddModList, typename RemoveList, typename ConvertElem, typename GetId>
180std::enable_if_t<is_iterable<toAddModList>::value> calculate_addmodremlist_diff(toAddModList& add_diff_list,
181 RemoveList& rem_diff_list,
182 const List& prev_list,
183 const List& next_list,
184 const ConvertElem& convert_func,
185 const GetId& id_func)
186{
187 if (&prev_list == &next_list) {
188 // No difference because src and target are the same list. Early return.
189 add_diff_list.clear();
190 rem_diff_list.clear();
191 return;
192 }
193
194 add_diff_list.clear();
195 rem_diff_list.clear();
196
197 using it_t = typename List::const_iterator;
198 auto rem_func = [&rem_diff_list, &id_func](it_t rem_it) { rem_diff_list.push_back(id_func(*rem_it)); };
199 auto add_func = [&add_diff_list, &convert_func](it_t add_it) { add_diff_list.push_back(convert_func(*add_it)); };
200 auto mod_func = [&add_diff_list, &convert_func](it_t src_it, it_t target_it) {
201 if (not(*src_it == *target_it)) {
202 add_diff_list.push_back(convert_func(*target_it));
203 }
204 };
205 calculate_addmodremlist_diff(rem_func, add_func, mod_func, prev_list, next_list, id_func);
206}

208/// \brief Generate toAddModList and toReleaseList based on differences between prev_list and next_list.
209/// \tparam toAddModList List of objects added or modified.
210/// \tparam RemoveList List of IDs for objects removed.
211/// \tparam GetId Function to obtain ID of an object.
212/// \param[out] add_diff_list Difference in terms of added/modified objects between \c prev_list and \c next_list.
213/// \param[out] rem_diff_list Difference in terms of removed objects between \c prev_list and \c next_list.
214/// \param[in] prev_list Previous list of objects.
215/// \param[in] next_list Next list of objects.
216/// \param[in] id_func function to extract the ID from an object with type equal to the value_type of \c List.
217template <typename toAddModList, typename RemoveList, typename GetId>
218void calculate_addmodremlist_diff(toAddModList& add_diff_list,
219 RemoveList& rem_diff_list,
220 const toAddModList& prev_list,
221 const toAddModList& next_list,
222 const GetId& id_func)
223{
224 using elem_type = decltype(*prev_list.begin());
225
226 if (&prev_list == &add_diff_list) {
227 // use const prev_list instead.
228 toAddModList prev_list2 = prev_list;
229 calculate_addmodremlist_diff(add_diff_list, rem_diff_list, prev_list2, next_list, id_func);
230 return;
231 }
232
233 if (&next_list == &add_diff_list) {
234 // use const next_list
235 toAddModList next_list2 = next_list;
236 calculate_addmodremlist_diff(add_diff_list, rem_diff_list, prev_list, next_list2, id_func);
237 return;
238 }
239
240 calculate_addmodremlist_diff(
241 add_diff_list, rem_diff_list, prev_list, next_list, [](const elem_type& e) { return e; }, id_func);
242}

244/// \brief Sets up the setup/release ASN.1 object based on the difference between prev and next.
245///
246/// \param setup_rel Setup/release ASN.1 object.
247/// \param prev Previous optional value.
248/// \param next New optional value.
249/// \tparam ConvertElem Function to convert element of \c OptionalElem to element of \c Asn1Type.
250/// \return Whether changes were made to Setup/Release ASN.1 object.
251template <typename Asn1Type, typename OptionalElem, typename ConvertElem>
252bool calculate_setup_release(asn1::setup_release_c<Asn1Type>& setup_rel,
253 const optional<OptionalElem>& prev,
254 const optional<OptionalElem>& next,
255 const ConvertElem& convert_func)
256{
257 if (next.has_value()) {
258 if (not prev.has_value() or *prev != *next) {
259 setup_rel.set_setup() = convert_func(*next);
260 }
261 } else if (prev.has_value()) {
262 setup_rel.set_release();
263 } else {
264 setup_rel.set(asn1::setup_release_opts::nulltype);
265 }
266 return setup_rel.type().value != asn1::setup_release_opts::nulltype;
267}
