33/// Maximum number of candidates per aggregation level in a SS as per TS38.331 SearchSpace.
34static constexpr unsigned PDCCH_MAX_NOF_CANDIDATES_SS = 8;

40/// \brief Gets the maximum number of monitored PDCCH candidates per slot for a given subcarrier spacing, for a single
41/// serving cell, as per TS 38.213, Table 10.1-2.
42inline unsigned max_nof_monitored_pdcch_candidates(subcarrier_spacing scs)
43{
44 const static std::array<uint8_t, 4> max_monitored_pdcch_candidates_per_slot = {44, 36, 22, 20};
45 return max_monitored_pdcch_candidates_per_slot[to_numerology_value(scs)];
46}

48/// \brief Parameters for calculating the PDCCH lowest Control Channel Element (CCE).
49///
50/// The parameters in this struct allow computing the PDCCH lowest CCE, for all candidates and for a Common SS,
51/// following the specifics in TS38.213 Section 10.1.
52///
53/// Consider an active DL BWP of a serving cell with carrier indicator field \f$n_{CI}\f$. In the search space set
54/// \f$s\f$, associated with CORESET \f$p\f$ in slot \f$n_{s,f}^\mu\f$, the lowest CCE index \f$n_{CCE}\f$ corresponding
55/// to the PDCCH candidate \f$m_{s,n_{CI}}\f$ is given by
56/// \[
57/// n_{CCE} = L\cdot \left \{ \left ( Y_{p,n_{s,f}^\mu} + \left \lfloor \frac{m_{s,n_{CI}}\cdot N_{CCE,p}}{L\cdot
58/// M_{s,max}} \right \rfloor +n_{CI} \right ) mod \left \lfloor N_{CCE,p}/L \right \rfloor \right \}
59/// \]
60/// where:
61/// - \f$Y_{p,n_{s,f}^\mu}=0\f$,
62/// - \f$n_{CI}=0\f$,
63/// - \f$m_{s,n_{CI}}\in \{0,...,M_{s,n_{CI}}^{(L)}\}\f$,
64struct pdcch_candidates_common_ss_configuration {
65 /// Aggregation level \f$L\epsilon\{1,2,4,8,16\}\f$.
66 aggregation_level al;
67 /// \brief Number of PDCCH candidates \f$M_{s,max}^{(L)}\f$ the UE is configured to monitor for aggregation level
68 /// \f$L\f$ of a search space set \f$s\f$ for a serving cell corresponding to \f$n_{CI}\f$.
69 unsigned nof_candidates;
70 /// Number of CCEs contained in CORESET \f$p\f$, \f$N_{CCE,p}\f$.
71 unsigned nof_cce_coreset;
72};

74/// \brief Parameters for calculating the PDCCH lowest Control Channel Element (CCE).
75///
76/// The parameters in this struct allow computing the PDCCH lowest CCE, for all candidates and for a UE-Specific SS,
77/// following the specifics in TS38.213 Section 10.1.
78///
79/// Consider an active DL BWP of a serving cell with carrier indicator field \f$n_{CI}\f$. In the search space set
80/// \f$s\f$, associated with CORESET \f$p\f$ in slot \f$n_{s,f}^\mu\f$, the lowest CCE index \f$n_{CCE}\f$ corresponding
81/// to the PDCCH candidate \f$m_{s,n_{CI}}\f$ is given by
82/// \[
83/// n_{CCE} = L\cdot \left \{ \left ( Y_{p,n_{s,f}^\mu} + \left \lfloor \frac{m_{s,n_{CI}}\cdot N_{CCE,p}}{L\cdot
84/// M_{s,max}} \right \rfloor +n_{CI} \right ) mod \left \lfloor N_{CCE,p}/L \right \rfloor \right \}
85/// \]
86/// where:
87/// - \f$Y_{p,n_{s,f}^\mu}\f$ is calculated as a function of parameters \f$p\f$, \f$n_\textup{RNTI}\f$ and
88/// \f$n_{s,f}^\mu\f$;
89/// - \f$n_{CI}=0\f$,
90/// - \f$m_{s,n_{CI}}\in \{0,...,M_{s,n_{CI}}^{(L)}\}\f$.
91struct pdcch_candidates_ue_ss_configuration {
92 /// Aggregation level \f$L\epsilon\{1,2,4,8,16\}\f$.
93 aggregation_level al;
94 /// \brief Number of PDCCH candidates \f$M_{s,max}^{(L)}\f$ the UE is configured to monitor for aggregation level
95 /// \f$L\f$ of a search space set \f$s\f$ for a serving cell corresponding to \f$n_{CI}\f$.
96 unsigned nof_candidates;
97 /// Number of CCEs contained in CORESET \f$p\f$, \f$N_{CCE,p}\f$.
98 unsigned nof_cce_coreset;
99 /// CORESET identifier. Parameter \f$p\f$.
100 coreset_id cs_id;
101 /// Parameter \f$n_\textup{RNTI}\f$.
102 rnti_t rnti;
103 /// Slot index within the radio frame. Parameter \f$n_{s,f}^\mu\f$.
104 unsigned slot_index;
105};
