51/// Adapter between E1AP and CU-CP
52class e1ap_cu_cp_adapter : public e1ap_cu_cp_notifier
53{
54public:
55 void connect_cu_cp(cu_cp_e1ap_handler& cu_cp_handler_) { cu_cp_handler = &cu_cp_handler_; }
56
57 /// \brief Notifies about the creation of an E1AP.
58 /// \param[in] bearer_context_manager The E1AP Bearer Context Manager interface.
59 /// \param[in] bearer_removal_handler The E1AP bearer context removal handler.
60 /// \param[in] e1ap_statistic_handler The E1AP statistic interface.
61 void on_e1ap_created(e1ap_bearer_context_manager& bearer_context_manager,
62 e1ap_bearer_context_removal_handler& bearer_removal_handler,
63 e1ap_statistics_handler& e1ap_statistic_handler) override
64 {
65 srsran_assert(cu_cp_handler != nullptr, "E1AP handler must not be nullptr");
66 cu_cp_handler->handle_e1ap_created(bearer_context_manager, bearer_removal_handler, e1ap_statistic_handler);
67 }
68
69 void on_bearer_context_inactivity_notification_received(const cu_cp_inactivity_notification& msg) override
70 {
71 srsran_assert(cu_cp_handler != nullptr, "E1AP handler must not be nullptr");
72 cu_cp_handler->handle_bearer_context_inactivity_notification(msg);
73 }
74
75private:
76 cu_cp_e1ap_handler* cu_cp_handler = nullptr;
77};
