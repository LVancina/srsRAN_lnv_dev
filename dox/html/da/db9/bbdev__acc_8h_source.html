34/// Maximum number of queues supported by a bbdev-based hardware-accelerator.
35static constexpr unsigned MAX_NOF_BBDEV_QUEUES = 128;
36/// Maximum number of operations that can be stored in a hardware queue at a given time.
37static constexpr unsigned MAX_NOF_OP_IN_QUEUE = 16;
38
39/// Configuration parameters and objects tied to a bbdev-based hardware-accelerator.
40struct bbdev_acc_configuration {
41 /// ID of the bbdev-based hardware-accelerator.
42 unsigned id;
43 /// Number of lcores available to the hardware-accelerated LDPC encoder (disabled if 0).
44 unsigned nof_ldpc_enc_lcores = 0;
45 /// Number of lcores available to the hardware-accelerated LDPC decoder (disabled if 0).
46 unsigned nof_ldpc_dec_lcores = 0;
47 /// Number of lcores available to the hardware-accelerated FFT (disabled if 0).
48 unsigned nof_fft_lcores = 0;
49 /// Size of each mbuf used to exchange unencoded and unrate-matched messages with the accelerator in bytes.
50 /// Note that by default it is initialized to the maximum size supported by an mbuf.
51 unsigned msg_mbuf_size = RTE_BBDEV_LDPC_E_MAX_MBUF;
52 /// Size of each mbuf used to exchange encoded and rate-matched messages with the accelerator in bytes.
53 /// Note that by default it is initialized to the maximum size supported by an mbuf.
54 unsigned rm_mbuf_size = RTE_BBDEV_LDPC_E_MAX_MBUF;
55 /// Number of mbufs in each memory pool.
56 unsigned nof_mbuf = 256;
57};

59/// Abstracted interfacing to bbdev-based hardware-accelerators.
60class bbdev_acc
61{
62public:
63 /// Constructor.
64 /// \param[in] dpdk Pointer to a dpdk-based hardware-accelerator interface.
65 /// \param[in] info bbdev Device information.
66 /// \param[in] cfg Configuration parameters of the bbdev-based hardware-accelerator.
67 explicit bbdev_acc(const bbdev_acc_configuration& cfg, const ::rte_bbdev_info& info_, srslog::basic_logger& logger);
68
69 /// Destructor.
70 ~bbdev_acc();
71
72 /// Returns the ID of the bbdev-based hardware-accelerator device.
73 /// \return Device ID.
74 unsigned get_device_id() const { return id; }
75
76 /// Returns the ID of the socket used by the bbdev-based hardware-accelerator.
77 /// \return Socket ID.
78 int get_socket_id() const { return info.socket_id; }
79
80 /// Returns the number of LDPC encoder cores provided by the bbdev-based hardware-accelerator.
81 /// \return Number of LDPC encoder cores.
82 unsigned get_nof_ldpc_enc_cores() const { return nof_ldpc_enc_lcores; }
83
84 /// Returns the number of LDPC decoder cores provided by the bbdev-based hardware-accelerator.
85 /// \return Number of LDPC decoder cores.
86 unsigned get_nof_ldpc_dec_cores() const { return nof_ldpc_dec_lcores; }
87
88 /// Returns the number of FFT cores provided by the bbdev-based hardware-accelerator.
89 /// \return Number of FFT cores.
90 unsigned get_nof_fft_cores() const { return nof_fft_lcores; }
91
92 /// Returns the size of the (external) HARQ buffer size embedded in the hardware-accelerator.
93 /// \return HARQ buffer size (in bytes).
94 units::bytes get_harq_buff_size() const { return units::bytes(1024 * info.drv.harq_buffer_size); }
95
96 /// Returns the size of each mbuf used to exchange unencoded and unrate-matched messages with the accelerator.
97 /// \return Unencoded and unrate-matched mbuf size (in bytes).
98 units::bytes get_msg_mbuf_size() const { return units::bytes(msg_mbuf_size); }
99
100 /// Returns the size of each mbuf used to exchange encoded and rate-matched messages with the accelerator.
101 /// \return Encoded and rate-matched mbuf size (in bytes).
102 units::bytes get_rm_mbuf_size() const { return units::bytes(rm_mbuf_size); }
103
104 /// Returns the number of mbufs in each memory pool used to exchange data with the accelerator.
105 /// \return Number of mbufs.
106 unsigned get_nof_mbuf() const { return nof_mbuf; }
107
108 /// Returns the internal SRS logger.
109 /// \return SRS logger.
110 srslog::basic_logger& get_logger() { return logger; }
111
112 /// Reserves a free queue to be used by a specific hardware-accelerated channel processor function.
113 /// \param[in] op_type Type of bbdev op.
114 /// \return ID of the reserved queue.
115 int reserve_queue(rte_bbdev_op_type op_type);
116
117 /// Frees a queue used by a specific hardware-accelerated channel processor function.
118 /// \param[in] queue_id ID of the queue to be freed.
119 void free_queue(unsigned queue_id);
120
121private:
122 /// ID of the bbdev-based hardware-accelerator.
123 unsigned id;
124 /// Structure providing device information.
125 ::rte_bbdev_info info;
126 /// Number of lcores available to the hardware-accelerated LDPC encoder (disabled if 0).
127 unsigned nof_ldpc_enc_lcores;
128 /// Number of lcores available to the hardware-accelerated LDPC decoder (disabled if 0).
129 unsigned nof_ldpc_dec_lcores;
130 /// Number of lcores available to the hardware-accelerated FFT (disabled if 0).
131 unsigned nof_fft_lcores;
132 /// Array indicating the queue availability.
133 bounded_bitset<MAX_NOF_BBDEV_QUEUES> queue_used;
134 /// Size of each mbuf used to exchange unencoded and unrate-matched messages with the accelerator in bytes.
135 unsigned msg_mbuf_size;
136 /// Size of each mbuf used to exchange encoded and rate-matched messages with the accelerator in bytes.
137 unsigned rm_mbuf_size;
138 /// Number of mbufs in each memory pool.
139 unsigned nof_mbuf;
140 /// SRS logger.
141 srslog::basic_logger& logger;
142};
