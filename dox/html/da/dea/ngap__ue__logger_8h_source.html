34class ngap_ue_log_prefix
35{
36public:
37 ngap_ue_log_prefix(ue_index_t ue_index,
38 ran_ue_id_t ran_ue_id = ran_ue_id_t::invalid,
39 amf_ue_id_t amf_ue_id = amf_ue_id_t::invalid)
40 {
41 fmt::memory_buffer buffer;
42 fmt::format_to(buffer,
43 "ue={}{}{}: ",
44 ue_index,
45 ran_ue_id != ran_ue_id_t::invalid ? fmt::format(" ran_ue_id={}", ran_ue_id) : "",
46 amf_ue_id != amf_ue_id_t::invalid ? fmt::format(" amf_ue_id={}", amf_ue_id) : "");
47 prefix = srsran::to_c_str(buffer);
48 }
49 const char* to_c_str() const { return prefix.c_str(); }
50
51private:
52 std::string prefix;
53};
