410 /*
411 * - update RX_Next_Reassembly to the SN of the first SN >= RX_Timer_Trigger that has not been reassembled;
412 * - discard all segments with SN < updated RX_Next_Reassembly;
413 */
414 uint32_t sn_upd = st.rx_next_reassembly; // lowest element from which we have to clean the rx_window
415 for (; rx_mod_base(sn_upd) < rx_mod_base(st.rx_next_highest); sn_upd = (sn_upd + 1) % mod) {
416 if (rx_mod_base(sn_upd) < rx_mod_base(st.rx_timer_trigger)) {
417 // remove anything below rx_timer_trigger
418 if (rx_window->has_sn(sn_upd)) {
419 if (not(*rx_window)[sn_upd].fully_received) {
420 // count incomplete SDUs as lost
421 logger.log_info("Discarding incomplete SDU. sn={}", sn_upd);
422 metrics.metrics_add_lost_pdus(1);
423 }
424 rx_window->remove_sn(sn_upd);
425 } else {
426 // count non-existing SDUs as lost
427 logger.log_info("Discarding SDU. sn={}", sn_upd);
428 metrics.metrics_add_lost_pdus(1);
429 }
430 } else {
431 // continue removing fully received SDUs starting from rx_timer_trigger; stop at first incomplete or unseen SDU.
432 if (rx_window->has_sn(sn_upd) && (*rx_window)[sn_upd].fully_received) {

434 } else {
435 break; // first SDU not fully received
436 }
437 }
438 }
