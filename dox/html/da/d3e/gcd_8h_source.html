30/// Calculates the greatest common divisor (GCD) of two integers.
31template <typename Integer>
32Integer gcd(Integer a, Integer b)
33{
34 while (true) {
35 if (a == 0) {
36 return b;
37 }
38 b %= a;
39 if (b == 0) {
40 return a;
41 }
42 a %= b;
43 }
44}
45
46/// Calculates the least common multiplier (LCM) of two integers.
47template <typename Integer>
48Integer lcm(Integer a, Integer b)
49{
50 Integer temp = gcd(a, b);
51
52 return temp != 0 ? (a / temp * b) : 0;
53}
54
55/// Calculates the least common multiplier (LCM) for a range of integers.
56template <typename Integer>
57Integer lcm(span<const Integer> values)
58{
59 return std::accumulate(
60 values.begin(), values.end(), Integer(1), [](Integer a, Integer b) { return lcm<Integer>(a, b); });
61}
