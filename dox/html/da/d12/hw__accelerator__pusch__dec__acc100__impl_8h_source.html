113public:
114 /// Constructor taking care of obtaining a bbdev-based hardware-accelerator queue and allocating the required
115 /// resources.
116 hw_accelerator_pusch_dec_acc100_impl(std::shared_ptr<srsran::dpdk::bbdev_acc> bbdev_accelerator_,
117 bool ext_softbuffer_,
118 std::shared_ptr<ext_harq_buffer_context_repository> harq_buffer_context_) :
119 bbdev_accelerator(std::move(bbdev_accelerator_)),
120 ext_softbuffer(ext_softbuffer_),
121 harq_buffer_context(std::move(harq_buffer_context_))
122 {
123 int qid = bbdev_accelerator->reserve_queue(RTE_BBDEV_OP_LDPC_DEC);
124 srsran_assert(qid >= 0, "No free bbdev queues available.");
125 queue_id = static_cast<uint16_t>(qid);
126 device_id = bbdev_accelerator->get_device_id();
127 allocate_resources();
128 drop_op.resize(MAX_NOF_SEGMENTS);
129 drop_op.reset();
130 harq_context_entries.reserve(MAX_NOF_SEGMENTS);
131
132 // HAL logging.
133 srslog::basic_logger& logger = bbdev_accelerator->get_logger();
134 logger.info("[acc100] new decoder: queue={}.", queue_id);
135 }
136
137 /// Destructor taking care of freeing the utilized resources and hardware-accelerator queue.
138 ~hw_accelerator_pusch_dec_acc100_impl()
139 {
140 bbdev_accelerator->free_queue(queue_id);
141
142 // HAL logging.
143 srslog::basic_logger& logger = bbdev_accelerator->get_logger();
144 logger.info("[acc100] destroyed decoder: queue={}.", queue_id);
145 }
