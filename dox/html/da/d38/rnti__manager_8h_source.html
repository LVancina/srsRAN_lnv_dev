31using du_rnti_table = rnti_value_table<du_ue_index_t, du_ue_index_t::INVALID_DU_UE_INDEX>;
32
33/// \brief Extends DU RNTI Table with the ability to allocate unique RNTIs for UEs.
34class rnti_manager : public du_rnti_table
35{
36public:
37 /// \brief Allocates new unique TC-RNTI.
38 rnti_t allocate()
39 {
40 if (this->nof_ues() >= MAX_NOF_DU_UES) {
41 // If the number of UEs is already maximum, ignore RACH.
42 return rnti_t::INVALID_RNTI;
43 }
44 // Increments rnti counter until it finds an available temp C-RNTI.
45 rnti_t temp_crnti;
46 do {
47 uint16_t prev_counter = rnti_counter.fetch_add(1, std::memory_order_relaxed) % CRNTI_RANGE;
48 temp_crnti = to_rnti(prev_counter + to_value(rnti_t::MIN_CRNTI));
49 } while (this->has_rnti(temp_crnti));
50 return temp_crnti;
51 }
52
53private:
54 static constexpr int CRNTI_RANGE = to_value(rnti_t::MAX_CRNTI) + 1 - to_value(rnti_t::MIN_CRNTI);
55 static constexpr rnti_t INITIAL_RNTI = to_rnti(0x4601);
56
57 std::atomic<std::underlying_type_t<rnti_t>> rnti_counter{to_value(INITIAL_RNTI) - to_value(rnti_t::MIN_CRNTI)};
58};
