51/// \brief Generates the list of cell PUCCH resources (Format 1 and 2) given the available PRBs.
52///
53/// The function attempts to allocate the maximum number of resources given the number of PRBs. PUCCH resources F1 and
54/// F2 are allocated on different RBs. The function attempts to spread the resources on both
55///// sides of the BWP.
56///
57/// \param[in] max_pucch_f1_rbs number of available RBs where to allocate PUCCH F1 resources.
58/// \param[in] max_pucch_f2_rbs number of available RBs where to allocate PUCCH F2 resources.
59/// \param[in] f1_params PUCCH F1 resource parameters.
60/// \param[in] f1_params PUCCH F2 resource parameters.
61/// \param[in] bwp_size_rbs Size of the BWP in RBs.
62/// \return The list of PUCCH resources for a cell. The list has the PUCCH Format 1 resources in front of the list, and
63/// the PUCCH Format 2 in the back of the list.
64/// \remark The function returns an empty list in the following cases: (i) If the given number of RBs is not enough to
65/// allocate at least 1 resource. (ii) If the RBs occupancy is larger than the BWP size. (iii) If F2 intra-slot
66/// frequency hopping is enabled with only 1 symbol.
67std::vector<pucch_resource> generate_cell_pucch_res_list_given_rbs(unsigned max_pucch_f1_rbs,
68 unsigned max_pucch_f2_rbs,
69 pucch_f1_params f1_params,
70 pucch_f2_params f2_params,
71 unsigned bwp_size_rbs);

93/// \brief Generates the list of PUCCH resources for a given UE.
94///
95/// This function generates the list of PUCCH F1 and F2 resources for a given UE, including the resources for HARQ-ACK
96/// reporting, SR and CSI. It also updates the PUCCH resource sets accordingly, as well as the pointers to the PUCCH F1
97/// resource for SR and to the PUCCH F2 resource for CSI.
98/// This function overwrites the default \c ServingCellConfig passed as a function input.
99///
100/// The UE's PUCCH resource list composed of:
101/// - \ref nof_ue_pucch_f1_res_harq PUCCH Format 1 resources for HARQ-ACK reporting, chosen from
102/// \ref nof_harq_pucch_sets possible sets of PUCCH Format 1 cell resources.
103/// - 1 PUCCH Format 1 resource for SR chosen from \ref nof_cell_pucch_f1_res_sr possible sets of PUCCH Format 1
104/// cell resources.
105/// - \ref nof_ue_pucch_f1_res_harq PUCCH Format 2 resources for HARQ-ACK reporting, chosen from
106/// \ref nof_harq_pucch_sets possible sets of PUCCH Format 2 cell resources.
107/// - 1 PUCCH Format 2 resource for CSI chosen from \ref nof_cell_pucch_f2_res_csi possible sets of PUCCH Format 1
108/// cell resources.
109///
110/// The returned UE PUCCH resource list \ref pucch_res_list contains the following resources, sorted as follows:
111/// [ F1-HARQ_0 ... F1-HARQ_N-1 F1-SR F2-HARQ_0 ... F2-HARQ_M-1 F2-CSI ]
112/// where N = nof_ue_pucch_f1_res_harq and M = nof_ue_pucch_f2_res_harq,
113/// and with the following indices \ref res_id:
114/// - The first \ref nof_ue_pucch_f1_res_harq are the PUCCH F1 resources for HARQ-ACK and have index
115/// [ (du_harq_set_idx % nof_harq_pucch_sets) * nof_ue_pucch_f1_res_harq,
116/// (du_harq_set_idx % nof_harq_pucch_sets) * nof_ue_pucch_f1_res_harq + nof_ue_pucch_f1_res_harq ).
117/// - The next resource in the list is the PUCCH F1 resource for SR, which have index:
118/// nof_harq_pucch_sets * nof_ue_pucch_f1_res_harq + du_sr_res_idx % nof_cell_pucch_f1_res_sr.
119/// - The next \ref nof_ue_pucch_f2_res_harq are the PUCCH F2 resources for HARQ-ACK and have index
120/// [ nof_harq_pucch_sets * nof_ue_pucch_f1_res_harq + nof_cell_pucch_f1_res_sr +
121/// (du_harq_set_idx % nof_harq_pucch_sets) * nof_ue_pucch_f2_res_harq,
122/// nof_harq_pucch_sets * nof_ue_pucch_f1_res_harq + nof_cell_pucch_f1_res_sr +
123/// (du_harq_set_idx % nof_harq_pucch_sets) * nof_ue_pucch_f2_res_harq + nof_ue_pucch_f2_res_harq).
124/// - The last resource in the list is the PUCCH F2 resource for CSI, which has index:
125//// nof_harq_pucch_sets * nof_ue_pucch_f1_res_harq + nof_cell_pucch_f1_res_sr +
126/// nof_ue_pucch_f2_res_harq * nof_harq_pucch_sets + du_csi_res_idx % nof_cell_pucch_f2_res_csi.
127///
128/// \param[in,out] serv_cell_cfg default \c ServingCellConfig that will be overwritten by this function.
129/// \param[in] res_list cell PUCCH resource list from which the function picks the UE PUCCH resources.
130/// \param[in] du_harq_set_idx defines which PUCCH resource set for HARQ to be assigned to this UE among
131/// \ref nof_harq_pucch_sets possible ones; the chosen set for this UE has index
132/// du_harq_set_idx % nof_harq_pucch_sets.
133/// \param[in] du_sr_res_idx defines which PUCCH resource for SR to be assigned to this UE among
134/// \ref nof_cell_pucch_f1_res_sr possible ones. Values: {0, ..., nof_cell_pucch_f1_res_sr-1}.
135/// \param[in] du_csi_res_idx defines which PUCCH resource for CSI to be assigned to this UE among
136/// \ref nof_cell_pucch_f2_res_csi possible ones. Values: {0, ..., nof_cell_pucch_f2_res_csi-1}.
137/// \param[in] nof_ue_pucch_f1_res_harq desired number of UE PUCCH F1 resources (HARQ-ACK) in UE configuration.
138/// \param[in] nof_ue_pucch_f2_res_harq desired number of UE PUCCH F2 resources (HARQ-ACK) in UE configuration.
139/// \param[in] nof_cell_pucch_f1_res_sr number of PUCCH F1 resources for SR available in the cell.
140/// \param[in] nof_cell_pucch_f2_res_csi number of PUCCH F2 resources for CSI available in the cell.
141/// \return true if the building is successful, false otherwise.
142bool ue_pucch_config_builder(serving_cell_config& serv_cell_cfg,
143 const std::vector<pucch_resource>& res_list,
144 unsigned du_harq_set_idx,
145 unsigned du_sr_res_idx,
146 unsigned du_csi_res_idx,
147 bounded_integer<unsigned, 1, max_ue_f1_res_harq> nof_ue_pucch_f1_res_harq,
148 bounded_integer<unsigned, 1, max_ue_f2_res_harq> nof_ue_pucch_f2_res_harq,
149 unsigned nof_harq_pucch_sets,
150 unsigned nof_cell_pucch_f1_res_sr,
151 unsigned nof_cell_pucch_f2_res_csi = 1);
