31/// Implements a generic physical layer receive symbol adapter.
32class phy_rx_symbol_adapter : public lower_phy_rx_symbol_notifier
33{
34private:
35 upper_phy_rx_symbol_handler* rx_symbol_handler = nullptr;
36
37public:
38 /// Connects the adaptor to the upper physical layer handler.
39 void connect(upper_phy_rx_symbol_handler* upper_handler) { rx_symbol_handler = upper_handler; }
40
41 // See interface for documentation.
42 void on_rx_symbol(const lower_phy_rx_symbol_context& context, const resource_grid_reader& grid) override
43 {
44 report_fatal_error_if_not(rx_symbol_handler, "Adapter is not connected.");
45 upper_phy_rx_symbol_context upper_context;
46 upper_context.slot = context.slot;
47 upper_context.sector = context.sector;
48 upper_context.symbol = context.nof_symbols;
49 rx_symbol_handler->handle_rx_symbol(upper_context, grid);
50 }
51
52 // See interface for documentation.
53 void on_rx_prach_window(const prach_buffer_context& context, const prach_buffer& buffer) override
54 {
55 report_fatal_error_if_not(rx_symbol_handler, "Adapter is not connected.");
56 rx_symbol_handler->handle_rx_prach_window(context, buffer);
57 }
58
59 // See interface for documentation.
60 void on_rx_srs_symbol(const lower_phy_rx_symbol_context& context) override
61 {
62 report_fatal_error_if_not(rx_symbol_handler, "Adapter is not connected.");
63 upper_phy_rx_symbol_context upper_context;
64 upper_context.slot = context.slot;
65 upper_context.sector = context.sector;
66 upper_context.symbol = context.nof_symbols;
67 rx_symbol_handler->handle_rx_srs_symbol(upper_context);
68 }
69};
70
