31/// Length of the generator state in bits.
32static constexpr unsigned pseudo_random_generator_state_size = 31;
33
34/// State sequence \f$x_1(n)\f$ initializer.
35class pseudo_random_initializer_x1
36{
37private:
38 /// Memory register for sequence \f$x_1(n)\f$.
39 std::array<uint32_t, pseudo_random_generator_state_size> table;
40
41public:
42 /// \brief Initializes the first 31 elements of \f$x_1(n)\f$ and advances to position \f$N_{\textup{C}}\f$.
43 ///
44 /// \param[in] Nc Parameter \f$N_{\textup{C}}\f$.
45 pseudo_random_initializer_x1(unsigned Nc)
46 {
47 static constexpr unsigned max_step_size = pseudo_random_generator_sequence::get_max_step_size();
48
49 // For each bit of the initial state.
50 for (unsigned i = 0; i != pseudo_random_generator_state_size; ++i) {
51 pseudo_random_generator_sequence sequence(1 << (31 - i), 0);
52
53 unsigned n = 0;
54 for (unsigned n_end = (Nc / max_step_size) * max_step_size; n != n_end; n += max_step_size) {
55 sequence.step<max_step_size>();
56 }
57 for (; n != Nc; ++n) {
58 sequence.step();
59 }
60
61 table[i] = sequence.get_x1();
62 }
63 }
64
65 /// \brief Gets the \f$x_1(n)\f$ state register after initialization.
66 ///
67 /// \param[in] c_init Initial \f$x_1(n)\f$ state as specified by TS38.211 Section 5.2.1. Set to 1 by default.
68 /// \return The \f$x_1(n)\f$ state register after initialization.
69 uint32_t get_reverse(unsigned c_init = 1) const
70 {
71 uint32_t ret = 0;
72 for (unsigned i = 0; i != pseudo_random_generator_state_size; ++i) {
73 if ((c_init >> i) & 1U) {
74 ret ^= table[i];
75 }
76 }
77
78 return ret;
79 }
80
81 /// \brief Gets the \f$x_1(n)\f$ state register after initialization.
82 ///
83 /// \param[in] init_state Initial \f$x_1(n)\f$ state.
84 /// \return The \f$x_1(n)\f$ state register after initialization.
85 uint32_t get(unsigned init_state) const
86 {
87 uint32_t ret = 0;
88 for (unsigned i = 0; i != pseudo_random_generator_state_size; ++i) {
89 if ((init_state << i) & (1U << 31U)) {
90 ret ^= table[i];
91 }
92 }
93
94 return ret;
95 }
96};

98/// \brief State sequence \f$x_2(n)\f$ initializer.
99///
100/// Similarly to pseudo_random_initializer_x1, this class is used to initialize the sequence \f$x_2(n)\f$ and advance it
101/// until position \f$n = N_{\textup{C}}\f$. Here, however, the process is carried out simultaneously for all possible
102/// seeds of the form \f$c_{\textup{init}} = 2^k\f$ for \f$k=0,1,\dots,30\f$. The initial state corresponding to any
103/// other seed can easily be computed from these basic ones after noticing that the map sending a seed to the
104/// corresponding sequence defines a group isomorphism between the set of seeds and the set of sequences (both groups
105/// under bitwise XOR). That is, if \f{align*} c_{\textup{init}}^{(1)} &\mapsto x_2^{(1)}(n) &&\text{and} &
106/// c_{\textup{init}}^{(2)} &\mapsto x_2^{(2)}(n)
107/// \f}
108/// then
109/// \f[
110/// c_{\textup{init}}^{(1)} \oplus c_{\textup{init}}^{(2)} \mapsto x_2^{(1)}(n) \oplus x_2^{(2)}(n)
111/// \f]
112/// with the XOR operator acting bitwise between seeds and for all \f$n\f$ between sequences.
113class pseudo_random_initializer_x2
114{
115private:
116 /// Memory register for sequence \f$x_2(n)\f$, for all basic seeds.
117 std::array<uint32_t, pseudo_random_generator_state_size> table;
118
119public:
120 /// Initializes the first 31 elements of \f$x_2(n)\f$ and advances to position \f$N_{\textup{C}}\f$.
121 pseudo_random_initializer_x2(unsigned Nc)
122 {
123 static constexpr unsigned max_step_size = pseudo_random_generator_sequence::get_max_step_size();
124
125 // For each bit of the seed.
126 for (uint32_t i = 0; i != pseudo_random_generator_state_size; ++i) {
127 pseudo_random_generator_sequence sequence(0, 1 << (31 - i));
128
129 unsigned n = 0;
130 for (unsigned n_end = (Nc / max_step_size) * max_step_size; n != n_end; n += max_step_size) {
131 sequence.step<max_step_size>();
132 }
133 for (; n != Nc; ++n) {
134 sequence.step();
135 }
136
137 table[i] = sequence.get_x2();
138 }
139 }
140
141 /// \brief Gets the \f$x_2(n)\f$ state register after initialization.
142 ///
143 /// \param[in] c_init Initial \f$x_2(n)\f$ state as specified by TS38.211 Section 5.2.1.
144 /// \return The \f$x_2(n)\f$ state register after initialization.
145 uint32_t get_reverse(unsigned c_init) const
146 {
147 uint32_t ret = 0;
148 for (unsigned i = 0; i != pseudo_random_generator_state_size; ++i) {
149 if ((c_init >> i) & 1U) {
150 ret ^= table[i];
151 }
152 }
153
154 return ret;
155 }
156
157 /// \brief Gets the \f$x_2(n)\f$ state register after initialization.
158 ///
159 /// \param[in] init_state Initial \f$x_2(n)\f$ state.
160 /// \return The \f$x_2(n)\f$ state register after initialization.
161 uint32_t get(unsigned init_state) const
162 {
163 uint32_t ret = 0;
164 for (unsigned i = 0; i != pseudo_random_generator_state_size; ++i) {
165 if ((init_state << i) & (1U << 31U)) {
166 ret ^= table[i];
167 }
168 }
169
170 return ret;
171 }
172};
