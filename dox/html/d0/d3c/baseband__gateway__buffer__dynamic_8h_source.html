92/// \brief Describes a baseband buffer implementation that comprises a fix number of channels that can be dynamically
93/// resized.
94///
95/// It contains a fix get_nof_channels() number of channels that contain the same number of get_nof_samples() samples.
96/// The number of samples can be changed in runtime without re-allocating memory using resize(). The number samples
97/// shall never exceed the maximum number of samples indicated in the constructor.
98class baseband_gateway_buffer_dynamic
99{
100public:
101 /// Gets the number of channels.
102 unsigned get_nof_channels() const { return data.get_dimension_size(detail::baseband_gateway_buffer_dims::channel); }
103
104 /// Gets the current number of samples.
105 unsigned get_nof_samples() const { return data.get_dimension_size(detail::baseband_gateway_buffer_dims::sample); }
106
107 /// Gets the reader interface.
108 const baseband_gateway_buffer_reader& get_reader() const { return reader; }
109
110 /// Gets the writer interface.
111 baseband_gateway_buffer_writer& get_writer() { return writer; }
112
113 /// Gets a data view of a channel.
114 span<cf_t> operator[](unsigned i_channel) { return writer.get_channel_buffer(i_channel); }
115
116 /// \brief Resize buffer.
117 /// \param[in] new_nof_samples Indicates the new number of samples per channel.
118 /// \note The new number of samples must be greater than 0 and must not exceed the maximum number of samples.
119 void resize(unsigned new_nof_samples) { data.resize({new_nof_samples, get_nof_channels()}); }
120
121 /// \brief Default constructor.
122 /// \param[in] nof_channels_ Indicates the number of channels to create.
123 /// \param[in] max_nof_samples_ Indicates the maximum number of samples.
124 baseband_gateway_buffer_dynamic(unsigned nof_channels, unsigned max_nof_samples) :
125 data({max_nof_samples, nof_channels}), reader(data), writer(data)
126 {
127 }
128
129 /// Move constructor.
130 baseband_gateway_buffer_dynamic(baseband_gateway_buffer_dynamic&& other) :
131 data(std::move(other.data)), reader(data), writer(data)
132 {
133 }
134
135private:
136 dynamic_tensor<static_cast<unsigned>(detail::baseband_gateway_buffer_dims::all),
137 cf_t,
138 detail::baseband_gateway_buffer_dims>
139 data;
140 detail::baseband_gateway_buffer_reader_tensor reader;
141 detail::baseband_gateway_buffer_writer_tensor writer;
142};
