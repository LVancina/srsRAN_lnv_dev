262 }

264 // Protect concurrent alignment and transmit.
265 std::unique_lock<std::mutex> lock(transmit_alignment_mutex);
266
267 // If the channel has never transmitted, skip wait.
268 if (is_tx_enabled && (timeout.count() != 0)) {
269 // Otherwise, wait for the transmitter to transmit.
270 bool is_not_timeout =
271 transmit_alignment_cvar.wait_for(lock, timeout, [this, timestamp]() { return sample_count >= timestamp; });
272 if (is_not_timeout) {
273 return sample_count > timestamp;
274 }
275 logger.debug("Transmit timed out. Enabling transmit zeros.");
276 is_tx_enabled = false;
277 }
278
279 std::array<cf_t, 1024> zero_buffer = {};
280 // Transmit zeros until the sample count reaches the timestamp.
281 while (sample_count < timestamp && state_fsm.is_running()) {
282 unsigned to_send = std::min(zero_buffer.size(), timestamp - sample_count);
283 span<cf_t> zeros = span<cf_t>(zero_buffer.begin(), to_send);
284 transmit_samples(zeros);
285 count += zeros.size();
286 }
287
288 if (count > 0) {
289 logger.debug("Aligned with {} zeros.", count);
290 }
291 return false;
