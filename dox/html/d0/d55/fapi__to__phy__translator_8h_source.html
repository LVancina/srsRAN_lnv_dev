88/// \brief FAPI-to-PHY message translator.
89///
90/// This class receives slot-based FAPI messages and translates them into PHY specific data types, which can then be
91/// processed by the upper PHY.
92///
93/// Translating and processing slot-based FAPI messages involves converting them to the suitable data types for the PHY
94/// layer, validation of the contents for each incoming message and selecting and reserving resources in the upper PHY
95/// to process the messages.
96///
97/// \note This class is thread safe and allows calling the \ref handle_new_slot method and the message handlers from
98/// different threads.
99/// \note The translator assumes that only one message of each type can be received within a slot. Receiving multiple
100/// messages of the same type in one slot results in undefined behavior.
101/// \note The translator is designed to work for a sector and subcarrier spacing. Supporting more than one sector and/or
102/// subcarrier spacing will require more instances of the translator.
103class fapi_to_phy_translator : public fapi::slot_message_gateway
104{
105 /// \brief Slot-based upper PHY controller.
106 ///
107 /// Takes care of the opening and closing procedures required by the upper PHY within each slot.
108 ///
109 /// \note This class follows the RAII (resource acquisition is initialization) programming principle.
110 /// \note The lifetime of any instantiation of this class is meant to be a single slot.
111 class slot_based_upper_phy_controller
112 {
113 slot_point slot;
114 std::reference_wrapper<downlink_processor> dl_processor;
115
116 public:
117 slot_based_upper_phy_controller();
118
119 slot_based_upper_phy_controller(downlink_processor_pool& dl_processor_pool,
120 resource_grid_pool& rg_pool,
121 slot_point slot_,
122 unsigned sector_id);
123
124 slot_based_upper_phy_controller(slot_based_upper_phy_controller&& other) = delete;
125
126 slot_based_upper_phy_controller& operator=(slot_based_upper_phy_controller&& other);
127
128 ~slot_based_upper_phy_controller();
129
130 /// Overloaded member of pointer operator.
131 downlink_processor* operator->() { return &dl_processor.get(); }
132 /// Overloaded member of pointer operator.
133 const downlink_processor* operator->() const { return &dl_processor.get(); }
134 };
135
136 /// Manages slot based controllers.
137 class slot_based_upper_phy_controller_manager
138 {
139 slot_based_upper_phy_controller dummy;
140 downlink_processor_pool& dl_processor_pool;
141 resource_grid_pool& rg_pool;
142 const unsigned sector_id;
143 std::vector<slot_based_upper_phy_controller> controllers;
144
145 /// Returns the controller for the given slot.
146 slot_based_upper_phy_controller& controller(slot_point slot)
147 {
148 return controllers[slot.system_slot() % controllers.size()];
149 }
150
151 /// Returns the controller for the given slot.
152 const slot_based_upper_phy_controller& controller(slot_point slot) const
153 {
154 return controllers[slot.system_slot() % controllers.size()];
155 }
156
157 public:
158 slot_based_upper_phy_controller_manager(downlink_processor_pool& dl_processor_pool_,
159 resource_grid_pool& rg_pool_,
160 unsigned sector_id_,
161 unsigned nof_slots_request_headroom);
162
163 /// Acquires and returns the controller for the given slot.
164 slot_based_upper_phy_controller& acquire_controller(slot_point slot);
165
166 /// \brief Releases the controller for the given slot.
167 ///
168 /// If the controller has already been released, this function does nothing.
169 void release_controller(slot_point slot);
170
171 /// Returns the controller for the given slot.
172 slot_based_upper_phy_controller& get_controller(slot_point slot);
173 };

175 /// \brief PDSCH PDU repository.
176 ///
177 /// Stores the PDSCH PDUs for later processing.
178 struct pdsch_pdu_repository {
179 slot_point slot;
180 static_vector<pdsch_processor::pdu_t, MAX_PDSCH_PDUS_PER_SLOT> pdus;
181
182 /// Returns true if the repository is empty, otherwise false.
183 bool empty() const { return pdus.empty(); }
184
185 /// Clears the repository.
186 void clear()
187 {
188 slot = slot_point();
189 pdus.clear();
190 }
191
192 /// Resets the repository by clearing the PDUs and setting the slot to the given one.
193 void reset(slot_point new_slot)
194 {
195 pdus.clear();
196 slot = new_slot;
197 }
198 };
199
200public:
201 fapi_to_phy_translator(const fapi_to_phy_translator_config& config,
202 fapi_to_phy_translator_dependencies&& dependencies);
203
204 // See interface for documentation.

212
213 // See interface for documentation.
214 void tx_data_request(const fapi::tx_data_request_message& msg) override;
215
216 /// \brief Handles a new slot.
217 ///
218 /// Handling a new slot consists of the following steps.
219 /// - Finish processing the PDUs from a past slot.
220 /// - Update the current slot value to the new one.
221 /// - Reset the contents of the uplink PDU repository.
222 ///
223 /// \param[in] slot Identifies the new slot.
224 /// \note This method is thread safe and may be called from different threads.
225 void handle_new_slot(slot_point slot);
226
227 /// Configures the FAPI slot-based, error-specific notifier to the given one.
228 void set_slot_error_message_notifier(fapi::slot_error_message_notifier& fapi_error_notifier)
229 {
230 error_notifier = std::ref(fapi_error_notifier);
231 }
232
233private:
234 /// Returns true if the given message arrived in time, otherwise returns false.
235 template <typename T>
236 bool is_message_in_time(const T& msg) const;
237
238 /// Returns this adaptor current slot.
239 slot_point get_current_slot() const
240 {
241 return slot_point(scs, current_slot_count_val.load(std::memory_order_acquire));
242 }
243
244 /// Updates this adaptor current slot to the given value.
245 void update_current_slot(slot_point slot)
246 {
247 // Update the atomic variable that holds the slot point.
248 current_slot_count_val.store(slot.system_slot(), std::memory_order_release);
249 }

251private:
252 /// Sector identifier.
253 const unsigned sector_id;
254 /// Request headroom size in slots.
255 const unsigned nof_slots_request_headroom;
256 /// Logger.
257 srslog::basic_logger& logger;
258 /// Downlink PDU validator.
259 const downlink_pdu_validator& dl_pdu_validator;
260 /// Uplink request processor.
261 uplink_request_processor& ul_request_processor;
262 /// Uplink resource grid pool.
263 resource_grid_pool& ul_rg_pool;
264 /// Uplink PDU validator.
265 const uplink_pdu_validator& ul_pdu_validator;
266 /// Uplink slot PDU repository.
267 uplink_slot_pdu_repository& ul_pdu_repository;
268 /// Current slot count value.
269 std::atomic<uint32_t> current_slot_count_val;
270 /// Slot controller manager.
271 slot_based_upper_phy_controller_manager slot_controller_mngr;
272 /// Precoding matrix repository.
273 std::unique_ptr<precoding_matrix_repository> pm_repo;
274 /// UCI Part2 correspondence repository.
275 std::unique_ptr<uci_part2_correspondence_repository> part2_repo;
276 /// Error indication notifier.
277 std::reference_wrapper<fapi::slot_error_message_notifier> error_notifier;
278 /// Subcarrier spacing as per TS38.211 Section 4.2.
279 const subcarrier_spacing scs;
280 /// Common subcarrier spacing as per TS38.331 Section 6.2.2.
281 const subcarrier_spacing scs_common;
282 /// PRACH configuration as per SCF-222 v4.0 section 3.3.2.4.
283 const fapi::prach_config prach_cfg;
284 /// Carrier configuration as per SCF-222 v4.0 section 3.3.2.4.
285 const fapi::carrier_config carrier_cfg;
286 /// PRACH receive ports.
287 const static_vector<uint8_t, MAX_PORTS> prach_ports;
288 /// PDSCH PDU repository.
289 pdsch_pdu_repository pdsch_repository;
290};
