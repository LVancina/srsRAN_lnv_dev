32/// \brief Port channel estimator that averages all OFDM symbols containing DM-RS.
33///
34/// This estimator considers the channel constant in time over the entire slot. As for the frequency domain, the channel
35/// coefficients corresponding to subcarriers containing DM-RS pilots are LSE-estimated, while the remaining
36/// coefficients are estimated by interpolating the previous ones.
37class port_channel_estimator_average_impl : public port_channel_estimator
38{
39public:
40 /// \brief Size of the internal inverse Discrete Fourier Transform.
41 ///
42 /// The inverse DFT is used to estimate the time alignment. A DFT size of 4096 points allows of a resolution of 16.3
43 /// and 8.1 nanoseconds with a subcarrier spacing of 15 kHz and 30 kHz, respectively.
44 static constexpr unsigned DFT_SIZE = 4096;
45 /// \brief Maximum SINR in decibels.
46 ///
47 /// The SINR is bounded above to avoid a zero noise variance.
48 static constexpr float MAX_SINR_DB = 100;
49
50 /// Constructor - Sets the internal interpolator and inverse DFT processor of size \c DFT_SIZE.
51 port_channel_estimator_average_impl(std::unique_ptr<interpolator> interp,
52 std::unique_ptr<dft_processor> idft_proc,
53 port_channel_estimator_fd_smoothing_strategy fd_smoothing_strategy_) :
54 fd_smoothing_strategy(fd_smoothing_strategy_), freq_interpolator(std::move(interp)), idft(std::move(idft_proc))
55 {
56 srsran_assert(freq_interpolator, "Invalid interpolator.");
57 srsran_assert(idft->get_direction() == dft_processor::direction::INVERSE,
58 "The port channel estimator requires an inverse (not direct) DFT processor.");
59 srsran_assert(idft->get_size() == DFT_SIZE,
60 "The port channel estimator requires an iDFT of size {}, provided {}.",
61 static_cast<unsigned>(DFT_SIZE),
62 idft->get_size());
63 }
64
65 // See interface for documentation.
66 void compute(channel_estimate& estimate,
67 const resource_grid_reader& grid,
68 unsigned port,
69 const dmrs_symbol_list& pilots,
70 const configuration& cfg) override;
71
72private:
73 /// Specializes \ref compute for one layer and one hop.
74 void compute_layer_hop(channel_estimate& estimate,
75 const resource_grid_reader& grid,
76 unsigned port,
77 const dmrs_symbol_list& pilots,
78 const configuration& cfg,
79 unsigned hop,
80 unsigned layer);
81
82 /// Frequency domain smoothing strategy.
83 port_channel_estimator_fd_smoothing_strategy fd_smoothing_strategy;
84
85 /// \brief Interpolator.
86 ///
87 /// When DM-RS pilots do not occupy all REs in an OFDM symbol, the interpolator is used to estimate the channel of the
88 /// REs without pilots.
89 std::unique_ptr<interpolator> freq_interpolator;
90
91 /// Inverse DFT processor of size \c DFT_SIZE.
92 std::unique_ptr<dft_processor> idft;
93
94 /// Buffer of received signal samples corresponding to pilots.
95 dmrs_symbol_list rx_pilots;
96
97 /// Buffer of frequency response coefficients.
98 std::array<cf_t, MAX_RB * NRE> freq_response;
99
100 /// Estimated RSRP value (single layer).
101 float rsrp = 0;
102
103 /// \brief Observed average DM-RS EPRE.
104 ///
105 /// \remark The EPRE is defined as the average received power (including noise) across all REs carrying DM-RS.
106 float epre = 0;
107
108 /// Estimated noise variance (single layer).
109 float noise_var = 0;
110
111 /// Estimated time alignment in seconds.
112 float time_alignment_s = 0;
113};
