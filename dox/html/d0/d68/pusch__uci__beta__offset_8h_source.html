27/// \brief Maps the integer \c beta_offset value for HARQ-ACK reporting into the corresponding float value.
28///
29/// The integer \c beta_offset value is passed by the \c PUSCH-Config, TS 38.331, and the mapping into the corresponding
30/// float value is determined as per Table 9.3-1, TS 38.213.
31/// \param[in] beta_uint_val is the integer value as per \c BetaOffsets, TS 38.331.
32/// \remark The caller must ensure \c beta_uint_val < 16, so as not to incur in reserved values, as per Table 9.3-1,
33/// TS 38.213.
34/// \return The corresponding float value as per Table 9.3-1, TS 38.213.
35float beta_harq_ack_to_float(unsigned beta_uint_val);
36
37/// \brief Maps the integer \c beta_offset value for CSI reporting into the corresponding float value.
38///
39/// The integer \c beta_offset value is passed by the \c PUSCH-Config, TS 38.331, and the mapping into the corresponding
40/// float value is determined as per Table 9.3-2, TS 38.213.
41/// \param[in] beta_uint_val is the integer value as per \c BetaOffsets, TS 38.331.
42/// \remark The caller must ensure \c beta_uint_val < 19, so as not to incur in reserved values, as per Table 9.3-2,
43/// TS 38.213.
44/// \return The corresponding float value as per Table 9.3-2, TS 38.213.
45float beta_csi_to_float(unsigned beta_uint_val);
