35/// \brief Conversion of CRBs to VRBs as per TS38.211, clause 7.3.1.6.
36/// \param crbs CRB interval to be converted to VRB interval.
37/// \param bwp_crb_start Start of the BWP CRBs.
38/// \param coreset_crb_start Start of the coreset CRBs.
39/// \param dci_fmt DCI DL format.
40/// \param ss_type type of Search Space.
41/// \return VRB interval.
42inline vrb_interval crb_to_vrb_dl_non_interleaved(crb_interval crbs,
43 unsigned bwp_crb_start,
44 unsigned coreset_crb_start,
45 dci_dl_format dci_fmt,
46 bool is_common_ss)
47{
48 if (dci_fmt == dci_dl_format::f1_0 and is_common_ss) {
49 return crb_to_vrb_f1_0_common_ss_non_interleaved(crbs, coreset_crb_start);
50 }
51 return vrb_interval{crbs.start() - bwp_crb_start, crbs.stop() - bwp_crb_start};
52}
53
54/// \brief Conversion of CRBs to VRBs as per TS38.211, clause 7.3.1.6.
55/// \param crbs CRB interval to be converted to VRB interval.
56/// \param bwp_crb_start Start of the BWP CRBs.
57/// \return VRB interval.
58inline vrb_interval crb_to_vrb_ul_non_interleaved(crb_interval crbs, unsigned bwp_crb_start)
59{
60 srsran_sanity_check(crbs.start() >= bwp_crb_start, "Invalid CRB start");
61 return vrb_interval{crbs.start() - bwp_crb_start, crbs.stop() - bwp_crb_start};
62}
63
64inline crb_interval vrb_to_crb_ul_non_interleaved(vrb_interval vrbs, unsigned bwp_crb_start)
65{
66 return crb_interval{vrbs.start() + bwp_crb_start, vrbs.stop() + bwp_crb_start};
67}
