32/// \brief This class manages the allocation of RAN PUCCH Resources to DU UEs. For instance, this class ensures that
33/// UEs do not collide in the usage of the PUCCH for SRs and CSI. For HARQ-ACKs, we rely on the MAC scheduler to ensure
34/// no collisions take place in the PUCCH.
35class du_pucch_resource_manager
36{
37public:
38 explicit du_pucch_resource_manager(span<const du_cell_config> cell_cfg_list_, unsigned max_pucch_grants_per_slot_);
39 du_pucch_resource_manager(const du_pucch_resource_manager&) = delete;
40 du_pucch_resource_manager(du_pucch_resource_manager&&) = delete;
41 du_pucch_resource_manager& operator=(const du_pucch_resource_manager&) = delete;
42 du_pucch_resource_manager& operator=(du_pucch_resource_manager&&) = delete;
43
44 /// \brief Allocate PUCCH resources for a given UE. The resources are stored in the UE's cell group config.
45 /// \return true if allocation was successful.
46 bool alloc_resources(cell_group_config& cell_grp_cfg);
47
48 /// \brief Deallocate PUCCH resources previously given to a UE. The resources are returned back to a pool.
49 void dealloc_resources(cell_group_config& cell_grp_cfg);
50
51private:
52 /// \brief Computes the DU index for PUCCH SR resource from the UE's PUCCH-Config \ref res_id index.
53 ///
54 /// Each cell has nof_cell_pucch_f1_res_sr PUCCH Format 1 resources that can be used for SR. Within the DU, these
55 /// resources are indexed with the values: {0, ..., nof_cell_pucch_f1_res_sr-1}. However, in the UE's PUCCH-Config,
56 /// the PUCCH F1 resources use different indices (see \ref res_id in \ref pucch_resource). The mapping between the DU
57 /// index and the UE's PUCCH-Config for SR PUCCH resources is defined in \ref srs_du::ue_pucch_config_builder.
58 unsigned pucch_res_idx_to_sr_du_res_idx(unsigned pucch_res_idx) const;
59
60 /// \brief Computes the DU index for PUCCH CSI resource from the UE's PUCCH-Config \ref res_id index.
61 ///
62 /// Each cell has nof_cell_pucch_f2_res_csi PUCCH Format 2 resources that can be used for CSI. Within the DU, these
63 /// resources are indexed with the values: {0, ..., nof_cell_pucch_f2_res_csi-1}. However, in the UE's PUCCH-Config,
64 /// the PUCCH F2 resources use different indices (see \ref res_id in \ref pucch_resource). The mapping between the DU
65 /// index and the UE's PUCCH-Config for CSI PUCCH resources is defined in \ref srs_du::ue_pucch_config_builder.
66 unsigned pucch_res_idx_to_csi_du_res_idx(unsigned pucch_res_idx) const;
67
68 std::vector<std::pair<unsigned, unsigned>>::const_iterator
69 find_optimal_csi_report_slot_offset(const std::vector<std::pair<unsigned, unsigned>>& available_csi_slot_offsets,
70 unsigned candidate_sr_offset,
71 const csi_meas_config& csi_meas_cfg);
72
73 /// Computes the CSI resource ID and offset, under the following constraints: (i) the PUCCH grants counter doesn't
74 /// exceed the max_pucch_grants_per_slot; (ii) the SR and CSI offsets will result in the PUCCH resource not exceeding
75 /// the maximum PUCCH F2 payload.
76 std::vector<std::pair<unsigned, unsigned>>::const_iterator
77 get_csi_resource_offset(const csi_meas_config& csi_meas_cfg,
78 unsigned candidate_sr_offset,
79 const std::vector<std::pair<unsigned, unsigned>>& free_csi_list);
80
81 /// Computes the SR and CSI PUCCH offsets and their repetitions within a given period, which is the Least Common
82 /// Multiple of SR and CSI periods. If SR and CSI results in having common offsets, this will be counted only once.
83 std::set<unsigned> compute_sr_csi_pucch_offsets(unsigned sr_offset, unsigned csi_offset = 0);
84
85 // Parameters for PUCCH configuration passed by the user.
86 const pucch_builder_params user_defined_pucch_cfg;
87 const std::vector<pucch_resource> default_pucch_res_list;
88 const pucch_config default_pucch_cfg;
89 const optional<csi_report_config> default_csi_report_cfg;
90 const unsigned max_pucch_grants_per_slot;
91 unsigned lcm_csi_sr_period;
92 unsigned sr_period_slots = 0;
93 unsigned csi_period_slots = 0;
94
95 struct cell_resource_context {
96 /// \brief Pool of PUCCH SR offsets currently available to be allocated to UEs. Each element is represented by a
97 /// pair (pucch_resource_id, slot_offset).
98 std::vector<std::pair<unsigned, unsigned>> sr_res_offset_free_list;
99 /// Pool of PUCCH CSI offsets currently available to be allocated to UEs.
100 std::vector<std::pair<unsigned, unsigned>> csi_res_offset_free_list;
101 /// UE index for randomization of resources.
102 unsigned ue_idx = 0;
103 std::vector<unsigned> pucch_grants_per_slot_cnt;
104 };
105
106 /// Resources for the different cells of the DU.
107 static_vector<cell_resource_context, MAX_NOF_DU_CELLS> cells;
108};
