31/// \brief Radio configuration validator for uhd based radios.
32///
33/// This validator validates that parameters values are generally valid, for example frequencies are valid numbers and
34/// positive or stream arguments follow a certain pattern. However, it does not validate whether the actual values are
35/// supported by the UHD device.
36///
37/// It is up to the user to select parameters within ranges. These can be consulted using the UHD application \e
38/// uhd_usrp_probe.
39class radio_config_uhd_config_validator : public radio_configuration::validator
40{
41public:
42 // See interface for documentation.
43 bool is_configuration_valid(const radio_configuration::radio& config) const override;
44};
