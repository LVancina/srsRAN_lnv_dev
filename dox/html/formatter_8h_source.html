77 template <typename... Ts>
78 void format_ctx(const srslog::context<Ts...>& ctx, detail::log_entry_metadata&& metadata, fmt::memory_buffer& buffer)
79 {
80 format_context_begin(metadata, ctx.name(), sizeof...(Ts), buffer);
81 iterate_tuple(ctx.contents(), 1, buffer, detail::make_index_sequence<sizeof...(Ts)>{});
82 format_context_end(metadata, ctx.name(), buffer);
83 }
84
85private:
87 template <typename... Ts, std::size_t... Is>
88 void
89 iterate_tuple(const std::tuple<Ts...>& t, unsigned level, fmt::memory_buffer& buffer, detail::index_sequence<Is...>)
90 {
91 (void)std::initializer_list<int>{(process_element(std::get<Is>(t), level, buffer), 0)...};
92 }
93
95 template <typename Name, typename... Ts>
96 void process_element(const metric_set<Name, Ts...>& ms, unsigned level, fmt::memory_buffer& buffer)
97 {
98 format_metric_set_begin(ms.name(), sizeof...(Ts), level, buffer);
99 iterate_tuple(ms.contents(), level + 1, buffer, detail::make_index_sequence<sizeof...(Ts)>{});
100 format_metric_set_end(ms.name(), level, buffer);
101 }
102
104 template <typename Name, typename T>
105 void process_element(const metric_list<Name, T>& list, unsigned level, fmt::memory_buffer& buffer)
106 {
107 format_list_begin(list.name(), list.size(), level, buffer);
108 for (const auto& elem : list) {
109 process_element(elem, level + 1, buffer);
110 }
111 format_list_end(list.name(), level, buffer);
112 }
113
115 template <typename Ty, typename Name, typename Units>
116 void process_element(const metric<Ty, Name, Units>& t, unsigned level, fmt::memory_buffer& buffer)
117 {
118 fmt::memory_buffer value;
119 metric_value_formatter<typename std::decay<decltype(t)>::type>{}.format(t.value, value);
120 value.push_back('\0');
121
122 format_metric(t.name(), value.data(), t.units(), t.kind(), level, buffer);
123 }
