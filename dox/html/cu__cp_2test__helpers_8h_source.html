193 void fill_pdu_session_setup_list(
194 slotted_id_vector<pdu_session_id_t, e1ap_pdu_session_resource_setup_modification_item>& e1ap_setup_list,
195 const std::list<unsigned>& outcome_setup_list)
196 {
197 for (const auto& psi : outcome_setup_list) {
198 // add only the most relevant items
199 e1ap_pdu_session_resource_setup_modification_item res_setup_item;
200 res_setup_item.pdu_session_id = uint_to_pdu_session_id(psi);
201
202 // add a single DRB with the same ID like the PDU session it belongs to
203 drb_id_t drb_id = uint_to_drb_id(psi); // Note: we use the PDU session ID here also as DRB ID
204 e1ap_drb_setup_item_ng_ran drb_item;
205 drb_item.drb_id = drb_id;
206
207 // add a QoS flow
208 e1ap_qos_flow_item qos_item;
209 qos_item.qos_flow_id = uint_to_qos_flow_id(psi); // Note: use the PSI again as QoS flow ID
210 drb_item.flow_setup_list.emplace(qos_item.qos_flow_id, qos_item);
211
212 // add one UP transport item
213 e1ap_up_params_item up_item;
214 up_item.cell_group_id = 0;
215 up_item.up_tnl_info = {transport_layer_address{"127.0.0.1"}, int_to_gtpu_teid(0x1)};
216 drb_item.ul_up_transport_params.push_back(up_item);
217 res_setup_item.drb_setup_list_ng_ran.emplace(drb_id, drb_item);
218
219 e1ap_setup_list.emplace(res_setup_item.pdu_session_id, res_setup_item);
220 }
221 }

241 void fill_pdu_session_modified_list(
242 slotted_id_vector<pdu_session_id_t, e1ap_pdu_session_resource_modified_item>& e1ap_modified_list,
243 const std::list<pdu_session_modified_outcome_t>& outcome_modified_list)
244 {
245 for (const auto& modified_item : outcome_modified_list) {
246 // add only the most relevant items
247 e1ap_pdu_session_resource_modified_item res_mod_item;
248 res_mod_item.pdu_session_id = modified_item.psi;
249
250 for (const auto& drb_id : modified_item.drb_added) {
251 // add a single DRB with the same ID like the PDU session it belongs to
252 e1ap_drb_setup_item_ng_ran drb_item;
253 drb_item.drb_id = drb_id;
254
255 // add a QoS flow
256 e1ap_qos_flow_item qos_item;
257 qos_item.qos_flow_id = uint_to_qos_flow_id(drb_id_to_uint(drb_id)); // QoS flow has same ID like DRB
258 drb_item.flow_setup_list.emplace(qos_item.qos_flow_id, qos_item);
259
260 // add one UP transport item
261 e1ap_up_params_item up_item;
262 drb_item.ul_up_transport_params.push_back(up_item);
263 res_mod_item.drb_setup_list_ng_ran.emplace(drb_id, drb_item);
264 }
265
266 e1ap_modified_list.emplace(res_mod_item.pdu_session_id, res_mod_item);
267 }

279 async_task<e1ap_bearer_context_setup_response>
280 on_bearer_context_setup_request(const e1ap_bearer_context_setup_request& msg) override
281 {
282 logger.info("Received a new bearer context setup request");
283
284 first_e1ap_request = msg; // store msg to verify content in TC
285
286 return launch_async([res = e1ap_bearer_context_setup_response{}, msg, this](
287 coro_context<async_task<e1ap_bearer_context_setup_response>>& ctx) mutable {
288 CORO_BEGIN(ctx);
289
290 if (first_e1ap_response.has_value()) {
291 const auto& response = first_e1ap_response.value();
292 fill_bearer_context_response(res, response);
293
294 // Invalidate E1 response so it's not consumed again.
295 first_e1ap_response.reset();
296 }
297
298 CORO_RETURN(res);
299 });
300 }
301
302 async_task<e1ap_bearer_context_modification_response>
303 on_bearer_context_modification_request(const e1ap_bearer_context_modification_request& msg) override
304 {
305 logger.info("Received a new bearer context modification request");
306
307 // store msg to be verify content in TC
308 if (first_e1ap_request.has_value()) {
309 second_e1ap_request = msg;
310 } else {
311 first_e1ap_request = msg;
312 }
313
314 return launch_async([res = e1ap_bearer_context_modification_response{},
315 this](coro_context<async_task<e1ap_bearer_context_modification_response>>& ctx) mutable {
316 CORO_BEGIN(ctx);
317
318 if (first_e1ap_response.has_value()) {
319 // first E1AP message is already a bearer modification
320 const auto& response = first_e1ap_response.value();
321 fill_bearer_context_response(res, response);
322 first_e1ap_response.reset(); // Make sure it's not consumed again.
323 } else if (second_e1ap_response.has_value()) {
324 // second E1AP message is a bearer modification
325 const auto& response = second_e1ap_response.value();
326 fill_bearer_context_response(res, response);
327 second_e1ap_response.reset(); // Make sure it's not consumed again.
328 }
329 CORO_RETURN(res);
330 });
331 }
