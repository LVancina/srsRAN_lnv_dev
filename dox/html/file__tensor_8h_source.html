36template <unsigned NDIMS, typename T, typename Index_type = unsigned>
37class file_tensor
38{
39private:
40 const char* file_name;
41 std::array<unsigned, NDIMS> dimension_sizes;
42 openmode op;
43
44public:
53 file_tensor(const char* filename, const std::array<unsigned, NDIMS> dim_sizes, openmode op_ = openmode::read_only) :
54 file_name(filename), dimension_sizes(dim_sizes), op(op_)
55 {
56 }
57
59 const char* get_file_name() const { return file_name; }
60
66 dynamic_tensor<NDIMS, T, Index_type> read() const
67 {
68 report_fatal_error_if_not(op != openmode::write_only, "The file_vector was created with write-only permissions.");
69 std::ifstream binary_file(file_name, std::ios::in | std::ios::binary);
70
71 report_fatal_error_if_not(binary_file.is_open(), "Error opening file '{}'. {}.", file_name, strerror(errno));
72
73 dynamic_tensor<NDIMS, T, Index_type> read_data(dimension_sizes);
74 span<T> flat_tensor_view(read_data.get_data());
75
76 T read_value;
77 while (binary_file.read(reinterpret_cast<char*>(&read_value), sizeof(T))) {
78 flat_tensor_view.front() = read_value;
79 flat_tensor_view = flat_tensor_view.last(flat_tensor_view.size() - 1);
80 }
81
82 report_fatal_error_if_not(flat_tensor_view.size() == 0, "Not all tensor elements have been filled.");
83
84 return read_data;
85 }
86};
