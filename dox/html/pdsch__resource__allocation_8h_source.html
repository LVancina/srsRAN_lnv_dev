41inline crb_interval get_ra_crb_limits(dci_dl_format dci_fmt,
42 const bwp_downlink_common& init_dl_bwp,
43 const bwp_downlink_common& active_dl_bwp,
44 const search_space_configuration& ss_cfg,
45 const coreset_configuration& cs_cfg)
46{
47 crb_interval crbs = active_dl_bwp.generic_params.crbs;
48
49 if (dci_fmt == dci_dl_format::f1_0 and ss_cfg.is_common_search_space()) {
50 // See TS 38.211, 7.3.1.6 Mapping from virtual to physical resource blocks and TS38.214, 5.1.2.2. Resource
51 // Allocation in frequency domain.
52 crbs = {cs_cfg.get_coreset_start_crb(), crbs.stop()};
53
54 // See TS 38.214, 5.1.2.2.2, Downlink resource allocation type 1.
55 if (init_dl_bwp.pdcch_common.coreset0.has_value()) {
56 crbs.resize(std::min(crbs.length(), init_dl_bwp.pdcch_common.coreset0->coreset0_crbs().length()));
57 } else {
58 crbs.resize(std::min(crbs.length(), init_dl_bwp.generic_params.crbs.length()));
59 }
60 }
61 return crbs;
62}

73inline crb_interval get_ra_crb_limits_common(const bwp_downlink_common& init_dl_bwp, search_space_id ss_id)
74{
75 const search_space_configuration& ss_cfg = init_dl_bwp.pdcch_common.search_spaces.at(ss_id);
76 const coreset_configuration& cs_cfg = ss_cfg.get_coreset_id() == to_coreset_id(0)
77 ? init_dl_bwp.pdcch_common.coreset0.value()
78 : init_dl_bwp.pdcch_common.common_coreset.value();
79 srsran_assert(
80 ss_cfg.is_common_search_space() and
81 variant_get<search_space_configuration::common_dci_format>(ss_cfg.get_monitored_dci_formats()).f0_0_and_f1_0,
82 "Invalid SearchSpace type");
83
84 return get_ra_crb_limits(dci_dl_format::f1_0, init_dl_bwp, init_dl_bwp, ss_cfg, cs_cfg);
85}
