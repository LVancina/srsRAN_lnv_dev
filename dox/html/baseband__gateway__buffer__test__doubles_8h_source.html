34class baseband_gateway_buffer_read_only : public baseband_gateway_buffer_reader
35{
36private:
37 unsigned nof_channels;
38 unsigned nof_samples;
39 std::vector<cf_t> data;
40
41public:
42 // See interface for documentation.
43 unsigned get_nof_channels() const override { return nof_channels; }
44
45 // See interface for documentation.
46 unsigned get_nof_samples() const override { return nof_samples; }
47
48 // See interface for documentation.
49 span<const cf_t> get_channel_buffer(unsigned channel_idx) const override
50 {
51 report_fatal_error_if_not(!data.empty(), "Data is empty. Was the buffer moved?");
52 report_fatal_error_if_not(
53 channel_idx < nof_channels, "Channel index ({}) is out-of-range ({}).", channel_idx, nof_channels);
54 unsigned offset = nof_samples * channel_idx;
55 return span<const cf_t>(data).subspan(offset, nof_samples);
56 }
57
59 baseband_gateway_buffer_read_only() : nof_channels(0), nof_samples(0)
60 {
61 // Do nothing.
62 }
63
65 baseband_gateway_buffer_read_only(baseband_gateway_buffer_read_only&& other) noexcept :
66 nof_channels(other.nof_channels), nof_samples(other.nof_samples), data(std::move(other.data))
67 {
68 // Reset the other number of channels and samples to 0 to be consistent with other.data.size().
69 other.nof_channels = 0;
70 other.nof_samples = 0;
71 }
72
74 baseband_gateway_buffer_read_only(const baseband_gateway_buffer_read_only& other) noexcept :
75 nof_channels(other.nof_channels), nof_samples(other.nof_samples), data(std::move(other.data))
76 {
77 // Do nothing.
78 }
79
81 baseband_gateway_buffer_read_only(const baseband_gateway_buffer_reader& other) noexcept :
82 nof_channels(other.get_nof_channels()), nof_samples(other.get_nof_samples()), data(nof_channels * nof_samples)
83 {
84 for (unsigned channel = 0; channel != nof_channels; ++channel) {
85 span<cf_t> buffer = span<cf_t>(data).subspan(nof_samples * channel, nof_samples);
86 srsvec::copy(buffer, other[channel]);
87 }
88 }
89
91 baseband_gateway_buffer_read_only(baseband_gateway_buffer_writer& other) noexcept :
92 nof_channels(other.get_nof_channels()), nof_samples(other.get_nof_samples())
93 {
94 data.clear();
95 data.reserve(nof_channels * nof_samples);
96 for (unsigned channel = 0; channel != nof_channels; ++channel) {
97 span<const cf_t> source = other[channel];
98 data.insert(data.end(), source.begin(), source.end());
99 }
100 }
101
103 baseband_gateway_buffer_read_only operator=(const baseband_gateway_buffer_read_only& other)
104 {
105 nof_channels = other.nof_channels;
106 nof_samples = other.nof_samples;
107 data = other.data;
108 return *this;
109 }
110};
