32class manual_task_worker : public task_executor
33{
34public:
35 manual_task_worker(size_t q_size, bool blocking_mode_ = true) :
36 t_id(std::this_thread::get_id()), pending_tasks(q_size), blocking_mode(blocking_mode_)
37 {
38 }
39
40 std::thread::id get_thread_id() const { return t_id; }
41
42 bool execute(unique_task task) override
43 {
44 if (std::this_thread::get_id() == t_id) {
45 task();
46 return true;
47 }
48 return defer(std::move(task));
49 }
50
51 bool defer(unique_task task) override
52 {
53 if (blocking_mode) {
54 pending_tasks.push_blocking(std::move(task));
55 return true;
56 }
57 return pending_tasks.try_push(std::move(task)).has_value();
58 }
59
60 size_t max_pending_tasks() const { return pending_tasks.max_size(); }
61
62 bool has_pending_tasks() const { return not pending_tasks.empty(); }
63
64 bool is_stopped() const { return pending_tasks.is_stopped(); }
65
66 void stop()
67 {
68 if (not is_stopped()) {
69 pending_tasks.stop();
70 }
71 }
72
73 void request_stop()
74 {
75 defer([this]() { stop(); });
76 }
77
79 bool run_pending_tasks()
80 {
81 assert_thread_id();
82 bool ret = false, success = false;
83 do {
84 unique_task t;
85 success = pending_tasks.try_pop(t);
86 if (success) {
87 t();
88 ret = true;
89 }
90 } while (success);
91 return ret;
92 }
93
95 bool try_run_next()
96 {
97 assert_thread_id();
98 unique_task t;
99 bool success = pending_tasks.try_pop(t);
100 if (not success) {
101 return false;
102 }
103 t();
104 return true;
105 }
106
108 bool run_next_blocking()
109 {
110 assert_thread_id();
111 bool success = false;
112 unique_task t = pending_tasks.pop_blocking(&success);
113 if (not success) {
114 return false;
115 }
116 t();
117 return true;
118 }
119
120private:
121 bool has_thread_id() const { return t_id != std::thread::id{}; }
122
123 void assert_thread_id()
124 {
125 srsran_assert(t_id == std::this_thread::get_id(), "run() caller thread should not change.");
126 }
127
128 std::thread::id t_id;
129 blocking_queue<unique_task> pending_tasks;
130 bool blocking_mode;
131};
132
133} // namespace srsran
