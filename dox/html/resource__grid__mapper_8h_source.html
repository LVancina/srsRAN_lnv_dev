62 class symbol_buffer_adapter : public symbol_buffer
63 {
64 public:
66 symbol_buffer_adapter(span<const ci8_t> symbols_) : symbols(symbols_) {}
67
69 ~symbol_buffer_adapter() { srsran_assert(symbols.empty(), "{} symbols are still in the buffer.", symbols.size()); }
70
71 // See interface for documentation.
72 unsigned get_max_block_size() const override { return symbols.size(); }
73
74 // See interface for documentation.
75 bool empty() const override { return symbols.empty(); }
76
77 // See interface for documentation.
78 span<const ci8_t> pop_symbols(unsigned block_size) override
79 {
80 // Make sure the block size does not exceed the number of symbols.
81 srsran_assert(symbols.size() >= block_size,
82 "The block size (i.e., {}) exceeds the number of available symbols (i.e., {}).",
83 block_size,
84 symbols.size());
85
86 // Select view of the symbols to return.
87 span<const ci8_t> ret = symbols.first(block_size);
88
89 // Cut symbols.
90 symbols = symbols.last(symbols.size() - block_size);
91
92 return ret;
93 }
94
95 private:
96 span<const ci8_t> symbols;
97 };
