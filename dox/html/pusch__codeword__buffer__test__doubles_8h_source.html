30class pusch_codeword_buffer_spy : public pusch_codeword_buffer
31{
32public:
33 pusch_codeword_buffer_spy() = default;
34
35 pusch_codeword_buffer_spy(unsigned buffer_size) : data(buffer_size), scrambling_seq(buffer_size) {}
36
37 span<log_likelihood_ratio> get_next_block_view(unsigned block_size) override
38 {
39 srsran_assert(!completed, "Wrong state.");
40 block_size = std::min(block_size, static_cast<unsigned>(data.size()) - count);
41 return span<log_likelihood_ratio>(data).subspan(count, block_size);
42 }
43
44 void on_new_block(span<const log_likelihood_ratio> new_data, const bit_buffer& new_sequence) override
45 {
46 srsran_assert(!completed, "Wrong state.");
47 srsran_assert(new_sequence.size() == new_data.size(), "Sizes must be equal.");
48 unsigned block_size = new_data.size();
49
50 // Append soft bits.
51 span<log_likelihood_ratio> dst_data = span<log_likelihood_ratio>(data).subspan(count, block_size);
52 srsvec::copy(dst_data, new_data);
53
54 // Append scrambling sequence.
55 srsvec::copy_offset(scrambling_seq, count, new_sequence, 0, new_sequence.size());
56
57 count += block_size;
58 }
59
60 void on_end_codeword() override
61 {
62 srsran_assert(!completed, "Wrong state.");
63 completed = true;
64 }
65
66 span<const log_likelihood_ratio> get_data() const
67 {
68 srsran_assert(completed, "Wrong state.");
69 return span<const log_likelihood_ratio>(data).first(count);
70 }
71
72 const bit_buffer& get_scrambling_seq() const
73 {
74 srsran_assert(completed, "Wrong state.");
75 return scrambling_seq;
76 }
77
78private:
79 bool completed = false;
80 unsigned count = 0;
81 std::vector<log_likelihood_ratio> data;
82 dynamic_bit_buffer scrambling_seq;
83};

85} // namespace srsran
