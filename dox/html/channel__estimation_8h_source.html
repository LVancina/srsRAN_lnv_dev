82 explicit channel_estimate(const channel_estimate_dimensions& dims) :
83 max_dims(dims),
84 nof_subcarriers(dims.nof_prb * NRE),
85 nof_symbols(dims.nof_symbols),
86 nof_rx_ports(dims.nof_rx_ports),
87 nof_tx_layers(dims.nof_tx_layers),
88 ce({nof_subcarriers, nof_symbols, nof_rx_ports, nof_tx_layers})
89 {
90 srsran_assert(dims.nof_prb <= MAX_RB, "Requested {} RBs, but at most {} are allowed.", dims.nof_prb, MAX_RB);
91 srsran_assert(dims.nof_symbols <= MAX_NSYMB_PER_SLOT,
92 "Requested {} OFDM symbols, but at most {} are allowed.",
93 dims.nof_symbols,
94 MAX_NSYMB_PER_SLOT);
95 srsran_assert(dims.nof_rx_ports <= MAX_RX_PORTS,
96 "Requested {} receive ports, but at most {} are supported.",
97 dims.nof_rx_ports,
98 static_cast<unsigned>(MAX_RX_PORTS));
99 srsran_assert(dims.nof_tx_layers <= MAX_TX_LAYERS,
100 "Requested {} transmission layers, but at most {} are supported.",
101 dims.nof_tx_layers,
102 static_cast<unsigned>(MAX_TX_LAYERS));
103
104 unsigned nof_paths = dims.nof_tx_layers * dims.nof_rx_ports;
105
106 // Reserve memory for channel estimates and initialize with 1.0.
107 ce.resize({dims.nof_prb * NRE, dims.nof_symbols, dims.nof_rx_ports, dims.nof_tx_layers});
108
109 // Set all reserved data to one.
110 span<cf_t> data = ce.get_view<4>({});
111 std::fill(data.begin(), data.end(), 1.0F);
112
113 // Reserve memory for the rest of channel statistics.
114 noise_variance.reserve(MAX_TX_RX_PATHS);
115 noise_variance.resize(nof_paths);
116 epre.reserve(MAX_TX_RX_PATHS);
117 epre.resize(nof_paths);
118 rsrp.reserve(MAX_TX_RX_PATHS);
119 rsrp.resize(nof_paths);
120 snr.reserve(MAX_TX_RX_PATHS);
121 snr.resize(nof_paths);
122 time_alignment.reserve(MAX_TX_RX_PATHS);
123 time_alignment.resize(nof_paths);
124 }

169 float get_layer_average_snr(unsigned tx_layer = 0) const
170 {
171 float noise_var_all_ports = 0.0F;
172 float rsrp_all_ports = 0.0F;
173
174 // Add the noise and signal power contributions of all Rx ports.
175 for (unsigned i_rx_port = 0; i_rx_port != nof_rx_ports; ++i_rx_port) {
176 noise_var_all_ports += noise_variance[path_to_index(i_rx_port, tx_layer)];
177 rsrp_all_ports += rsrp[path_to_index(i_rx_port, tx_layer)];
178 }
179
180 if (std::isnormal(noise_var_all_ports)) {
181 return rsrp_all_ports / noise_var_all_ports;
182 }
183
184 // If noise variance is 0, report and SNR of 60 dB.
185 return 1e6;
186 }

235 void get_channel_state_information(channel_state_information& csi) const
236 {
237 // EPRE, RSRP and time alignment are reported as a linear average of the results for all Rx ports.
238 float epre_lin = 0.0F;
239 float rsrp_lin = 0.0F;
240 unsigned best_rx_port = 0;
241 float best_path_snr = 0.0F;
242 for (unsigned i_rx_port = 0; i_rx_port != nof_rx_ports; ++i_rx_port) {
243 // Accumulate EPRE and RSRP values.
244 epre_lin += get_epre(i_rx_port, 0);
245 rsrp_lin += get_rsrp(i_rx_port, 0);
246
247 // Determine the Rx port with better SNR.
248 float port_snr = get_snr(i_rx_port, 0);
249 if (port_snr > best_path_snr) {
250 best_path_snr = port_snr;
251 best_rx_port = i_rx_port;
252 }
253 }
254
255 epre_lin /= static_cast<float>(nof_rx_ports);
256 rsrp_lin /= static_cast<float>(nof_rx_ports);
257
258 csi.set_epre(convert_power_to_dB(epre_lin));
259 csi.set_rsrp(convert_power_to_dB(rsrp_lin));
260
261 // Use the time alignment of the channel path with better SNR.
262 csi.set_time_alignment(get_time_alignment(best_rx_port, 0));
263
264 // SINR is reported by averaging the signal and noise power contributions of all Rx ports.
265 csi.set_sinr_dB(channel_state_information::sinr_type::channel_estimator,
266 convert_power_to_dB(get_layer_average_snr(0)));
267 }

311 void resize(const channel_estimate_dimensions& dims)
312 {
313 nof_subcarriers = dims.nof_prb * NRE;
314 nof_symbols = dims.nof_symbols;
315 nof_rx_ports = dims.nof_rx_ports;
316 nof_tx_layers = dims.nof_tx_layers;
317
318 unsigned nof_paths = nof_rx_ports * nof_tx_layers;
319 srsran_assert(nof_paths <= MAX_TX_RX_PATHS,
320 "Total requested paths ({}) exceed maximum available ({}).",
321 nof_paths,
322 static_cast<unsigned>(MAX_TX_RX_PATHS));
323 noise_variance.resize(nof_paths);
324 epre.resize(nof_paths);
325 rsrp.resize(nof_paths);
326 snr.resize(nof_paths);
327
328 unsigned nof_res = nof_paths * nof_subcarriers * nof_symbols;
329 srsran_assert(nof_res <= MAX_BUFFER_SIZE,
330 "Total requested REs ({}) exceed maximum available ({}).",
331 nof_res,
332 static_cast<unsigned>(MAX_BUFFER_SIZE));
333 ce.resize({NRE * dims.nof_prb, dims.nof_symbols, dims.nof_rx_ports, dims.nof_tx_layers});
334 }

395 unsigned path_to_index(unsigned rx_port, unsigned tx_layer = 0) const
396 {
397 srsran_assert(
398 rx_port < nof_rx_ports, "Requested Rx antenna port {} is out of bound (max. {})", rx_port, nof_rx_ports);
399 srsran_assert(
400 tx_layer < nof_tx_layers, "Requested Rx antenna port {} is out of bound (max. {})", rx_port, nof_tx_layers);
401 return rx_port + nof_rx_ports * tx_layer;
402 }
403
405 unsigned coords_to_index(unsigned subcarrier, unsigned symbol, unsigned rx_port = 0, unsigned tx_layer = 0) const
406 {
407 srsran_assert(
408 subcarrier < nof_subcarriers, "Requested subcarrier {} is out of bound (max. {})", subcarrier, nof_subcarriers);
409 srsran_assert(symbol < nof_symbols, "Requested OFDM symbol {} is out of bound (max. {})", symbol, nof_symbols);
410 srsran_assert(
411 rx_port < nof_rx_ports, "Requested Rx antenna port {} is out of bound (max. {})", rx_port, nof_rx_ports);
412 srsran_assert(tx_layer < nof_tx_layers, "Requested Tx layer {} is out of bound (max. {})", tx_layer, nof_tx_layers);
413 return subcarrier + nof_subcarriers * (symbol + nof_symbols * (rx_port + nof_rx_ports * tx_layer));
414 }
