29class ofdm_prach_demodulator_spy : public ofdm_prach_demodulator
30{
31public:
32 struct demodulate_entry {
33 prach_buffer* buffer;
34 std::vector<cf_t> input;
35 configuration config;
36 };
37
38 void demodulate(prach_buffer& buffer, span<const cf_t> input, const configuration& config) override
39 {
40 demodulate_entries.emplace_back();
41 demodulate_entry& entry = demodulate_entries.back();
42 entry.buffer = &buffer;
43 entry.input.resize(input.size());
44 std::copy(input.begin(), input.end(), entry.input.begin());
45 entry.config = config;
46 }
47
48 const std::vector<demodulate_entry> get_demodulate_entries() const { return demodulate_entries; }
49
50private:
51 std::vector<demodulate_entry> demodulate_entries;
52};

54class ofdm_prach_demodulator_factory_spy : public ofdm_prach_demodulator_factory
55{
56public:
57 std::unique_ptr<ofdm_prach_demodulator> create() override
58 {
59 // Create spy.
60 std::unique_ptr<ofdm_prach_demodulator_spy> demodulator_spy = std::make_unique<ofdm_prach_demodulator_spy>();
61
62 // Save spy reference.
63 entries.push_back(demodulator_spy.get());
64
65 // Return the smart pointer.
66 return demodulator_spy;
67 }
68
69 std::vector<ofdm_prach_demodulator_spy*>& get_entries() { return entries; }
70
71 unsigned get_nof_total_demodulate_entries() const
72 {
73 unsigned count = 0;
74
75 for (const ofdm_prach_demodulator_spy* spy : entries) {
76 count += spy->get_demodulate_entries().size();
77 }
78
79 return count;
80 }
81
82 std::vector<ofdm_prach_demodulator_spy::demodulate_entry> get_total_demodulate_entries() const
83 {
84 std::vector<ofdm_prach_demodulator_spy::demodulate_entry> result;
85 result.reserve(get_nof_total_demodulate_entries());
86
87 for (const ofdm_prach_demodulator_spy* spy : entries) {
88 const std::vector<ofdm_prach_demodulator_spy::demodulate_entry>& demod_entries = spy->get_demodulate_entries();
89 result.insert(result.end(), demod_entries.begin(), demod_entries.end());
90 }
91
92 return result;
93 }
94
95private:
96 std::vector<ofdm_prach_demodulator_spy*> entries;
97};
