32class lcid_dl_sch_t
33{
34 using underlying_type = std::underlying_type_t<lcid_t>;
35
36public:
38 enum options : underlying_type {
39 CCCH = 0b000000,
40
42 LCID1 = 1,
43 // ...
44 LCID32 = 32,
45
47 MIN_RESERVED = 33,
48 MAX_RESERVED = 51,
49
50 RECOMMENDED_BIT_RATE = 0b101111,
51
52 // TODO: Add remaining.
53
54 SCELL_ACTIV_4_OCTET = 0b111001,
55 SCELL_ACTIV_1_OCTET = 0b111010,
56
57 LONG_DRX_CMD = 0b111011,
58 DRX_CMD = 0b111100,
59 TA_CMD = 0b111101,
60 UE_CON_RES_ID = 0b111110,
61 PADDING = 0b111111
62 };
63
64 lcid_dl_sch_t(underlying_type lcid_ = PADDING) : lcid_val(lcid_) { srsran_assert(lcid_ <= PADDING, "Invalid LCID"); }
65 lcid_dl_sch_t& operator=(underlying_type lcid)
66 {
67 srsran_assert(lcid <= PADDING, "Invalid LCID");
68 lcid_val = lcid;
69 return *this;
70 }
71
73 explicit operator underlying_type() const { return lcid_val; }
74
76 underlying_type value() const { return lcid_val; }
77
79 bool is_ce() const { return lcid_val <= PADDING and lcid_val >= RECOMMENDED_BIT_RATE; }
80
82 bool is_sdu() const { return lcid_val <= LCID32 and lcid_val >= CCCH; }
83
84 lcid_t to_lcid() const
85 {
86 srsran_assert(is_sdu(), "Invalid to_lcid() access");
87 return (lcid_t)lcid_val;
88 }
89
91 bool is_valid() const { return lcid_val <= PADDING and (lcid_val < MIN_RESERVED or lcid_val > MAX_RESERVED); }
92
93 bool is_var_len_ce() const { return false; }
94
95 uint32_t sizeof_ce() const
96 {
97 // Values taken from TS38.321, Section 6.1.3.
98 switch (lcid_val) {
99 case SCELL_ACTIV_4_OCTET:
100 return 4;
101 case SCELL_ACTIV_1_OCTET:
102 return 1;
103 case LONG_DRX_CMD:
104 case DRX_CMD:
105 return 0;
106 case TA_CMD:
107 return 1;
108 case UE_CON_RES_ID:
109 return 6;
110 case PADDING:
111 return 0;
112 default:
113 break;
114 }
115 return 0;
116 }
117
118 bool operator==(lcid_dl_sch_t other) const { return lcid_val == other.lcid_val; }
119 bool operator!=(lcid_dl_sch_t other) const { return lcid_val != other.lcid_val; }
120
121private:
122 underlying_type lcid_val;
123};
