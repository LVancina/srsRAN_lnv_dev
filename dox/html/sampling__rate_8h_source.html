36class sampling_rate
37{
38public:
40 using value_type = uint64_t;
41
43 sampling_rate() = default;
44
50 template <typename T>
51 constexpr bool is_valid(T scs_Hz) const
52 {
53 static_assert(std::is_convertible<T, value_type>::value, "Invalid data type.");
54 if (value == 0 || scs_Hz == 0) {
55 return false;
56 }
57 return value % static_cast<value_type>(scs_Hz) == 0;
58 }
59
64 constexpr bool is_valid(subcarrier_spacing scs) const { return is_valid(scs_to_khz(scs) * 1000); }
65
73 template <typename T>
74 constexpr unsigned get_dft_size(T scs_Hz) const
75 {
76 srsran_assert(is_valid(scs_Hz), "Subcarrier spacing must be valid.");
77 return value / static_cast<value_type>(scs_Hz);
78 }
79
86 constexpr unsigned get_dft_size(subcarrier_spacing scs) const { return get_dft_size(scs_to_khz(scs) * 1000); }
87
91 template <typename T = unsigned>
92 constexpr T to_Hz() const
93 {
94 static_assert(std::is_convertible<value_type, T>::value, "Invalid data type.");
95 return static_cast<T>(value);
96 }
97
101 template <typename T = unsigned>
102 constexpr T to_kHz() const
103 {
104 static_assert(std::is_convertible<value_type, T>::value, "Invalid data type.");
105 return static_cast<T>(value / 1000);
106 }
107
109 constexpr double to_MHz() const { return static_cast<double>(value) / 1e6; }
110
119 template <typename T>
120 static sampling_rate from_Hz(T value_Hz)
121 {
122 static_assert(std::is_convertible<T, value_type>::value, "Invalid data type.");
123 return sampling_rate(static_cast<value_type>(value_Hz));
124 }
125
134 template <typename T>
135 static sampling_rate from_kHz(T value_kHz)
136 {
137 return from_Hz(value_kHz * 1e3);
138 }
139
148 template <typename T>
149 static sampling_rate from_MHz(T value_MHz)
150 {
151 return from_Hz(value_MHz * 1e6);
152 }
153
155 constexpr bool operator==(sampling_rate other) const { return value == other.value; }
156
158 constexpr bool operator!=(sampling_rate other) const { return value != other.value; }
159
160private:
162 constexpr explicit sampling_rate(value_type value_) : value(value_)
163 {
164 srsran_assert(is_valid(15000), "The sampling rate must be multiple of 15kHz.");
165 }
166
168 value_type value = 0;
169};
