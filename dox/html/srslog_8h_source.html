102template <typename T, typename... Args>
103inline T& fetch_logger(const std::string& id, Args&&... args)
104{
105 static_assert(detail::is_logger<T>::value, "T should be a logger type");
106 if (auto* logger = find_logger<T>(id)) {
107 return *logger;
108 }
109 auto logger = detail::make_any<T>(id, std::forward<Args>(args)...);
110 detail::any* p = detail::fetch_logger(id, std::move(logger));
111
112 return *detail::any_cast<T>(p);
113}
