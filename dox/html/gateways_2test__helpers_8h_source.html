42byte_buffer make_tx_byte_buffer(uint32_t length)
43{
44 byte_buffer pdu{};
45 for (uint32_t i = 0; i < length; ++i) {
46 if (not pdu.append((uint8_t)i)) {
47 pdu.clear();
48 break;
49 }
50 }
51 return pdu;
52}

69void fill_sctp_hints(struct addrinfo& hints)
70{
71 // support ipv4, ipv6 and hostnames
72 hints.ai_family = AF_UNSPEC;

74 hints.ai_flags = 0;
