43class rx_payload_buffer_pool
44{
46 static constexpr size_t nof_slots = 40U;
48 static constexpr units::bits max_buffer_size = units::bits(MAX_RB * 156 * 8);
50 static constexpr unsigned min_block_size = 64;
51
52public:
54 span<uint8_t> acquire_payload_buffer(units::bytes size)
55 {
56 // Convert the maximum buffer size from bits to bytes for comparison and allocation.
57 static constexpr units::bytes max_buffer_size_bytes = max_buffer_size.truncate_to_bytes();
58
59 srsran_assert(size <= max_buffer_size_bytes, "Buffer size (i.e., {}) exceeds maximum {}.", size, pool.size());
60
61 // Round the number of consumed bytes to the next block.
62 size_t count = divide_ceil(size.value(), min_block_size) * min_block_size;
63
64 // Reset the available bytes if the pool is exhausted.
65 if (available.size() < count) {
66 available = pool;
67 }
68
69 // Select the first bytes as the payload for the transmission.
70 span<uint8_t> payload = available.first(size.value());
71
72 // Advance available bytes.
73 available = available.last(available.size() - count);
74 return payload;
75 }
76
77private:
79 std::array<uint8_t, max_buffer_size.truncate_to_bytes().value() * nof_slots> pool;
81 span<uint8_t> available;
82};
