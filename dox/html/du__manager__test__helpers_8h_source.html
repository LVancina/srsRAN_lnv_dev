216class f1u_gateway_dummy : public f1u_du_gateway
217{
218public:
219 bool next_bearer_is_created = true;
220
221 srs_du::f1u_bearer* create_du_bearer(uint32_t ue_index,
222 drb_id_t drb_id,
223 srs_du::f1u_config config,
224 const up_transport_layer_info& dl_tnl_info,
225 const up_transport_layer_info& ul_tnl_info,
226 srs_du::f1u_rx_sdu_notifier& du_rx,
227 timer_factory timers) override
228 {
229 if (next_bearer_is_created and f1u_bearers.count(dl_tnl_info) == 0) {
230 f1u_bearers.insert(std::make_pair(dl_tnl_info, std::map<up_transport_layer_info, f1u_bearer_dummy>{}));
231 f1u_bearers[dl_tnl_info].emplace(ul_tnl_info, du_rx);
232 return &f1u_bearers.at(dl_tnl_info).at(ul_tnl_info);
233 }
234 return nullptr;
235 }
236
237 void remove_du_bearer(const up_transport_layer_info& dl_tnl_info) override
238 {
239 auto bearer_it = f1u_bearers.find(dl_tnl_info);
240 if (bearer_it == f1u_bearers.end()) {
241 srslog::fetch_basic_logger("TEST").warning("Could not find DL-TEID at DU to remove. DL-TEID={}",
242 dl_tnl_info.gtp_teid);
243 return;
244 }
245 f1u_bearers.erase(bearer_it);
246 }
247
248 std::map<up_transport_layer_info, std::map<up_transport_layer_info, f1u_bearer_dummy>> f1u_bearers;
249};
