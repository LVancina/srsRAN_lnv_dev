49const uint16_t MAX_NOF_UES_PER_DU = 1024;
51const uint16_t MAX_NOF_DUS = 65535;
53#define MAX_NOF_CU_UES (MAX_NOF_DUS * MAX_NOF_UES_PER_DU)
55const uint16_t MAX_NOF_CU_UPS = 65535;
57const uint16_t MAX_NOF_DU_CELLS = 16;
58
61enum class ue_index_t : uint64_t { min = 0, max = MAX_NOF_CU_UES - 1, invalid = MAX_NOF_CU_UES };
62
64inline uint64_t ue_index_to_uint(ue_index_t index)
65{
66 return static_cast<uint64_t>(index);
67}
68
70inline ue_index_t uint_to_ue_index(std::underlying_type_t<ue_index_t> index)
71{
72 return static_cast<ue_index_t>(index);
73}
74
76enum class du_index_t : uint16_t { min = 0, max = MAX_NOF_DUS - 1, invalid = MAX_NOF_DUS };
77
79constexpr inline du_index_t uint_to_du_index(std::underlying_type_t<du_index_t> index)
80{
81 return static_cast<du_index_t>(index);
82}
83
85constexpr inline std::underlying_type_t<du_index_t> du_index_to_uint(du_index_t du_index)
86{
87 return static_cast<std::underlying_type_t<du_index_t>>(du_index);
88}
89
91inline du_index_t get_du_index_from_ue_index(ue_index_t index)
92{
93 if (index == ue_index_t::invalid) {
94 return du_index_t::invalid;
95 }
96 return uint_to_du_index((ue_index_to_uint(index) / MAX_NOF_UES_PER_DU));
97}
98
100inline ue_index_t generate_ue_index(du_index_t du_index, uint16_t index)
101{
102 return uint_to_ue_index(du_index_to_uint(du_index) * MAX_NOF_UES_PER_DU + index);
103}
