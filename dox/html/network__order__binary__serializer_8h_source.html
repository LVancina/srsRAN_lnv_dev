35class network_order_binary_serializer
36{
37public:
38 explicit network_order_binary_serializer(uint8_t* ptr_) : ptr(ptr_)
39 {
40 srsran_assert(ptr, "Invalid pointer to buffer");
41 }
42
44 void write(uint16_t x)
45 {
46 // Convert to network order.
47 uint16_t net_x = htons(x);
48 std::memcpy(ptr, &net_x, sizeof(net_x));
49 advance(sizeof(net_x));
50 }
51
53 void write(int16_t x)
54 {
55 // Convert to network order.
56 uint16_t net_x = htons(x);
57 std::memcpy(ptr, &net_x, sizeof(net_x));
58 advance(sizeof(net_x));
59 }
60
62 void write(uint8_t x)
63 {
64 std::memcpy(ptr, &x, sizeof(x));
65 advance(sizeof(x));
66 }
67
69 template <typename T>
70 void write(span<T> s)
71 {
72 for (const auto elem : s) {
73 write(elem);
74 }
75 }
76
78 void write(span<const uint8_t> s)
79 {
80 std::memcpy(ptr, s.data(), s.size() * sizeof(uint8_t));
81 advance(s.size() * sizeof(uint8_t));
82 }
83
85 void write(span<uint8_t> s) { write(span<const uint8_t>(s)); }
86
88 void advance(unsigned x)
89 {
90 ptr += x;
91 offset += x;
92 }
93
95 unsigned get_offset() const { return offset; }
96
97private:
98 uint8_t* ptr;
99 unsigned offset = 0;
100};
