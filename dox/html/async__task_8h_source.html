37 struct promise_type : public detail::promise_data<result_type, detail::task_promise_base> {
38 struct final_awaiter {
40 bool await_ready() const { return false; }
41
44 coro_handle<> await_suspend(coro_handle<promise_type> cb) { return cb.promise().continuation; }
45
46 void await_resume() {}
47
49 final_awaiter& get_awaiter() { return *this; }
50 };
51
53 suspend_always initial_suspend() { return {}; }
54
56 final_awaiter final_suspend() { return {}; }
57
58 async_task<R> get_return_object()
59 {
60 auto corohandle = coro_handle<promise_type>::from_promise(this);
61 corohandle.resume();
62 return async_task<R>{corohandle};
63 }
64 };
65
66 async_task() = default;
67 explicit async_task(coro_handle<promise_type> cb) : handle(cb) {}
68
70 async_task<R>& get_awaiter() { return *this; }
71
76 coro_handle<> await_suspend(coro_handle<> h) noexcept
77 {
78 srsran_sanity_check(not this->empty(), "Awaiting an empty task");
79 srsran_sanity_check(handle.promise().continuation.empty(), "Lazy task can only be awaited once.");
80
81 // Store continuation in promise, so that it gets called at "this" coroutine final suspension point.
82 handle.promise().continuation = h;
83
84 // We tail-resume the current awaiter task's coroutine, which is currently suspended at initial suspension point.
85 return *handle;
86 }
87
88private:
89 friend class detail::common_task_crtp<async_task<R>, R>;
90
91 unique_coroutine<promise_type> handle;
92};
