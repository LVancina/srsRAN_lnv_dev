50template <typename Range, typename IdentifierGetter>
51bool has_unique_ids(const Range& r, const IdentifierGetter& id_get)
52{
53 for (auto it = r.begin(); it != r.end(); ++it) {
54 auto it2 = it;
55 ++it2;
56 for (; it2 != r.end(); ++it2) {
57 if (id_get(*it) == id_get(*it2)) {
58 return false;
59 }
60 }
61 }
62 return true;
63}

70template <typename Range, typename ValueType = typename Range::value_type, typename IdType>
71bool has_unique_ids(const Range& r, IdType ValueType::*id_field)
72{
73 return has_unique_ids(r, [id_field](const auto& e) { return e.*id_field; });

92template <typename RangeSet,

99find_disconnected_id(const RangeSet& set, const Range& r, IdList SetValueType::*id_set_list, IdType ValueType::*idfield)
100{
101 for (const auto& es : set) {
102 for (const auto& id : es.*id_set_list) {
103 if (std::none_of(r.begin(), r.end(), [&](const auto& er) { return er.*idfield == id; })) {
104 return id;
105 }
106 }
107 }
108 return {};
