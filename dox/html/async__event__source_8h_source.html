36template <typename T>
37class async_event_source
38{
39public:
40 explicit async_event_source(timer_factory timer_db) : running_timer(timer_db.create_timer()) {}
41 async_event_source(const async_event_source&) = delete;
42 async_event_source(async_event_source&&) = delete;
43 async_event_source& operator=(const async_event_source&) = delete;
44 async_event_source& operator=(async_event_source&&) = delete;
45 ~async_event_source() { srsran_assert(not has_subscriber(), "Observers must not outlive event sources"); }
46
48 bool has_subscriber() const { return sub != nullptr; }
49
51 template <typename U>
52 bool set(U&& u)
53 {
54 if (has_subscriber()) {
55 running_timer.stop();
56 auto* old_sub = sub;
57 remove_observer();
58 old_sub->event.set(std::forward<U>(u));
59 return true;
60 }
61 return false;
62 }
63
64private:
65 friend class async_single_event_observer<T>;
66
67 void set_observer(async_single_event_observer<T>& sub_)
68 {
69 srsran_assert(not has_subscriber(), "This class only allows one observer/sink/listener per transaction");
70 srsran_assert(not sub_.connected(), "The observer cannot be already registered");
71 if (sub_.complete()) {
72 // observer finished the previous transaction. Its last result can be reset.
73 sub_.reset();
74 }
75 sub = &sub_;
76 sub->parent = this;
77 }
78
79 template <typename U>
80 void set_observer(async_single_event_observer<T>& sub_, std::chrono::milliseconds time_to_cancel, U&& cancelled_value)
81 {
82 set_observer(sub_);
83 // Setup timeout.
84 running_timer.set(time_to_cancel, [this, c = std::forward<U>(cancelled_value)](timer_id_t ) mutable {
85 set(std::forward<U>(c));
86 });
87 running_timer.run();
88 }
89
90 void remove_observer()
91 {
92 srsran_assert(has_subscriber(), "Unsubscribe called but no subscriber is registered");
93 sub->parent = nullptr;
94 sub = nullptr;
95 }
96
97 async_single_event_observer<T>* sub = nullptr;
98 unique_timer running_timer;
99};

103template <typename T>
104class async_single_event_observer
105{
106public:
107 using result_type = T;
108 using awaiter_type = typename manual_event<result_type>::awaiter_type;
109
110 async_single_event_observer() = default;
111 async_single_event_observer(async_event_source<T>& publisher) { subscribe_to(publisher); }
112 async_single_event_observer(async_single_event_observer&& other) = delete;
113 async_single_event_observer(const async_single_event_observer& other) = delete;
114 async_single_event_observer& operator=(async_single_event_observer&& other) = delete;
115 async_single_event_observer& operator=(const async_single_event_observer& other) = delete;
116 ~async_single_event_observer()
117 {
118 if (parent != nullptr) {
119 parent->remove_observer();
120 }
121 }
122
125 void subscribe_to(async_event_source<T>& publisher) { publisher.set_observer(*this); }
126
129 template <typename U>
130 void subscribe_to(async_event_source<T>& publisher, std::chrono::milliseconds time_to_cancel, U&& cancelled_value)
131 {
132 publisher.set_observer(*this, time_to_cancel, std::forward<U>(cancelled_value));
133 }
134
136 bool connected() const { return parent != nullptr; }
137
139 bool complete() const { return event.is_set(); }
140
142 void reset() { event.reset(); }
143
145 const T& result() const&
146 {
147 srsran_assert(complete(), "Trying to fetch result of incomplete transaction");
148 return event.get();
149 }
150 T result() &&
151 {
152 srsran_assert(complete(), "Trying to fetch result of incomplete transaction");
153 return std::move(event).get();
154 }
155
157 awaiter_type get_awaiter() { return event.get_awaiter(); }
158
159private:
160 friend class async_event_source<T>;
161
162 async_event_source<T>* parent = nullptr;
163 manual_event<T> event;
164};
