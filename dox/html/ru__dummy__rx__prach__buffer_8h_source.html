36class ru_dummy_rx_prach_buffer : private prach_buffer
37{
38public:
39 ru_dummy_rx_prach_buffer(unsigned seed, unsigned max_nof_ports)
40 {
41 // Resize PRACH buffer data tp the maximum buffer size.
42 data.resize({prach_constants::SHORT_SEQUENCE_LENGTH,
43 prach_constants::SHORT_SEQUENCE_MAX_NOF_SYMBOLS,
44 prach_constants::MAX_NOF_PRACH_FD_OCCASIONS,
45 prach_constants::MAX_NOF_PRACH_TD_OCCASIONS,
46 max_nof_ports});
47
48 // Fill PRACH buffer with random data.
49 complex_normal_distribution<cf_t> dist(0, 1);
50 std::mt19937 rgen(seed);
51 span<cf_t> data_view = data.get_data();
52 std::generate(data_view.begin(), data_view.end(), [&dist, &rgen]() { return dist(rgen); });
53 }
54
56 prach_buffer& get_buffer(const prach_buffer_context& context)
57 {
58 unsigned nof_symbols, sequence_length;
59
60 // Derive parameters from the PRACH format.
61 if (is_long_preamble(context.format)) {
62 prach_preamble_information info = get_prach_preamble_long_info(context.format);
63 nof_symbols = static_cast<unsigned>(info.symbol_length.to_seconds() * ra_scs_to_Hz(info.scs));
64 sequence_length = info.sequence_length;
65 } else {
66 prach_preamble_information info =
67 get_prach_preamble_short_info(context.format, to_ra_subcarrier_spacing(context.pusch_scs), false);
68 nof_symbols = static_cast<unsigned>(info.symbol_length.to_seconds() * ra_scs_to_Hz(info.scs));
69 sequence_length = info.sequence_length;
70 }
71
72 // Derive other parameters.
73 unsigned nof_ports = context.ports.size();
74 unsigned nof_td_occasions = context.nof_td_occasions;
75 unsigned nof_fd_occasions = context.nof_fd_occasions;
76
77 // Resize data to match context specifications.
78 if ((sequence_length != data.get_dimension_size(dims::re)) ||
79 (nof_symbols != data.get_dimension_size(dims::symbol)) ||
80 (nof_fd_occasions != data.get_dimension_size(dims::fd_occasion)) ||
81 (nof_td_occasions != data.get_dimension_size(dims::td_occasion)) ||
82 (nof_ports != data.get_dimension_size(dims::port))) {
83 data.resize({sequence_length, nof_symbols, nof_fd_occasions, nof_td_occasions, nof_ports});
84 }
85
86 return *this;
87 }
