32class baseband_gateway_buffer_reader_view : public baseband_gateway_buffer_reader
33{
34public:
39 baseband_gateway_buffer_reader_view(const baseband_gateway_buffer_reader& buffer_,
40 unsigned offset_,
41 unsigned nof_samples_) :
42 buffer(buffer_), offset(offset_), nof_samples(nof_samples_)
43 {
44 srsran_assert(buffer.get_nof_samples() >= offset + nof_samples,
45 "The offset (i.e., {}) plus the number of samples (i.e., {}) exceed the buffer size (i.e., {}).",
46 offset,
47 nof_samples,
48 buffer.get_nof_samples());
49 }
50
51 // See interface for documentation.
52 unsigned get_nof_channels() const override { return buffer.get_nof_channels(); }
53
54 // See interface for documentation.
55 unsigned get_nof_samples() const override { return nof_samples; }
56
57 // See interface for documentation.
58 span<const cf_t> get_channel_buffer(unsigned i_channel) const override
59 {
60 return buffer.get_channel_buffer(i_channel).subspan(offset, nof_samples);
61 }
62
63private:
64 const baseband_gateway_buffer_reader& buffer;
65 unsigned offset;
66 unsigned nof_samples;
