37class olla_algorithm
38{
39public:
45 olla_algorithm(float target_bler_, float snr_inc_step_db, float max_snr_offset_db) :
46 target_bler(target_bler_),
47 delta_down(snr_inc_step_db),
48 delta_up((1 - target_bler) * snr_inc_step_db / target_bler),
49 max_snr_offset(max_snr_offset_db)
50 {
51 srsran_assert(target_bler > 0 and target_bler < 0.5, "Invalid target BLER");
52 // See Francisco Blanquez-Casado, Gerardo Gomez, Maria del Carmen Aguayo-Torres and Jose Tomas Entrambasaguas,
53 // "eOLLA: an enhanced outer loop link adaptation for cellular networks", EURASIP Journal on Wireless
54 // Communications and Networking, 2016.
55 srsran_assert(delta_down < 2 * std::exp(1) * target_bler / 1.03,
56 "Convergence guarantees of the OLLA algorithm have not been met");
57 srsran_assert(max_snr_offset_db > 0.0f, "Invalid maximum SNR offset");
58 }
59
61 float update(bool ack, sch_mcs_index used_mcs, interval<sch_mcs_index, true> mcs_boundaries)
62 {
63 // The OLLA algorithm does not account for the saturation points of the MCS. For instance, if we applied the OLLA
64 // formula when MCS=28, and average BLER=0, the SNR offset would increase without bounds. As a workaround, we
65 // forbid increases in the SNR offset for MCS=maxMCS, and decreases in the SNR offset when MCS=minMCS.
66 float eff_delta_down = used_mcs >= mcs_boundaries.stop() ? 0 : delta_down;
67 float eff_delta_up = used_mcs <= mcs_boundaries.start() ? 0 : delta_up;
68 current_offset += ack ? eff_delta_down : -eff_delta_up;
69 current_offset = std::max(std::min(current_offset, max_snr_offset), -max_snr_offset);
70 return current_offset;
71 }
72
74 float offset_db() const { return current_offset; }
75
76private:
78 float target_bler;
80 float delta_down, delta_up;
82 float max_snr_offset;
83
85 float current_offset = 0;
86};
