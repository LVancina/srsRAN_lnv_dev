33template <typename Executor>
34class sync_task_executor final : public task_executor
35{
36 Executor executor;
37
38public:
39 template <typename E>
40 explicit sync_task_executor(E&& executor_) : executor(std::forward<E>(executor_))
41 {
42 }
43
44 bool execute(unique_task task) override
45 {
46 std::mutex mutex;
47 std::condition_variable cvar;
48 bool done = false;
49
50 bool ret = get(executor).execute([&mutex, &cvar, &done, task = std::move(task)]() {
51 task();
52
53 mutex.lock();
54 done = true;
55 cvar.notify_one();
56 mutex.unlock();
57 });
58
59 // Nothing to wait for if the task was not enqueued.
60 if (not ret) {
61 return false;
62 }
63
64 // Wait for the condition variable to be set.
65 std::unique_lock<std::mutex> lock(mutex);
66 while (not done) {
67 cvar.wait(lock);
68 }
69
70 return ret;
71 }
72
73 bool defer(unique_task task) override { return execute(std::move(task)); }
74
75private:
76 template <typename U>
77 static U& get(U* u)
78 {
79 return *u;
80 }
81
82 template <typename U>
83 static U& get(std::unique_ptr<U>& u)
84 {
85 return *u;
86 }
87
88 template <typename U>
89 static U& get(U& u)
90 {
91 return u;
92 }
93};

104template <typename Exec, typename Task, typename OnTaskDispatchFailure>
105void force_blocking_execute(Exec&& exec,
106 Task&& task,
107 OnTaskDispatchFailure&& fail_func,
108 unsigned max_attempts = std::numeric_limits<unsigned>::max())
109{
110 static_assert(std::is_copy_constructible<Task>::value, "Task must be copy assignable");
111
112 // Create a sync executor that will block the thread until the task is executed.
113 sync_task_executor<Exec> sync_task_exec(std::forward<Exec>(exec));
114
115 for (unsigned count = 0; count < max_attempts and not sync_task_exec.execute(task); ++count) {
116 fail_func();
117 }
118}
