86struct rlc_tx_metrics_atomic_lower {
87 std::atomic<uint32_t> num_pdus_no_segmentation;
88 std::atomic<size_t> num_pdu_bytes_no_segmentation;
89
91 std::atomic<rlc_mode> mode;
92
97 union {
98 rlc_tm_tx_metrics_atomic_lower tm;
99 rlc_um_tx_metrics_atomic_lower um;
100 rlc_am_tx_metrics_atomic_lower am;
101 } mode_specific;
102
103 void reset()
104 {
105 num_pdus_no_segmentation.store(0, std::memory_order_relaxed);
106 num_pdu_bytes_no_segmentation.store(0, std::memory_order_relaxed);
107
108 // reset mode-specific values
109 switch (mode.load(std::memory_order_relaxed)) {
110 case rlc_mode::tm:
111 mode_specific.tm.reset();
112 break;
113 case rlc_mode::um_bidir:
114 case rlc_mode::um_unidir_dl:
115 mode_specific.um.reset();
116 break;
117 case rlc_mode::am:
118 mode_specific.am.reset();
119 break;
120 default:
121 // nothing to do here
122 break;
123 }
124 }
125};

254 // Metrics getters and setters
255 rlc_tx_metrics get_metrics()
256 {
257 srsran_assert(enabled, "Trying to get metrics, but metrics are disabled.");
258 if (not enabled) {
259 return {};
260 }
261 rlc_tx_metrics ret;
262
263 // Metrics accessed from higher layer
264 ret.num_sdus = metrics_hi.num_sdus.load(std::memory_order_relaxed);
265 ret.num_of_pulled_sdus = metrics_hi.num_of_pulled_sdus.load(std::memory_order_relaxed);
266 ret.num_sdu_bytes = metrics_hi.num_sdu_bytes.load(std::memory_order_relaxed);
267 ret.num_dropped_sdus = metrics_hi.num_dropped_sdus.load(std::memory_order_relaxed);
268 ret.num_discarded_sdus = metrics_hi.num_discarded_sdus.load(std::memory_order_relaxed);
269 ret.num_discard_failures = metrics_hi.num_discard_failures.load(std::memory_order_relaxed);
270 ret.sum_sdu_latency_us = metrics_hi.sum_sdu_latency_us.load(std::memory_order_relaxed);
271
272 // Metrics accessed from lower layer
273 ret.num_pdus_no_segmentation = metrics_lo.num_pdus_no_segmentation.load(std::memory_order_relaxed);
274 ret.num_pdu_bytes_no_segmentation = metrics_lo.num_pdu_bytes_no_segmentation.load(std::memory_order_relaxed);
275 ret.mode = metrics_lo.mode.load(std::memory_order_relaxed);
276 switch (ret.mode) {
277 case rlc_mode::tm:
278 ret.mode_specific.tm.num_small_allocs =
279 metrics_lo.mode_specific.tm.num_small_allocs.load(std::memory_order_relaxed);
280 break;
281 case rlc_mode::um_bidir:
282 case rlc_mode::um_unidir_dl:
283 ret.mode_specific.um.num_pdus_with_segmentation =
284 metrics_lo.mode_specific.um.num_pdus_with_segmentation.load(std::memory_order_relaxed);
285 ret.mode_specific.um.num_pdu_bytes_with_segmentation =
286 metrics_lo.mode_specific.um.num_pdu_bytes_with_segmentation.load(std::memory_order_relaxed);
287 break;
288 case rlc_mode::am:
289 ret.mode_specific.am.num_pdus_with_segmentation =
290 metrics_lo.mode_specific.am.num_pdus_with_segmentation.load(std::memory_order_relaxed);
291 ret.mode_specific.am.num_pdu_bytes_with_segmentation =
292 metrics_lo.mode_specific.am.num_pdu_bytes_with_segmentation.load(std::memory_order_relaxed);
293 ret.mode_specific.am.num_retx_pdus = metrics_lo.mode_specific.am.num_retx_pdus.load(std::memory_order_relaxed);
294 ret.mode_specific.am.num_retx_pdu_bytes =
295 metrics_lo.mode_specific.am.num_retx_pdu_bytes.load(std::memory_order_relaxed);
296 ret.mode_specific.am.num_ctrl_pdus = metrics_lo.mode_specific.am.num_ctrl_pdus.load(std::memory_order_relaxed);
297 ret.mode_specific.am.num_ctrl_pdu_bytes =
298 metrics_lo.mode_specific.am.num_ctrl_pdu_bytes.load(std::memory_order_relaxed);
299 break;
300 default:
301 // nothing to do here
302 break;
303 }
304
305 return ret;
306 }
307
308 void reset_metrics()
309 {
310 srsran_assert(enabled, "Trying to reset metrics, but metrics are disabled.");
311 if (not enabled) {
312 return;
313 }
314 metrics_hi.reset();
315 metrics_lo.reset();
316 }
317
318 rlc_tx_metrics get_and_reset_metrics()
319 {
320 srsran_assert(enabled, "Trying to get metrics, but metrics are disabled.");
321 if (not enabled) {
322 return {};
323 }
324 rlc_tx_metrics ret = get_metrics();
325 reset_metrics();
326 return ret;
327 }
