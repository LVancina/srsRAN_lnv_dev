31class channel_state_information
32{
33public:
35 enum class sinr_type {
37 channel_estimator,
39 post_equalization,
41 evm
42 };
43
45 channel_state_information() : sinr_report_type(sinr_type::channel_estimator), epre_dB(NAN), rsrp_dB(NAN)
46 {
47 // Do nothing.
48 }
49
51 explicit channel_state_information(sinr_type sinr_report_type_) :
52 sinr_report_type(sinr_report_type_), epre_dB(NAN), rsrp_dB(NAN)
53 {
54 // Do nothing.
55 }
56
58 channel_state_information(const channel_state_information& other) = default;
59
61 static sinr_type sinr_type_from_string(const std::string& sinr_type_str)
62 {
63 if (sinr_type_str == "channel_estimator") {
64 return sinr_type::channel_estimator;
65 }
66 if (sinr_type_str == "post_equalization") {
67 return sinr_type::post_equalization;
68 }
69 if (sinr_type_str == "evm") {
70 return sinr_type::evm;
71 }
72 srsran_assertion_failure("Invalid SINR type");
73 return sinr_type::channel_estimator;
74 }
75
77 float get_sinr_dB() const
78 {
79 switch (sinr_report_type) {
80 case sinr_type::channel_estimator:
81 srsran_assert(sinr_ch_estimator_dB.has_value(),
82 "Selected CSI SINR type, i.e., channel estimator, is not available.");
83 return sinr_ch_estimator_dB.value();
84 case sinr_type::post_equalization:
85 srsran_assert(sinr_post_eq_dB.has_value(),
86 "Selected CSI SINR type, i.e., post equalization, is not available.");
87 return sinr_post_eq_dB.value();
88 case sinr_type::evm:
89 default:
90 srsran_assert(sinr_evm_dB.has_value(), "Selected CSI SINR type, i.e., from EVM, is not available.");
91 return sinr_evm_dB.value();
92 }
93 }
94
99 void set_sinr_dB(sinr_type type, float sinr_dB)
100 {
101 switch (type) {
102 case sinr_type::channel_estimator:
103 sinr_ch_estimator_dB.emplace(sinr_dB);
104 break;
105 case sinr_type::post_equalization:
106 sinr_post_eq_dB.emplace(sinr_dB);
107 break;
108 case sinr_type::evm:
109 sinr_evm_dB.emplace(sinr_dB);
110 break;
111 }
112 }
113
115 void set_evm(float evm_) { evm.emplace(evm_); }
116
118 void set_time_alignment(const phy_time_unit& time_alignment_) { time_alignment = time_alignment_; }
120 phy_time_unit get_time_alignment() const { return time_alignment; }
121
123 void set_epre(float epre_dB_) { epre_dB = epre_dB_; }
125 float get_epre_dB() const { return epre_dB; }
126
128 void set_rsrp(float rsrp_dB_) { rsrp_dB = rsrp_dB_; }
130 float get_rsrp_dB() const { return rsrp_dB; }
131
132private:
133 friend struct fmt::formatter<channel_state_information>;
138 sinr_type sinr_report_type;
140 phy_time_unit time_alignment;
142 optional<float> sinr_post_eq_dB;
144 optional<float> evm;
146 optional<float> sinr_evm_dB;
148 optional<float> sinr_ch_estimator_dB;
150 float epre_dB;
152 float rsrp_dB;
153};
