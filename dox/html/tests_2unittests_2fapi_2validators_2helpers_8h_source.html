180template <typename T, typename F>
181class validate_fapi_field
182{
183public:
184 virtual ~validate_fapi_field() = default;
185
186protected:
187 void execute_test(pdu_field_data<T> property, test_case_data params, std::function<T()> builder, F validator)
188 {
189 T pdu = given_the_pdu(property, params, builder);
190 bool result = when_executing_the_validation(pdu, validator);
191 then_check_the_report(result, property.property, params.result);
192 }
193
194 void execute_test_pdu(pdu_field_data<T> property,
195 test_case_data params,
196 std::function<T()> builder,
197 F validator,
198 srsran::fapi::dl_pdu_type pdu_type)
199 {
200 T pdu = given_the_pdu(property, params, builder);
201 bool result = when_executing_the_validation(pdu, validator, pdu_type);
202 then_check_the_report(result, property.property, params.result);
203 }
204
205private:
206 T given_the_pdu(pdu_field_data<T> property, test_case_data params, std::function<T()> builder)
207 {
208 T pdu = builder();
209 property.fun(pdu, params.value);
210 return pdu;
211 };
212
213 virtual bool when_executing_the_validation(T pdu, F validator) { return true; };
214
215 virtual bool when_executing_the_validation(T pdu, F validator, srsran::fapi::dl_pdu_type pdu_type) { return true; };
216
217 void then_check_the_report(bool result, const std::string& property, bool expected_result)
218 {
219 EXPECT_EQ(result, expected_result);
220 if (!result) {
221 EXPECT_EQ(1U, report.reports.size());
222 const auto& rep = report.reports.back();
223 EXPECT_EQ(property, rep.property_name);
224 } else {
225 EXPECT_TRUE(report.reports.empty());
226 }
227 };
228
229protected:
230 validator_report report = {0, 0};
231};

233using fapi_error = srsran::error_type<validator_report>;

235template <typename T>
236class validate_fapi_message : public validate_fapi_field<T, std::function<fapi_error(T& pdu)>>
237{
238 using base = validate_fapi_field<T, std::function<fapi_error(T& pdu)>>;
239
240public:
241 void execute_test(pdu_field_data<T> property,
242 test_case_data params,
243 std::function<T()> builder,
244 std::function<fapi_error(T& pdu)> validator,
245 message_type_id msg_type_id)
246 {
247 base::execute_test(property, params, builder, validator);
248
249 check_message_params(params.result, msg_type_id);
250 }
251
252private:
253 bool when_executing_the_validation(T pdu, std::function<fapi_error(T& pdu)> validator) override
254 {
255 fapi_error error = validator(pdu);
256 if (error.is_error()) {
257 base::report = error.error();
258 }
259
260 return !error.is_error();
261 };
262
263 void check_message_params(bool result, message_type_id msg_type_id) const
264 {
265 // In case of error, check the PDU type.
266 if (base::report.reports.size()) {
267 // Base class checks all the parameters but pdu_type, so it gets checked here. Base also checks that only one
268 // error report exists, so the last (or the first) of them can be used for the check.
269 const error_report& rep = base::report.reports.back();
270 EXPECT_EQ(static_cast<unsigned>(msg_type_id), static_cast<unsigned>(rep.message_type));
271 }
272 }
273};
