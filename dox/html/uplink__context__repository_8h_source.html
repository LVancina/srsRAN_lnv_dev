41class uplink_context
42{
43public:
45 struct uplink_context_resource_grid_info {
46 resource_grid_context context;
47 resource_grid* grid = nullptr;
48 };
49
51 uplink_context() = default;
52
54 uplink_context(unsigned symbol_, const resource_grid_context& context_, resource_grid& grid_) :
55 symbol(symbol_), grid({context_, &grid_})
56 {
57 re_written = static_vector<bounded_bitset<MAX_NOF_PRBS * NRE>, MAX_NOF_SUPPORTED_EAXC>(
58 grid_.get_writer().get_nof_ports(),
59 bounded_bitset<MAX_NOF_PRBS * NRE>(size_t(grid_.get_writer().get_nof_subc())));
60 }
61
63 bool empty() const { return grid.grid == nullptr; }
64
66 unsigned get_grid_nof_prbs() const
67 {
68 return (grid.grid) ? (grid.grid->get_writer().get_nof_subc() / NOF_SUBCARRIERS_PER_RB) : 0U;
69 }
70
72 const resource_grid_context& get_grid_context() const { return grid.context; }
73
76 span<const bounded_bitset<MAX_NOF_PRBS * NRE>> get_re_written_mask() const { return re_written; }
77
79 void write_grid(unsigned port, unsigned start_re, span<const cf_t> re_iq_buffer)
80 {
81 srsran_assert(grid.grid, "Invalid resource grid");
82
83 // Skip writing if the given port does not fit in the grid.
84 if (port >= grid.grid->get_writer().get_nof_ports()) {
85 return;
86 }
87
88 grid.grid->get_writer().put(port, symbol, start_re, re_iq_buffer);
89 re_written[port].fill(start_re, start_re + re_iq_buffer.size());
90 }
91
94 expected<uplink_context_resource_grid_info> try_getting_complete_resource_grid() const
95 {
96 if (!grid.grid) {
97 return default_error_t({});
98 }
99
100 if (!have_all_prbs_been_written()) {
101 return default_error_t({});
102 }
103
104 return {grid};
105 }
106
107private:
109 bool have_all_prbs_been_written() const
110 {
111 return std::all_of(
112 re_written.begin(), re_written.end(), [](const auto& port_re_written) { return port_re_written.all(); });
113 }
114
115private:
116 unsigned symbol;
117 uplink_context_resource_grid_info grid;
118 static_vector<bounded_bitset<MAX_NOF_PRBS * NRE>, MAX_NOF_SUPPORTED_EAXC> re_written;
119};

122class uplink_context_repository
123{
125 static constexpr unsigned SFN_MAX_VALUE = 1U << 8;
126
127 srslog::basic_logger* logger;
128 std::vector<std::array<uplink_context, MAX_NSYMB_PER_SLOT>> buffer;
129 //: TODO: make this lock free
130 mutable std::mutex mutex;
131
133 uplink_context& entry(slot_point slot, unsigned symbol)
134 {
135 srsran_assert(symbol < MAX_NSYMB_PER_SLOT, "Invalid symbol index '{}'", symbol);
136
137 slot_point entry_slot(slot.numerology(), slot.sfn() % SFN_MAX_VALUE, slot.slot_index());
138 unsigned index = entry_slot.system_slot() % buffer.size();
139 return buffer[index][symbol];
140 }
141
143 const uplink_context& entry(slot_point slot, unsigned symbol) const
144 {
145 srsran_assert(symbol < MAX_NSYMB_PER_SLOT, "Invalid symbol index '{}'", symbol);
146
147 slot_point entry_slot(slot.numerology(), slot.sfn() % SFN_MAX_VALUE, slot.slot_index());
148 unsigned index = entry_slot.system_slot() % buffer.size();
149 return buffer[index][symbol];
150 }
151
152public:
153 explicit uplink_context_repository(unsigned size_, srslog::basic_logger* logger_ = nullptr) :
154 logger(logger_), buffer(size_)
155 {
156 }
157
159 void add(const resource_grid_context& context, resource_grid& grid)
160 {
161 std::lock_guard<std::mutex> lock(mutex);
162 for (unsigned symbol_id = 0, symbol_end = grid.get_reader().get_nof_symbols(); symbol_id != symbol_end;
163 ++symbol_id) {
164 if (logger) {
165 if (!entry(context.slot, symbol_id).empty()) {
166 const resource_grid_context& previous_context = entry(context.slot, symbol_id).get_grid_context();
167 logger->warning("Missed incoming User-Plane uplink messages for slot '{}', symbol '{}' and sector#{}",
168 previous_context.slot,
169 symbol_id,
170 previous_context.sector);
171 }
172 }
173
174 entry(context.slot, symbol_id) = uplink_context(symbol_id, context, grid);
175 }
176 }
177
179 void write_grid(slot_point slot, unsigned port, unsigned symbol, unsigned start_re, span<const cf_t> re_iq_buffer)
180 {
181 std::lock_guard<std::mutex> lock(mutex);
182 entry(slot, symbol).write_grid(port, start_re, re_iq_buffer);
183 }
184
186 uplink_context get(slot_point slot, unsigned symbol) const
187 {
188 std::lock_guard<std::mutex> lock(mutex);
189 return entry(slot, symbol);
190 }
191
193 void clear(slot_point slot, unsigned symbol)
194 {
195 std::lock_guard<std::mutex> lock(mutex);
196 entry(slot, symbol) = {};
197 }
198};
