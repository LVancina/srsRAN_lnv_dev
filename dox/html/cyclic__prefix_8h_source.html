35class cyclic_prefix
36{
37public:
39 enum options {
41 NORMAL = 0,
43 EXTENDED
44 };
45
47 options value = NORMAL;
48
50 constexpr cyclic_prefix() = default;
51
53 constexpr cyclic_prefix(options opt) : value(opt) {}
54
56 constexpr operator options() const { return value; }
57
59 std::string to_string() const
60 {
61 if (value == NORMAL) {
62 return "normal";
63 }
64 return "extended";
65 }
66
74 bool is_valid(subcarrier_spacing scs, unsigned dft_size) const
75 {
76 // All units are in multiples of the constant kappa.
77 unsigned cp_len = 144U >> to_numerology_value(scs);
78 if (value == EXTENDED) {
79 cp_len = 512U >> to_numerology_value(scs);
80 }
81
82 unsigned sampling_rate_Hz = to_sampling_rate_Hz<unsigned>(scs, dft_size);
83
84 // The cyclic prefix lengths must result in an integer number of samples.
85 return phy_time_unit::from_units_of_kappa(cp_len).is_sample_accurate(sampling_rate_Hz) &&
86 phy_time_unit::from_units_of_kappa(16).is_sample_accurate(sampling_rate_Hz);
87 }
88
93 constexpr phy_time_unit get_length(unsigned symbol_idx, subcarrier_spacing scs) const
94 {
95 // All units are in multiples of the constant kappa.
96 unsigned cp_len = 144U >> to_numerology_value(scs);
97 if (value == EXTENDED) {
98 cp_len = 512U >> to_numerology_value(scs);
99 } else if (symbol_idx == 0 || symbol_idx == 7 * pow2(to_numerology_value(scs))) {
100 cp_len += 16;
101 }
102
103 return phy_time_unit::from_units_of_kappa(cp_len);
104 }
105};
106
108inline constexpr unsigned get_nsymb_per_slot(const cyclic_prefix cp)
109{
110 if (cp == cyclic_prefix::EXTENDED) {
111 return 12;
112 }
113 return 14;
114}
115
117static constexpr unsigned MAX_NSYMB_PER_SLOT =
118 std::max(get_nsymb_per_slot(cyclic_prefix::NORMAL), get_nsymb_per_slot(cyclic_prefix::EXTENDED));
