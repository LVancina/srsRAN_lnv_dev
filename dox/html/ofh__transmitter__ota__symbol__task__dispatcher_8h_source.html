33class transmitter_ota_symbol_task_dispatcher : public ota_symbol_boundary_notifier
34{
35public:
36 transmitter_ota_symbol_task_dispatcher(task_executor& executor_,
37 ota_symbol_boundary_notifier& window_checker_,
38 ota_symbol_boundary_notifier& symbol_handler_) :
39 executor(executor_), window_checker(window_checker_), symbol_handler(symbol_handler_)
40 {
41 }
42
43 void on_new_symbol(slot_symbol_point symbol_point) override
44 {
45 window_checker.on_new_symbol(symbol_point);
46
47 if (!executor.execute([&, symbol_point]() { symbol_handler.on_new_symbol(symbol_point); })) {
48 srslog::fetch_basic_logger("OFH").warning(
49 "Failed to dispatch new symbol task in the message transmitter for slot '{}' and symbol '{}'",
50 symbol_point.get_slot(),
51 symbol_point.get_symbol_index());
52 }
53 }
54
55private:
56 task_executor& executor;
57 ota_symbol_boundary_notifier& window_checker;
58 ota_symbol_boundary_notifier& symbol_handler;
59};
