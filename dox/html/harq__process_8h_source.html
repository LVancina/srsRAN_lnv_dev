598 void save_dl_harq_info(const dl_harq_process& harq, const slot_point& _slot_tx)
599 {
600 if (harq.empty()) {
601 return;
602 }
603 if (harq.get_slot_ack_timeout() <= _slot_tx) {
604 int idx = (harq.slot_ack().to_uint() + ntn_cs_koffset) % slot_ack_info.size();
605 slot_ack_info.at(idx).tbs_bytes = harq.last_alloc_params().tb[0]->tbs_bytes;
606 slot_ack_info.at(idx).h_id = harq.id;
607 slot_ack_info.at(idx).mcs = harq.last_alloc_params().tb[0]->mcs;
608 slot_ack_info.at(idx).mcs_table = harq.last_alloc_params().tb[0]->mcs_table;
609 slot_ack_info.at(idx).update = dl_harq_process::status_update::acked;
610 }
611 }

615 {
616 if (harq.empty()) {
617 return;
618 }
619 if (harq.get_slot_ack_timeout() <= _slot_tx) {
620 int idx = (harq.slot_ack().to_uint() + ntn_cs_koffset) % slot_tbs.size();
621 slot_tbs.at(idx) = harq.get_tbs_bytes();
622 }

626 dl_harq_process::dl_ack_info_result pop_ack_info(const slot_point& _slot_tx, mac_harq_ack_report_status ack)
627 {
628 auto ret = slot_ack_info.at(_slot_tx.to_uint() % slot_ack_info.size());
629 if (ret.tbs_bytes) {
630 slot_ack_info.at(_slot_tx.to_uint() % slot_ack_info.size()) = {};
631 } else {
632 return {};
633 }
634 ret.update = ack == mac_harq_ack_report_status::ack ? dl_harq_process::status_update::acked
635 : dl_harq_process::status_update::nacked;
636 return ret;
637 }

641 {
642 unsigned ret = slot_tbs.at(_slot_tx.to_uint() % slot_tbs.size());
643 if (ret) {

645 } else {
646 return -1;
647 }
648 return ret;
