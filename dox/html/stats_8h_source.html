57 void update(T value)
58 {
59 float delta = value - mean;
60 ++nof_observations;
61 mean += delta / nof_observations;
62
63 // The following update formulas are inspired by
64 // Philippe Pebay, "Formulas for Robust, One-Pass Parallel Computation of Covariances and Arbitrary-Order
65 // Statistical Moments," SANDIA REPORT SAND2008-6212 (2008).
66 // [Online]. Available: https://www.osti.gov/servlets/purl/1028931
67 float nof_obs_sqrd = static_cast<float>(nof_observations) * static_cast<float>(nof_observations);
68 float delta_sqrd = delta * delta;
69 centered_power_sum_ord4 += -4 * centered_power_sum_ord3 / nof_observations * delta +
70 6 * centered_power_sum_ord2 / nof_obs_sqrd * delta_sqrd +
71 (nof_observations - 1.0F) * (nof_obs_sqrd - 3.0F * nof_observations + 3.0F) /
72 (nof_obs_sqrd * nof_observations) * delta_sqrd * delta_sqrd;
73
74 centered_power_sum_ord3 += -3 * centered_power_sum_ord2 / nof_observations * delta +
75 (nof_obs_sqrd - 3.0F * nof_observations + 2.0F) / nof_obs_sqrd * delta_sqrd * delta;
76
77 centered_power_sum_ord2 += (nof_observations - 1.0F) / nof_observations * delta_sqrd;
78
79 max = std::max(max, value);
80 min = std::min(min, value);
81 }

119 float get_variance(bias biased = bias::UNBIASED) const
120 {
121 if (nof_observations < 2) {
122 return NAN;
123 }
124
125 if (biased == bias::UNBIASED) {
126 return centered_power_sum_ord2 / (nof_observations - 1);
127 }
128 return centered_power_sum_ord2 / nof_observations;
129 }

136 float get_std(bias biased = bias::UNBIASED) const
137 {
138 if (nof_observations < 2) {
139 return NAN;
140 }
141 return std::sqrt(get_variance(biased));
142 }
143
155 float get_sem(bias biased = bias::UNBIASED) const
156 {
157 if (nof_observations < 2) {
158 return NAN;
159 }
160 return get_std(biased) / std::sqrt(nof_observations);
161 }
162
185 float get_skewness(bias biased = bias::UNBIASED) const
186 {
187 if (nof_observations < 3) {
188 return NAN;
189 }
190 float num = centered_power_sum_ord3 / nof_observations;
191 float den_tmp = std::sqrt(centered_power_sum_ord2 / nof_observations);
192 float den = den_tmp * den_tmp * den_tmp;
193
194 float result = num / den;
195
196 if (biased == bias::UNBIASED) {
197 result *= std::sqrt(nof_observations * (nof_observations - 1.0F)) / (nof_observations - 2.0F);
198 }
199 return result;
200 }
201
225 float get_kurtosis(bias biased = bias::UNBIASED) const
226 {
227 if (nof_observations < 4) {
228 return NAN;
229 }
230 float num = centered_power_sum_ord4 / nof_observations;
231 float den_tmp = centered_power_sum_ord2 / nof_observations;
232 float den = den_tmp * den_tmp;
233
234 float result = num / den;
235
236 if (biased == bias::UNBIASED) {
237 result = result * (nof_observations + 1) - 3.0F * (nof_observations - 1);
238 result = result * (nof_observations - 1) / ((nof_observations - 2.0F) * (nof_observations - 3.0F)) + 3.0F;
239 }
240 return result;
241 }

243private:

257 float centered_power_sum_ord3 = 0;
259 float centered_power_sum_ord4 = 0;
