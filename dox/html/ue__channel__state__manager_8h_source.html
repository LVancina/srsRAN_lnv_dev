56 optional<pdsch_precoding_info> get_precoding(unsigned chosen_nof_layers, prb_interval pdsch_prbs) const
57 {
58 srsran_assert(chosen_nof_layers <= nof_dl_ports, "Invalid number of layers chosen");
59 optional<pdsch_precoding_info> precoding_info;
60 if (nof_dl_ports <= 1) {
61 // In case of 1 DL port, no precoding is used.
62 return precoding_info;
63 }
64 precoding_info.emplace();
65 precoding_info->nof_rbs_per_prg = pdsch_prbs.length();
66 precoding_info->prg_infos.emplace_back(recommended_prg_info[nof_layers_to_index(chosen_nof_layers)]);
67 return precoding_info;
68 }
