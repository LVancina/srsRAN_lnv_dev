44public:

56 pusch_decoder_buffer&
57 new_transmission(unsigned message_length, modulation_scheme modulation, pusch_uci_decoder_notifier& notifier_)
58 {
59 notifier = &notifier_;
60 softbits_count = 0;
61 message = span<uint8_t>(temp_message).first(message_length);
62 decoder_config.modulation = modulation;
63 return *this;
64 }
65
66private:
68 uci_decoder& decoder;
70 pusch_uci_decoder_notifier* notifier = nullptr;
72 std::vector<log_likelihood_ratio> softbits_buffer;
74 unsigned softbits_count;
76 std::array<uint8_t, uci_constants::MAX_NOF_PAYLOAD_BITS> temp_message;
78 span<uint8_t> message;
80 uci_decoder::configuration decoder_config;
81
82 // See interface for documentation.
83 span<log_likelihood_ratio> get_next_block_view(unsigned block_size) override
84 {
85 // Makes sure the block size does not overflow the buffer.
86 srsran_assert(softbits_count + block_size <= softbits_buffer.size(),
87 "The sum of current buffer number of elements (i.e., {}) and the block size (i.e., {}), exceeds the "
88 "total number of elements of the buffer (i.e., {}).",
89 softbits_count,
90 block_size,
91 softbits_buffer.size());
92
93 return span<log_likelihood_ratio>(softbits_buffer).subspan(softbits_count, block_size);
94 }
95
96 // See interface for documentation.
97 void on_new_softbits(span<const log_likelihood_ratio> softbits) override
98 {
99 span<log_likelihood_ratio> block = get_next_block_view(softbits.size());
100
101 // Copy only if the soft bits do not match.
102 if (block.data() != softbits.data()) {
103 srsvec::copy(block, softbits);
104 }
105
106 softbits_count += softbits.size();
107 }
108
109 // See interface for documentation.
110 void on_end_softbits() override
111 {
112 // Select codeword from the soft bit buffer.
113 span<const log_likelihood_ratio> codeword = span<const log_likelihood_ratio>(softbits_buffer).first(softbits_count);
114
115 // Actual decode.
116 uci_status status = decoder.decode(message, codeword, decoder_config);
117
118 // Notify decoded UCI.
119 srsran_assert(notifier != nullptr, "Invalid notifier.");
120 notifier->on_uci_decoded(message, status);
121 }
122};
