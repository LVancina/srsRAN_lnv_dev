32class test_e1ap_message_notifier : public e1ap_message_notifier
33{
34public:
35 test_e1ap_message_notifier(bool cu_cp_to_cu_up_dir_,
36 std::vector<e1ap_message>* last_pdus_,
37 std::unique_ptr<e1ap_message_notifier> decorated = nullptr) :
38 cu_cp_to_cu_up_dir(cu_cp_to_cu_up_dir_), last_pdus(last_pdus_), notifier(std::move(decorated))
39 {
40 }
41
42 void on_new_message(const e1ap_message& msg) override
43 {
44 if (cu_cp_to_cu_up_dir) {
45 logger.info("E1AP Forwarding CU-CP > CU-UP: {}", msg.pdu.type().to_string());
46 } else {
47 logger.info("E1AP Forwarding CU-UP > CU-CP: {}", msg.pdu.type().to_string());
48 }
49 if (last_pdus != nullptr) {
50 last_pdus->push_back(msg);
51 }
52 if (notifier != nullptr) {
53 notifier->on_new_message(msg);
54 }
55 }
56
57private:
58 srslog::basic_logger& logger = srslog::fetch_basic_logger("TEST");
59 bool cu_cp_to_cu_up_dir;
60 std::vector<e1ap_message>* last_pdus;
61 std::unique_ptr<e1ap_message_notifier> notifier;
62};

66class e1ap_test_local_gateway : public srs_cu_up::e1ap_connection_client
67{
68public:
69 e1ap_test_local_gateway() = default;
70 explicit e1ap_test_local_gateway(srs_cu_cp::cu_cp_e1_handler& cu_cp_cu_up_mng_) : cu_cp_cu_up_mng(&cu_cp_cu_up_mng_)
71 {
72 }
73
74 void attach_cu_cp_cu_up_repo(srs_cu_cp::cu_cp_e1_handler& cu_cp_cu_up_mng_) { cu_cp_cu_up_mng = &cu_cp_cu_up_mng_; }
75
76 std::unique_ptr<e1ap_message_notifier>
77 handle_cu_up_connection_request(std::unique_ptr<e1ap_message_notifier> cu_up_rx_pdu_notifier) override
78 {
79 connections.emplace_back(std::make_unique<cu_up_connection_test_context>());
80 auto& conn = connections.back();
81
82 auto decorated_cu_up_rx_pdu_notifier =
83 std::make_unique<test_e1ap_message_notifier>(true, &conn->last_cu_up_rx_pdus, std::move(cu_up_rx_pdu_notifier));
84
85 auto cu_rx_pdu_notifier = cu_cp_cu_up_mng->handle_new_cu_up_connection(std::move(decorated_cu_up_rx_pdu_notifier));
86
87 return std::make_unique<test_e1ap_message_notifier>(
88 false, &conn->last_cu_cp_rx_pdus, std::move(cu_rx_pdu_notifier));
89 }
90
91 span<const e1ap_message> get_last_cu_cp_rx_pdus(std::size_t connection_index) const
92 {
93 return connections.at(connection_index)->last_cu_cp_rx_pdus;
94 }
95 span<const e1ap_message> get_last_cu_cp_tx_pdus(std::size_t connection_index) const
96 {
97 return connections.at(connection_index)->last_cu_up_rx_pdus;
98 }
99
100 void clear_messages()
101 {
102 for (unsigned i = 0; i != connections.size(); ++i) {
103 connections.at(i)->last_cu_cp_rx_pdus.clear();
104 connections.at(i)->last_cu_up_rx_pdus.clear();
105 }
106 }
107
108private:
109 struct cu_up_connection_test_context {
111 std::vector<e1ap_message> last_cu_cp_rx_pdus;
112
114 std::vector<e1ap_message> last_cu_up_rx_pdus;
115 };
116
117 srs_cu_cp::cu_cp_e1_handler* cu_cp_cu_up_mng = nullptr;
118
119 std::vector<std::unique_ptr<cu_up_connection_test_context>> connections;
120};
