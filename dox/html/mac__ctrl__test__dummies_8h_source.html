124class dummy_mac_event_indicator : public mac_ul_ccch_notifier
125{
126public:
127 optional<ul_ccch_indication_message> last_ccch_ind;
128
129 void on_ul_ccch_msg_received(const ul_ccch_indication_message& msg) override { last_ccch_ind = msg; }
130
131 bool verify_ul_ccch_msg(const ul_ccch_indication_message& test_msg)
132 {
133 if (not last_ccch_ind.has_value()) {
134 return false;
135 }
136 bool test = last_ccch_ind.value().tc_rnti == test_msg.tc_rnti &&
137 last_ccch_ind.value().cell_index == test_msg.cell_index &&
138 last_ccch_ind.value().slot_rx == test_msg.slot_rx && last_ccch_ind.value().subpdu == test_msg.subpdu;
139 return test;
140 }
141
142 bool verify_no_ul_ccch_msg() const { return not last_ccch_ind.has_value(); }
143};
