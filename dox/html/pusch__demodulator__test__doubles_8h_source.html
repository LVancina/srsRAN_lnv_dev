34class pusch_demodulator_spy : public pusch_demodulator
35{
36public:
37 struct entry_t {
38 std::vector<log_likelihood_ratio> codeword;
39 dynamic_bit_buffer scrambling_seq;
40 const resource_grid_reader* grid;
41 const channel_estimate* estimates;
42 configuration config;
43 };
44
45 pusch_demodulator_spy() : llr_dist(log_likelihood_ratio::min().to_int(), log_likelihood_ratio::max().to_int()) {}
46
47 void demodulate(pusch_codeword_buffer& codeword_buffer,
48 pusch_demodulator_notifier& notifier,
49 const resource_grid_reader& grid,
50 const channel_estimate& estimates,
51 const configuration& config) override
52 {
53 entries.emplace_back();
54 entry_t& entry = entries.back();
55 entry.codeword.resize(codeword_size);
56 entry.scrambling_seq.resize(codeword_size);
57 entry.grid = &grid;
58 entry.estimates = &estimates;
59 entry.config = config;
60
61 std::generate(
62 entry.codeword.begin(), entry.codeword.end(), [this]() { return log_likelihood_ratio(llr_dist(rgen)); });
63 std::generate(entry.scrambling_seq.get_buffer().begin(), entry.scrambling_seq.get_buffer().end(), [this]() {
64 return rgen();
65 });
66 codeword_buffer.on_new_block(entry.codeword, entry.scrambling_seq);
67 codeword_buffer.on_end_codeword();
68
69 // Report empty demodulation statistics.
70 notifier.on_end_stats(pusch_demodulator_notifier::demodulation_stats());
71 }
72
73 void set_codeword_size(unsigned codeword_size_) { codeword_size = codeword_size_; }
74
75 const std::vector<entry_t>& get_entries() const { return entries; }
76
77 void clear() { entries.clear(); }
78
79private:
80 unsigned codeword_size = 0;
81 std::vector<entry_t> entries;
82 std::mt19937 rgen;
83 std::uniform_int_distribution<log_likelihood_ratio::value_type> llr_dist;
84};
