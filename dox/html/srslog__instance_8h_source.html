35class srslog_instance : public sink_repository
36{
37 srslog_instance()
38 {
39 // stdout and stderr sinks are always present.
40 auto& stdout_sink =
41 sink_repo.emplace(std::piecewise_construct,
42 std::forward_as_tuple("stdout"),
43 std::forward_as_tuple(new stream_sink(sink_stream_type::stdout,
44 std::unique_ptr<log_formatter>(new text_formatter))));
45 default_sink = stdout_sink.get();
46
47 sink_repo.emplace(std::piecewise_construct,
48 std::forward_as_tuple("stderr"),
49 std::forward_as_tuple(new stream_sink(sink_stream_type::stderr,
50 std::unique_ptr<log_formatter>(new text_formatter))));
51
52 // Initialize the default formatter pointer with a text formatter.
53 {
54 detail::scoped_lock lock(formatter_mutex);
55 default_formatter = std::unique_ptr<log_formatter>(new text_formatter);
56 }
57 }

112 std::vector<sink*> contents() override
113 {
114 auto temp_contents = sink_repo.contents();
115
116 std::vector<sink*> contents;
117 contents.reserve(temp_contents.size());
118 for (auto& elem : temp_contents) {
119 contents.push_back(elem->get());
120 }
121
122 return contents;
123 }

134 std::unique_ptr<log_formatter> default_formatter;
