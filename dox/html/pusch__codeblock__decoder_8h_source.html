74 crc_calculator* select_crc(crc_generator_poly poly)
75 {
76 srsran_assert((poly == crc_generator_poly::CRC16) || (poly == crc_generator_poly::CRC24A) ||
77 (poly == crc_generator_poly::CRC24B),
78 "Invalid CRC polynomial.");
79 if (poly == crc_generator_poly::CRC16) {
80 return crc_set.crc16.get();
81 }
82 if (poly == crc_generator_poly::CRC24A) {
83 return crc_set.crc24A.get();
84 }
85 if (poly == crc_generator_poly::CRC24B) {
86 return crc_set.crc24B.get();
87 }
88 return nullptr;
89 }
