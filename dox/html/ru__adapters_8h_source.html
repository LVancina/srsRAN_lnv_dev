104 srsran_assert(context.sector < handlers.size(), "Unsupported sector {}", context.sector);
105 handlers[context.sector]->handle_rx_srs_symbol({context.sector, context.slot, context.symbol_id});
106 }
107
109 void map_handler(unsigned sector, upper_phy_rx_symbol_handler& hndlr)

111 srsran_assert(sector < handlers.size(), "Unsupported sector {}", sector);

113 handlers[sector] = &hndlr;
