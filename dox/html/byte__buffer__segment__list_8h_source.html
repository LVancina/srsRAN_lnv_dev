32class byte_buffer_segment_list
33{
34public:
36 struct node_t : public byte_buffer_segment {
37 node_t* next = nullptr;
38
39 using byte_buffer_segment::byte_buffer_segment;
40 };
41
43 node_t* head = nullptr;
44
46 node_t* tail = nullptr;
47
49 byte_buffer_segment_list() = default;
50
52 byte_buffer_segment_list(byte_buffer_segment_list&& other) noexcept :
53 head(std::exchange(other.head, nullptr)), tail(std::exchange(other.tail, nullptr))
54 {
55 }
56
58 byte_buffer_segment_list& operator=(byte_buffer_segment_list&& other) noexcept
59 {
60 head = std::exchange(other.head, nullptr);
61 tail = std::exchange(other.tail, nullptr);
62 return *this;
63 }
64
66 void push_back(node_t& seg) noexcept
67 {
68 seg.next = nullptr;
69 if (head != nullptr) {
70 tail->next = &seg;
71 tail = &seg;
72 } else {
73 head = &seg;
74 tail = &seg;
75 }
76 }
77
79 void push_front(node_t& seg)
80 {
81 seg.next = head;
82 if (tail == nullptr) {
83 tail = &seg;
84 }
85 head = &seg;
86 }
87
89 void pop_front()
90 {
91 if (head == tail) {
92 head = nullptr;
93 tail = nullptr;
94 } else {
95 head = head->next;
96 }
97 }
98
100 void pop_back()
101 {
102 if (head == tail) {
103 head = nullptr;
104 tail = nullptr;
105 return;
106 }
107
108 // tail needs to be recomputed.
109 node_t* new_tail = head;
110 for (; new_tail->next != tail; new_tail = new_tail->next) {
111 }
112 tail = new_tail;
113 tail->next = nullptr;
114 }
115
117 bool empty() const { return head == nullptr; }
118
120 node_t& front() { return *head; }
121 const node_t& front() const { return *head; }
122
124 node_t& back() { return *tail; }
125 const node_t& back() const { return *tail; }
126};

133{
134public:
135 using iterator_type = byte_buffer_segment_byte_iterator_impl<T>;

144 current_segment(start_segment)
145 {

148 current_segment(start_segment), offset(offset_)

153 template <typename U, std::enable_if_t<not std::is_same<U, T>::value, bool> = true>
154 byte_buffer_segment_byte_iterator_impl(const byte_buffer_segment_byte_iterator_impl<U>& other) :
155 current_segment(other.current_segment), offset(other.offset)
156 {
157 }
158
159 reference operator*() { return *(current_segment->data() + offset); }
160 reference operator*() const { return *(current_segment->data() + offset); }
161 pointer operator->() { return (current_segment->data() + offset); }
162 pointer operator->() const { return (current_segment->data() + offset); }
163
164 byte_buffer_segment_byte_iterator_impl& operator++()
165 {
166 offset++;
167 if (offset >= current_segment->length()) {
168 offset = 0;
169 current_segment = current_segment->next;
170 }
171 return *this;
172 }
173 byte_buffer_segment_byte_iterator_impl operator++(int)
174 {
175 byte_buffer_segment_byte_iterator_impl tmp(*this);
176 ++(*this);
177 return tmp;
178 }
179 byte_buffer_segment_byte_iterator_impl operator+(unsigned n) const
180 {
181 byte_buffer_segment_byte_iterator_impl tmp(*this);
182 tmp += n;
183 return tmp;
184 }
185 byte_buffer_segment_byte_iterator_impl& operator+=(unsigned n)
186 {
187 offset += n;
188 while (current_segment != nullptr and offset >= current_segment->length()) {
189 offset -= current_segment->length();
190 current_segment = current_segment->next;
191 }
192 srsran_assert(current_segment != nullptr or offset == 0, "Out-of-bounds Access");
193 return *this;
194 }
195
197 difference_type operator-(const byte_buffer_segment_byte_iterator_impl<T>& other) const
198 {
199 difference_type diff = 0;
200 byte_buffer_segment_list::node_t* seg = other.current_segment;
201 for (; seg != current_segment; seg = seg->next) {
202 diff += seg->length();
203 }
204 diff += offset - other.offset;
205 return diff;
206 }
207
208 bool operator==(const byte_buffer_segment_byte_iterator_impl<T>& other) const
209 {
210 return current_segment == other.current_segment and offset == other.offset;
211 }
212 bool operator!=(const byte_buffer_segment_byte_iterator_impl<T>& other) const { return !(*this == other); }
213
214private:
215 template <typename OtherT>
216 friend class byte_buffer_segment_byte_iterator_impl;
217 template <typename U>
218 friend class byte_buffer_segment_list_span_iterator_impl;
219
220 byte_buffer_segment_list::node_t* current_segment = nullptr;
221 size_t offset = 0;
