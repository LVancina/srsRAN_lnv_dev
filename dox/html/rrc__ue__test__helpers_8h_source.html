40// Free-function to generate dummy security context (used by e.g. Mobility tests)
41static security::security_context generate_security_context()
42{
43 const char* sk_gnb_cstr = "8d2abb1a4349319ea4276295c33d107a6e274495cb9bc2433fb7d7ca4c3f7646";
44
45 // Pack hex strings into srsgnb types
46 security::sec_key sk_gnb = make_sec_key(sk_gnb_cstr);
47
48 // Initialize security context and capabilities.
49 security::security_context sec_ctxt = {};
50 sec_ctxt.k = sk_gnb;
51 std::fill(sec_ctxt.supported_int_algos.begin(), sec_ctxt.supported_int_algos.end(), true);
52 std::fill(sec_ctxt.supported_enc_algos.begin(), sec_ctxt.supported_enc_algos.end(), true);
53
54 // Select preferred integrity algorithm.
55 security::preferred_integrity_algorithms inc_algo_pref_list = {security::integrity_algorithm::nia2,
56 security::integrity_algorithm::nia1,
57 security::integrity_algorithm::nia3,
58 security::integrity_algorithm::nia0};
59 security::preferred_ciphering_algorithms ciph_algo_pref_list = {security::ciphering_algorithm::nea0,
60 security::ciphering_algorithm::nea2,
61 security::ciphering_algorithm::nea1,
62 security::ciphering_algorithm::nea3};
63
64 sec_ctxt.select_algorithms(inc_algo_pref_list, ciph_algo_pref_list);
65
66 // Generate K_rrc_enc and K_rrc_int
67 sec_ctxt.generate_as_keys();
68
69 return sec_ctxt;
70}
