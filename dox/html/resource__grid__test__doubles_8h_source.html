83 // See interface for documentation.
84 void put(unsigned port, span<const resource_grid_coordinate> coordinates, span<const cf_t> symbols) override
85 {
86 std::unique_lock<std::mutex> lock(entries_mutex);
87 ++count;
88 const cf_t* symbol_ptr = symbols.begin();
89 for (const resource_grid_coordinate& coordinate : coordinates) {
90 put(port, coordinate.symbol, coordinate.subcarrier, *(symbol_ptr++));
91 }
92 fmt::print("entries.size()={}\n", entries.size());
93 }

95 // See interface for documentation.
96 span<const cf_t>
97 put(unsigned port, unsigned l, unsigned k_init, span<const bool> mask, span<const cf_t> symbols) override
98 {
99 std::unique_lock<std::mutex> lock(entries_mutex);
100 TESTASSERT(k_init + mask.size() <= max_prb * NRE,
101 "The mask staring at {} for {} subcarriers exceeds the resource grid bandwidth (max {}).",
102 k_init,
103 mask.size(),
104 max_prb * NRE);
105 ++count;
106 unsigned i_symb = 0;
107 for (unsigned k = 0; k != mask.size(); ++k) {
108 if (mask[k]) {
109 put(port, l, k_init + k, symbols[i_symb]);
110 i_symb++;
111 }
112 }
113
114 // Consume buffer.
115 return symbols.last(symbols.size() - i_symb);
116 }
117
118 span<const cf_t> put(unsigned port,
119 unsigned l,
120 unsigned k_init,
121 const bounded_bitset<NRE * MAX_RB>& mask,
122 span<const cf_t> symbols) override
123 {
124 std::unique_lock<std::mutex> lock(entries_mutex);
125 ++count;
126 unsigned i_symb = 0;
127 for (unsigned k = 0; k != mask.size(); ++k) {
128 // Skip if subcarrier is not active.
129 if (!mask.test(k)) {
130 continue;
131 }
132
133 // Skip Symbol if the symbol is 0.
134 if ((symbols[i_symb].real() != 0) || (symbols[i_symb].imag() != 0)) {
135 put(port, l, k + k_init, symbols[i_symb]);
136 }
137
138 ++i_symb;
139 }
140
141 // Consume buffer.
142 return symbols.last(symbols.size() - i_symb);
143 }

155 void put(unsigned port, unsigned l, unsigned k_init, unsigned stride, span<const cf_t> symbols) override

158 ++count;
159 for (unsigned i_symb = 0; i_symb != symbols.size(); ++i_symb) {
160 if ((symbols[i_symb].real() != 0) || (symbols[i_symb].imag() != 0)) {
161 put(port, l, k_init + (i_symb * stride), symbols[i_symb]);
162 }
163 }

173 void assert_entries(span<const expected_entry_t> expected_entries) const
174 {
175 // Make sure the number of elements match.
176 TESTASSERT_EQ(expected_entries.size(), entries.size());
177
178 // Iterate each expected entry, check that there is an entry and that the expected value error is below a threshold.
179 for (const auto& entry : expected_entries) {
180 entry_key_t key = {entry.port, entry.symbol, entry.subcarrier};
181 TESTASSERT(entries.count(key),
182 "No resource element was written for port={}, symbol={} and subcarrier={}.",
183 entry.port,
184 entry.symbol,
185 entry.subcarrier);
186
187 cf_t value = entries.at(key);
188 float err = std::abs(entry.value - value);
189 TESTASSERT(err < ASSERT_MAX_ERROR, "Mismatched value {} but expected {}", value, entry.value);
190 }
191 }
192
201 void assert_entries(span<const expected_entry_t> expected_entries, float max_error) const
202 {
203 // Make sure the number of elements match.
204 TESTASSERT_EQ(expected_entries.size(), entries.size());
205
206 // Iterate each expected entry, check that there is an entry and that the expected value error is below a threshold.
207 for (const auto& entry : expected_entries) {
208 entry_key_t key = {entry.port, entry.symbol, entry.subcarrier};
209 TESTASSERT(entries.count(key),
210 "No resource element was written for port={}, symbol={} and subcarrier={}.",
211 entry.port,
212 entry.symbol,
213 entry.subcarrier);
214
215 cf_t value = entries.at(key);
216 float err = std::abs(entry.value - value);
217 TESTASSERT(err < max_error, "Mismatched value {} but expected {}", value, entry.value);
218 }
219 }

222 unsigned get_count() const { return count; }

225 void reset()

228 count = 0;

233 using entry_key_t = std::tuple<uint8_t, uint8_t, uint16_t>;
234
236 static constexpr float ASSERT_MAX_ERROR = 1e-6;
237
239 std::map<entry_key_t, cf_t> entries;
240
242 std::mutex entries_mutex;
243
245 unsigned count = 0;
246
248 unsigned max_ports;
249
251 unsigned max_symb;
252
254 unsigned max_prb;
255
257 srslog::basic_logger& logger;
258
266 void put(uint8_t port, uint8_t symbol, uint16_t subcarrier, cf_t value)
267 {
268 // Generate key.
269 entry_key_t key{port, symbol, subcarrier};
270
271 // Ensure the port, symbol and subcarrier indexes are in range.
272 TESTASSERT(port < max_ports, "Port index {} exceeded maximum {}.", port, max_ports);
273 TESTASSERT(symbol < max_symb, "Symbol index {} exceeded maximum {}.", symbol, max_symb);
274 TESTASSERT(subcarrier < max_prb * NRE, "Subcarrier index {} exceeded maximum {}.", subcarrier, max_prb * NRE);
275
276 // Ensure the resource element does not exist.
277 TESTASSERT(entries.count(key) == 0,
278 "Detected resource grid overwrite for port={}, symbol={} and subcarrier={}.",
279 port,
280 symbol,
281 subcarrier);
282
283 // Debug trace.
284 logger.debug("[put] port={:>2}; symbol={:>2}; subcarrier={:>4}; value={}; count={};",
285 port,
286 symbol,
287 subcarrier,
288 value,
289 entries.size() + 1);
290
291 // Write element.
292 entries.emplace(key, value);
293 }

300 using expected_entry_t = resource_grid_writer_spy::expected_entry_t;
301
302 resource_grid_reader_spy(unsigned max_ports_ = 0, unsigned max_symb_ = 0, unsigned max_prb_ = 0) :
303 max_ports(max_ports_), max_symb(max_symb_), max_prb(max_prb_)

307 unsigned get_nof_ports() const override { return max_ports; }
308
309 unsigned get_nof_subc() const override { return max_prb * NRE; }
310
311 unsigned get_nof_symbols() const override { return max_symb; }
312
313 bool is_empty(unsigned port) const override { return entries.empty(); }
314
315 bool is_empty() const override { return entries.empty(); }
316
317 span<cf_t> get(span<cf_t> symbols, unsigned port, unsigned l, unsigned k_init, span<const bool> mask) const override
318 {
319 ++count;
320 unsigned i_symb = 0;
321 for (unsigned k = 0; k != mask.size(); ++k) {
322 if (mask[k]) {
323 symbols[i_symb] = get(static_cast<uint8_t>(port), l, k_init + k);
324 i_symb++;
325 }
326 }
327
328 // Consume buffer.
329 return symbols.last(symbols.size() - i_symb);
330 }
331
332 span<cf_t> get(span<cf_t> symbols,
333 unsigned port,
334 unsigned l,
335 unsigned k_init,
336 const bounded_bitset<MAX_RB * NRE>& mask) const override
337 {
338 ++count;
339 mask.for_each(0, mask.size(), [&](unsigned i_subc) {
340 symbols.front() = get(static_cast<uint8_t>(port), l, k_init + i_subc);
341 symbols = symbols.last(symbols.size() - 1);
342 });
343
344 // Consume buffer.
345 return symbols;
346 }
347
348 void get(span<cf_t> symbols, unsigned port, unsigned l, unsigned k_init) const override
349 {
350 ++count;
351 cf_t* symbol_ptr = symbols.data();
352 for (unsigned k = k_init, k_end = k_init + symbols.size(); k != k_end; ++k) {
353 *(symbol_ptr++) = get(port, l, k);
354 }
355 }
356
357 span<const cf_t> get_view(unsigned port, unsigned l) const override
358 {
359 srsran_assert(false, "Unimplemented method");
360 return {};
361 }
362
363 void write(span<const expected_entry_t> entries_)
364 {
365 for (const expected_entry_t& e : entries_) {
366 write(e);
367 }
368 }
369
370 void write(const expected_entry_t& entry)
371 {
372 entry_key_t key = {entry.port, entry.symbol, entry.subcarrier};
373
374 entries.emplace(key, entry.value);
375 }
376
377 unsigned get_count() const { return count; }
378
380 void reset()
381 {
382 entries.clear();
383 count = 0;
384 }
385
386private:
387 unsigned max_ports;
388 unsigned max_symb;
389 unsigned max_prb;
390 mutable unsigned count = 0;
391
393 using entry_key_t = std::tuple<uint8_t, uint8_t, uint16_t>;
394
396 std::map<entry_key_t, cf_t> entries;
397
398 cf_t get(uint8_t port, uint8_t symbol, uint16_t subcarrier) const
399 {
400 // Generate key.
401 entry_key_t key{port, symbol, subcarrier};
402
403 // Ensure the resource element exist.
404 srsran_assert(entries.count(key) == 1,
405 "Resource grid for port={}, symbol={} and subcarrier={} does not exist.",
406 port,
407 symbol,
408 subcarrier);
409
410 return entries.at(key);
411 }
