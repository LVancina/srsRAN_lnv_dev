33class prach_processor_notifier_spy : public prach_processor_notifier
34{
35private:
36 struct rx_prach_window_entry {
37 prach_buffer_context context;
38 const prach_buffer* buffer;
39 };
40
41 srslog::basic_logger& logger;
42 std::vector<prach_buffer_context> request_late_entries;
43 std::vector<prach_buffer_context> request_overflow_entries;
44 std::vector<rx_prach_window_entry> rx_prach_window_entries;
45
46public:
47 explicit prach_processor_notifier_spy(const std::string& log_level = "info") :
48 logger(srslog::fetch_basic_logger("Notifier", false))
49 {
50 srslog::init();
51 logger.set_level(srslog::str_to_basic_level(log_level));
52 }
53
54 void on_prach_request_late(const prach_buffer_context& context) override
55 {
56 logger.info("PRACH request is late. Sector/Port {}/{}. Slot/Symbol {}/{}.",
57 context.sector,
58 context.ports,
59 context.slot,
60 context.start_symbol);
61 request_late_entries.push_back(context);
62 }
63
64 void on_prach_request_overflow(const prach_buffer_context& context) override
65 {
66 logger.info("PRACH request discarded. Sector/Port {}/{}. Slot/Symbol {}/{}.",
67 context.sector,
68 context.ports,
69 context.slot,
70 context.start_symbol);
71 request_overflow_entries.push_back(context);
72 }
73
74 void on_rx_prach_window(const prach_buffer& buffer, const prach_buffer_context& context) override
75 {
76 logger.info("PRACH Rx Window Processed. Sector/Port {}/{}. Slot/Symbol {}/{}.",
77 context.sector,
78 context.ports,
79 context.slot,
80 context.start_symbol);
81 rx_prach_window_entries.emplace_back();
82 rx_prach_window_entry& entry = rx_prach_window_entries.back();
83 entry.context = context;
84 entry.buffer = &buffer;
85 }
86
87 unsigned get_nof_notifications() const
88 {
89 return request_late_entries.size() + request_overflow_entries.size() + rx_prach_window_entries.size();
90 }
91
92 void clear_notifications()
93 {
94 request_late_entries.clear();
95 request_overflow_entries.clear();
96 rx_prach_window_entries.clear();
97 }
98
99 const std::vector<prach_buffer_context>& get_request_late_entries() const { return request_late_entries; }
100
101 const std::vector<prach_buffer_context>& get_request_overflow_entries() const { return request_overflow_entries; }
102
103 const std::vector<rx_prach_window_entry>& get_rx_prach_window_entries() const { return rx_prach_window_entries; }
104};
105
106} // namespace srsran
