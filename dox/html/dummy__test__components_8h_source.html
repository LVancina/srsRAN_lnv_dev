34class dummy_pdcch_resource_allocator : public pdcch_resource_allocator
35{
36public:
37 rnti_t last_dl_pdcch_rnti;
38 pdcch_dl_information next_ue_pdcch_alloc;
39 pdcch_ul_information next_ue_ul_pdcch_alloc;
40
41 pdcch_dl_information* alloc_dl_pdcch_common(cell_slot_resource_allocator& slot_alloc,
42 rnti_t rnti,
43 search_space_id ss_id,
44 aggregation_level aggr_lvl) override
45 {
46 TESTASSERT_EQ(ss_id, slot_alloc.cfg.dl_cfg_common.init_dl_bwp.pdcch_common.ra_search_space_id);
47 slot_alloc.result.dl.dl_pdcchs.emplace_back();
48 slot_alloc.result.dl.dl_pdcchs.back().ctx.rnti = rnti;
49 slot_alloc.result.dl.dl_pdcchs.back().ctx.bwp_cfg = &slot_alloc.cfg.dl_cfg_common.init_dl_bwp.generic_params;
50 slot_alloc.result.dl.dl_pdcchs.back().ctx.coreset_cfg =
51 &*slot_alloc.cfg.dl_cfg_common.init_dl_bwp.pdcch_common.coreset0;
52 slot_alloc.result.dl.dl_pdcchs.back().ctx.cces = {get_ncce(slot_alloc.slot), srsran::aggregation_level::n4};
53 return &slot_alloc.result.dl.dl_pdcchs[0];
54 }
55
56 pdcch_dl_information* alloc_dl_pdcch_ue(cell_slot_resource_allocator& slot_alloc,
57 rnti_t rnti,
58 const ue_cell_configuration& user,
59 search_space_id ss_id,
60 aggregation_level aggr_lvl) override
61 {
62 last_dl_pdcch_rnti = rnti;
63 return &next_ue_pdcch_alloc;
64 }
65
66 pdcch_ul_information* alloc_ul_pdcch_ue(cell_slot_resource_allocator& slot_alloc,
67 rnti_t rnti,
68 const ue_cell_configuration& user,
69 search_space_id ss_id,
70 aggregation_level aggr_lvl) override
71 {
72 srsran_terminate("UE-dedicated PDCCHs not supported");
73 return nullptr;
74 }
75
76 pdcch_ul_information* alloc_ul_pdcch_common(cell_slot_resource_allocator& slot_alloc,
77 rnti_t rnti,
78 search_space_id ss_id,
79 aggregation_level aggr_lvl) override
80 {
81 return &next_ue_ul_pdcch_alloc;
82 }
83
84 bool cancel_last_pdcch(cell_slot_resource_allocator& slot_alloc) override
85 {
86 srsran_terminate("Cancellations not supported");
87 return false;
88 }
89
90private:
91 unsigned get_ncce(slot_point pdcch_sl)
92 {
93 if (not last_sl.valid() or pdcch_sl != last_sl) {
94 next_ncce = to_nof_cces(srsran::aggregation_level::n4);
95 last_sl = pdcch_sl;
96 return 0;
97 }
98 unsigned ret = next_ncce;
99 next_ncce += to_nof_cces(srsran::aggregation_level::n4);
100 return ret;
101 }
102
103 unsigned next_ncce = 0;
104 slot_point last_sl;
105};
