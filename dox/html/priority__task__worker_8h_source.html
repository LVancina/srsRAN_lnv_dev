44template <concurrent_queue_policy... QueuePolicies>
45class priority_task_worker
46{
47public:
54 priority_task_worker(std::string thread_name,
55 const std::array<unsigned, sizeof...(QueuePolicies)>& task_queue_sizes,
56 std::chrono::microseconds wait_interval,
57 os_thread_realtime_priority prio = os_thread_realtime_priority::no_realtime(),
58 const os_sched_affinity_bitmask& mask = {}) :
59 task_queue(task_queue_sizes, wait_interval), t_handle(thread_name, prio, mask, [this]() { run_pop_task_loop(); })
60 {
61 }
62 ~priority_task_worker() { stop(); }
63
65 void stop()
66 {
67 if (t_handle.running()) {
68 task_queue.request_stop();
69 t_handle.join();
70 }
71 }
72
74 template <task_priority Priority>
75 SRSRAN_NODISCARD bool push_task(unique_task task)
76 {
77 return task_queue.template try_push<Priority>(std::move(task));
78 }
79
81 template <task_priority Priority>
82 SRSRAN_NODISCARD size_t queue_capacity() const
83 {
84 return task_queue.template capacity<Priority>();
85 }
86
88 std::thread::id get_id() const { return t_handle.get_id(); }
89
91 const char* worker_name() const { return t_handle.get_name(); }
92
94 static constexpr size_t nof_priority_levels() { return sizeof...(QueuePolicies); }
95
97 template <task_priority Priority>
98 auto get_enqueuer()
99 {
100 return task_queue.template get_enqueuer<Priority>();
101 }
102
103private:
104 void run_pop_task_loop()
105 {
106 while (task_queue.call_on_pop_blocking([](unique_task& task) { task(); })) {
107 }
108 srslog::fetch_basic_logger("ALL").info("Task worker \"{}\" finished.", this_thread_name());
109 }
110
111 // Task queues with different priorities. The first queue is the highest priority queue.
112 concurrent_priority_queue<unique_task, QueuePolicies...> task_queue;
113
114 // Worker thread
115 unique_thread t_handle;
116};

123template <concurrent_queue_policy QueuePolicy>
124class priority_task_worker_executor : public task_executor
125{
126public:
127 priority_task_worker_executor(priority_enqueuer<unique_task, QueuePolicy> enqueuer_,
128 task_priority prio_,
129 std::thread::id tid_,
130 const char* worker_name_) :
131 enqueuer(std::move(enqueuer_)), prio(prio_), worker_tid(tid_), worker_name(worker_name_)
132 {
133 }
134
135 SRSRAN_NODISCARD bool execute(unique_task task) override
136 {
137 if (can_run_task_inline()) {
138 // If same thread and highest priority task, run task right away.
139 task();
140 return true;
141 }
142 return defer(std::move(task));
143 }
144
145 SRSRAN_NODISCARD bool defer(unique_task task) override { return enqueuer.try_push(std::move(task)); }
146
147 // Check whether task can be run inline or it needs to be dispatched to a queue.
148 bool can_run_task_inline() const { return prio == task_priority::max and worker_tid == std::this_thread::get_id(); }
149
150private:
152 priority_enqueuer<unique_task, QueuePolicy> enqueuer;
153 task_priority prio;
154 std::thread::id worker_tid;
155 const char* worker_name;
156};

176template <typename Func, concurrent_queue_policy... QueuePolicies, size_t... Is>
177void visit_executor(priority_task_worker<QueuePolicies...>& w,
178 task_priority priority,
179 const Func& func,
180 std::index_sequence<Is...> /*unused*/)
181{
182 const size_t idx = detail::enqueue_priority_to_queue_index(priority, sizeof...(QueuePolicies));
183 (void)std::initializer_list<int>{[idx, &w, &func]() {
184 if (idx == Is) {
185 func(
186 make_priority_task_worker_executor<detail::queue_index_to_enqueue_priority(Is, sizeof...(QueuePolicies))>(w));
187 }
188 return 0;
189 }()...};
190}
191
192} // namespace detail

195template <typename Func, concurrent_queue_policy... QueuePolicies>
196void visit_executor(priority_task_worker<QueuePolicies...>& worker, task_priority priority, const Func& func)
197{
198 detail::visit_executor(worker, priority, func, std::make_index_sequence<sizeof...(QueuePolicies)>{});
199}

202template <concurrent_queue_policy... QueuePolicies>
203std::unique_ptr<task_executor> make_priority_task_executor_ptr(priority_task_worker<QueuePolicies...>& worker,
204 task_priority priority)
205{
206 std::unique_ptr<task_executor> exec;
207 detail::visit_executor(
208 worker,
209 priority,
210 [&exec](const auto& e) { exec = std::make_unique<decltype(e)>(e); },
211 std::make_index_sequence<sizeof...(QueuePolicies)>{});
212 return exec;
213}
