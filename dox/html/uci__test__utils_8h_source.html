37// Helper function to create a sched_cell_configuration_request_message that allows a configuration with either 15kHz or
38// 30kHz SCS. By default, it creates a bandwidth of 20MHz.
39inline sched_cell_configuration_request_message
40make_default_sched_cell_configuration_request_scs(subcarrier_spacing scs, bool tdd_mode = false)
41{
42 cell_config_builder_params params{
43 .scs_common = scs, .channel_bw_mhz = bs_channel_bandwidth_fr1::MHz20, .nof_dl_ports = 1};
44 if (scs == subcarrier_spacing::kHz15) {
45 // Band n5 for FDD, band n41 for TDD.
46 params.dl_arfcn = tdd_mode ? 499200 : 530000;
47 params.band = band_helper::get_band_from_dl_arfcn(params.dl_arfcn);
48 } else if (scs == subcarrier_spacing::kHz30) {
49 // Band n5 for FDD, band n77 or n78 for TDD.
50 params.dl_arfcn = tdd_mode ? 630000 : 176000;
51 params.band = band_helper::get_band_from_dl_arfcn(params.dl_arfcn);
52 }
53 return sched_cell_configuration_request_message{test_helpers::make_default_sched_cell_configuration_request(params)};
54}

82inline sched_cell_configuration_request_message make_custom_sched_cell_configuration_request(unsigned pucch_res_common,
83 bool is_tdd = false)
84{
85 sched_cell_configuration_request_message req = test_helpers::make_default_sched_cell_configuration_request(
86 cell_config_builder_params{.scs_common = is_tdd ? subcarrier_spacing::kHz30 : subcarrier_spacing::kHz15,
87 .channel_bw_mhz = bs_channel_bandwidth_fr1::MHz10,
88 .dl_arfcn = is_tdd ? 520000U : 365000U});
89 req.ul_cfg_common.init_ul_bwp.pucch_cfg_common->pucch_resource_common = pucch_res_common;
90 return req;
91}
