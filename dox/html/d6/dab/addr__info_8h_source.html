27inline const char* ipproto_to_string(int protocol)
28{
29 switch (protocol) {
30 case IPPROTO_SCTP:
31 return "SCTP";
32 case IPPROTO_UDP:
33 return "UDP";
34 case IPPROTO_IP:
35 return "IP";
36 case IPPROTO_IPV6:
37 return "IPV6";
38 case IPPROTO_ICMP:
39 return "ICMP";
40 case IPPROTO_TCP:
41 return "TCP";
42 default:
43 break;
44 }
45 return "unknown protocol";
46}
