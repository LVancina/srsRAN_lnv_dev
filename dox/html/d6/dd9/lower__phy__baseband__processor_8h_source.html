101private:
102 /// Internal finite state machine to control the internal state.
103 class internal_fsm
104 {
105 public:
106 /// Default destructor - It reports a fatal error if the state is \c running or \c wait_stop.
107 ~internal_fsm()
108 {
109 report_fatal_error_if_not((state != states::running) && (state != states::wait_stop), "Unexpected state.");
110 }
111
112 /// \brief Notifies the start of the processing.
113 /// \remark A fatal error is reported if start() is called while the processor is not in \c idle state.
114 void start()
115 {
116 states previous_state = state.exchange(states::running);
117 report_fatal_error_if_not(previous_state == states::idle, "Unexpected state.");
118 }
119
120 /// \brief Notifies that the asynchronous execution has stopped.
121 ///
122 /// Changes the state to \c stopped and sends the corresponding notification.
123 /// \remark A fatal error is reported if notify_stop() is called while the processor is not in \c wait_stop.
124 void notify_stop()
125 {
126 states previous_state = state.exchange(states::stopped);
127 report_fatal_error_if_not(previous_state == states::wait_stop, "Unexpected state.");
128 }
129
130 /// \brief Requests all asynchronous processing to stop.
131 /// \remark A fatal error is reported if request_stop() is called more than once.
132 void request_stop()
133 {
134 states previous_state = state.exchange(states::wait_stop);
135
136 report_fatal_error_if_not((previous_state == states::running) || (previous_state == states::idle),
137 "Unexpected state.");
138
139 // Transition to stop state directly if idle.
140 if (previous_state == states::idle) {
141 state = states::stopped;
142 }
143 }
144
145 /// \brief Waits for all asynchronous processing to stop.
146 /// \remark A fatal error is reported if wait_stop() is called while it is \c idle or \c running without calling
147 /// first request_stop().
148 void wait_stop()
149 {
150 report_fatal_error_if_not((state == states::wait_stop) || (state == states::stopped), "Unexpected state.");
151
152 while (state == states::wait_stop) {
153 std::this_thread::sleep_for(std::chrono::microseconds(10));
154 }
155 }
156
157 /// Returns true if the asynchronous execution is running.
158 bool is_running() const { return state == states::running; }
159
160 private:
161 /// Possible states.
162 enum class states {
163 /// The asynchronous processing has not been started.
164 idle = 0,
165 /// The asynchronous processing has started and not notify_stop.
166 running,
167 /// A stop command has been issued - waiting for the asynchronous tasks to finish.
168 wait_stop,
169 /// The asynchronous processing started, ran and it is currently notify_stop.
170 stopped
171 };
172
173 /// Actual state.
174 std::atomic<states> state{states::idle};
175 };
