41/// \brief Uplink processor interface.
42///
43/// The uplink processor is in charge of handling incoming requests to process the physical uplink channels within
44/// a certain slot.
45///
46/// Requests are dispatched asynchronously as they get enqueued for execution, and generate an event through the \ref
47/// upper_phy_rx_results_notifier interface upon completion.
48class uplink_processor
49{
50public:
51 /// PUSCH PDU configuration.
52 struct pusch_pdu {
53 /// HARQ process number.
54 unsigned harq_id;
55 /// Transport block size.
56 units::bytes tb_size;
57 /// PUSCH processor PDU.
58 pusch_processor::pdu_t pdu;
59 };
60
61 /// PUCCH PDU configuration.
62 struct pucch_pdu {
63 /// PUCCH context;
64 ul_pucch_context context;
65 // :TODO: the formats will use a variant when available.
66 /// PUCCH format 0.
67 pucch_processor::format0_configuration format0;
68 /// PUCCH format 1.
69 pucch_processor::format1_configuration format1;
70 /// PUCCH format 2.
71 pucch_processor::format2_configuration format2;
72 /// PUCCH format 3.
73 pucch_processor::format3_configuration format3;
74 /// PUCCH format 4.
75 pucch_processor::format4_configuration format4;
76 };
77
78 virtual ~uplink_processor() = default;
79
80 /// \brief Processes the PRACH using the given configuration and context.
81 ///
82 /// The PRACH detection results will be notified by the upper_phy_rx_results_notifier with event \ref
83 /// upper_phy_rx_results_notifier::on_new_prach_results "on_new_prach_results".
84 ///
85 /// \param[in] notifier Event notification interface.
86 /// \param[in] buffer Channel symbols the PRACH detection is performed on.
87 /// \param[in] context Context used by the underlying PRACH detector.
88 virtual void process_prach(upper_phy_rx_results_notifier& notifier,
89 const prach_buffer& buffer,
90 const prach_buffer_context& context) = 0;
91
92 /// \brief Processes a PUSCH transmission.
93 ///
94 /// The size of each transport block is determined by <tt>data[TB index].size()</tt>.
95 ///
96 /// The PUSCH results will be notified by the upper_phy_rx_results_notifier with event \ref
97 /// upper_phy_rx_results_notifier::on_new_pusch_results "on_new_pusch_results".
98 ///
99 /// \param[out] data Received transport block.
100 /// \param[in,out] rm_buffer Rate matcher buffer.
101 /// \param[in] notifier Event notification interface.
102 /// \param[in] grid Grid the capture data is stored in.
103 /// \param[in] pdu PUSCH transmission parameters.
104 virtual void process_pusch(span<uint8_t> data,
105 unique_rx_buffer rm_buffer,
106 upper_phy_rx_results_notifier& notifier,
107 const resource_grid_reader& grid,
108 const uplink_processor::pusch_pdu& pdu) = 0;
109
110 /// \brief Processes a PUCCH transmission.
111 ///
112 /// The PUCCH results will be notified by the upper_phy_rx_results_notifier with event \ref
113 /// upper_phy_rx_results_notifier::on_new_pucch_results "on_new_pucch_results".
114 ///
115 /// \param[in] notifier Event notification interface.
116 /// \param[in] grid Resource grid.
117 /// \param[in] pdu PUCCH transmission parameters.
118 virtual void
119 process_pucch(upper_phy_rx_results_notifier& notifier, const resource_grid_reader& grid, const pucch_pdu& pdu) = 0;
120};
