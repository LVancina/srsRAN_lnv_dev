121// cause_t formatter
122template <>
123struct formatter<srsran::cause_t> {
124 template <typename ParseContext>
125 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
126 {
127 return ctx.begin();
128 }
129
130 template <typename FormatContext>
131 auto format(srsran::cause_t o, FormatContext& ctx) -> decltype(std::declval<FormatContext>().out())
132 {
133 if (srsran::variant_holds_alternative<srsran::cause_radio_network_t>(o)) {
134 return format_to(ctx.out(), "radio_network-id{}", srsran::variant_get<srsran::cause_radio_network_t>(o));
135 } else if (srsran::variant_holds_alternative<srsran::cause_transport_t>(o)) {
136 return format_to(ctx.out(), "transport-id{}", srsran::variant_get<srsran::cause_transport_t>(o));
137 } else if (srsran::variant_holds_alternative<srsran::cause_nas_t>(o)) {
138 return format_to(ctx.out(), "nas-id{}", srsran::variant_get<srsran::cause_nas_t>(o));
139 } else if (srsran::variant_holds_alternative<srsran::cause_protocol_t>(o)) {
140 return format_to(ctx.out(), "protocol-id{}", srsran::variant_get<srsran::cause_protocol_t>(o));
141 } else {
142 return format_to(ctx.out(), "misc-id{}", srsran::variant_get<srsran::cause_misc_t>(o));
143 }
144 }
145};
