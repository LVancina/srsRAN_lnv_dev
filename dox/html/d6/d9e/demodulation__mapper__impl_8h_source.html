32/// \brief Approximate-LLR demodulator.
33///
34/// Implementation of the interface provided by demodulation_mapper. Instead of the defining expression, the
35/// log-likelihood ratios are computed using the widely accepted approximation
36/// \f[
37/// \ell(b_{m+i}) = -\frac{1}{\sigma_m^2} \biggl( \min_{s\in \mathcal{S}_{i,0}} \abs{y_m - s}^2
38/// - \min_{s\in \mathcal{S}_{i,1}} \abs{y_m - s}^2 \biggr)
39/// \f]
40/// where \f$y_m\f$ denotes the noisy modulation symbol, \f$\sigma_m^2\f$ is the equivalent (after equalization) noise
41/// variance and \f$\mathcal{S}_{i,B}\f$ is the set of constellation points for which \f$b_{m+i}=B\in\{0,1\}\f$.
42class demodulation_mapper_impl : public demodulation_mapper
43{
44public:
45 // See interface for the documentation.
46 void demodulate_soft(span<log_likelihood_ratio> llrs,
47 span<const cf_t> symbols,
48 span<const float> noise_vars,
49 modulation_scheme mod) override;
50};
