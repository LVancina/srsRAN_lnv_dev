34/// Class used for transmitting GTP-U bearers.
35class gtpu_tunnel_base_tx
36{
37public:
38 gtpu_tunnel_base_tx(gtpu_tunnel_log_prefix log_prefix,
39 dlt_pcap& gtpu_pcap_,
40 gtpu_tunnel_tx_upper_layer_notifier& upper_dn_) :
41 logger("GTPU", log_prefix), gtpu_pcap(gtpu_pcap_), upper_dn(upper_dn_)
42 {
43 }
44
45 bool to_sockaddr(sockaddr_storage& out_sockaddr, const char* addr, uint16_t port)
46 {
47 // Validate configuration
48 if (inet_pton(AF_INET, addr, &((::sockaddr_in*)&out_sockaddr)->sin_addr) == 1) {
49 ((::sockaddr_in*)&out_sockaddr)->sin_family = AF_INET;
50 ((::sockaddr_in*)&out_sockaddr)->sin_port = htons(port);
51 } else if (inet_pton(AF_INET6, addr, &((::sockaddr_in6*)&out_sockaddr)->sin6_addr) == 1) {
52 ((::sockaddr_in6*)&out_sockaddr)->sin6_family = AF_INET6;
53 ((::sockaddr_in6*)&out_sockaddr)->sin6_port = htons(port);
54 } else {
55 logger.log_error("Invalid address or port. addr={} port={} errno={}", addr, port, strerror(errno));
56 return false;
57 }
58 return true;
59 }
60
61protected:
62 gtpu_tunnel_logger logger;
63
64 void send_pdu(byte_buffer buf, const ::sockaddr_storage& peer_sockaddr)
65 {
66 if (gtpu_pcap.is_write_enabled()) {
67 gtpu_pcap.push_pdu(buf.deep_copy());
68 }
69
70 upper_dn.on_new_pdu(std::move(buf), peer_sockaddr);
71 }
72
73private:
74 dlt_pcap& gtpu_pcap;
75 gtpu_tunnel_tx_upper_layer_notifier& upper_dn;
76};
77} // namespace srsran
