35/// Implements a gateway receiver based on UHD receive stream.
36class radio_uhd_rx_stream : public uhd_exception_handler, public baseband_gateway_receiver
37{
38private:
39 /// Receive timeout in seconds.
40 static constexpr double RECEIVE_TIMEOUT_S = 0.2f;
41 /// Set to true for receiving data in a single packet.
42 static constexpr bool ONE_PACKET = false;
43
44 /// Defines the Rx stream internal states.
45 enum class states { UNINITIALIZED, SUCCESSFUL_INIT, STREAMING, STOP };
46 /// Indicates the current stream state.
47 std::atomic<states> state = {states::UNINITIALIZED};
48 /// Indicates the stream identification for notifications.
49 unsigned id;
50 /// Sampling rate in hertz.
51 double srate_Hz;
52 /// Radio notification interface.
53 radio_notification_handler& notifier;
54 /// Owns the UHD Tx stream.
55 uhd::rx_streamer::sptr stream;
56 /// Maximum number of samples in a single packet.
57 unsigned max_packet_size;
58 /// Indicates the number of channels.
59 unsigned nof_channels;
60 /// Protects stream from concurrent receive and stop.
61 std::mutex stream_mutex;
62
63 /// \brief Receives a single block of baseband samples.
64 /// \param[out] nof_rxd_samples Indicate the number of samples received in the block.
65 /// \param[in,out] buffs Provides the reception buffers.
66 /// \param[in] buffer_offset Indicates the data offset in the reception buffers.
67 /// \param[in] metadata Provides the reception metadata.
68 /// \return True if no exception is caught. Otherwise false.
69 bool receive_block(unsigned& nof_rxd_samples,
70 baseband_gateway_buffer_writer& buffs,
71 unsigned buffer_offset,
72 uhd::rx_metadata_t& metadata);
73
74public:
75 /// Describes the necessary parameters to create an UHD transmit stream.
76 struct stream_description {
77 /// Identifies the stream.
78 unsigned id;
79 /// Sampling rate in hertz.
80 double srate_Hz;
81 /// Over-the-wire format.
82 radio_configuration::over_the_wire_format otw_format;
83 /// Stream arguments.
84 std::string args;
85 /// Indicates the port indexes for the stream.
86 std::vector<size_t> ports;
87 };
88
89 /// \brief Constructs a receive UHD stream.
90 /// \param[in] usrp Provides the USRP context.
91 /// \param[in] description Provides the stream configuration parameters.
92 /// \param[in] notifier_ Provides the radio event notification handler.
93 radio_uhd_rx_stream(uhd::usrp::multi_usrp::sptr& usrp,
94 const stream_description& description,
95 radio_notification_handler& notifier_);
96
97 /// \brief Starts the stream reception.
98 /// \param[in] time_spec Indicates the start time of the stream.
99 /// \return True if no exception is caught. Otherwise false.
100 bool start(const uhd::time_spec_t& time_spec);
101
102 /// Gets the optimal transmitter buffer size.
103 unsigned get_buffer_size() const;
104
105 // See interface for documentation.
106 metadata receive(baseband_gateway_buffer_writer& data) override;
107
108 /// \brief Stops the reception stream.
109 /// \return True if no exception is caught. Otherwise false.
110 bool stop();
111
112 /// Wait until radio is notify_stop.
113 void wait_stop();
114};
115} // namespace srsran
