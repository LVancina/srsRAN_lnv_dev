31/// Number of subcarriers within a SSB block, as per TS 38.211, Section 7.4.3.1.
32const unsigned NOF_SSB_SUBCARRIERS = NOF_SSB_PRBS * NOF_SUBCARRIERS_PER_RB;
33
34/// Contains the parameters defining the SSB position within the band; returned by the DU config generator.
35struct ssb_freq_location {
36 /// Tells whether the set of parameters represent a valid configuration.
37 bool is_valid;
38 /// <em>offsetToPointA<\em>, as per Section 4.4.4.2, TS 38.211.
39 ssb_offset_to_pointA offset_to_point_A;
40 /// \f$k_{SSB}\f$, as per Section 7.4.3.1, TS 38.211.
41 ssb_subcarrier_offset k_ssb;
42 /// SSB central frequency, or SS_ref (as per TS 38.104, Section 5.4.3.1.), expressed in Hz.
43 double ss_ref;
44};
45
46/// Class that implements utilities that compute the SSB position within the band.
47class ssb_freq_position_generator
48{
49public:
50 explicit ssb_freq_position_generator(unsigned dl_arfcn,
51 nr_band nr_band_,
52 unsigned n_rbs_,
53 subcarrier_spacing scs_common,
54 subcarrier_spacing scs_ssb);
55
56 /// Return the next viable position of the SSB on the raster, so that the SSB falls within the band CRBs and the SSB
57 /// subcarriers are aligned with those of the CRBs.
58 ssb_freq_location get_next_ssb_location();
59 unsigned get_ssb_first_symbol() const { return ssb_first_symbol; };
60 unsigned get_max_ss0_idx() const { return max_ss0_idx; };
61
62private:
63 /// Finds the \f$M\f$ parameter of the sync raster so that the SSB subcarriers are aligned to those of the CRBs.
64 unsigned find_M_raster();
65 /// Get the SSB central frequency, or SS_ref, given the parameters N, M, as per Table 5.4.3.1-1, TS 38.104.
66 double get_ss_ref_hz(unsigned N, unsigned M) const;
67 /// Increase N_raster according to the raster step given by the band, as per Table 5.4.3.3-1, TS 38.104, ver 17.8.0.
68 void increase_N_raster();
69 /// Implements \ref get_next_ssb_location() for the specific case of bands n46, n96 and n102, which don't have a
70 /// standard raster.
71 ssb_freq_location get_next_ssb_location_special_raster();
72
73 /// Parameters that are passed to the constructor.
74 const unsigned dl_arfcn;
75 const nr_band band;
76 /// Transmission bandwidth configuration \f$N_{RB}\f$, as per Table 5.3.2-1, TS 38.104.
77 const unsigned n_rbs;
78 const subcarrier_spacing scs_common;
79 const subcarrier_spacing scs_ssb;
80
81 /// Point A, as per TS 38.211, Section 4.4.4.2, expressed in Hz.
82 double point_A_hz;
83 /// RF reference frequency (a.k.a. centre frequency), as per TS 38.104, Section 5.4.2.1.
84 double f_ref_hz;
85
86 /// Band upper-bound expressed in Hz. We consider the last subcarrier of the last CRB as band upper-bound.
87 double bw_ub_hz;
88 /// SSB case, as per TS 38.213, Section 4.1.
89 ssb_pattern_case ssb_case;
90 /// First symbol where the 1st SSB within the burst gets allocated, as per TS 38.211, Section 4.1.
91 unsigned ssb_first_symbol;
92 /// Maximum value of row index of Table 13-11, TS 38.213 that can be addressed for a specific configuration.
93 uint8_t max_ss0_idx;
94
95 /// This is the \f$N\f$ parameter in the synchronization raster, as per Table 5.4.3.1-1, TS 38.104.
96 unsigned N_raster;
97 /// This is the \f$M\f$ parameter in the synchronization raster, as per Table 5.4.3.1-1, TS 38.104.
98 unsigned M_raster;
99 /// Some bands have a non-regular set of GSCN values for the synchronization raster. This index keeps track of what
100 /// GSCN value has been used to generate the latest SSB frequency location.
101 size_t gscn_raster_idx;
102};
