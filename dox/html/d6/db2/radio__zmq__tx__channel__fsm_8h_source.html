25#include "radio_zmq_timer.h"
26#include "srsran/support/srsran_assert.h"
27#include <condition_variable>
28#include <mutex>
29
30namespace srsran {
31class radio_zmq_tx_channel_fsm
32{
33private:
34 /// Describes the possible states.
35 enum class states {
36 /// Indicates it failed to initialize.
37 UNINITIALIZED = 0,
38 /// The channel has been successfully initialised and it is waiting to start.
39 WAIT_START,
40 /// The channel waits for a request.
41 WAIT_REQUEST,
42 /// The channel waits for data to transmit.
43 WAIT_DATA,
44 /// Waits for the asynchronous thread to notify a stop.
45 WAIT_STOP,
46 /// Indicates it is notify_stop.
47 STOPPED
48 };
49
50 /// Indicates the current state.
51 states state = states::UNINITIALIZED;
52 /// Protects the state and transitions.
53 mutable std::mutex mutex;
54 /// Condition variable to wait for certain states.
55 std::condition_variable cvar;
56 /// State waiting timer.
57 radio_zmq_timer timer;
58
59 /// Same than is_running().
60 bool is_running_unprotected() const { return state == states::WAIT_REQUEST || state == states::WAIT_DATA; }
61
62public:
63 /// Notifies that the transmit stream has been initialized successfully.
64 void init_successful()
65 {
66 std::unique_lock<std::mutex> lock(mutex);
67 srsran_assert(state == states::UNINITIALIZED, "Invalid state.");
68 state = states::WAIT_START;
69 }
70
71 /// Notifies the asynchronous processing has started.
72 void on_start()
73 {
74 std::unique_lock<std::mutex> lock(mutex);
75 srsran_assert(state == states::WAIT_START, "Invalid state.");
76 state = states::WAIT_REQUEST;
77 timer.start();
78 }
79
80 /// Notifies that the stream detected an error it cannot recover from.
81 void on_error()
82 {
83 std::unique_lock<std::mutex> lock(mutex);
84 state = states::STOPPED;
85 cvar.notify_all();
86 }
87
88 /// Notifies that a request was received.
89 void request_received()
90 {
91 std::unique_lock<std::mutex> lock(mutex);
92 if (state == states::WAIT_REQUEST) {
93 state = states::WAIT_DATA;
94 timer.start();
95 }
96 }
97
98 /// Notifies a block of data was send.
99 void data_sent()
100 {
101 std::unique_lock<std::mutex> lock(mutex);
102 if (state == states::WAIT_DATA) {
103 state = states::WAIT_REQUEST;
104 timer.start();
105 }
106 }
107
108 /// Notifies a stop was signaled and wait for asynchronous executor to stop.
109 void stop()
110 {
111 std::unique_lock<std::mutex> lock(mutex);
112 if (is_running_unprotected()) {
113 state = states::WAIT_STOP;
114 }
115 }
116
117 /// Waits for the asynchronous tasks notifies that has notify_stop.
118 void wait_stop()
119 {
120 std::unique_lock<std::mutex> lock(mutex);
121 while (state != states::STOPPED) {
122 cvar.wait(lock);
123 }
124 }
125
126 /// Notifies the asynchronous task has notify_stop.
127 void async_task_stopped()
128 {
129 std::unique_lock<std::mutex> lock(mutex);
130 state = states::STOPPED;
131 cvar.notify_all();
132 }
133
134 /// Gets if the channel has been successfully initialised.
135 bool is_initiated() const
136 {
137 std::unique_lock<std::mutex> lock(mutex);
138 return state > states::UNINITIALIZED;
139 }
140
141 /// Indicates whether the current state is operational.
142 bool is_running() const
143 {
144 std::unique_lock<std::mutex> lock(mutex);
145 return is_running_unprotected();
146 }
147
148 /// Indicates whether it has a pending request.
149 bool has_pending_request() const
150 {
151 std::unique_lock<std::mutex> lock(mutex);
152 return state == states::WAIT_DATA;
153 }
154
155 /// \brief Checks if the waiting time for request or data has expired.
156 ///
157 /// The request and data wait expires only if:
158 /// - the current state is waiting for a request; or
159 /// - the current state is waiting for data.
160 ///
161 /// The expiration occurs when a certain time had past since:
162 /// - the last state transition from waiting data to waiting for request; or
163 /// - last time an expiration was detected.
164 bool has_wait_expired()
165 {
166 std::unique_lock<std::mutex> lock(mutex);
167
168 // It can only be expired if it is currently running.
169 if (!is_running_unprotected()) {
170 return false;
171 }
172
173 // Check if the timer has not expired.
174 if (!timer.is_expired()) {
175 return false;
176 }
177
178 // Update timer with new time.
179 timer.start();
180 return true;
181 }
182};
183} // namespace srsran
