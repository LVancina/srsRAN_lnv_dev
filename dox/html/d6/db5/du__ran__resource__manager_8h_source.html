31/// \brief Outcome report of an DU UE Resource allocation request.
32struct du_ue_resource_update_response {
33 /// \brief Defines whether the UE release is required due to an error during the update procedure.
34 /// If \c procedure_error doesn't contain any error string, then the UE resource update was successful.
35 error_type<std::string> procedure_error = {};
36 std::vector<srb_id_t> failed_srbs;
37 std::vector<drb_id_t> failed_drbs;
38 std::vector<serv_cell_index_t> failed_scells;
39
40 bool release_required() const { return procedure_error.is_error(); }
41};

43/// \brief This class manages the PHY (e.g. RB and symbols used for PUCCH), MAC (e.g. LCIDs) and RLC resources used
44/// by an UE. It provides an API to update the UE resources on arrival of new UE Context Update Requests, and
45/// returns resources back to the DU RAN Resource Manager when the UE is destroyed.
46class ue_ran_resource_configurator
47{
48public:
49 /// \brief Interface used to update the UE Resources on Reconfiguration and return the resources back to the pool,
50 /// on UE deletion.
51 struct resource_updater {
52 virtual ~resource_updater() = default;
53 virtual du_ue_resource_update_response update(du_cell_index_t pcell_index,
54 const f1ap_ue_context_update_request& upd_req) = 0;
55 virtual const cell_group_config& get() = 0;
56 };
57
58 explicit ue_ran_resource_configurator(std::unique_ptr<resource_updater> ue_res_, std::string error = {}) :
59 ue_res_impl(std::move(ue_res_)),
60 cached_res(ue_res_impl != nullptr ? &ue_res_impl->get() : nullptr),
61 configurator_error(ue_res_impl != nullptr ? std::string{} : error)
62 {
63 }
64
65 /// \brief Initiates the update of the resources (PCell, SCells, Bearers) used by the UE.
66 ///
67 /// \param pcell_index DU Cell Index of the UE's PCell.
68 /// \param upd_req UE Context Update Request for a given UE.
69 /// \return Outcome of the configuration.
70 du_ue_resource_update_response update(du_cell_index_t pcell_index, const f1ap_ue_context_update_request& upd_req)
71 {
72 return ue_res_impl->update(pcell_index, upd_req);
73 }
74
75 /// \brief Checks whether the UE resources have been correctly allocated.
76 bool empty() const { return ue_res_impl == nullptr; }
77
78 /// \brief Returns the configurator error, which non-empty string only if the procedure failed.
79 std::string get_error() const { return empty() ? configurator_error : std::string{}; }
80
81 const cell_group_config& value() const { return *cached_res; }
82 const cell_group_config& operator*() const { return *cached_res; }
83 const cell_group_config* operator->() const { return cached_res; }
84
85private:
86 std::unique_ptr<resource_updater> ue_res_impl;
87 const cell_group_config* cached_res;
88 std::string configurator_error;
89};
