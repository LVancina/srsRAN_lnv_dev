30/// \brief Calculates the ratio between the Physical Shared Channel EPRE and the DM-RS EPRE.
31///
32/// The calculation is common for PDSCH and PUSCH, as defined in TS38.214 Table 4.1-1 and Table 6.2.2-1.
33///
34/// \param[in] nof_cdm_groups_without_data Number of CDM groups without data.
35/// \return Parameter \f$\beta_{\textup{DMRS}}\f$ in decibels.
36inline float get_sch_to_dmrs_ratio_dB(unsigned nof_cdm_groups_without_data)
37{
38 srsran_assert(nof_cdm_groups_without_data > 0 && nof_cdm_groups_without_data < 4,
39 "Invalid number of DMRS CDM groups without data.");
40 static constexpr std::array<float, 4> beta_dmrs_values = {NAN, 0, -3, -4.77};
41 return beta_dmrs_values[nof_cdm_groups_without_data];
42}
