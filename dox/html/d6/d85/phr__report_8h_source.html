33/// \brief PH value expressed in dB interval as defined in Table 10.1.17.1-1 of TS 38.133.
34using ph_db_range = interval<int>;
35
36/// \brief P_CMAX,f,c value expressed in dBm interval as defined in Table 10.1.18.1-1 of TS 38.133.
37/// \remark P_CMAX,f,c is the UE configured maximum output power used in computation of Power Headroom level. See
38/// TS 38.321, 6.1.3.8 and TS 38.213, 7.1.1.
39using p_cmax_dbm_range = interval<int>;
40
41/// \brief Type of PH field reported by UE.
42/// \remark See TS 38.321, 6.1.3.8 and 6.1.3.9.
43enum class ph_field_type_t { type1, type2, type3 };
44
45/// UL Power Headroom Report (PHR) of a cell.
46struct cell_ph_report {
47 /// Serving cell ID of the cell for which PH is reported. du_cell_index_t == 0 for SE-PHR.
48 du_cell_index_t serv_cell_id;
49 /// PH type. For SE-PHR its type1.
50 ph_field_type_t ph_type;
51 /// Indicates the power headroom level in dB interval.
52 ph_db_range ph;
53 /// UE configured maximum output power used in computation of Power Headroom level. This field maps to P_CMAX,f,c in
54 /// table 6.1.3.8-2 of TS 38.321. This field is optional only in case of Multiple Entry PHR.
55 optional<p_cmax_dbm_range> p_cmax;
56};
57
58/// UL Power Headroom Report (PHR).
59struct phr_report {
60 /// \brief Returns Single Entry PHR report.
61 cell_ph_report get_se_phr() const
62 {
63 srsran_assert(not ph_reports.empty(), "Invalid access to SE-PHR report. Cause: No PHR report set.");
64 return ph_reports.back();
65 }
66
67 /// \brief Sets Single-Entry PHR report.
68 void set_se_phr(const cell_ph_report& cell_phr)
69 {
70 srsran_assert(cell_phr.serv_cell_id == to_du_cell_index(0), "Invalid serving cell id set in SE-PHR.");
71 srsran_assert(cell_phr.ph_type == ph_field_type_t::type1, "Invalid PH type set in SE-PHR.");
72 srsran_assert(cell_phr.p_cmax.has_value(), "P_CMAX must be set for SE-PHR.");
73 ph_reports.clear();
74 ph_reports.push_back(cell_phr);
75 }
76
77 /// \brief Returns whether the PHR is of type Single Entry PHR or not.
78 /// \remark See TS 38.321, 6.1.3.8. A Single Entry PHR consists of a single PH report for PCell with Type1 PH field.
79 bool is_se_phr() const
80 {
81 srsran_assert(ph_reports.size() > 0, "PHR report not set.");
82 return ph_reports.size() == 1 and ph_reports.back().ph_type == ph_field_type_t::type1;
83 }
84
85 // TODO: Handle Multiple Entry PHR.
86
87 /// \brief Returns PHR as list of PH report per cell.
88 /// \remark Applies for both Single Entry PHR and multiple Entry PHR.
89 span<const cell_ph_report> get_phr() const { return ph_reports; }
90
91private:
92 /// List containing PH reports. Only 1 report present in case of Single Entry PHR and can be more than 1 in case of
93 /// Multiple Entry PHR.
94 static_vector<cell_ph_report, MAX_NOF_DU_CELLS> ph_reports;
95};
