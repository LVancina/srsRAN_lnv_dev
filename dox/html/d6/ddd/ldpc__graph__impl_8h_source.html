33/// Placeholder for invalid lifting-size values.
34static constexpr uint8_t VOID_LIFTSIZE = 255;
35/// Placeholder for missing edges in the Tanner graph.
36static constexpr uint16_t NO_EDGE = 0xffff;
37
38/// Number of parity-check nodes in base graph BG1.
39static constexpr unsigned BG1_M = 46;
40/// Number of variable nodes in base graph BG1, before shortening.
41static constexpr unsigned BG1_N_FULL = 68;
42/// Number of variable nodes in base graph BG1, after shortening.
43static constexpr unsigned BG1_N_SHORT = 66;
44/// Number of parity-check nodes in base graph BG2.
45static constexpr unsigned BG2_M = 42;
46/// Number of variable nodes in base graph BG2, before shortening.
47static constexpr unsigned BG2_N_FULL = 52;
48/// Number of variable nodes in base graph BG2, after shortening.
49static constexpr unsigned BG2_N_SHORT = 50;
50/// Maximum number of variable nodes in a base graph.
51static constexpr unsigned MAX_BG_N_FULL = std::max(BG1_N_FULL, BG2_N_FULL);
52/// Maximum number of parity check nodes in a base graph.
53static constexpr unsigned MAX_BG_M = std::max(BG1_M, BG2_M);
54/// Maximum lifting size.
55static constexpr unsigned MAX_LIFTING_SIZE = 384;
56
57/// \brief Representation of a graph, inspired by TS38.212 Tables 5.3.2-2 and 5.3.2-3.
58///
59/// Graphs are represented by a \c max_BG_M \f$\times\f$ \c max_BG_N_full matrix. A number between 0 and 383
60/// at entry \f$(m, n)\f$ denotes a connection between check node \f$m\f$ and variable node \f$n\f$
61/// in the base graph. The value of the number represents the circular shift applied to the connections
62/// in the lifted graph. If entry \f$(m, n)\f$ is equal to \c NO_EDGE, there is no connection in either the
63/// base graph or the lifted one.
64using BG_matrix_t = std::array<std::array<uint16_t, MAX_BG_N_FULL>, MAX_BG_M>;
65
66/// Maximum degree (number of incident edges) of a check node, in the base graphs.
67static constexpr unsigned MAX_BG_CHECK_EDGES = 20;
68
69/// \brief Represents a list of variable nodes connected to a check node.
70using BG_adjacency_row_t = std::array<uint16_t, MAX_BG_CHECK_EDGES>;
71
72/// \brief For check node \c m, BG_adjacency_matrix_t[m] provides a list of the variable nodes connected to it.
73using BG_adjacency_matrix_t = std::array<BG_adjacency_row_t, MAX_BG_M>;
74
75/// Number of base graphs.
76static constexpr unsigned NOF_BASE_GRAPHS = 2;
77/// Total number of lifted graphs.
78static constexpr unsigned TOTAL_NOF_GRAPHS = NOF_LIFTING_SIZES * NOF_BASE_GRAPHS;
79} // namespace ldpc

81/// Describes the Tanner graph of an LDPC code (implementation).
82class ldpc_graph_impl
83{
84public:
85 /// \name Constructors and destructors
86
87 ///@{
88 ldpc_graph_impl(ldpc_base_graph_type _bg, ldpc::lifting_size_t _ls);
89 ldpc_graph_impl() : ldpc_graph_impl(ldpc_base_graph_type::BG1, ldpc::lifting_size_t::LS2) {}
90 ~ldpc_graph_impl() = default;
91 ///@}
92
93 /// \name Copy constructor and operator
94 /// Deleted
95
96 ///@{
97 ldpc_graph_impl(ldpc_graph_impl&) = delete;
98 ldpc_graph_impl& operator=(ldpc_graph_impl&) = delete;
99 ///@}
100
101 /// \name Move constructor and operator
102 /// Default
103
104 ///@{
105 ldpc_graph_impl(ldpc_graph_impl&&) = default;
106 ldpc_graph_impl& operator=(ldpc_graph_impl&&) = default;
107 ///@}
108
109 /// Returns the base graph generating the current graph.
110 ldpc_base_graph_type get_base_graph() const { return base_graph; }
111
112 /// Returns the lifting size generating the current graph.
113 ldpc::lifting_size_t get_lifting_size() const { return lifting_size; }
114
115 /// Returns the lifting index corresponding to the current lifting size (a number between 0 and 7).
116 uint8_t get_lifting_index() const { return lifting_index; }
117
118 /// \brief Returns the compact representation of the lifted node at position \f$(m,n)\f$.
119 ///
120 /// If the base graph has an edge between check node \c m and variable node \c n, the returned value is
121 /// the shift applied to the lifted node. Conversely, if there is no edge, then the returned value is
122 /// srsran::ldpc::NO_EDGE.
123 uint16_t get_lifted_node(unsigned m, unsigned n) const { return parity_check_matrix[m][n]; }
124
125 /// Returns the number of check nodes of the underlying base graph.
126 unsigned get_nof_BG_check_nodes() const { return nof_BG_check_nodes; }
127
128 /// Returns the number of variable nodes of the underlying base graph.
129 unsigned get_nof_BG_var_nodes_full() const { return nof_BG_var_nodes_full; }
130
131 /// Returns the number of variable nodes of the underlying base graph after shortening.
132 unsigned get_nof_BG_var_nodes_short() const { return nof_BG_var_nodes_short; }
133
134 /// Returns the number of variable nodes of the underlying base graph associated to information bits.
135 unsigned get_nof_BG_info_nodes() const { return nof_BG_var_nodes_full - nof_BG_check_nodes; }
136
137 /// Returns a reference to the selected row of ldpc_graph_impl::adjacency_matrix.
138 const ldpc::BG_adjacency_row_t& get_adjacency_row(unsigned m) const { return (*adjacency_matrix)[m]; }
139
140 const std::vector<std::tuple<uint16_t, uint16_t, uint16_t>>& get_parity_check_sparse() const
141 {
142 return parity_check_sparse;
143 }
144
145private:
146 /// Base graph.
147 ldpc_base_graph_type base_graph;
148 /// Lifting size.
149 ldpc::lifting_size_t lifting_size;
150 /// Lifting index.
151 uint8_t lifting_index;
152 /// Number of check nodes in the base graph.
153 unsigned nof_BG_check_nodes;
154 /// Number of variable nodes in the base graph, before shortening.
155 unsigned nof_BG_var_nodes_full;
156 /// Number of variable nodes in the base graph, after shortening.
157 unsigned nof_BG_var_nodes_short;
158
159 /// Parity check matrix of the graph (BG representation).
160 ldpc::BG_matrix_t parity_check_matrix;
161
162 std::vector<std::tuple<uint16_t, uint16_t, uint16_t>> parity_check_sparse;
163
164 /// Adjacency matrix of the base graph (sparse representation).
165 const ldpc::BG_adjacency_matrix_t* adjacency_matrix;
166};

169/// \brief Creates an array with all possible LDPC graphs.
170///
171/// The graphs will be ordered by base graph and lifting size. For instance, the graph corresponding to BG2 and LS4 will
172/// be at entry ::NOF_LIFTING_SIZES <tt> + 3 </tt>, since \c LS4 is the third one in the list of all possible lifting
173/// sizes (i.e., ::all_lifting_sizes).
174std::array<ldpc_graph_impl, TOTAL_NOF_GRAPHS> create_graph_array();
