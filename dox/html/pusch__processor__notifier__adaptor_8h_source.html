142 class pusch_demodulator_notifier_impl : public pusch_demodulator_notifier
143 {
144 public:
145 pusch_demodulator_notifier_impl(channel_state_information& csi_) : csi(csi_) {}
146
147 // See interface for documentation.
148 void on_provisional_stats(const demodulation_stats& stats) override { update_csi(stats); }
149
150 // See interface for documentation.
151 void on_end_stats(const demodulation_stats& stats) override { update_csi(stats); }
152
153 private:
154 void update_csi(const demodulation_stats& stats)
155 {
156 if (stats.sinr_dB.has_value()) {
157 csi.set_sinr_dB(channel_state_information::sinr_type::post_equalization, stats.sinr_dB.value());
158 }
159 if (stats.evm.has_value()) {
160 csi.set_evm(stats.evm.value());
161 csi.set_sinr_dB(channel_state_information::sinr_type::evm, -20.0F * log10f(stats.evm.value()) - 3.7F);
162 }
163 }
164
165 channel_state_information& csi;
166 };

238 void on_sch_data(const pusch_decoder_result& result) override
239 {
240 pusch_processor_result_data result_data;
241 result_data.data = result;
242 result_data.csi = uci_payload.csi;
243 notifier->on_sch(result_data);
244 }
245
246 // See detail::pusch_processor_notifier_callback for documentation.
247 void on_harq_ack(const pusch_uci_field& field) override
248 {
249 // Check HARQ ACK is pending.
250 srsran_assert(pending_harq_ack, "HARQ ACK is not pending.");
251
252 // Set HARQ ACK field.
253 uci_payload.harq_ack = field;
254
255 // HARQ ACK is no longer pending.
256 pending_harq_ack = false;
257
258 // Check if the rest of fields are pending and notify the results.
259 check_and_notify_uci();
260 }
261
262 // See detail::pusch_processor_notifier_callback for documentation.
263 void on_csi_part1(const pusch_uci_field& field) override
264 {
265 // Check CSI Part 1 is pending.
266 srsran_assert(pending_csi_part1, "CSI Part 1 is not pending.");
267
268 if (field.status == uci_status::valid) {
269 srsran_assert(csi_part_1_feedback != nullptr, "Invalid CSI Part 1 feedback.");
270 csi_part_1_feedback->on_csi_part1(field.payload);
271 }
272
273 // Set CSI Part 1 field.
274 uci_payload.csi_part1 = field;
275
276 // CSI Part 1 is no longer pending.
277 pending_csi_part1 = false;
278
279 // Check if the rest of fields are pending and notify the results.
280 check_and_notify_uci();
281 }
282
283 // See detail::pusch_processor_notifier_callback for documentation.
284 void on_csi_part2(const pusch_uci_field& field) override
285 {
286 // Check CSI Part 2 is pending.
287 srsran_assert(pending_csi_part2, "CSI Part 2 is not pending.");
288
289 // Set CSI Part 2 field.
290 uci_payload.csi_part2 = field;
291
292 // CSI Part 2 is no longer pending.
293 pending_csi_part2 = false;
294
295 // Check if the rest of fields are pending and notify the results.
296 check_and_notify_uci();
297 }
298
300 void check_and_notify_uci()
301 {
302 srsran_assert(notifier != nullptr, "Invalid notifier.");
303 if (!pending_harq_ack && !pending_csi_part1 && !pending_csi_part2) {
304 notifier->on_uci(uci_payload);
305 }
306 }
307
309 bool pending_harq_ack = false;
311 bool pending_csi_part1 = false;
313 bool pending_csi_part2 = false;
315 pusch_processor_result_control uci_payload;
317 pusch_processor_result_notifier* notifier;
319 pusch_processor_csi_part1_feedback* csi_part_1_feedback;
321 pusch_demodulator_notifier_impl pusch_demod_notifier;
323 sch_data_notifier_impl sch_data_notifier;
325 harq_ack_notifier_impl harq_ack_notifier;
327 csi_part1_notifier_impl csi_part_1_notifier;
329 csi_part2_notifier_impl csi_part_2_notifier;
330};
