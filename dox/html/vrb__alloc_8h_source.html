52 vrb_alloc& operator=(const vrb_alloc& other) noexcept
53 {
54 if (this == &other) {
55 return *this;
56 }
57 if (other.alloc_type_0) {
58 *this = other.type0();
59 } else {
60 *this = other.type1();

62 return *this;
63 }
64 vrb_alloc& operator=(const vrb_interval& vrbs)
65 {
66 if (alloc_type_0) {
67 alloc_type_0 = false;
68 alloc.rbgs.~rbg_bitmap();
69 new (&alloc.interv) vrb_interval(vrbs);

73 return *this;
74 }
75 vrb_alloc& operator=(const rbg_bitmap& rbgs)

80 alloc_type_0 = true;
81 alloc.interv.~vrb_interval();

83 }
84 return *this;
85 }

87 {
88 if (is_type0()) {
89 alloc.rbgs.~rbg_bitmap();

91 alloc.interv.~vrb_interval();

124 return alloc.interv;
125 }
126
128 bool any() const { return is_type0() ? type0().count() : type1().empty(); }
