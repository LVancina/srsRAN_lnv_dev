32class tx_window_checker : public ota_symbol_boundary_notifier
33{
34public:
35 tx_window_checker(srslog::basic_logger& logger_,
36 uint32_t advance_time_in_symbols_,
37 uint32_t nof_symbols_,
38 uint32_t numerology_) :
39 logger(logger_),
40 advance_time_in_symbols(advance_time_in_symbols_),
41 nof_symbols(nof_symbols_),
42 numerology(numerology_)
43 {
44 }
45
46 // See interface for documentation.
47 void on_new_symbol(slot_symbol_point symbol_point) override
48 {
49 // This atomic is only written from a single thread.
50 count_val.store(symbol_point.system_slot(), std::memory_order::memory_order_release);
51 }
52
54 bool is_late(slot_point slot) const
55 {
56 slot_symbol_point ota_symbol_point(
57 numerology, count_val.load(std::memory_order::memory_order_acquire), nof_symbols);
58
59 // Use symbol 0 as the worst case for the resource grid slot.
60 slot_symbol_point rg_point(slot, 0, nof_symbols);
61 rg_point -= advance_time_in_symbols;
62
63 if (ota_symbol_point < rg_point) {
64 return false;
65 }
66
67 logger.debug("A late upper-PHY downlink request arrived to OFH in slot '{}_{}' with current ota_slot='{}_{}', "
68 "OFH processing time requires a minimum of '{}' symbols",
69 slot,
70 0,
71 ota_symbol_point.get_slot(),
72 ota_symbol_point.get_symbol_index(),
73 advance_time_in_symbols);
74
75 return true;
76 }
77
78private:
79 srslog::basic_logger& logger;
80 const uint32_t advance_time_in_symbols;
81 const uint32_t nof_symbols;
82 const uint32_t numerology;
83 std::atomic<uint32_t> count_val;
84};
