36class pdcp_tx_test_frame : public pdcp_rx_status_provider,
37 public pdcp_tx_lower_notifier,
38 public pdcp_tx_upper_control_notifier
39{
40public:
41 std::queue<pdcp_tx_pdu> pdu_queue = {};
42 uint32_t nof_max_count_reached = 0;
43 uint32_t nof_protocol_failure = 0;
44 std::queue<uint32_t> sdu_discard_queue = {};
45
46 // PDCP RX status provider
47 byte_buffer compile_status_report() final
48 {
49 // Build status report dummy to be picked by the TX entity
50 byte_buffer buf = {};
51 bit_encoder enc(buf);
52
53 // Pack PDU header
54 enc.pack(to_number(pdcp_dc_field::control), 1);
55 enc.pack(to_number(pdcp_control_pdu_type::status_report), 3);
56 enc.pack(0b0000, 4);
57
58 // Pack something into FMC field
59 enc.pack(0xc0cac01a, 32);
60
61 // Pack some bitmap
62 enc.pack(0xcafe, 16);
63
64 return buf;
65 }
66
67 // PDCP TX upper layer control notifier
68 void on_max_count_reached() final { nof_max_count_reached++; }
69 void on_protocol_failure() final { nof_protocol_failure++; }
70
71 // PDCP TX lower layer data notifier
72 void on_new_pdu(pdcp_tx_pdu pdu) final { pdu_queue.push(std::move(pdu)); }
73 void on_discard_pdu(uint32_t pdcp_sn) final { sdu_discard_queue.push(pdcp_sn); }
74};

90 void init(std::tuple<pdcp_sn_size, unsigned> cfg_tuple,
91 pdcp_rb_type rb_type_ = pdcp_rb_type::drb,
92 pdcp_rlc_mode rlc_mode_ = pdcp_rlc_mode::am,
93 pdcp_discard_timer discard_timer = pdcp_discard_timer::ms10,
94 pdcp_max_count max_count = {pdcp_tx_default_max_count_notify, pdcp_tx_default_max_count_hard})
95 {
96 logger.info("Creating PDCP TX ({} bit, NIA{}, NEA{})",
97 pdcp_sn_size_to_uint(std::get<pdcp_sn_size>(cfg_tuple)),
98 std::get<unsigned>(cfg_tuple),
99 std::get<unsigned>(cfg_tuple));
100
101 sn_size = std::get<pdcp_sn_size>(cfg_tuple);
102 algo = std::get<unsigned>(cfg_tuple);
103 pdu_hdr_len = pdcp_data_pdu_header_size(sn_size); // Round up division
104
105 // Set TX config
106 config.sn_size = sn_size;
107 config.rb_type = rb_type_;
108 config.rlc_mode = rlc_mode_;
109 config.direction = pdcp_security_direction::downlink;
110 config.discard_timer = discard_timer;
111 config.custom.max_count = max_count;
112 config.status_report_required = true;
113
114 // RB_id and security domain
115 rb_id_t rb_id;
116 switch (rb_type_) {
117 case pdcp_rb_type::srb:
118 sec_cfg.domain = security::sec_domain::rrc;
119 rb_id = srb_id_t::srb1;
120 break;
121 case pdcp_rb_type::drb:
122 sec_cfg.domain = security::sec_domain::up;
123 rb_id = drb_id_t::drb1;
124 break;
125 }
126
127 // Set security keys
128 sec_cfg.k_128_int = k_128_int;
129 sec_cfg.k_128_enc = k_128_enc;
130
131 // Set encription/integrity algorithms
132 sec_cfg.integ_algo = static_cast<security::integrity_algorithm>(std::get<unsigned>(cfg_tuple));
133 sec_cfg.cipher_algo = static_cast<security::ciphering_algorithm>(std::get<unsigned>(cfg_tuple));
134
135 // Allow for config adjustments
136 init_adjustments();
137
138 // Create PDCP entity
139 pdcp_tx = std::make_unique<pdcp_entity_tx>(0, rb_id, config, test_frame, test_frame, timer_factory{timers, worker});
140 pdcp_tx->set_status_provider(&test_frame);
141 }
