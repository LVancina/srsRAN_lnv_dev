47inline void fill_asn1_ng_setup_request(asn1::ngap::ng_setup_request_s& asn1_request,
48 const ngap_ng_setup_request& request)
49{
50 // fill global ran node id
51 asn1_request->global_ran_node_id.set_global_gnb_id();
52 asn1_request->global_ran_node_id.global_gnb_id().gnb_id.set_gnb_id();
53 asn1_request->global_ran_node_id.global_gnb_id().gnb_id.gnb_id().from_number(request.global_ran_node_id.gnb_id);
54 asn1_request->global_ran_node_id.global_gnb_id().plmn_id.from_number(
55 plmn_string_to_bcd(request.global_ran_node_id.plmn_id));
56
57 // fill ran node name
58 asn1_request->ran_node_name_present = true;
59 asn1_request->ran_node_name.from_string(request.ran_node_name);
60
61 // fill supported ta list
62 for (const auto& supported_ta_item : request.supported_ta_list) {
63 asn1::ngap::supported_ta_item_s asn1_supported_ta_item = {};
64
65 // fill tac
66 asn1_supported_ta_item.tac.from_number(supported_ta_item.tac);
67
68 // fill broadcast plmn list
69 for (const auto& broadcast_plmn_item : supported_ta_item.broadcast_plmn_list) {
70 asn1::ngap::broadcast_plmn_item_s asn1_broadcast_plmn_item = {};
71
72 // fill plmn id
73 asn1_broadcast_plmn_item.plmn_id.from_number(plmn_string_to_bcd(broadcast_plmn_item.plmn_id));
74
75 // fill tai slice support list
76 for (const auto& slice_support_item : broadcast_plmn_item.tai_slice_support_list) {
77 // fill s_nssai
78 asn1::ngap::slice_support_item_s asn1_slice_support_item = {};
79 asn1_slice_support_item.s_nssai = s_nssai_to_asn1(slice_support_item.s_nssai);
80
81 asn1_broadcast_plmn_item.tai_slice_support_list.push_back(asn1_slice_support_item);
82 }
83 asn1_supported_ta_item.broadcast_plmn_list.push_back(asn1_broadcast_plmn_item);
84 }
85 asn1_request->supported_ta_list.push_back(asn1_supported_ta_item);
86 }
87
88 // fill paging drx
89 asn1::number_to_enum(asn1_request->default_paging_drx, request.default_paging_drx);
90}

95inline void fill_ngap_ng_setup_result(ngap_ng_setup_result& result, const asn1::ngap::ng_setup_resp_s& asn1_response)
96{
97 ngap_ng_setup_response response;
98
99 // fill amf name
100 response.amf_name = asn1_response->amf_name.to_string();
101
102 // fill served guami list
103 for (const auto& asn1_served_guami_item : asn1_response->served_guami_list) {
104 ngap_served_guami_item served_guami_item = {};
105 served_guami_item.guami = asn1_to_guami(asn1_served_guami_item.guami);
106 if (asn1_served_guami_item.backup_amf_name_present) {
107 served_guami_item.backup_amf_name = asn1_served_guami_item.backup_amf_name.to_string();
108 }
109 response.served_guami_list.push_back(served_guami_item);
110 }
111
112 // fill relative amf capacity
113 response.relative_amf_capacity = asn1_response->relative_amf_capacity;
114
115 // fill plmn support list
116 for (const auto& asn1_plmn_support_item : asn1_response->plmn_support_list) {
117 ngap_plmn_support_item plmn_support_item = {};
118 plmn_support_item.plmn_id = asn1_plmn_support_item.plmn_id.to_string();
119
120 for (const auto& asn1_slice_support_item : asn1_plmn_support_item.slice_support_list) {
121 slice_support_item_t slice_support_item = {};
122 slice_support_item.s_nssai.sst = asn1_slice_support_item.s_nssai.sst.to_number();
123 if (asn1_slice_support_item.s_nssai.sd_present) {
124 slice_support_item.s_nssai.sd = asn1_slice_support_item.s_nssai.sd.to_number();
125 }
126 plmn_support_item.slice_support_list.push_back(slice_support_item);
127 }
128 response.plmn_support_list.push_back(plmn_support_item);
129 }
130
131 // TODO: Fill crit diagnostics
132
133 // TODO: Add missing optional values
134
135 result = response;
136}

199inline bool fill_cu_cp_pdu_session_resource_setup_item_base(cu_cp_pdu_session_res_setup_item& setup_item,
200 const template_asn1_item& asn1_session_item,
201 const byte_buffer asn1_request_transfer)
202{
203 // pDUSessionID
204 setup_item.pdu_session_id = uint_to_pdu_session_id(asn1_session_item.pdu_session_id);
205
206 // pDUSessionNAS-PDU / NAS-PDU will be added in a separate function
207
208 // s-NSSAI
209 if (asn1_session_item.s_nssai.sd_present) {
210 setup_item.s_nssai.sd = asn1_session_item.s_nssai.sd.to_number();
211 }
212 setup_item.s_nssai.sst = asn1_session_item.s_nssai.sst.to_number();
213
214 // pDUSessionResourceSetupRequestTransfer
215 asn1::ngap::pdu_session_res_setup_request_transfer_s asn1_setup_req_transfer;
216 asn1::cbit_ref bref({asn1_request_transfer.begin(), asn1_request_transfer.end()});
217
218 if (asn1_setup_req_transfer.unpack(bref) != asn1::SRSASN_SUCCESS) {
219 srslog::fetch_basic_logger("NGAP").error("Couldn't unpack PDU Session Resource Setup Request Transfer PDU");
220 return false;
221 }
222
223 if (asn1_setup_req_transfer->pdu_session_aggr_max_bit_rate_present) {
224 // id-PDUSessionAggregateMaximumBitRate
225 setup_item.pdu_session_aggregate_maximum_bit_rate_dl =
226 asn1_setup_req_transfer->pdu_session_aggr_max_bit_rate.pdu_session_aggr_max_bit_rate_dl;
227 setup_item.pdu_session_aggregate_maximum_bit_rate_ul =
228 asn1_setup_req_transfer->pdu_session_aggr_max_bit_rate.pdu_session_aggr_max_bit_rate_ul;
229 }
230
231 // id-UL-NGU-UP-TNLInformation
232 setup_item.ul_ngu_up_tnl_info = asn1_to_up_transport_layer_info(asn1_setup_req_transfer->ul_ngu_up_tnl_info);
233
234 // id-PDUSessionType
235 setup_item.pdu_session_type = asn1_setup_req_transfer->pdu_session_type.to_string();
236
237 // id-SecurityIndication
238 if (asn1_setup_req_transfer->security_ind_present) {
239 security_indication_t ind = {};
240 asn1_to_security_indication(ind, asn1_setup_req_transfer->security_ind);
241 setup_item.security_ind = ind;
242 }
243
244 // id-QosFlowSetupRequestList
245 for (const auto& asn1_flow_item : asn1_setup_req_transfer->qos_flow_setup_request_list) {
246 qos_flow_setup_request_item qos_flow_setup_req_item;
247
248 // qosFlowIdentifier
249 qos_flow_setup_req_item.qos_flow_id = uint_to_qos_flow_id(asn1_flow_item.qos_flow_id);
250
251 // qosFlowLevelQosParameters
252 if (asn1_flow_item.qos_flow_level_qos_params.qos_characteristics.type() ==
253 asn1::ngap::qos_characteristics_c::types::dyn5qi) {
254 dyn_5qi_descriptor_t dyn_5qi = {};
255 if (asn1_flow_item.qos_flow_level_qos_params.qos_characteristics.dyn5qi().five_qi_present) {
256 dyn_5qi.five_qi =
257 uint_to_five_qi(asn1_flow_item.qos_flow_level_qos_params.qos_characteristics.dyn5qi().five_qi);
258 }
259 // TODO: Add optional values
260
261 qos_flow_setup_req_item.qos_flow_level_qos_params.qos_characteristics.dyn_5qi = dyn_5qi;
262
263 // TODO: Add optional values
264
265 } else if (asn1_flow_item.qos_flow_level_qos_params.qos_characteristics.type() ==
266 asn1::ngap::qos_characteristics_c::types::non_dyn5qi) {
267 non_dyn_5qi_descriptor_t non_dyn_5qi = {};
268 non_dyn_5qi.five_qi =
269 uint_to_five_qi(asn1_flow_item.qos_flow_level_qos_params.qos_characteristics.non_dyn5qi().five_qi);
270 qos_flow_setup_req_item.qos_flow_level_qos_params.qos_characteristics.non_dyn_5qi = non_dyn_5qi;
271
272 // TODO: Add optional values
273 }
274
275 // allocationAndRetentionPriority
276 qos_flow_setup_req_item.qos_flow_level_qos_params.alloc_and_retention_prio.prio_level_arp =
277 asn1_flow_item.qos_flow_level_qos_params.alloc_and_retention_prio.prio_level_arp;
278 qos_flow_setup_req_item.qos_flow_level_qos_params.alloc_and_retention_prio.pre_emption_cap =
279 asn1_flow_item.qos_flow_level_qos_params.alloc_and_retention_prio.pre_emption_cap.to_string();
280 qos_flow_setup_req_item.qos_flow_level_qos_params.alloc_and_retention_prio.pre_emption_vulnerability =
281 asn1_flow_item.qos_flow_level_qos_params.alloc_and_retention_prio.pre_emption_vulnerability.to_string();
282
283 // Optional Parameters
284 if (asn1_flow_item.qos_flow_level_qos_params.add_qos_flow_info_present) {
285 qos_flow_setup_req_item.qos_flow_level_qos_params.add_qos_flow_info =
286 asn1_flow_item.qos_flow_level_qos_params.add_qos_flow_info.to_string();
287 }
288
289 if (asn1_flow_item.qos_flow_level_qos_params.gbr_qos_info_present) {
290 // TODO: Add to common type
291 }
292
293 if (asn1_flow_item.qos_flow_level_qos_params.reflective_qos_attribute_present) {
294 qos_flow_setup_req_item.qos_flow_level_qos_params.reflective_qos_attribute =
295 asn1::enum_to_bool(asn1_flow_item.qos_flow_level_qos_params.reflective_qos_attribute);
296 }
297
298 if (asn1_flow_item.erab_id_present) {
299 qos_flow_setup_req_item.erab_id = asn1_flow_item.erab_id;
300 }
301
302 setup_item.qos_flow_setup_request_items.emplace(qos_flow_setup_req_item.qos_flow_id, qos_flow_setup_req_item);
303 }
304 return true;
305}

311inline bool fill_cu_cp_pdu_session_resource_setup_request(
312 cu_cp_pdu_session_resource_setup_request& cu_cp_pdu_session_resource_setup_msg,
313 const asn1::dyn_seq_of<asn1::ngap::pdu_session_res_setup_item_su_req_s, 1U, 256U, true>&
314 asn1_pdu_session_res_setup_list)
315{
316 if (asn1_pdu_session_res_setup_list.size() == 0) {
317 return false;
318 }
319
320 for (const auto& asn1_session_item : asn1_pdu_session_res_setup_list) {
321 cu_cp_pdu_session_res_setup_item setup_item;
322
323 if (!fill_cu_cp_pdu_session_resource_setup_item_base(
324 setup_item, asn1_session_item, asn1_session_item.pdu_session_res_setup_request_transfer.copy())) {
325 return false;
326 }
327
328 // pDUSessionNAS-PDU
329 if (!asn1_session_item.pdu_session_nas_pdu.empty()) {
330 setup_item.pdu_session_nas_pdu.resize(asn1_session_item.pdu_session_nas_pdu.size());
331 std::copy(asn1_session_item.pdu_session_nas_pdu.begin(),
332 asn1_session_item.pdu_session_nas_pdu.end(),
333 setup_item.pdu_session_nas_pdu.begin());
334 }
335
336 cu_cp_pdu_session_resource_setup_msg.pdu_session_res_setup_items.emplace(setup_item.pdu_session_id,
337 std::move(setup_item));
338 }
339 return true;
340}

378{

382 }
383
384 // ue_aggr_max_bit_rate
385 if (asn1_request->ue_aggr_max_bit_rate_present) {
386 request.ue_aggr_max_bit_rate.emplace();
387 request.ue_aggr_max_bit_rate.value().ue_aggr_max_bit_rate_dl =

409
410 // security_context
411 copy_asn1_key(request.security_context.k, asn1_request->security_key);

417 // ue_radio_cap

419 request.ue_radio_cap = asn1_request->ue_radio_cap.copy();
420 }
421
422 // idx_to_rfsp
423 if (asn1_request->idx_to_rfsp_present) {
424 request.idx_to_rfsp = asn1_request->idx_to_rfsp;
425 }
426
427 // masked_imeisv

429 request.masked_imeisv = asn1_request->masked_imeisv.to_number();
430 }
431
432 // nas_pdu
433 if (asn1_request->nas_pdu_present) {
434 request.nas_pdu = asn1_request->nas_pdu.copy();
435 }
436
437 // ue_radio_cap_for_paging
438 if (asn1_request->ue_radio_cap_for_paging_present) {
439 cu_cp_ue_radio_cap_for_paging ue_radio_cap_for_paging;
440 ue_radio_cap_for_paging.ue_radio_cap_for_paging_of_nr =
441 asn1_request->ue_radio_cap_for_paging.ue_radio_cap_for_paging_of_nr.copy();
442
443 request.ue_radio_cap_for_paging = ue_radio_cap_for_paging;
444 }
445
446 // TODO: Add missing optional values
447
448 return true;
449}
