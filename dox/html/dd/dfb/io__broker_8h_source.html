37/// \brief Describes the base interface for an (async) IO broker.
38/// The IO broker is responsible for handling all IO events, including
39/// sockets, signals, and system timers.
40/// File descriptors can be registered with the IO broker via register_fd()
41/// and unregistered via unregister_fd().
42class io_broker
43{
44protected:
45 /// Allow default constructor for child.
46 io_broker() = default;
47
48public:
49 /// Callback called when socket fd (passed as argument) has data
50 using recv_callback_t = std::function<void(int)>;
51
52 /// Provides default destructor.
53 virtual ~io_broker() = default;
54
55 /// Forbid copy constructor.
56 io_broker(const io_broker& other) = delete;
57
58 /// Forbid move constructor.
59 io_broker(const io_broker&& other) = delete;
60
61 /// Forbid copy assigment operator.
62 io_broker& operator=(const io_broker&) = delete;
63
64 /// Forbid move assigment operator.
65 io_broker& operator=(io_broker&&) = delete;
66
67 /// \brief Register a file descriptor to be handled by the IO interface.
68 SRSRAN_NODISCARD virtual bool register_fd(int fd, recv_callback_t handler) = 0;
69
70 /// \brief Unregister a file descriptor from the IO interface.
71 SRSRAN_NODISCARD virtual bool unregister_fd(int fd) = 0;
72};
