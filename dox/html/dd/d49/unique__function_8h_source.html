30/// Size of the small buffer embedded in "unique_function<R(Args...)>", used to store functors without calling "new"
31constexpr size_t default_unique_function_buffer_size = 32;
32
33/// unique_function<R(Args...)> declaration
34/// @tparam Signature Functor signature with format: R(Args...)
35/// @tparam Capacity Small buffer size embedded into the unique_function type
36/// @tparam ForbidAlloc Determines whether malloc of the Functor is allowed or forbidden. In case of ForbidAlloc==true,
37/// the compiler forbids storing functors, such that "sizeof(Functor) > Capacity"
38template <class Signature, size_t Capacity = default_unique_function_buffer_size, bool ForbidAlloc = false>
39class unique_function;
40

43/// Base vtable for move/call/destroy operations over the functor stored in "unique_function<R(Args...)"
44template <typename R, typename... Args>
45class oper_table_t
46{
47public:
48 constexpr oper_table_t() = default;
49 virtual ~oper_table_t() = default;
50 virtual R call(void* src, Args... args) const = 0;
51 virtual void move(void* src, void* dest) const = 0;
52 virtual void dtor(void* src) const = 0;
53 virtual bool is_in_small_buffer() const = 0;
54};
55
56/// Specialization of move/call/destroy operations for when the "unique_function<R(Args...)>" is empty
57template <typename R, typename... Args>
58class empty_table_t final : public oper_table_t<R, Args...>
59{
60public:
61 constexpr empty_table_t() = default;
62 R call(void* src, Args... args) const override
63 {
64 srsran_terminate("bad function call (cause: function ptr is empty)");
65 }
66 void move(void* src, void* dest) const override {}
67 void dtor(void* src) const override {}
68 bool is_in_small_buffer() const override { return true; }
69};
70
71/// specialization of move/call/destroy operations for when the functor is stored in "unique_function<R(Args...)>"
72/// buffer
73template <typename FunT, typename R, typename... Args>
74class smallbuffer_table_t final : public oper_table_t<R, Args...>
75{
76public:
77 constexpr smallbuffer_table_t() = default;
78 R call(void* src, Args... args) const override { return (*static_cast<FunT*>(src))(std::forward<Args>(args)...); }
79 void move(void* src, void* dest) const override
80 {
81 ::new (dest) FunT(std::move(*static_cast<FunT*>(src)));
82 static_cast<FunT*>(src)->~FunT();
83 }
84 void dtor(void* src) const override { static_cast<FunT*>(src)->~FunT(); }
85 bool is_in_small_buffer() const override { return true; }
86};
87
88/// move/call/destroy operations for when the functor is stored outside of "unique_function<R(Args...)>" buffer
89template <typename FunT, typename R, typename... Args>
90class heap_table_t final : public oper_table_t<R, Args...>
91{
92public:
93 constexpr heap_table_t() = default;
94 R call(void* src, Args... args) const override { return (**static_cast<FunT**>(src))(std::forward<Args>(args)...); }
95 void move(void* src, void* dest) const override
96 {
97 *static_cast<FunT**>(dest) = *static_cast<FunT**>(src);
98 *static_cast<FunT**>(src) = nullptr;
99 }
100 void dtor(void* src) const override { delete (*static_cast<FunT**>(src)); }
101 bool is_in_small_buffer() const override { return false; }
102};

112/// Metafunctions to enable different ctor implementations depending on whether the callback fits the small buffer
113template <typename T, size_t Cap, typename FunT = typename std::decay<T>::type>
114using enable_if_small_capture =
115 typename std::enable_if<sizeof(FunT) <= Cap and not is_unique_function<FunT>::value, bool>::type;
116template <typename T, size_t Cap, typename FunT = typename std::decay<T>::type>
117using enable_if_big_capture =
118 typename std::enable_if < Cap<sizeof(FunT) and not is_unique_function<FunT>::value, bool>::type;

122template <class R, class... Args, size_t Capacity, bool ForbidAlloc>
123class unique_function<R(Args...), Capacity, ForbidAlloc>
124{
125 static_assert(Capacity >= sizeof(void*), "unique_function embedded buffer is too small");
126 using storage_t = std::aligned_storage_t<Capacity, alignof(std::max_align_t)>;
127 using oper_table_t = task_details::oper_table_t<R, Args...>;
128 static const task_details::empty_table_t<R, Args...> empty_table;
129
130public:
131 static constexpr size_t capacity = Capacity; ///< size of buffer
132
133 constexpr unique_function() noexcept : oper_ptr(&empty_table) {}
134
135 /// Called when T capture fits the unique_function small buffer
136 template <typename T, task_details::enable_if_small_capture<T, capacity> = true>
137 unique_function(T&& function) noexcept(std::is_nothrow_move_constructible<T>::value)
138 {
139 using FunT = typename std::decay<T>::type;
140 static const task_details::smallbuffer_table_t<FunT, R, Args...> small_oper_table{};
141 oper_ptr = &small_oper_table;
142 ::new (&buffer) FunT(std::forward<T>(function));
143 }
144
145 /// Called when T capture does not fit the unique_function small buffer
146 template <typename T, task_details::enable_if_big_capture<T, capacity> = true>
147 unique_function(T&& function) noexcept(std::is_nothrow_move_constructible<T>::value)
148 {
149 static_assert(
150 not ForbidAlloc,
151 "Failed to store provided callback in std::unique_function specialization that forbids heap allocations.");
152 using FunT = typename std::decay<T>::type;
153 static const task_details::heap_table_t<FunT, R, Args...> heap_oper_table{};
154 oper_ptr = &heap_oper_table;
155 ptr = static_cast<void*>(new FunT{std::forward<T>(function)});
156 }
157
158 unique_function(unique_function&& other) noexcept : oper_ptr(other.oper_ptr)
159 {
160 other.oper_ptr = &empty_table;
161 oper_ptr->move(&other.buffer, &buffer);
162 }
163
164 ~unique_function() { oper_ptr->dtor(&buffer); }
165
166 unique_function& operator=(unique_function&& other) noexcept
167 {
168 oper_ptr->dtor(&buffer);
169 oper_ptr = other.oper_ptr;
170 other.oper_ptr = &empty_table;
171 oper_ptr->move(&other.buffer, &buffer);
172 return *this;
173 }
174
175 /// Call operator
176 R operator()(Args... args) const noexcept { return oper_ptr->call(&buffer, std::forward<Args>(args)...); }
177
178 bool is_empty() const noexcept { return oper_ptr == &empty_table; }
179 bool is_in_small_buffer() const noexcept { return oper_ptr->is_in_small_buffer(); }
180
181private:
182 union {
183 mutable storage_t buffer;
184 void* ptr;
185 };
186 const oper_table_t* oper_ptr;
187};
188
189template <typename R, typename... Args, size_t Capacity, bool ForbidAlloc>
190const task_details::empty_table_t<R, Args...> unique_function<R(Args...), Capacity, ForbidAlloc>::empty_table{};
