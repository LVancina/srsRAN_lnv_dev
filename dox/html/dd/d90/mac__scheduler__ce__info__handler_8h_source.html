66/// \brief Interface between MAC and scheduler that is used by MAC to forward MAC CE information and force UL grants.
67class mac_scheduler_ce_info_handler

69public:

71
72 /// \brief Forward to scheduler any decoded UL BSRs for a given UE.

75 /// \brief Force the UL grant scheduling for a given UE.
76 virtual void handle_ul_sched_command(const mac_ul_scheduling_command& sched_cmd) = 0;
77
78 /// \brief Command scheduling of DL MAC CE for a given UE.
