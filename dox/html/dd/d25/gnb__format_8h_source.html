197/// FMT formatter of slot_point type.
198template <>
199struct formatter<srsran::ue_event_prefix> {
200 template <typename ParseContext>
201 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
202 {
203 return ctx.begin();
204 }
205 template <typename FormatContext>
206 auto format(const srsran::ue_event_prefix& ue_prefix, FormatContext& ctx)
207 -> decltype(std::declval<FormatContext>().out())
208 {
209 using namespace srsran;
210 auto ret = format_to(ctx.out(), "{:<4}", ue_prefix.direction);
211 if (ue_prefix.ue_index != srsran::MAX_NOF_DU_UES) {
212 ret = format_to(ctx.out(), " ueId={}", ue_prefix.ue_index);
213 } else {
214 ret = format_to(ctx.out(), "{: <7}", "");
215 }
216 if (ue_prefix.rnti != srsran::rnti_t::INVALID_RNTI) {
217 ret = format_to(ctx.out(), " {}", ue_prefix.rnti);
218 } else {
219 ret = format_to(ctx.out(), " {: <6}", "");
220 }
221 if (ue_prefix.cell_index != srsran::MAX_NOF_DU_CELLS) {
222 ret = format_to(ctx.out(), " cell={}", ue_prefix.cell_index);
223 }
224 if (ue_prefix.channel != nullptr) {
225 ret = format_to(ctx.out(), " {}", ue_prefix.channel);
226 }
227 if (ue_prefix.lcid <= MAX_LCID) {
228 ret = format_to(ctx.out(), " {}", ue_prefix.lcid);
229 }
230 return ret;
231 }
232};
