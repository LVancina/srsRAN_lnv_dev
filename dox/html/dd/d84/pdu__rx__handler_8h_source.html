39/// Stores MAC RX PDU, as well as any contextual or temporary information related to the PDU decoding.
40struct decoded_mac_rx_pdu {
41 decoded_mac_rx_pdu() = default;
42 decoded_mac_rx_pdu(slot_point slot_rx_,
43 du_cell_index_t cell_idx_,
44 mac_rx_pdu pdu_rx_,
45 du_ue_index_t ue_index_ = MAX_NOF_DU_UES) :
46 slot_rx(slot_rx_), cell_index_rx(cell_idx_), ue_index(ue_index_), pdu_rx(std::move(pdu_rx_))
47 {
48 srsran_sanity_check(not pdu_rx.pdu.empty(), "Received empty PDU");
49 }
50
51 /// Clear PDU context.
52 void clear()
53 {
54 slot_rx = {};
55 cell_index_rx = MAX_NOF_DU_CELLS;
56 ue_index = MAX_NOF_DU_UES;
57 pdu_rx.pdu.clear();
58 decoded_subpdus.clear();
59 }
60
61 /// Slot when PDU was received in the PHY.
62 slot_point slot_rx;
63
64 /// Cell where PDU was decoded by the PHY.
65 du_cell_index_t cell_index_rx;
66
67 /// UE index for which PDU is directed. ue_index = MAX_NOF_UES if no UE with provided RNTI exists.
68 du_ue_index_t ue_index;
69
70 /// Received MAC PDU content.
71 /// Note: C-RNTI may be later altered, depending on whether a CRNTI MAC CE is present.
72 mac_rx_pdu pdu_rx;
73
74 /// View of decoded subPDUs of the MAC PDU.
75 mac_ul_sch_pdu decoded_subpdus;
76};

78/// Decodes MAC Rx PDUs and handles the respective CEs and SDUs by dispatching them to the Scheduler or UE UL bearers.
79/// This class is thread-safe, i.e. it can be concurrently called for multiple MAC UL PDUs, iff its methods are called
80/// from within the respective MAC PDU RNTI executor.
81class pdu_rx_handler
82{
83public:
84 pdu_rx_handler(mac_ul_ccch_notifier& ccch_notifier_,
85 du_high_ue_executor_mapper& ue_exec_mapper_,
86 mac_scheduler_ce_info_handler& sched_,
87 mac_ul_ue_manager& ue_manager_,
88 du_rnti_table& rnti_table_,
89 mac_pcap& pcap_);
90
91 /// Decode MAC Rx PDU, log contents and handle subPDUs.
92 /// \param sl_rx Slot when MAC UL PDU was received.
93 /// \param cell_index Cell index where MAC UL PDU was received.
94 /// \param pdu MAC UL PDU received.
95 /// \return true if all subPDUs were correctly handled.
96 bool handle_rx_pdu(slot_point sl_rx, du_cell_index_t cell_index, mac_rx_pdu pdu);
97
98 /// Called to push stored UL CCCH to upper layers, once the UE has been created in the DU manager.
99 bool push_ul_ccch_msg(du_ue_index_t ue_index, byte_buffer ul_ccch_msg);
100
101private:
102 /// Handle subPDUs contained in a MAC UL PDU.
103 /// \param pdu MAC UL PDU being processed.
104 /// \return true if all subPDUs were correctly handled.
105 bool handle_rx_subpdus(const decoded_mac_rx_pdu& pdu);
106
107 /// Handle UL SDU targetted at LCIDs >= 1.
108 /// \param pdu MAC UL PDU being processed.
109 /// \param subpdu subPDU of PDU where CRNTI CE was detected.
110 /// \param ue MAC UL UE object for which the sdu is directed.
111 /// \return true if correctly handled.
112 bool handle_sdu(const decoded_mac_rx_pdu& ctx, const mac_ul_sch_subpdu& sdu, mac_ul_ue_context* ue);
113
114 /// Handle UL MAC CE.
115 /// \param pdu MAC UL PDU being processed.
116 /// \param subpdu subPDU of PDU where CRNTI CE was detected.
117 /// \return true if correctly handled.
118 bool handle_mac_ce(const decoded_mac_rx_pdu& pdu, const mac_ul_sch_subpdu& subpdu);
119
120 /// Handle UL CCCH Message
121 /// \param pdu MAC UL PDU being processed.
122 /// \param subpdu subPDU of PDU where CRNTI CE was detected.
123 /// \return true if correctly handled.
124 bool handle_ccch_msg(const decoded_mac_rx_pdu& pdu, const mac_ul_sch_subpdu& sdu);
125
126 /// Handle CRNTI MAC CE
127 /// \remark see TS 38.321, 6.1.3.2 - C-RNTI MAC CE
128 /// \param pdu MAC UL PDU being processed.
129 /// \param subpdu subPDU of PDU where CRNTI CE was detected.
130 /// \return true if correctly handled.
131 bool handle_crnti_ce(const decoded_mac_rx_pdu& pdu, const mac_ul_sch_subpdu& subpdu);
132
133 /// Handle PDU to PCAP file
134 void write_pcap_rx_pdu(const slot_point& sl_rx, const mac_rx_pdu& pdu);
135
136 mac_ul_ccch_notifier& ccch_notifier;
137 du_high_ue_executor_mapper& ue_exec_mapper;
138 srslog::basic_logger& logger;
139 mac_scheduler_ce_info_handler& sched;
140 mac_ul_ue_manager& ue_manager;
141 du_rnti_table& rnti_table;
142 mac_pcap& pcap;
143};
