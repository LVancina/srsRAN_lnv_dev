57/// \brief Scales packed 8-bit integers in \c a by the scaling factor \c sf.
58///
59/// Values of \c a larger than \c max or smaller than \c -max are forwarded unaltered.
60/// \param[in] a Vector of packed 8-bit integers.
61/// \param[in] sf Scaling factor (0, 1].
62/// \param[in] max Maximum input value (in absolute value) to which the scaling is applied [0, 127).
63/// \return Vector of packed 8-bit integers with the scaling result.
64inline int8x16_t scale_s8(int8x16_t a, float sf, uint8_t max)
65{
66 srsran_assert((sf > 0) && (sf <= 1), "Scaling factor out of range.");
67 srsran_assert(max < 127, "Parameter max out of range.");
68
69 if (sf >= .9999) {
70 return a;
71 }
72 // FLOAT2INT = 2^8 = 256
73 static constexpr unsigned FLOAT2INT = 1U << 8U;
74 const int8x16_t MAX_s8 = vdupq_n_s8(max);
75 const int8x16_t MIN_s8 = vdupq_n_s8(-max);
76 static const uint16x8_t mask_even_bytes = vdupq_n_u16(0x00ff);
77 static const uint16x8_t mask_odd_bytes = vdupq_n_u16(0xff00);
78
79 // Scaling factor times FLOAT2INT. Note that the upper bound on sf implies
80 // that (sf * FLOAT2INT) can be represented over 8 bits.
81 uint8x8_t sf_u8 = vdup_n_u8(static_cast<uint8_t>(sf * FLOAT2INT));
82 uint8x16x2_t even_odd_deinterlvd = vuzpq_u8(vreinterpretq_u8_s8(a), vreinterpretq_u8_s8(a));
83 // The above instruction results in:
84 // even bytes of input vector placed in interleaved.val[0]
85 // odd bytes of input vector placed in interleaved.val[1]
86 uint16x8_t p_even_u16 = vmull_u8(vget_low_u8(even_odd_deinterlvd.val[0]), sf_u8);
87 uint16x8_t p_odd_u16 = vmull_u8(vget_low_u8(even_odd_deinterlvd.val[1]), sf_u8);
88 // Shift right MSB byte in every element of p_even_u16, mask it and xor with odd bytes stored in p_odd_u16.
89 p_odd_u16 = vandq_u16(p_odd_u16, mask_odd_bytes);
90 p_even_u16 = vandq_u16(vshrq_n_u16(p_even_u16, 8), mask_even_bytes);
91 int8x16_t result = vreinterpretq_s8_u16(vorrq_u16(p_even_u16, p_odd_u16));
92
93 // Replace values corresponding to "large" inputs with their original values.
94 uint8x16_t mask_u8 = vcgtq_s8(a, MAX_s8);
95 result = vbslq_s8(mask_u8, a, result);
96 mask_u8 = vcgtq_s8(MIN_s8, a);
97 result = vbslq_s8(mask_u8, a, result);
98 return result;
99}
