167enum class integrity_protection_indication_t { required, preferred, not_needed };
168inline bool from_string(integrity_protection_indication_t& integrity_protection_ind, const std::string& str)
169{
170 if (str == "required") {
171 integrity_protection_ind = integrity_protection_indication_t::required;
172 return true;
173 }
174 if (str == "preferred") {
175 integrity_protection_ind = integrity_protection_indication_t::preferred;
176 return true;
177 }
178 if (str == "not_needed") {
179 integrity_protection_ind = integrity_protection_indication_t::not_needed;
180 return true;
181 }
182 return false;
183}

185enum class confidentiality_protection_indication_t { required, preferred, not_needed };
186inline bool from_string(confidentiality_protection_indication_t& confidentiality_protection_ind, const std::string& str)
187{

189 confidentiality_protection_ind = confidentiality_protection_indication_t::required;
190 return true;
191 }

193 confidentiality_protection_ind = confidentiality_protection_indication_t::preferred;
194 return true;
195 }

203struct security_indication_t {
204 integrity_protection_indication_t integrity_protection_ind;
205 confidentiality_protection_indication_t confidentiality_protection_ind;
206};
207
208/// \brief Checks whether a \c security_result shall be sent.
209///
210/// Helper function to determine whether the \c security_indication shall be replied with a \c security_result by the
211/// peer entity, i.e. if either integrity or confidentiality are set to 'preferred' so the peer entity can decide
212/// according to its capabilities.
213/// Ref: TS 38.413 Sec. 8.2.1.2, TS 38.463 Sec. 8.3.1.2
214///
215/// \param security_indication The security_indication to be checked.
216/// \return True if either integrity or confidentiality are set to 'preferred'; False otherwise.
217inline bool security_result_required(const security_indication_t& security_indication)
218{
219 return security_indication.integrity_protection_ind == integrity_protection_indication_t::preferred ||
220 security_indication.confidentiality_protection_ind == confidentiality_protection_indication_t::preferred;
221}
