53 {
54 srsran_assert(block_detector, "Invalid short block detector.");
55 srsran_assert(code, "Invalid polar code instance.");
56 srsran_assert(rate_dematcher, "Invalid polar rate dematcher.");
57 srsran_assert(decoder, "Invalid polar decoder.");
58 srsran_assert(deallocator, "Invalid polar deallocator.");
59 srsran_assert(crc6, "Invalid CRC6 calculator.");

61 srsran_assert(crc11, "Invalid CRC11 calculator.");
62 srsran_assert(crc11->get_generator_poly() == crc_generator_poly::CRC11, "Invalid CRC11 calculator.");
63 };
