29/// \brief Custom formatter for \c pusch_context.
30template <>
31struct formatter<srsran::pusch_context> {
32public:
33 template <typename ParseContext>
34 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
35 {
36 return helper.parse(ctx);
37 }
38
39 template <typename FormatContext>
40 auto format(const srsran::pusch_context& context, FormatContext& ctx) -> decltype(std::declval<FormatContext>().out())
41 {
42 helper.format_always(ctx, "rnti={}", context.rnti);
43 helper.format_always(ctx, "h_id={}", context.h_id);
44 return ctx.out();
45 }
46
47private:
48 srsran::delimited_formatter helper;
49};
