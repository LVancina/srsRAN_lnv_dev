34class e1ap_ue_log_prefix
35{
36public:
37 e1ap_ue_log_prefix(ue_index_t ue_index,
38 gnb_cu_up_ue_e1ap_id_t cu_up_ue_e1ap_id = gnb_cu_up_ue_e1ap_id_t::invalid,
39 gnb_cu_cp_ue_e1ap_id_t cu_cp_ue_e1ap_id = gnb_cu_cp_ue_e1ap_id_t::invalid)
40 {
41 fmt::memory_buffer buffer;
42 fmt::format_to(
43 buffer,
44 "ue={}{}{}: ",
45 ue_index,
46 cu_up_ue_e1ap_id != gnb_cu_up_ue_e1ap_id_t::invalid ? fmt::format(" cu_up_ue_e1ap_id={}", cu_up_ue_e1ap_id)
47 : "",
48 cu_cp_ue_e1ap_id != gnb_cu_cp_ue_e1ap_id_t::invalid ? fmt::format(" cu_cp_ue_e1ap_id={}", cu_cp_ue_e1ap_id)
49 : "");
50 prefix = srsran::to_c_str(buffer);
51 }
52 const char* to_c_str() const { return prefix.c_str(); }
53
54private:
55 std::string prefix;
56};
