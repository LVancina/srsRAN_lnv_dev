33/// \brief Table used by MAC to convert from RNTI to a value in a thread-safe manner.
34template <typename T, T SentinelValue>
35class rnti_value_table
36{
37public:
38 rnti_value_table() : rnti_to_ue_index_map(std::make_unique<array_type>())
39 {
40 for (unsigned idx = 0; idx < rnti_to_ue_index_map->size(); ++idx) {
41 (*rnti_to_ue_index_map)[idx].store(SentinelValue, std::memory_order_relaxed);
42 }
43 }
44 rnti_value_table(const rnti_value_table&) = delete;
45 rnti_value_table(rnti_value_table&&) = delete;
46 rnti_value_table& operator=(const rnti_value_table&) = delete;
47 rnti_value_table& operator=(rnti_value_table&&) = delete;
48
49 bool add_ue(rnti_t crnti, T value)
50 {
51 srsran_assert(is_crnti(crnti), "Invalid c-rnti={}", crnti);
52 srsran_assert(value != SentinelValue, "Invalid rnti_value_table value={}", value);
53 std::atomic<T>& ue_pos = get(crnti);
54 T prev_ue_idx = ue_pos.exchange(value, std::memory_order_relaxed);
55 if (prev_ue_idx == SentinelValue) {
56 nof_ues_.fetch_add(1, std::memory_order_relaxed);
57 return true;
58 }
59 return false;
60 }
61
62 void rem_ue(rnti_t crnti)
63 {
64 srsran_assert(is_crnti(crnti), "Invalid c-rnti={}", crnti);
65 std::atomic<T>& ue_pos = get(crnti);
66 T prev_ue_idx = ue_pos.exchange(SentinelValue, std::memory_order_relaxed);
67 if (prev_ue_idx != SentinelValue) {
68 nof_ues_.fetch_sub(1, std::memory_order_relaxed);
69 srsran_assert(nof_ues_.load(std::memory_order_relaxed) <= RNTI_RANGE, "Invalid rnti_table state");
70 }
71 }
72
73 /// Get an estimate of the current number of UEs present in the table.
74 size_t nof_ues() const { return nof_ues_.load(std::memory_order_relaxed); }
75
76 /// Checks whether passed RNTI is registered as a UE.
77 bool has_rnti(rnti_t crnti) const { return this->operator[](crnti) != SentinelValue; }
78
79 T operator[](rnti_t crnti) const
80 {
81 srsran_sanity_check(is_crnti(crnti), "Invalid c-rnti={}", crnti);
82 const std::atomic<T>& ue_pos = get(crnti);
83 return ue_pos.load(std::memory_order_relaxed);
84 }
85
86private:
87 static constexpr rnti_t MIN_CRNTI = rnti_t::MIN_CRNTI;
88 static constexpr rnti_t MAX_CRNTI = rnti_t::MAX_CRNTI;
89 static constexpr int RNTI_RANGE = to_value(rnti_t::MAX_CRNTI) + 1 - to_value(rnti_t::MIN_CRNTI);
90 using array_type = std::array<std::atomic<T>, RNTI_RANGE>;
91
92 std::atomic<T>& get(rnti_t rnti) { return (*rnti_to_ue_index_map)[to_value(rnti) - to_value(rnti_t::MIN_CRNTI)]; }
93 const std::atomic<T>& get(rnti_t rnti) const
94 {
95 return (*rnti_to_ue_index_map)[to_value(rnti) - to_value(rnti_t::MIN_CRNTI)];
96 }
97
98 // Table of RNTI -> UE index with size 65535.
99 std::unique_ptr<array_type> rnti_to_ue_index_map;
100
101 std::atomic<size_t> nof_ues_{0};
102};
