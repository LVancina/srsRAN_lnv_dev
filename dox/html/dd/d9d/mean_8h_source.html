33/// \brief Computes the mean of the values in the given sequence.
34///
35/// \tparam T Type of the input sequence, compatible with a span of arithmetic or cf_t typed values.
36/// \param x Sequence of values to compute the mean of.
37/// \return The mean value of the given sequence.
38/// \remark For sequences of floating-point (including complex) values, the output type is the same as the value type
39/// of the input. For integer-valued sequences, the output type is \c float.
40template <typename T>
41auto mean(const T& x)
42{
43 static_assert(detail::is_arithmetic_span_compatible<T>::value || detail::is_complex_span_compatible<T>::value,
44 "The input type is not compatible with a span of arithmetic/cf_t values.");
45
46 using DataType = typename T::value_type;
47
48 using MeanType1 = typename std::conditional<is_complex<typename T::value_type>::value, DataType, float>::type;
49 using MeanType = typename std::conditional<std::is_floating_point<DataType>::value, DataType, MeanType1>::type;
50
51 srsran_assert(!x.empty(), "The input span cannot be empty.");
52
53 return std::accumulate(x.begin(), x.end(), static_cast<MeanType>(0)) / static_cast<MeanType>(x.size());
54}
