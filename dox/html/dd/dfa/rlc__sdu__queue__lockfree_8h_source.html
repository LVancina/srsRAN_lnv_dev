31/// \brief Lockfree RLC SDU queue
32///
33/// This class is used as FIFO buffer for RLC SDUs from upper layers that shall be transmitted by the RLC Tx entity.
34/// It implements SPSC (single producer single consumer) semantics, i.e. allows concurrent access by one upper-layer
35/// thread and one lower-layer thread. Internally, this class wraps a lock-free SPSC queue to store the actual SDUs and
36/// involves additional bookkeeping to support discard of SDUs and to track the amount of buffered SDUs and bytes.
37///
38/// From the perspective of the upper-layer thread this class provides methods to write RLC SDUs (with or without PDCP
39/// SN) and to discard SDUs by their PDCP SN. SDUs without PDCP SN cannot be discarded.
40/// SDUs that are marked as discarded remain in the internal queue until they are popped (and dropped) via \c read.
41/// Writing to the queue fails if the internal queue is either full or the queue already contains an SDU (whether valid
42/// or marked as discarded) with the same value for [PDCP_SN mod capacity].
43///
44/// From the perspective of the lower-layer thread it provides methods to read RLC SDUs and to query the total number of
45/// buffered SDUs and bytes.
46/// SDUs are read in the same order as they were written into the queue. There is no reordering by PDCP SN.
47class rlc_sdu_queue_lockfree
48{
49public:
50 explicit rlc_sdu_queue_lockfree(uint16_t capacity_, rlc_bearer_logger& logger_) : logger(logger_), capacity(capacity_)
51 {
52 sdu_states = std::make_unique<std::atomic<uint32_t>[]>(capacity);
53 sdu_sizes = std::make_unique<std::atomic<size_t>[]>(capacity);
54
55 for (uint16_t i = 0; i < capacity; i++) {
56 sdu_states[i].store(STATE_FREE, std::memory_order_relaxed);
57 }
58
59 queue = std::make_unique<
60 concurrent_queue<rlc_sdu, concurrent_queue_policy::lockfree_spsc, concurrent_queue_wait_policy::non_blocking>>(
61 capacity);
62 }
63
64 /// \brief Writes an RLC SDU (with optional PDCP SN) to the queue, if possible.
65 ///
66 /// This function may be called by the upper-layer thread.
67 ///
68 /// The write fails (returns false) in the following cases:
69 /// - The internal queue is full.
70 /// - Another SDU with same value of [PDCP_SN mod capacity] exists (either valid or discarded) in the queue.
71 ///
72 /// \param sdu The RLC SDU that shall be written.
73 /// \return True if the RLC SDU was successfully written to the queue, otherwise false.
74 bool write(rlc_sdu sdu)
75 {
76 // if the SDU has a PDCP SN, first check the slot is available
77 optional<uint32_t> pdcp_sn = sdu.pdcp_sn;
78 const size_t sdu_size = sdu.buf.length();
79 if (pdcp_sn.has_value()) {
80 const uint32_t pdcp_sn_value = sdu.pdcp_sn.value();
81 // load slot state (memory_order_acquire ensures sdu_size is written after this)
82 uint32_t slot_state = sdu_states[pdcp_sn_value % capacity].load(std::memory_order_acquire);
83 if (slot_state != STATE_FREE) {
84 logger.log_debug("SDU queue failed to enqueue pdcp_sn={}. Slot holds pdcp_sn={}", pdcp_sn_value, slot_state);
85 return false;
86 }
87
88 // slot is free, we can safely store the SDU size
89 sdu_sizes[pdcp_sn_value % capacity].store(sdu_size, std::memory_order_relaxed);
90
91 // allocate slot by writing the PDCP SN into it (memory_order_relaxed, don't care about order because of "release"
92 // semantics of the push to queue)
93 sdu_states[pdcp_sn_value % capacity].store(pdcp_sn_value, std::memory_order_relaxed);
94 }
95
96 // push SDU to queue
97 bool pushed = queue->try_push(std::move(sdu));
98 if (not pushed) {
99 logger.log_debug("SDU queue failed to enqueue pdcp_sn={}. Queue is full", pdcp_sn);
100 // if the SDU has a PDCP SN, release the slot (memory_order_relaxed, don't care about value of sdu_size)
101 if (pdcp_sn.has_value()) {
102 sdu_states[pdcp_sn.value() % capacity].store(STATE_FREE, std::memory_order_relaxed);
103 }
104 return false;
105 }
106
107 // update totals
108 n_sdus.fetch_add(1, std::memory_order_relaxed);
109 n_bytes.fetch_add(sdu_size, std::memory_order_relaxed);
110 return true;
111 }
112
113 /// \brief Marks an RLC SDU as "discarded", if possible.
114 ///
115 /// This function may be called by the upper-layer thread.
116 ///
117 /// The function fails (returns false) in the following cases:
118 /// - The SDU with the PDCP SN is already marked as discarded.
119 /// - The SDU with the PDCP SN is not in the queue.
120 ///
121 /// In case of success, the number of SDUs/bytes will be updated immediately.
122 /// However, the actual SDU remains in the internal queue until it is popped and finally discarded via \c read.
123 /// Meanwhile, no further SDU with the same value of [PDCP_SN mod capacity] can be added to the queue.
124 ///
125 /// \param pdcp_sn The PDCP SN of the SDU that shall be discarded.
126 /// \return True if the RLC SDU was successfully discarded, otherwise false.
127 bool try_discard(uint32_t pdcp_sn)
128 {
129 uint32_t expected_state = pdcp_sn;
130 uint32_t desired_state = STATE_DISCARDED;
131 // set slot state as "discarded" only if it holds the PDCP SN; otherwise load the existing state
132 // - on success: memory_order_acquire ensures sdu_size is up to date
133 // - on failure: memory_order_relaxed, don't care about value of sdu_size
134 bool success = sdu_states[pdcp_sn % capacity].compare_exchange_strong(
135 expected_state, desired_state, std::memory_order_acquire, std::memory_order_relaxed);
136 uint32_t sdu_size = sdu_sizes[pdcp_sn % capacity].load(std::memory_order_relaxed);
137
138 if (not success) {
139 switch (expected_state) {
140 case STATE_DISCARDED:
141 logger.log_debug("SDU queue cannot discard pdcp_sn={}. Slot is already discarded", pdcp_sn);
142 break;
143 case STATE_FREE:
144 logger.log_debug("SDU queue cannot discard pdcp_sn={}. Slot is already free", pdcp_sn);
145 break;
146 default:
147 logger.log_debug("SDU queue cannot discard pdcp_sn={}. Slot holds pdcp_sn={}", pdcp_sn, expected_state);
148 break;
149 }
150 return false;
151 }
152
153 // update totals
154 n_sdus.fetch_sub(1, std::memory_order_relaxed);
155 n_bytes.fetch_sub(sdu_size, std::memory_order_relaxed);
156 return true;
157 }
158
159 /// \brief Reads an RLC SDU (with optional PDCP SN) from the queue, if possible.
160 ///
161 /// This function may be called by the lower-layer thread.
162 ///
163 /// The read fails (returns false) in the following cases:
164 /// - The internal queue only contains SDUs that are marked as discarded.
165 /// - The internal queue is empty.
166 ///
167 /// Each call of this function pops and drops SDUs that are marked as discarded from the internal queue in a loop
168 /// until a valid SDU is popped (return true) or the queue is empty (returns false).
169 ///
170 /// \param sdu Reference to a \c rlc_sdu object that will be filled with the read RLC SDU.
171 /// \return True if an RLC SDU was successfully read from the queue, otherwise false.
172 bool read(rlc_sdu& sdu)
173 {
174 bool sdu_is_valid = true;
175 do {
176 // first try to pop front (SDU can still get discarded from upper layers)
177 bool popped = queue->try_pop(sdu);
178 if (not popped) {
179 // queue is empty
180 return false;
181 }
182
183 if (sdu.pdcp_sn.has_value()) {
184 // Check if the SDU is still valid (i.e. the PDCP SN was not already discarded) and release the slot
185 const uint32_t pdcp_sn = sdu.pdcp_sn.value();
186 sdu_is_valid = check_and_release(pdcp_sn); // this also updates totals
187 } else {
188 // SDUs without PDCP SN are alway valid as they can't be discarded
189 sdu_is_valid = true;
190
191 // update totals
192 n_sdus.fetch_sub(1, std::memory_order_relaxed);
193 n_bytes.fetch_sub(sdu.buf.length(), std::memory_order_relaxed);
194 }
195 // try again if SDU is not valid
196 } while (not sdu_is_valid);
197 return true;
198 }
199
200 /// \brief Reads the number of buffered SDUs that are not marked as discarded.
201 ///
202 /// This function may be called by any thread.
203 ///
204 /// \return The number of buffered SDUs that are not marked as discarded.
205 uint32_t size_sdus() const { return n_sdus.load(std::memory_order_relaxed); }
206
207 /// \brief Reads the number of buffered SDU bytes that are not marked as discarded.
208 ///
209 /// This function may be called by any thread.
210 ///
211 /// \return The number of buffered SDU bytes that are not marked as discarded.
212 uint32_t size_bytes() const { return n_bytes.load(std::memory_order_relaxed); }
213
214 /// \brief Checks if the internal queue is empty.
215 ///
216 /// This function may be called by any thread.
217 ///
218 /// \return True if the internal queue is empty, otherwise false.
219 bool is_empty() { return queue->empty(); }
220
221 /// \brief Checks if the internal queue is full.
222 ///
223 /// This function may be called by any thread.
224 ///
225 /// \return True if the internal queue is full, otherwise false.
226 bool is_full() { return queue->size() >= capacity; }
227
228private:
229 /// \brief Checks if the RLC SDU with a PDCP SN is valid (i.e. not marked as discarded) and marks the respective slot
230 /// as free.
231 ///
232 /// This private function may be called by \c read by the lower-layer thread.
233 ///
234 /// The function fails (returns false) in the following cases:
235 /// - The SDU with the PDCP SN is already marked as discarded.
236 /// - The SDU with the PDCP SN is not in the queue.
237 ///
238 /// The number of SDUs/bytes will be updated unless the SDU is marked as discarded (in that case it was already
239 /// updated by \c try_discard).
240 ///
241 /// \param pdcp_sn The PDCP SN of the SDU that shall be checked and released.
242 /// \return True if the RLC SDU with given PDCP SN is valid, otherwise false.
243 bool check_and_release(uint32_t pdcp_sn)
244 {
245 uint32_t sdu_size = sdu_sizes[pdcp_sn % capacity].load(std::memory_order_relaxed);
246 // free the slot (memory_order_release ensures sdu_size is read before the state is set to free)
247 uint32_t old_state = sdu_states[pdcp_sn % capacity].exchange(STATE_FREE, std::memory_order_release);
248
249 bool sdu_is_valid;
250 if (old_state == pdcp_sn) {
251 sdu_is_valid = true;
252 logger.log_debug("SDU queue released valid pdcp_sn={}", pdcp_sn);
253 }
254 if (old_state != pdcp_sn) {
255 sdu_is_valid = false;
256 switch (old_state) {
257 case STATE_DISCARDED:
258 logger.log_debug("SDU queue released discarded pdcp_sn={}", pdcp_sn);
259 return false; // totals were already updated on discard
260 case STATE_FREE:
261 logger.log_error("SDU queue error: Invalid release of pdcp_sn={}. Slot was already free", pdcp_sn);
262 break;
263 default:
264 logger.log_error("SDU queue error: Invalid release of pdcp_sn={}. Slot holds pdcp_sn={}", pdcp_sn, old_state);
265 break;
266 }
267 }
268
269 // update totals
270 n_sdus.fetch_sub(1, std::memory_order_relaxed);
271 n_bytes.fetch_sub(sdu_size, std::memory_order_relaxed);
272 return sdu_is_valid;
273 }
274
275 static constexpr uint32_t STATE_FREE = 0xffffffff; ///< Sentinel value to mark a slot as free.
276 static constexpr uint32_t STATE_DISCARDED = 0xfffffffe; ///< Sentinel value to mark a slot as discarded.
277
278 rlc_bearer_logger& logger;
279
280 uint16_t capacity;
281 std::atomic<uint32_t> n_bytes = {0};
282 std::atomic<uint32_t> n_sdus = {0};
283
284 std::unique_ptr<
285 concurrent_queue<rlc_sdu, concurrent_queue_policy::lockfree_spsc, concurrent_queue_wait_policy::non_blocking>>
286 queue;
287
288 std::unique_ptr<std::atomic<uint32_t>[]> sdu_states;
289 std::unique_ptr<std::atomic<size_t>[]> sdu_sizes;
290};
