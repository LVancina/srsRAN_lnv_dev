40/// \brief Lower PHY PRACH processor worker subcomponent.
41///
42/// The worker contains a finite-state machine with four possible states: idle, wait, collecting and processing. See
43/// \ref states for more information.
44///
45/// The implementation assumes that:
46/// - handle_request() and is_available() are called from a first thread, and
47/// - process_symbol() is called from a second, different thread.
48class prach_processor_worker
49{
50private:
51 /// Worker internal states.
52 enum class states {
53 /// The worker has not been configured yet with any request or the previously configured request has already been
54 /// processed. It transitions to \c wait when a request is set.
55 idle,
56 /// A PRACH request is configured and the worker is waiting for the start of the PRACH occasion window. The worker
57 /// transitions to \c collecting if it detects the beginning of the PRACH occasion window for the configured PRACH
58 /// context. When in this state, the worker also detects late requests, in which case it notifies the event and
59 /// transitions back to \c idle.
60 wait,
61 /// The PRACH occasion window started and baseband samples are collected. Once the internal buffer has been filled
62 /// with the PRACH occasion window samples, the worker transitions to \c processing for the demodulation phase.
63 collecting,
64 /// The PRACH occasion has been captured and it is processed by the asynchronous task executor. The worker
65 /// transitions to \c idle as soon as the PRACH occasion is processed and notified.
66 processing
67 };
68
69 /// OFDM PRACH demodulator.
70 std::unique_ptr<ofdm_prach_demodulator> demodulator;
71 /// Asynchronous task executor.
72 task_executor& async_task_executor;
73 /// Sampling rate in Hz.
74 unsigned sampling_rate_Hz = 0;
75 /// Temporary baseband buffer of size \ref TEMP_BASEBAND_BUFFER_SIZE.
76 baseband_gateway_buffer_dynamic temp_baseband;
77 /// PRACH processor notifier.
78 prach_processor_notifier* notifier = nullptr;
79 /// Current worker state.
80 std::atomic<states> state = {states::idle};
81 /// Current context.
82 prach_buffer_context prach_context;
83 /// Current PRACH buffer.
84 prach_buffer* buffer = nullptr;
85 /// Current PRACH occasion window length.
86 unsigned window_length = 0;
87 /// Current number of collected samples.
88 unsigned nof_samples = 0;
89
90 /// Runs state \c wait.
91 void run_state_wait(const baseband_gateway_buffer_reader& samples,
92 const prach_processor_baseband::symbol_context& context);
93
94 /// Runs state \c collecting.
95 void run_state_collecting(const baseband_gateway_buffer_reader& samples,
96 const prach_processor_baseband::symbol_context& context);
97
98 /// Accumulates \c samples in the internal buffer.
99 void accumulate_samples(const baseband_gateway_buffer_reader& samples);
100
101public:
102 /// Creates a PRACH processor worker.
103 prach_processor_worker(std::unique_ptr<ofdm_prach_demodulator> demodulator_,
104 task_executor& async_task_executor_,
105 sampling_rate srate,
106 unsigned max_nof_ports) :
107 demodulator(std::move(demodulator_)),
108 async_task_executor(async_task_executor_),
109 sampling_rate_Hz(srate.to_Hz()),
110 temp_baseband(max_nof_ports, prach_constants::MAX_WINDOW_LENGTH.to_samples(sampling_rate_Hz))
111 {
112 srsran_assert(sampling_rate_Hz && prach_constants::MAX_WINDOW_LENGTH.is_sample_accurate(sampling_rate_Hz),
113 "Invalid sampling rate of {} Hz.",
114 sampling_rate_Hz);
115 }
116
117 /// Connects the worker with the given PRACH processor notifier.
118 void connect(prach_processor_notifier& notifier_) { notifier = &notifier_; }
119
120 /// \brief Handles a PRACH occasion request.
121 /// \param[in] buffer PRACH buffer.
122 /// \param[in] context PRACH occasion context.
123 void handle_request(prach_buffer& buffer, const prach_buffer_context& context);
124
125 /// \brief Processes an OFDM symbol.
126 /// \param[in] samples Baseband samples.
127 /// \param[in] context OFDM symbol context.
128 void process_symbol(const baseband_gateway_buffer_reader& samples,
129 const prach_processor_baseband::symbol_context& context);
130
131 /// \brief Determines whether the PRACH processor worker is available.
132 ///
133 /// A PRACH processor is available when it is \c idle. See \ref states for more information regarding the PRACH worker
134 /// internal states.
135 bool is_available() const { return state == states::idle; }
136};
