76 re_prb_mask get_dmrs_prb_mask(unsigned nof_cdm_groups_without_data) const
77 {
78 if (value == TYPE1) {
79 bounded_bitset<2> cdm_groups_pattern(2);
80 cdm_groups_pattern.fill(0, nof_cdm_groups_without_data);
81 bounded_bitset<6> groups(6);
82 groups.fill(0, 6, true);
83 return groups.kronecker_product<2>(cdm_groups_pattern);
84 }
85
86 bounded_bitset<6> cdm_groups_pattern(6);
87 cdm_groups_pattern.fill(0, 2 * nof_cdm_groups_without_data);
88 bounded_bitset<2> groups(2);
89 groups.fill(0, 2, true);
90 return groups.kronecker_product<6>(cdm_groups_pattern);
91 }

102 re_pattern get_dmrs_pattern(unsigned bwp_start_rb,
103 unsigned bwp_size_rb,
104 unsigned nof_cdm_groups_without_data,
105 symbol_slot_mask symbol_mask) const
106 {
107 // Temporal pattern.
108 re_pattern dmrs_pattern;
109
110 // Validate symbol mask size.
111 srsran_assert(symbol_mask.size() <= dmrs_pattern.symbols.size(),
112 "The symbol mask size {} exceeds the maximum size {}.",
113 symbol_mask.size(),
114 dmrs_pattern.symbols.size());
115
116 // Generate DMRS pattern.
117 dmrs_pattern.prb_mask.resize(bwp_start_rb + bwp_size_rb);
118 dmrs_pattern.prb_mask.fill(bwp_start_rb, bwp_start_rb + bwp_size_rb);
119 dmrs_pattern.re_mask = get_dmrs_prb_mask(nof_cdm_groups_without_data);
120 dmrs_pattern.symbols = symbol_mask;
121
122 return dmrs_pattern;
123 }
