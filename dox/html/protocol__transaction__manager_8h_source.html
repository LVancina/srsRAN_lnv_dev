51template <typename ResponseType>
52class protocol_transaction
53{
54 using manager_type = protocol_transaction_manager<ResponseType>;
55
56public:
57 using response_type = ResponseType;
58 using outcome_type = expected<ResponseType, protocol_transaction_failure>;
59 using event_type = manual_event<outcome_type>;
60 using awaiter_type = typename event_type ::awaiter_type;
61
62 protocol_transaction() = default;
63 protocol_transaction(manager_type& parent_, unsigned transaction_id_, event_type& ev_) :
64 parent(&parent_), transaction_id(transaction_id_), ev(&ev_)
65 {
66 }
67 protocol_transaction(const protocol_transaction&) = delete;
68 protocol_transaction(protocol_transaction&& other) noexcept :
69 parent(std::exchange(other.parent, nullptr)),
70 transaction_id(other.transaction_id),
71 ev(std::exchange(other.ev, nullptr))
72 {
73 }
74 ~protocol_transaction() { release(); }
75 protocol_transaction& operator=(const protocol_transaction&) = delete;
76 protocol_transaction& operator=(protocol_transaction&& other) noexcept
77 {
78 release();
79 parent = std::exchange(other.parent, nullptr);
80 transaction_id = other.transaction_id;
81 ev = std::exchange(other.ev, nullptr);
82 return *this;
83 }
84
86 unsigned id() const { return transaction_id; }
87
89 bool valid() const { return parent != nullptr; }
90
92 bool complete() const
93 {
94 srsran_assert(valid(), "Trying to check completion of invalid transaction");
95 return ev->is_set();
96 }
97
99 bool has_response() const
100 {
101 srsran_assert(valid(), "Trying to check completion of invalid transaction");
102 return complete() and ev->get().has_value();
103 }
104
106 bool aborted() const
107 {
108 srsran_assert(valid(), "Trying to check completion of invalid transaction");
109 return complete() and ev->get().is_error();
110 }
111
113 const protocol_transaction_failure& failure_cause() const
114 {
115 srsran_assert(aborted(), "Trying to get result of invalid transaction");
116 return ev->get().error();
117 }
118
120 const response_type& response() const&
121 {
122 srsran_assert(has_response(), "Trying to get response of transaction without a response");
123 return ev->get().value();
124 }
125 response_type response() &&
126 {
127 srsran_assert(has_response(), "Trying to get result of transaction without a response");
128 return std::move(ev)->get();
129 }
130
131 const outcome_type& outcome() const&
132 {
133 srsran_assert(complete(), "Trying to fetch outcome of incomplete transaction");
134 return ev->get();
135 }
136 outcome_type outcome() &&
137 {
138 srsran_assert(complete(), "Trying to fetch outcome of incomplete transaction");
139 return std::move(ev)->get();
140 }
141
143 awaiter_type get_awaiter()
144 {
145 srsran_assert(valid(), "Trying to fetch awaiter of invalid transaction");
146 return ev->get_awaiter();
147 }
148
150 void release()
151 {
152 if (parent != nullptr) {
153 parent->remove_transaction(transaction_id);
154 parent = nullptr;
155 ev = nullptr;
156 }
157 }
158
159private:
160 manager_type* parent = nullptr;
161 protocol_transaction_id_t transaction_id = invalid_protocol_transaction_id;
162 event_type* ev = nullptr;
163};

172template <typename T>
173class protocol_transaction_manager
174{
175 using outcome_type = typename protocol_transaction<T>::outcome_type;
176
177public:
179 const static std::chrono::milliseconds max_timeout;
180
181 explicit protocol_transaction_manager(protocol_transaction_id_t nof_transaction_ids_, timer_factory timer_service_) :
182 nof_transaction_ids(nof_transaction_ids_), timer_service(timer_service_)
183 {
184 }
185 ~protocol_transaction_manager()
186 {
187 if (not running_transactions.empty()) {
188 srslog::fetch_basic_logger("ALL").error("Protocol transaction manager destroyed with {} running transactions",
189 running_transactions.size());
190 }
191 }
192
195 SRSRAN_NODISCARD protocol_transaction<T> create_transaction(std::chrono::milliseconds time_to_cancel = max_timeout)
196 {
197 // Allocate a new transaction id for the protocol transaction.
198 unsigned transaction_id = invalid_protocol_transaction_id;
199 for (unsigned count = 0; count < nof_transaction_ids and transaction_id == invalid_protocol_transaction_id;
200 ++count) {
201 if (running_transactions.count(next_transaction_id) == 0) {
202 transaction_id = next_transaction_id;
203 }
204 next_transaction_id = (next_transaction_id + 1) % nof_transaction_ids;
205 }
206 if (transaction_id == invalid_protocol_transaction_id) {
207 srslog::fetch_basic_logger("ALL").error("Failed to allocate new transaction id");
208 return {};
209 }
210
211 return create_transaction(transaction_id, time_to_cancel);
212 }
213
215 SRSRAN_NODISCARD protocol_transaction<T> create_transaction(protocol_transaction_id_t transaction_id,
216 std::chrono::milliseconds time_to_cancel = max_timeout)
217 {
218 // Create new transaction instance.
219 auto ret = running_transactions.emplace(
220 std::piecewise_construct, std::forward_as_tuple(transaction_id), std::forward_as_tuple());
221 if (not ret.second) {
222 srslog::fetch_basic_logger("ALL").error("Failed to allocate transaction id={}", transaction_id);
223 return {};
224 }
225
226 // Create timer, set timeout and callback, and start running it.
227 unique_timer& timer = ret.first->second.timer;
228 timer = timer_service.create_timer();
229 timer.set(time_to_cancel, [this, transaction_id](timer_id_t /*unused*/) {
230 if (not set_transaction_outcome(transaction_id, protocol_transaction_failure::timeout)) {
231 srslog::fetch_basic_logger("ALL").warning("Transaction id={} timeout but transaction is already completed",
232 transaction_id);
233 }
234 });
235 timer.run();
236
237 return {*this, ret.first->first, ret.first->second.event};
238 }

244 template <typename U>
245 SRSRAN_NODISCARD bool set_response(unsigned transaction_id, U&& result)
246 {
247 static_assert(std::is_convertible<U, T>::value, "Invalid transaction response type being set");
248 return set_transaction_outcome(transaction_id, std::forward<U>(result));
249 }
250
252 bool cancel_transaction(unsigned transaction_id)
253 {
254 return set_transaction_outcome(transaction_id, protocol_transaction_failure::cancel);
255 }
256
257private:
258 friend class protocol_transaction<T>;
259
260 template <typename U>
261 SRSRAN_NODISCARD bool set_transaction_outcome(unsigned transaction_id, U&& result)
262 {
263 auto it = running_transactions.find(transaction_id);
264 if (it == running_transactions.end()) {
265 // No transaction with provided transaction id exists.
266 return false;
267 }
268
269 // Stop running timer, so it doesn't try to overwrite result.
270 it->second.timer.stop();
271
272 // Store result.
273 it->second.event.set(std::forward<U>(result));
274 return true;
275 }
276
277 void remove_transaction(protocol_transaction_id_t transaction_id)
278 {
279 auto it = running_transactions.find(transaction_id);
280 if (it == running_transactions.end()) {
281 srslog::fetch_basic_logger("ALL").warning("Trying to remove non-existing transaction id={}", transaction_id);
282 return;
283 }
284 running_transactions.erase(it);
285 }

287 const protocol_transaction_id_t nof_transaction_ids;
288 timer_factory timer_service;

290 struct transaction_context {
291 unique_timer timer;
292 manual_event<outcome_type> event;
293 };
294
295 protocol_transaction_id_t next_transaction_id{0};
296
297 std::unordered_map<protocol_transaction_id_t, transaction_context> running_transactions;

335template <typename SuccessResp, typename FailureResp = no_fail_response_path>
336class protocol_transaction_outcome_observer
337{
338 static_assert(not std::is_same<SuccessResp, protocol_transaction_failure>::value, "Invalid Success Response");
339 static_assert(not std::is_same<FailureResp, protocol_transaction_failure>::value, "Invalid Success Response");
340
341 using observer_type = async_single_event_observer<variant<SuccessResp, FailureResp, protocol_transaction_failure>>;
342
343public:
344 using success_response_type = SuccessResp;
345 using failure_response_type = FailureResp;
346 using event_source_type = protocol_transaction_event_source<success_response_type, failure_response_type>;
347 using awaiter_type = typename observer_type::awaiter_type;
348
351 void subscribe_to(event_source_type& publisher) { observer.subscribe_to(publisher.ev_source); }
352
355 void subscribe_to(event_source_type& publisher, std::chrono::milliseconds time_to_cancel)
356 {
357 observer.subscribe_to(publisher.ev_source, time_to_cancel, protocol_transaction_failure::timeout);
358 }
359
361 bool connected() const { return observer.connected(); }
362
364 bool complete() const { return observer.complete(); }
365
367 bool successful() const { return complete() and variant_holds_alternative<success_response_type>(observer.result()); }
368
370 bool failed() const { return complete() and variant_holds_alternative<failure_response_type>(observer.result()); }
371
374 bool protocol_transaction_failed() const
375 {
376 return complete() and variant_holds_alternative<protocol_transaction_failure>(observer.result());
377 }
378
380 bool timeout_expired() const
381 {
382 return protocol_transaction_failed() and
383 variant_get<protocol_transaction_failure>(observer.result()) == protocol_transaction_failure::timeout;
384 }
385
387 const success_response_type& response() const
388 {
389 srsran_assert(successful(), "Trying to fetch incorrect transaction result");
390 return variant_get<success_response_type>(observer.result());
391 }
392 const failure_response_type& failure() const
393 {
394 srsran_assert(failed(), "Trying to fetch incorrect transaction result");
395 return variant_get<failure_response_type>(observer.result());
396 }
397
399 awaiter_type get_awaiter() { return observer.get_awaiter(); }
400
401private:
402 observer_type observer;
403};
