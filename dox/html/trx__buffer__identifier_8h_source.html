32class trx_buffer_identifier
33{
35 using id_type = uint32_t;
36
38 static constexpr id_type invalid_id = std::numeric_limits<id_type>::max();
40 static constexpr id_type unknown_id = invalid_id - 1;
41
43 static constexpr uint32_t to_id(uint16_t rnti, uint8_t harq)
44 {
45 return (static_cast<id_type>(rnti) << 8U) | static_cast<id_type>(harq);
46 }
47
49 constexpr trx_buffer_identifier(uint32_t id_) : id(id_) {}
50
51public:
53 static constexpr trx_buffer_identifier invalid() { return invalid_id; }
54
56 static constexpr trx_buffer_identifier unknown() { return unknown_id; }
57
59 explicit trx_buffer_identifier() noexcept : id(invalid_id) {}
60
62 explicit constexpr trx_buffer_identifier(uint16_t rnti, uint8_t harq) noexcept : id(to_id(rnti, harq)) {}
63
65 constexpr bool operator==(const trx_buffer_identifier& other) const { return other.id == id; }
66
68 constexpr bool operator!=(const trx_buffer_identifier& other) const { return !(*this == other); }
69
71 uint16_t get_rnti() const { return static_cast<uint16_t>(id >> 8); }
72
74 uint8_t get_harq() const { return static_cast<uint8_t>(id & 0xff); }
75
76private:
78 id_type id;
79};
