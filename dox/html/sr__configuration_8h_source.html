65inline unsigned sr_periodicity_to_slot(sr_periodicity period)
66{
67 if (period == sr_periodicity::sym_2 or period == sr_periodicity::sym_6_or_7) {
68 return static_cast<unsigned>(sr_periodicity::sl_1);
