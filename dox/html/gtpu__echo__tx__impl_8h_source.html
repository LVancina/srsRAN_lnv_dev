45 void send_echo_request(const sockaddr_storage& dst_addr) override
46 {
47 byte_buffer buf;
48
49 // TODO: Add optional IE for "private extension"
50
51 gtpu_header hdr = {};
52 hdr.flags.version = GTPU_FLAGS_VERSION_V1;
53 hdr.flags.protocol_type = GTPU_FLAGS_GTP_PROTOCOL;
54 hdr.flags.ext_hdr = false;
55 hdr.flags.seq_number = true;
56 hdr.message_type = GTPU_MSG_ECHO_REQUEST;
57 hdr.length = 0; // length will be computed automatically
58 hdr.teid = GTPU_PATH_MANAGEMENT_TEID;
59 hdr.seq_number = sn_next;
60
61 bool write_ok = gtpu_write_header(buf, hdr, logger);
62
63 if (!write_ok) {
64 logger.log_error("Discarded SDU. Cause: Error writing GTP-U header of echo request.");
65 return;
66 }
67
68 sn_next++;
69 logger.log_info(buf.begin(), buf.end(), "TX echo request. sn={} pdu_len={}", hdr.seq_number, buf.length());
70 send_pdu(std::move(buf), dst_addr);
71 }
72
73 void send_echo_response(const sockaddr_storage& dst_addr, uint16_t sn) override
74 {
75 byte_buffer buf;
76
77 // Add information element for "recovery" for backward compatibility. See TS 29.281 Sec. 8.2
78 gtpu_ie_recovery ie_recovery = {};
79 gtpu_write_ie_recovery(buf, ie_recovery, logger);
80
81 // TODO: Add optional IE for "private extension"
82
83 // Add header
84 gtpu_header hdr = {};
85 hdr.flags.version = GTPU_FLAGS_VERSION_V1;
86 hdr.flags.protocol_type = GTPU_FLAGS_GTP_PROTOCOL;
87 hdr.flags.ext_hdr = false;
88 hdr.flags.seq_number = true;
89 hdr.message_type = GTPU_MSG_ECHO_RESPONSE;
90 hdr.length = 0; // length will be computed automatically
91 hdr.teid = GTPU_PATH_MANAGEMENT_TEID;
92 hdr.seq_number = sn; // responses copy the SN of the request, TS 29.281 Sec. 4.3.1
93
94 bool write_ok = gtpu_write_header(buf, hdr, logger);
95
96 if (!write_ok) {
97 logger.log_error("Discarded SDU. Cause: Error writing GTP-U header of echo response. sn={}", sn);
98 return;
99 }
100
101 logger.log_info(buf.begin(), buf.end(), "TX echo response. sn={} pdu_len={}", sn, buf.length());
102 send_pdu(std::move(buf), dst_addr);
103 }
104
105 void handle_echo_response(const sockaddr_storage& src_addr, uint16_t sn) override
106 {
107 logger.log_info("handle_echo_response was called. sn={}", sn);
108 // TODO: disarm t3_response
109 }
110
111private:
112 uint32_t sn_next = 0;
113};
