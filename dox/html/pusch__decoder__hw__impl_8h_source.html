40// Maximum accepted transport block size.
41static constexpr unsigned MAX_TBS = 1277992;
42
43// Number of bits in the long CRC. A CRC of this length is used either for TB CRCs, when the TB is longer than
44// MAX_BITS_CRC16, or as a codeblock CRC, when the TB consists of multiple codeblocks.
45static constexpr unsigned LONG_CRC_LENGTH = 24;

48class pusch_decoder_hw_impl : public pusch_decoder, private pusch_decoder_buffer
49{
50public:
52 struct sch_crc {
54 std::unique_ptr<crc_calculator> crc16;
56 std::unique_ptr<crc_calculator> crc24A;
58 std::unique_ptr<crc_calculator> crc24B;
59 };
60
69 pusch_decoder_hw_impl(std::unique_ptr<ldpc_segmenter_rx> seg,
70 sch_crc& c,
71 std::unique_ptr<hal::hw_accelerator_pusch_dec> hw) :
72 segmenter(std::move(seg)),
73 crc_set({std::move(c.crc16), std::move(c.crc24A), std::move(c.crc24B)}),
74 decoder(std::move(hw)),
75 softbits_buffer(pdsch_constants::CODEWORD_MAX_SIZE.value())
76 {
77 srsran_assert(segmenter, "Invalid LDPC segmenter factory.");
78 srsran_assert(crc_set.crc16, "Invalid CRC16 calculator.");
79 srsran_assert(crc_set.crc24A, "Invalid CRC24A calculator.");
80 srsran_assert(crc_set.crc24B, "Invalid CRC24B calculator.");
81 srsran_assert(crc_set.crc16->get_generator_poly() == crc_generator_poly::CRC16,
82 "Not a CRC generator of type CRC16.");
83 srsran_assert(crc_set.crc24A->get_generator_poly() == crc_generator_poly::CRC24A,
84 "Not a CRC generator of type CRC24A.");
85 srsran_assert(crc_set.crc24B->get_generator_poly() == crc_generator_poly::CRC24B,
86 "Not a CRC generator of type CRC24B.");
87 srsran_assert(decoder, "Invalid hardware-accelerated PUSCH decoder.");
88 }
