36struct ul_cplane_context {
38 cplane_radio_application_header radio_hdr;
40 uint16_t prb_start;
42 unsigned nof_prb;
44 uint8_t nof_symbols;
45};
46
48class uplink_cplane_context_repository
49{
51 static constexpr unsigned SFN_MAX_VALUE = 1U << 8;
52
53 using repo_entry = std::array<ul_cplane_context, MAX_SUPPORTED_EAXC_ID_VALUE>;
54
55 std::vector<repo_entry> repo;
56 unsigned size = 0;
57 //: TODO: make this lock free
58 mutable std::mutex mutex;
59
61 ul_cplane_context& entry(slot_point slot, unsigned eaxc)
62 {
63 slot_point entry_slot(slot.numerology(), slot.sfn() % SFN_MAX_VALUE, slot.slot_index());
64 unsigned index = entry_slot.system_slot() % size;
65 return repo[index][eaxc];
66 }
67
69 const ul_cplane_context& entry(slot_point slot, unsigned eaxc) const
70 {
71 slot_point entry_slot(slot.numerology(), slot.sfn() % SFN_MAX_VALUE, slot.slot_index());
72 unsigned index = entry_slot.system_slot() % size;
73 return repo[index][eaxc];
74 }
75
76public:
77 explicit uplink_cplane_context_repository(unsigned size_) : repo(size_), size(size_) {}
78
80 void add(slot_point slot, unsigned eaxc, const ul_cplane_context& context)
81 {
82 std::lock_guard<std::mutex> lock(mutex);
83 entry(slot, eaxc) = context;
84 }
85
88 expected<ul_cplane_context> get(slot_point slot, unsigned symbol, filter_index_type filter_index, unsigned eaxc) const
89 {
90 std::lock_guard<std::mutex> lock(mutex);
91
92 const ul_cplane_context& cp_param = entry(slot, eaxc);
93 if (symbol >= cp_param.radio_hdr.start_symbol &&
94 symbol < (cp_param.radio_hdr.start_symbol + cp_param.nof_symbols) &&
95 filter_index == cp_param.radio_hdr.filter_index) {
96 return cp_param;
97 }
98
99 return default_error_t{};
100 }
101};
