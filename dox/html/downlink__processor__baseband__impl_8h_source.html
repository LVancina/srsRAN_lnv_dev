59class baseband_symbol_buffer
60{
61public:
63 baseband_symbol_buffer(unsigned nof_ports_, unsigned symbol_size) :
64 buffer(nof_ports_, symbol_size), nof_ports(nof_ports_)
65 {
66 empty = true;
67 }
68
70 void clear()
71 {
72 // Mark buffer as empty.
73 empty = true;
74 }
75
86 unsigned read(baseband_gateway_buffer_writer& out, baseband_gateway_timestamp timestamp)
87 {
88 srsran_assert(!empty, "Attempting to read from empty buffer.");
89 srsran_assert(out.get_nof_channels() == nof_ports, "Number of ports does not match.");
90
91 // Clear the buffer if attempting to read samples in the past.
92 if (timestamp < symbol_start_timestamp) {
93 clear();
94 return 0;
95 }
96
97 unsigned read_index = timestamp - symbol_start_timestamp;
98
99 // Clear the buffer if attempting to read samples in the future.
100 if (read_index >= buffer.get_nof_samples()) {
101 clear();
102 return 0;
103 }
104
105 // Select the minimum among the samples to process and the number of stored samples that have not been read.
106 unsigned nof_available_samples = buffer.get_nof_samples() - read_index;
107 unsigned count = std::min(out.get_nof_samples(), nof_available_samples);
108
109 // For each port, concatenate samples.
110 for (unsigned i_port = 0; i_port != nof_ports; ++i_port) {
111 // Select view of the temporary buffer.
112 span<const cf_t> temp_buffer_src = buffer[i_port].subspan(read_index, count);
113
114 // Select view of the output samples.
115 span<cf_t> temp_buffer_dst = out.get_channel_buffer(i_port).first(count);
116
117 srsvec::copy(temp_buffer_dst, temp_buffer_src);
118 }
119
120 // Clear the buffer if all available samples have been read.
121 if (count == nof_available_samples) {
122 clear();
123 }
124
125 // Return the number of read samples.
126 return count;
127 }
128
138 baseband_gateway_buffer_writer& write_symbol(baseband_gateway_timestamp symbol_timestamp, unsigned symbol_size)
139 {
140 srsran_assert(empty, "Attempting to write into non-empty buffer.");
141 srsran_assert(symbol_size > 0, "Symbol size cannot be zero.");
142
143 buffer.resize(symbol_size);
144 symbol_start_timestamp = symbol_timestamp;
145 empty = false;
146
147 return buffer.get_writer();
148 }
149
157 unsigned get_nof_available_samples(baseband_gateway_timestamp timestamp) const
158 {
159 // No available samples if the buffer is empty, or for timestamps in the past or in the future.
160 if (empty || (timestamp < symbol_start_timestamp) ||
161 (timestamp >= symbol_start_timestamp + buffer.get_nof_samples())) {
162 return 0;
163 }
164
165 unsigned read_index = timestamp - symbol_start_timestamp;
166 return buffer.get_nof_samples() - read_index;
167 }
168
169private:
171 baseband_gateway_buffer_dynamic buffer;
172
174 baseband_gateway_timestamp symbol_start_timestamp;
176 unsigned nof_ports;
178 bool empty;
179};
