41struct os_thread_realtime_priority_tag {};
42class os_thread_realtime_priority
43 : public strong_type<int, struct os_thread_realtime_priority_tag, strong_equality, strong_comparison>
44{
45 static int max_posix_prio()
46 {
47 // Subtract one to the priority offset to avoid scheduling threads with the highest priority that could contend with
48 // OS critical tasks.
49 static int value = sched_get_priority_max(SCHED_FIFO) - 1;
50 return value;
51 }
52 static int min_posix_prio()
53 {
54 static int value = sched_get_priority_min(SCHED_FIFO);
55 return value;
56 }
57
58public:
60 os_thread_realtime_priority() : strong_type(0) {}
61
63 static os_thread_realtime_priority no_realtime() { return os_thread_realtime_priority{}; }
64
66 static os_thread_realtime_priority max() { return os_thread_realtime_priority{max_posix_prio()}; }
67
69 static os_thread_realtime_priority min() { return os_thread_realtime_priority{min_posix_prio()}; }
70
72 os_thread_realtime_priority& operator+=(int prio_diff)
73 {
74 *this = os_thread_realtime_priority{value() + prio_diff};
75 return *this;
76 }
77 os_thread_realtime_priority operator+(int prio_diff) const
78 {
79 os_thread_realtime_priority ret{*this};
80 return ret += prio_diff;
81 }
82
84 os_thread_realtime_priority& operator-=(int prio_diff)
85 {
86 *this = os_thread_realtime_priority{value() - prio_diff};
87 return *this;
88 }
89 os_thread_realtime_priority operator-(int prio_diff) const
90 {
91 os_thread_realtime_priority ret{*this};
92 return ret -= prio_diff;
93 }
94
96 int native() const { return strong_type::value(); }
97
98 int native_sched_policy() const { return *this == no_realtime() ? SCHED_OTHER : SCHED_FIFO; }
99
100private:
104 explicit os_thread_realtime_priority(int posix_prio) :
105 strong_type(std::max(min_posix_prio(), std::min(max_posix_prio(), posix_prio)))
106 {
107 }
108};
