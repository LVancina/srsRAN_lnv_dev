79 std::pair<uint64_t, uint64_t> get_max_meas_time_ns() const
80 {
81 auto it = std::max_element(
82 benchmark_results.begin(), benchmark_results.end(), [](const benchmark_result& a, const benchmark_result& b) {
83 return a.measurements.back() < b.measurements.back();
84 });
85 return {it->measurements.back(), it->size};
86 }
87
89 unsigned get_percentile_width_time(double scaling = 1.0) const
90 {
91 uint64_t max_meas_ns = get_max_meas_time_ns().first;
92 unsigned percentile_width =
93 static_cast<unsigned>(std::ceil(std::log10(static_cast<double>(max_meas_ns) * scaling)));
94 percentile_width = std::max(percentile_width, 6U);
95 return percentile_width + 2;
96 }
97
99 unsigned get_percentile_width_throughput(double scaling) const
100 {
101 std::pair<uint64_t, uint64_t> max_meas = get_max_meas_time_ns();
102 max_meas.first = std::max(max_meas.first, 1UL);
103 double throughput_max = convert_to_throughput(max_meas.first, max_meas.second);
104
105 unsigned percentile_width = static_cast<unsigned>(std::ceil(std::log10(throughput_max * scaling))) + 2U;
106 percentile_width = std::max(percentile_width, 8U);
107 return percentile_width;
108 }
109
111 unsigned get_description_width() const
112 {
113 unsigned descr_width = std::max_element(benchmark_results.begin(),
114 benchmark_results.end(),
115 [](const benchmark_result& a, const benchmark_result& b) {
116 return a.description.size() < b.description.size();
117 })
118 ->description.size();
119 descr_width = std::max(descr_width, 13U);
120 return descr_width;
121 }

156 static inline double convert_to_throughput(uint64_t time_ns, size_t size)
157 {
158 time_ns = std::max(time_ns, 1UL);
159
160 // To get a more precise result with one decimal place, the throughput is calculated in hundreds of thousands of
161 // elements per second and then divided by ten when it is converted to a floating point number.
162 uint64_t throughput = (size * 10000UL) / time_ns;
163
164 // Converts to floating point.
165 return static_cast<double>(throughput) * 0.1;
166 }

186 void print_percentiles_time(std::string units = "nanoseconds", double scaling = 1.0) const
187 {
188 if (benchmark_results.empty()) {
189 return;
190 }
191
192 // Get width plus 2 characters for 1 decimal position.
193 unsigned percentile_width = get_percentile_width_time(scaling);
194 unsigned descr_width = get_description_width();
195
196 print_percentile_header(descr_width, percentile_width, units);
197 for (const benchmark_result& result : benchmark_results) {
198 fmt::print(" {:{}}|{:{}.1f}|{:{}.1f}|{:{}.1f}|{:{}.1f}|{:{}.1f}|{:{}.1f}|{:{}.1f}|\n",
199 result.description,
200 descr_width,
201 result.measurements[static_cast<size_t>(nof_repetitions * 0.5)] * scaling,
202 percentile_width,
203 result.measurements[static_cast<size_t>(nof_repetitions * 0.75)] * scaling,
204 percentile_width,
205 result.measurements[static_cast<size_t>(nof_repetitions * 0.9)] * scaling,
206 percentile_width,
207 result.measurements[static_cast<size_t>(nof_repetitions * 0.99)] * scaling,
208 percentile_width,
209 result.measurements[static_cast<size_t>(nof_repetitions * 0.999)] * scaling,
210 percentile_width,
211 result.measurements[static_cast<size_t>(nof_repetitions * 0.9999)] * scaling,

220 void print_percentiles_throughput(const std::string& units, double scaling = 1.0) const
221 {
222 if (benchmark_results.empty()) {
223 fmt::print("No benchmark results for {}.\n", title);
224 return;
225 }
226
227 unsigned percentile_width = get_percentile_width_throughput(scaling);
228 unsigned descr_width = get_description_width();
229
230 print_percentile_header(descr_width, percentile_width, "mega" + units + " per second");
231 for (const benchmark_result& result : benchmark_results) {
232 fmt::print(
233 " {:{}}|{:{}.1f}|{:{}.1f}|{:{}.1f}|{:{}.1f}|{:{}.1f}|{:{}.1f}|{:{}.1f}|\n",
234 result.description,
235 descr_width,
236 convert_to_throughput(result.measurements[static_cast<size_t>(nof_repetitions * 0.5)], result.size) * scaling,
237 percentile_width,
238 convert_to_throughput(result.measurements[static_cast<size_t>(nof_repetitions * 0.75)], result.size) *
239 scaling,
240 percentile_width,
241 convert_to_throughput(result.measurements[static_cast<size_t>(nof_repetitions * 0.9)], result.size) * scaling,
242 percentile_width,
243 convert_to_throughput(result.measurements[static_cast<size_t>(nof_repetitions * 0.99)], result.size) *
244 scaling,
245 percentile_width,
246 convert_to_throughput(result.measurements[static_cast<size_t>(nof_repetitions * 0.999)], result.size) *
247 scaling,
248 percentile_width,
249 convert_to_throughput(result.measurements[static_cast<size_t>(nof_repetitions * 0.9999)], result.size) *
250 scaling,
251 percentile_width,

265 template <typename Func, typename PostFunc = do_nothing_functor>
266 void new_measure(const std::string& description,
267 uint64_t size,
268 Func&& function,
269 PostFunc&& post_func = do_nothing_functor{})
270 {
271 benchmark_result result;
272 result.description = description;
273 result.size = size;
274 result.measurements.reserve(nof_repetitions);
275
276 for (uint64_t rep = 0; rep != nof_repetitions; ++rep) {
277 auto start = std::chrono::high_resolution_clock::now();
278 function();
279 auto end = std::chrono::high_resolution_clock::now();
280 result.measurements.push_back(std::chrono::duration_cast<std::chrono::nanoseconds>(end - start).count());
281 post_func();
282 }
283
284 std::sort(result.measurements.begin(), result.measurements.end());
285
286 benchmark_results.push_back(std::move(result));
287 }

302

304 context_function();
305 auto start = std::chrono::high_resolution_clock::now();
306 function();
307 auto end = std::chrono::high_resolution_clock::now();
308 result.measurements.push_back(std::chrono::duration_cast<std::chrono::nanoseconds>(end - start).count());
309 }
310
311 std::sort(result.measurements.begin(), result.measurements.end());
312
313 benchmark_results.push_back(std::move(result));
314 }
