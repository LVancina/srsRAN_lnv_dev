59 bool contains(ue_index_t ue_index) const

61 if (ue_index_to_ue_f1ap_id.find(ue_index) == ue_index_to_ue_f1ap_id.end()) {

106 f1ap_ue_context& add_ue(ue_index_t ue_index, gnb_cu_ue_f1ap_id_t cu_ue_id)
107 {
108 srsran_assert(ue_index != ue_index_t::invalid, "Invalid ue_index={}", ue_index);
109 srsran_assert(cu_ue_id != gnb_cu_ue_f1ap_id_t::invalid, "Invalid cu_ue_id={}", cu_ue_id);
110
111 logger.debug("ue={} cu_ue_f1ap_id={}: Adding F1AP UE context", ue_index, cu_ue_id);
112 ues.emplace(
113 std::piecewise_construct, std::forward_as_tuple(cu_ue_id), std::forward_as_tuple(ue_index, cu_ue_id, timers));
114 ue_index_to_ue_f1ap_id.emplace(ue_index, cu_ue_id);
115 return ues.at(cu_ue_id);

131 void remove_ue(ue_index_t ue_index)
132 {
133 srsran_assert(ue_index != ue_index_t::invalid, "Invalid ue_index={}", ue_index);
134
135 if (ue_index_to_ue_f1ap_id.find(ue_index) == ue_index_to_ue_f1ap_id.end()) {
136 logger.warning("ue={}: gNB-CU-UE-F1AP-ID not found", ue_index);
137 return;
138 }
139
140 // Remove UE from lookup
141 gnb_cu_ue_f1ap_id_t cu_ue_id = ue_index_to_ue_f1ap_id.at(ue_index);
142 ue_index_to_ue_f1ap_id.erase(ue_index);
143
144 if (ues.find(cu_ue_id) == ues.end()) {
145 logger.warning("cu_ue_f1ap_id={}: F1AP UE context not found", cu_ue_id);
146 return;
147 }
148
149 logger.debug("ue={} cu_ue_f1ap_id={}: Removing F1AP UE context", ue_index, cu_ue_id);
150 ues.erase(cu_ue_id);
151 }

168 gnb_cu_ue_f1ap_id_t next_gnb_cu_ue_f1ap_id()
169 {
170 // return invalid when no cu ue f1ap id is available
171 if (ue_index_to_ue_f1ap_id.size() == MAX_NOF_UES_PER_DU) {
172 return gnb_cu_ue_f1ap_id_t::invalid;
173 }
174
175 // Check if the next_cu_ue_f1ap_id is available
176 if (ues.find(next_cu_ue_f1ap_id) == ues.end()) {
177 gnb_cu_ue_f1ap_id_t ret = next_cu_ue_f1ap_id;
178 // increase the next cu ue f1ap id
179 increase_next_cu_ue_f1ap_id();
180 return ret;
181 }
182
183 // Find holes in the allocated IDs by iterating over all ids starting with the next_cu_ue_f1ap_id to find the
184 // available id
185 while (true) {
186 // Only iterate over ue_index_to_ue_f1ap_id (size=MAX_NOF_UES_PER_DU)
187 // to avoid iterating over all possible values of gnb_cu_ue_f1ap_id_t (size=2^32-1)
188 auto it = std::find_if(ue_index_to_ue_f1ap_id.begin(), ue_index_to_ue_f1ap_id.end(), [this](auto& u) {
189 return u.second == next_cu_ue_f1ap_id;
190 });
191
192 // return the id if it is not already used
193 if (it == ue_index_to_ue_f1ap_id.end()) {
194 gnb_cu_ue_f1ap_id_t ret = next_cu_ue_f1ap_id;
195 // increase the next cu ue f1ap id
196 increase_next_cu_ue_f1ap_id();
197 return ret;
198 }
199
200 // increase the next cu ue f1ap id and try again
201 increase_next_cu_ue_f1ap_id();
202 }
203
204 return gnb_cu_ue_f1ap_id_t::invalid;
205 }

207protected:
208 gnb_cu_ue_f1ap_id_t next_cu_ue_f1ap_id = gnb_cu_ue_f1ap_id_t::min;
209
210private:
211 timer_factory timers;
212 srslog::basic_logger& logger;
213
214 inline void increase_next_cu_ue_f1ap_id()
215 {
216 if (next_cu_ue_f1ap_id == gnb_cu_ue_f1ap_id_t::max) {
217 // reset cu ue f1ap id counter
218 next_cu_ue_f1ap_id = gnb_cu_ue_f1ap_id_t::min;
219 } else {
220 // increase cu ue f1ap id counter
221 next_cu_ue_f1ap_id = int_to_gnb_cu_ue_f1ap_id(gnb_cu_ue_f1ap_id_to_uint(next_cu_ue_f1ap_id) + 1);
222 }
223 }
