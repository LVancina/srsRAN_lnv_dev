43inline void fill_asn1_ue_context_modification_request(asn1::f1ap::ue_context_mod_request_s& asn1_request,
44 const f1ap_ue_context_modification_request& request)
45{
46 // sp cell id
47 if (request.sp_cell_id.has_value()) {
48 asn1_request->sp_cell_id_present = true;
49 asn1_request->sp_cell_id.nr_cell_id.from_number(request.sp_cell_id.value().nci);
50 asn1_request->sp_cell_id.plmn_id.from_string(request.sp_cell_id.value().plmn_hex);
51 }
52
53 // serv cell idx
54 if (request.serv_cell_idx.has_value()) {
55 asn1_request->serv_cell_idx_present = true;
56 asn1_request->serv_cell_idx = request.serv_cell_idx.value();
57 }
58
59 // sp cell ul cfg
60 if (request.sp_cell_ul_cfg.has_value()) {
61 asn1_request->sp_cell_ul_cfg_present = true;
62 asn1_request->sp_cell_ul_cfg = cell_ul_cfg_to_asn1(request.sp_cell_ul_cfg.value());
63 }
64
65 // drx cycle
66 if (request.drx_cycle.has_value()) {
67 asn1_request->drx_cycle_present = true;
68 asn1::number_to_enum(asn1_request->drx_cycle.long_drx_cycle_len, request.drx_cycle.value().long_drx_cycle_len);
69
70 if (request.drx_cycle.value().short_drx_cycle_len.has_value()) {
71 asn1_request->drx_cycle.short_drx_cycle_len_present = true;
72 asn1::number_to_enum(asn1_request->drx_cycle.short_drx_cycle_len,
73 request.drx_cycle.value().short_drx_cycle_len.value());
74 }
75
76 if (request.drx_cycle.value().short_drx_cycle_timer.has_value()) {
77 asn1_request->drx_cycle.short_drx_cycle_timer_present = true;
78 asn1_request->drx_cycle.short_drx_cycle_timer = request.drx_cycle.value().short_drx_cycle_timer.value();
79 }
80 }
81
82 // cu to du rrc info
83 if (request.cu_to_du_rrc_info.has_value()) {
84 asn1_request->cu_to_du_rrc_info_present = true;
85 asn1_request->cu_to_du_rrc_info.cg_cfg_info = request.cu_to_du_rrc_info.value().cg_cfg_info.copy();
86 asn1_request->cu_to_du_rrc_info.ue_cap_rat_container_list =
87 request.cu_to_du_rrc_info.value().ue_cap_rat_container_list.copy();
88 asn1_request->cu_to_du_rrc_info.meas_cfg = request.cu_to_du_rrc_info.value().meas_cfg.copy();
89 }
90
91 // tx action ind
92 if (request.tx_action_ind.has_value()) {
93 asn1_request->tx_action_ind_present = true;
94 asn1_request->tx_action_ind = f1ap_tx_action_ind_to_asn1(request.tx_action_ind.value());
95 }
96
97 // res coordination transfer container
98 if (!request.res_coordination_transfer_container.empty()) {
99 asn1_request->res_coordination_transfer_container_present = true;
100 asn1_request->res_coordination_transfer_container = request.res_coordination_transfer_container.copy();
101 }
102
103 // rrc recfg complete ind
104 if (request.rrc_recfg_complete_ind.has_value()) {
105 asn1_request->rrc_recfg_complete_ind_present = true;
106 asn1_request->rrc_recfg_complete_ind = f1ap_rrc_recfg_complete_ind_to_asn1(request.rrc_recfg_complete_ind.value());
107 }
108
109 // rrc container
110 if (!request.rrc_container.empty()) {
111 asn1_request->rrc_container_present = true;
112 asn1_request->rrc_container = request.rrc_container.copy();
113 }
114
115 // scell to be setup mod list
116 if (!request.scell_to_be_setup_mod_list.empty()) {
117 asn1_request->scell_to_be_setup_mod_list_present = true;
118 for (const auto& scell_to_be_setup_mod_item : request.scell_to_be_setup_mod_list) {

120 asn1_scell_to_be_setup_mod_item_container;

123 auto& asn1_scell_to_be_setup_mod_item =
124 asn1_scell_to_be_setup_mod_item_container.value().scell_to_be_setup_mod_item();
125

127
128 asn1_request->scell_to_be_setup_mod_list.push_back(asn1_scell_to_be_setup_mod_item_container);
129 }

137 asn1_scell_to_be_remd_item_container;
138 asn1_scell_to_be_remd_item_container.load_info_obj(ASN1_F1AP_ID_SCELL_TO_BE_REMD_ITEM);

330inline void fill_f1ap_ue_context_modification_response(f1ap_ue_context_modification_response& res,
331 const asn1::f1ap::ue_context_mod_resp_s& asn1_response)
332{
333 res.success = true;
334
335 // DUtoCURRCInformation
336 if (asn1_response->du_to_cu_rrc_info_present) {
337 res.du_to_cu_rrc_info.cell_group_cfg = asn1_response->du_to_cu_rrc_info.cell_group_cfg.copy();
338 res.du_to_cu_rrc_info.meas_gap_cfg = asn1_response->du_to_cu_rrc_info.meas_gap_cfg.copy();
339 res.du_to_cu_rrc_info.requested_p_max_fr1 = asn1_response->du_to_cu_rrc_info.requested_p_max_fr1.copy();
340 }
341
342 // Add DRBs setup mod list
343 if (asn1_response->drbs_setup_mod_list_present) {
344 for (auto asn1_drb_setup_mod_list_item : asn1_response->drbs_setup_mod_list) {
345 auto& asn1_drb_mod_item = asn1_drb_setup_mod_list_item.value().drbs_setup_mod_item();
346
347 f1ap_drbs_setup_mod_item drb_setup_mod_item = asn1_to_f1ap_drbs_setup_mod_item(asn1_drb_mod_item);
348
349 res.drbs_setup_mod_list.emplace(drb_setup_mod_item.drb_id, drb_setup_mod_item);
350 }
351 }
352
353 // Add DRBs modified list
354 if (asn1_response->drbs_modified_list_present) {
355 for (auto asn1_drbs_modified_list_item : asn1_response->drbs_modified_list) {
356 auto& asn1_drb_mod_item = asn1_drbs_modified_list_item.value().drbs_modified_item();
357
358 f1ap_drbs_setup_mod_item drb_setup_mod_item = asn1_to_f1ap_drbs_setup_mod_item(asn1_drb_mod_item);
359
360 res.drbs_modified_list.emplace(drb_setup_mod_item.drb_id, drb_setup_mod_item);
361 }
362 }
363
364 // Add SRBs failed to be setup mod list
365 if (asn1_response->srbs_failed_to_be_setup_mod_list_present) {
366 for (auto asn1_srbs_failed_setup_mod_list_item : asn1_response->srbs_failed_to_be_setup_mod_list) {
367 auto& asn1_srb_failed_item = asn1_srbs_failed_setup_mod_list_item.value().srbs_failed_to_be_setup_mod_item();
368
369 f1ap_srbs_failed_to_be_setup_mod_item srb_failed_item =
370 asn1_to_f1ap_srbs_failed_to_be_setup_mod_item(asn1_srb_failed_item);
371
372 res.srbs_failed_to_be_setup_mod_list.emplace(srb_failed_item.srb_id, srb_failed_item);
373 }
374 }
375
376 // Add DRBs failed to be setup mod list
377 if (asn1_response->drbs_failed_to_be_setup_mod_list_present) {
378 for (auto asn1_drbs_failed_setup_mod_list_item : asn1_response->drbs_failed_to_be_setup_mod_list) {
379 auto& asn1_drb_failed_item = asn1_drbs_failed_setup_mod_list_item.value().drbs_failed_to_be_setup_mod_item();
380
381 f1ap_drbs_failed_to_be_setup_mod_item drb_failed_item =
382 asn1_to_f1ap_drbs_failed_to_be_setup_mod_item(asn1_drb_failed_item);
383
384 res.drbs_failed_to_be_setup_mod_list.emplace(drb_failed_item.drb_id, drb_failed_item);
385 }
386 }
387
388 // Add SCell failed to be setup mod list
389 if (asn1_response->scell_failedto_setup_mod_list_present) {
390 for (auto asn1_scell_failed_setup_mod_list_item : asn1_response->scell_failedto_setup_mod_list) {
391 auto& asn1_scell_failed_item = asn1_scell_failed_setup_mod_list_item.value().scell_failedto_setup_mod_item();
392
393 f1ap_scell_failed_to_setup_mod_item scell_failed_item;
394 scell_failed_item.scell_id = cgi_from_asn1(asn1_scell_failed_item.scell_id);
395 if (asn1_scell_failed_item.cause_present) {
396 scell_failed_item.cause = f1ap_asn1_to_cause(asn1_scell_failed_item.cause);
397 }
398 res.scell_failed_to_setup_mod_list.push_back(scell_failed_item);
399 }
400 }
401
402 // Add DRBs failed to be modified list
403 if (asn1_response->drbs_failed_to_be_modified_list_present) {
404 for (auto asn1_drbs_failed_modified_list_item : asn1_response->drbs_failed_to_be_modified_list) {
405 auto& asn1_drb_failed_item = asn1_drbs_failed_modified_list_item.value().drbs_failed_to_be_modified_item();
406
407 f1ap_drbs_failed_to_be_setup_mod_item drb_failed_item =
408 asn1_to_f1ap_drbs_failed_to_be_setup_mod_item(asn1_drb_failed_item);
409
410 res.drbs_failed_to_be_modified_list.emplace(drb_failed_item.drb_id, drb_failed_item);
411 }
412 }
413
414 // Add inactivity monitoring response
415 if (asn1_response->inactivity_monitoring_resp_present) {
416 res.inactivity_monitoring_resp = asn1::enum_to_bool(asn1_response->inactivity_monitoring_resp);
417 }
418
419 // Add C-RNTI
420 if (asn1_response->c_rnti_present) {
421 res.c_rnti = to_rnti(asn1_response->c_rnti);
422 }
423
424 // Add associated SCell list
425 if (asn1_response->associated_scell_list_present) {
426 for (auto asn1_associated_scell_list_item : asn1_response->associated_scell_list) {
427 auto& asn1_associated_scell_item = asn1_associated_scell_list_item.value().associated_scell_item();
428
429 f1ap_associated_scell_item associated_scell_item;
