39inline sched_cell_configuration_request_message
40make_default_sched_cell_configuration_request(const config_helpers::cell_config_builder_params_extended& params = {})
41{
42 sched_cell_configuration_request_message sched_req{};
43 sched_req.cell_index = to_du_cell_index(0);
44 sched_req.pci = params.pci;
45 sched_req.scs_common = params.scs_common;
46 sched_req.dl_carrier = config_helpers::make_default_dl_carrier_configuration(params);
47 sched_req.ul_carrier = config_helpers::make_default_ul_carrier_configuration(params);
48 sched_req.dl_cfg_common = config_helpers::make_default_dl_config_common(params);
49 sched_req.ul_cfg_common = config_helpers::make_default_ul_config_common(params);
50 sched_req.ssb_config = config_helpers::make_default_ssb_config(params);
51 sched_req.tdd_ul_dl_cfg_common = params.tdd_ul_dl_cfg_common;
52
53 // The CORESET duration of 3 symbols is only permitted if dmrs-typeA-Position is set to 3. Refer TS 38.211, 7.3.2.2.
54 const pdcch_type0_css_coreset_description coreset0_desc = pdcch_type0_css_coreset_get(
55 *params.band, params.ssb_scs, params.scs_common, *params.coreset0_index, params.k_ssb->value());
56 sched_req.dmrs_typeA_pos =
57 coreset0_desc.nof_symb_coreset == 3U ? dmrs_typeA_position::pos3 : dmrs_typeA_position::pos2;
58
59 sched_req.nof_beams = 1;
60
61 // SIB1 parameters.
62 sched_req.coreset0 = *params.coreset0_index;
63 sched_req.searchspace0 = params.search_space0_index;
64 sched_req.sib1_payload_size = 101; // Random size.
65
66 pucch_builder_params default_pucch_builder_params = du_cell_config{}.pucch_cfg;
67 default_pucch_builder_params.nof_ue_pucch_f1_res_harq = 3;
68 default_pucch_builder_params.nof_ue_pucch_f2_res_harq = 6;
69 default_pucch_builder_params.nof_sr_resources = 2;
70
71 sched_req.pucch_guardbands = config_helpers::build_pucch_guardbands_list(
72 default_pucch_builder_params, sched_req.ul_cfg_common.init_ul_bwp.generic_params.crbs.length());
73
74 if (params.csi_rs_enabled) {
75 csi_helper::csi_builder_params csi_params{};
76 csi_params.pci = params.pci;
77 csi_params.nof_rbs = sched_req.dl_cfg_common.init_dl_bwp.generic_params.crbs.length();
78 csi_params.nof_ports = params.nof_dl_ports;
79 csi_params.csi_rs_period = csi_helper::get_max_csi_rs_period(params.scs_common);
80
81 sched_req.zp_csi_rs_list = csi_helper::make_periodic_zp_csi_rs_resource_list(csi_params);
82 sched_req.nzp_csi_rs_res_list = csi_helper::make_nzp_csi_rs_resource_list(csi_params);
83 }
84
85 return sched_req;
86}

96inline sched_ue_creation_request_message
97create_default_sched_ue_creation_request(const cell_config_builder_params& params = {},
98 const std::initializer_list<lcid_t>& lcid_to_cfg = {})
99{
100 sched_ue_creation_request_message msg{};
101
102 msg.ue_index = to_du_ue_index(0);
103 msg.crnti = to_rnti(0x4601);
104
105 scheduling_request_to_addmod sr_0{.sr_id = scheduling_request_id::SR_ID_MIN, .max_tx = sr_max_tx::n64};
106 msg.cfg.sched_request_config_list.emplace();
107 msg.cfg.sched_request_config_list->push_back(sr_0);
108
109 msg.cfg.cells.emplace();
110 msg.cfg.cells->push_back(create_test_initial_ue_spcell_cell_config(params));
111
112 msg.cfg.lc_config_list.emplace();
113 msg.cfg.lc_config_list->resize(2);
114 (*msg.cfg.lc_config_list)[0] = config_helpers::create_default_logical_channel_config(lcid_t::LCID_SRB0);
115 (*msg.cfg.lc_config_list)[1] = config_helpers::create_default_logical_channel_config(lcid_t::LCID_SRB1);
116 for (lcid_t lcid : lcid_to_cfg) {
117 if (lcid >= lcid_t::LCID_SRB2) {
118 msg.cfg.lc_config_list->push_back(config_helpers::create_default_logical_channel_config(lcid));
119 }
120 }
121
122 return msg;
123}

132 msg.occasions.back().start_symbol = 0;

138}

140inline uplink_config make_test_ue_uplink_config(const config_helpers::cell_config_builder_params_extended& params)
141
142{
143 // > UL Config.
144 uplink_config ul_config{};
145 ul_config.init_ul_bwp.pucch_cfg.emplace();
146
147 // >> PUCCH.
148 auto& pucch_cfg = ul_config.init_ul_bwp.pucch_cfg.value();
149 // PUCCH Resource Set ID 0.
150 auto& pucch_res_set_0 = pucch_cfg.pucch_res_set.emplace_back();
151 pucch_res_set_0.pucch_res_set_id = 0;
152 pucch_res_set_0.pucch_res_id_list.emplace_back(pucch_res_id_t{0, 0});
153 pucch_res_set_0.pucch_res_id_list.emplace_back(pucch_res_id_t{1, 1});
154 pucch_res_set_0.pucch_res_id_list.emplace_back(pucch_res_id_t{2, 2});
155
156 auto& pucch_res_set_1 = pucch_cfg.pucch_res_set.emplace_back();
157 pucch_res_set_1.pucch_res_set_id = 1;
158 pucch_res_set_1.pucch_res_id_list.emplace_back(pucch_res_id_t{3, 3});
159 pucch_res_set_1.pucch_res_id_list.emplace_back(pucch_res_id_t{4, 4});
160 pucch_res_set_1.pucch_res_id_list.emplace_back(pucch_res_id_t{5, 5});
161 pucch_res_set_1.pucch_res_id_list.emplace_back(pucch_res_id_t{6, 6});
162 pucch_res_set_1.pucch_res_id_list.emplace_back(pucch_res_id_t{7, 7});
163 pucch_res_set_1.pucch_res_id_list.emplace_back(pucch_res_id_t{8, 8});
164
165 unsigned nof_rbs = params.cell_nof_crbs;
166
167 // PUCCH resource format 1, for HARQ-ACK.
168 // >>> PUCCH resource 0.
169 pucch_resource res_basic{
170 .res_id = pucch_res_id_t{0, 0}, .starting_prb = nof_rbs - 1, .format = pucch_format::FORMAT_1};
171
172 res_basic.format_params.emplace<pucch_format_1_cfg>(
173 pucch_format_1_cfg{.initial_cyclic_shift = 0, .nof_symbols = 14, .starting_sym_idx = 0, .time_domain_occ = 0});
174 pucch_cfg.pucch_res_list.push_back(res_basic);
175 // >>> PUCCH resource 1.
176 pucch_cfg.pucch_res_list.push_back(res_basic);
177 pucch_resource& res1 = pucch_cfg.pucch_res_list.back();
178 res1.res_id = pucch_res_id_t{1, 1};
179 res1.starting_prb = 1;
180 res1.second_hop_prb.emplace(nof_rbs - res1.starting_prb - 1);
181 // >>> PUCCH resource 2.
182 pucch_cfg.pucch_res_list.push_back(res_basic);
183 pucch_resource& res2 = pucch_cfg.pucch_res_list.back();
184 res2.res_id = pucch_res_id_t{2, 2};
185 res2.second_hop_prb.emplace(1);
186 res2.starting_prb = nof_rbs - res2.second_hop_prb.value() - 1;
187
188 // PUCCH resource format 2, for HARQ-ACK and CSI.
189 // >>> PUCCH resource 3.
190 pucch_resource res_basic_f2{.starting_prb = 2, .format = pucch_format::FORMAT_2};
191 res_basic_f2.res_id = pucch_res_id_t{3, 3};
192 res_basic_f2.format_params.emplace<pucch_format_2_3_cfg>(
193 pucch_format_2_3_cfg{.nof_prbs = 1, .nof_symbols = 2, .starting_sym_idx = 0});
194 pucch_cfg.pucch_res_list.push_back(res_basic_f2);
195 // >>> PUCCH resource 4.
196 pucch_cfg.pucch_res_list.push_back(res_basic_f2);
197 pucch_resource& res4 = pucch_cfg.pucch_res_list.back();
198 res4.res_id = pucch_res_id_t{4, 4};
199 variant_get<pucch_format_2_3_cfg>(res4.format_params).starting_sym_idx = 2;
200 // >>> PUCCH resource 5.
201 pucch_cfg.pucch_res_list.push_back(res_basic_f2);
202 pucch_resource& res5 = pucch_cfg.pucch_res_list.back();
203 res5.res_id = pucch_res_id_t{5, 5};
204 variant_get<pucch_format_2_3_cfg>(res5.format_params).starting_sym_idx = 4;
205 // >>> PUCCH resource 6.
206 pucch_cfg.pucch_res_list.push_back(res_basic_f2);
207 pucch_resource& res6 = pucch_cfg.pucch_res_list.back();
208 res6.res_id = pucch_res_id_t{6, 6};
209 variant_get<pucch_format_2_3_cfg>(res6.format_params).starting_sym_idx = 6;
210 // >>> PUCCH resource 7.
211 pucch_cfg.pucch_res_list.push_back(res_basic_f2);
212 pucch_resource& res7 = pucch_cfg.pucch_res_list.back();
213 res7.res_id = pucch_res_id_t{7, 7};
214 variant_get<pucch_format_2_3_cfg>(res7.format_params).starting_sym_idx = 8;
215 // >>> PUCCH resource 8.
216 pucch_cfg.pucch_res_list.push_back(res_basic_f2);
217 pucch_resource& res8 = pucch_cfg.pucch_res_list.back();
218 res8.res_id = pucch_res_id_t{8, 8};
219 variant_get<pucch_format_2_3_cfg>(res8.format_params).starting_sym_idx = 10;
220 // >>> PUCCH resource 9.
221 pucch_cfg.pucch_res_list.push_back(res_basic_f2);
222 pucch_resource& res9 = pucch_cfg.pucch_res_list.back();
223 res9.res_id = pucch_res_id_t{9, 9};
224 variant_get<pucch_format_2_3_cfg>(res9.format_params).starting_sym_idx = 12;
225
226 // >>> PUCCH resource 10.
227 pucch_cfg.pucch_res_list.push_back(res_basic);
228 pucch_resource& res10 = pucch_cfg.pucch_res_list.back();
229 res10.res_id = pucch_res_id_t{10, 10};
230 res10.starting_prb = 0;
231
232 // >>> PUCCH resource 11.
233 pucch_cfg.pucch_res_list.push_back(res_basic);
234 pucch_resource& res11 = pucch_cfg.pucch_res_list.back();
235 res11.res_id = pucch_res_id_t{11, 11};
236 res11.starting_prb = nof_rbs - 3;
237
238 // TODO: add more PUCCH resources.
239
240 // >>> SR Resource.
241 const unsigned pucch_sr_res_id = pucch_cfg.pucch_res_list.size() - 1;
242 pucch_cfg.sr_res_list.push_back(
243 scheduling_request_resource_config{.sr_res_id = 1,
244 .sr_id = uint_to_sched_req_id(0),
245 .period = sr_periodicity::sl_40,
246 .offset = 0,
247 .pucch_res_id = pucch_res_id_t{pucch_sr_res_id, pucch_sr_res_id}});
248
249 pucch_cfg.format_1_common_param.emplace();
250 pucch_cfg.format_2_common_param.emplace(
251 pucch_common_all_formats{.max_c_rate = max_pucch_code_rate::dot_25, .simultaneous_harq_ack_csi = true});
252
253 // >>> dl-DataToUl-Ack
254 // TS38.213, 9.1.2.1 - "If a UE is provided dl-DataToUL-ACK, the UE does not expect to be indicated by DCI format 1_0
255 // a slot timing value for transmission of HARQ-ACK information that does not belong to the intersection of the set
256 // of slot timing values {1, 2, 3, 4, 5, 6, 7, 8} and the set of slot timing values provided by dl-DataToUL-ACK for
257 // the active DL BWP of a corresponding serving cell.
258 // Inactive for format1_0."
259 // Note2: Only k1 >= 4 supported.
260 nr_band band = params.band.has_value() ? params.band.value() : band_helper::get_band_from_dl_arfcn(params.dl_arfcn);
261 if (band_helper::get_duplex_mode(band) == duplex_mode::FDD) {
262 pucch_cfg.dl_data_to_ul_ack = {params.min_k1};
263 } else {
264 // TDD
265 pucch_cfg.dl_data_to_ul_ack = config_helpers::generate_k1_candidates(params.tdd_ul_dl_cfg_common.value());
266 }
267
268 // > PUSCH config.
269 ul_config.init_ul_bwp.pusch_cfg.emplace(config_helpers::make_default_pusch_config(params));
270 if (band_helper::get_duplex_mode(band) == duplex_mode::TDD) {
271 ul_config.init_ul_bwp.pusch_cfg->pusch_td_alloc_list =
272 config_helpers::generate_k2_candidates(cyclic_prefix::NORMAL, params.tdd_ul_dl_cfg_common.value());
273 }
274
275 // > SRS config.
276 ul_config.init_ul_bwp.srs_cfg.emplace(config_helpers::make_default_srs_config(params));
277
278 return ul_config;
279}

281inline slot_point generate_random_slot_point(subcarrier_spacing scs)
282{
283 static std::array<std::uniform_int_distribution<uint32_t>, NOF_NUMEROLOGIES> scs_dists = {
284 std::uniform_int_distribution<uint32_t>{0, (10240 * get_nof_slots_per_subframe(subcarrier_spacing::kHz15)) - 1},
285 std::uniform_int_distribution<uint32_t>{0, (10240 * get_nof_slots_per_subframe(subcarrier_spacing::kHz30)) - 1},
286 std::uniform_int_distribution<uint32_t>{0, (10240 * get_nof_slots_per_subframe(subcarrier_spacing::kHz60)) - 1},
287 std::uniform_int_distribution<uint32_t>{0, (10240 * get_nof_slots_per_subframe(subcarrier_spacing::kHz120)) - 1},
288 std::uniform_int_distribution<uint32_t>{0, (10240 * get_nof_slots_per_subframe(subcarrier_spacing::kHz240)) - 1}};
289
290 uint32_t count = scs_dists[to_numerology_value(scs)](test_rgen::get());
291 return slot_point{scs, count};
292}
