32class udp_sink : public sink
33{
34public:
35 udp_sink(std::string remote_ip_, unsigned port_, std::unique_ptr<log_formatter> f) :
36 sink(std::move(f)), remote_ip(std::move(remote_ip_)), port(port_)
37 {
38 }
39
40 ~udp_sink() { ::close(socket_fd); }
41
42 udp_sink(const udp_sink& other) = delete;
43 udp_sink& operator=(const udp_sink& other) = delete;
44
45 detail::error_string write(detail::memory_buffer buffer) override
46 {
47 if (socket_fd == -1) {
48 socket_fd = ::socket(AF_INET, SOCK_DGRAM, 0);
49 if (socket_fd < 0) {
50 return "Cannot create UDP socket";
51 }
52
53 remote_address = {};
54 remote_address.sin_family = AF_INET;
55 remote_address.sin_port = ::htons(port);
56 if (::inet_pton(AF_INET, remote_ip.c_str(), &remote_address.sin_addr) < 1) {
57 return "Invalid IP address format";
58 }
59 }
60
61 ::sendto(socket_fd,
62 buffer.data(),
63 buffer.size(),
64 0,
65 reinterpret_cast<const struct sockaddr*>(&remote_address),
66 sizeof(remote_address));
67
68 return {};
69 }
70
71 detail::error_string flush() override { return {}; }
72
73private:
74 std::string remote_ip;
75 unsigned port;
76 int socket_fd = -1;
77 ::sockaddr_in remote_address;
78};
