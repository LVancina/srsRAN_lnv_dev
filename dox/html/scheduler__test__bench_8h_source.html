87 void run_slot(du_cell_index_t cell_idx = to_du_cell_index(0))
88 {
89 srsran_assert(cell_cfg_list.size() > cell_idx, "Invalid cellId={}", cell_idx);
90 logger.set_context(next_slot.sfn(), next_slot.slot_index());
91 last_sched_res_list[cell_idx] = &sched->slot_indication(next_slot, cell_idx);
92 test_scheduler_result_consistency(cell_cfg_list[cell_idx], next_slot, *last_sched_res_list[cell_idx]);
93 ++next_slot;

96 template <typename StopCondition>
97 bool run_slot_until(const StopCondition& cond_func, unsigned slot_timeout = 1000)
98 {
99 unsigned count = 0;
100 for (; count < slot_timeout; ++count) {
101 run_slot();
102 if (cond_func()) {
103 break;
104 }
105 }
106 return count < slot_timeout;
107 }
