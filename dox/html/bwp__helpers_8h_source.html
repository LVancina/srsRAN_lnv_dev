40inline unsigned prb_to_crb(const bwp_configuration& bwp_cfg, unsigned prb)
41{
42 srsran_sanity_check(prb <= bwp_cfg.crbs.length(), "PRB={} falls outside BWP limits={}", prb, bwp_cfg.crbs);
43 return prb + bwp_cfg.crbs.start();
44}
