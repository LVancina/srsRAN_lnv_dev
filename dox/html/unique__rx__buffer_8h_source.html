84 unique_rx_buffer& operator=(unique_rx_buffer&& other) noexcept
85 {
86 // Unlock current soft buffer if it is actually not unlocked.
87 if (ptr != nullptr) {
88 ptr->unlock();
89 }
90
91 // Move the other soft buffer ownership to the current soft buffer.
92 ptr = other.ptr;
93 other.ptr = nullptr;
