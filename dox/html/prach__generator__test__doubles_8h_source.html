32class prach_generator_spy : public prach_generator
33{
34public:
35 struct generate_entry {
36 configuration config;
37 std::vector<cf_t> result;
38 };
39
40 span<const cf_t> generate(const configuration& config) override
41 {
42 // Extract sequence length.
43 unsigned sequence_length = is_long_preamble(config.format) ? prach_constants::LONG_SEQUENCE_LENGTH
44 : prach_constants::SHORT_SEQUENCE_LENGTH;
45
46 // Create entry.
47 entries.emplace_back();
48 generate_entry& entry = entries.back();
49 entry.config = config;
50 entry.result.resize(sequence_length);
51
52 // Generates the random result.
53 std::generate(entry.result.begin(), entry.result.end(), [&, n = 0]() mutable {
54 return std::polar(1.0F, static_cast<float>((n++) * config.preamble_index) / static_cast<float>(sequence_length));
55 });
56
57 return entry.result;
58 }
59
60 const std::vector<generate_entry>& get_entries() const { return entries; }
61
62 void clear_entries() { entries.clear(); }
63
64private:
65 std::vector<generate_entry> entries;
66};
