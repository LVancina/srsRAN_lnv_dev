51 precoding_weight_matrix(unsigned nof_layers, unsigned nof_ports) : data({nof_layers, nof_ports})

91 precoding_weight_matrix(span<const cf_t> weights, unsigned nof_layers, unsigned nof_ports) :
92 data({nof_layers, nof_ports})
93 {
94 srsran_assert(
95 weights.size() == nof_ports * nof_layers,
96 "The number of weights, i.e., {}, does not match the specified matrix dimensions, i.e., {} layers, {} ports.",
97 weights.size(),

99 nof_ports);

104 precoding_constants::MAX_NOF_LAYERS);

107 nof_ports,
108 precoding_constants::MAX_NOF_PORTS);

110 // Copy the weights into the tensor.
111 srsvec::copy(data.get_view<static_cast<unsigned>(dims::all)>({}), weights);

124 precoding_weight_matrix& operator=(const precoding_weight_matrix& other)
125 {
126 if (this == &other) {
127 return *this;
128 }
129
130 // Resize the tensor.
131 resize(other.get_nof_layers(), other.get_nof_ports());
132 // Copy the weights into the tensor.
133 srsvec::copy(data.get_view<static_cast<unsigned>(dims::all)>({}),
134 other.data.get_view<static_cast<unsigned>(dims::all)>({}));
135 return *this;
136 }

141 bool operator==(const precoding_weight_matrix& other) const
142 {
143 unsigned nof_layers = get_nof_layers();
144 unsigned nof_ports = get_nof_ports();
145
146 if (nof_layers != other.get_nof_layers()) {
147 return false;
148 }
149 if (nof_ports != other.get_nof_ports()) {
150 return false;
151 }
152
153 for (unsigned i_layer = 0; i_layer != nof_layers; ++i_layer) {
154 for (unsigned i_port = 0; i_port != nof_ports; ++i_port) {
155 if (get_coefficient(i_layer, i_port) != other.get_coefficient(i_layer, i_port)) {
156 return false;
157 }
158 }
159 }
160
161 return true;
162 }
163
165 bool operator!=(const precoding_weight_matrix& other) const { return !(*this == other); }

208 cf_t get_coefficient(unsigned i_layer, unsigned i_port) const
209 {
210 srsran_assert(i_layer < get_nof_layers(),
211 "The layer identifier (i.e., {}) exceeds the maximum (i.e., {}).",
212 i_layer,
213 get_nof_layers() - 1);
214 srsran_assert(i_port < get_nof_ports(),
215 "The ports identifier (i.e., {}) exceeds the maximum (i.e., {}).",
216 i_port,
217 get_nof_ports() - 1);
218 return data[{i_layer, i_port}];
219 }

225 void set_coefficient(cf_t coefficient, unsigned i_layer, unsigned i_port)
226 {
227 srsran_assert(i_layer < get_nof_layers(),
228 "The layer identifier (i.e., {}) exceeds the maximum (i.e., {}).",
229 i_layer,
230 get_nof_layers() - 1);
231 srsran_assert(i_port < get_nof_ports(),
232 "The ports identifier (i.e., {}) exceeds the maximum (i.e., {}).",
233 i_port,
234 get_nof_ports() - 1);
235 data[{i_layer, i_port}] = coefficient;
236 }

239 precoding_weight_matrix& operator*=(float scale)

241 srsvec::sc_prod(data.get_data(), scale, data.get_data());

253 srsran_assert(nof_layers <= precoding_constants::MAX_NOF_LAYERS,

256 precoding_constants::MAX_NOF_LAYERS);

261
262 data.resize({nof_layers, nof_ports});
263 }
