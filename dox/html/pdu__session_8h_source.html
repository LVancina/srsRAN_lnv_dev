38struct pdu_session {
39 pdu_session(const e1ap_pdu_session_res_to_setup_item& session, gtpu_demux_ctrl& gtpu_rx_demux_) :
40 pdu_session_id(session.pdu_session_id),
41 session_type(session.pdu_session_type),
42 snssai(session.snssai),
43 security_ind(session.security_ind),
44 ul_tunnel_info(session.ng_ul_up_tnl_info),
45 gtpu_rx_demux(gtpu_rx_demux_)
46 {
47 if (session.pdu_session_res_dl_ambr.has_value()) {
48 pdu_session_res_ambr = session.pdu_session_res_dl_ambr.value();
49 }
50 };
51 ~pdu_session()
52 {
53 // Remove GTP-U tunnel from GTP-U demux.
54 gtpu_rx_demux.remove_tunnel(local_teid);
55 }
56
57 std::unique_ptr<sdap_entity> sdap;
58 std::unique_ptr<gtpu_tunnel_ngu> gtpu;
59
60 // Adapters between SDAP and GTPU
61 gtpu_sdap_adapter gtpu_to_sdap_adapter;
62 sdap_gtpu_adapter sdap_to_gtpu_adapter;
63
64 pdcp_f1u_adapter pdcp_to_f1u_adapter;
65 pdu_session_id_t pdu_session_id; //< PDU session ID (0-255)
66 std::string session_type;
67 s_nssai_t snssai;
68 security_indication_t security_ind;
69 uint64_t pdu_session_res_ambr = 0;
70
71 // Tunneling info used by all DRBs/QoS flows in this PDU session
72 up_transport_layer_info ul_tunnel_info; // the peer GTP-U address and TEID
73 gtpu_teid_t local_teid; // the local teid used by the gNB for this PDU session
74 gtpu_demux_ctrl& gtpu_rx_demux; // The demux entity to register/remove the tunnel.
75
76 drb_context* default_drb = nullptr; // non-owning pointer to default DRB, if any
77
78 std::map<drb_id_t, std::unique_ptr<drb_context>> drbs; // key is drb_id
79};
