35inline asn1::e1ap::gnb_cu_up_e1_setup_request_s cu_up_e1_setup_request_to_asn1(const cu_up_e1_setup_request& request)
36{
37 asn1::e1ap::gnb_cu_up_e1_setup_request_s asn1_request;
38
39 // gnb cu up id
40 asn1_request->gnb_cu_up_id = request.gnb_cu_up_id;
41
42 // gnb cu up name
43 if (request.gnb_cu_up_name.has_value()) {
44 asn1_request->gnb_cu_up_name_present = true;
45 asn1_request->gnb_cu_up_name.from_string(request.gnb_cu_up_name.value());
46 }
47
48 // cn support
49 asn1_request->cn_support.value = static_cast<asn1::e1ap::cn_support_opts::options>(request.cn_support);
50
51 // supported plmns
52 for (const auto& plmn_item : request.supported_plmns) {
53 asn1::e1ap::supported_plmns_item_s asn1_plmn_item;
54
55 // plmn id
56 asn1_plmn_item.plmn_id.from_number(plmn_string_to_bcd(plmn_item.plmn_id));
57
58 // slice support list
59 for (const auto& slice_item : plmn_item.slice_support_list) {
60 asn1::e1ap::slice_support_item_s asn1_slice_item;
61
62 // s nssai
63 asn1_slice_item.snssai = snssai_to_e1ap_asn1(slice_item.s_nssai);
64
65 asn1_plmn_item.slice_support_list.push_back(asn1_slice_item);
66 }
67
68 // nr cgi support list
69 for (const auto& cgi_item : plmn_item.nr_cgi_support_list) {
70 asn1::e1ap::nr_cgi_support_item_s asn1_cgi_item;
71
72 // nr cgi
73 asn1_cgi_item.nr_cgi = nr_cgi_to_e1ap_asn1(cgi_item.nr_cgi);
74
75 asn1_plmn_item.nr_cgi_support_list.push_back(asn1_cgi_item);
76 }
77
78 // ng ran qos support list
79 if (!plmn_item.ng_ran_qos_support_list.empty()) {
80 asn1_plmn_item.qos_params_support_list_present = true;
81
82 for (const auto& ran_item : plmn_item.ng_ran_qos_support_list) {
83 asn1::e1ap::ng_ran_qos_support_item_s asn1_ran_item;
84
85 // five qi
86 asn1_ran_item.non_dyn_5qi_descriptor.five_qi = five_qi_to_uint(ran_item.non_dyn_5qi_descriptor.five_qi);
87
88 // qos prio level
89 if (ran_item.non_dyn_5qi_descriptor.qos_prio_level.has_value()) {
90 asn1_ran_item.non_dyn_5qi_descriptor.qos_prio_level_present = true;
91 asn1_ran_item.non_dyn_5qi_descriptor.qos_prio_level =
92 qos_prio_level_to_uint(ran_item.non_dyn_5qi_descriptor.qos_prio_level.value());
93 }
94
95 // averaging win
96 if (ran_item.non_dyn_5qi_descriptor.averaging_win.has_value()) {
97 asn1_ran_item.non_dyn_5qi_descriptor.averaging_win_present = true;
98 asn1_ran_item.non_dyn_5qi_descriptor.averaging_win = ran_item.non_dyn_5qi_descriptor.averaging_win.value();
99 }
100
101 // max data burst
102 if (ran_item.non_dyn_5qi_descriptor.max_data_burst_volume.has_value()) {
103 asn1_ran_item.non_dyn_5qi_descriptor.max_data_burst_volume_present = true;
104 asn1_ran_item.non_dyn_5qi_descriptor.max_data_burst_volume =
105 ran_item.non_dyn_5qi_descriptor.max_data_burst_volume.value();
106 }
107
108 asn1_plmn_item.qos_params_support_list.ng_ran_qos_support_list.push_back(asn1_ran_item);
109 }
110 }
111
112 asn1_request->supported_plmns.push_back(asn1_plmn_item);
113 }
114
115 // gnb cu up capacity
116 if (request.gnb_cu_up_capacity.has_value()) {
117 asn1_request->gnb_cu_up_capacity_present = true;
118 asn1_request->gnb_cu_up_capacity = request.gnb_cu_up_capacity.value();
119 }
120
121 return asn1_request;
122}

124inline bool
125check_e1ap_bearer_context_setup_request_valid(const asn1::e1ap::bearer_context_setup_request_s& asn1_request,
126 srslog::basic_logger& logger)
127{
128 // We only support NG-RAN Bearer
129 if (asn1_request->sys_bearer_context_setup_request.type() !=
130 asn1::e1ap::sys_bearer_context_setup_request_c::types::ng_ran_bearer_context_setup_request) {
131 logger.error("Not handling E-UTRAN Bearers");
132 return false;
133 }
134
135 // Check activity level
136 if (asn1_request->activity_notif_level.value != asn1::e1ap::activity_notif_level_e::ue) {
137 logger.warning("Unsupported activity notification level: {}", asn1_request->activity_notif_level.to_string());
138 return false;
139 }
140 if (!asn1_request->ue_inactivity_timer_present) {
141 logger.warning("Activity notification level is UE, but no UE inactivity timer present.");
142 return false;
143 }
144 return true;
145}

147inline void fill_e1ap_drb_to_setup_item(e1ap_drb_to_setup_item_ng_ran& drb_to_setup_item,
148 const asn1::e1ap::drb_to_setup_item_ng_ran_s& asn1_drb_to_setup_item)
149{

151
152 // sdap config
153 drb_to_setup_item.sdap_cfg = e1ap_asn1_to_sdap_config(asn1_drb_to_setup_item.sdap_cfg);
154
155 // pdcp config
156 drb_to_setup_item.pdcp_cfg = e1ap_asn1_to_pdcp_config(asn1_drb_to_setup_item.pdcp_cfg);
157
158 // cell group info
159 e1ap_asn1_to_cell_group_info(drb_to_setup_item.cell_group_info, asn1_drb_to_setup_item.cell_group_info);
160
161 // qos flow info to be setup
162 e1ap_asn1_to_flow_map_info(drb_to_setup_item.qos_flow_info_to_be_setup,
163 asn1_drb_to_setup_item.qos_flow_info_to_be_setup);
164
165 // drb data forwarding info request
166 if (asn1_drb_to_setup_item.drb_data_forwarding_info_request_present) {
167 drb_to_setup_item.drb_data_forwarding_info_request =
168 e1ap_asn1_to_data_forwarding_info_request(asn1_drb_to_setup_item.drb_data_forwarding_info_request);
169 }
170
171 // drb inactivity timer
172 if (asn1_drb_to_setup_item.drb_inactivity_timer_present) {
173 drb_to_setup_item.drb_inactivity_timer = asn1_drb_to_setup_item.drb_inactivity_timer;
174 }
175
176 // pdcp sn status info
177 if (asn1_drb_to_setup_item.pdcp_sn_status_info_present) {
178 drb_to_setup_item.pdcp_sn_status_info =
179 e1ap_asn1_to_pdcp_sn_status_info(asn1_drb_to_setup_item.pdcp_sn_status_info);
180 }

183inline void fill_e1ap_bearer_context_setup_request(e1ap_bearer_context_setup_request& request,
184 const asn1::e1ap::bearer_context_setup_request_s& asn1_request)
185{
186 // security info
187 request.security_info.security_algorithm.ciphering_algo =
188 e1ap_asn1_to_ciphering_algorithm(asn1_request->security_info.security_algorithm.ciphering_algorithm);
189
190 if (asn1_request->security_info.security_algorithm.integrity_protection_algorithm_present) {
191 request.security_info.security_algorithm.integrity_protection_algorithm =
192 e1ap_asn1_to_integrity_algorithm(asn1_request->security_info.security_algorithm.integrity_protection_algorithm);
193 }
194
195 request.security_info.up_security_key.encryption_key =
196 asn1_request->security_info.up_securitykey.encryption_key.copy();
197
198 request.security_info.up_security_key.integrity_protection_key =
199 asn1_request->security_info.up_securitykey.integrity_protection_key.copy();
200
201 // ue dl aggregate maximum bit rate
202 request.ue_dl_aggregate_maximum_bit_rate = asn1_request->ue_dl_aggr_max_bit_rate;
203
204 // serving plmn
205 request.serving_plmn = plmn_bcd_to_string(asn1_request->serving_plmn.to_number());
206
207 // activity notification level
208 request.activity_notif_level = asn1_to_activity_notification_level(asn1_request->activity_notif_level.value);
209 if (request.activity_notif_level == activity_notification_level_t::ue) {
210 request.ue_inactivity_timer = asn1_request->ue_inactivity_timer;
211 }
212
213 // pdu session resource to setup list
214 for (const auto& asn1_pdu_session_res_list_item :
215 asn1_request->sys_bearer_context_setup_request.ng_ran_bearer_context_setup_request()) {
216 for (const auto& asn1_pdu_session_res_item : asn1_pdu_session_res_list_item->pdu_session_res_to_setup_list()) {
217 e1ap_pdu_session_res_to_setup_item pdu_session_res_item;
218
219 // pdu session id
220 pdu_session_res_item.pdu_session_id = uint_to_pdu_session_id(asn1_pdu_session_res_item.pdu_session_id);
221
222 // pdu session type
223 pdu_session_res_item.pdu_session_type = asn1_pdu_session_res_item.pdu_session_type.to_string();
224
225 // s-nssai
226 pdu_session_res_item.snssai = e1ap_asn1_to_snssai(asn1_pdu_session_res_item.snssai);
227
228 // ng ul up transport layer information
229 pdu_session_res_item.ng_ul_up_tnl_info =
230 asn1_to_up_transport_layer_info(asn1_pdu_session_res_item.ng_ul_up_tnl_info);
231
232 // security indication
233 asn1_to_security_indication(pdu_session_res_item.security_ind, asn1_pdu_session_res_item.security_ind);
234
235 // drb to setup list ng ran
236 for (const auto& asn1_drb_to_setup_item : asn1_pdu_session_res_item.drb_to_setup_list_ng_ran) {
237 e1ap_drb_to_setup_item_ng_ran drb_to_setup_item;
238 fill_e1ap_drb_to_setup_item(drb_to_setup_item, asn1_drb_to_setup_item);
239 pdu_session_res_item.drb_to_setup_list_ng_ran.emplace(uint_to_drb_id(asn1_drb_to_setup_item.drb_id),
240 drb_to_setup_item);
241 }
242
243 // pdu session resource dl aggregate maximum bit rate
244 if (asn1_pdu_session_res_item.pdu_session_res_dl_ambr_present) {
245 pdu_session_res_item.pdu_session_res_dl_ambr = asn1_pdu_session_res_item.pdu_session_res_dl_ambr;
246 }
247
248 // pdu session data forwarding info request
249 if (asn1_pdu_session_res_item.pdu_session_data_forwarding_info_request_present) {
250 e1ap_data_forwarding_info_request data_forwarding_info_req = {};
251 data_forwarding_info_req.data_forwarding_request =
252 asn1_pdu_session_res_item.pdu_session_data_forwarding_info_request.data_forwarding_request.to_string();
253
254 for (const auto& asn1_qos_flows_forwarded_item :
255 asn1_pdu_session_res_item.pdu_session_data_forwarding_info_request.qos_flows_forwarded_on_fwd_tunnels) {
256 e1ap_qos_flow_map_item qos_flows_forwarded_item =
257 asn1_e1ap_to_qos_flow_map_item(asn1_qos_flows_forwarded_item);
258
259 data_forwarding_info_req.qos_flows_forwarded_on_fwd_tunnels.emplace(
260 uint_to_qos_flow_id(asn1_qos_flows_forwarded_item.qos_flow_id), qos_flows_forwarded_item);
261 }
262 }
263
264 // pdu session inactivity timer
265 if (asn1_pdu_session_res_item.pdu_session_inactivity_timer_present) {
266 pdu_session_res_item.pdu_session_inactivity_timer = asn1_pdu_session_res_item.pdu_session_inactivity_timer;
267 }
268
269 // existing allocated ng dl up tnl info
270 if (asn1_pdu_session_res_item.existing_allocated_ng_dl_up_tnl_info_present) {
271 pdu_session_res_item.existing_allocated_ng_dl_up_tnl_info =
272 asn1_to_up_transport_layer_info(asn1_pdu_session_res_item.existing_allocated_ng_dl_up_tnl_info);
273 }
274
275 // network instance
276 if (asn1_pdu_session_res_item.network_instance_present) {
277 pdu_session_res_item.network_instance = asn1_pdu_session_res_item.network_instance;
278 }
279 request.pdu_session_res_to_setup_list.emplace(uint_to_pdu_session_id(asn1_pdu_session_res_item.pdu_session_id),
280 pdu_session_res_item);
281 }
282 }
283
284 // ue dl maximum integrity protection data rate
285 if (asn1_request->ue_dl_max_integrity_protected_data_rate_present) {
286 request.ue_dl_maximum_integrity_protected_data_rate = asn1_request->ue_dl_max_integrity_protected_data_rate;
287 }
288
289 // ue inactivity timer
290 if (asn1_request->ue_inactivity_timer_present) {
291 request.ue_inactivity_timer = asn1_request->ue_inactivity_timer;
292 }
293
294 // bearer context status change
295 if (asn1_request->bearer_context_status_change_present) {
296 request.bearer_context_status_change = asn1_request->bearer_context_status_change.to_string();
297 }
298
299 // ran ue id
300 if (asn1_request->ran_ue_id_present) {
301 request.ran_ue_id = uint_to_ran_ue_id(asn1_request->ran_ue_id.to_number());
302 }
303
304 // gnb du id
305 if (asn1_request->gnb_du_id_present) {
306 request.gnb_du_id = asn1_request->gnb_du_id;
307 }
308}

310inline void fill_asn1_bearer_context_setup_response(asn1::e1ap::sys_bearer_context_setup_resp_c& asn1_response,
311 const e1ap_bearer_context_setup_response& response)
312{
313 auto& asn1_bearer_context_setup_response = asn1_response.set_ng_ran_bearer_context_setup_resp();
314
315 // Add PDU Session Resource Setup List
316 for (const auto& res_setup_item : response.pdu_session_resource_setup_list) {
317 asn1::e1ap::pdu_session_res_setup_item_s asn1_res_setup_item;
318
319 // Add PDU Session ID
320 asn1_res_setup_item.pdu_session_id = pdu_session_id_to_uint(res_setup_item.pdu_session_id);
321
322 // Add NG DL UP TNL Info
323 up_transport_layer_info_to_asn1(asn1_res_setup_item.ng_dl_up_tnl_info, res_setup_item.ng_dl_up_tnl_info);
324
325 // Add DRB Setup List NG RAN
326 e1ap_drb_item_list_to_asn1(asn1_res_setup_item.drb_setup_list_ng_ran, res_setup_item.drb_setup_list_ng_ran);
327
328 // Add DRB Failed List NG RAN
329 e1ap_drb_failed_item_list_to_asn1(asn1_res_setup_item.drb_failed_list_ng_ran,
330 res_setup_item.drb_failed_list_ng_ran);
331
332 // Add Security Result
333 if (res_setup_item.security_result.has_value()) {
334 asn1_res_setup_item.security_result_present = true;
335 security_result_to_asn1(asn1_res_setup_item.security_result, res_setup_item.security_result.value());
336 }
337
338 // Add PDU Session Data Forwarding Info Response
339 if (res_setup_item.pdu_session_data_forwarding_info_resp.has_value()) {
340 asn1_res_setup_item.pdu_session_data_forwarding_info_resp_present = true;
341 if (res_setup_item.pdu_session_data_forwarding_info_resp.value().ul_data_forwarding.has_value()) {
342 asn1_res_setup_item.pdu_session_data_forwarding_info_resp.ul_data_forwarding_present = true;
343 up_transport_layer_info_to_asn1(
344 asn1_res_setup_item.pdu_session_data_forwarding_info_resp.ul_data_forwarding,
345 res_setup_item.pdu_session_data_forwarding_info_resp.value().ul_data_forwarding.value());
346 }
347 if (res_setup_item.pdu_session_data_forwarding_info_resp.value().dl_data_forwarding.has_value()) {
348 asn1_res_setup_item.pdu_session_data_forwarding_info_resp.dl_data_forwarding_present = true;
349 up_transport_layer_info_to_asn1(
350 asn1_res_setup_item.pdu_session_data_forwarding_info_resp.dl_data_forwarding,
351 res_setup_item.pdu_session_data_forwarding_info_resp.value().dl_data_forwarding.value());
352 }
353 }
354
355 // Add NG DL UP Unchanged
356 if (res_setup_item.ng_dl_up_unchanged.has_value()) {
357 asn1_res_setup_item.ng_dl_up_unchanged_present = true;
358 if (res_setup_item.ng_dl_up_unchanged.value()) {
359 asn1_res_setup_item.ng_dl_up_unchanged.value =
360 asn1::e1ap::pdu_session_res_setup_item_s::ng_dl_up_unchanged_opts::options::true_value;
361 } else {
362 asn1_res_setup_item.ng_dl_up_unchanged.value =
363 asn1::e1ap::pdu_session_res_setup_item_s::ng_dl_up_unchanged_opts::options::nulltype;
364 }
365 }
366
367 asn1_bearer_context_setup_response.pdu_session_res_setup_list.push_back(asn1_res_setup_item);
368 }
369
370 // Add pdu session res failed list
371 if (!response.pdu_session_resource_failed_list.empty()) {
372 asn1_bearer_context_setup_response.pdu_session_res_failed_list_present = true;
373 for (const auto& failed_item : response.pdu_session_resource_failed_list) {
374 asn1::e1ap::pdu_session_res_failed_item_s asn1_failed_item;
375 asn1_failed_item.pdu_session_id = pdu_session_id_to_uint(failed_item.pdu_session_id);
376 asn1_failed_item.cause = cause_to_asn1_cause(failed_item.cause);
377
378 asn1_bearer_context_setup_response.pdu_session_res_failed_list.push_back(asn1_failed_item);
379 }
380 }

383inline void fill_e1ap_bearer_context_modification_request(e1ap_bearer_context_modification_request& request,
384 const asn1::e1ap::bearer_context_mod_request_s& asn1_request)
385{
386 // security info
387 if (asn1_request->security_info_present) {
388 e1ap_security_info security_info;
389 security_info.security_algorithm.ciphering_algo =
390 e1ap_asn1_to_ciphering_algorithm(asn1_request->security_info.security_algorithm.ciphering_algorithm);
391 security_info.security_algorithm.integrity_protection_algorithm =
392 e1ap_asn1_to_integrity_algorithm(asn1_request->security_info.security_algorithm.integrity_protection_algorithm);
393 security_info.up_security_key.encryption_key = asn1_request->security_info.up_securitykey.encryption_key.copy();
394 security_info.up_security_key.integrity_protection_key =
395 asn1_request->security_info.up_securitykey.integrity_protection_key.copy();
396
397 request.security_info = security_info;
398 }
399
400 // ue dl aggr max bit rate
401 if (asn1_request->ue_dl_aggr_max_bit_rate_present) {
402 request.ue_dl_aggr_max_bit_rate = asn1_request->ue_dl_aggr_max_bit_rate;
403 }
404
405 // ue dl max integrity protected data rate
406 if (asn1_request->ue_dl_max_integrity_protected_data_rate_present) {
407 request.ue_dl_max_integrity_protected_data_rate = asn1_request->ue_dl_max_integrity_protected_data_rate;
408 }
409
410 // bearer context status change
411 if (asn1_request->bearer_context_status_change_present) {
412 request.bearer_context_status_change = asn1_request->bearer_context_status_change.to_string();
413 }
414
415 // new ul tnl info required
416 if (asn1_request->new_ul_tnl_info_required_present) {
417 request.new_ul_tnl_info_required = asn1_request->new_ul_tnl_info_required.to_string();
418 }
419
420 // ue inactivity timer
421 if (asn1_request->ue_inactivity_timer_present) {
422 request.ue_inactivity_timer = asn1_request->ue_inactivity_timer;
423 }
424
425 // data discard required
426 if (asn1_request->data_discard_required_present) {
427 request.data_discard_required = asn1_request->data_discard_required.to_string();
428 }
429
430 // ng ran bearer context mod
431 if (asn1_request->sys_bearer_context_mod_request_present) {
432 e1ap_ng_ran_bearer_context_mod_request ng_ran_bearer_context_mod_request;
433
434 const auto& asn1_ng_ran_bearer_context_mod_request =
435 asn1_request->sys_bearer_context_mod_request.ng_ran_bearer_context_mod_request();
436
437 // pdu session res to setup mod list
438 if (asn1_ng_ran_bearer_context_mod_request.pdu_session_res_to_setup_mod_list_present) {
439 for (const auto& asn1_res_to_setup_mod_item :
440 asn1_ng_ran_bearer_context_mod_request.pdu_session_res_to_setup_mod_list) {
441 e1ap_pdu_session_res_to_setup_item pdu_session_res_to_setup_mod_item;
442
443 // pdu session id
444 pdu_session_res_to_setup_mod_item.pdu_session_id =
445 uint_to_pdu_session_id(asn1_res_to_setup_mod_item.pdu_session_id);
446
447 // pdu session type
448 pdu_session_res_to_setup_mod_item.pdu_session_type = asn1_res_to_setup_mod_item.pdu_session_type.to_string();
449
450 // s nssai
451 pdu_session_res_to_setup_mod_item.snssai = e1ap_asn1_to_snssai(asn1_res_to_setup_mod_item.snssai);
452
453 // security ind
454 asn1_to_security_indication(pdu_session_res_to_setup_mod_item.security_ind,
