35/// \brief Maximum number of RE per resource block in a PDSCH transmission.
36///
37/// As per TS38.214 Section 5.1.3.2.
38static constexpr unsigned MAX_NRE_PER_RB = 156;
39
40/// \brief Maximum modulation order supported on PDSCH transmissions.
41///
42/// As per TS38.214 Section 5.1.3.1 with \c mcs-Table set to \c qam256.
43static constexpr unsigned MAX_MODULATION_ORDER = 8;
44
45/// \brief Maximum number of PDSCH transmission layers per codeword.
46///
47/// As per TS38.211 Table 7.3.1.3-1.
48static constexpr unsigned CODEWORD_MAX_NOF_LAYERS = 4;
49
50/// \brief Maximuym number of codewords that can be contained in a PDSCH transmission.
51///
52/// As per TS38.211 Table 7.3.1.3-1.
53static constexpr unsigned MAX_NOF_CODEWORDS = 2;
54
55/// \brief Maximum number of PDSCH transmission layers.
56///
57/// As per TS38.211 Table 7.3.1.3-1.
58static constexpr unsigned MAX_NOF_LAYERS = CODEWORD_MAX_NOF_LAYERS * MAX_NOF_CODEWORDS;
59
60/// \brief Maximum number of resource elements that can be mapped into a single codeword.
61///
62/// Calculated as the product of maximum number or RE per PRB and the maximum number of PRB in the a resource grid.
63static constexpr unsigned CODEWORD_MAX_NOF_RE = MAX_NRE_PER_RB * MAX_NOF_PRBS;
64
65/// \brief Maximum number of symbols that can be mapped into a single codeword.
66///
67/// Calculated as the product of the maximum number of resource elements and the maximum layers
68static constexpr unsigned CODEWORD_MAX_SYMBOLS = CODEWORD_MAX_NOF_RE * CODEWORD_MAX_NOF_LAYERS;
69
70/// \brief Maximum number of bits that can be modulated into a single codeword.
71///
72/// Calculated as the product of the maximum number of resource elements and the maximum modulation order.
73static constexpr units::bits CODEWORD_MAX_SIZE{CODEWORD_MAX_SYMBOLS * MAX_MODULATION_ORDER};
74
75} // namespace pdsch_constants
