29/// \brief Unsynchronized (not thread-safe) memory pool of memory blocks of equal size.
30class unsync_fixed_size_memory_block_pool
31{
32public:
33 /// Ctor of the memory pool.
34 explicit unsync_fixed_size_memory_block_pool(size_t nof_blocks_, size_t memory_block_size_) :
35 mblock_size(align_next(memory_block_size_, alignof(std::max_align_t))), nof_blocks(nof_blocks_)
36 {
37 srsran_assert(mblock_size > free_memory_block_list::min_memory_block_align(),
38 "Segment size is too small ({} <= {})",
39 mblock_size,
40 free_memory_block_list::min_memory_block_align());
41
42 // Allocate the required memory for the given number of segments and segment size.
43 size_t total_mem = mblock_size * nof_blocks;
44 allocated_memory.resize(total_mem);
45
46 // Push all segments to the central cache.
47 for (unsigned i = 0; i != nof_blocks; ++i) {
48 block_free_list.push(static_cast<void*>(allocated_memory.data() + (mblock_size * i)));
49 }
50 }
51 unsync_fixed_size_memory_block_pool(const unsync_fixed_size_memory_block_pool&) = delete;
52 unsync_fixed_size_memory_block_pool(unsync_fixed_size_memory_block_pool&&) = delete;
53 unsync_fixed_size_memory_block_pool& operator=(const unsync_fixed_size_memory_block_pool&) = delete;
54 unsync_fixed_size_memory_block_pool& operator=(unsync_fixed_size_memory_block_pool&&) = delete;
55
56 /// Memory block size in bytes.
57 size_t memory_block_size() const { return mblock_size; }
58
59 /// Number of memory blocks contained in this memory pool.
60 size_t nof_memory_blocks() const { return nof_blocks; }
61
62 /// Number of blocks currently available to be allocated.
63 size_t nof_blocks_available() const { return block_free_list.size(); }
64
65 /// Whether the pool is empty.
66 size_t empty() const { return block_free_list.empty(); }
67
68 /// Allocate a node from the memory pool with the provided size.
69 void* allocate_node(size_t sz) noexcept
70 {
71 srsran_assert(sz <= mblock_size, "Allocated node size={} exceeds max object size={}", sz, mblock_size);
72
73 // pop from free memory block list. If the free list is empty, null is returned.
74 void* node = block_free_list.try_pop();
75 return node;
76 }
77
78 /// Deallocate node by returning it back to the memory pool.
79 void deallocate_node(void* p)
80 {
81 srsran_assert(p != nullptr, "Deallocated nodes must have valid address");
82
83 // push to free memory block list.
84 block_free_list.push(p);
85 }
86
87private:
88 const size_t mblock_size;
89 const size_t nof_blocks;
90
91 std::vector<uint8_t> allocated_memory;
92 free_memory_block_list block_free_list;
93};
