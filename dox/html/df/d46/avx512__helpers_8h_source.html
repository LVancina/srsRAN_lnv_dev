34/// Type of AVX512 intrinsic calculating min or max values between two vectors.
35using _mm512_comparator = __m512i (*)(__m512i a, __m512i b);

37/// \brief Finds the maximum (or minimum) 16bit sample in each of the four input resource blocks, returns its absolute
38/// value.
39///
40/// Finds absolute maximum (or minimum, depending on the comparator passed as the template argument) values for each of
41/// the four RBs passed in three AVX512 registers.
42///
43/// AVX512 register bits are evenly divided between four RBs so that IQ samples of each RB occupy one 128bit lane.
44/// The following diagram shows the input format. Here RBx stands for one unique RE (a pair of IQ samples, 32 bits long)
45/// pertaining to a respective RB:
46/// | | | | | | | | | |
47/// | ----- | ------- | ------- | ------- | ------- | ------- | ------- | ------- | ------- |
48/// | \c v0_epi16: | RB0 RB0 | RB0 RB0 | RB1 RB1 | RB1 RB1 | RB2 RB2 | RB2 RB2 | RB3 RB3 | RB3 RB3 |
49/// | \c v1_epi16: | RB0 RB0 | RB0 RB0 | RB1 RB1 | RB1 RB1 | RB2 RB2 | RB2 RB2 | RB3 RB3 | RB3 RB3 |
50/// | \c v2_epi16: | RB0 RB0 | RB0 RB0 | RB1 RB1 | RB1 RB1 | RB2 RB2 | RB2 RB2 | RB3 RB3 | RB3 RB3 |
51///
52/// The function first performs vertical min/max search across the three registers (comparing 16bit samples), then
53/// horizontal min/max across elements in each 128bit lane of the resulting vector. Finally, the function takes the
54/// absolute value of the four 16bit samples computed thus far.
55///
56/// \tparam COMPARE AVX512 intrinsic used for comparison, either \c _mm512_min_epi16 or \c _mm512_max_epi16.
57/// \param[in] v0_epi16 AVX512 register storing first four IQ pairs of each of the four RB.
58/// \param[in] v1_epi16 AVX512 register storing second four IQ pairs of each of the four RB.
59/// \param[in] v2_epi16 AVX512 register storing last four IQ pairs of each of the four RB.
60///
61/// \return An AVX512 register storing the result for each of the four RBs in every 128bit lane.
62template <_mm512_comparator COMPARE>
63inline __m512i find_abs_min_max_values(__m512i v0_epi16, __m512i v1_epi16, __m512i v2_epi16)
64{
65 // Find vertical min/max across every 8 16bit values pertaining to the same PRB.
66 __m512i cmp_8val_epi16 = COMPARE(COMPARE(v0_epi16, v1_epi16), v2_epi16);
67
68 // Swap 64bit words inside 128bit lanes and find 4 min/max values.
69 __m512i cmp_8val_swp_epi16 =
70 _mm512_mask_shuffle_epi32(_mm512_set1_epi64(0), 0xffff, cmp_8val_epi16, (_MM_PERM_ENUM)_MM_SHUFFLE(1, 0, 3, 2));
71 __m512i cmp_4val_epi16 = COMPARE(cmp_8val_epi16, cmp_8val_swp_epi16);
72
73 // Swap 32bit words and find 2 min/max values.
74 __m512i cmp_4val_swp_epi16 =
75 _mm512_mask_shuffle_epi32(_mm512_set1_epi64(0), 0xffff, cmp_4val_epi16, (_MM_PERM_ENUM)_MM_SHUFFLE(2, 3, 0, 1));
76 __m512i cmp_2val_epi16 = COMPARE(cmp_4val_epi16, cmp_4val_swp_epi16);
77
78 // Swap 16bit words and find final min/max values for each PRB.
79 const __m512i shuffle_mask = _mm512_setr_epi64(0x0504070601000302,
80 0x0d0c0f0e09080b0a,
81 0x0504070601000302,
82 0x0d0c0f0e09080b0a,
83 0x0504070601000302,
84 0x0d0c0f0e09080b0a,
85 0x0504070601000302,
86 0x0d0c0f0e09080b0a);
87
88 __m512i cmp_2val_swp_epi16 = _mm512_shuffle_epi8(cmp_2val_epi16, shuffle_mask);
89 return _mm512_abs_epi16(COMPARE(cmp_2val_epi16, cmp_2val_swp_epi16));
90}

92/// \brief Finds maximum absolute value across 16bit IQ samples in each of the four input resource blocks passed in
93/// three AVX512 registers.
94///
95/// The content of the three input AVX512 registers is represented in the table below. Here RBx stands for one unique RE
96/// (pair of IQ samples, 32 bits long) pertaining to a respective RB:
97/// | | | | | | | | | |
98/// | ----- | ------- | ------- | ------- | ------- | ------- | ------- | ------- | ------- |
99/// | \c r0_epi16: | RB0 RB0 | RB0 RB0 | RB0 RB0 | RB0 RB0 | RB0 RB0 | RB0 RB0 | RB1 RB1 | RB1 RB1 |
100/// | \c r1_epi16: | RB1 RB1 | RB1 RB1 | RB1 RB1 | RB1 RB1 | RB2 RB2 | RB2 RB2 | RB2 RB2 | RB2 RB2 |
101/// | \c r2_epi16: | RB2 RB2 | RB2 RB2 | RB3 RB3 | RB3 RB3 | RB3 RB3 | RB3 RB3 | RB3 RB3 | RB3 RB3 |
102///
103/// \param[in] r0_epi16 AVX512 register storing 16bit IQ pairs of the first and second RBs.
104/// \param[in] r1_epi16 AVX512 register storing 16bit IQ pairs of the second and third RBs.
105/// \param[in] r2_epi16 AVX512 register storing 16bit IQ pairs of the third and fourth RBs.
106/// \return An AVX512 register storing the result for each of the four RBs in every 128bit lane.
107inline __m512i calculate_max_abs(__m512i r0_epi16, __m512i r1_epi16, __m512i r2_epi16)
108{
109 const __m512i ONE_EPI16 = _mm512_set1_epi16(1);
110
111 // Permute input vectors so that the pairwise comparison can be used. The resulting vectors look as follows:
112 // [ RB0 RB0 | RB0 RB0 | RB1 RB1 | RB1 RB1 | RB2 RB2 | RB2 RB2 | RB3 RB3 | RB3 RB3 ]
113 // [ RB0 RB0 | RB0 RB0 | RB1 RB1 | RB1 RB1 | RB2 RB2 | RB2 RB2 | RB3 RB3 | RB3 RB3 ]
114 // [ RB0 RB0 | RB0 RB0 | RB1 RB1 | RB1 RB1 | RB2 RB2 | RB2 RB2 | RB3 RB3 | RB3 RB3 ]
115 __m512i v0_epi16 =
116 _mm512_mask_shuffle_i64x2(r0_epi16, 0b11111100, r1_epi16, r2_epi16, (_MM_PERM_ENUM)_MM_SHUFFLE(1, 0, 0, 0));
117 __m512i v1_epi16 =
118 _mm512_mask_shuffle_i64x2(r1_epi16, 0b11000011, r0_epi16, r2_epi16, (_MM_PERM_ENUM)_MM_SHUFFLE(2, 0, 0, 1));
119 __m512i v2_epi16 =
120 _mm512_mask_shuffle_i64x2(r2_epi16, 0b00111111, r0_epi16, r1_epi16, (_MM_PERM_ENUM)_MM_SHUFFLE(0, 3, 3, 2));
121
122 __m512i abs_min_epu16 = find_abs_min_max_values<_mm512_min_epi16>(v0_epi16, v1_epi16, v2_epi16);
123 __m512i abs_max_epu16 = find_abs_min_max_values<_mm512_max_epi16>(v0_epi16, v1_epi16, v2_epi16);
124
125 // Subtract 1 from absolute minimum value.
126 abs_min_epu16 = _mm512_subs_epu16(abs_min_epu16, ONE_EPI16);
127
128 return _mm512_max_epu16(abs_min_epu16, abs_max_epu16);
129}
130/// \brief Determines BFP exponent for each of the four input RBs passed in three AVX512 registers.
131///
132/// \param[in] r0_epi16 AVX512 register storing 16bit IQ pairs of the first and second RBs.
133/// \param[in] r1_epi16 AVX512 register storing 16bit IQ pairs of the second and third RBs.
134/// \param[in] r2_epi16 AVX512 register storing 16bit IQ pairs of the third and fourth RBs.
135/// \param[in] data_width Compressed samples bit width.
136///
137/// \return Four exponents in the LSB bits of every 128bit lane.
138inline __m512i determine_bfp_exponent(__m512i r0_epi16, __m512i r1_epi16, __m512i r2_epi16, unsigned data_width)
139{
140 // Max allowed exponent (account for the extra 16 MSBs and a sign bit counted by _mm512_lzcnt_epi32 intrinsic).
141 unsigned max_lzcnt = (16 - data_width) + 16 + 1;
142 __m512i max_exp_epu32 = _mm512_set1_epi32(max_lzcnt);
143
144 // Calculate maximum absolute values in the four PRBs.
145 __m512i max_abs_epu16 = calculate_max_abs(r0_epi16, r1_epi16, r2_epi16);
146
147 // Mask out 16 MSB bits in each 32bit word.
148 max_abs_epu16 = _mm512_and_epi32(max_abs_epu16, _mm512_set1_epi32(0x0000ffff));
149
150 // Compute BFP exponent as a number of leading zeros.
151 __m512i lz_count_epi32 = _mm512_lzcnt_epi32(max_abs_epu16);
152
153 // Subtract with saturation.
154 return _mm512_subs_epu16(max_exp_epu32, lz_count_epi32);
155}
