40/// Helper class to fill the transmission precoding and beamforming parameters specified in SCF-222 v4.0
41/// section 3.4.2.5.
42class tx_precoding_and_beamforming_pdu_builder
43{
44 tx_precoding_and_beamforming_pdu& pdu;
45
46public:
47 explicit tx_precoding_and_beamforming_pdu_builder(tx_precoding_and_beamforming_pdu& pdu_) : pdu(pdu_)
48 {
49 // Mark the tx precoding and beamforming pdu as used when this builder is called.
50 pdu.trp_scheme = 0U;
51 // Initialize number of digital beamforming interfaces.
52 pdu.dig_bf_interfaces = 0U;
53 }
54
55 /// Sets the basic parameters for the fields of the tranmission precoding and beamforming PDU.
56 /// \note These parameters are specified in SCF-222 v4.0 section 3.4.2.5, in table Tx precoding and beamforming PDU.
57 tx_precoding_and_beamforming_pdu_builder& set_basic_parameters(unsigned prg_size, unsigned dig_bf_interfaces)
58 {
59 pdu.prg_size = prg_size;
60 pdu.dig_bf_interfaces = dig_bf_interfaces;
61
62 return *this;
63 }
64
65 /// Adds a PRG to the transmission precoding and beamforming PDU.
66 /// \note These parameters are specified in SCF-222 v4.0 section 3.4.2.5, in table Tx precoding and beamforming PDU.
67 tx_precoding_and_beamforming_pdu_builder& add_prg(unsigned pm_index, span<const uint16_t> beam_index)
68 {
69 tx_precoding_and_beamforming_pdu::prgs_info& prg = pdu.prgs.emplace_back();
70
71 srsran_assert(pdu.dig_bf_interfaces == beam_index.size(),
72 "Error number of beam indexes={} does not match the expected={}",
73 beam_index.size(),
74 pdu.dig_bf_interfaces);
75
76 prg.pm_index = pm_index;
77 prg.beam_index.assign(beam_index.begin(), beam_index.end());
78
79 return *this;
80 }

93 /// Sets the basic parameters for the fields of the SSB/PBCH PDU.
94 /// \note These parameters are specified in SCF-222 v4.0 section 3.4.2.4, in table SSB/PBCH PDU.
95 dl_ssb_pdu_builder& set_basic_parameters(pci_t phys_cell_id,

97 uint8_t ssb_block_index,
98 uint8_t ssb_subcarrier_offset,
99 ssb_offset_to_pointA ssb_offset_pointA)

101 pdu.phys_cell_id = phys_cell_id;

104 pdu.ssb_subcarrier_offset = ssb_subcarrier_offset;

110 /// Sets the BCH payload configured by the MAC and returns a reference to the builder.
111 /// \note Use this function when the MAC generates the full PBCH payload.
112 /// \note These parameters are specified in SCF-222 v4.0 section 3.4.2.4, in table MAC generated MIB PDU.
113 /// \note This function assumes that given bch_payload value is codified as: a0,a1,a2,...,a29,a30,a31, with the most
114 /// significant bit being the leftmost (in this case a0 in position 31 of the uint32_t).
115 dl_ssb_pdu_builder& set_bch_payload_mac_full(uint32_t bch_payload)
116 {
117 // Configure the BCH payload as fully generated by the MAC.
118 pdu.bch_payload_flag = bch_payload_type::mac_full;
119 pdu.bch_payload.bch_payload = bch_payload;
120
121 return *this;
122 }
123
124 /// Sets the BCH payload and returns a reference to the builder. PHY configures the timing PBCH bits.
125 /// \note Use this function when the PHY generates the timing PBCH information.
126 /// \note These parameters are specified in SCF-222 v4.0 section 3.4.2.4, in table MAC generated MIB PDU.
127 /// \note This function assumes that given bch_payload value is codified as: 0,0,0,0,0,0,0,0,a0,a1,a2,...,a21,a22,a23,
128 /// with the most significant bit being the leftmost (in this case a0 in position 24 of the uint32_t).
129 dl_ssb_pdu_builder& set_bch_payload_phy_timing_info(uint32_t bch_payload)
130 {
131 static constexpr unsigned MAX_SIZE = (1U << 24U);
132
133 srsran_assert(bch_payload < MAX_SIZE, "BCH payload value out of bounds");
134
135 pdu.bch_payload_flag = bch_payload_type::phy_timing_info;
136 // Clear unused bits in the high part of the payload.
137 pdu.bch_payload.bch_payload = bch_payload & (MAX_SIZE - 1);
138
139 return *this;
140 }
141
142 /// Sets the BCH payload configured by the PHY and returns a reference to the builder.
143 /// \note Use this function when the PHY generates the full PBCH payload.
144 /// \note These parameters are specified in SCF-222 v4.0 section 3.4.2.4, in table PHY generated MIB PDU.
145 dl_ssb_pdu_builder& set_bch_payload_phy_full(dmrs_typeA_position dmrs_type_a_position,
146 uint8_t pdcch_config_sib1,
147 bool cell_barred,
148 bool intra_freq_reselection)
149 {
150 pdu.bch_payload_flag = bch_payload_type::phy_full;
151 pdu.bch_payload.phy_mib_pdu.dmrs_typeA_position =
152 (dmrs_type_a_position == dmrs_typeA_position::pos2) ? dmrs_typeA_pos::pos2 : dmrs_typeA_pos::pos3;
153 pdu.bch_payload.phy_mib_pdu.pdcch_config_sib1 = pdcch_config_sib1;
154 pdu.bch_payload.phy_mib_pdu.cell_barred = cell_barred;
155 pdu.bch_payload.phy_mib_pdu.intrafreq_reselection = intra_freq_reselection;
156
157 return *this;
158 }

161 /// \note These parameters are specified in SCF-222 v4.0 section 3.4.2.4, in table SSB/PBCH PDU maintenance FAPIv3.
162 /// \note ssbPduIndex field is automatically filled when adding a new SSB PDU to the DL TTI request message.
163 dl_ssb_pdu_builder&
164 set_maintenance_v3_basic_parameters(ssb_pattern_case case_type, subcarrier_spacing scs, uint8_t L_max)

173 /// Sets the SSB power information and returns a reference to the builder.
174 /// \note These parameters are specified in SCF-222 v4.0 section 3.4.2.4, in table SSB/PBCH PDU maintenance FAPIv3.
175 dl_ssb_pdu_builder& set_maintenance_v3_tx_power_info(optional<float> power_scaling_ss_pbch_dB,
176 optional<float> pss_to_sss_ratio_dB)
177 {
178 // Power scaling in SS-PBCH in hundredths of dBs.
179 int ss_block_power = (power_scaling_ss_pbch_dB) ? static_cast<int>(power_scaling_ss_pbch_dB.value() * 100) : -32768;
180 srsran_assert(ss_block_power <= std::numeric_limits<decltype(v3.ss_pbch_block_power_scaling)>::max(),
181 "SS PBCH block power scaling ({}) exceeds the maximum ({}).",
182 ss_block_power,
183 std::numeric_limits<decltype(v3.ss_pbch_block_power_scaling)>::max());
184 srsran_assert(ss_block_power >= std::numeric_limits<decltype(v3.ss_pbch_block_power_scaling)>::min(),
185 "SS PBCH block power scaling ({}) does not reach the minimum ({}).",
186 ss_block_power,
187 std::numeric_limits<decltype(v3.ss_pbch_block_power_scaling)>::min());
188 v3.ss_pbch_block_power_scaling = static_cast<decltype(v3.ss_pbch_block_power_scaling)>(ss_block_power);
189
190 // SSS to PSS ratio in thousandths of dBs.
191 int beta_pss = (pss_to_sss_ratio_dB) ? static_cast<int>(pss_to_sss_ratio_dB.value() * 1000) : -32768;
192 srsran_assert(beta_pss <= std::numeric_limits<decltype(v3.beta_pss_profile_sss)>::max(),
193 "PSS to SSS ratio ({}) exceeds the maximum ({}).",
194 beta_pss,
195 std::numeric_limits<decltype(v3.beta_pss_profile_sss)>::max());
196 srsran_assert(beta_pss >= std::numeric_limits<decltype(v3.beta_pss_profile_sss)>::min(),
197 "PSS to SSS ratio ({}) does not reach the minimum ({}).",
198 beta_pss,
199 std::numeric_limits<decltype(v3.beta_pss_profile_sss)>::min());
200
201 v3.beta_pss_profile_sss = static_cast<decltype(v3.beta_pss_profile_sss)>(beta_pss);
202
203 return *this;
204 }
205
206 /// Returns a transmission precoding and beamforming PDU builder of this SSB PDU.

211 return builder;
