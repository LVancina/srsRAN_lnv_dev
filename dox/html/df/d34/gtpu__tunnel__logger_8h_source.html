32class gtpu_tunnel_log_prefix
33{
34public:
35 gtpu_tunnel_log_prefix(optional<srs_cu_up::ue_index_t> ue_index, gtpu_teid_t teid, const char* dir)
36 {
37 fmt::memory_buffer buffer;
38 if (ue_index.has_value()) {
39 fmt::format_to(buffer, "ue={} {} teid={}: ", ue_index, dir, teid);
40 } else {
41 fmt::format_to(buffer, "{} teid={}: ", dir, teid);
42 }
43 prefix = srsran::to_c_str(buffer);
44 }
45 const char* to_c_str() const { return prefix.c_str(); }

47private:
48 std::string prefix;
