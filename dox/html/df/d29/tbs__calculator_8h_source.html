34/// \brief Collection of parameters required to calculate the TBS for a shared channel transmission.
35///
36/// The configuration applies to the determination of the transport block size (TBS) for both PDSCH, as described in
37/// TS38.214 Section 5.1.3.2, and PUSCH, as described in TS38.214 Section 6.1.4.2.
38struct tbs_calculator_configuration {
39 /// \brief Number of OFDM symbols allocated to the SCH within the slot.
40 ///
41 /// Parameter \f$N_{\textup{symb}}^{\textup{sh}}\f$ for both PDSCH and PUSCH.
42 unsigned nof_symb_sh;
43 /// \brief Number of REs for DM-RS per PRB in the allocated duration.
44 ///
45 /// Parameter \f$N_{\textup{DMRS}}^{\textup{PRB}}\f$ for both PDSCH and PUSCH. It includes the overhead of the DM-RS
46 /// CDM groups without data.
47 unsigned nof_dmrs_prb;
48 /// \brief Number of REs configured as overhead.
49 ///
50 /// Parameter \f$N_{\textup{oh}}^{\textup{PRB}}\f$. Possible values are {0, 6, 12, 18}.
51 ///
52 /// Specifically, for PDSCH, the value is zero if:
53 /// - the PDSCH is scheduled by PDCCH with a CRC scrambled by SI-RNTI, RA-RNTI or P-RNTI, or
54 /// - the higher layer parameter \e xOverhead in \e PDSCH-ServingCellconfig is not configured.
55 ///
56 /// Otherwise, the value is provided by the parameter \e xOverhead in \e PDSCH-ServingCellconfig.
57 ///
58 /// Similarly, for PUSCH, the value is zero if:
59 /// - the transport block corresponds to a Msg3 transmission, or
60 /// - the higher layer parameter \e xOverhead in \e PUSCH-ServingCellConfig is not configured.
61 ///
62 /// Otherwise, the value is provided by the parameter \e xOverhead in \e PUSCH-ServingCellConfig.
63 unsigned nof_oh_prb;
64 /// \brief Modulation and coding scheme for the SCH transmission.
65 ///
66 /// The modulation scheme yields the number of bits per RE, parameter \f$Q_m\f$ for both PDSCH and PUSCH. Possible
67 /// values are:
68 /// - QPSK,
69 /// - QAM-16,
70 /// - QAM-64, and
71 /// - 256-QAM.
72 sch_mcs_description mcs_descr;
73 /// \brief Number of layers.
74 ///
75 /// Parameter \f$\nu\f$. For PDSCH, the possible values are:
76 /// - {1, ..., 4} for one codeword, and
77 /// - {5, ..., 8} for two codewords.
78 ///
79 /// For PUSCH, which only admits single-codeword transmissions, the only possible values are {1, ..., 4}.
80 unsigned nof_layers;
81 /// \brief TB scaling field.
82 ///
83 /// Provides the scaling factor \f$S\f$ according to TS38.214 Table 5.1.3.2-2. Possible values are {0, 1, 2}.
84 ///
85 /// For the PDSCH assigned by a PDCCH with DCI format 1_0 with CRC scrambled by P-RNTI or RA-RNTI, the value is
86 /// provided by the <em>TB scaling</em> field of the DCI. For other PDSCHs, it must be set to zero.
87 ///
88 /// For PUSCH, the parameter is meaningless and must be set to zero.
89 unsigned tb_scaling_field;
90 /// \brief Total number of PRBs allocated to the UE.
91 ///
92 /// Parameter \f$n_{PRB}\f$. Possibles values are {1, ..., 275}.
93 unsigned n_prb;
94};
95

99float tbs_calculator_pdsch_get_scaling_factor(unsigned scaling);

101/// \brief Gets the smallest TBS in bits (from TS 38.214 Table 5.1.3.2-1) greater than or equal to a given value.
102///
103/// \param[in] nof_info_prime Quantized intermediate number of information bits, as per TS38.214 Section 5.1.3.2.
104/// \remark \c nof_info_prime must not be greater than 3824 bits.
