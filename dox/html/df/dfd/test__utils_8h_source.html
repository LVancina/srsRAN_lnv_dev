33/// This custom sink intercepts log messages to count error and warning log entries.
34class log_sink_spy : public srslog::sink
35{
36public:
37 explicit log_sink_spy(std::unique_ptr<srslog::log_formatter> f) :
38 srslog::sink(std::move(f)), s(srslog::get_default_sink())
39 {
40 error_counter.store(0);
41 warning_counter.store(0);
42 }
43
44 /// Identifier of this custom sink.
45 static const char* name() { return "log_sink_spy"; }
46
47 /// Returns the number of log entries tagged as errors.
48 unsigned get_error_counter() const
49 {
50 // Flush to make sure all entries have been processed by the backend.
51 srslog::flush();
52 return error_counter.load();
53 }
54
55 /// Returns the number of log entries tagged as warnings.
56 unsigned get_warning_counter() const
57 {
58 // Flush to make sure all entries have been processed by the backend.
59 srslog::flush();
60 return warning_counter.load();
61 }
62
63 /// Resets the counters back to 0.
64 void reset_counters()
65 {
66 // Flush to make sure all entries have been processed by the backend.
67 srslog::flush();
68 error_counter.store(0);
69 warning_counter.store(0);
70 }
71
72 srslog::detail::error_string write(srslog::detail::memory_buffer buffer) override
73 {
74 std::string entry(buffer.data(), buffer.size());
75 if (entry.find("[E]") != std::string::npos) {
76 error_counter.fetch_add(1);
77 } else if (entry.find("[W]") != std::string::npos) {
78 warning_counter.fetch_add(1);
79 }
80
81 return s.write(buffer);
82 }
83
84 srslog::detail::error_string flush() override { return s.flush(); }
85
86private:
87 srslog::sink& s;
88 std::atomic<unsigned> error_counter;
89 std::atomic<unsigned> warning_counter;
90};

92/// This custom sink intercepts log messages allowing users to check if a certain log entry has been generated.
93/// Calling spy.has_message("something") will return true if any log entries generated so far contain the string
94/// "something".
95/// The log entries history can be cleared with reset so old entries can be discarded.
96class log_sink_message_spy : public srslog::sink
97{
98public:
99 explicit log_sink_message_spy(std::unique_ptr<srslog::log_formatter> f) :
100 srslog::sink(std::move(f)), s(srslog::get_default_sink())
101 {
102 }
103
104 /// Identifier of this custom sink.
105 static const char* name() { return "log_sink_message_spy"; }
106
107 /// Discards all registered log entries.
108 void reset()
109 {
110 // Flush to make sure all entries have been processed by the backend.
111 srslog::flush();
112 entries.clear();
113 }
114
115 /// Returns true if the string in msg is found in the registered log entries.
116 bool has_message(const std::string& msg) const
117 {
118 srslog::flush();
119 return std::find_if(entries.cbegin(), entries.cend(), [&](const std::string& entry) {
120 return entry.find(msg) != std::string::npos;
121 }) != entries.cend();
122 }
123
124 srslog::detail::error_string write(srslog::detail::memory_buffer buffer) override
125 {
126 entries.emplace_back(buffer.data(), buffer.size());
127
128 return s.write(buffer);
129 }
130
131 srslog::detail::error_string flush() override { return s.flush(); }
132
133private:
134 srslog::sink& s;
135 std::vector<std::string> entries;
136};

138/**
139 * Delimits beginning/ending of a test with the following console output:
140 * ============= [Test <Name of the Test>] ===============
141 * <test log>
142 * =======================================================
143 */
144class test_delimit_logger
145{
146 const size_t delimiter_length = 128;
147
148public:
149 template <typename... Args>
150 explicit test_delimit_logger(const char* test_name_fmt, Args&&... args)
151 {
152 test_name = fmt::format(test_name_fmt, std::forward<Args>(args)...);
153 std::string name_str = fmt::format("[ Test \"{}\" ]", test_name);
154 double nof_repeats = (delimiter_length - name_str.size()) / 2.0;
155 fmt::print("{0:=>{1}}{2}{0:=>{3}}\n", "", (int)floor(nof_repeats), name_str, (int)ceil(nof_repeats));
156 }
157 test_delimit_logger(const test_delimit_logger&) = delete;
158 test_delimit_logger(test_delimit_logger&&) = delete;
159 test_delimit_logger& operator=(const test_delimit_logger&) = delete;
160 test_delimit_logger& operator=(test_delimit_logger&&) = delete;
161 ~test_delimit_logger()
162 {
163 srslog::flush();
164 fmt::print("{:=>{}}\n", "", delimiter_length);
165 }
166
167private:
168 std::string test_name;
169};

249/// \brief This class creates a random generation interface that is suitable for unit tests. The user has the
250/// ability to set a random seed to reproduce tests.
251class test_rgen
252{
253public:
254 /// \brief Get test pseudo-random generator.
255 static std::mt19937& get()
256 {
257 thread_local std::mt19937& random_generator = init(std::random_device{}());
258 return random_generator;
259 }
260
261 /// \brief Set random seed for the test pseudo-random generator.
262 /// If this function is never called, the pseudo-random generator will be initialized with a random seed using
263 /// std::random_device. If it is called once, the seed will be the one passed as an argument. If it is called more
264 /// than once, the test will abort. Thus, there should not be race conditions in seed initialization.
265 static void set_seed(uint32_t seed)
266 {
267 init(seed);
268 if (get_seed() != seed) {
269 report_fatal_error("Trying to set a random seed multiple times (not thread-safe)");
270 }
271 }
272
273 /// \brief Return a random integer with uniform distribution within the specified bounds.
274 template <typename Integer>
275 static Integer uniform_int(Integer min, Integer max)
276 {
277 return std::uniform_int_distribution<Integer>{min, max}(get());
278 }
279
280 template <typename Integer>
281 static Integer uniform_int()
282 {
283 return uniform_int(std::numeric_limits<Integer>::min(), std::numeric_limits<Integer>::max());
284 }
285
286 /// \brief Return a vector of integers with specified size filled with random values.
287 template <typename Integer>
288 static std::vector<Integer> random_vector(size_t sz)
289 {
290 std::vector<Integer> vec(sz);
291 for (unsigned i = 0; i != sz; ++i) {
292 vec[i] = test_rgen::uniform_int<Integer>();
293 }
294 return vec;
295 }
296
297private:
298 static std::mt19937& init(uint32_t seed)
299 {
300 thread_local std::mt19937 random_generator(get_seed(seed));
301 return random_generator;
302 }
303
304 static uint32_t get_seed(uint32_t seed = 0)
305 {
306 static uint32_t locked_seed = [](uint32_t seed_) {
307 fmt::print("-- TEST random generator seed: {}\n", seed_);
308 return seed_;
309 }(seed);
310 return locked_seed;
311 }
312};

316 static const bool dummy = []() {
317 srsran::test_rgen::set_seed(seed);
