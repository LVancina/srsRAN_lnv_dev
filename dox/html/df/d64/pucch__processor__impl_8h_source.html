34/// Implements a parameter validator for \ref pucch_processor_impl.
35class pucch_pdu_validator_impl : public pucch_pdu_validator
36{
37public:
38 /// \brief Constructs a PUCCH processor validator.
39 ///

41 /// layers.
42 ///
43 /// \param[in] ce_dims_ Provides the channel estimates dimensions.

45 {
46 // Do nothing.
47 }
48
49 // See interface for documentation.
50 bool is_valid(const pucch_processor::format0_configuration& config) const override { return true; };
51 bool is_valid(const pucch_processor::format1_configuration& config) const override;
52 bool is_valid(const pucch_processor::format2_configuration& config) const override;
53 bool is_valid(const pucch_processor::format3_configuration& config) const override { return true; }

55
56private:
57 /// Maximum transmit and receive resource grid dimensions handled by the PUCCH processor.
58 channel_estimate::channel_estimate_dimensions ce_dims;
59};

61/// Implementation of the PUCCH processor interface.
62class pucch_processor_impl : public pucch_processor
63{
64public:
65 /// Maximum supported PUCCH Format 2 UCI payload length in number of bits.
66 static constexpr unsigned FORMAT2_MAX_UCI_NBITS = 1706;
67
68 // See interface for documentation.
69 pucch_processor_result process(const resource_grid_reader& grid, const format0_configuration& config) override
70 {
71 pucch_detector::format0_configuration detector_config;
72 std::pair<pucch_uci_message, channel_state_information> detection_result = detector->detect(grid, detector_config);
73
74 pucch_processor_result result;
75 result.message = detection_result.first;
76 result.csi = detection_result.second;
77
78 return result;
79 }
80
81 // See interface for documentation.
82 pucch_processor_result process(const resource_grid_reader& grid, const format1_configuration& config) override;
83
84 // See interface for documentation.
85 pucch_processor_result process(const resource_grid_reader& grid, const format2_configuration& config) override;
86
87 // See interface for documentation.
88 pucch_processor_result process(const resource_grid_reader& grid, const format3_configuration& config) override
89 {
90 // TBD.
91 return pucch_processor_result();
92 }
93
94 // See interface for documentation.
95 pucch_processor_result process(const resource_grid_reader& grid, const format4_configuration& config) override
96 {
97 // TBD.
98 return pucch_processor_result();
99 }
100
101 /// PUCCH processor constructor.
102 pucch_processor_impl(std::unique_ptr<dmrs_pucch_processor> channel_estimator_format_1_,
103 std::unique_ptr<dmrs_pucch_processor> channel_estimator_format_2_,
104 std::unique_ptr<pucch_detector> detector_,
105 std::unique_ptr<pucch_demodulator> demodulator_,
106 std::unique_ptr<uci_decoder> decoder_,
107 const channel_estimate::channel_estimate_dimensions& estimates_dimensions) :
108 channel_estimator_format_1(std::move(channel_estimator_format_1_)),
109 channel_estimator_format_2(std::move(channel_estimator_format_2_)),
110 detector(std::move(detector_)),
111 demodulator(std::move(demodulator_)),
112 decoder(std::move(decoder_)),
113 estimates(estimates_dimensions),
114 max_sizes(estimates_dimensions)
115 {
116 srsran_assert(channel_estimator_format_1, "Invalid channel estimator for PUCCH Format 1.");
117 srsran_assert(channel_estimator_format_2, "Invalid channel estimator for PUCCH Format 2.");
118 srsran_assert(detector, "Invalid detector.");
119 srsran_assert(demodulator, "Invalid PUCCH demodulator.");
120 srsran_assert(decoder, "Invalid UCI decoder.");
121 }
122
123private:
124 /// Validates PUCCH Format 1 configuration.
125 void assert_format1_config(const pucch_processor::format1_configuration& config);
126 /// Validates PUCCH Format 2 configuration.
127 void assert_format2_config(const pucch_processor::format2_configuration& config);
128
129 /// Channel estimator for PUCCH Format 1.
130 std::unique_ptr<dmrs_pucch_processor> channel_estimator_format_1;
131 /// Channel estimator for PUCCH Format 2.
132 std::unique_ptr<dmrs_pucch_processor> channel_estimator_format_2;
133 /// PUCCH detector for 1 or 2 bits, using format 0 and 1.
134 std::unique_ptr<pucch_detector> detector;
135 /// PUCCH demodulator for more than 2 bits, using formats 2, 3 and 4.
136 std::unique_ptr<pucch_demodulator> demodulator;
137 /// UCI decoder for more than 2 bits, using formats 2, 3 and 4.
138 std::unique_ptr<uci_decoder> decoder;
139 /// Temporal channel estimates.
140 channel_estimate estimates;
141 /// Maximum RB, symbol and channel sizes handled by the processor.
142 channel_estimate::channel_estimate_dimensions max_sizes;
143 /// Temporal LLR storage.
144 std::array<log_likelihood_ratio, pucch_constants::MAX_NOF_LLR> temp_llr;
145};
