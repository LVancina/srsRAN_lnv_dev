28/// PUSCH demodulator interface to notify the demodulation statistics of a codeword.
29class pusch_demodulator_notifier
30{
31public:
32 /// Collects PUSCH demodulation statistics.
33 struct demodulation_stats {
34 /// Estimated Signal-to-Interference-plus-Noise Ratio (SINR) at the output of the equalizer.
35 optional<float> sinr_dB;
36 /// Measured EVM.
37 optional<float> evm;
38 };
39
40 /// Default destructor.
41 virtual ~pusch_demodulator_notifier() = default;
42
43 /// Notifies intermediate PUSCH demodulator statistics.
44 virtual void on_provisional_stats(const demodulation_stats& stats) = 0;
45
46 /// Notifies the end of PUSCH processing and the final demodulator statistics.
47 virtual void on_end_stats(const demodulation_stats& stats) = 0;
48};
