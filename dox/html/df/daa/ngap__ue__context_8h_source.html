60 /// \brief Checks whether a UE with the given RAN UE ID exists.

65 /// \brief Checks whether a UE with the given UE index exists.
66 /// \param[in] ue_index The UE index used to find the UE.
67 /// \return True when a UE for the given UE index exists, false otherwise.
68 bool contains(ue_index_t ue_index) const
69 {
70 if (ue_index_to_ran_ue_id.find(ue_index) == ue_index_to_ran_ue_id.end()) {
71 return false;
72 }
73 if (ues.find(ue_index_to_ran_ue_id.at(ue_index)) == ues.end()) {
74 return false;
75 }
76 return true;
77 }

120 ngap_ue_context& add_ue(ue_index_t ue_index, ran_ue_id_t ran_ue_id, timer_manager& timers, task_executor& task_exec)
121 {
122 srsran_assert(ue_index != ue_index_t::invalid, "Invalid ue_index={}", ue_index);
123 srsran_assert(ran_ue_id != ran_ue_id_t::invalid, "Invalid ran_ue_id={}", ran_ue_id);
124
125 logger.debug("ue={} ran_ue_id={}: Adding NGAP UE context", ue_index, ran_ue_id);
126 ues.emplace(std::piecewise_construct,
127 std::forward_as_tuple(ran_ue_id),
128 std::forward_as_tuple(ue_index, ran_ue_id, timers, task_exec));
129 ue_index_to_ran_ue_id.emplace(ue_index, ran_ue_id);
130 return ues.at(ran_ue_id);
131 }

147 void update_ue_index(ue_index_t new_ue_index, ue_index_t old_ue_index)
148 {
149 srsran_assert(new_ue_index != ue_index_t::invalid, "Invalid new_ue_index={}", new_ue_index);
150 srsran_assert(old_ue_index != ue_index_t::invalid, "Invalid old_ue_index={}", old_ue_index);
151 srsran_assert(ue_index_to_ran_ue_id.find(old_ue_index) != ue_index_to_ran_ue_id.end(),
152 "ue={}: RAN-UE-ID not found",
153 old_ue_index);
154
155 ran_ue_id_t ran_ue_id = ue_index_to_ran_ue_id.at(old_ue_index);
156
157 srsran_assert(ues.find(ran_ue_id) != ues.end(), "ran_ue_id={}: NGAP UE context not found", ran_ue_id);
158
159 // Update UE context
160 ues.at(ran_ue_id).ue_ids.ue_index = new_ue_index;
161
162 // Update lookups
163 ue_index_to_ran_ue_id.emplace(new_ue_index, ran_ue_id);
164 ue_index_to_ran_ue_id.erase(old_ue_index);
165
166 ues.at(ran_ue_id).logger.set_prefix(
167 {ues.at(ran_ue_id).ue_ids.ue_index, ran_ue_id, ues.at(ran_ue_id).ue_ids.amf_ue_id});
168
169 ues.at(ran_ue_id).logger.log_debug("Updated UE index from ue_index={}", old_ue_index);
170 }

172 void remove_ue_context(ue_index_t ue_index)
173 {
174 srsran_assert(ue_index != ue_index_t::invalid, "Invalid ue_index={}", ue_index);
175
176 if (ue_index_to_ran_ue_id.find(ue_index) == ue_index_to_ran_ue_id.end()) {
177 logger.warning("ue={}: RAN-UE-ID not found", ue_index);
178 return;
179 }
180
181 // Remove UE from lookup
182 ran_ue_id_t ran_ue_id = ue_index_to_ran_ue_id.at(ue_index);
183 ue_index_to_ran_ue_id.erase(ue_index);
184
185 if (ues.find(ran_ue_id) == ues.end()) {
186 logger.warning("ran_ue_id={}: NGAP UE context not found", ran_ue_id);
187 return;
188 }
189
190 ues.at(ran_ue_id).logger.log_debug("Removing NGAP UE context");
191
192 if (ues.at(ran_ue_id).ue_ids.amf_ue_id != amf_ue_id_t::invalid) {
193 amf_ue_id_to_ran_ue_id.erase(ues.at(ran_ue_id).ue_ids.amf_ue_id);
194 }
195 ues.erase(ran_ue_id);
196 }

200 /// \brief Get the next available RAN-UE-ID.
201 ran_ue_id_t allocate_ran_ue_id()
202 {
203 // return invalid when no RAN-UE-ID is available
204 if (ue_index_to_ran_ue_id.size() == MAX_NOF_RAN_UES) {
205 return ran_ue_id_t::invalid;
206 }
207
208 // iterate over all ids starting with the next_ran_ue_id to find the available id
209 while (true) {
210 // Only iterate over ue_index_to_ran_ue_id (size=MAX_NOF_UES_PER_DU)
211 // to avoid iterating over all possible values of ran_ue_id_t (size=2^32-1)
212 auto it = std::find_if(ue_index_to_ran_ue_id.begin(), ue_index_to_ran_ue_id.end(), [this](auto& u) {
213 return u.second == next_ran_ue_id;
214 });
215
216 // return the id if it is not already used
217 if (it == ue_index_to_ran_ue_id.end()) {
218 ran_ue_id_t ret = next_ran_ue_id;
219 // increase the next cu ue f1ap id
220 increase_next_ran_ue_id();
221 return ret;
222 }
223
224 // increase the next cu ue f1ap id and try again
225 increase_next_ran_ue_id();
226 }
227
228 return ran_ue_id_t::invalid;
229 }
230
231protected:
232 ran_ue_id_t next_ran_ue_id = ran_ue_id_t::min;
233
234private:
235 srslog::basic_logger& logger;
236
237 inline void increase_next_ran_ue_id()
238 {
239 if (next_ran_ue_id == ran_ue_id_t::max) {
240 // reset ran ue id counter
241 next_ran_ue_id = ran_ue_id_t::min;
242 } else {
243 // increase ran ue id counter
244 next_ran_ue_id = uint_to_ran_ue_id(ran_ue_id_to_uint(next_ran_ue_id) + 1);
245 }
246 }
247
248 // Note: Given that UEs will self-remove from the map, we don't want to destructor to clear the lookups beforehand.
249 std::unordered_map<ue_index_t, ran_ue_id_t> ue_index_to_ran_ue_id; // indexed by ue_index
250 std::unordered_map<amf_ue_id_t, ran_ue_id_t> amf_ue_id_to_ran_ue_id; // indexed by amf_ue_id_t
251 std::unordered_map<ran_ue_id_t, ngap_ue_context> ues; // indexed by ran_ue_id_t
252};
