30/// This class is a wrapper of a file handle that buffers the input data into an internal buffer and writes its contents
31/// to the file once the buffer is full or in object destruction.
32class buffered_file_sink : public sink
33{
34public:
35 buffered_file_sink(std::string filename_, std::size_t capacity, std::unique_ptr<log_formatter> f) :
36 sink(std::move(f)), filename(std::move(filename_))
37 {
38 buffer.reserve(capacity);
39 }
40
41 ~buffered_file_sink() override { flush_buffer(); }
42
43 buffered_file_sink(const buffered_file_sink& other) = delete;
44 buffered_file_sink& operator=(const buffered_file_sink& other) = delete;
45
46 detail::error_string write(detail::memory_buffer input_buffer) override
47 {
48 // Create a new file the first time we hit this method.
49 if (!is_file_created) {
50 is_file_created = true;
51 assert(!handler && "No handler should be created yet");
52 if (auto err_str = handler.create(filename)) {
53 return err_str;
54 }
55 }
56
57 if (has_room_for(input_buffer.size())) {
58 buffer.insert(buffer.end(), input_buffer.begin(), input_buffer.end());
59 return {};
60 }
61
62 return flush_buffer();
63 }
64
65 detail::error_string write_error(detail::memory_buffer input_buffer) override { return write(input_buffer); }
66
67 detail::error_string flush() override
68 {
69 if (auto err = flush_buffer()) {
70 return err;
71 }
72 return handler.flush();
73 }
74
75private:
76 /// Returns true if the internal buffer has room for the specified input size,
77 /// otherwise returns false.
78 bool has_room_for(std::size_t s) const { return s + buffer.size() < buffer.capacity(); }
79
80 /// Flushes the buffer contents into the file.
81 detail::error_string flush_buffer()
82 {
83 if (buffer.empty()) {
84 return {};
85 }
86 auto err = handler.write(detail::memory_buffer(buffer.data(), buffer.size()));
87 buffer.clear();
88 return err;
89 }
90
91private:
92 const std::string filename;
93 file_utils::file handler;
94 std::vector<char> buffer;
95 bool is_file_created = false;
96};
