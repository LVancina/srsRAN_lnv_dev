138struct sec_128_as_config {
139 sec_domain domain;
140 optional<sec_128_key> k_128_int; // Optional in E1AP, see TS 38.463 Sec. 9.4.5
141 sec_128_key k_128_enc;
142 optional<integrity_algorithm> integ_algo; // Optional in E1AP, see TS 38.463 Sec. 9.4.5
143 ciphering_algorithm cipher_algo;
144
145 bool operator==(const sec_128_as_config& rhs) const
146 {
147 if (integ_algo != rhs.integ_algo) {
148 return false;
149 }
150 if (cipher_algo != rhs.cipher_algo) {
151 return false;
152 }
153 if (domain != rhs.domain) {
154 return false;
155 }
156 if (k_128_int != rhs.k_128_int) {
157 return false;
158 }
159 if (k_128_enc != rhs.k_128_enc) {
160 return false;
161 }
162 return true;
163 }

231/// \brief Fill tailing bits of the last byte of a contiguous memory with zeros.
232/// \param[inout] data Pointer to the contiguous memory to operate on
233/// \param[in] length_bits Number of occupied bits in the whole memory buffer that shall not be zeroed
234inline void zero_tailing_bits(uint8_t* data, uint32_t length_bits)
235{
236 uint8_t bits = (8 - (length_bits & 0x07)) & 0x07;
237 data[(length_bits + 7) / 8 - 1] &= (uint8_t)(0xff << bits);
238}
239
240/// \brief Fill tailing bits of a given byte with zeros.
241/// \param[inout] tail_byte Reference to the byte to operate on
242/// \param[in] length_bits Number of occupied bits in the tail_byte that shall not be zeroed
243inline void zero_tailing_bits(uint8_t& tail_byte, uint8_t length_bits)
244{
245 uint8_t bits = (8 - (length_bits & 0x07)) & 0x07;
246 tail_byte &= (uint8_t)(0xff << bits);
247}

249inline std::string sec_as_key_to_string(const sec_key& key)
250{
251 std::stringstream ss;
252 for (auto val : key) {
253 ss << std::hex << std::setw(2) << std::setfill('0') << (int)val;
254 }
255 return ss.str();
256}
