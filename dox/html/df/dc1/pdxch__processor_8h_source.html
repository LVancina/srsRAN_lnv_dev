32/// \brief PDxCH processor main interface.
33///
34/// Provides a single access point to all downlink channel processor interfaces, that is \ref pdxch_processor_notifier,
35/// \ref pdxch_processor_request_handler and \ref pdxch_processor_baseband.
36///
37/// \note All implementations of this interface must trigger an assertion error if the PDxCH processor is not connected
38/// to a notifier before calling \ref get_request_handler() or \ref get_baseband().
39class pdxch_processor
40{
41public:
42 /// Default destructor.
43 virtual ~pdxch_processor() = default;
44
45 /// Connects the PDxCH processor with a notifier.
46 virtual void connect(pdxch_processor_notifier& notifier) = 0;
47
48 /// \brief Gets the PDxCH processor request handler.
49 /// \return A reference to the internal PDxCH processor request handler.
50 virtual pdxch_processor_request_handler& get_request_handler() = 0;
51
52 /// \brief Gets the PDxCH processor baseband interface.
53 /// \return A reference to the internal PDxCH processor baseband interface.
54 virtual pdxch_processor_baseband& get_baseband() = 0;
55};
