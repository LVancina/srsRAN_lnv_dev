58/// \brief Convolution between two sequences.
59///
60/// Computes the convolution between two sequences \f$x\f$ and \f$y\f$, namely \f$z[n] = \sum_{i} x[i] y[n-i]\f$ and
61/// returns its central part, of the \e same size as the first input sequence.
62/// \tparam V Output type (must be compatible with a span of arithmetic/complex values).
63/// \tparam T Type of the first input sequence (must be compatible with a span of arithmetic/complex values).
64/// \tparam U Type of the second input sequence (must be compatible with a span of arithmetic/complex values).
65/// \param[out] out Output sequence - convolution between the input sequences.
66/// \param[in] x_v First input sequence.
67/// \param[in] y_v Second input sequence.
68/// \remark An assertion is raised if the length of the second sequence is larger than the length of the first one.
69/// \remark This implementation of the convolution is not based on FFT and should only be used when the second input
70/// sequence is "very" short.
71template <typename V, typename T, typename U>
72void convolution_same(V&& out, const T& x_v, const U& y_v)
73{
74 static_assert((detail::is_arithmetic_span_compatible<T>::value || detail::is_complex_span_compatible<T>::value),
75 "Template type is not compatible with a span of arithmetics or complex floats.");
76 static_assert((detail::is_arithmetic_span_compatible<U>::value || detail::is_complex_span_compatible<U>::value),
77 "Template type is not compatible with a span of arithmetics or complex floats.");
78 static_assert((detail::is_arithmetic_span_compatible<V>::value || detail::is_complex_span_compatible<V>::value),
79 "Template type is not compatible with a span of arithmetics or complex floats.");
80 srsran_srsvec_assert_size(out, x_v);
81
82 unsigned x_size = x_v.size();
83 unsigned y_size = y_v.size();
84 srsran_assert((y_size <= x_size),
85 "The current implementation of the convolution is only defined when the second input is not longer "
86 "than the first one.");
87
88 using Type_y = typename U::value_type;
89 using Type_x = typename T::value_type;
90 using Type_out = typename std::remove_reference<V>::type::value_type;
91
92 srsran_assert(detail::check_different(out, x_v), "Cannot override input with output.");
93
94 srsvec::zero(out);
95 span<const Type_x> x(x_v);
96 span<const Type_y> y(y_v);
97 Type_out init = 0;
98
99 // At the beginning, x and y do not fully overlap.
100 unsigned y_mid = y_size / 2;
101 unsigned i_out = 0;
102 unsigned out_end = x.size() - y_mid;
103 for (unsigned n_els = y_mid + 1; n_els != y_size; ++n_els) {
104 span<const Type_y> y_local = y.first(n_els);
105 span<const Type_x> x_local = x.first(n_els);
106 // Note that y_local is reversed.
107 out[i_out++] = std::inner_product(x_local.begin(), x_local.end(), y_local.rbegin(), init);
108 }
109
110 // In the central part, y fully overlaps with x.
111 detail::multiply_and_accumulate(out, x, y);
112
113 // For the final part, we again take into account the partial overlap.
114 unsigned start = i_out;
115 for (unsigned n_els = x_size - out_end + start; n_els != start; --n_els) {
116 span<const Type_y> y_local = y.last(n_els);
117 span<const Type_x> x_local = x.last(n_els);
118 // Note that y_local is reversed.
119 out[out_end++] = std::inner_product(x_local.begin(), x_local.end(), y_local.rbegin(), init);
120 }
121}
