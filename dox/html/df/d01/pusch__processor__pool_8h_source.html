34/// Free PUSCH processor identifier list type.
35using pusch_processor_free_list =
36 concurrent_queue<unsigned, concurrent_queue_policy::lockfree_mpmc, concurrent_queue_wait_policy::sleep>;

38/// PUSCH processor wrapper. It appends its identifier into the free list when the processing is finished.
39class pusch_processor_wrapper : public pusch_processor, private pusch_processor_result_notifier
40{
41public:
42 /// Creates a pusch processor wrapper from another pusch processor.
43 explicit pusch_processor_wrapper(unsigned index_,
44 pusch_processor_free_list& free_list_,
45 std::unique_ptr<pusch_processor> processor_) :
46 index(index_), free_list(free_list_), notifier(nullptr), processor(std::move(processor_))
47 {
48 srsran_assert(processor, "Invalid pusch processor.");
49 }
50
51 /// Creates a PUSCH processor wrapper from another PUSCH processor wrapper.
52 pusch_processor_wrapper(pusch_processor_wrapper&& other) :
53 index(other.index), free_list(other.free_list), notifier(other.notifier), processor(std::move(other.processor))
54 {
55 other.notifier = nullptr;
56 }
57
58 // See pusch_processor interface for documentation.
59 void process(span<uint8_t> data,
60 unique_rx_buffer rm_buffer,
61 pusch_processor_result_notifier& notifier_,
62 const resource_grid_reader& grid,
63 const pdu_t& pdu) override
64 {
65 // Save original notifier.
66 notifier = &notifier_;
67
68 // Process.
69 processor->process(data, std::move(rm_buffer), *this, grid, pdu);
70 }
71
72private:
73 // See pusch_processor_result_notifier for documentation.
74 void on_uci(const pusch_processor_result_control& uci) override
75 {
76 srsran_assert(notifier != nullptr, "Invalid notifier.");
77 notifier->on_uci(uci);
78 }
79
80 // See pusch_processor_result_notifier for documentation.
81 void on_sch(const pusch_processor_result_data& sch) override
82 {
83 srsran_assert(notifier != nullptr, "Invalid notifier.");
84
85 // Notify the completion of the processing.
86 notifier->on_sch(sch);
87
88 // Return the pusch processor identifier to the free list.
89 free_list.push_blocking(index);
90 }
91
92 /// Processor identifier within the pool.
93 unsigned index;
94 /// List of free processors.
95 pusch_processor_free_list& free_list;
96 /// Current pusch processor notifier.
97 pusch_processor_result_notifier* notifier = nullptr;
98 /// Wrapped pusch processor.
99 std::unique_ptr<pusch_processor> processor;
100};
101} // namespace detail

103/// \brief PUSCH processor pool
104///
105/// It contains PUSCH processors that are asynchronously executed. The processing of a PUSCH transmission is dropped if
106/// there are no free PUSCH processors available.
107///
108class pusch_processor_pool : public pusch_processor
109{
110public:
111 /// Creates a PUSCH processor pool from a list of processors. Ownership is transferred to the pool.
112 explicit pusch_processor_pool(span<std::unique_ptr<pusch_processor>> processors_) : free_list(processors_.size())
113 {
114 unsigned index = 0;
115 for (std::unique_ptr<pusch_processor>& processor : processors_) {
116 free_list.push_blocking(index);
117 processors.emplace_back(detail::pusch_processor_wrapper(index++, free_list, std::move(processor)));
118 }
119 }
120
121 // See interface for documentation.
122 void process(span<uint8_t> data,
123 unique_rx_buffer rm_buffer,
124 pusch_processor_result_notifier& notifier,
125 const resource_grid_reader& grid,
126 const pdu_t& pdu) override
127 {
128 // Try to get a worker.
129 optional<unsigned> index = free_list.try_pop();
130
131 // If no worker is available.
132 if (!index.has_value()) {
133 srslog::fetch_basic_logger("PHY").warning("Insufficient number of PUSCH processors. Dropping PUSCH {:s}.", pdu);
134 return;
135 }
136
137 // Process pusch.
138 processors[index.value()].process(data, std::move(rm_buffer), notifier, grid, pdu);
139 }
140
141private:
142 std::vector<detail::pusch_processor_wrapper> processors;
143 detail::pusch_processor_free_list free_list;
144};
