36/// Parameters that affect the detection threshold value.
37struct threshold_params {
38 /// Number of receive antenna ports.
39 unsigned nof_rx_ports = 0;
40 /// Subcarrier spacing.
41 prach_subcarrier_spacing scs = prach_subcarrier_spacing::invalid;
42 /// PRACH format type.
43 prach_format_type format = prach_format_type::invalid;
44 /// Zero correlation zone.
45 unsigned zero_correlation_zone = 0;
46 /// Flag to activate non-coherent combining of PRACH symbols.
47 bool combine_symbols = false;
48};
49
50/// Overloaded less-than comparison operator.
51inline bool operator<(const threshold_params& lhs, const threshold_params& rhs)
52{
53 if (lhs.nof_rx_ports < rhs.nof_rx_ports) {
54 return true;
55 }
56 if (lhs.nof_rx_ports > rhs.nof_rx_ports) {
57 return false;
58 }
59
60 // If we are here, nof_rx_ports is equal.
61 if (lhs.scs < rhs.scs) {
62 return true;
63 }
64 if (lhs.scs > rhs.scs) {
65 return false;
66 }
67
68 // If we are here, nof_rx_ports and scs are equal.
69 if (lhs.format < rhs.format) {
70 return true;
71 }
72 if (lhs.format > rhs.format) {
73 return false;
74 }
75
76 // If we are here, nof_rx_ports, scs and format are equal.
77 if (lhs.zero_correlation_zone < rhs.zero_correlation_zone) {
78 return true;
79 }
80 if (lhs.zero_correlation_zone > rhs.zero_correlation_zone) {
81 return false;
82 }
83
84 // If we are here, nof_rx_ports, scs, format and zero_correlation_zone are equal.
85 if (!lhs.combine_symbols && rhs.combine_symbols) {
86 return true;
87 }
88 return false;
89}
90
91/// Overloaded equal-to comparison operator.
92inline bool operator==(const threshold_params& lhs, const threshold_params& rhs)
93{
94 if (lhs.nof_rx_ports != rhs.nof_rx_ports) {
95 return false;
96 }
97
98 // If we are here, nof_rx_ports is equal.
99 if (lhs.scs != rhs.scs) {
100 return false;
101 }
102
103 // If we are here, nof_rx_ports and scs are equal.
104 if (lhs.format != rhs.format) {
105 return false;
106 }
107
108 // If we are here, nof_rx_ports, scs and format are equal.
109 if (lhs.zero_correlation_zone != rhs.zero_correlation_zone) {
110 return false;
111 }
112
113 // If we are here, nof_rx_ports, scs, format and zero_correlation_zone are equal.
114 if (lhs.combine_symbols != rhs.combine_symbols) {
115 return false;
116 }
117 return true;
118}

120/// Manages the mapping between PRACH configuration and the (threshold, margin) pairs.
121class threshold_and_margin_finder
122{
123public:
124 /// Flag applied to each threshold entry.
125 enum class threshold_flag {
126 /// Threshold entry not suitable for proper PRACH detection.
127 red,
128 /// Threshold entry with suboptimal PRACH detection performance.
129 orange,
130 /// Threshold entry meeting PRACH detection requirements.
131 green
132 };
133
134 using threshold_and_margin_type = std::pair<float, unsigned>;
135
136 struct threshold_entry {
137 threshold_params configuration;
138 threshold_and_margin_type threshold_and_margin;
139 threshold_flag flag;
140 };
141
142 /// Constructor: receives the list of pairings and ensures it is sorted.
143 explicit threshold_and_margin_finder(span<const threshold_entry> in) :
144 sorted_thresholds_and_margins(in.begin(), in.end())
145 {
146 std::sort(sorted_thresholds_and_margins.begin(),
147 sorted_thresholds_and_margins.end(),
148 [](const threshold_entry& a, const threshold_entry& b) { return (a.configuration < b.configuration); });
149 }
150
151 /// Retrieves the (threshold, margin) pair corresponding to the given configuration.
152 threshold_and_margin_type get(const threshold_params& params) const
153 {
154 auto it =
155 std::lower_bound(sorted_thresholds_and_margins.begin(),
156 sorted_thresholds_and_margins.end(),
157 params,
158 [](const threshold_entry& a, const threshold_params& b) { return (a.configuration < b); });
159
160 if (it != sorted_thresholds_and_margins.end()) {
161 return it->threshold_and_margin;
162 }
163
164 // todo(david): once all cases are covered, replace this by a srsran_assert.
165 if (is_long_preamble(params.format)) {
166 return {/* threshold */ 2.0F, /* margin */ 5};
167 }
168 return {/* threshold */ 0.3F, /* margin */ 12};
169 }
170
171 threshold_flag check_flag(const threshold_params& params) const
172 {
173 auto it =
174 std::lower_bound(sorted_thresholds_and_margins.begin(),
175 sorted_thresholds_and_margins.end(),
176 params,
177 [](const threshold_entry& a, const threshold_params& b) { return (a.configuration < b); });
178
179 if ((it != sorted_thresholds_and_margins.end()) && (it->configuration == params)) {
180 return it->flag;
181 }
182 return threshold_flag::red;
183 }
184
185private:
186 std::vector<threshold_entry> sorted_thresholds_and_margins;
187};
