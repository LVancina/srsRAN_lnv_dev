27/// \file
28/// \brief Interfaces and implementations of a Resource Element buffer.
29///
30/// The RE buffer stores a number of resource elements for a given amount of slices. A slice can either be a transmit
31/// layer or an antenna port, depending on the context.

37/// Interface for accessing the dimension sizes of a Resource Element buffer.
38class re_buffer_dimensions
39{
40public:
41 /// Default destructor.
42 virtual ~re_buffer_dimensions() = default;
43
44 /// Gets the current number of slices.
45 virtual unsigned get_nof_slices() const = 0;
46
47 /// Gets the current number of Resource Elements.
48 virtual unsigned get_nof_re() const = 0;
49};

51} // namespace detail

53/// Resource Element buffer read-only interface.
54class re_buffer_reader : public detail::re_buffer_dimensions
55{
56public:
57 /// \brief Gets a read-only view of a specific slice.
58 /// \param[in] i_slice Slice identifier.
59 /// \remark An assertion is triggered if the slice identifier is greater than or equal to get_nof_slices().
60 /// \return A read-only view of the specified slice.
61 virtual span<const cf_t> get_slice(unsigned i_slice) const = 0;
62};

64/// Resource Element buffer read-write interface.
65class re_buffer_writer : public detail::re_buffer_dimensions
66{
67public:
68 /// \brief Gets a read-write view of a specific slice.
69 /// \param[in] i_slice Slice identifier.
70 /// \remark An assertion is triggered if the slice identifier is greater than or equal to get_nof_slices().
71 /// \return A read-write view of the specified slice.
72 virtual span<cf_t> get_slice(unsigned i_slice) = 0;
73};

75/// Implements a resource element buffer reader view, with read-only access methods.
76class re_buffer_reader_view : public re_buffer_reader
77{
78public:
79 /// \brief Constructs a resource element buffer view that gives access to a subspan of the REs for all slices.
80 ///
81 /// \param[in] buffer RE buffer accessed by the view.
82 /// \param[in] offset Offset in number of RE. It determines the start of the RE view across each slice.
83 /// \param[in] count Count in number of RE. It determines the number of RE spanned by the view across each slice.
84 re_buffer_reader_view(const re_buffer_reader& buffer, unsigned offset, unsigned count) :
85 view_offset(offset), view_count(count), data(buffer)
86 {
87 srsran_assert((buffer.get_nof_re() >= offset + count),
88 "A view over REs [{}, {}) exceeds the RE dimension size, i.e., {}.",
89 offset,
90 offset + count,
91 buffer.get_nof_re());
92 }
93
94 // See interface for documentation.
95 unsigned get_nof_slices() const override { return data.get_nof_slices(); }
96
97 // See interface for documentation.
98 unsigned get_nof_re() const override { return view_count; }
99
100 /// \brief Gets a read-only view of a specific slice.
101 ///
102 /// The view spans the buffer REs in the range [\ref view_offset, \ref view_count). An assertion is triggered if the
103 /// slice identifier is greater than or equal to get_nof_slices().
104 ///
105 /// \param[in] i_slice Slice identifier.
106 /// \return A read-only view of the specified slice.
107 span<const cf_t> get_slice(unsigned i_slice) const override
108 {
109 return data.get_slice(i_slice).subspan(view_offset, view_count);
110 }
111
112private:
113 /// Offset of the view across the Resource Elements.
114 unsigned view_offset;
115 /// Number of Resource Elements per slice included in the view.
116 unsigned view_count;
117
118 /// Internal data storage.
119 const re_buffer_reader& data;
120};

166/// Implements a dynamic resource element buffer.
167class dynamic_re_buffer : public re_buffer_reader, public re_buffer_writer
168{
169public:
170 /// \brief Constructs a resource element buffer for a given maximum number of slices and resource elements.
171 /// \param[in] max_nof_slices_ Maximum number of slices.
172 /// \param[in] max_nof_re_ Maximum number of Resource Elements.
173 dynamic_re_buffer(unsigned max_nof_slices_, unsigned max_nof_re_) :
174 max_nof_slices(max_nof_slices_), max_nof_re(max_nof_re_), data({max_nof_re, max_nof_slices})
175 {
176 }
177
178 /// \brief Resizes the buffer to a desired number of RE and slices.
179 /// \param[in] nof_slices Number of slices.
180 /// \param[in] nof_re Number of resource elements.
181 /// \remark An assertion is triggered if the number of slices exceeds \ref max_nof_slices.
182 /// \remark An assertion is triggered if the number of resource elements exceeds \ref max_nof_re.
183 void resize(unsigned nof_slices, unsigned nof_re)
184 {
185 srsran_assert(nof_slices <= max_nof_slices,
186 "The number of slices (i.e., {}) exceeds the maximum (i.e., {}).",
187 nof_slices,
188 max_nof_slices);
189 srsran_assert(nof_re <= max_nof_re,
190 "The number of resource elements (i.e., {}) exceeds the maximum (i.e., {}).",
191 nof_re,
192 max_nof_re);
193 data.resize({nof_re, nof_slices});
194 }
195
196 // See interface for documentation.
197 unsigned get_nof_slices() const override { return data.get_dimension_size(dims::slice); }
198
199 // See interface for documentation.
200 unsigned get_nof_re() const override { return data.get_dimension_size(dims::re); }
201
202 // See interface for documentation.
203 span<cf_t> get_slice(unsigned i_slice) override { return data.get_view<1>({i_slice}); }
204
205 // See interface for documentation.
206 span<const cf_t> get_slice(unsigned i_slice) const override { return data.get_view<1>({i_slice}); }
207
208private:
209 /// Internal tensor dimensions.
210 enum class dims : unsigned { re = 0, slice = 1, all = 2 };
211
212 /// Maximum number of slices.
213 unsigned max_nof_slices;
214 /// Maximum number of resource elements.
215 unsigned max_nof_re;
216
217 /// Internal data storage.
218 dynamic_tensor<static_cast<unsigned>(dims::all), cf_t, dims> data;
219};

221/// Implements a static resource element buffer.
222template <unsigned MaxNofSlices, unsigned MaxNofRe>
223class static_re_buffer : public re_buffer_reader, public re_buffer_writer
224{
225public:
226 /// Default constructor - creates an empty resource element buffer.
227 static_re_buffer() = default;
228
229 /// \brief Creates a resource element buffer for a number of slices and resource elements.
230 /// \param[in] nof_slices Number of slices.
231 /// \param[in] nof_re Number of resource elements.
232 /// \remark An assertion is triggered if the number of slices exceeds \c MaxNofSlices.
233 /// \remark An assertion is triggered if the number of resource elements exceeds \c MaxNofRe.
234 static_re_buffer(unsigned nof_slices, unsigned nof_re) : data({nof_re, nof_slices}) {}
235
236 /// \brief Resizes the number of coefficients to a desired number of slices and ports.
237 /// \param[in] nof_slices Number of slices.
238 /// \param[in] nof_re Number of resource elements.
239 /// \remark An assertion is triggered if the number of slices exceeds \c MaxNofSlices.
240 /// \remark An assertion is triggered if the number of resource elements exceeds \c MaxNofRe.
241 void resize(unsigned nof_slices, unsigned nof_re)
242 {
243 srsran_assert(nof_slices <= MaxNofSlices,
244 "The number of slices (i.e., {}) exceeds the maximum (i.e., {}).",
245 nof_slices,
246 MaxNofSlices);
247 srsran_assert(nof_re <= MaxNofRe,
248 "The number of resource elements (i.e., {}) exceeds the maximum (i.e., {}).",
249 nof_re,
250 MaxNofRe);
251 data.resize({nof_re, nof_slices});
252 }
253
254 // See interface for documentation.
255 unsigned get_nof_slices() const override { return data.get_dimension_size(dims::slice); }
256
257 // See interface for documentation.
258 unsigned get_nof_re() const override { return data.get_dimension_size(dims::re); }
259
260 // See interface for documentation.
261 span<cf_t> get_slice(unsigned i_slice) override { return data.get_view({i_slice}); }
262
263 // See interface for documentation.
264 span<const cf_t> get_slice(unsigned i_slice) const override { return data.get_view({i_slice}); }
265
266private:
267 /// Internal tensor dimensions.
268 enum class dims : unsigned { re = 0, slice = 1, all = 2 };
269
270 /// Internal data storage.
271 static_tensor<static_cast<unsigned>(dims::all), cf_t, MaxNofSlices * MaxNofRe, dims> data;
272};
