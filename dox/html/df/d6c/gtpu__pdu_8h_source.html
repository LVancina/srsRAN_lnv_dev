33/****************************************************************************
34 * GTPU Header
35 * Ref: 3GPP TS 29.281 v10.1.0 Section 5
36 *
37 * | 8 | 7 | 6 | 5 | 4 | 3 | 2 | 1 |
38 *
39 * 1 | Version |PT | * | E | S |PN |
40 * 2 | Message Type |
41 * 3 | Length (1st Octet) |
42 * 4 | Length (2nd Octet) |
43 * 5 | TEID (1st Octet) |
44 * 6 | TEID (2nd Octet) |
45 * 7 | TEID (3rd Octet) |
46 * 8 | TEID (4th Octet) |
47 * 9 | Seq Number (1st Octet) |
48 * 10 | Seq Number (2st Octet) |
49 * 11 | N-PDU |
50 * 12 | Next Extension Header Type |
51 ***************************************************************************/

73// GTP-U extension header types. See TS 29.281 v16.2.0, figure 5.2.1-3
74enum class gtpu_extension_header_type : uint8_t {
75 no_more_extension_headers = 0b00000000,
76 reserved_0 = 0b00000001,
77 reserved_1 = 0b00000010,
78 long_pdcp_pdu_number_0 = 0b00000011,
79 service_class_indicator = 0b00100000,
80 udp_port = 0b01000000,
81 ran_container = 0b10000001,
82 long_pdcp_pdu_number_1 = 0b10000010,
83 xw_ran_container = 0b10000011,
84 nr_ran_container = 0b10000100,
85 pdu_session_container = 0b10000101,
86 pdcp_pdu_number = 0b11000000,
87 reserved_2 = 0b11000001,
88 reserved_3 = 0b11000010
89};
90inline const char* to_string(gtpu_extension_header_type type)
91{
92 switch (type) {
93 case gtpu_extension_header_type::no_more_extension_headers:
94 return "no more extension headers";
95 case gtpu_extension_header_type::service_class_indicator:
96 return "service class indicator";
97 case gtpu_extension_header_type::udp_port:
98 return "UDP port";
99 case gtpu_extension_header_type::ran_container:
100 return "RAN container";
101 case gtpu_extension_header_type::long_pdcp_pdu_number_0:
102 case gtpu_extension_header_type::long_pdcp_pdu_number_1:
103 return "long PDCP PDU number";
104 case gtpu_extension_header_type::xw_ran_container:
105 return "XW RAN container";
106 case gtpu_extension_header_type::nr_ran_container:
107 return "NR RAN container";
108 case gtpu_extension_header_type::pdu_session_container:
109 return "PDU session container";
110 case gtpu_extension_header_type::pdcp_pdu_number:
111 return "PDCP PDU number";
112 case gtpu_extension_header_type::reserved_0:
113 case gtpu_extension_header_type::reserved_1:
114 case gtpu_extension_header_type::reserved_2:
115 case gtpu_extension_header_type::reserved_3:
116 return "reserved";
117 default:
118 return "invalid";
119 }
120};

122// 00 Comprehension of this extension header is not required. An Intermediate Node shall forward it to any Receiver
123// Endpoint
124// 01 Comprehension of this extension header is not required. An Intermediate Node shall discard the
125// Extension Header Content and not forward it to any Receiver Endpoint. Other extension headers shall be treated
126// independently of this extension header.
127// 10 Comprehension of this extension header is required by the Endpoint
128// Receiver but not by an Intermediate Node. An Intermediate Node shall forward the whole field to the Endpoint
129// Receiver.
130// 11 Comprehension of this header type is required by recipient (either Endpoint Receiver or Intermediate Node)
131enum class gtpu_comprehension : uint8_t {
132 not_required_intermediate_node_forward = 0b00000000,
133 not_required_intermediate_node_discard = 0b00000001,
134 required_at_endpoint_not_intermediate_node = 0b00000010,
135 required_at_endpoint_and_intermediate_node = 0b00000011
136};

178/// GTP-U information element for "Private Extension". See TS 29.281 Sec. 8.6

187/// GTP-U header, including extensions
188struct gtpu_header {
189 struct gtpu_flags {
190 uint8_t version = 1;
191 uint8_t protocol_type = 0;
192 bool ext_hdr = false;
193 bool seq_number = false;
194 bool n_pdu = false;
195 } flags;
196 /// This field indicates the type of GTP-U message.
197 uint8_t message_type = 0;
198 /// This field indicates the length in octets of the payload, i.e. the rest of the packet following the mandatory
199 /// part of the GTP header (that is the first 8 octets). The Sequence Number, the N-PDU Number or any Extension
200 /// headers shall be considered to be part of the payload, i.e. included in the length count.
201 uint16_t length = 0;
202 /// Tunnel Endpoint Identifier.
203 gtpu_teid_t teid = {};
204 /// Sequence Number.
205 uint16_t seq_number = 0;
206 /// N-PDU Number.
207 uint8_t n_pdu = 0;
208 /// Next Extension Header Type: This field defines the type of Extension Header that follows this field in the
209 /// GTP-PDU.
210 gtpu_extension_header_type next_ext_hdr_type = gtpu_extension_header_type::no_more_extension_headers;
211 /// Collection of included GTP-U header extensions
212 static_vector<gtpu_extension_header, GTPU_MAX_NUM_HEADER_EXTENSIONS> ext_list;
213};

215/// Intermediate representation of a received GTP-U PDU to access the unpacked header and to the raw content of the
216/// extension headers. This container holds a buffer with the original PDU to which the views of the extension header
217/// content are mapped. After processing the extension headers, the T-PDU can be extracted via \c gtpu_extract_t_pdu
218/// which advances the PDU buffer to the start of the T-PDU according to \c hdr_len.
219struct gtpu_dissected_pdu {
220 /// Storage of the original GTP-U PDU to which the views of the extension header content is mapped
221 byte_buffer buf;
222 /// Access to the dissected content of the GTP-U PDU
223 gtpu_header hdr;
224 /// Total header length (including all extension headers); marks the start of the T-PDU.
225 size_t hdr_len = 0;
226};
