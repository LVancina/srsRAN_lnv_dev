29/// Provides access to status functions of the RLC RX AM entity.
30/// The RLC TX AM entity uses this class to
31/// - get status PDUs for transmission to lower layers
32/// - get length of status PDUs for buffer state calculation
33/// - check whether a status report has been polled by the peer
34class rlc_rx_am_status_provider
35{
36public:
37 virtual ~rlc_rx_am_status_provider() = default;
38
39 /// \brief Gets a reference to the latest status report.
40 ///
41 /// Important Note: The returned status report is only guaranteed to be valid if \c status_report_required returns
42 /// \c true. For (unit-)testing not checking this condition, a valid status report can be expected
43 /// (a) once after receiving one (or more) PDUs, or (b) once after t-reassembly has expired.
44 ///
45 /// The returned reference stays valid until the next call of this function or the destruction of the RLC RX entity.
46 /// \return The latest status report
47 virtual rlc_am_status_pdu& get_status_pdu() = 0;
48 virtual uint32_t get_status_pdu_length() = 0;
49 virtual bool status_report_required() = 0;
50};
51
52/// This interface represents the status PDU entry point of a RLC TX AM entity.
53/// The RLC RX AM entity uses this class to forward received status PDUs to the RLC TX AM entity
54class rlc_tx_am_status_handler
55{
56public:
57 virtual ~rlc_tx_am_status_handler() = default;
58
59 /// \brief Status PDU handler for TX entity
60 /// \param status The status PDU to be processed by the RLC TX AM entity
61 virtual void on_status_pdu(rlc_am_status_pdu status) = 0;
62};
63
64/// This interface represents the RLC TX AM entity that the RLC RX AM must notify to inform the RLC TX AM entity that a
65/// status report is triggered (either by timeout or by request of the other peer) or its size has changed (e.g. further
66/// PDUs have been received)
67class rlc_tx_am_status_notifier
68{
69public:
70 virtual ~rlc_tx_am_status_notifier() = default;
71
72 /// \brief Informs the TX entity that a status report is triggered or its size has changed, so that the buffer status
73 /// report needs to be updated.
74 virtual void on_status_report_changed() = 0;
75};
