71{
72 // NOTE:
73 // - [Implementation defined]
74 // The UE monitors the SearchSpaceSet 0 for SIB1 in 2 consecutive slots, starting from n0.
75 // In this function, it is assumed that the GNB only allocates the PDCCH/DCI_1_0 for SIB1 in the first slot, i.e.,
76 // in n0 if channel BW > 10 Mhz. Otherwise, in (n0 + 1) slot to make space for SSB.
77 // - [Implementation defined] We assume the SIB1 is (re)transmitted every 20ms if the SSB periodicity <= 20ms.
78 // Else, we set the (re)transmission periodicity as the SSB's.
79
80 // The sib1_period_slots is expressed in unit of slots.
81 // NOTE: As sib1_period_slots is expressed in milliseconds or subframes, we need to this these into slots.
82 const unsigned sib1_period_slots = sib1_period * static_cast<unsigned>(sl_point.nof_slots_per_subframe());
83
84 // For each beam, check if the SIB1 needs to be allocated in this slot.
85 for (unsigned ssb_idx = 0; ssb_idx < MAX_NUM_BEAMS; ssb_idx++) {
86 // Do not schedule the SIB1 for the SSB indices that are not used.
87 if (not is_nth_ssb_beam_active(cell_cfg.ssb_cfg.ssb_bitmap, ssb_idx)) {
88 continue;
89 }
90
91 if (sl_point.to_uint() % sib1_period_slots == sib1_type0_pdcch_css_slots[ssb_idx].to_uint()) {
92 // Ensure slot for SIB1 has DL enabled.
93 if (not cell_cfg.is_dl_enabled(sl_point)) {
94 logger.error("Could not allocate SIB1 for beam idx {} as slot is not DL enabled.", ssb_idx);
95 return;
96 }
