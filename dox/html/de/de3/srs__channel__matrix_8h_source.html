37/// \brief SRS-based estimated channel matrix.
38///
39/// Wideband channel matrix estimated from the Sounding Reference Signals (SRS). The complex channel coefficients are
40/// arranged by i) receive ports and ii) transmit ports.
41class srs_channel_matrix
42{
43public:
44 /// Channel matrix dimensions.
45 enum class dims : uint8_t { rx_port = 0, tx_port, all };
46
47 /// Default constructor - constructs a channel matrix with no coefficients.
48 srs_channel_matrix() = default;
49
50 /// \brief Constructs a channel matrix with the desired number of transmit and receive ports.
51 /// \param[in] nof_rx_ports Number of receive ports.
52 /// \param[in] nof_tx_ports Number of transmit ports.
53 /// \remark An assertion is triggered if the number of receive ports exceeds \ref srs_constants::max_nof_rx_ports.
54 /// \remark An assertion is triggered if the number of transmit ports exceeds \ref srs_constants::max_nof_tx_ports.
55 srs_channel_matrix(unsigned nof_rx_ports, unsigned nof_tx_ports) : data({nof_rx_ports, nof_tx_ports})
56 {
57 srsran_assert(nof_rx_ports <= srs_constants::max_nof_rx_ports,
58 "The number of receive ports (i.e., {}) exceeds the maximum (i.e., {}).",
59 nof_rx_ports,
60 srs_constants::max_nof_rx_ports);
61 srsran_assert(nof_tx_ports <= srs_constants::max_nof_tx_ports,
62 "The number of transmit ports (i.e., {}) exceeds the maximum (i.e., {}).",
63 nof_rx_ports,
64 srs_constants::max_nof_rx_ports);
65 }
66
67 /// \brief Constructs a channel matrix with the desired number of receive and transmit ports.
68 ///
69 /// Creates a channel matrix with the specified dimensions, and sets its contents to the provided coefficients
70 /// values.
71 ///
72 /// \param[in] coefficients Channel coefficient list, arranged by i) receive port and ii) transmit port.
73 /// \param[in] nof_rx_ports Number of receive ports.
74 /// \param[in] nof_tx_ports Number of transmit ports.
75 /// \remark An assertion is triggered if the number of receive ports exceeds \ref srs_constants::max_nof_rx_ports.
76 /// \remark An assertion is triggered if the number of transmit ports exceeds \ref srs_constants::max_nof_tx_ports.
77 srs_channel_matrix(const std::initializer_list<cf_t>& coefficients, unsigned nof_rx_ports, unsigned nof_tx_ports) :
78 srs_channel_matrix(span<const cf_t>(coefficients.begin(), coefficients.end()), nof_rx_ports, nof_tx_ports)
79 {
80 }
81
82 /// \brief Constructs a channel matrix with the desired number of receive and transmit ports.
83 ///
84 /// Creates a channel matrix with the specified dimensions, and sets its contents to the provided coefficients
85 /// values.
86 ///
87 /// \param[in] coefficients Channel coefficient list, arranged by i) receive port and ii) transmit port.
88 /// \param[in] nof_rx_ports Number of receive ports.
89 /// \param[in] nof_tx_ports Number of transmit ports.
90 /// \remark An assertion is triggered if the number of receive ports exceeds \ref srs_constants::max_nof_rx_ports.
91 /// \remark An assertion is triggered if the number of transmit ports exceeds \ref srs_constants::max_nof_tx_ports.
92 srs_channel_matrix(span<const cf_t> coefficients, unsigned nof_rx_ports, unsigned nof_tx_ports) :
93 data({nof_rx_ports, nof_tx_ports})
94 {
95 srsran_assert(coefficients.size() == nof_rx_ports * nof_tx_ports,
96 "The number of coefficients, i.e., {}, does not match the specified matrix dimensions, i.e., {} "
97 "transmit ports, {} receive ports.",
98 coefficients.size(),
99 nof_tx_ports,
100 nof_rx_ports);
101
102 srsran_assert(nof_rx_ports <= srs_constants::max_nof_rx_ports,
103 "The number of receive ports (i.e., {}) exceeds the maximum (i.e., {}).",
104 nof_rx_ports,
105 srs_constants::max_nof_rx_ports);
106 srsran_assert(nof_tx_ports <= srs_constants::max_nof_tx_ports,
107 "The number of transmit ports (i.e., {}) exceeds the maximum (i.e., {}).",
108 nof_tx_ports,
109 srs_constants::max_nof_tx_ports);
110
111 // Copy the weights into the tensor.
112 srsvec::copy(data.get_view<static_cast<unsigned>(dims::all)>({}), coefficients);
113 }
114
115 /// Copy constructor.
116 srs_channel_matrix(const srs_channel_matrix& other) : data({other.get_nof_rx_ports(), other.get_nof_tx_ports()})
117 {
118 // Copy the weights into the tensor.
119 srsvec::copy(data.get_view<static_cast<unsigned>(dims::all)>({}),
120 other.data.get_view<static_cast<unsigned>(dims::all)>({}));
121 }
122
123 /// \brief Overload assignment operator.
124 /// \param[in] other Channel matrix to copy.
125 srs_channel_matrix& operator=(const srs_channel_matrix& other)
126 {
127 if (this == &other) {
128 return *this;
129 }
130
131 // Resize the tensor.
132 resize(other.get_nof_rx_ports(), other.get_nof_tx_ports());
133 // Copy the weights into the tensor.
134 srsvec::copy(data.get_view<static_cast<unsigned>(dims::all)>({}),
135 other.data.get_view<static_cast<unsigned>(dims::all)>({}));
136 return *this;
137 }
138
139 /// \brief Overload equality comparison operator.
140 /// \param[in] other Channel matrix to compare against.
141 /// \return \c true if both channel matrices are exactly the same, \c false otherwise.
142 bool operator==(const srs_channel_matrix& other) const
143 {
144 unsigned nof_rx_ports = get_nof_rx_ports();
145 unsigned nof_tx_ports = get_nof_tx_ports();
146
147 if (nof_rx_ports != other.get_nof_rx_ports()) {
148 return false;
149 }
150 if (nof_tx_ports != other.get_nof_tx_ports()) {
151 return false;
152 }
153
154 for (unsigned i_layer = 0; i_layer != nof_rx_ports; ++i_layer) {
155 for (unsigned i_port = 0; i_port != nof_tx_ports; ++i_port) {
156 if (get_coefficient(i_layer, i_port) != other.get_coefficient(i_layer, i_port)) {
157 return false;
158 }
159 }
160 }
161
162 return true;
163 }
164
165 /// Overload inequality comparison operator.
166 bool operator!=(const srs_channel_matrix& other) const { return !(*this == other); }
167
168 /// Gets the current number of receive ports.
169 unsigned get_nof_rx_ports() const { return data.get_dimension_size(dims::rx_port); }
170
171 /// Gets the current number of transmit ports.
172 unsigned get_nof_tx_ports() const { return data.get_dimension_size(dims::tx_port); }
173
174 /// \brief Gets a channel coefficient from the matrix.
175 ///
176 /// \param[in] i_rx_port Receive port index.
177 /// \param[in] i_tx_port Transmit port index.
178 /// \return The channel coefficient for the given transmit and receive ports.
179 cf_t get_coefficient(unsigned i_rx_port, unsigned i_tx_port) const
180 {
181 srsran_assert(i_rx_port < get_nof_rx_ports(),
182 "The receive port index (i.e., {}) exceeds the maximum (i.e., {}).",
183 i_rx_port,
184 get_nof_rx_ports() - 1);
185 srsran_assert(i_tx_port < get_nof_tx_ports(),
186 "The transmit port index (i.e., {}) exceeds the maximum (i.e., {}).",
187 i_tx_port,
188 get_nof_tx_ports() - 1);
189 return data[{i_rx_port, i_tx_port}];
190 }
191
192 /// \brief Sets a channel coefficient in the matrix to a specified value.
193 ///
194 /// \param[in] coefficient Channel coefficient to set.
195 /// \param[in] i_rx_port Receive port index.
196 /// \param[in] i_tx_port Transmit port index.
197 void set_coefficient(cf_t coefficient, unsigned i_rx_port, unsigned i_tx_port)
198 {
199 srsran_assert(i_rx_port < get_nof_rx_ports(),
200 "The receive port index (i.e., {}) exceeds the maximum (i.e., {}).",
201 i_rx_port,
202 get_nof_rx_ports() - 1);
203 srsran_assert(i_tx_port < get_nof_tx_ports(),
204 "The transmit port index (i.e., {}) exceeds the maximum (i.e., {}).",
205 i_tx_port,
206 get_nof_tx_ports() - 1);
207 data[{i_rx_port, i_tx_port}] = coefficient;
208 }
209
210 /// Scales all the coefficients by a scaling factor.
211 srs_channel_matrix& operator*=(float scale)
212 {
213 srsvec::sc_prod(data.get_data(), scale, data.get_data());
214 return *this;
215 }
216
217private:
218 /// \brief Resizes the number of coefficients to a desired number of receive and transmit ports.
219 /// \param[in] nof_rx_ports Number of receive ports.
220 /// \param[in] nof_tx_ports Number of transmit ports.
221 /// \remark An assertion is triggered if the number of receive ports exceeds \ref srs_constants::max_nof_rx_ports.
222 /// \remark An assertion is triggered if the number of transmit ports exceeds \ref srs_constants::max_nof_tx_ports.
223 void resize(unsigned nof_rx_ports, unsigned nof_tx_ports)
224 {
225 srsran_assert(nof_rx_ports <= srs_constants::max_nof_rx_ports,
226 "The number of receive ports (i.e., {}) exceeds the maximum (i.e., {}).",
227 nof_rx_ports,
228 srs_constants::max_nof_rx_ports);
229 srsran_assert(nof_tx_ports <= srs_constants::max_nof_tx_ports,
230 "The number of transmit ports (i.e., {}) exceeds the maximum (i.e., {}).",
231 nof_tx_ports,
232 srs_constants::max_nof_tx_ports);
233
234 data.resize({nof_rx_ports, nof_tx_ports});
235 }
236
237 /// Internal data storage.
238 static_tensor<static_cast<unsigned>(dims::all),
239 cf_t,
240 srs_constants::max_nof_rx_ports * srs_constants::max_nof_tx_ports,
241 dims>
242 data;
243};
244
245} // namespace srsran
