34/// PUSCH Transmit Precoding Matrix Indication (TPMI) information.
35class pusch_tpmi_select_info
36{
37public:
38 struct tpmi_info {
39 /// Most suitable Transmit Precoding Matrix Indicator.
40 unsigned tpmi;
41 /// Average Signal-to-Interference-plus-Noise Ratio (SINR) in decibels.
42 float sinr_dB;
43 };
44
45 /// Gets the maximum number of layers.
46 unsigned get_max_nof_layers() const { return info.size(); }
47
48 /// Determines if the information is valid.
49 bool is_valid() const { return !info.empty(); }
50
51 /// \brief Gets the TPMI information for a number of layers.
52 ///
53 /// \remark An assertion is triggered if the number of layers is out of range.
54 const tpmi_info& get_tpmi_select(unsigned nof_layers) const
55 {
56 interval<unsigned, true> nof_layers_range(1, get_max_nof_layers());
57 srsran_assert(nof_layers_range.contains(nof_layers),
58 "The number of layers (i.e., {}) is out-of-range {}",
59 nof_layers,
60 nof_layers_range);
61
62 return info[nof_layers - 1];
63 }
64
65 /// Constructs a PUSCH TPMI information from an initializer list.
66 pusch_tpmi_select_info(const std::initializer_list<tpmi_info>& info_) : info(info_.begin(), info_.end()) {}
67
68private:
69 /// TPMI information for each number of layers.
70 static_vector<tpmi_info, pusch_constants::MAX_NOF_LAYERS> info;
71};
