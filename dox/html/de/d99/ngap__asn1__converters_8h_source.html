38// Helper to create PDU from NGAP message
39template <class T>
40byte_buffer pack_into_pdu(const T& msg, const char* context_name = nullptr)
41{
42 context_name = context_name == nullptr ? __FUNCTION__ : context_name;
43 byte_buffer pdu{};
44 asn1::bit_ref bref{pdu};
45 if (msg.pack(bref) == asn1::SRSASN_ERROR_ENCODE_FAIL) {
46 srslog::fetch_basic_logger("NGAP").error("Failed to pack message in {} - discarding it", context_name);
47 }
48 return pdu;
49}

51/// \brief Convert CU-CP security result to NGAP security result.
52/// \param security_result The CU-CP security result.
53/// \return The NGAP security result.
54inline asn1::ngap::security_result_s cu_cp_security_result_to_ngap_security_result(security_result_t security_result)
55{
56 asn1::ngap::security_result_s ngap_security_result;
57
58 if (security_result.confidentiality_protection_result == confidentiality_protection_result_t::performed) {
59 ngap_security_result.confidentiality_protection_result =
60 asn1::ngap::confidentiality_protection_result_opts::options::performed;
61 } else {
62 ngap_security_result.confidentiality_protection_result =
63 asn1::ngap::confidentiality_protection_result_opts::options::not_performed;
64 }
65
66 if (security_result.integrity_protection_result == integrity_protection_result_t::performed) {
67 ngap_security_result.integrity_protection_result.value =
68 asn1::ngap::integrity_protection_result_opts::options::performed;
69 } else {
70 ngap_security_result.integrity_protection_result =
71 asn1::ngap::integrity_protection_result_opts::options::not_performed;
72 }
73
74 return ngap_security_result;
75}

77/// \brief Convert CU-CP Associated QoS Flow to NGAP Associated QoS Flow Item.
78/// \param cu_cp_qos_flow The CU-CP Associated QoS Flow.
79/// \return The NGAP Associated QoS Flow Item.
80inline asn1::ngap::associated_qos_flow_item_s
81cu_cp_assoc_qos_flow_to_ngap_assoc_qos_flow_item(cu_cp_associated_qos_flow cu_cp_qos_flow)
82{
83 asn1::ngap::associated_qos_flow_item_s asn1_assoc_qos_item;
84
85 asn1_assoc_qos_item.qos_flow_id = qos_flow_id_to_uint(cu_cp_qos_flow.qos_flow_id);
86
87 if (cu_cp_qos_flow.qos_flow_map_ind.has_value()) {
88 asn1_assoc_qos_item.qos_flow_map_ind_present = true;
89
90 if (cu_cp_qos_flow.qos_flow_map_ind.value() == cu_cp_qos_flow_map_ind::ul) {
91 asn1_assoc_qos_item.qos_flow_map_ind.value =
92 asn1::ngap::associated_qos_flow_item_s::qos_flow_map_ind_opts::options::ul;
93 } else {
94 asn1_assoc_qos_item.qos_flow_map_ind.value =
95 asn1::ngap::associated_qos_flow_item_s::qos_flow_map_ind_opts::options::dl;
96 }
97 }
98
99 return asn1_assoc_qos_item;
100}

122/// \brief Convert \c cause_t type to NGAP cause.
123/// \param cause The cause_t type.
124/// \return The NGAP cause.
125inline asn1::ngap::cause_c cause_to_asn1(cause_t cause)
126{
127 asn1::ngap::cause_c ngap_cause;
128
129 if (variant_holds_alternative<cause_radio_network_t>(cause)) {
130 ngap_cause.set_radio_network() =
131 static_cast<asn1::ngap::cause_radio_network_opts::options>(variant_get<cause_radio_network_t>(cause));
132 } else if (variant_holds_alternative<cause_transport_t>(cause)) {
133 ngap_cause.set_transport() =
134 static_cast<asn1::ngap::cause_transport_opts::options>(variant_get<cause_transport_t>(cause));
135 } else if (variant_holds_alternative<cause_nas_t>(cause)) {
136 ngap_cause.set_nas() = static_cast<asn1::ngap::cause_nas_opts::options>(variant_get<cause_nas_t>(cause));
137 } else if (variant_holds_alternative<cause_protocol_t>(cause)) {
138 ngap_cause.set_protocol() =
139 static_cast<asn1::ngap::cause_protocol_opts::options>(variant_get<cause_protocol_t>(cause));
140 } else if (variant_holds_alternative<cause_misc_t>(cause)) {
141 ngap_cause.set_misc() = static_cast<asn1::ngap::cause_misc_opts::options>(variant_get<cause_misc_t>(cause));
142 } else {
143 report_fatal_error("Cannot convert cause to NGAP type");
144 }
145
146 return ngap_cause;
147}

149/// \brief Convert NGAP ASN1 cause to \c cause_t type.
150/// \param ngap_cause The ASN1 NGAP cause.
151/// \return The cause_t type.
152inline cause_t asn1_to_cause(asn1::ngap::cause_c ngap_cause)
153{
154 cause_t cause;
155
156 switch (ngap_cause.type()) {
157 case asn1::ngap::cause_c::types_opts::radio_network:
158 cause = static_cast<cause_radio_network_t>(ngap_cause.radio_network().value);
159 break;
160 case asn1::ngap::cause_c::types_opts::transport:
161 cause = static_cast<cause_transport_t>(ngap_cause.transport().value);
162 break;
163 case asn1::ngap::cause_c::types_opts::nas:
164 cause = static_cast<cause_nas_t>(ngap_cause.nas().value);
165 break;
166 case asn1::ngap::cause_c::types_opts::protocol:
167 cause = static_cast<cause_protocol_t>(ngap_cause.protocol().value);
168 break;
169 case asn1::ngap::cause_c::types_opts::misc:
170 cause = static_cast<cause_misc_t>(ngap_cause.misc().value);
171 break;
172 default:
173 report_fatal_error("Cannot convert NGAP ASN.1 cause {} to common type", ngap_cause.type());
174 }
175
176 return cause;
177}

200/// \brief Convert CU-CP NRCGI to NR Cell Identity.
201/// \param ngap_cgi The NGAP NRCGI.
202/// \return The NR Cell Identity.
203inline asn1::ngap::user_location_info_nr_s
204cu_cp_user_location_info_to_asn1(const cu_cp_user_location_info_nr& cu_cp_user_location_info)
205{
206 asn1::ngap::user_location_info_nr_s asn1_user_location_info;
207
208 // add nr cgi
209 asn1_user_location_info.nr_cgi.nr_cell_id.from_number(cu_cp_user_location_info.nr_cgi.nci);
210 asn1_user_location_info.nr_cgi.plmn_id.from_string(cu_cp_user_location_info.nr_cgi.plmn_hex);
211 // add tai
212 asn1_user_location_info.tai.plmn_id.from_string(cu_cp_user_location_info.tai.plmn_id);
213 asn1_user_location_info.tai.tac.from_number(cu_cp_user_location_info.tai.tac);
214 // add timestamp
215 if (cu_cp_user_location_info.time_stamp.has_value()) {
216 asn1_user_location_info.time_stamp_present = true;
217 asn1_user_location_info.time_stamp.from_number(cu_cp_user_location_info.time_stamp.value());
218 }
219
220 return asn1_user_location_info;
221}

223/// \brief Convert ASN.1 cause to a human-readable string.
224/// \param cause The ASN.1 cause.
225/// \return The humand-readable string.
226inline std::string asn1_cause_to_string(const asn1::ngap::cause_c& cause)
227{
228 std::string cause_str = "";
229
230 switch (cause.type()) {
231 case asn1::ngap::cause_c::types_opts::radio_network:
232 cause_str = cause.radio_network().to_string();
233 break;
234 case asn1::ngap::cause_c::types_opts::transport:
235 cause_str = cause.transport().to_string();
236 break;
237 case asn1::ngap::cause_c::types_opts::protocol:
238 cause_str = cause.protocol().to_string();
239 break;
240 case asn1::ngap::cause_c::types_opts::nas:
241 cause_str = cause.nas().to_string();
242 break;
243 case asn1::ngap::cause_c::types_opts::misc:
244 cause_str = cause.misc().to_string();
245 break;
246 default:
247 cause_str = "unknown";
248 break;
249 }
250
251 return cause_str;
252}

254/// \brief Convert common type Initial Context Setup Response message to NGAP Initial Context Setup Response
255/// message.
256/// \param[out] resp The ASN1 NGAP Initial Context Setup Response message.
257/// \param[in] cu_cp_resp The CU-CP Initial Context Setup Response message.
258template <typename template_asn1_item>
259inline void pdu_session_res_setup_response_item_to_asn1(template_asn1_item& asn1_resp,
260 const cu_cp_pdu_session_res_setup_response_item resp)
261{
262 asn1_resp.pdu_session_id = pdu_session_id_to_uint(resp.pdu_session_id);
263
264 asn1::ngap::pdu_session_res_setup_resp_transfer_s response_transfer;
265
266 // Add dLQosFlowPerTNLInformation
267 response_transfer.dl_qos_flow_per_tnl_info = cu_cp_qos_flow_per_tnl_info_to_ngap_qos_flow_per_tnl_info(
268 resp.pdu_session_resource_setup_response_transfer.dlqos_flow_per_tnl_info);
269
270 // Add AdditionalDLQosFlowPerTNLInformation
271 for (const auto& cu_cp_qos_flow_info :
272 resp.pdu_session_resource_setup_response_transfer.add_dl_qos_flow_per_tnl_info) {
273 asn1::ngap::qos_flow_per_tnl_info_item_s ngap_qos_flow_info;
274 ngap_qos_flow_info.qos_flow_per_tnl_info =
275 cu_cp_qos_flow_per_tnl_info_to_ngap_qos_flow_per_tnl_info(cu_cp_qos_flow_info);
276 response_transfer.add_dl_qos_flow_per_tnl_info.push_back(ngap_qos_flow_info);
277 }
278
279 // Add QosFlowFailedToSetupList
280 for (const auto& failed_item : resp.pdu_session_resource_setup_response_transfer.qos_flow_failed_to_setup_list) {
281 asn1::ngap::qos_flow_with_cause_item_s asn1_failed_item =
282 cu_cp_qos_flow_failed_to_setup_item_to_ngap_qos_flow_with_cause_item(failed_item);
283 response_transfer.qos_flow_failed_to_setup_list.push_back(asn1_failed_item);
284 }
285
286 // Add SecurityResult
287 if (resp.pdu_session_resource_setup_response_transfer.security_result.has_value()) {
288 response_transfer.security_result_present = true;
289 response_transfer.security_result = cu_cp_security_result_to_ngap_security_result(
290 resp.pdu_session_resource_setup_response_transfer.security_result.value());
291 }
292
293 // Pack pdu_session_res_setup_resp_transfer_s
294 byte_buffer pdu = pack_into_pdu(response_transfer);
295
296 asn1_resp.pdu_session_res_setup_resp_transfer.resize(pdu.length());
297 std::copy(pdu.begin(), pdu.end(), asn1_resp.pdu_session_res_setup_resp_transfer.begin());
298}

300/// \brief Convert common type modify response item to ASN1 type message.
301/// \param[out] asn1_resp The ASN1 NGAP struct.
302/// \param[in] resp The common type struct.
303template <typename template_asn1_item>
304inline void pdu_session_res_modify_response_item_to_asn1(template_asn1_item& asn1_resp,
305 const cu_cp_pdu_session_resource_modify_response_item resp)
306{
307 asn1_resp.pdu_session_id = pdu_session_id_to_uint(resp.pdu_session_id);
308
309 asn1::ngap::pdu_session_res_modify_resp_transfer_s response_transfer;
310
311 // Add AdditionalDLQosFlowPerTNLInformation
312 for (const auto& cu_cp_qos_flow_info : resp.transfer.add_dl_qos_flow_per_tnl_info) {
313 asn1::ngap::qos_flow_per_tnl_info_item_s ngap_qos_flow_info;
314 ngap_qos_flow_info.qos_flow_per_tnl_info =
315 cu_cp_qos_flow_per_tnl_info_to_ngap_qos_flow_per_tnl_info(cu_cp_qos_flow_info);
316 response_transfer.add_dl_qos_flow_per_tnl_info.push_back(ngap_qos_flow_info);
317 }
318
319 if (resp.transfer.qos_flow_add_or_modify_response_list.has_value()) {
320 for (const auto& qos_flow : resp.transfer.qos_flow_add_or_modify_response_list.value()) {
321 asn1::ngap::qos_flow_add_or_modify_resp_item_s asn1_item;
322 asn1_item.qos_flow_id = qos_flow_id_to_uint(qos_flow.qos_flow_id);
323 response_transfer.qos_flow_add_or_modify_resp_list.push_back(asn1_item);
324 }
325 }
326
327 // Pack pdu_session_res_modify_resp_transfer_s
328 byte_buffer pdu = pack_into_pdu(response_transfer);
329
330 asn1_resp.pdu_session_res_modify_resp_transfer.resize(pdu.length());
331 std::copy(pdu.begin(), pdu.end(), asn1_resp.pdu_session_res_modify_resp_transfer.begin());
332}

389/// \brief Converts type \c security_indication to an ASN.1 type.

391/// \param security_indication Security Indication IE contents.
392inline void security_indication_to_asn1(asn1::ngap::security_ind_s& asn1obj, const security_indication_t& security_ind)
393{
394 switch (security_ind.integrity_protection_ind) {
395 case integrity_protection_indication_t::not_needed:
396 case integrity_protection_indication_t::preferred:
397 case integrity_protection_indication_t::required:
398 asn1obj.integrity_protection_ind.value =
399 static_cast<asn1::ngap::integrity_protection_ind_opts::options>(security_ind.integrity_protection_ind);
400 break;
401 default:
402 report_fatal_error("Cannot convert security indication to NGAP type");
403 }
404
405 switch (security_ind.confidentiality_protection_ind) {
406 case confidentiality_protection_indication_t::not_needed:
407 case confidentiality_protection_indication_t::preferred:
408 case confidentiality_protection_indication_t::required:
409 asn1obj.confidentiality_protection_ind.value =
410 static_cast<asn1::ngap::confidentiality_protection_ind_opts::options>(security_ind.integrity_protection_ind);
411 break;
412 default:
413 report_fatal_error("Cannot convert security indication to NGAP type");
414 }

776 return false;

779 return true;
