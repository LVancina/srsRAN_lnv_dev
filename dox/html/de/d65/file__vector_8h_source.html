32/// \c file_vector open modes.
33enum class openmode {
34 /// Opens the file in read-only mode.
35 read_only,
36 /// Opens the file in write-only mode.
37 write_only,
38 /// \brief Opens the file in read-write mode.
39 ///
40 /// Any previous content of the file will be lost.
41 read_write
42};
43
44template <typename T>
45/// \brief Simple binary file input-output interface.
46///
47/// An object of this file allows reading and writing values of type \c T from and to a binary file.
48class file_vector
49{
50private:
51 const char* file_name;
52 openmode op;
53
54public:
55 /// \brief Constructor.
56 ///
57 /// Sets the name and the mode (default is \e read-only) of the binary file. No operations are performed on the file.
58 file_vector(const char* filename, openmode op_ = openmode::read_only) : file_name(filename), op(op_) {}
59
60 /// Returns the file name.
61 const char* get_file_name() const { return file_name; }
62
63 /// \brief Reads the file.
64 ///
65 /// The contents of the file are dumped to a \c std::vector<T>. The file is closed when the operation is completed.
66 /// \warning An exception arises if the file cannot be opened or if the write-only open mode was specified at
67 /// creation.
68 std::vector<T> read() const
69 {
70 report_fatal_error_if_not(op != openmode::write_only, "The file_vector was created with write-only permissions.");
71 std::ifstream binary_file(file_name, std::ios::in | std::ios::binary);
72
73 report_fatal_error_if_not(binary_file.is_open(), "Error opening file '{}'. {}.", file_name, strerror(errno));
74 std::vector<T> read_data;
75
76 T read_value;
77 while (binary_file.read(reinterpret_cast<char*>(&read_value), sizeof(T))) {
78 read_data.push_back(read_value);
79 }
80
81 return read_data;
82 }
83
84 /// \brief Writes data to the file.
85 ///
86 /// The sequence of \c T values viewed by \c write_data are written to the file.
87 /// If the specified file does not exist, it is created. If it exists, previous contents will be erased.
88 /// The file is closed when the operation is completed.
89 ///
90 /// \warning An exception arises if the file cannot be opened or if the read-only open mode was specified at creation.
91 void write(span<const T> write_data) const
92 {
93 report_fatal_error_if_not(op != openmode::read_only, "The file_vector was created with read-only permissions.");
94 std::ofstream binary_file(file_name, std::ios::out | std::ios::binary);
95
96 report_fatal_error_if_not(binary_file.is_open(), "Error opening file '{}'. {}.", file_name, strerror(errno));
97
98 binary_file.write(reinterpret_cast<const char*>(write_data.begin()), write_data.size() * sizeof(T));
99 }
100};
