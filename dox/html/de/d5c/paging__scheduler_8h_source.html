33/// Defines Paging scheduler that is used to allocate resources to send paging information to UE in a given slot.
34class paging_scheduler
35{
36public:
37 // 48 bits (6 Bytes) 5G-S-TMSI + 1 Byte for Paging record header size estimate.
38 static constexpr unsigned RRC_CN_PAGING_ID_RECORD_SIZE = 7U;
39 // 40 bits (5 Bytes) Full-I-RNTI + 1 Byte for Paging record header size estimate.
40 static constexpr unsigned RRC_RAN_PAGING_ID_RECORD_SIZE = 6U;
41
42 /// NG-5G-S-TMSI (48 bits) or I-RNTI-Value (40 bits).
43 using ue_paging_id = uint64_t;
44 using paging_retries_count_type = unsigned;

47 const cell_configuration& cell_cfg_,

51 /// \brief Performs paging (if any) scheduling for the current slot.
52 ///
53 /// \param[out,in] res_grid Resource grid with current allocations and scheduling results.
54 void schedule_paging(cell_resource_allocator& res_grid);

56 /// Handles Paging information reported by upper layers.
57 /// \param[in] paging_info Per UE paging information to be scheduled.
58 void handle_paging_information(const sched_paging_information& paging_info);
59
60private:
61 struct ue_paging_info {
62 sched_paging_information info;
63 paging_retries_count_type retry_count;
64 };

66 /// \brief Checks paging conditions for a UE in SearchSpace > 0 i.e pagingSearchSpace > 0 in its active BWP config.
67 ///
68 /// \param[in] pdcch_slot Slot at which the paging scheduler is called.
69 /// \param[in] i_s Index of the Paging Occasion.
70 /// \return True if paging conditions are satisfied, false otherwise.
71 bool is_paging_slot_in_search_space_id_gt_0(slot_point pdcch_slot, unsigned i_s);
72
73 /// \brief Checks paging conditions for a UE in SearchSpace 0 when pagingSearchSpace is 0 in its active BWP config.
74 ///
75 /// \param[in] sl_point Slot at which the paging scheduler is called.
76 /// \param[in] i_s Index of the Paging Occasion.
77 /// \return True if paging conditions are satisfied, false otherwise.
78 bool is_paging_slot_in_search_space0(slot_point pdcch_slot, unsigned i_s);
79
80 /// \brief Helper function to get sum of paging payload size of each UE scheduled at a particular slot.
81 /// \param[in] ues_paging_info List of UE scheduled at a particular slot for Paging.
82 /// \return Sum of payload size.
83 static unsigned get_accumulated_paging_msg_size(span<const sched_paging_information*> ues_paging_info);
84
85 /// \brief Checks whether there is space for PDSCH to allocate Paging grant.
86 ///
87 /// \param[out,in] res_grid Resource grid with current allocations and scheduling results.
88 /// \param[in] pdsch_time_res Slot at which PDSCH needs to be scheduled.
89 /// \param[in] msg_size Paging message size.
90 /// \return True if there is space to allocate Paging grant, false otherwise.
91 bool
92 is_there_space_available_for_paging(cell_resource_allocator& res_grid, unsigned pdsch_time_res, unsigned msg_size);
93
94 /// \brief Allocates PDSCH and PDCCH for Paging.
95 ///
96 /// \param[out,in] res_grid Resource grid with current allocations and scheduling results.
97 /// \param[in] pdsch_time_res Slot at which PDSCH needs to be scheduled.
98 /// \param[in] ues_paging_info List of UE scheduled at a particular slot for Paging.
99 /// \param[in] ss_id Search Space Id used in scheduling paging message.
100 /// \return True if paging allocation is successful, false otherwise.
101 bool allocate_paging(cell_resource_allocator& res_grid,
102 unsigned pdsch_time_res,
103 span<const sched_paging_information*> ues_paging_info,
104 search_space_id ss_id);
105
106 /// \brief Fills the Paging grant.
107 ///
108 /// \param[out,in] pg_grant Paging grant to be filled.
109 /// \param[out,in] pdcch Allocated PDCCH for Paging.
110 /// \param[in] crbs_grant Paging grant in CRBs.
111 /// \param[in] time_resource Slot at which PDSCH needs to be scheduled.
112 /// \param[in] ues_paging_info List of UE scheduled at a particular slot for Paging.
113 /// \param[in] dmrs_info DMRS information related to the scheduled grant.
114 /// \param[in] tbs TBS information of the Paging grant.
115 void fill_paging_grant(dl_paging_allocation& pg_grant,
116 pdcch_dl_information& pdcch,
117 crb_interval crbs_grant,
118 unsigned time_resource,
119 span<const sched_paging_information*> ues_paging_info,
120 const dmrs_information& dmrs_info,
121 unsigned tbs);

123 /// \brief Helper function to precompute Type2-PDCCH Monitoring slots when pagingSearchSpace > 0.
124 ///
125 /// \param[in] scs_common SCS of PDCCH.
126 void precompute_type2_pdcch_slots(subcarrier_spacing scs_common);

128 // Args.
129 const scheduler_expert_config& expert_cfg;
130 const cell_configuration& cell_cfg;
131 pdcch_resource_allocator& pdcch_sch;

133 // Derived args.
134
135 /// Default Paging cycle (DRX cycle) configured in SIB1, expressed in nof. radio frames. This value is used if F1AP
136 /// does not provide Paging DRX IE in Paging message.
137 unsigned default_paging_cycle;
138 /// Number of paging frames per DRX cycle. Value of N in the equation in clause 7.1 of TS 38.304.
139 unsigned nof_pf_per_drx_cycle;
140 /// Paging frame offset. Value of PF_offset in the equation in clause 7.1 of TS 38.304.
141 unsigned paging_frame_offset{};
142 /// Number of paging occasions per paging frame. Value of Ns in the equation in clause 7.1 of TS 38.304.
143 uint8_t nof_po_per_pf;
144
145 /// Array of Type0-PDCCH CSS monitoring slots (1 per beam) that will be used for Paging scheduling if
146 /// pagingSearchSpace is 0 [TS 38.213, Section 13].
147 std::array<slot_point, MAX_NUM_BEAMS> type0_pdcch_css_slots;
148
149 /// Array of Type2-PDCCH CSS monitoring slots that will be used for Paging scheduling if pagingSearchSpace > 0.
150 /// NOTE1: nrofPDCCH-MonitoringOccasionPerSSB-InPO is always 1. See \c
151 /// pcch_config::NR_OF_PDCCH_MONITORING_OCCASION_PER_SSB_IN_PO.
152 /// NOTE2: Row number corresponds to SSB beam index and column number corresponds to Paging Occasion index (i_s).
153 std::array<static_vector<slot_point, pcch_config::MAX_PO_PER_PF>, MAX_NUM_BEAMS> type2_pdcch_css_slots;
154 /// Search Space configuration when pagingSearchSpace > 0, if configured.
155 const search_space_configuration* ss_cfg = nullptr;
156
157 /// BWP configuration of CORESET used for Paging (applies for both pagingSearchSpace = 0 and pagingSearchSpace > 0).
158 /// It's used for CRB-to-PRB conversion.
159 bwp_configuration bwp_cfg;
160 /// PDSCH time domain resource allocation list.
161 span<const pdsch_time_domain_resource_allocation> pdsch_td_alloc_list;
162
163 /// List of notifications from upper layers containing Paging information.
164 /// This is used only to avoid data race between threads.
165 slot_event_list<sched_paging_information> new_paging_notifications;
166 /// Contains paging information of UEs yet to be scheduled.
167 std::unordered_map<ue_paging_id, ue_paging_info> paging_pending_ues;
168 /// Lookup to keep track of scheduled paging UEs at a particular PDSCH time resource index. Index of \c
169 /// pdsch_time_res_idx_to_scheduled_ues_lookup corresponds to PDSCH Time Domain Resource Index.
170 static_vector<std::vector<const sched_paging_information*>, MAX_NOF_PDSCH_TD_RESOURCE_ALLOCATIONS>
171 pdsch_time_res_idx_to_scheduled_ues_lookup;
172
173 srslog::basic_logger& logger;
174};
