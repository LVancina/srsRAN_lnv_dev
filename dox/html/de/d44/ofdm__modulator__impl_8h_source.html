39/// Describes a generic OFDM symbol modulator.
40class ofdm_symbol_modulator_impl : public ofdm_symbol_modulator
41{
42private:
43 /// Indicates the DFT size.
44 unsigned dft_size;
45 /// Indicates the resource grid bandwidth in resource elements.
46 unsigned rg_size;
47 /// Cyclic prefix type.
48 cyclic_prefix cp;
49 /// Numerology.
50 subcarrier_spacing scs;
51 /// Sampling rate in Hz.
52 unsigned sampling_rate_Hz;
53 /// Scaling factor at the DFT output.
54 float scale;
55 /// DFT processor.
56 std::unique_ptr<dft_processor> dft;
57 /// Phase compensation look-up table.
58 phase_compensation_lut phase_compensation_table;
59
60public:
61 /// \brief Constructs an OFDM symbol modulator.
62 /// \param[in] common_config Provides specific configuration parameters from the factory.
63 /// \param[in] ofdm_config Provides generic OFDM configuration parameters.
64 ofdm_symbol_modulator_impl(ofdm_modulator_common_configuration& common_config,
65 const ofdm_modulator_configuration& ofdm_config);
66
67 // See interface for documentation.
68 unsigned get_symbol_size(unsigned symbol_index) const override
69 {
70 return cp.get_length(symbol_index, scs).to_samples(sampling_rate_Hz) + dft_size;
71 }
72
73 // See interface for documentation.
74 void
75 modulate(span<cf_t> ouput, const resource_grid_reader& grid, unsigned port_index, unsigned symbol_index) override;
76};
