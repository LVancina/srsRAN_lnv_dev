32/// \brief AMF_UE_ID (non ASN1 type of AMF_UE_NGAP_ID) used to identify the UE in the AMF.
33/// \remark See TS 38.413 Section 9.3.3.1: AMF_UE_NGAP_ID valid values: (0..2^40-1)
34constexpr static uint64_t MAX_NOF_AMF_UES = ((uint64_t)1 << 40);
35enum class amf_ue_id_t : uint64_t { min = 0, max = MAX_NOF_AMF_UES - 1, invalid = 0x1ffffffffff };
36
37/// Convert AMF_UE_ID type to integer.
38inline uint64_t amf_ue_id_to_uint(amf_ue_id_t id)
39{
40 return static_cast<uint64_t>(id);
41}
42
43/// Convert integer to AMF_UE_ID type.
44inline amf_ue_id_t uint_to_amf_ue_id(std::underlying_type_t<amf_ue_id_t> id)
45{
46 return static_cast<amf_ue_id_t>(id);
47}
