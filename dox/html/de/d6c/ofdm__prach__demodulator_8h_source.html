31/// \brief OFDM PRACH demodulator.
32///
33/// Performs OFDM demodulation for all the PRACH frequency- and time-domain occasions within a PRACH slot as per
34/// TS38.211 Section 5.3.2.
35///
36class ofdm_prach_demodulator
37{
38public:
39 /// Collects the necessary parameters to demodulate PRACH frequency- and time-domain occasions within a slot.
40 struct configuration {
41 /// PRACH preamble format.
42 prach_format_type format;
43 /// Number of time-domain occasions.
44 unsigned nof_td_occasions;
45 /// Number of frequency-domain occasions.
46 unsigned nof_fd_occasions;
47 /// Starting symbol index for the first PRACH TD occasion within the slot {0, ..., 13} at \c pusch_scs.
48 unsigned start_symbol;
49 /// PRACH frequency location (see \ref prach_buffer_context::rb_offset).
50 unsigned rb_offset;
51 /// Uplink resource grid size (see \ref prach_buffer_context::nof_prb_ul_grid).
52 unsigned nof_prb_ul_grid;
53 /// Uplink resource grid subcarrier spacing. Expresses the numerology \f$\mu\f$.
54 subcarrier_spacing pusch_scs;
55 /// Destination port identifier.
56 unsigned port;
57 };
58
59 /// Default destructor.
60 virtual ~ofdm_prach_demodulator() = default;
61
62 /// \brief Demodulates a PRACH occasion as described in TS38.211 Section 5.3.2.
63 ///
64 /// For long preambles, the input shall contain the cyclic prefix of length \f$N_{CP,l}^{RA}\f$) and a time-domain
65 /// sequence of length \f$N_u\f$. Also, the number of time-domain occasions must be one.
66 ///
67 /// For short preamble formats, the input shall correspond to one slot at \ref configuration::pusch_scs.
68 ///
69 /// An assertion is triggered if:
70 /// - the input samples are not sufficient,
71 /// - the buffer dimensions cannot contain the demodulated sequences, or
72 /// - the port identifier exceeds the number of ports contained in the destination buffer.
73 ///
74 /// \param[out] buffer Frequency-domain signal corresponding to one PRACH receive occasion.
75 /// \param[in] input PRACH receive window baseband samples.
76 /// \param[in] config PRACH demodulator configuration parameters.
77 virtual void demodulate(prach_buffer& buffer, span<const cf_t> input, const configuration& config) = 0;
78};
79} // namespace srsran
