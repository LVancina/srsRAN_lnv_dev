40/// \brief Custom formatter for \c pdcch_processor::coreset_description.
41template <>
42struct formatter<srsran::pdcch_processor::coreset_description> {
43 /// Helper used to parse formatting options and format fields.
44 srsran::delimited_formatter helper;
45
46 /// Default constructor.
47 formatter() = default;
48
49 template <typename ParseContext>
50 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
51 {
52 return helper.parse(ctx);
53 }
54
55 template <typename FormatContext>
56 auto format(const srsran::pdcch_processor::coreset_description& coreset, FormatContext& ctx)
57 -> decltype(std::declval<FormatContext>().out())
58 {
59 helper.format_always(ctx, "bwp=[{}, {})", coreset.bwp_start_rb, coreset.bwp_start_rb + coreset.bwp_size_rb);
60 helper.format_always(
61 ctx, "symb=[{}, {})", coreset.start_symbol_index, coreset.start_symbol_index + coreset.duration);
62 helper.format_always(ctx, "f_re={}", coreset.frequency_resources);
63
64 switch (coreset.cce_to_reg_mapping) {
65 case srsran::pdcch_processor::cce_to_reg_mapping_type::CORESET0:
66 helper.format_if_verbose(ctx, "mapping=coreset0");
67 break;
68 case srsran::pdcch_processor::cce_to_reg_mapping_type::INTERLEAVED:
69 helper.format_if_verbose(ctx, "mapping=interleaved");
70 helper.format_if_verbose(ctx, "reg_bundle_size={}", coreset.reg_bundle_size);
71 helper.format_if_verbose(ctx, "interleaver_size={}", coreset.interleaver_size);
72 break;
73 case srsran::pdcch_processor::cce_to_reg_mapping_type::NON_INTERLEAVED:
74 helper.format_if_verbose(ctx, "mapping=non-interleaved");
75 break;
76 }
77
78 helper.format_if_verbose(ctx, "shift_idx={}", coreset.shift_index);
79
80 return ctx.out();
81 }
82};

116 helper.format_if_verbose(ctx, "power_dmrs={:+.1f}dB", pdu.dci.dmrs_power_offset_dB);
117 helper.format_if_verbose(ctx, "power_data={:+.1f}dB", pdu.dci.data_power_offset_dB);

149/// \brief Custom formatter for \c pdsch_processor::pdu_t.
150template <>
151struct formatter<srsran::pdsch_processor::pdu_t> {
152 /// Helper used to parse formatting options and format fields.
153 srsran::delimited_formatter helper;
154
155 /// Default constructor.
156 formatter() = default;
157
158 template <typename ParseContext>
159 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
160 {
161 return helper.parse(ctx);
162 }
163
164 template <typename FormatContext>
165 auto format(const srsran::pdsch_processor::pdu_t& pdu, FormatContext& ctx)
166 -> decltype(std::declval<FormatContext>().out())
167 {
168 helper.format_always(ctx, "rnti=0x{:04x}", pdu.rnti);
169 if (pdu.context.has_value()) {
170 helper.format_always(ctx, pdu.context.value());
171 }
172 helper.format_if_verbose(ctx, "bwp=[{}, {})", pdu.bwp_start_rb, pdu.bwp_start_rb + pdu.bwp_size_rb);
173 helper.format_always(ctx, "prb={}", pdu.freq_alloc);
174 helper.format_always(ctx, "symb=[{}, {})", pdu.start_symbol_index, pdu.start_symbol_index + pdu.nof_symbols);
175 helper.format_always(ctx, srsran::span<const srsran::pdsch_processor::codeword_description>(pdu.codewords));
176
177 helper.format_if_verbose(ctx, "n_id={}", pdu.n_id);
178 helper.format_if_verbose(
179 ctx, "ref_point={}", (pdu.ref_point == srsran::pdsch_processor::pdu_t::CRB0) ? "CRB0" : "PRB0");
180 helper.format_if_verbose(ctx, "dmrs_type={}", (pdu.dmrs == srsran::dmrs_type::TYPE1) ? 1 : 2);
181 helper.format_if_verbose(ctx, "dmrs_mask={}", pdu.dmrs_symbol_mask);
182 helper.format_if_verbose(ctx, "n_scidid={}", pdu.scrambling_id);
183 helper.format_if_verbose(ctx, "n_scid={}", pdu.n_scid);
184 helper.format_if_verbose(ctx, "ncgwd={}", pdu.nof_cdm_groups_without_data);
185 helper.format_if_verbose(ctx, "bg={}", (pdu.ldpc_base_graph == srsran::ldpc_base_graph_type::BG1) ? "BG1" : "BG2");
186 helper.format_if_verbose(ctx, "lbrm={}bytes", pdu.tbs_lbrm_bytes);
187 helper.format_if_verbose(ctx, "power_dmrs={:+.1f}dB", -pdu.ratio_pdsch_dmrs_to_sss_dB);
188 helper.format_if_verbose(ctx, "power_data={:+.1f}dB", pdu.ratio_pdsch_data_to_sss_dB);
189 helper.format_if_verbose(ctx, "slot={}", pdu.slot);
190 helper.format_if_verbose(ctx, "cp={}", pdu.cp.to_string());
191 helper.format_if_verbose(ctx, "precoding={}", pdu.precoding);
192 if (pdu.reserved.get_nof_entries() > 0) {
193 helper.format_if_verbose(ctx, "reserved=[{:,}]", pdu.reserved.get_re_patterns());
194 }
195 return ctx.out();
196 }

418/// \brief Custom formatter for \c pucch_processor_result.
419template <>
420struct formatter<srsran::pucch_processor_result> {
421 /// Helper used to parse formatting options and format fields.
422 srsran::delimited_formatter helper;
423
424 /// Default constructor.
425 formatter() = default;
426
427 template <typename ParseContext>
428 auto parse(ParseContext& ctx) -> decltype(ctx.begin())
429 {
430 return helper.parse(ctx);
431 }
432
433 template <typename FormatContext>
434 auto format(const srsran::pucch_processor_result& result, FormatContext& ctx)
435 -> decltype(std::declval<FormatContext>().out())
436 {
437 unsigned nof_sr = result.message.get_expected_nof_sr_bits();
438 unsigned nof_harq_ack = result.message.get_expected_nof_harq_ack_bits();
439 unsigned nof_csi_part1 = result.message.get_expected_nof_csi_part1_bits();
440 unsigned nof_csi_part2 = result.message.get_expected_nof_csi_part2_bits();
441
442 helper.format_if_verbose(ctx, "status={}", to_string(result.message.get_status()));
443
444 // PUCCH can carry a scheduling request without HARQ-ACK. In that case, the UE only transmits for positive SR.
445 if ((nof_sr == 0) && (nof_harq_ack == 0) && (nof_csi_part1 == 0) && (nof_csi_part2 == 0)) {
446 if (result.message.get_status() == srsran::uci_status::valid) {
447 helper.format_always(ctx, "sr=yes");
448 } else {
449 helper.format_always(ctx, "sr=no");
450 }
451 }
452
453 if (result.message.get_status() == srsran::uci_status::valid) {
454 // Valid UCI payload.
455 if (nof_harq_ack) {
456 helper.format_always(ctx, "ack={:#}", srsran::span<const uint8_t>(result.message.get_harq_ack_bits()));
457 }
458 if (nof_sr) {
459 helper.format_always(ctx, "sr={:#}", srsran::span<const uint8_t>(result.message.get_sr_bits()));
460 }
461 if (nof_csi_part1) {
462 helper.format_always(ctx, "csi1={:#}", srsran::span<const uint8_t>(result.message.get_csi_part1_bits()));
463 }
464 if (nof_csi_part2) {
465 helper.format_always(ctx, "csi2={:#}", srsran::span<const uint8_t>(result.message.get_csi_part2_bits()));
466 }
467 } else {
468 // Bad UCI payload.
469 std::array<uint8_t, srsran::uci_constants::MAX_NOF_PAYLOAD_BITS> bad_payload;
470 std::fill(bad_payload.begin(), bad_payload.end(), 2U);
471
472 if (nof_harq_ack) {
473 helper.format_always(ctx, "ack={:#}", srsran::span<const uint8_t>(bad_payload).first(nof_harq_ack));
474 }
475 if (nof_sr) {
476 helper.format_always(ctx, "sr={:#}", srsran::span<const uint8_t>(bad_payload).first(nof_sr));
477 }
478 if (nof_csi_part1) {
479 helper.format_always(ctx, "csi1={:#}", srsran::span<const uint8_t>(bad_payload).first(nof_csi_part1));
480 }
481 if (nof_csi_part2) {
482 helper.format_always(ctx, "csi2={:#}", srsran::span<const uint8_t>(bad_payload).first(nof_csi_part2));
483 }
484 }
485
486 if (result.detection_metric.has_value()) {
487 helper.format_if_verbose(ctx, "detection_metric={:.1f}", result.detection_metric.value());
488 }
489
490 // Channel State Information.
491 helper.format_if_verbose(ctx, "epre={:+.1f}dB", result.csi.get_epre_dB());
492 helper.format_if_verbose(ctx, "rsrp={:+.1f}dB", result.csi.get_rsrp_dB());
493 helper.format_if_verbose(ctx, "sinr={:+.1f}dB", result.csi.get_sinr_dB());
494 helper.format_if_verbose(ctx, "t_align={:.1f}us", result.csi.get_time_alignment().to_seconds() * 1e6);
495
496 return ctx.out();
497 }
498};
