149/// \brief Downlink processor software factory configuration.
150///
151/// \remark Default values are empty strings which are invalid.
152struct downlink_processor_factory_sw_config {
153 /// \brief LDPC encoder type.
154 ///
155 /// Use of there options:
156 /// - \c auto: let the factory select the most efficient given the CPU architecture, or
157 /// - \c generic: for using unoptimized LDPC encoder, or
158 /// - \c avx2: for using AVX2 optimized LDPC encoder (x86_64 CPUs only), or
159 /// - \c neon: for using NEON optimized LDPC encoder (ARM CPUs only).
160 std::string ldpc_encoder_type;
161 /// \brief CRC calculator type.
162 ///
163 /// Use of there options:
164 /// - \c auto: let the factory select the most efficient given the CPU architecture, or
165 /// - \c lut: for using a look-up table CRC calculator, or
166 /// - \c clmul: for using a look-up table CRC calculator (x86_64 CPUs only).
167 std::string crc_calculator_type;
168 /// \brief PDSCH processor type.
169 ///
170 /// Use of there options:
171 /// - \c generic: for using unoptimized PDSCH processing, or
172 /// - \c concurrent: for using a processor that processes code blocks in parallel, or
173 /// - \c lite: for using a memory optimized processor.
174 variant<pdsch_processor_generic_configuration,
175 pdsch_processor_concurrent_configuration,
176 pdsch_processor_lite_configuration>
177 pdsch_processor;
178 /// Number of concurrent threads processing downlink transmissions.
179 unsigned nof_concurrent_threads;
180};

219/// Upper PHY configuration parameters used to create a new upper PHY object.
220struct upper_phy_config {
221 /// \brief Logging level.
222 /// - \c none: No logging is enabled, or
223 /// - \c info: all processing PDUs and their results are logged.
224 srslog::basic_levels log_level;
225 /// Enable logging broadcast channels such as SSB, PDSCH and PDCCH channels with broadcast RNTIs.
226 bool enable_logging_broadcast;
227 /// Logger maximum hexadecimal dump size. Set to zero for none.
228 unsigned logger_max_hex_size;
229 /// Selects the PUSCH SINR calculation method used for choosing the modulation and coding scheme.
230 channel_state_information::sinr_type pusch_sinr_calc_method;
231 /// Receive symbol printer. Leave empty to disable.
232 std::string rx_symbol_printer_filename;
233 /// Receive port the UL symbols are dumped from. Leave emtpy for all ports.
234 optional<unsigned> rx_symbol_printer_port;
235 /// Boolean flag for dumping PRACH symbols when set to true.
236 bool rx_symbol_printer_prach;
237 /// \brief LDPC decoder type.
238 ///
239 /// Use of there options:
240 /// - \c auto: let the factory select the most efficient given the CPU architecture, or
241 /// - \c generic: for using generic instructions, or
242 /// - \c avx2: for using AVX2 instructions (x86_64 CPUs only), or
243 /// - \c avx512: for using AVX512 instructions (x86_64 CPUs only), or
244 /// - \c neon: for using NEON instructions (ARM CPUs only).
245 std::string ldpc_decoder_type;
246 /// \brief LDPC rate dematcher type.
247 ///
248 /// Use of there options:
249 /// - \c auto: let the factory select the most efficient given the CPU architecture, or
250 /// - \c generic: for using generic instructions, or
251 /// - \c avx2: for using AVX2 instructions (x86_64 CPUs only), or
252 /// - \c avx512: for using AVX512 instructions (x86_64 CPUs only), or
253 /// - \c neon: for using NEON instructions (ARM CPUs only).
254 std::string ldpc_rate_dematcher_type;
255 /// \brief CRC calculator type.
256 ///
257 /// Use of there options:
258 /// - \c auto: let the factory select the most efficient given the CPU architecture, or
259 /// - \c lut: for using a look-up table CRC calculator, or
260 /// - \c clmul: for using a look-up table CRC calculator (x86_64 CPUs only).
261 std::string crc_calculator_type;
262 /// Number of LDPC decoder iterations.
263 unsigned ldpc_decoder_iterations;
264 /// Set to true to enable the LDPC decoder early stop.
265 bool ldpc_decoder_early_stop;
266 /// Radio sector identifier.
267 unsigned sector_id;
268 /// Number of transmit antenna ports.
269 unsigned nof_tx_ports;
270 /// Number of receive antenna ports.
271 unsigned nof_rx_ports;
272 /// Number of downlink resource grids. Downlink resource grids minimum reuse time is \c dl_rg_expire_timeout_slots
273 /// slots.
274 unsigned nof_dl_rg;
275 /// Downlink resource grid timeout expiration in number of slots.
276 unsigned dl_rg_expire_timeout_slots;
277 /// Number of uplink resource grids. They are reused after \c nof_ul_rg slots.
278 unsigned nof_ul_rg;
279 /// Number of PRACH buffer.
280 unsigned nof_prach_buffer;
281 /// Maximum number of time-domain occasions.
282 unsigned max_nof_td_prach_occasions;
283 /// Maximum number of frequency-domain occasions.
284 unsigned max_nof_fd_prach_occasions;
285 /// Set to true if the PRACH preamble is long.
286 bool is_prach_long_format;
287 /// Maximum number of concurrent downlink processes.
288 unsigned nof_dl_processors;
289 /// Maximum uplink processor thread concurrency.
290 unsigned max_ul_thread_concurrency;
291 /// Maximum asynchronous PUSCH processing concurrency for each UL processor.
292 unsigned max_pusch_concurrency;
293 /// Number of threads that simultaneously use a PUSCH decoder.
294 unsigned nof_pusch_decoder_threads;
295 /// Number of RBs for downlink.
296 unsigned dl_bw_rb;
297 /// Number of RBs for uplink.
298 unsigned ul_bw_rb;
299 /// Request headroom size in slots.
300 unsigned nof_slots_request_headroom;
301 /// List of active subcarrier spacing, indexed by numerology.
302 std::array<bool, to_numerology_value(subcarrier_spacing::invalid)> active_scs;
303 /// Receive buffer pool configuration.
304 rx_buffer_pool_config rx_buffer_config;
305 /// Upper PHY resource grid gateway.
306 upper_phy_rg_gateway* rg_gateway;
307 /// Downlink task executors.
308 span<task_executor*> dl_executors;
309 /// PUCCH task executor.
310 task_executor* pucch_executor;
311 /// PUSCH task executor.
312 task_executor* pusch_executor;
313 /// PUSCH decoder task executor.
314 task_executor* pusch_decoder_executor;
315 /// PRACH task executor.
316 task_executor* prach_executor;
317 /// Received symbol request notifier.
318 upper_phy_rx_symbol_request_notifier* rx_symbol_request_notifier;
319};
