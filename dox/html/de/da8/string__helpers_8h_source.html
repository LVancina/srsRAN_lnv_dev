29/// Splits a given string into multiple elements given a delimiter. The elements are casted to the specified type.
30/// Insertable It is the list data-type. It needs to implement insert(iterator, element)
31/// \param input It is the input string
32/// \param delimiter Character used for indicating the end of the strings
33/// \param list contains the parsed values
34template <class Insertable>
35inline void string_parse_list(const std::string& input, char delimiter, Insertable& list)
36{
37 std::stringstream ss(input);
38
39 // Removes all possible elements of the list
40 list.clear();
41
42 while (ss.good()) {
43 std::string substr;
44 std::getline(ss, substr, delimiter);
45
46 if (not substr.empty()) {
47 list.insert(list.end(), substr);
48 }
49 }
50}
