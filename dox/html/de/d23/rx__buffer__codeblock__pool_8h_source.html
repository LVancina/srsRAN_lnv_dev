37/// Manages a codeblock buffer pool.
38class rx_buffer_codeblock_pool
39{
40private:
41 /// Codeblock identifier list type.
42 using codeblock_identifier_list =
43 concurrent_queue<unsigned, concurrent_queue_policy::lockfree_mpmc, concurrent_queue_wait_policy::non_blocking>;
44
45 /// Describes a codeblock buffer entry.
46 struct entry {
47 /// Contains the codeblock soft bits.
48 std::vector<log_likelihood_ratio> soft_bits;
49 /// Contains the codeblock data bits.
50 dynamic_bit_buffer data_bits;
51 };
52
53 /// Stores all codeblock entries.
54 std::vector<entry> entries;
55 /// List containing the free codeblocks identifiers.
56 codeblock_identifier_list free_list;
57
58public:
59 /// \brief Creates a receive buffer codeblock pool.
60 /// \param[in] nof_codeblocks Indicates the maximum number of codeblocks.
61 /// \param[in] max_codeblock_size Indicates the maximum codeblock size.
62 /// \param[in] external_soft_bits Set to true to indicate that soft bits are not stored in the buffer.
63 rx_buffer_codeblock_pool(unsigned nof_codeblocks, unsigned max_codeblock_size, bool external_soft_bits) :
64 free_list(nof_codeblocks)
65 {
66 entries.resize(nof_codeblocks);
67 unsigned cb_id = 0;
68 for (entry& e : entries) {
69 e.soft_bits.resize(external_soft_bits ? 0 : max_codeblock_size);
70 // The maximum number of data bits is
71 // max_codeblock_size * max(BG coding rate) = max_codeblock_size * (1/3)
72 e.data_bits.resize(divide_ceil(max_codeblock_size, 3));
73 // Push codeblock identifier into the free list.
74 while (!free_list.try_push(cb_id++)) {
75 }
76 }
77 }
78
79 /// \brief Reserves a codeblock buffer.
80 /// \return The codeblock identifier in the pool if it is reserved successfully. Otherwise, \c nullopt
81 optional<unsigned> reserve()
82 {
83 // Try to get an available codeblock.
84 return free_list.try_pop();
85 }
86
87 /// \brief Frees a codeblock buffer.
88 /// \param[in] cb_id Indicates the codeblock identifier in the pool.
89 void free(unsigned cb_id)
90 {
91 // Push codeblock identifier back in the pool.
92 while (!free_list.try_push(cb_id)) {
93 }
94 }
95
96 /// \brief Gets a codeblock soft-bit buffer.
97 /// \param[in] cb_id Indicates the codeblock identifier.
98 /// \return A view to the codeblock soft-bit buffer.
99 span<log_likelihood_ratio> get_soft_bits(unsigned cb_id)
100 {
101 srsran_assert(cb_id < entries.size(), "Codeblock index ({}) is out of range ({}).", cb_id, entries.size());
102 return entries[cb_id].soft_bits;
103 }
104
105 /// \brief Gets a codeblock data-bit buffer.
106 /// \param[in] cb_id Indicates the codeblock identifier.
107 /// \return A view to the codeblock data-bit buffer.
108 bit_buffer& get_data_bits(unsigned cb_id)
109 {
110 srsran_assert(cb_id < entries.size(), "Codeblock index ({}) is out of range ({}).", cb_id, entries.size());
111 return entries[cb_id].data_bits;
112 }
113};
