37/// \brief UE context setup configuration
38struct ue_context_cfg {
39 security::sec_as_config security_info;
40 activity_notification_level_t activity_level;
41 optional<std::chrono::seconds> ue_inactivity_timeout;
42 std::map<five_qi_t, srs_cu_up::cu_up_qos_config> qos;
43};

45/// \brief Context for a UE within the CU-UP with storage for all active PDU sessions.
46class ue_context : public pdu_session_manager_ctrl
47{
48public:
49 ue_context(ue_index_t index_,
50 ue_context_cfg cfg_,
51 e1ap_control_message_handler& e1ap_,
52 network_interface_config& net_config_,
53 n3_interface_config& n3_config_,
54 timer_factory timers_,
55 f1u_cu_up_gateway& f1u_gw_,
56 gtpu_teid_pool& f1u_teid_allocator_,
57 gtpu_tunnel_tx_upper_layer_notifier& gtpu_tx_notifier_,
58 gtpu_demux_ctrl& gtpu_rx_demux_,
59 dlt_pcap& gtpu_pcap) :
60 index(index_),
61 cfg(cfg_),
62 logger("CU-UP", {index_}),
63 e1ap(e1ap_),
64 pdu_session_manager(index,
65 cfg.qos,
66 cfg.security_info,
67 net_config_,
68 n3_config_,
69 logger,
70 ue_inactivity_timer,
71 timers_,
72 f1u_gw_,
73 f1u_teid_allocator_,
74 gtpu_tx_notifier_,
75 gtpu_rx_demux_,
76 gtpu_pcap),
77 timers(timers_)
78 {
79 if (cfg.activity_level == activity_notification_level_t::ue) {
80 if (not cfg.ue_inactivity_timeout.has_value()) {
81 report_error(
82 "Failed to create UE context. Activity notification level is UE, but no UE inactivity timer configured\n");
83 }
84 ue_inactivity_timer = timers.create_timer();
85 ue_inactivity_timer.set(*cfg.ue_inactivity_timeout,
86 [this](timer_id_t /*tid*/) { on_ue_inactivity_timer_expired(); });
87 ue_inactivity_timer.run();
88 }
89 }
90 ~ue_context() override = default;
91
92 // pdu_session_manager_ctrl
93 pdu_session_setup_result setup_pdu_session(const e1ap_pdu_session_res_to_setup_item& session) override
94 {
95 return pdu_session_manager.setup_pdu_session(session);
96 }
97 pdu_session_modification_result modify_pdu_session(const e1ap_pdu_session_res_to_modify_item& session,
98 bool new_ul_tnl_info_required) override
99 {
100 return pdu_session_manager.modify_pdu_session(session, new_ul_tnl_info_required);
101 }
102 void remove_pdu_session(pdu_session_id_t pdu_session_id) override
103 {
104 pdu_session_manager.remove_pdu_session(pdu_session_id);
105 }
106 size_t get_nof_pdu_sessions() override { return pdu_session_manager.get_nof_pdu_sessions(); }
107
108 ue_index_t get_index() const { return index; };
109
110 const cu_up_ue_logger& get_logger() const { return logger; };
111
112private:
113 ue_index_t index;
114 ue_context_cfg cfg;
115
116 cu_up_ue_logger logger;
117
118 e1ap_control_message_handler& e1ap;
119 pdu_session_manager_impl pdu_session_manager;
120
121 timer_factory timers;
122 unique_timer ue_inactivity_timer;
123 void on_ue_inactivity_timer_expired()
124 {
125 e1ap_bearer_context_inactivity_notification msg = {};
126 msg.ue_index = index;
127 e1ap.handle_bearer_context_inactivity_notification(msg);
128 }
129};
