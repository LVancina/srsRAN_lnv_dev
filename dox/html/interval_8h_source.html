34template <typename T, bool RightClosed = false>
35class interval
36{
37 static_assert(std::is_trivially_copyable<T>::value, "Expected to be trivially copyable");
38
39public:
41 using is_real = std::is_floating_point<T>;
43 using length_type =
44 std::conditional_t<is_real::value, T, std::conditional_t<(sizeof(T) > sizeof(uint32_t)), uint64_t, uint32_t>>;
45
46 constexpr interval() : start_(T{}), stop_(T{}) {}
47
48 template <typename U, typename V>
49 constexpr interval(U start_point, V stop_point) : start_(start_point), stop_(stop_point)
50 {
51 static_assert(std::is_convertible<U, T>::value, "Invalid interval start point type");
52 static_assert(std::is_convertible<V, T>::value, "Invalid interval stop point type");
53 srsran_assert(start_ <= stop_, "Invalid interval [{}, {})", start_, stop_);
54 }
55
56 T start() const { return start_; }
57
58 T stop() const { return stop_; }
59
60 bool empty() const { return not RightClosed and stop_ == start_; }
61
63 length_type length() const
64 {
65 return static_cast<length_type>(stop_ - start_ + ((RightClosed and not is_real::value) ? 1 : 0));
66 }
67
68 void set(T start_point, T stop_point)
69 {
70 srsran_assert(stop_point >= start_point, "interval::set called for invalid range points");
71 start_ = start_point;
72 stop_ = stop_point;
73 }
74
78 void extend(T length_increment)
79 {
80 // Detect length overflows
81 srsran_assert(std::is_unsigned<T>::value or (length_increment >= 0 or length() >= -length_increment),
82 "Resulting interval would be invalid");
83 stop_ += length_increment;
84 }
85
87 void resize(length_type new_length) { stop_ = start_ + new_length; }
88
90 void displace_by(int offset)
91 {
92 srsran_assert(
93 std::is_signed<T>::value or static_cast<int64_t>(start_) >= -offset,
94 "Cannot have negative starting_points in case interval<T> underlying type is unsigned. Start={} < offset={}",
95 start_,
96 -offset);
97 start_ += offset;
98 stop_ += offset;
99 }
100
102 void displace_to(T start_point)
103 {
104 stop_ = start_point + length();
105 start_ = start_point;
106 }
107
109 bool overlaps(interval other) const
110 {
111 // Note: if one or both intervals are empty, this will always return false
112 return RightClosed ? start_ <= other.stop_ and other.start_ <= stop_
113 : start_ < other.stop_ and other.start_ < stop_;
114 }
115
117 bool contains(T point) const { return start_ <= point and (point < stop_ or (RightClosed and point == stop_)); }
118
120 bool contains(interval other) const { return start_ <= other.start_ and other.stop_ <= stop_; }
121
123 interval<T>& intersect(const interval<T>& other)
124 {
125 if (not overlaps(other)) {
126 *this = interval<T>{};
127 } else {
128 start_ = std::max(start(), other.start());
129 stop_ = std::min(stop(), other.stop());
130 }
131 return *this;
132 }
133
134private:
135 T start_;
136 T stop_;
137};
138
139template <typename T>
140bool operator==(const interval<T>& lhs, const interval<T>& rhs)
141{
142 return lhs.start() == rhs.start() and lhs.stop() == rhs.stop();
143}
144
145template <typename T>
146bool operator!=(const interval<T>& lhs, const interval<T>& rhs)
147{
148 return not(lhs == rhs);
149}
150
153template <typename T>
154bool operator<(const interval<T>& lhs, const interval<T>& rhs)
155{
156 return lhs.start() < rhs.start() or (lhs.start() == rhs.start() and lhs.stop() < rhs.stop());
157}
158
160template <typename T>
161interval<T> operator|(const interval<T>& lhs, const interval<T>& rhs)
162{
163 if (not lhs.overlaps(rhs)) {
164 return interval<T>{};
165 }
166 return {std::min(lhs.start(), rhs.start()), std::max(lhs.stop(), rhs.stop())};
167}
168
170template <typename T>
171interval<T> operator&(const interval<T>& lhs, const interval<T>& rhs)
172{
173 if (not lhs.overlaps(rhs)) {
174 return interval<T>{};
175 }
176 return interval<T>{std::max(lhs.start(), rhs.start()), std::min(lhs.stop(), rhs.stop())};
177}
