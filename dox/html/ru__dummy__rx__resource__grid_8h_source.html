34class ru_dummy_rx_resource_grid : public resource_grid_reader
35{
36public:
37 ru_dummy_rx_resource_grid(unsigned seed, unsigned nof_subc, unsigned nof_symbols, unsigned nof_ports) :
38 rg_buffer({nof_subc, nof_symbols, nof_ports})
39 {
40 std::mt19937 rgen(seed);
41 complex_normal_distribution<cf_t> dist(0, 1);
42
43 span<cf_t> data = rg_buffer.get_data();
44 std::generate(data.begin(), data.end(), [&rgen, &dist]() { return dist(rgen); });
45 }
46
47 // See resource_grid_base for documentation.
48 unsigned get_nof_ports() const override { return rg_buffer.get_dimension_size(resource_grid_dimensions::port); }
49
50 // See resource_grid_base for documentation.
51 unsigned get_nof_subc() const override { return rg_buffer.get_dimension_size(resource_grid_dimensions::subc); }
52
53 // See resource_grid_base for documentation.
54 unsigned get_nof_symbols() const override { return rg_buffer.get_dimension_size(resource_grid_dimensions::symbol); }
55
56 // See resource_grid_reader for documentation.
57 bool is_empty(unsigned port) const override
58 {
59 // This grid is never empty.
60 return false;
61 }
62
63 // See resource_grid_reader for documentation.
64 bool is_empty() const override
65 {
66 // This grid is never empty.
67 return false;
68 }
69
70 // See resource_grid_reader for documentation.
71 span<cf_t> get(span<cf_t> symbols, unsigned port, unsigned l, unsigned k_init, span<const bool> mask) const override
72 {
73 // Count number of elements.
74 unsigned nof_symbols = std::accumulate(
75 mask.begin(), mask.end(), 0U, [](unsigned count, bool active_re) { return count + (active_re ? 1U : 0U); });
76
77 return get(symbols, nof_symbols);
78 }
79
80 // See resource_grid_reader for documentation.
81 span<cf_t> get(span<cf_t> symbols,
82 unsigned port,
83 unsigned l,
84 unsigned k_init,
85 const bounded_bitset<MAX_RB * NRE>& mask) const override
86 {
87 // Count number of elements.
88 unsigned nof_symbols = mask.count();
89
90 return get(symbols, nof_symbols);
91 }
92
93 // See resource_grid_reader for documentation.
94 void get(span<cf_t> symbols, unsigned port, unsigned l, unsigned k_init) const override
95 {
96 srsvec::copy(symbols, rg_buffer.get_data().first(symbols.size()));
97 }
98
99 // See resource_grid_reader for documentation.
100 span<const cf_t> get_view(unsigned port, unsigned l) const override { return rg_buffer.get_view({l, port}); }
101
102private:
104 span<cf_t> get(span<cf_t> symbols, unsigned nof_symbols) const
105 {
106 srsvec::copy(symbols.first(nof_symbols), rg_buffer.get_data().first(nof_symbols));
107
108 return symbols.last(symbols.size() - nof_symbols);
109 }
110
115 dynamic_tensor<static_cast<unsigned>(resource_grid_dimensions::all), cf_t, resource_grid_dimensions> rg_buffer;
116};
