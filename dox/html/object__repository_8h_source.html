35template <typename K, typename V>
36class object_repository
37{
38 mutable detail::mutex m;
39 std::unordered_map<K, V> repo;
40
41public:
44 V* insert(const K& key, V&& value)
45 {
46 detail::scoped_lock lock(m);
47 const auto& insertion = repo.emplace(key, std::move(value));
48 if (!insertion.second)
49 return nullptr;
50 return &insertion.first->second;
51 }

56 template <typename... Args>
57 V& emplace(Args&&... args)
58 {
59 detail::scoped_lock lock(m);
60 auto insertion = repo.emplace(std::forward<Args>(args)...);
61 return insertion.first->second;
62 }
63
66 V* find(const K& key)
67 {
68 detail::scoped_lock lock(m);
69 auto it = repo.find(key);
70 return (it != repo.end()) ? &it->second : nullptr;
71 }
72 const V* find(const K& key) const
73 {
74 detail::scoped_lock lock(m);
75 const auto it = repo.find(key);
76 return (it != repo.cend()) ? &it->second : nullptr;
77 }
78
80 std::vector<V*> contents()
81 {
82 detail::scoped_lock lock(m);
83
84 std::vector<V*> data;
85 data.reserve(repo.size());
86 for (auto& elem : repo) {
87 data.push_back(&elem.second);
88 }
89
90 return data;

93 {
94 detail::scoped_lock lock(m);
95

98 for (const auto& elem : repo) {
99 data.push_back(&elem.second);
100 }
101

103 }
