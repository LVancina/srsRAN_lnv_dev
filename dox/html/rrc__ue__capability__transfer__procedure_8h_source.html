69inline void fill_asn1_rrc_ue_capability_enquiry(asn1::rrc_nr::ue_cap_enquiry_s& rrc_ue_cap_enquiry,
70 uint8_t rrc_transaction_id,
71 std::vector<nr_band> bands)
72{
73 using namespace asn1::rrc_nr;
74 ue_cap_enquiry_ies_s& ue_cap_ies = rrc_ue_cap_enquiry.crit_exts.set_ue_cap_enquiry();
75 rrc_ue_cap_enquiry.rrc_transaction_id = rrc_transaction_id;
76
77 asn1::rrc_nr::ue_cap_rat_request_s ue_cap_rat_request;
78 ue_cap_rat_request.rat_type = asn1::rrc_nr::rat_type_e::nr;
79
80 ue_cap_request_filt_nr_s filter;
81 for (const auto& band : bands) {
82 freq_band_info_c band_info;
83 band_info.set_band_info_nr().band_nr = nr_band_to_uint(band);
84 filter.freq_band_list_filt.push_back(band_info);
85 }
86
87 byte_buffer filter_packed;
88 asn1::bit_ref bref(filter_packed);
89 filter.pack(bref); // possible packing error fill be shown in ASN1 logs
90 ue_cap_rat_request.cap_request_filt = std::move(filter_packed);
91
92 ue_cap_ies.ue_cap_rat_request_list.push_back(ue_cap_rat_request);
93}
