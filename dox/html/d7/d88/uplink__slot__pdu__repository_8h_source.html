32/// Defines an entry of the uplink slot PDU repository.
33using uplink_slot_pdu_entry = variant<uplink_processor::pusch_pdu, uplink_processor::pucch_pdu>;
34
35/// \brief Uplink slot PDU repository.
36///
37/// This class registers PUSCH and PUCCH PDUs. PDUs are indexed by the uplink slot they will be processed in.
38// :TODO: move this class to private when the uplink processor gets refactorized.
39class uplink_slot_pdu_repository
40{
41 using slot_entry = static_vector<uplink_slot_pdu_entry, MAX_UL_PDUS_PER_SLOT>;
42
43public:
44 /// \brief Constructs an uplink slot pdu repository that supports the given number of slots.
45 ///
46 /// \param nof_slots_[in] Number of slots supported by the repository.
47 explicit uplink_slot_pdu_repository(unsigned nof_slots_) : nof_slots(nof_slots_), repository(nof_slots_) {}
48
49 /// Adds the given PUSCH PDU to the repository at the given slot.
50 void add_pusch_pdu(slot_point slot, const uplink_processor::pusch_pdu& pdu)
51 {
52 assert_slot(slot);
53 repository[slot.to_uint() % nof_slots].push_back(pdu);
54 }
55
56 /// Adds the given PUCCH PDU to the repository at the given slot.
57 void add_pucch_pdu(slot_point slot, const uplink_processor::pucch_pdu& pdu)
58 {
59 assert_slot(slot);
60 repository[slot.to_uint() % nof_slots].push_back(pdu);
61 }
62
63 /// Clears the given slot of the registry.
64 void clear_slot(slot_point slot)
65 {
66 assert_slot(slot);
67 repository[slot.to_uint() % nof_slots].clear();
68 }
69
70 /// Returns a span that contains the PDUs for the given slot.
71 span<const uplink_slot_pdu_entry> get_pdus(slot_point slot) const
72 {
73 assert_slot(slot);
74 return repository[slot.to_uint() % nof_slots];
75 }
76
77private:
78 /// Number of slots.
79 const size_t nof_slots;
80 /// Repository that contains the PDUs.
81 std::vector<slot_entry> repository;
82
83 /// Asserts the slot numerology with the pool dimensions.
84 void assert_slot(slot_point slot) const
85 {
86 srsran_assert(slot.nof_slots_per_system_frame() % nof_slots == 0,
87 "The minimum supported number of slots per system frame (i.e., {}) must be divisible by the number "
88 "of slots (i.e., {})",
89 slot.nof_slots_per_system_frame(),
90 nof_slots);
91 }
92};
93} // namespace srsran
