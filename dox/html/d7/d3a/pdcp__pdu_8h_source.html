40/// PDCP Data/Control (D/C) field
41/// Ref: TS 38.323 Sec. 6.3.7
42enum class pdcp_dc_field : unsigned {
43 control = 0b00, ///< Control PDU
44 data = 0b01 ///< Data PDU
45};
46constexpr unsigned to_number(pdcp_dc_field dc)
47{
48 return static_cast<unsigned>(dc);
49}
50
51/// \brief Reads the D/C field from the first (header) byte of a PDCP PDU
52/// \param first_byte First byte of the PDU (passed by value)
53/// \return Value of the D/C field
54constexpr pdcp_dc_field pdcp_pdu_get_dc(uint8_t first_byte)
55{
56 return static_cast<pdcp_dc_field>((first_byte >> 7U) & 0x01U);
57}
58
59/// PDCP Control PDU type
60/// Ref: TS 38.323 Sec. 6.3.8
61enum class pdcp_control_pdu_type : unsigned {
62 status_report = 0b000, ///< PDCP status report
63 interspersed_rohc_feedback = 0b001, ///< Interspersed ROHC feedback
64 ehc_feedback = 0b010 ///< EHC feedback
65};
66constexpr uint16_t to_number(pdcp_control_pdu_type type)
67{
68 return static_cast<uint16_t>(type);
69}
70
71/// \brief Reads the CPT field from the first (header) byte of a PDCP control PDU
72/// \param first_byte First byte of the PDU (passed by value)
73/// \return Value of the CPT field
74constexpr pdcp_control_pdu_type pdcp_control_pdu_get_cpt(uint8_t first_byte)
75{
76 return static_cast<pdcp_control_pdu_type>((first_byte >> 4U) & 0x07U);
77}

79/// PDCP Data PDU header
80/// Ref: TS 38.323 Sec. 6.2.2
81struct pdcp_data_pdu_header {
82 uint32_t sn; ///< Sequence number
83};

85/// PDCP Control PDU header
86/// Ref: TS 38.323 Sec. 6.2.3
87struct pdcp_control_pdu_header {
88 pdcp_control_pdu_type cpt; ///< Control PDU type (control PDU only, ignored for data PDUs)
89};
