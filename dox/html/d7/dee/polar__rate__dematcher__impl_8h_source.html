33/// \brief Polar code rate dematching implementation.
34///
35/// It reverts the rate matching procedure described in TS38.212 Section 5.4.1.
36class polar_rate_dematcher_impl : public polar_rate_dematcher
37{
38private:
39 /// Internal buffer that contains both codeblock and rate-matched codeblock.
40 std::array<log_likelihood_ratio, polar_code::NMAX + polar_code::EMAX> y_e;
41 /// Pointer to the rate-matched part of the buffer.
42 log_likelihood_ratio* e = &y_e[polar_code::NMAX];
43
44public:
45 /// Default constructor.
46 polar_rate_dematcher_impl() = default;
47 /// Default destructor.
48 ~polar_rate_dematcher_impl() override = default;
49
50 // See interface for the documentation.
51 void rate_dematch(span<log_likelihood_ratio> output,
52 span<const log_likelihood_ratio> input,
53 const polar_code& code) override;
54};
