31/// \brief Absolute values.
32/// \param[in] value Input single-precision AVX register.
33/// \return A single-precision AVX register with the absolute value.
34inline __m256 abs_ps(__m256 value)
35{
36 const __m256 mask = _mm256_castsi256_ps(_mm256_set1_epi32(0x7fffffff));
37 return _mm256_and_ps(value, mask);
38}

40/// \brief Copy sign from a single-precision AVX register.
41/// \param[in] value0 Single-precision AVX register.
42/// \param[in] value1 Single-precision AVX register.
43/// \return A single-precision AVX register with the magnitudes of \c value0 and the signs of \c value1.
44/// \remark A zero in the second argument is considered as a positive number, following the convention of \c
45/// std::copysign.
46inline __m256 copysign_ps(__m256 value0, __m256 value1)
47{
48 __m256 abs_value0 = abs_ps(value0);
49 __m256 sign_value1 = _mm256_and_ps(value1, _mm256_set1_ps(-0.0));
50 return _mm256_or_ps(abs_value0, sign_value1);
51}

53/// \brief Clips the values of a single-precision AVX register.
54///
55/// The values greater than \c range_ceil or lower than \c range_floor are substituted by their corresponding range
56/// limits.
57///
58/// \param[in] value Input values.
59/// \param[in] range_ceil Ceiling values.
60/// \param[in] range_floor Floor values.
61/// \return A single-precision AVX register containing the clipped values.
62inline __m256 clip_ps(__m256 value, __m256 range_ceil, __m256 range_floor)
63{
64 value = _mm256_blendv_ps(value, range_ceil, _mm256_cmp_ps(value, range_ceil, _CMP_GT_OS));
65 value = _mm256_blendv_ps(value, range_floor, _mm256_cmp_ps(value, range_floor, _CMP_LT_OS));
66 return value;
67}

69/// \brief Clips the values of an AVX register carrying eight signed 32-bit integers.
70///
71/// The values greater than \c range_ceil or lower than \c range_floor are substituted by their corresponding range
72/// limits.
73///
74/// \param[in] value Input values.
75/// \param[in] range_ceil Ceiling values.
76/// \param[in] range_floor Floor values.
77/// \return An AVX register containing the clipped values.
78inline __m256i clip_epi32(__m256i value, __m256i range_ceil, __m256i range_floor)
79{
80 value = _mm256_blendv_epi8(value, range_ceil, _mm256_cmpgt_epi32(value, range_ceil));
81 value = _mm256_blendv_epi8(value, range_floor, _mm256_cmpgt_epi32(range_floor, value));
82 return value;
83}

85/// \brief Ensures that 32-bit integers are bounded, otherwise set them to zero.
86///
87/// \param[in] value Integers to be tested.
88/// \param[in] bound_up Upper bound.
89/// \param[in] bound_low Lower bound.
90/// \return The input integers with zeros in place of the out-of-bound values.
91inline __m256i check_bounds_epi32(__m256i value, __m256i bound_up, __m256i bound_low)
92{
93 __m256i ZEROS = _mm256_set1_epi8(0);
94 __m256i mask = _mm256_cmpgt_epi32(value, bound_up);
95 // Since the mask is set for all bytes of the int32_t integers, we can use the byte-wise blend.
96 value = _mm256_blendv_epi8(value, ZEROS, mask);
97
98 mask = _mm256_cmpgt_epi32(bound_low, value);
99 // Since the mask is set for all bytes of the int32_t integers, we can use the byte-wise blend.
100 value = _mm256_blendv_epi8(value, ZEROS, mask);
101
102 return value;
103}

105/// \brief Clips and quantizes four single-precision AVX registers, continuous log-likelihood ratio to the discrete
106/// representation of type \c log_likelihood_ratio in a single AVX register.
107///
108/// \param[in] value_0 Single-precision AVX register with the first eight continuous log-likelihood ratio.
109/// \param[in] value_1 Single-precision AVX register with the second eight continuous log-likelihood ratio.
110/// \param[in] value_2 Single-precision AVX register with the third eight continuous log-likelihood ratio.
111/// \param[in] value_3 Single-precision AVX register with the fourth eight continuous log-likelihood ratio.
112/// \param[in] range_limit The input value mapped to \ref log_likelihood_ratio::max().
113/// \return A quantized representation of the input values as \c log_likelihood_ratio quantity.
114/// \note The quantization in the range <tt>(-range_limit, range_limit)</tt> is [mid-tread
115/// uniform](https://en.wikipedia.org/wiki/Quantization_(signal_processing)#Mid-riser_and_mid-tread_uniform_quantizers),
116/// with quantization step <tt> range_limit / LLR_MAX </tt>.
117/// \note All values larger (in magnitude) than \c range_limit, will be clipped and mapped to
118/// <tt>&plusmn;LLR_MAX</tt>, depending on their sign.
119inline __m256i quantize_ps(__m256 value_0, __m256 value_1, __m256 value_2, __m256 value_3, float range_limit)
120{
121 // Scale.
122 __m256 SCALE = _mm256_set1_ps(static_cast<float>(log_likelihood_ratio::max().to_int()) / range_limit);
123 value_0 = _mm256_mul_ps(value_0, SCALE);
124 value_1 = _mm256_mul_ps(value_1, SCALE);
125 value_2 = _mm256_mul_ps(value_2, SCALE);
126 value_3 = _mm256_mul_ps(value_3, SCALE);
127
128 // Clip and round to the nearest integer.
129 __m256 RANGE_CEIL = _mm256_set1_ps(log_likelihood_ratio::max().to_int());
130 __m256 RANGE_FLOOR = _mm256_set1_ps(log_likelihood_ratio::min().to_int());
131 value_0 = _mm256_round_ps(clip_ps(value_0, RANGE_CEIL, RANGE_FLOOR), _MM_FROUND_NINT);
132 value_1 = _mm256_round_ps(clip_ps(value_1, RANGE_CEIL, RANGE_FLOOR), _MM_FROUND_NINT);
133 value_2 = _mm256_round_ps(clip_ps(value_2, RANGE_CEIL, RANGE_FLOOR), _MM_FROUND_NINT);
134 value_3 = _mm256_round_ps(clip_ps(value_3, RANGE_CEIL, RANGE_FLOOR), _MM_FROUND_NINT);
135
136 // Convert to 32 bit.
137 __m256i llr_i32_0 = _mm256_cvtps_epi32(value_0);
138 __m256i llr_i32_1 = _mm256_cvtps_epi32(value_1);
139 __m256i llr_i32_2 = _mm256_cvtps_epi32(value_2);
140 __m256i llr_i32_3 = _mm256_cvtps_epi32(value_3);
141
142 // Check bounds one more time: if value_X contains a NaN, its casting to int32_t is indeterminate - set it to zero.
143 __m256i BOUND_UP = _mm256_set1_epi32(log_likelihood_ratio::max().to_int());
144 __m256i BOUND_LOW = _mm256_set1_epi32(log_likelihood_ratio::min().to_int());
145 llr_i32_0 = check_bounds_epi32(llr_i32_0, BOUND_UP, BOUND_LOW);
146 llr_i32_1 = check_bounds_epi32(llr_i32_1, BOUND_UP, BOUND_LOW);
147 llr_i32_2 = check_bounds_epi32(llr_i32_2, BOUND_UP, BOUND_LOW);
148 llr_i32_3 = check_bounds_epi32(llr_i32_3, BOUND_UP, BOUND_LOW);
149
150 // Re-collocate SSE registers.
151 __m256i llr_i32_0_ = _mm256_permute2f128_si256(llr_i32_0, llr_i32_1, 0x20);
152 __m256i llr_i32_1_ = _mm256_permute2f128_si256(llr_i32_0, llr_i32_1, 0x31);
153 __m256i llr_i32_2_ = _mm256_permute2f128_si256(llr_i32_2, llr_i32_3, 0x20);
154 __m256i llr_i32_3_ = _mm256_permute2f128_si256(llr_i32_2, llr_i32_3, 0x31);
155
156 // Conversion to 16 bit.
157 __m256i llr_i16_0 = _mm256_packs_epi32(llr_i32_0_, llr_i32_1_);
158 __m256i llr_i16_1 = _mm256_packs_epi32(llr_i32_2_, llr_i32_3_);
159
160 // Re-collocate SSE registers.
161 __m256i llr_i16_0_ = _mm256_permute2f128_si256(llr_i16_0, llr_i16_1, 0x20);
162 __m256i llr_i16_1_ = _mm256_permute2f128_si256(llr_i16_0, llr_i16_1, 0x31);
163
164 // Conversion to 8 bit.
165 return _mm256_packs_epi16(llr_i16_0_, llr_i16_1_);
166}

168/// \brief Helper function to calculate an interval index from single-precision AVX register values.
169/// \param[in] value Input AVX register.
170/// \param[in] interval_width Interval width.
171/// \param[in] nof_intervals Number of intervals.
172/// \return An AVX register carrying eight signed 32-bit integers with the corresponding interval indexes.
173inline __m256i compute_interval_idx(__m256 value, float interval_width, int nof_intervals)
174{
175 // Scale.
176 value = _mm256_mul_ps(value, _mm256_set1_ps(1.0F / interval_width));
177
178 // Round to the lowest integer.
179 value = _mm256_round_ps(value, _MM_FROUND_FLOOR);
180
181 // Convert to int32.
182 __m256i idx = _mm256_cvtps_epi32(value);
183
184 // Add interval offset.
185 idx = _mm256_add_epi32(idx, _mm256_set1_epi32(nof_intervals / 2));
186
187 // Clip index.
188 idx = clip_epi32(idx, _mm256_set1_epi32(nof_intervals - 1), _mm256_setzero_si256());
189
190 // Clip integer and return.
191 return idx;
192}

194/// \brief Get values from a look-up table.
195/// \param[in] table Look-up table containing eight single-precision values.
196/// \param[in] indexes AVX register containing eight indexes.
197/// \return A single-precision AVX register containing the eight values corresponding to the indexes.
198inline __m256 look_up_table(const std::array<float, 8>& table, __m256i indexes)
199{
200 return _mm256_permutevar8x32_ps(_mm256_loadu_ps(table.data()), indexes);
201}
202
203/// \brief Get values from a look-up table.
204/// \param[in] table Look-up table containing sixteen single-precision values.
205/// \param[in] indexes AVX register containing eight indexes.
206/// \return A single-precision AVX register containing the eight values corresponding to the indexes.
207inline __m256 look_up_table(const std::array<float, 16>& table, __m256i indexes)
208{
209 // Get lower part of the table.
210 __m256 lower = _mm256_permutevar8x32_ps(_mm256_loadu_ps(table.data()), indexes);
211
212 // Get upper part of the table.
213 __m256 upper = _mm256_permutevar8x32_ps(_mm256_loadu_ps(table.data() + 8), indexes);
214
215 // Generate mask. Set to true if the MSB of the index is greater than zero.
216 __m256i mask = _mm256_cmpgt_epi32(indexes, _mm256_set1_epi32(0b111));
217
218 // Select upper or lower value.
219 return _mm256_blendv_ps(lower, upper, (__m256)mask);
220}

222/// \brief Applies an interval function to a series of values.
223/// \tparam Table Look-up table type. All tables mut be of the same type.
224/// \param[in] value Single-precision AVX register with eight input values.
225/// \param[in] rcp_noise Single-precision AVX register with the reciprocal noise corresponding to the values.
226/// \param[in] nof_intervals Number of intervals.
227/// \param[in] interval_width Interval width to quantify the interval indexes.
228/// \param[in] slopes Table with the slope of each interval.
229/// \param[in] intercepts Table with the interception points of each interval.
230/// \return A single-precision AVX register containing the results of the interval function.
231/// \remark The number of intervals must be lower than or equal to \c Table size.
232template <typename Table>
233inline __m256 interval_function(__m256 value,
234 __m256 rcp_noise,
235 float interval_width,
236 unsigned nof_intervals,
237 const Table& slopes,
238 const Table& intercepts)
239{
240 __m256i interval_index = mm256::compute_interval_idx(value, interval_width, nof_intervals);
241
242 __m256 slope = mm256::look_up_table(slopes, interval_index);
243 __m256 intercept = mm256::look_up_table(intercepts, interval_index);
244
245 return _mm256_mul_ps(_mm256_add_ps(_mm256_mul_ps(slope, value), intercept), rcp_noise);
246}

248/// \brief Safe division.
249///
250/// \return <tt>dividend / divisor</tt> if \c divisor is greater than zero, \c 0 otherwise.
251inline __m256 safe_div(__m256 dividend, __m256 divisor)
252{
253 static const __m256 all_zero = _mm256_setzero_ps();
254 // _CMP_GT_OQ: compare greater than, ordered (nan is false) and quite (no exceptions raised).
255#if defined(__AVX512F__) && defined(__AVX512VL__)
256 __mmask8 mask = _mm256_cmp_ps_mask(divisor, all_zero, _CMP_GT_OQ);
257 return _mm256_maskz_div_ps(mask, dividend, divisor);
258#else
259 __m256 mask = _mm256_cmp_ps(divisor, all_zero, _CMP_GT_OQ);
260 __m256 result = _mm256_div_ps(dividend, divisor);
261 return _mm256_blendv_ps(all_zero, result, mask);
262#endif
263}
