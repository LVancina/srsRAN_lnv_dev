71 /// Maximum processing block size in bits.

73 /// Data type for representing an RE mask within an OFDM symbol.
74 using re_symbol_mask_type = bounded_bitset<MAX_RB * NRE>;

76 /// \brief Gets channel data Resource Elements from the resource grid.
77 ///
78 /// Extracts the PUSCH data RE's from the provided resource grid. The DM-RS symbols are skipped. The extracted RE's
79 /// are arranged in two dimensions, i.e., resource element and receive antenna port, as the channel equalizer expects.
80 ///
81 /// \param[out] data_re PUSCH channel data symbols, organized by receive antenna port.
82 /// \param[in] grid Resource grid for the current slot.
83 /// \param[in] i_symbol OFDM symbol index relative to the beginning of the slot.
84 /// \param[in] init_subc Initial subcarrier index relative to Point A.
85 /// \param[in] re_mask Resource element mask, it selects the RE elements to extract.
86 /// \param[in] rx_ports Receive ports.
87 static void get_ch_data_re(channel_equalizer::re_list& data_re,
88 const resource_grid_reader& grid,
89 unsigned i_symbol,
90 unsigned init_subc,
91 const re_symbol_mask_type& re_mask,
92 const static_vector<uint8_t, MAX_PORTS>& rx_ports)
93 {
94 // Extract RE for each port and symbol.
95 for (unsigned i_port = 0, i_port_end = rx_ports.size(); i_port != i_port_end; ++i_port) {
96 // Get a view of the port data RE.
97 span<cf_t> re_port_buffer = data_re.get_view<>({i_port});
98
99 // Copy grid data resource elements into the buffer.
100 re_port_buffer = grid.get(re_port_buffer, rx_ports[i_port], i_symbol, init_subc, re_mask);
101
102 // Verify buffer size.
103 srsran_assert(
104 re_port_buffer.empty(), "Invalid number of RE read from the grid. {} RE are missing.", re_port_buffer.size());
105 }
106 }
107
108 /// \brief Gets channel data estimates.
109 ///
110 /// Extracts the channel estimation coefficients corresponding to the PUSCH data Resource Elements from the channel
111 /// estimate. The DM-RS symbols are skipped. The extracted channel coefficients are arranged in three dimensions,
112 /// i.e., resource element, receive port and transmit layer, as the channel equalizer expects.
113 ///
114 /// \param[out] data_estimates Channel estimates of the data symbols, organized by receive port and transmit layer.
115 /// \param[in] channel_estimate Channel estimation object.
116 /// \param[in] i_symbol OFDM symbol index relative to the beginning of the slot.
117 /// \param[in] init_subc Initial subcarrier index relative to Point A.
118 /// \param[in] re_mask Resource element mask, it selects the RE elements to extract.
119 /// \param[in] rx_ports Receive ports.
120 static void get_ch_data_estimates(channel_equalizer::ch_est_list& data_estimates,
121 const channel_estimate& channel_estimate,
122 unsigned i_symbol,
123 unsigned init_subc,
124 const re_symbol_mask_type& re_mask,
125 const static_vector<uint8_t, MAX_PORTS>& rx_ports)
126 {
127 // Extract data RE coefficients from the channel estimation.
128 for (unsigned i_layer = 0, i_layer_end = channel_estimate.size().nof_tx_layers; i_layer != i_layer_end; ++i_layer) {
129 for (unsigned i_port = 0, i_port_end = rx_ports.size(); i_port != i_port_end; ++i_port) {
130 // Get a view of the channel estimates buffer for a single Rx port.
131 span<cf_t> ch_port_buffer = data_estimates.get_view({i_port, i_layer});
132
133 // View of the channel estimation for an OFDM symbol.
134 span<const cf_t> symbol_estimates = channel_estimate.get_symbol_ch_estimate(i_symbol, i_port, i_layer);
135
136 // Get view of the selected area of the grid.
137 symbol_estimates = symbol_estimates.subspan(init_subc, re_mask.size());
138
139 if (re_mask.is_contiguous()) {
140 int begin = re_mask.find_lowest();
141 int end = re_mask.find_highest();
142
143 srsran_assert(begin >= 0, "Invalid mask.");
144 srsran_assert(begin <= end, "Invalid mask.");
145
146 symbol_estimates = symbol_estimates.subspan(begin, end - begin + 1);
147
148 srsvec::copy(ch_port_buffer, symbol_estimates);
149 continue;
150 }
151
152 // Skip DM-RS estimates.
153 re_mask.for_each(0, re_mask.size(), [&symbol_estimates, &ch_port_buffer](unsigned i_re) {
154 // Copy RE.
155 ch_port_buffer.front() = symbol_estimates[i_re];
156
157 // Advance buffer.
158 ch_port_buffer = ch_port_buffer.last(ch_port_buffer.size() - 1);
159 });
160
161 // Verify buffer size.
162 srsran_assert(ch_port_buffer.empty(),
163 "Invalid number of RE read from the channel estimates. {} RE are missing.",
164 ch_port_buffer.size());
165 }
166 }
167 }
