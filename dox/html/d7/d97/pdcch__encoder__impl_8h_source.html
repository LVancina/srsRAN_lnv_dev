37/// \brief PDCCH encoder, as per TS38.212 Section 7.3.
38///
39/// For the PHY Downlink Control CHannel (PDCCH) it carries out CRC attachment,
40/// Polar encoding and rate matching.
41class pdcch_encoder_impl : public pdcch_encoder
42{
43private:
44 /// Defines CRC length in bits.
45 static constexpr unsigned CRC_LEN = 24;
46
47 /// Defines RNTI length in bits.
48 static constexpr unsigned RNTI_LEN = 16;
49
50 /// Defines the maximum payload size in bits, including 24 CRC bits.
51 static constexpr unsigned MAX_K = pdcch_constants::MAX_DCI_PAYLOAD_SIZE + CRC_LEN;
52
53 std::unique_ptr<crc_calculator> crc24c;
54 std::unique_ptr<polar_interleaver> interleaver;
55 std::unique_ptr<polar_allocator> alloc;
56 std::unique_ptr<polar_code> code;
57 std::unique_ptr<polar_encoder> encoder;
58 std::unique_ptr<polar_rate_matcher> rm;
59
60 /// Temporary payload with CRC appended.
61 std::array<uint8_t, MAX_K + CRC_LEN> temp_c;
62 /// Temporary encoded data prior to rate matching.
63 std::array<uint8_t, polar_code::NMAX> temp_d;
64
65 /// \brief CRC attachment
66 ///
67 /// \param[out] c Payload with CRC attached
68 /// \param[in] a Payload
69 /// \param[in] rnti RNTI used for CRC bits scrambling
70 ///
71 /// \remark CRC attachment follows TS 38.312 section 7.3.2
72 void crc_attach(span<uint8_t>& c, span<const uint8_t> a, unsigned rnti);
73
74 /// \brief Payload encoding
75 ///
76 /// \param[out] d Data after encoding
77 /// \param[in] c Payload after CRC attachment
78 ///
79 /// \remark Channel codding follows TS 38.312 section 7.3.3
80 void channel_coding(span<uint8_t> d, span<const uint8_t> c);
81
82 /// \brief Rate matching of the encoded data
83 ///
84 /// \param[out] f Data after rate matching
85 /// \param[in] d Data after encoding
86 ///
87 /// \remark Rate matching follows TS 38.312 section 7.3.4
88 void rate_matching(span<uint8_t> f, span<const uint8_t> d);
89
90public:
91 pdcch_encoder_impl(std::unique_ptr<crc_calculator> crc24c_,
92 std::unique_ptr<polar_interleaver> interleaver_,
93 std::unique_ptr<polar_allocator> alloc_,
94 std::unique_ptr<polar_code> code_,
95 std::unique_ptr<polar_encoder> encoder_,
96 std::unique_ptr<polar_rate_matcher> rm_) :
97 crc24c(std::move(crc24c_)),
98 interleaver(std::move(interleaver_)),
99 alloc(std::move(alloc_)),
100 code(std::move(code_)),
101 encoder(std::move(encoder_)),
102 rm(std::move(rm_))
103 {
104 report_fatal_error_if_not(crc24c, "Invalid CRC calculator.");
105 report_fatal_error_if_not(crc24c->get_generator_poly() == crc_generator_poly::CRC24C, "Invalid CRC calculator.");
106 report_fatal_error_if_not(interleaver, "Invalid polar interleaver.");
107 report_fatal_error_if_not(alloc, "Invalid polar allocator.");
108 report_fatal_error_if_not(code, "Invalid polar code.");
109 report_fatal_error_if_not(encoder, "Invalid polar encoder.");
110 report_fatal_error_if_not(rm, "Invalid polar rate matcher.");
111 }
112
113 // See interface for the documentation.
114 void encode(span<uint8_t> encoded, span<const uint8_t> data, const config_t& config) override;
115};
