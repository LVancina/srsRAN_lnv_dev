80// Struct that contains all fields required to update the UP config based on an incoming
81// PDU sessions resource setup request over NGAP. This config is then used to:
82// * Initiate or modifiy the CU-UPs bearer context over E1AP
83// * Modify the DU's UE context over F1AP
84// * Modify the CU-UPs bearer context over E1AP (update TEIDs, etc)
85// * Modify the UE's configuration over RRC signaling
86// For PDU sessions to be setup the entire session context is included in the struct as this has been allocated by UP
87// resource manager. For removal of PDU sessions or DRBs only the respective identifiers are included.
88struct up_config_update {
89 bool initial_context_creation = true; // True if this is the first PDU session to be created.
90 bool context_removal_required =
91 false; // Set to true if all UP resources will be removed and the context can be deleted.
92 std::map<pdu_session_id_t, up_pdu_session_context_update>
93 pdu_sessions_to_setup_list; // List of PDU sessions to be added.
94 std::map<pdu_session_id_t, up_pdu_session_context_update>
95 pdu_sessions_to_modify_list; // List of PDU sessions to be modified.
96 std::vector<pdu_session_id_t> pdu_sessions_to_remove_list; // List of PDU sessions to be removed.
97 std::vector<pdu_session_id_t> pdu_sessions_failed_to_modify_list; // List of PDU sessions that failed to be modified.
98 std::vector<drb_id_t> drb_to_remove_list; // List of DRBs to be removed.
99};
100
101// Response given back to the UP resource manager containing the full context
102// that could be setup.
103struct up_config_update_result {
104 std::vector<up_pdu_session_context_update> pdu_sessions_added_list; // List of sessions that have been added.
105 std::vector<up_pdu_session_context_update> pdu_sessions_modified_list; // List of sessions that have been modified.
106 std::vector<pdu_session_id_t> pdu_sessions_removed_list; // List of sessions that have been removed.
107};
