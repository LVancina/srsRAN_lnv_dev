35/// \brief Uplink processor implementation with a task dispatcher per channel.
36///
37/// In this implementation, every task of a physical channel is processed by the same executor. When a task finishes,
38/// the results are notified by the upper_phy_rx_results_notifier interface.
39class uplink_processor_task_dispatcher : public uplink_processor
40{
41public:
42 uplink_processor_task_dispatcher(std::unique_ptr<uplink_processor> processor_,
43 task_executor& pucch_executor_,
44 task_executor& pusch_executor_,
45 task_executor& prach_executor_) :
46 processor(std::move(processor_)),
47 pucch_executor(pucch_executor_),
48 pusch_executor(pusch_executor_),
49 prach_executor(prach_executor_)
50 {
51 srsran_assert(processor, "A valid uplink processor must be provided");
52 }
53
54 // See interface for documentation.
55 void process_prach(upper_phy_rx_results_notifier& notifier,
56 const prach_buffer& buffer,
57 const prach_buffer_context& context) override
58 {
59 bool success = prach_executor.execute(
60 [&notifier, &buffer, context, this]() { processor->process_prach(notifier, buffer, context); });
61
62 if (!success) {
63 logger.warning(context.slot.sfn(), context.slot.slot_index(), "Failed to execute PRACH. Ignoring detection.");
64 }
65 }
66
67 // See interface for documentation.
68 void process_pusch(span<uint8_t> data,
69 unique_rx_buffer rm_buffer,
70 upper_phy_rx_results_notifier& notifier,
71 const resource_grid_reader& grid,
72 const uplink_processor::pusch_pdu& pdu) override
73 {
74 bool success =
75 pusch_executor.execute([data, rm_buffer2 = std::move(rm_buffer), &notifier, &grid, pdu, this]() mutable {
76 processor->process_pusch(data, std::move(rm_buffer2), notifier, grid, pdu);
77 });
78
79 if (!success) {
80 logger.warning(pdu.pdu.slot.sfn(), pdu.pdu.slot.slot_index(), "Failed to execute PUSCH. Ignoring processing.");
81 }
82 }
83
84 // See interface for documentation.
85 void process_pucch(upper_phy_rx_results_notifier& notifier,
86 const resource_grid_reader& grid,
87 const pucch_pdu& pdu) override
88 {
89 bool success =
90 pucch_executor.execute([&notifier, &grid, pdu, this]() { processor->process_pucch(notifier, grid, pdu); });
91 if (!success) {
92 logger.warning(
93 pdu.context.slot.sfn(), pdu.context.slot.slot_index(), "Failed to execute PUCCH. Ignoring processing.");
94 }
95 }
96
97private:
98 /// Default logger.
99 srslog::basic_logger& logger = srslog::fetch_basic_logger("PHY");
100 /// Uplink processor detector.
101 std::unique_ptr<uplink_processor> processor;
102 /// Executor for the PUCCH tasks generated within this uplink processor.
103 task_executor& pucch_executor;
104 /// Executor for the PUSCH tasks generated within this uplink processor.
105 task_executor& pusch_executor;
106 /// Executor for the PRACH tasks generated within this uplink processor.
107 task_executor& prach_executor;
108};
