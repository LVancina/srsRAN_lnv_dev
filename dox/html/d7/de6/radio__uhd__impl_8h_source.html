35/// Describes a radio session based on UHD that also implements the management and data plane functions.
36class radio_session_uhd_impl : public radio_session, private radio_management_plane
37{
38private:
39 /// Wait at most 1s for external clock locking.
40 static constexpr int CLOCK_TIMEOUT_MS = 1000;
41 /// Enumerates possible UHD session states.
42 enum class states { UNINITIALIZED = 0, SUCCESSFUL_INIT, STOP };
43 /// Maps ports to stream and channel indexes.
44 using port_to_stream_channel = std::pair<unsigned, unsigned>;
45 /// Indicates the current state.
46 std::atomic<states> state = {states::UNINITIALIZED};
47 /// Wraps the UHD device functions.
48 radio_uhd_device device;
49 /// Indexes the transmitter port indexes into stream and channel index as first and second.
50 static_vector<port_to_stream_channel, RADIO_MAX_NOF_PORTS> tx_port_map;
51 /// Indexes the receiver port indexes into stream and channel index as first and second.
52 static_vector<port_to_stream_channel, RADIO_MAX_NOF_PORTS> rx_port_map;
53 /// Baseband gateways.
54 std::vector<std::unique_ptr<radio_uhd_baseband_gateway>> bb_gateways;
55 double actual_sampling_rate_Hz;
56 /// Protects the stream start.
57 std::mutex stream_start_mutex;
58 /// Indicates if the reception streams require start.
59 bool stream_start_required = true;
60 /// Asynchronous executor.
61 task_executor& async_executor;
62 /// Event notifier.
63 radio_notification_handler& notifier;
64
65 /// \brief Set the synchronization time to GPS mode.
66 /// \return True if no exception is caught. Otherwise false.
67 bool set_time_to_gps_time();
68
69 /// \brief Waits for a sensor to be locked.
70 /// \param[in] sensor_name Indicates the sensor name.
71 /// \param[in] is_mboard Indicates if the sensor is from the motherboard or daughterboard.
72 /// \param[in] timeout Indicates the amount of time to wait.
73 /// \return True if sensor is found and locked. Otherwise false.
74 bool wait_sensor_locked(const std::string& sensor_name, bool is_mboard, int timeout);
75
76 /// \brief Set transmission gain from the class itself.
77 /// \param[in] port_idx Indicates the port index.
78 /// \param[in] gain_dB Indicates the gain value.
79 /// \return True if the port index and gain value are valid, and no exception is caught. Otherwise false.
80 bool set_tx_gain_unprotected(unsigned port_idx, double gain_dB);
81
82 /// \brief Set reception gain from the class itself.
83 /// \param[in] port_idx Indicates the port index.
84 /// \param[in] gain_dB Indicates the gain value.
85 /// \return True if the port index and gain value are valid, and no exception is caught. Otherwise false.
86 bool set_rx_gain_unprotected(unsigned port_idx, double gain_dB);
87
88 /// \brief Set transmission frequency.
89 /// \param[in] port_idx Indicates the port index.
90 /// \param[in] frequency Provides the frequency tuning parameters
91 /// \return True if the port index and frequency value are valid, and no exception is caught. Otherwise false.
92 bool set_tx_freq(unsigned port_idx, radio_configuration::lo_frequency frequency);
93
94 /// \brief Set reception frequency.
95 /// \param[in] port_idx Indicates the port index.
96 /// \param[in] frequency Provides the frequency tuning parameters
97 /// \return True if the port index and frequency value are valid, and no exception is caught. Otherwise false.
98 bool set_rx_freq(unsigned port_idx, radio_configuration::lo_frequency frequency);
99
100 /// \brief Start receive streams.
101 /// \param[in] init_time Time in which the stream shall start.
102 /// \return True if no exception is caught. Otherwise false.
103 bool start_rx_stream(baseband_gateway_timestamp init_time);
104
105public:
106 baseband_gateway_timestamp read_current_time() override;
107
108public:
109 /// Constructs a radio session based on UHD.
110 radio_session_uhd_impl(const radio_configuration::radio& radio_config,
111 task_executor& async_executor,
112 radio_notification_handler& notifier_);
113
114 /// \brief Indicates that the radio session was initialized succesfully.
115 /// \return True if no exception is caught during initialization. Otherwise false.
116 bool is_successful() const { return (state != states::UNINITIALIZED); }
117
118 // See interface for documentation.
119 radio_management_plane& get_management_plane() override { return *this; }
120
121 // See interface for documentation.
122 baseband_gateway& get_baseband_gateway(unsigned stream_id) override
123 {
124 srsran_assert(stream_id < bb_gateways.size(),
125 "Stream identifier (i.e., {}) exceeds the number of baseband gateways (i.e., {})",
126 stream_id,
127 bb_gateways.size());
128 return *bb_gateways[stream_id];
129 }
130
131 // See interface for documentation.
132 void start(baseband_gateway_timestamp init_time) override;
133
134 // See interface for documentation.
135 void stop() override;
136
137 // See interface for documentation.
138 bool set_tx_gain(unsigned port_idx, double gain_dB) override { return set_tx_gain_unprotected(port_idx, gain_dB); }
139
140 // See interface for documentation.
141 bool set_rx_gain(unsigned port_idx, double gain_dB) override { return set_rx_gain_unprotected(port_idx, gain_dB); }
142};
