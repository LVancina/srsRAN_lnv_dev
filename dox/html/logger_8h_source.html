35template <typename T, typename Enum>
36class logger_impl : public T
37{
38 static_assert(std::is_enum<Enum>::value, "Expected enum type");
39
40 using enum_base_type = typename std::underlying_type<Enum>::type;
41 static constexpr unsigned size = static_cast<enum_base_type>(Enum::LAST) - 1;
42
43public:
44 template <typename... Args>
45 explicit logger_impl(std::string id, Args&&... args) :
46 T{std::forward<Args>(args)...}, logger_id(std::move(id)), channels{&args...}
47 {
48 static_assert(sizeof...(args) == size, "Number of levels in enum does not match number of log channels");
49 }
50
51 logger_impl(const logger_impl& other) = delete;
52 logger_impl& operator=(const logger_impl& other) = delete;
53
55 const std::string& id() const { return logger_id; }
56
58 void set_level(Enum lvl)
59 {
60 detail::scoped_lock lock(m);
61
62 // Disable all channels ...
63 if (lvl == Enum::none) {
64 for (unsigned i = 0, e = channels.size(); i != e; ++i) {
65 channels[i]->set_enabled(false);
66 }
67 return;
68 }
69
70 // ... otherwise do it selectively.
71 for (unsigned i = 0, e = channels.size(); i != e; ++i) {
72 channels[i]->set_enabled(static_cast<Enum>(i + 1) <= lvl);
73 }
74 }
75
77 void set_context(uint32_t a, uint32_t b)
78 {
79 detail::scoped_lock lock(m);
80 for (auto channel : channels) {
81 channel->set_context(a, b);
82 }
83 }
84
87 void set_hex_dump_max_size(int x)
88 {
89 detail::scoped_lock lock(m);
90 for (auto channel : channels) {
91 channel->set_hex_dump_max_size(x);
92 }
93 }
94
95private:
97 friend bool operator<=(Enum lhs, Enum rhs)
98 {
99 return static_cast<enum_base_type>(lhs) <= static_cast<enum_base_type>(rhs);
100 }
101
102private:
103 const std::string logger_id;
104 const std::array<log_channel*, size> channels;
105 mutable detail::mutex m;
106};

153enum class basic_levels { none, error, warning, info, debug, LAST };
154struct basic_logger_channels {
155 log_channel& error;

160using basic_logger = build_logger_type<basic_logger_channels, basic_levels>;

163inline basic_levels str_to_basic_level(std::string s)
164{
165 std::transform(s.begin(), s.end(), s.begin(), ::toupper);
166
167 if ("NONE" == s) {
168 return basic_levels::none;
169 }
170 if ("ERROR" == s) {
171 return basic_levels::error;
172 }
173 if ("WARNING" == s) {
174 return basic_levels::warning;
175 }
176 if ("INFO" == s) {
177 return basic_levels::info;
178 }
179 if ("DEBUG" == s) {
180 return basic_levels::debug;
181 }
182 return basic_levels::none;
183}
184
186inline const char* basic_level_to_string(basic_levels level)
187{
188 switch (level) {
189 case basic_levels::debug:
190 return "DEBUG";
191 case basic_levels::info:
192 return "INFO";
193 case basic_levels::warning:
194 return "WARNING";
195 case basic_levels::error:
196 return "ERROR";
197 default:
198 break;
199 }
200 return "NONE";
201}
