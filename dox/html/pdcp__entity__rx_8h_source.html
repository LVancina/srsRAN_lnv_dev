101 /*
102 * Security configuration
103 */
104 void configure_security(security::sec_128_as_config sec_cfg_) final
105 {
106 srsran_assert((is_srb() && sec_cfg_.domain == security::sec_domain::rrc) ||
107 (is_drb() && sec_cfg_.domain == security::sec_domain::up),
108 "Invalid sec_domain={} for {} in {}",
109 sec_cfg.domain,
110 rb_type,
111 rb_id);
112 // The 'NULL' integrity protection algorithm (nia0) is used only for SRBs and for the UE in limited service mode,
113 // see TS 33.501 [11] and when used for SRBs, integrity protection is disabled for DRBs. In case the â€²NULL'
114 // integrity protection algorithm is used, 'NULL' ciphering algorithm is also used.
115 // Ref: TS 38.331 Sec. 5.3.1.2
116 if ((sec_cfg_.integ_algo == security::integrity_algorithm::nia0) &&
117 (is_drb() || (is_srb() && sec_cfg_.cipher_algo != security::ciphering_algorithm::nea0))) {
118 logger.log_error(
119 "Integrity algorithm NIA0 is only permitted for SRBs configured with NEA0. is_srb={} NIA{} NEA{}",
120 is_srb(),
121 sec_cfg_.integ_algo,
122 sec_cfg_.cipher_algo);
123 }
124
125 sec_cfg = sec_cfg_;
126 logger.log_info(
127 "Security configured: NIA{} NEA{} domain={}", sec_cfg.integ_algo, sec_cfg.cipher_algo, sec_cfg.domain);
128 if (sec_cfg.k_128_int.has_value()) {
129 logger.log_info(sec_cfg.k_128_int.value().data(), 16, "128 K_int");
130 }
131 logger.log_info(sec_cfg.k_128_enc.data(), 16, "128 K_enc");
132 }
133
134 void set_integrity_protection(security::integrity_enabled integrity_enabled_) final
135 {
136 if (integrity_enabled_ == security::integrity_enabled::on) {
137 if (!sec_cfg.k_128_int.has_value()) {
138 logger.log_error("Cannot enable integrity protection: Integrity key is not configured.");
139 return;
140 }
141 if (!sec_cfg.integ_algo.has_value()) {
142 logger.log_error("Cannot enable integrity protection: Integrity algorithm is not configured.");
143 return;
144 }
145 }
146 integrity_enabled = integrity_enabled_;
147 logger.log_info("Set integrity_enabled={}", integrity_enabled);
148 }
149 void set_ciphering(security::ciphering_enabled ciphering_enabled_) final
150 {
151 ciphering_enabled = ciphering_enabled_;
152 logger.log_info("Set ciphering_enabled={}", ciphering_enabled);
153 }
