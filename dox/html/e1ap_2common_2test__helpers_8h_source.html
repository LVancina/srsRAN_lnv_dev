69 srs_cu_up::e1ap_bearer_context_setup_response
70 on_bearer_context_setup_request_received(const srs_cu_up::e1ap_bearer_context_setup_request& msg) override
71 {
72 logger.info("Received BearerContextSetupRequest");
73 last_bearer_context_setup_request.security_info.security_algorithm = msg.security_info.security_algorithm;
74 last_bearer_context_setup_request.security_info.up_security_key.encryption_key =
75 msg.security_info.up_security_key.encryption_key.copy();
76 last_bearer_context_setup_request.security_info.up_security_key.integrity_protection_key =
77 msg.security_info.up_security_key.integrity_protection_key.copy();
78 last_bearer_context_setup_request.ue_dl_aggregate_maximum_bit_rate = msg.ue_dl_aggregate_maximum_bit_rate;
79 last_bearer_context_setup_request.serving_plmn = msg.serving_plmn;
80 last_bearer_context_setup_request.activity_notif_level = msg.activity_notif_level;
81 last_bearer_context_setup_request.pdu_session_res_to_setup_list = msg.pdu_session_res_to_setup_list;
82
83 srs_cu_up::e1ap_bearer_context_setup_response response = {};
84 response.ue_index = ue_index;
85 response.success = true;
86 for (const auto& request_setup_item : msg.pdu_session_res_to_setup_list) {
87 e1ap_pdu_session_resource_setup_modification_item response_setup_item;
88 response_setup_item.pdu_session_id = request_setup_item.pdu_session_id;
89 response_setup_item.ng_dl_up_tnl_info.gtp_teid = int_to_gtpu_teid(1);
90 response_setup_item.ng_dl_up_tnl_info.tp_address = transport_layer_address{"127.0.0.1"};
91
92 for (const auto& request_drb_item : request_setup_item.drb_to_setup_list_ng_ran) {
93 e1ap_drb_setup_item_ng_ran response_drb_item;
94 response_drb_item.drb_id = request_drb_item.drb_id;
95
96 response_setup_item.drb_setup_list_ng_ran.emplace(request_drb_item.drb_id, response_drb_item);
97 }
98
99 response.pdu_session_resource_setup_list.emplace(request_setup_item.pdu_session_id, response_setup_item);
100 }
101
102 return response;
103 }

105 srs_cu_up::e1ap_bearer_context_modification_response on_bearer_context_modification_request_received(
106 const srs_cu_up::e1ap_bearer_context_modification_request& msg) override
107 {
108 logger.info("Received BearerContextModificationRequest");
109
110 // copy message
111 if (last_bearer_context_modification_request.security_info.has_value()) {
112 e1ap_security_info security_info = {};
113 security_info.security_algorithm = msg.security_info.value().security_algorithm;
114 security_info.up_security_key.encryption_key = msg.security_info.value().up_security_key.encryption_key.copy();
115 security_info.up_security_key.integrity_protection_key =
116 msg.security_info.value().up_security_key.integrity_protection_key.copy();
117 last_bearer_context_modification_request.security_info = security_info;
118 }
119 if (msg.ue_dl_aggr_max_bit_rate.has_value()) {
120 last_bearer_context_modification_request.ue_dl_aggr_max_bit_rate = msg.ue_dl_aggr_max_bit_rate.value();
121 }
122 if (msg.ue_dl_max_integrity_protected_data_rate.has_value()) {
123 last_bearer_context_modification_request.ue_dl_max_integrity_protected_data_rate =
124 msg.ue_dl_max_integrity_protected_data_rate.value();
125 }
126 if (msg.bearer_context_status_change.has_value()) {
127 last_bearer_context_modification_request.bearer_context_status_change = msg.bearer_context_status_change.value();
128 }
129 if (msg.new_ul_tnl_info_required.has_value()) {
130 last_bearer_context_modification_request.new_ul_tnl_info_required = msg.new_ul_tnl_info_required.value();
131 }
132 if (msg.ue_inactivity_timer.has_value()) {
133 last_bearer_context_modification_request.ue_inactivity_timer = msg.ue_inactivity_timer.value();
134 }
135 if (msg.data_discard_required.has_value()) {
136 last_bearer_context_modification_request.data_discard_required = msg.data_discard_required.value();
137 }
138 if (msg.ng_ran_bearer_context_mod_request.has_value()) {
139 last_bearer_context_modification_request.ng_ran_bearer_context_mod_request =
140 msg.ng_ran_bearer_context_mod_request.value();
141 }
142 if (msg.ran_ue_id.has_value()) {
143 last_bearer_context_modification_request.ran_ue_id = msg.ran_ue_id.value();
144 }
145 if (msg.gnb_du_id.has_value()) {
146 last_bearer_context_modification_request.gnb_du_id = msg.gnb_du_id.value();
147 }
148 if (msg.activity_notif_level.has_value()) {
149 last_bearer_context_modification_request.activity_notif_level = msg.activity_notif_level.value();
150 }
151
152 srs_cu_up::e1ap_bearer_context_modification_response response = {};
153 response.ue_index = ue_index;
154 response.success = true;
155
156 return response;
157 }
