56class ticking_ring_buffer_pool
57{
58public:
59 using tick_t = unsigned;
60
61private:
62 struct overflow_checker {
63 std::vector<unsigned> n_alloc_per_tick;
64 size_t pool_size;
65 size_t max_bytes_per_tick;
66 tick_t tic_wraparound;
67
68 bool first_tick = true;
69 size_t n_alloc = 0;
70 unsigned ring_index = 0;
71 tick_t last_tick_index = 0;
72
73 explicit overflow_checker(size_t nof_bytes_per_tick, size_t ring_of_slots_size, tick_t tic_wraparound_) :
74 n_alloc_per_tick(ring_of_slots_size, 0),
75 pool_size(nof_bytes_per_tick * ring_of_slots_size),
76 max_bytes_per_tick(nof_bytes_per_tick),
77 tic_wraparound(tic_wraparound_)
78 {
79 }
80
82 void register_alloc(size_t sz) noexcept
83 {
84 if (n_alloc + sz > pool_size) {
85 report_fatal_error("POOL: Slot ring buffer overflown");
86 }
87 if (sz > max_bytes_per_tick) {
88 report_fatal_error(
89 "POOL: Trying to allocate too many bytes in one single tick ({} > {})", sz, max_bytes_per_tick);
90 }
91 n_alloc_per_tick[ring_index] += sz;
92 n_alloc += sz;
93 }
94
95 void tick(tick_t tic_index) noexcept
96 {
97 srsran_assert(first_tick or tic_index != last_tick_index,
98 "This function cannot be called multiple times for the same tick");
99 if (SRSRAN_UNLIKELY(first_tick)) {
100 // first tick call.
101 last_tick_index = tic_index - 1;
102 first_tick = false;
103 } else if (SRSRAN_UNLIKELY(last_tick_index > tic_index)) {
104 // wrap-around case.
105 last_tick_index = -(tic_wraparound - last_tick_index);
106 }
107 for (; last_tick_index != tic_index; ++last_tick_index) {
108 ring_index = (ring_index + 1) % n_alloc_per_tick.size();
109 n_alloc -= n_alloc_per_tick[ring_index];
110 n_alloc_per_tick[ring_index] = 0;
111 }
112 }
113 };
114
115 struct no_overflow_checker {
116 explicit no_overflow_checker(size_t nof_bytes_per_tick, size_t ring_of_slots_size, tick_t ) {}
117
118 void register_alloc(size_t sz) noexcept {}
119 void tick(unsigned tic_index) noexcept {}
120 };
121
122public:
123 explicit ticking_ring_buffer_pool(size_t nof_bytes_per_tick,
124 size_t buffer_duration_in_ticks,
125 tick_t tic_wraparound_) :
126 pool(nof_bytes_per_tick * buffer_duration_in_ticks),
127 overflow_detector(nof_bytes_per_tick, buffer_duration_in_ticks, tic_wraparound_)
128 {
129 }
130
134 span<uint8_t> allocate_buffer(size_t sz) noexcept
135 {
136 overflow_detector.register_alloc(sz);
137 return pool.allocate_buffer(sz);
138 }
139
141 void tick(tick_t tic_index) noexcept { overflow_detector.tick(tic_index); }
142
144 size_t size() const { return pool.size(); }
145
146private:
147 using overflow_detector_t =
148 std::conditional_t<SRSRAN_IS_DEFINED(ASSERTS_ENABLED), overflow_checker, no_overflow_checker>;
149
150 ring_buffer_pool pool;
151 overflow_detector_t overflow_detector;
152};
